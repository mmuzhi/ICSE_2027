#include <winsock2.h>
#include <ws2tcpip.h>
#include <iphlpapi.h>
#include <iostream>
#include <cstring>


#pragma comment(lib, "Ws2_32.lib")
#pragma comment(lib, "Iphlpapi.lib")

#include <string>

class IpUtil {
public:
    static bool is_valid_ipv4(const std::string& ip_address);
    static bool is_valid_ipv6(const std::string& ip_address);
    static std::string get_hostname(const std::string& ip_address);
};

bool IpUtil::is_valid_ipv4(const std::string& ip_address) {
    struct sockaddr_in sa;
    int result = inet_pton(AF_INET, ip_address.c_str(), &(sa.sin_addr));
    return result == 1;
}

bool IpUtil::is_valid_ipv6(const std::string& ip_address) {
    struct sockaddr_in6 sa;
    int result = inet_pton(AF_INET6, ip_address.c_str(), &(sa.sin6_addr));
    return result == 1;
}

std::string IpUtil::get_hostname(const std::string& ip_address) {
    if (ip_address == "0.0.0.0") {
        char hostname[NI_MAXHOST];
        if (gethostname(hostname, sizeof(hostname)) == 0) {
            
            return std::string(hostname);
        }
        else {
            
            return "";
        }
    }

    struct sockaddr_in sa;
    sa.sin_family = AF_INET;
    inet_pton(AF_INET, ip_address.c_str(), &(sa.sin_addr));

    char host[NI_MAXHOST];

    int result = getnameinfo((struct sockaddr*)&sa, sizeof(sa), host, sizeof(host), nullptr, 0, 0);
    if (result == 0) {
        return std::string(host);
    }
    else {
        return "";
    }
}


=== REASONING ===
<think>
We need to translate C++ code to Python. The C++ code uses Windows-specific sockets APIs: inet_pton, gethostname, getnameinfo. In Python, we can use socket module (standard library) which provides analogous functions on all platforms (including Windows). However, the behavior should be identical.

Key points:

- IpUtil class with static methods: is_valid_ipv4, is_valid_ipv6, get_hostname.
- is_valid_ipv4: uses inet_pton(AF_INET, ...). In Python, socket.inet_pton(socket.AF_INET, ip) returns the packed binary representation if valid, else raises OSError? Actually inet_pton raises OSError for invalid? In C++ it returns 1 on success, 0 on invalid input, -1 on error (but typically only 0/1). We need to replicate: return True if valid, False otherwise. In Python, we can try/except.
- is_valid_ipv6: similarly with AF_INET6.
- get_hostname: 
  - If ip_address == "0.0.0.0", it calls gethostname() (C++ gethostname). In Python, socket.gethostname() returns the hostname. If it fails? gethostname in C++ returns 0 on success, -1 on error. In Python, socket.gethostname() could raise OSError on failure. We'll catch and return empty string.
  - Otherwise, it uses getnameinfo on a sockaddr_in with AF_INET. Note that the C++ code only handles IPv4 (sockaddr_in, AF_INET). For IPv6, the code would need separate handling. The Python implementation should do the same: only IPv4 for getnameinfo. However, the function receives any ip string; if it's IPv6, the C++ code would likely fail because it only uses AF_INET.
</think>

=== RAW ANSWER ===
```
import socket

class IpUtil:
    @staticmethod
    def is_valid_ipv4(ip_address: str) -> bool:
        try:
            socket.inet_pton(socket.AF_INET, ip_address)
            return True
        except OSError:
            return False

    @staticmethod
    def is_valid_ipv6(ip_address: str) -> bool:
        try:
            socket.inet_pton(socket.AF_INET6, ip_address)
            return True
        except OSError:
            return False

    @staticmethod
    def get_hostname(ip_address: str) -> str:
        if ip_address == "0.0.0.0":
            try:
                return socket.gethostname()
            except OSError:
                return ""
        # Only IPv4 is handled in the C++ code
        try:
            packed = socket.inet_pton(socket.AF_INET, ip_address)
        except OSError:
            return ""
        # getnameinfo returns (hostname, service)
        try:
            hostname, _ = socket.getnameinfo((ip_address, 0), socket.NI_NAMEREQD)
            return hostname
        except OSError:
            return ""
```
