```python
from functools import lru_cache

class Solution:
    def numPermsDISequence(self, s: str) -> int:
        MOD = 10**9 + 7
        n = len(s)
        
        @lru_cache(None)
        def dp(i: int, prev: int) -> int:
            # i: number of elements placed so far (0..n)
            # prev: last placed element (0..i)
            if i == n:
                return 1
            total = 0
            if s[i] == 'D':
                # next must be smaller than prev
                for nxt in range(prev):
                    total = (total + dp(i + 1, nxt)) % MOD
            else:  # 'I'
                # next must be larger than prev
                for nxt in range(prev + 1, i + 2):
                    total = (total + dp(i + 1, nxt)) % MOD
            return total
        
        # initial placement: any of the n+1 numbers as first element
        ans = 0
        for first in range(n + 1):
            ans = (ans + dp(1, first)) % MOD
        return ans
```
