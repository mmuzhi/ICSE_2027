                    total = (total + dfs(pos + 1, new_less)) % MOD
            else:
                for new_less in range(less, n - pos):
                    total = (total + dfs(pos + 1, new_less)) % MOD
            memo[key] = total
            return total

        ans = 0
        for first_less in range(n + 1):
            ans = (ans + dfs(0, first_less)) % MOD
        return ans
