from typing import List

class Solution:
    def secondGreaterElement(self, nums: List[int]) -> List[int]:
        n = len(nums)
        ans = [-1] * n
        s = []   # waiting for first greater
        ss = []  # waiting for second greater (already got first greater)
        for i, x in enumerate(nums):
            # Check if x is the second greater for elements in ss
            while ss and nums[ss[-1]] < x:
                ans[ss.pop()] = x

            while s and nums[s[-1]] < x:
                buff.append(s.pop())
            # Elements in buff just found their first greater -> move to ss

                ss.append(buff.pop())
            s.append(i)
        return ans
