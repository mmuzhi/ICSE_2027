from typing import List

class Solution:
    def validPartition(self, nums: List[int]) -> bool:
        n = len(nums)
        if n < 2:
            return False
        # dp[-1], dp[0], dp[1]
        checks = (True, False, nums[0] == nums[1])
        for i in range(2, n):
            curr = nums[i]
            prev1 = nums[i-1]
