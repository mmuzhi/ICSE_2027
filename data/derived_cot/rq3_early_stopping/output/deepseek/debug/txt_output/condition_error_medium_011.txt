class Solution:
    def numWays(self, s: str) -> int:
        MOD = 10**9 + 7
        N = len(s)
        c1 = s.count('1')
        if c1 == 0:
            return ((N-1)*(N-2)//2) % MOD
        if c1 % 3 != 0:
            return 0
        
        # collect indices of all '1's
        ones = [i for i, ch in enumerate(s) if ch == '1']
        k = c1 // 3
        
        # first cut: between the k-th and (k+1)-th 1s
        i1 = ones[k-1]          # index of k-th '1'
        i2 = ones[k]            # index of (k+1)-th '1'
        
        # second cut: between the 2k-th and (2k+1)-th 1s
        j1 = ones[2*k-1]        # index of 2k-th '1'
        j2 = ones[2*k]          # index of (2k+1)-th '1'

        second_gap = j2 - j1
