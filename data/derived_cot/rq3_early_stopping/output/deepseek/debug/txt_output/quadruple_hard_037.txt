class Solution:
    def getLengthOfOptimalCompression(self, s: str, k: int) -> int:
        n = len(s)
        memo = {}

        def dp(i: int, prev: str, ct: int, rem: int) -> int:
            if rem < 0:
                return float('inf')
            if i == n:
                # end of string, compute length of the current run
                if ct > 1:
                    return len(str(ct)) + 1
                elif ct == 1:
                    return 1
                else:
                    return 0

            key = (i, prev, ct, rem)
            if key in memo:
                return memo[key]

            # option 1: keep current character
            if s[i] == prev:
                inc = dp(i + 1, prev, ct + 1, rem)
            else:
                # close the previous run
                x = 0
                if ct > 1:
                    x = len(str(ct)) + 1
                elif ct == 1:
                    x = 1
                inc = x + dp(i + 1, s[i], 1, rem)

            # option 2: delete current character

            res = min(inc, exc)
            memo[key] = res
