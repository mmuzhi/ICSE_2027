from functools import cache
from typing import List

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppingCosts *= 2

        @cache
        def fn(i: int, x: int) -> int:
            if i == len(toppingCosts):
                return 0
            # If we have already exceeded target, adding more only makes it worse
            if x <= 0:
                return 0
            # Option 1: skip current topping
            skip = fn(i + 1, x)
            # Option 2: take current topping (cannot take duplicate at i+1)
            take = toppingCosts[i] + fn(i + 2, x - toppingCosts[i])
            # Choose the one that is closer to x (tie-break with smaller sum)
            return min(skip, take, key=lambda y: (abs(y - x), y))

        ans = float('inf')

            candidate = bc + fn(0, target - bc)
            ans = min(ans, candidate, key=lambda x: (abs(x - target), x))

from functools import cache

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppingCosts *= 2

        @cache
        def fn(i: int, x: int) -> int:
            if i == len(toppingCosts):
