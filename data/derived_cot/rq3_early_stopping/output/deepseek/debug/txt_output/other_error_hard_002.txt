class Solution:
    def minimumTotalPrice(self, n: int, edges: List[List[int]], price: List[int], trips: List[List[int]]) -> int:

        freq = [0] * n
        level = [0] * n
        parent = [0] * n

            level[i] = l
            parent[i] = p
            for j in g[i]:
                if j != p:
                    dfs(j, l + 1, i)

        def LCA(a: int, b: int) -> int:
            if level[a] > level[b]:
                a, b = b, a
            d = level[b] - level[a]
            while d:
                b = parent[b]
                d -= 1
            if a == b:
                return a
            while a != b:
                a = parent[a]
                b = parent[b]
            return a

        dfs(0, 0, -1)
        for start, end in trips:
            lca = LCA(start, end)
            # mark path from start to lca (inclusive of start, exclusive of lca)
            while start != lca:
                freq[start] += 1
                start = parent[start]
            # mark path from end to lca (inclusive of end, exclusive of lca)
            while end != lca:
                freq[end] += 1
                end = parent[end]
            # mark lca itself
            freq[lca] += 1

        total = 0
        for i in range(n):
            total += freq[i] * price[i]

        # dp returns [max reduction when node not halved, max reduction when node halved]
        def dp(i: int, p: int) -> List[int]:

            halve = freq[i] * (price[i] - price[i] // 2)  # reduction from halving this node

                    not_halve += max(child[0], child[1])
                    halve += child[0]

        best_reduction = max(dp(0, -1))
