class Solution:
    def dp(self, i, s, prev, k, ct, n, dct):
        if k < 0:
            return float("infinity")
        if i >= n:
            x = 0
            if ct > 1:

            return x
        key = (i, prev, ct, k)
        if key in dct:
            return dct[key]
        if s[i] == prev:
            inc = self.dp(i + 1, s, prev, k, ct + 1, n, dct)
