from typing import List

class Solution:
    def sumSubarrayMins(self, arr: List[int]) -> int:
        MOD = 10**9 + 7
        n = len(arr)
        stack = []
        res = 0
        
        # Process each element as the next smaller element
        for i in range(n):
            while stack and arr[stack[-1]] > arr[i]:
                cur = stack.pop()
                left = stack[-1] if stack else -1
                # arr[cur] is minimum for subarrays starting in (left, cur] and ending in [cur, i)
                res = (res + arr[cur] * (cur - left) * (i - cur)) % MOD
            stack.append(i)
        
        # Process remaining elements (no next smaller element)
        while stack:
            cur = stack.pop()
            left = stack[-1] if stack else -1
            # right boundary is n (end of array)
            res = (res + arr[cur] * (cur - left) * (n - cur)) % MOD
        
        return res
