from typing import List
import collections

class Solution:
    def isPossibleToCutPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        # reverse DFS to mark cells that can reach the end
        def dfs(i: int, j: int) -> None:
            if i < 0 or i >= m or j < 0 or j >= n or grid[i][j] != 1:
                return
            grid[i][j] = 2
            dfs(i - 1, j)
            dfs(i, j - 1)
        dfs(m - 1, n - 1)
        # start must be reachable to the end
        if grid[0][0] != 2:
            return False
        # BFS from start, only moving down/right
        dq = collections.deque()
        dq.append((0, 0))
