from typing import List
import collections

class Solution:
    def isPossibleToCutPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        if grid[0][0] == 0 or grid[m-1][n-1] == 0:
            return True

        # DFS to find any path from (0,0) to (m-1,n-1)
        dirs = [(1,0), (0,1)]
        path = []
        found = False

        def dfs(i, j):
            nonlocal found
            if found:
                return
            if i == m-1 and j == n-1:
                path.append((i,j))
                found = True
                return
            if grid[i][j] == 0:
                return
            grid[i][j] = 0  # mark visited
            path.append((i,j))
            for di, dj in dirs:
                ni, nj = i+di, j+dj
                if 0 <= ni < m and 0 <= nj < n:
                    dfs(ni, nj)
                    if found:
                        return
            if not found:
                path.pop()

        dfs(0, 0)

        if not found:
            return True   # already disconnected

        # Try to flip each cell on the found path (except start and end?)
        # Actually flipping start or end also disconnects, so we can try those too.
        for x, y in path:
            # skip if it's start or end? They are included.
            # Backup grid? Instead, just modify and BFS again.
            # Since we only need to know if flipping that cell disconnects.
            # We can simply BFS without that cell.
            # Optimization: we can BFS only once from start, but easier to test.
            if (x, y) == (0, 0) or (x, y) == (m-1, n-1):
                # flipping start or end always disconnects
                continue  # we can return True immediately, but let's handle generically.
                # Actually we can return True right away.
                # But for completeness, we test.
                pass

            # Temporarily set cell to 0 and check if path still exists
            original = grid[x][y]
            grid[x][y] = 0
            # BFS from (0,0) to (m-1,n-1)
            dq = collections.deque()
