```python
from typing import Optional, List

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def splitListToParts(self, head: Optional[ListNode], k: int) -> List[Optional[ListNode]]:
        # calculate length
        length = 0
        ptr = head
        while ptr:
            length += 1
            ptr = ptr.next

        # determine size of each part
        base = length // k
        extra = length % k
        part_sizes = []
        for i in range(k):
            part_sizes.append(base + (1 if i < extra else 0))

        # split the list
        result = []
        ptr = head
        idx = 0
        while ptr:
            result.append(ptr)          # start of current part
            size = part_sizes[idx]
            for _ in range(size - 1):
                ptr = ptr.next
            # cut the list
            next_part = ptr.next
            ptr.next = None
            ptr = next_part
            idx += 1

        # pad remaining parts with None
        while len(result) < k:
            result.append(None)

        return result
```
