class Solution:
    def numPermsDISequence(self, s: str) -> int:
        MOD = 10**9 + 7
        mem = defaultdict(int)
        
        def dfs(i: int, last: int = 0) -> int:

                return 1
            key = (i, last)

                return mem[key]
