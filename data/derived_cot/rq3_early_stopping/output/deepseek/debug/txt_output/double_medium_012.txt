from typing import List
from collections import deque

class Solution:
    def isPossibleToCutPath(self, grid: List[List[int]]) -> bool:
        m, n = len(grid), len(grid[0])
        # Mark cells that can reach (m-1, n-1) by moving up/left
        def dfs(i: int, j: int) -> None:
            grid[i][j] = 2
            for di, dj in [(-1, 0), (0, -1)]:

            dfs(m-1, n-1)

        # BFS from (0,0) moving down/right, only stepping on marked cells

                if i == m - 1 and j == n - 1:
                    return False
                for di, dj in dirs:
                    ni, nj = i + di, j + dj
