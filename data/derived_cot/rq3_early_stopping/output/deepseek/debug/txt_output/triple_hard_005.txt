```python
from typing import List

class Solution:
    def longestValidSubstring(self, word: str, forbidden: List[str]) -> int:
        # build a trie of forbidden words
        trie = {}
        for f in forbidden:
            node = trie
            for ch in f:
                if ch not in node:
                    node[ch] = {}
                node = node[ch]
            node['#'] = True  # mark end of a word

        max_len = 0
        n = len(word)
        right = n  # current right boundary (exclusive)

        # iterate left boundary from right to left
        for left in range(n - 1, -1, -1):
            # check suffixes starting at 'left' that could be forbidden
            node = trie
            for offset in range(left, min(left + 10, right)):
                ch = word[offset]
                if ch not in node:
                    break
                node = node[ch]
                if '#' in node:
                    # found a forbidden word from left to offset (inclusive)
                    right = offset + 1  # adjust right so that this word is excluded
                    break
            # update max length of valid substring ending at right-1
            max_len = max(max_len, right - left)

        return max_len
```
