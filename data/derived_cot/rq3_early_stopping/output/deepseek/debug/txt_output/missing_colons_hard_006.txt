import collections

class Solution:
    def kSimilarity(self, s1: str, s2: str) -> int:
        deque = collections.deque([s1])
        seen = set([s1])
        ans = 0
        
        while deque:
            for _ in range(len(deque)):
                curr = deque.popleft()
                if curr == s2:
                    return ans

                        if new not in seen:

            ans += 1
