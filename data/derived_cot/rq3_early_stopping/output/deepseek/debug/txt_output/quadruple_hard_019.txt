            # Cycle detection
            if i in record:
                prev_count1, prev_count2 = record[i]
                cycle_s1 = count1 - prev_count1
                cycle_s2 = count2 - prev_count2
                if cycle_s1 == 0:
                    continue
                remaining_s1 = n1 - prev_count1
                cycles = remaining_s1 // cycle_s1
                count1 += cycles * cycle_s1
                count2 += cycles * cycle_s2
                # The current state is now after the cycles; break to handle remainder
                break
            else:
                record[i] = (count1, count2)
        
        # Handle the remaining s1 repetitions (if any) without further cycle detection
        while count1 < n1:
            for ch in s2:
                pos = s1.find(ch, i)
                if pos == -1:
                    count1 += 1
                    if count1 >= n1:
                        return count2 // n2
                    i = 0
                    pos = s1.find(ch)
                    if pos == -1:
                        return 0
                i = pos + 1
                if i == len1:
                    i = 0
                    count1 += 1
            count2 += 1
        
        return count2 // n2
