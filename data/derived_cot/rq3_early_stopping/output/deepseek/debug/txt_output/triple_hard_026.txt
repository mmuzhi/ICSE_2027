class Solution:
    def findTheString(self, lcp: List[List[int]]) -> str:

            for j in range(i + 1, n):

                    return ''
                if lcp[i][j] > n - j:
                    return ''
        for i in range(n):
            if lcp[i][i] != n - i:
                return ''

        # build pattern - the only possible candidate for answer
        pattern = [None] * n
        next_el_ind = 0
        for i in range(n):
            if pattern[i] is not None:
                continue
            pattern[i] = next_el_ind
            next_el_ind += 1
            for j in range(i + 1, n):
                if lcp[i][j] > 0:
                    if pattern[j] is not None and pattern[j] != pattern[i]:
                        return ''
                    pattern[j] = pattern[i]

        # check if lcp is valid - compute pattern's LCP and compare
        pattern_lcp = [[0] * n for _ in range(n)]
        for i in range(n - 1, -1, -1):
