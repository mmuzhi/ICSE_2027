from typing import List
from functools import cache

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppingCosts *= 2

        @cache
        def fn(i: int, x: int) -> int:
            """Return sum of subsequence of toppingCosts[i:] closest to x."""
            if x < 0 or i >= len(toppingCosts):
                return 0
            # skip current topping
            skip = fn(i+1, x)
            # take current topping
            take = toppingCosts[i] + fn(i+1, x - toppingCosts[i])
            # choose the one closer to x, tie-breaking by smaller sum
            return min(skip, take, key=lambda y: (abs(y - x), y))
        
        ans = float('inf')
        for bc in baseCosts:
            ans = min(ans, bc + fn(0, target - bc), key=lambda z: (abs(z - target), z))
        return ans
```
