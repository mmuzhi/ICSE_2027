    def isPossible(self, n: int, edges: list[list[int]]) -> bool:

        oddDegreesNodes = [i for i in range(n) if (len(neighbors[i]) % 2 == 1)]
        numOdd = len(oddDegreesNodes)
        if numOdd == 0:
            return True
        elif numOdd == 2:

            if o1 not in neighbors[o2]:
                return True

            return False
