=== RAW ANSWER ===
```python
from collections import deque
from typing import List

class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        if not edges:
            return 1.0 if target == 1 else 0.0

        # build adjacency list
        g = [[] for _ in range(n + 1)]
        for u, v in edges:
            g[u].append(v)
            g[v].append(u)

        visited = [False] * (n + 1)
        q = deque()
        q.append((1, 1.0))          # (node, probability)
        visited[1] = True

        for _ in range(t):
            level_size = len(q)
            # we will store stays for nodes that do not move
            stays = []
            for _ in range(level_size):
                node, prob = q.popleft()
                # count unvisited neighbors (children)
                children = [nei for nei in g[node] if not visited[nei]]
                cnt = len(children)
                if cnt == 0:
                    # leaf: frog stays, add back for next level
                    stays.append((node, prob))
                else:
                    # move to each child with equal probability
                    for nei in children:
                        visited[nei] = True
                        q.append((nei, prob / cnt))
            # add all staying nodes back to the queue
            q.extend(stays)

        # after t seconds, find target in the queue
        for node, prob in q:
            if node == target:
                return prob
        return 0.0
```
