        # and the current position pointer in s1
        pointer = 0

        track = defaultdict(int)  # maps pointer (position in s1) to index in rec

        # We will simulate matching s2 repeatedly and record the state after
        # each full match of s2 (i.e., after processing all its characters)
        while True:
            for c in s2:
                # Find c in s1 starting from current pointer
                pos = s1.find(c, pointer)
                if pos == -1:
                    # Need to consume another s1
