```python
from typing import List
from functools import lru_cache

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppingCosts *= 2
        
        @lru_cache(None)
        def fn(i, x):
            """Return sum of subsequence of toppingCosts[i:] closest to x."""
            if x < 0 or i == len(toppingCosts):
                return 0
            # Option 1: skip current topping
            skip = fn(i+1, x)
            # Option 2: take current topping
            take = toppingCosts[i] + fn(i+1, x - toppingCosts[i])
            # Choose the one whose sum is closer to x (with tie-breaker smaller sum)
            return min(skip, take, key=lambda y: (abs(y - x), y))
        
        ans = float('inf')
        for bc in baseCosts:
            ans = min(ans, bc + fn(0, target - bc), key=lambda x: (abs(x - target), x))
        return ans
```
