from collections import Counter

class Solution:
    def maxDifference(self, s: str) -> int:
        # Count the frequency of each character
        count = Counter(s)
        # Separate the frequencies into even and odd
        even_freqs = []
        odd_freqs = []
        for freq in count.values():
            if freq % 2 == 0:
                even_freqs.append(freq)
            else:
                odd_freqs.append(freq)
        # Since there's at least one even and one odd, we can safely take min and max
        return max(odd_freqs) - min(even_freqs)

from collections import Counter

class Solution:
    def maxDifference(self, s: str) -> int:
        count = Counter(s)
        even_freqs = [freq for freq in count.values() if freq % 2 == 0]
        odd_freqs = [freq for freq in count.values() if freq % 2 != 0]
        return max(odd_freqs) - min(even_freqs)
