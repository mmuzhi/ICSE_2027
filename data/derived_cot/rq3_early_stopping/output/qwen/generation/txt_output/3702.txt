        exp_map = {
            1: [0, 0, 0, 0],

            9: [0, 2, 0, 0],

        }
        n = len(nums)
        max_len = 0
        for start in range(n):
            min_vals = [10**9] * 4

                if min_vals[j] + max_vals[j] != total_vals[j]:
                    valid = False
                    break
            if valid:
                max_len = max(max_len, current_length)
            for end in range(start + 1, n):
