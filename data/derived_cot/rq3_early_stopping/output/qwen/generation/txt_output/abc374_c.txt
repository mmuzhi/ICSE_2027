=== ANSWER ===
n = int(input().strip())
k_list = list(map(int, input().split()))

total = sum(k_list)
best = total  # worst case: one group has all

# Iterate over all subsets (bitmask from 0 to (1<<n)-1)
for bitmask in range(1 << n):
    s = 0
    for i in range(n):
        if bitmask & (1 << i):
            s += k_list[i]
    candidate = max(s, total - s)
    if candidate < best:
        best = candidate

print(best)
