    nodes = [(0, 0)] + points
    total_nodes = 1 + 2*n  # indices 0 to 2*n

        endpoints[i] = [2*i+1, 2*i+2]

        for j in range(total_nodes):
            x1, y1 = nodes[i]
            x2, y2 = nodes[j]
            dx = x2 - x1
            dy = y2 - y1
            dist_matrix[i][j] = math.sqrt(dx*dx + dy*dy)

    num_masks = 1 << n
