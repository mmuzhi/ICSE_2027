Output: true

(s[2] + s[3]) % 10 = (0 + 2) % 10 = 2
s becomes "292"

(s[0] + s[1]) % 10 = (2 + 9) % 10 = 1
(s[1] + s[2]) % 10 = (9 + 2) % 10 = 1
s becomes "11"
