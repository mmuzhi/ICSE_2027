import collections

class Solution:
    def minimumObstacles(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        distance = [[float('inf')] * n for _ in range(m)]
        distance[0][0] = 0
        q = collections.deque()
        q.append((0, 0, 0))
        
        while q:
            d, i, j = q.popleft()
            if i == m - 1 and j == n - 1:
                return d
            if d != distance[i][j]:
                continue
            for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:

                if 0 <= ni < m and 0 <= nj < n:
