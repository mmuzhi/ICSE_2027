class Solution:
    def longestValidSubstring(self, word: str, forbidden: List[str]) -> int:
        def isForbidden(s):
            for f in forbidden:
                if f in s:
                    return True
            return False
        
        n = len(word)

        i = 0
        for j in range(n):
