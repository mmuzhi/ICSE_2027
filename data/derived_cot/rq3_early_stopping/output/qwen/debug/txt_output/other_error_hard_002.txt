                a = parent_arr[a]
                b = parent_arr[b]
            return a

            while i != lca:
                freq[i] += 1
                i = parent_arr[i]
            freq[lca] += 1
            while j != lca:
                freq[j] += 1
                j = parent_arr[j]

        dp1 = [0] * n
        stack = [0]
        parent_arr = [-1] * n
        order = []
