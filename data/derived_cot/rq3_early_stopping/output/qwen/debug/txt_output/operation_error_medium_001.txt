from collections import deque, defaultdict

class Solution:
    def findAllRecipes(self, recipes: List[str], ingredients: List[List[str]], supplies: List[str]) -> List[str]:
        # Build the set of all nodes
        nodes = set(recipes) | set(j for i in ingredients for j in i) | set(supplies)
        
        # Build the graph: for each node, if it is a recipe, then its ingredients are the outgoing edges; else, empty list.
        graph = defaultdict(list)
        for idx, recipe in enumerate(recipes):
            graph[recipe] = ingredients[idx]
        
        # Build the reverse graph: for each node, list of nodes that depend on it (i.e., recipes that require this ingredient)
        rev_graph = defaultdict(list)
        for node in nodes:
            for neighbor in graph[node]:
                rev_graph[neighbor].append(node)
        
        # Compute in_degree for each node
        in_degree = {}
        for node in nodes:
            if node in recipes:
                in_degree[node] = len(graph[node])
            else:
                in_degree[node] = 0
        
        # Kahn's algorithm
        q = deque()
        # Start with all nodes that are in the supplies (available initially)
        for node in nodes:
            if node in supplies:
                q.append(node)
