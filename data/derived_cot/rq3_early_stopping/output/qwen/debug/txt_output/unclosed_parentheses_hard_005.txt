    def longestCycle(self, edges: List[int]) -> int:
        n = len(edges)
        bl = [0] * n
        mp = {}
        mx = -1
        for i in range(n):
            if bl[i] == 0:
                x = i
                l = 0
                st = set()
                while x != -1 and bl[x] == 0:
