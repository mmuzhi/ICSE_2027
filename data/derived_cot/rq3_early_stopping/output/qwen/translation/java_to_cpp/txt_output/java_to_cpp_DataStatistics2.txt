    private double[] data;

    public DataStatistics2(List<Integer> data) {
        this.data = data.stream().mapToDouble(Integer::doubleValue).toArray();
    }

    public double getSum() {
        return DoubleStream.of(data).sum();
    }

    public double getMin() {
        return DoubleStream.of(data).min().orElse(Double.NaN);
    }

    public double getMax() {
        return DoubleStream.of(data).max().orElse(Double.NaN);
    }

    public double getVariance() {
        double mean = getSum() / data.length;
        return DoubleStream.of(data)
                .map(val -> (val - mean) * (val - mean))
                .average()
                .orElse(Double.NaN);
    }

    public double getStdDeviation() {
        return Math.sqrt(getVariance());
    }

    public double getCorrelation() {
        return 1.0;
    }
}

To solve this problem, we need to translate a Java class that computes various statistics (sum, min, max, variance, standard deviation, and correlation) from a list of integers into an equivalent C++ class. The solution must maintain the same behavior, including handling edge cases such as empty input data.

### Approach
1. **Class Structure**: The C++ class will be similar to the Java class, using a private member `std::vector<double>` to store the data as doubles for consistency with the Java version.
2. **Constructor**: The constructor will take a `std::vector<int>` and convert each element to a double, storing them in the private member.
3. **Statistical Methods**: Each method (getSum, getMin, getMax, getVariance, getStdDeviation, getCorrelation) will be implemented to perform the corresponding calculation. Special cases, such as empty input, are handled to return `std::numeric_limits<double>::quiet_NaN()` for min, max, and variance, and `0.0` for sum (as per the original Java behavior).
4. **Efficiency**: The implementations use standard algorithms (like `std::accumulate` and `std::min_element`) to ensure efficiency and correctness, mimicking the behavior of Java's streams.

### Solution Code
```cpp
#include <vector>
#include <cmath>
#include <limits>
#include <algorithm>
#include <numeric>

class DataStatistics2 {
private:
    std::vector<double> data;

public:
    explicit DataStatistics2(const std::vector<int>& data) {
        for (int i : data) {
            this->data.push_back(static_cast<double>(i));
        }
    }

    double getSum() const {
        if (data.empty()) {
            return 0.0;
        }
        return std::accumulate(data.begin(), data.end(), 0.0);
    }

    double getMin() const {
        if (data.empty()) {
            return std::numeric_limits<double>::quiet_NaN();
        }
        return *std::min_element(data.begin(), data.end());
    }

    double getMax() const {
        if (data.empty()) {
            return std::numeric_limits<double>::quiet_NaN();
        }
        return *std::max_element(data.begin(), data.end());
    }

    double getVariance() const {
        if (data.empty()) {
            return std::numeric_limits<double>::quiet_NaN();
        }
        double mean = getSum() / data.size();
        double sq_sum = 0.0;
        for (double val : data) {
            sq_sum += (val - mean) * (val - mean);
        }
        return sq_sum / data.size();
    }

    double getStdDeviation() const {
        double variance = getVariance();
        return std::sqrt(variance);
    }

    double getCorrelation() const {
        return 1.0;
    }
};
```

4. **getVariance**: Computes the mean of the data, then calculates the average of squared differences from the mean. Handles empty input by returning NaN.
5. **getStdDeviation**: Returns the square root of the variance. If the variance is NaN, the result will also be NaN.
