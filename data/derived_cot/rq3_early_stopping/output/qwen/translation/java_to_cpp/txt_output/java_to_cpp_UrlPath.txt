package org.example;

import java.net.URLDecoder;
import java.util.ArrayList;
import java.util.List;

public class UrlPath {
    private List<String> segments;
    private boolean withEndTag;

    public UrlPath() {
        this.segments = new ArrayList<>();
        this.withEndTag = false;
    }

    public void add(String segment) {
        this.segments.add(fixPath(segment));
    }

    public void parse(String path, String charset) {
        if (path != null && !path.isEmpty()) {
            if (path.endsWith("/")) {
                this.withEndTag = true;
            }

            path = fixPath(path);
            if (path != null && !path.isEmpty()) {
                String[] split = path.split("/");
                for (String seg : split) {
                    try {
                        String decodedSeg = URLDecoder.decode(seg, charset);
                        this.segments.add(decodedSeg);
                    } catch (Exception e) {
                        e.printStackTrace();
                    }
                }
            }
        }
    }

    public static String fixPath(String path) {
        if (path == null || path.isEmpty()) {
            return "";
        }

        String segmentStr = path.strip().replaceAll("^/+|/+$", "");
        return segmentStr;
    }

    public List<String> getSegments() {
        return segments;
    }

    public boolean isWithEndTag() {
        return withEndTag;
    }
}

class UrlPath {
private:
    std::vector<std::string> segments;
    bool withEndTag;

    static std::string fixPath(const std::string& path) {
        if (path.empty()) {
            return "";
        }

        // Trim leading and trailing whitespace
        size_t start = 0;
        while (start < path.size() && std::isspace(static_cast<unsigned char>(path[start]))) {
            ++start;
        }
        if (start == path.size()) {
            return "";
        }

        size_t end = path.size() - 1;
        while (end > start && std::isspace(static_cast<unsigned char>(path[end]))) {
            --end;
        }

        // Remove leading and trailing slashes
        std::string trimmed = path.substr(start, end - start + 1);
        trimmed = std::regex_replace(trimmed, std::regex("^/+|/+$"), "");
        return trimmed;
    }

    static std::string decodeURIComponent(const std::string& input, const std::string& charset) {
        if (input.empty()) {
            return "";
        }

        std::string result;
        result.reserve(input.size());

        for (size_t i = 0; i < input.size(); ) {
            if (input[i] == '%' && i + 2 < input.size()) {
                std::string hex = input.substr(i + 1, 2);
                try {
                    int codePoint = std::stoi(hex, nullptr, 16);
                    result += static_cast<char>(codePoint);
                    i += 3;
                } catch (...) {
                    result += '%';
                    result += hex;
                    i += 3;
                }
            } else {
                result += input[i];
                ++i;
            }
        }

        return result;
    }

public:
    UrlPath() : segments(), withEndTag(false) {}

    void add(const std::string& segment) {
        segments.push_back(fixPath(segment));
    }

    void parse(const std::string& path, const std::string& charset) {
        if (path.empty()) {
            return;
        }

        withEndTag = (path.back() == '/');
        std::string fixedPath = fixPath(path);
        if (fixedPath.empty()) {
            return;
        }

        std::vector<std::string> parts;
        std::sregex_token_iterator iter(fixedPath.begin(), fixedPath.end(), std::regex("/"), 0);
        std::sregex_token_iterator end;

        for (std::sregex_token_iterator i = iter; i != end; ++i) {
            if (!i->str().empty()) {
                try {
                    std::string decoded = decodeURIComponent(i->str(), charset);
                    segments.push_back(decoded);
                } catch (...) {
                    std::cerr << "Error decoding segment: " << i->str() << std::endl;
                }
            }
        }
    }

    }
