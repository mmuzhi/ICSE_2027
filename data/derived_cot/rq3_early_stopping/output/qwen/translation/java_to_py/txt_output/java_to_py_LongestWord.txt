package org.example;

import java.util.ArrayList;
import java.util.List;
import java.util.regex.Pattern;

public class LongestWord {
    List<String> wordList;

    public LongestWord() {
        wordList = new ArrayList<>();
    }

    public void addWord(String word) {
        wordList.add(word);
    }

    public String findLongestWord(String sentence) {
        String longestWord = "";
        sentence = sentence.toLowerCase();
        sentence = sentence.replaceAll("[" + Pattern.quote("!\"#$%&'()*+,-./:;<=>?@[\\]^_`{|}~") + "]", "");
        String[] words = sentence.split(" ");
        for (String word : words) {
            if (wordList.contains(word) && word.length() > longestWord.length()) {
                longestWord = word;
            }
        }
        return longestWord;
    }

    public static void main(String[] args) {
        LongestWord longestWord = new LongestWord();
        longestWord.addWord("A");
        longestWord.addWord("aM");
        System.out.println(longestWord.findLongestWord("I am a student."));
    }
}

=== RAW ANSWER ===

To solve this problem, we need to translate a Java program into Python while maintaining the same behavior. The Java program processes a list of words and finds the longest word present in a given sentence after removing punctuation and converting to lowercase. 

### Approach
1. **Data Structures**: Use a list in Python to store words, similar to the `ArrayList` in Java.
2. **String Processing**: Convert the sentence to lowercase and remove punctuation using regular expressions.
3. **Finding the Longest Word**: Iterate through the words in the processed sentence, checking if they exist in the word list and updating the longest word found.
4. **Main Functionality**: The `findLongestWord` method processes the input sentence and returns the longest word from the word list.

### Solution Code
```python
import re

class LongestWord:
    def __init__(self):
        self.word_list = []

    def add_word(self, word):
        self.word_list.append(word)

    def find_longest_word(self, sentence):
        # Convert sentence to lowercase and remove punctuation
        sentence = sentence.lower()
        # Escape special characters for regex pattern
        pattern_str = re.escape('!"#$%&\'()*+,-./:;<=>?@[\\]^_`{|}~')
        # Replace all punctuation with empty strings
        sentence = re.sub(f'[{pattern_str}]+', '', sentence)
        # Split the sentence into words
        words = sentence.split()
        longest_word = ""
        for word in words:
            if word in self.word_list and len(word) > len(longest_word):
                longest_word = word
        return longest_word

if __name__ == '__main__':
    longest_word = LongestWord()
    longest_word.add_word("A")
    longest_word.add_word("aM")
    print(longest_word.find_longest_word("I am a student."))
```
