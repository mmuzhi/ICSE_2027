        count = 1

                a -= (x ** (2 * k)) / self.factorial(2 * k)
            else:
                a += (x ** (2 * k)) / self.factorial(2 * k)

            n += 1

        return round(g, 10)

        if self.cos(x) != 0:
            result = self.sin(x) / self.cos(x)

#include <cmath>
#include <iostream>
#include <limits>

class TriCalculator {
public:
    double cos(double x) {
        return round(taylor(x, 50) * 10) / 10.0;
    }

    long long factorial(int a) {
        long long b = 1;
        while (a != 1) {
            b *= a;
            a -= 1;
        }
        return b;
    }

    double taylor(double x, int n) {
        double a = 1.0;
        x = x / 180 * M_PI;
        int count = 1;
        for (int k = 1; k < n; k++) {
            if (count % 2 != 0) {
                a -= (std::pow(x, 2 * k)) / static_cast<double>(factorial(2 * k));
            } else {
                a += (std::pow(x, 2 * k)) / static_cast<double>(factorial(2 * k));
            }
            count++;
        }
        return a;
    }

    double sin(double x) {
        x = x / 180 * M_PI;
        double g = 0.0;
        double t = x;
        int n = 1;

        while (std::fabs(t) >= 1e-15) {
            g += t;
            n++;
            t = -t * x * x / (2 * n - 1) / (2 * n - 2);
        }
        return round(g * 10) / 10.0;
    }

    double tan(double x) {
        if (cos(x) != 0.0) {
            double result = sin(x) / cos(x);
            return round(result * 10) / 10.0;
        } else {
            return std::numeric_limits<double>::quiet_NaN();
        }
    }
};

int main() {
    TriCalculator calc;
    std::cout << calc.cos(60) << std::endl;      // Output: 0.5
    std::cout << calc.sin(30) << std::endl;      // Output: 0.5
    std::cout << calc.tan(45) << std::endl;      // Output: 1.0
    return 0;
}
