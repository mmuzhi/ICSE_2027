import zipfile


class ZipFileProcessor:
    def __init__(self, file_name):
        self.file_name = file_name

    def read_zip_file(self):
        try:
            zip_file = zipfile.ZipFile(self.file_name, 'r')
            return zip_file
        except:
            return None

    def extract_all(self, output_path):
        try:
            with zipfile.ZipFile(self.file_name, 'r') as zip_file:
                zip_file.extractall(output_path)
            return True
        except:
            return False

    def extract_file(self, file_name, output_path):
        try:
            with zipfile.ZipFile(self.file_name, 'r') as zip_file:
                zip_file.extract(file_name, output_path)
            return True
        except:
            return False

    def create_zip_file(self, files, output_file_name):
        try:
            with zipfile.ZipFile(output_file_name, 'w') as zip_file:
                for file in files:
                    zip_file.write(file)
            return True
        except:
            return False

namespace fs = std::filesystem;

class ZipFileProcessor {
private:
    std::string file_name;

    // Helper function to check if a file exists
    bool fileExists(const std::string& name) {
        return fs::exists(name);
    }

public:
    ZipFileProcessor(const std::string& file_name) : file_name(file_name) {}

    // Open a zip file for reading
    std::unique_ptr<zip> read_zip_file() {
        if (!fileExists(file_name)) {
            return nullptr;
        }
        zip* zip_file = zip_open(file_name.c_str(), ZIP_RDONLY, nullptr);
        if (zip_file == nullptr) {
            return nullptr;
        }
        return std::unique_ptr<zip>(zip_file);
    }

    // Extract all files from the zip to the specified path
    bool extract_all(const std::string& output_path) {
        auto zip_file = read_zip_file();
        if (!zip_file) {
            return false;
        }

        // Create output directory if it doesn't exist
        if (!fs::exists(output_path)) {
            fs::create_directories(output_path);
        }

        zip_int64_t num_entries = zip_get_num_entries(zip_file);
        for (zip_int64_t i = 0; i < num_entries; ++i) {
            struct zip_stat zip_stat;
            if (zip_stat_index(zip_file, i, &zip_stat) != ZIP_OK) {
                return false;
            }

            std::string output_file = output_path + "/" + zip_stat.name;
            std::ofstream fout(output_file, std::ios::binary);
            if (!fout) {
                return false;
            }

            zip_filepos64_t pos = 0;
            zip_uint64_t size = zip_stat.size;
            while (size > 0) {
                zip_int64_t read_size = std::min(static_cast<zip_uint64_t>(size), static_cast<zip_uint64_t>(4096));
                zip_uint64_t read_bytes;
                zip_int64_t error = zip_read(zip_file, reinterpret_cast<char*>(&read_bytes), pos, read_size, &read_bytes);
                if (error != ZIP_OK || read_bytes == 0) {
                    return false;
                }
                fout.write(reinterpret_cast<char*>(read_bytes), read_bytes);
                size -= read_bytes;
                pos += read_bytes;
            }
        }
        return true;
    }

    // Extract a specific file from the zip to the specified path
    bool extract_file(const std::string& file_name, const std::string& output_path) {
        auto zip_file = read_zip_file();
        if (!zip_file) {
            return false;
        }

        // Create output directory if it doesn't exist
        if (!fs::exists(output_path)) {
            fs::create_directories(output_path);
        }

        struct zip_stat zip_stat;
        if (zip_stat_name(zip_file, file_name.c_str(), &zip_stat) != ZIP_OK) {
            return false;
        }

        std::string output_file = output_path + "/" + zip_stat.name;
        std::ofstream fout(output_file, std::ios::binary);
        if (!fout) {
            return false;
        }

        zip_filepos64_t pos = 0;
        zip_uint64_t size = zip_stat.size;
        while (size > 0) {
            zip_int64_t read_size = std::min(static_cast<zip_uint64_t>(size), static_cast<zip_uint64_t>(4096));
            zip_uint64_t read_bytes;
            zip_int64_t error = zip_read(zip_file, reinterpret_cast<char*>(&read_bytes), pos, read_size, &read_bytes);
            if (error != ZIP_OK || read_bytes == 0) {
                return false;
            }
            fout.write(reinterpret_cast<char*>(read_bytes), read_bytes);
            size -= read_bytes;
            pos += read_bytes;
        }
        return true;
    }

    // Create a zip file from a list of files
    bool create_zip_file(const std::vector<std::string>& files, const std::string& output_file_name) {
        // Ensure output file has .zip extension
        if (output_file_name.find(".zip") == std::string::npos) {
            output_file_name += ".zip";

            if (zip_stat_name(zip_file, file.c_str(), &zip_stat) != ZIP_OK) {
                zip_close(zip_file);
                return false;
            }

            zip_int64_t error = zip_add(zip_file, file.c_str(), file.c_str());
            if (error != ZIP_OK) {
                zip_close(zip_file);
                return false;
            }
        }

        zip_close(zip_file);
        return true;
    }
