                        // We use base 16 if is_hex, else base 10.
                        std::string num_str;
                        if (is_hex) {

                            }

                        out += string.substr(pos);
                        break;

                }
            }
            out += string[pos];
            pos++;
        }

        // Append the remaining part of the string
        if (pos < length) {

        }

    }

Note: This code uses `std::u32string` and `char32_t` to handle any Unicode code point. The `is_hex_char` function checks for ASCII hex digits. The conversion from the entity value to a `char32_t` is done using `static_cast<char32_t>(entity_value)`.
