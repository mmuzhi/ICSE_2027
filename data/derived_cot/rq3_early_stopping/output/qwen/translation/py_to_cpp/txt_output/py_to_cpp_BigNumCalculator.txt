        num1 = num1.zfill(max_length)
        num2 = num2.zfill(max_length)

            negative = False
        else:

                num1, num2 = num2, num1
                negative = True
            else:
                negative = False

        num1 = num1.zfill(max_length)
        num2 = num2.zfill(max_length)

        borrow = 0
        result = []

            digit_diff = int(num1[i]) - int(num2[i]) - borrow

                digit_diff += 10
                borrow = 1
            else:
                borrow = 0

            result.pop(0)

        if negative:
            result.insert(0, '-')

        result = [0] * (len1 + len2)

        for i in range(len1 - 1, -1, -1):

                mul = int(num1[i]) * int(num2[j])
                p1, p2 = i + j, i + j + 1
                total = mul + result[p2]

                result[p1] += total // 10
                result[p2] = total % 10
