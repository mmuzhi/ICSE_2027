            hex_str = hex_str.substr(1);
        }

        if (hex_str.empty()) {
            return 0;
        }

        int result = 0;
        for (char c : hex_str) {
            if (c >= '0' && c <= '9') {
                result = result * 16 + (c - '0');
            } else if (c >= 'A' && c <= 'F') {
                result = result * 16 + (c - 'A' + 10);
            } else if (c >= 'a' && c <= 'f') {
                result = result * 16 + (c - 'a' + 10);
            } else {
                throw std::invalid_argument("Invalid hex digit");
            }
        }

        if (negative) {
            result = -result;
        }
        return result;
    }
};
