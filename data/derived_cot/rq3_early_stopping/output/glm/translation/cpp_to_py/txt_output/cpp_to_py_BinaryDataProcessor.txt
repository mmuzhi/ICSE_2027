#include <algorithm>
#include <bitset>
#include <sstream>
#include <stdexcept>

#include <string>
#include <unordered_map>

class BinaryDataProcessor {
public:
    BinaryDataProcessor(const std::string& binary_string);
    void clean_non_binary_chars();
    std::unordered_map<std::string, double> calculate_binary_info();
    std::string convert_to_ascii();
    std::string convert_to_utf8();

    std::string binary_string;
};

BinaryDataProcessor::BinaryDataProcessor(const std::string& binary_string) {
    this->binary_string = binary_string;
    clean_non_binary_chars();
}

void BinaryDataProcessor::clean_non_binary_chars() {
    binary_string.erase(std::remove_if(binary_string.begin(), binary_string.end(),
        [](char c) { return c != '0' && c != '1'; }), binary_string.end());
}

std::unordered_map<std::string, double> BinaryDataProcessor::calculate_binary_info() {
    int zeroes_count = std::count(binary_string.begin(), binary_string.end(), '0');
    int ones_count = binary_string.size() - zeroes_count;
    int total_length = binary_string.size();

    double zeroes_percentage = static_cast<double>(zeroes_count) / total_length;
    double ones_percentage = static_cast<double>(ones_count) / total_length;

    return {
        {"Zeroes", zeroes_percentage},
        {"Ones", ones_percentage},
        {"Bit length", static_cast<double>(total_length)}
    };
}

std::string BinaryDataProcessor::convert_to_ascii() {
    std::stringstream ss;
    for (size_t i = 0; i < binary_string.size(); i += 8) {
        std::bitset<8> byte(binary_string.substr(i, 8));
        ss << static_cast<char>(byte.to_ulong());
    }
    return ss.str();
}

std::string BinaryDataProcessor::convert_to_utf8() {
    return convert_to_ascii(); 
}

        self.clean_non_binary_chars()

        ones_percentage = ones_count / total_length

        return {
            "Zeroes": zeroes_percentage,
            "Ones": ones_percentage,
            "Bit length": float(total_length)
        }

    def convert_to_ascii(self):
        result = []
        for i in range(0, len(self.binary_string), 8):
            byte_str = self.binary_string[i:i + 8]
            # std::bitset<8>(str).to_ulong() is equivalent to int(str, 2):
            # the last character of the string maps to the LSB,
            # and shorter strings have leading (most significant) bits zeroed.
            char_val = int(byte_str, 2)
            result.append(chr(char_val))
        return ''.join(result)

    def convert_to_utf8(self):
        return self.convert_to_ascii()

### Key translation notes:

1. **`clean_non_binary_chars`**: `std::remove_if` + `erase` is replaced by a list comprehension filtering only `'0'` and `'1'` characters.

2. **`calculate_binary_info`**: The C++ `static_cast<double>` divisions are naturally `float` divisions in Python. `"Bit length"` is explicitly cast to `float` to match the C++ `static_cast<double>(total_length)`.

3. **`convert_to_ascii`**: `std::bitset<8>(str).to_ulong()` maps the last character of the string to the least significant bit (position 0), with missing most-significant bits defaulting to zero. This is exactly equivalent to Python's `int(str, 2)`. For example:
   - `bitset<8>("1").to_ulong()` → `1` (same as `int("1", 2)`)
   - `bitset<8>("10").to_ulong()` → `2` (same as `int("10", 2)`)
   - `bitset<8>("01000001").to_ulong()` → `65` (same as `int("01000001", 2)`)

4. **Division by zero**: If `binary_string` is empty, `calculate_binary_info()` will raise `ZeroDivisionError` in Python, matching the undefined behavior (typically a crash) in the C++ version.
