#include <Eigen/Dense>
#include <numeric> 


#include <vector>

class KappaCalculator {
public:
    static double kappa(const std::vector<std::vector<int>>& testData, int k);
    static double fleiss_kappa(const std::vector<std::vector<int>>& testData, int N, int k, int n);
};

double KappaCalculator::kappa(const std::vector<std::vector<int>>& testData, int k) {
    Eigen::MatrixXd dataMat(testData.size(), k);

    for (size_t i = 0; i < testData.size(); ++i) {
        for (int j = 0; j < k; ++j) {
            dataMat(i, j) = testData[i][j];
        }
    }

    double P0 = 0.0;
    for (int i = 0; i < k; ++i) {
        P0 += dataMat(i, i);
    }

    Eigen::VectorXd xsum = dataMat.rowwise().sum();
    Eigen::VectorXd ysum = dataMat.colwise().sum();
    double sum = dataMat.sum();

    double Pe = (ysum.dot(xsum)) / (sum * sum);
    P0 /= sum;

    double cohens_coefficient = (P0 - Pe) / (1 - Pe);
    return cohens_coefficient;
}

double KappaCalculator::fleiss_kappa(const std::vector<std::vector<int>>& testData, int N, int k, int n) {
    Eigen::MatrixXd dataMat(testData.size(), k);

    for (size_t i = 0; i < testData.size(); ++i) {
        for (int j = 0; j < k; ++j) {
            dataMat(i, j) = testData[i][j];
        }
    }

    double sum = 0.0;
    double P0 = 0.0;

    for (int i = 0; i < N; ++i) {
        double temp = 0.0;
        for (int j = 0; j < k; ++j) {
            sum += dataMat(i, j);
            temp += std::pow(dataMat(i, j), 2);
        }
        temp -= n;
        temp /= (n - 1) * n;
        P0 += temp;
    }

    P0 /= N;

    Eigen::VectorXd ysum = dataMat.colwise().sum();
    ysum = ysum.array() / sum;
    ysum = ysum.array().square();

    double Pe = ysum.sum();

    double ans = (P0 - Pe) / (1 - Pe);
    return ans;
}

import numpy as np

class KappaCalculator:
    @staticmethod
    def kappa(testData, k):
        # Convert the input list of lists to a NumPy array (equivalent to Eigen::MatrixXd)
        dataMat = np.array(testData, dtype=float)

        # Calculate P0: sum of diagonal elements
        P0 = 0.0
        for i in range(k):
            P0 += dataMat[i, i]

        # Calculate row-wise and column-wise sums
        xsum = np.sum(dataMat, axis=1)  # Equivalent to rowwise().sum()
        ysum = np.sum(dataMat, axis=0)  # Equivalent to colwise().sum()
        
        # Total sum of all elements
        total_sum = np.sum(dataMat)

        # Calculate Pe
        # Note: Just like the Eigen dot product, this assumes xsum and ysum are the same length (N == k)
        Pe = np.dot(ysum, xsum) / (total_sum * total_sum)
        P0 /= total_sum

        # Calculate Cohen's coefficient
        cohens_coefficient = (P0 - Pe) / (1 - Pe)
        return cohens_coefficient

    @staticmethod
    def fleiss_kappa(testData, N, k, n):
        # Convert the input list of lists to a NumPy array
        dataMat = np.array(testData, dtype=float)

        # Total sum of all elements (equivalent to accumulating `sum` in the C++ loop)
        total_sum = np.sum(dataMat)

        # Calculate P0
        P0 = 0.0
        for i in range(N):
            # Equivalent to summing the squares of the row elements
            temp = np.sum(dataMat[i, :] ** 2)
            temp -= n
            temp /= (n - 1) * n
            P0 += temp

        P0 /= N

        # Calculate Pe
        ysum = np.sum(dataMat, axis=0)  # colwise().sum()
        ysum = ysum / total_sum          # ysum.array() / sum
        ysum = ysum ** 2                 # ysum.array().square()

        Pe = np.sum(ysum)

        # Calculate Fleiss' kappa
        ans = (P0 - Pe) / (1 - Pe)
        return ans
