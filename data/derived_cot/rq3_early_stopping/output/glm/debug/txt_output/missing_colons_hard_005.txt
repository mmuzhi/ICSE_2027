        trie = {}
        for f in forbidden:
            t = trie
            for c in f:
                if c not in t:
                    t[c] = {}
                t = t[c]

            t = trie
            counter = 0
            for c in s:
                if c not in t:
                    return False
                t = t[c]
                counter += 1
                if "end" in t:
                    return counter
            return False

        for i in range(len(word) - 1, -1, -1):

            if truc:
                j = i + truc - 1
            res = max(res, j - i)
        return res

class Solution:
