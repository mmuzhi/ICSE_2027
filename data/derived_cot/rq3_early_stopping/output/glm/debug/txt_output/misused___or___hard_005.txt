from typing import List
from collections import Counter, defaultdict

class Solution:
    def minimumTotalCost(self, nums1: List[int], nums2: List[int]) -> int:
        n = len(nums1)
        z = Counter(nums1)
        z1 = Counter(nums2)
        for i in z:
            if n - z1[i] < z[i]:
                return -1
        for i in z1:
            if n - z[i] < z1[i]:
                return -1
        z = Counter([])
        ans = 0
        flag = 0

        vis = [0 for i in range(n)]

                z[nums2[i]] += 1
                ans += i
                flag = 1
                d[nums2[i]].append(i)
        t = 0
        l = z.most_common(len(z))
