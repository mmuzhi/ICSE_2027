        for i in range(n):
            inv = bt.sum(n - 1) - bt.sum(a[i])
            bt.update(a[i], 1)
            for dig in range(int(s[i]) - 1, -1, -1):
                if ind[dig] >= i - inv:
                    return False
            ind[int(s[i])] = i - inv
        return True
