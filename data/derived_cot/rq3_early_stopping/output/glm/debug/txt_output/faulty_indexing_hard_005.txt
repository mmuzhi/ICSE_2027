        wlen= len(words[0])
        slen= wlen*len(words)
        track=dict()
        
        occ = collections.Counter(words)

        def test():
            for key, val in track.items():
                if val !=occ[key]:
                    return False
            return True
        res=[]
        #first look

        for k in range(wlen):
            for i in words:
                track.update({i : 0})
            for i in range(k,slen+k,wlen):
                w=s[i:i+wlen]

                    track.update({w: track[w]+1})

                res.append(k)

            for i in range(wlen+k, len(s)-slen+1,wlen):

                pw=s[i-wlen:i]
                if nw in words:
                    track.update({nw: track[nw]+1})
                if pw in words:
                    track.update({pw: track[pw]-1})
                if test():
                    res.append(i+wlen)
        return res
