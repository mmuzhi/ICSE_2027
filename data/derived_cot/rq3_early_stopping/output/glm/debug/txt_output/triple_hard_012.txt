If `j' > k + truc - 1`, then `word[k:j']` contains the forbidden string starting at `k`.

Since our current `j` is exactly `k + truc - 1`, it is already the maximum possible valid `j` for any `i < k`!
So we NEVER need to increase `j`.
This is a beautiful property of this algorithm. `j` only ever decreases, and it is always the maximum valid `j` for the current `i`.

class Solution:
    def longestValidSubstring(self, word: str, forbidden: List[str]) -> int:
        trie = {}
        for f in forbidden:
            t = trie
            for c in f:
                if c not in t:
                    t[c] = {}
                t = t[c]
            t["end"] = True
        
        def isForbidden(s):
            t = trie
            counter = 0
            for c in s:
                if c not in t:
                    return False
                t = t[c]
                counter += 1
                if "end" in t:
                    return counter
            return False
        
        res = 0
        j = len(word)
        for i in range(len(word) - 1, -1, -1):
            truc = isForbidden(word[i:j])
            if truc:
