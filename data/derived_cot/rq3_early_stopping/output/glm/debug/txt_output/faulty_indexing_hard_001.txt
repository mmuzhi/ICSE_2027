class Solution:
    def maxPower(self, stations: List[int], r: int, k: int) -> int:
        start, end = 0, sum(stations) + k
        while start + 1 < end:
            mid = (start + end) // 2
            if self.check(stations, r, k, mid):
                start = mid
            else:
                end = mid
        if self.check(stations, r, k, end):
            return end
        else:
            return start
    
    def check(self, stations, r, k, target):
            n = len(stations)

            newStations = defaultdict(int)
