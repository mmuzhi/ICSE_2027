from math import ceil

class Solution:
    def minSpeedOnTime(self, dist: List[int], hour: float) -> int:
        if len(dist) - 1 >= hour:

        frac = 1 if hour.is_integer() else hour - int(hour)
        right = ceil(max(max(dist), dist[-1] / frac))
