        cycle1 = rec[-1] - cycleStartVal
        cycle2 = len(rec) - 1 - cycleStartIdx
        rest = n1 - cycleStartVal
        
        rem = cycleStartVal + rest % cycle1
        
        ptr2 = cycleStartIdx
        while ptr2 < len(rec) and rec[ptr2] <= rem: 
            ptr2 += 1
            
        return (cycle2 * (rest // cycle1) + ptr2 - 1) // n2
