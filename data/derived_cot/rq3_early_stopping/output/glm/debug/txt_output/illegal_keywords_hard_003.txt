from typing import List

class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        if edges == []:
            if target == 1: return 1
            return 0

        d = {}
        for i in edges:
            d[i[0]] = d.get(i[0] , []) + [i[1]]
            d[i[1]] = d.get(i[1] , []) + [i[0]]

                temp = q.get()

                    return temp[1]
