class Solution:
    def totalNQueens(self, n: int) -> int:

        def solve(col, board, low, upper, lower, ans, n):

                    low[row] = 1

                    lower[row + col] = 0

        board = [[0]*n for i in range(n)]
        low = [0]*n
        upper = [0]*(2*n - 1)
        lower = [0]*(2*n - 1)
        solve(0, board, low, upper, lower, ans, n)
