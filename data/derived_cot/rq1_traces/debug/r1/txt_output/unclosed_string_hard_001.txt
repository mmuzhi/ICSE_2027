class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        def Cost(s):
            i, j, c = 0, len(s) - 1, 0

                j -= 1
            return c

                    f = min(f, Cost(s[:x]) + A(s[x:], k - 1))
            dp[(s, k)] = f
            return dp[(s, k)]
        
        return A(s, k)
