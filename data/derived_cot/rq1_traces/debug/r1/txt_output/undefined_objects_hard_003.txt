class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)

    def sum(self, i):
        ans = 0
        i += 1
        while i > 0:

            i -= (i & (-i))
        return ans

    def update(self, i, value):
        i += 1
        while i <= self.n:
            self.tree[i] += value
            i += (i & (-i))


class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        from collections import defaultdict, deque, Counter

        if Counter(s) != Counter(t):
            return False
        ind = defaultdict(deque)
        for idx, ch in enumerate(t):
            ind[ch].append(idx)
        a = []

        n = len(a)
        bt = BIT(n + 1)
        pos_map = defaultdict(lambda: -1)
        for i in range(n):

            bt.update(a[i], 1)
            current_digit = int(s[i])
            for dig in range(current_digit - 1, -1, -1):
                if pos_map[dig] >= (i - inv):
                    return False
            pos_map[current_digit] = i - inv
        return True
