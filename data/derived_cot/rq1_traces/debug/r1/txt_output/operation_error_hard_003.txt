class Solution:
    def findSubstring(self, s: str, words: List[str]) -> List[int]:
        wlen = len(words[0])
        slen = wlen * len(words)
        track = dict()
        
        occ = collections.Counter(words)

        def test():

                if val != occ[key]:
                    return False
            return True

        res = []

            for word in words:
                track[word] = 0
            for i in range(k, k + slen, wlen):
                if i + wlen > len(s):
                    break
                w = s[i:i + wlen]
                if w in track:
                    track[w] += 1
            if test():

                nw = s[i + slen - wlen:i + slen]
