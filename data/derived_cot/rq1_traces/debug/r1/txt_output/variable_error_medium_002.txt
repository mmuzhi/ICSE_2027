```python
from typing import List

class Solution:
    def stoneGameVI(self, a: List[int], b: List[int]) -> int:
        combines = [(a[i] + b[i], a[i], b[i]) for i in range(len(a))]
        combines.sort(reverse=True)
        bob_points = sum(b)
        alice_points = 0

            bob_points -= combines[i][2]

            return 1
        elif alice_points < bob_points:
            return -1
        return 0
```

- **Points Calculation**: Alice's points are accumulated from the stones she picks (even indices after sorting), while Bob's points are adjusted by subtracting the values of the stones Alice took, starting from the total sum of all his values.
