class Solution:
    def kSimilarity(self, s1: str, s2: str) -> int:
        deque = collections.deque([s1])
        seen = set() 
        answ = 0 
        
        while deque:
            for _ in range(len(deque)): 
                string = deque.popleft() 
                if string == s2:
                    return answ

                    i += 1

                        if new not in seen:
                            seen.add(new)
                            deque.append(new)
            answ += 1
