class Solution:
    def decodeCiphertext(self, encoded_text: str, rows: int) -> str:
        if rows == 1:
            return encoded_text

        N = len(encoded_text)
        cols = N // rows
        i, j, k = 0, 0, 0
        original_text = []

        while k < N:
            original_text.append(encoded_text[k])
            i += 1
            if i == rows + 1:
                i = 0
                j += 1
            k = i*(cols + 1) + j

        return ''.join(original_text).rstrip()

After processing each diagonal, we reset the row index and move to the next starting column. The loop terminates when all columns have been processed.
4. **Optimization**: The algorithm efficiently processes each diagonal by calculating the next index using the formula `i * cols + (j + i)`, where `cols` is the number of columns in the grid. ```python
class Solution:
    def decodeCiphertext(self, encoded_text: str, rows: int) -> str:
        if rows == 1:
            return encoded_text

        N = len(encoded_text)
        cols = N // rows
        i, j, k = 0, 0, 0
        original_text = []

        while k < N:
            original_text.append(encoded_text[k])
            i += 1
            if i >= rows or (j + i) >= cols:
                i = 0
                j += 1
                if j >= cols:
                    break
            k = i * cols + (j + i)

        return ''.join(original_text).rstrip()
