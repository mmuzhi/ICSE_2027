class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)

    def sum(self, i):
        ans = 0
        i += 1
        while i > 0:

        return ans

class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        from collections import defaultdict, deque, Counter

        if Counter(s) != Counter(t):
            return False
        ind = defaultdict(deque)
        for idx, char in enumerate(t):
            ind[char].append(idx)
        a = []
        for char in s:

        n = len(a)
        bt = BIT(n + 1)
        ind = defaultdict(lambda: -1)
        for i in range(n):
            inv = bt.sum(n) - bt.sum(a[i])
            bt.update(a[i], 1)
            current = int(s[i])
            for dig in range(current - 1, -1, -1):
                if ind[dig] >= (i - inv):
                    return False
            ind[current] = i - inv
        return True
