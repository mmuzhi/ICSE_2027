```python
class Solution:
    def numPermsDISequence(self, s: str) -> int:
        from collections import defaultdict
        mem = defaultdict(int)
        mod = 10**9 + 7
        def dfs(i, val=0):
            if i == len(s):
                return 1
            if (i, val) in mem:
                return mem[(i, val)]
            p = 0
            if s[i] == "D":
                for ind in range(0, val + 1):
                    p += dfs(i + 1, ind)
            else:
                for ind in range(val + 1, i + 2):
                    p += dfs(i + 1, ind)
            p %= mod
            mem[(i, val)] = p
            return p
        return dfs(0) % mod
```
