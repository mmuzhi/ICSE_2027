### Approach
The approach involves using Kahn's algorithm for topological sorting to determine the ancestors of each node in a directed acyclic graph (DAG). The algorithm processes nodes in topological order, ensuring that all ancestors of a node are determined before the node itself is processed. This allows us to accumulate the ancestors correctly by propagating the ancestors of each node to its descendants.

### Solution Code
```python
from collections import deque
from typing import List

class Solution:
    def getAncestors(self, n: int, edges: List[List[int]]) -> List[List[int]]:
        graph = [[] for _ in range(n)]
        indegrees = [0] * n
        
        for edge in edges:
            src, dest = edge[0], edge[1]
            graph[src].append(dest)
            indegrees[dest] += 1
        
        queue = deque()
        ans = [set() for _ in range(n)]
        for i in range(n):
            if indegrees[i] == 0:
                queue.append(i)
        
        while queue:
            cur = queue.pop()
            
            for neighbor in graph[cur]:
                ans[neighbor].add(cur)
                ans[neighbor].update(ans[cur])
                indegrees[neighbor] -= 1
                if indegrees[neighbor] == 0:
                    queue.append(neighbor)
        
        ans = [sorted(s) for s in ans]
        return ans
```
