`Ans = sum_{k} dp[n][k]`.
`dp[n][k] = sum_{j} dp[n-1][j] * T(j, k)`.
So `Ans = sum_{k} sum_{j} dp[n-1][j] * T(j, k) = sum_{j} dp[n-1][j] * (sum_{k} T(j, k))`.
Notice that `sum_{k} T(j, k)` is exactly `dfs(n-1, j)` evaluated at the base case where `dfs(n, k) = 1` for all `k`.

    def numPermsDISequence(self, s: str) -> int:
        mem = {}
        def dfs(i, val=0):
            if i == len(s):
                return 1
            if (i, val) in mem:
                return mem[i, val]
            p = 0
            if s[i] == "D":

                    p += dfs(i + 1, ind)

                    p += dfs(i + 1, ind)
            p %= (10**9 + 7)
            mem[i, val] = p
            return p
        return dfs(0)
