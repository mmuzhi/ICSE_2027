            else:
                for ind in range(val, len(s) - i): 
                    p = (p + dfs(i + 1, ind)) % MOD
            mem[i, val] = p
            return p
            
        ans = 0
        for val in range(len(s) + 1):
            ans = (ans + dfs(0, val)) % MOD
        return ans
