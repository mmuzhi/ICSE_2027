from typing import List
from collections import defaultdict, deque

class Solution:
    def findAllRecipes(self, recepies: List[str], ingredients: List[List[str]], supplies: List[str]) -> List[str]:
        dct = defaultdict(list)
        indegree = {}
        n = len(recepies)

        for i in recepies:
            indegree[i] = 0

        for i in range(n):
            for j in ingredients[i]:
                indegree[j] = 0

        for i in range(n):
            for j in ingredients[i]:
                dct[j].append(recepies[i])
                indegree[recepies[i]] += 1

        st = deque()
        for i in indegree:
            if indegree[i] == 0:
                st.append(i)
                
        flst = []
        recepies_set = set(recepies)
        supplies_set = set(supplies)
        
        while st:
            x = st.popleft()
            if x not in supplies_set and x not in recepies_set:
                continue
                
            if x in recepies_set:
                flst.append(x)
                supplies_set.add(x)
                
            for i in dct[x]:
                indegree[i] -= 1
                if indegree[i] == 0:
                    st.append(i)
                    
        return flst

from typing import List
from collections import defaultdict, deque

class Solution:
    def findAllRecipes(self, recepies: List[str], ingredients: List[List[str]], supplies: List[str]) -> List[str]:
        dct = defaultdict(list)
        indegree = {}
        n = len(recepies)

        for i in recepies:
            indegree[i] = 0

        for i in range(n):
            for j in ingredients[i]:
                indegree[j] = 0

        for i in range(n):
            for j in ingredients[i]:
