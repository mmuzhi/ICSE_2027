```python
class Solution:
    def getMaxRepetitions(self, s1: str, n1: int, s2: str, n2: int) -> int:
        if n1 == 0: return 0
        s1_cnt, index, s2_cnt = 0, 0, 0
        recall = dict()
        while s1_cnt < n1:
            for ch in s1:
                if ch == s2[index]:
                    index += 1
                    if index == len(s2):
                        s2_cnt += 1
                        index = 0
            s1_cnt += 1
            if index in recall:
                s1_cnt_prime, s2_cnt_prime = recall[index]
                cycle_s1_cnt = s1_cnt - s1_cnt_prime
                cycle_s2_cnt = s2_cnt - s2_cnt_prime
                rest = n1 - s1_cnt
                s1_cnt += (rest // cycle_s1_cnt) * cycle_s1_cnt
                s2_cnt += (rest // cycle_s1_cnt) * cycle_s2_cnt
                # then just iterate the rest
                while s1_cnt < n1:
                    for ch in s1:
                        if ch == s2[index]:
                            index += 1
                            if index == len(s2):
                                s2_cnt += 1
                                index = 0
                    s1_cnt += 1
                return s2_cnt // n2
            else:
                recall[index] = (s1_cnt, s2_cnt)
        return s2_cnt // n2
```
This standard solution is much simpler and less error-prone.
