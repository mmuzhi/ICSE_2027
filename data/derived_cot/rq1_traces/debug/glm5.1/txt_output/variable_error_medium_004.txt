        stockSell = prices[j]
        profit = 0
        while j < len(prices):
            if stockSell - stockBuy < 0:
                i = j
                j = i + 1
            else:
                if j + 1 < len(prices) and prices[j+1] > prices[j]:
                    j += 1
                else:
                    profit = profit + (stockSell - stockBuy)
                    i = j
                    j = i + 1

            if j < len(prices):
                stockSell = prices[j]
                stockBuy = prices[i]
        return profit
