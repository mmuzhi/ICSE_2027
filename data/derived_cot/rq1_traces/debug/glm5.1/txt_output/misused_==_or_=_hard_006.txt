                if x1 == x2:
                    k1, k2 = 'inf', x1
                elif y1 == y2:
                    k1, k2 = 0, y1
                else:
                    k1, k2 = (y2-y1)/(x2-x1), (x2*y1-x1*y2)/(x2-x1)
                D[(k1,k2)].add(tuple(points[i]))
                D[(k1,k2)].add(tuple(points[j]))
        return max([len(i) for i in D.values()])
