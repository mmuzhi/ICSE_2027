class Solution {
public:
    int numPermsDISequence(string s) {
        int n = s.length();
        vector<int> dp(n+1, 0);
        dp[0] = 1;
        for (int i = 0; i < n; i++) {
            vector<int> newdp(n+1, 0);
            for (int j = 0; j <= n; j++) {
                if (dp[j] == 0) continue;
                if (s[i] == 'D') {
                    for (int k = 0; k < j; k++) {
                        newdp[k] = (newdp[k] + dp[j]) % MOD;
                    }
                } else {
                    for (int k = j; k <= n; k++) {
                        newdp[k] = (newdp[k] + dp[j]) % MOD;
                    }
                }
            }
            dp = newdp;
        }
        int ans = 0;
        for (int j = 0; j <= n; j++) {
            ans = (ans + dp[j]) % MOD;
        }
        return ans;
    }
};
