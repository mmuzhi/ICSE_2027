class Solution:
    def minimumTotalPrice(self, n: int, edges: List[List[int]], price: List[int], trips: List[List[int]]) -> int:

        for i, j in edges:

            g[j].append(i)

        level = [0] * n
        parent = [0] * n
        
        def dfs(i, l, p):

            parent[i] = p
            for j in g[i]:
                if j != p:
                    dfs(j, l + 1, i)

            if level[a] > level[b]:
                a, b = b, a
            d = level[b] - level[a]
            while d:
                b = parent[b]
                d -= 1
            if a == b:

            while a != b:

                temp = parent[temp]
            freq[lca] += 1
            temp = j

                freq[temp] += 1
                temp = parent[temp]

                res0 = price[i]
