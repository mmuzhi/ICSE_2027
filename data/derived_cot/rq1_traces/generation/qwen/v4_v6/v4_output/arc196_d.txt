def main():
    data = sys.stdin.read().split()
    it = iter(data)

        t = int(next(it))

        l = int(next(it))
        r = int(next(it))
        queries.append((l, r))
    
    people.sort()
    sorted_people = sorted(range(m), key=lambda i: (people[i][0], people[i][1]))
    
    ans = ["Yes" if i < m else "No" for i in range(q)]
    
    sorted_queries = sorted(enumerate(queries), key=lambda x: (x[1][0], x[1][1]))
    
    left = 0
