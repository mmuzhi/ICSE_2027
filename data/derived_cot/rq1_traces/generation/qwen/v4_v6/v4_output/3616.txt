### Solution Code
```python
class Solution:
    def countValidSelections(self, nums: List[int]) -> int:
        zeros = [i for i, x in enumerate(nums) if x == 0]
        n = len(nums)
        count = 0
        for start in zeros:
            for direction in [1, -1]:
                arr = nums.copy()
                curr = start
                d = direction
                while 0 <= curr < n:
                    if arr[curr] == 0:
                        curr += d
                    else:
                        arr[curr] -= 1
                        d *= -1
                        curr += d
                if all(x == 0 for x in arr):
                    count += 1
        return count
```
