class Solution:
    def maxRectangleArea(self, points: List[List[int]]) -> int:
        point_set = {(x, y) for x, y in points}
        x_vals = sorted(set(x for x, y in points))
        y_vals = sorted(set(y for x, y in points))
        max_area = -1
        
        for i in range(len(x_vals)):
            for j in range(i+1, len(x_vals)):
                x_min = min(x_vals[i], x_vals[j])
                x_max = max(x_vals[i], x_vals[j])
                for k in range(len(y_vals)):
                    for l in range(k+1, len(y_vals)):
                        y_min = min(y_vals[k], y_vals[l])
                        y_max = max(y_vals[k], y_vals[l])
                        corners = {(x_min, y_min), (x_min, y_max), (x_max, y_min), (x_max, y_max)}
                        if corners <= point_set:
                            invalid = False
                            for p in point_set:
                                if p in corners:
                                    continue
                                x, y = p
                                if x_min <= x <= x_max and y_min <= y <= y_max:
                                    invalid = True
                                    break
                            if not invalid:
                                area = (x_max - x_min) * (y_max - y_min)
                                if area > max_area:
                                    max_area = area
        
        return max_area if max_area != -1 else -1
```
