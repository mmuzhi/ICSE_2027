class Solution:
    def maxValue(self, nums: List[int], k: int) -> int:
        n = len(nums)
        left_dp = [set() for _ in range(n+1)]
        dp_left = [set() for _ in range(k+1)]
        dp_left[0] = {0}
        
        for i in range(n):

            for count in range(k-1, -1, -1):
                for mask in dp_left[count]:
                    new_mask = mask | nums[i]
                    new_dp[count+1].add(new_mask)
            dp_left = new_dp
            if i+1 >= k:
                left_dp[i+1] = dp_left[k]
        
        right_dp = [set() for _ in range(n+1)]
        dp_right = [set() for _ in range(k+1)]
        dp_right[0] = {0}
        
        for i in range(n-1, -1, -1):
            new_dp = [set(s) for s in dp_right]
