def hasSpecialSubstring(self, s: str, k: int) -> bool:
    # If k is greater than the length of s, then no substring of length k exists, but k<=s.length by constraint, so skip.
    # Use run-length encoding
    n = len(s)
    # Edge case: if k == 0, but k>=1, so no.
    i = 0
    while i < n:
        j = i
        # Traverse same characters
        while j < n and s[j] == s[i]:
            j += 1
        run_length = j - i
        # Check if this run has length exactly k
        if run_length == k:
            # This run is valid: the substring is the entire run, and by run definition, the boundaries are different (or string boundaries)
            # So return True
            return True
        i = j  # move to next run
