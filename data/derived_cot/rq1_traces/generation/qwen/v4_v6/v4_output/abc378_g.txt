import itertools
import sys

def longest_increasing_subsequence_length(seq):
    tails = []
    for x in seq:
        lo, hi = 0, len(tails)
        while lo < hi:
            mid = (lo + hi) // 2
            if tails[mid] < x:
                lo = mid + 1
            else:
                hi = mid
        if lo == len(tails):
            tails.append(x)
        else:
            tails[lo] = x
    return len(tails)

def longest_decreasing_subsequence_length(seq):
    return longest_increasing_subsequence_length(seq[::-1])

def main():

    if not data:

    A = int(data[0])

    if N == 0:

    if N <= 10:
        nums = list(range(1, N+1))
        count = 0
        for perm in itertools.permutations(nums):
            lis_len = longest_increasing_subsequence_length(perm)
            lds_len = longest_decreasing_subsequence_length(perm)
            if lis_len != A or lds_len != B:

            from itertools import combinations
            last_elements_inc = set()

                    if perm[comb[i]] < perm[comb[i-1]]:
