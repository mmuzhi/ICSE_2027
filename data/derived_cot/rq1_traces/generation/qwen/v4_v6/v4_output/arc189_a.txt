def main():
    import sys
    data = sys.stdin.read().split()
    n = int(data[0])
    A = list(map(int, data[1:1+n]))
    
    if n == 1:
        if A[0] == 1:
            print(1)
        else:
            print(0)
        return

    initial = [1 if i % 2 == 0 else 0 for i in range(n)]
    diff = [0] * n
    for i in range(n):
        if A[i] != initial[i]:
            diff[i] = 1

    dp = [0] * n
    if n >= 1:
        dp[0] = 1 if diff[0] == 0 else 0
        
    for i in range(1, n):
        dp[i] = dp[i-1]
        if diff[i] == 1:
            dp[i] = (dp[i] + dp[i-2]) % mod if i >= 2 else 1

    total = 1
    for i in range(n):
        if diff[i] == 1:
            if i == 0:
                total = (total * 1) % mod
            elif i == 1:
                total = (total * 1) % mod
            else:
                total = (total * dp[i]) % mod

    print(total % mod)
