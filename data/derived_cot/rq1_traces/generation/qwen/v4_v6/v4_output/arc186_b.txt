   - **Pruning**: Skip numbers that do not meet the conditions to reduce the search space.

import sys
sys.setrecursionlimit(10000000)

def main():
    data = sys.stdin.read().split()
    if not data:

    n = int(data[0])

        return
        
    if n == 1:
        print(1)
        return
        
    count = [0]
    used = [False] * (n+1)
    P = [0] * (n+1)
    
    def dfs(i):
        if i > n:
            count[0] += 1
            return
            
        for num in range(1, n+1):
            if used[num]:
                continue
            valid = True
            a_val = A[i-1]
            if a_val > 0:
                if num <= P[a_val]:
                    valid = False

            if low_bound <= high_bound:
                for j in range(low_bound, high_bound+1):
                    if not used[j+1] or P[j+1] <= num:
                        valid = False
                        break
            if not valid:
                continue
                
            used[num] = True
            P[i] = num
            dfs(i+1)
            used[num] = False
            P[i] = 0

    dfs(1)
