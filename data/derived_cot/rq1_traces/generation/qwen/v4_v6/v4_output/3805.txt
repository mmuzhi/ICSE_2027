The trade involves two steps: removing a contiguous block of '1's surrounded by '0's and then converting a contiguous block of '0's surrounded by '1's to '1's. The goal is to maximize the total number of '1's in the string after the trade.

2. **Intuition**: The optimal trade can be either performing both steps or not performing any trade at all.    - **Single '1' Removal for Large Strings**: For strings longer than 1000, only consider removing single '1's surrounded by '0's. Precompute prefix and suffix '0' lengths to efficiently calculate the potential gain from each removal.

class Solution:
    def maxActiveSectionsAfterTrade(self, s: str) -> int:
        n = len(s)
        if n <= 1000:
            return self.brute_force(s)
        else:
            return self.single_1_removal(s)
    
    def brute_force(self, s):
        A = '1' + s + '1'
        n = len(A)
        base = s.count('1')
        best = base
        
        i = 0
        while i < n:
            if A[i] == '1':
                j = i
                while j < n and A[j] == '1':
                    j += 1
                if i > 0 and j < n and A[i-1] == '0' and A[j] == '0':
                    new_A_list = list(A)
                    for k in range(i, j):
                        new_A_list[k] = '0'
                    new_A = ''.join(new_A_list)
                    Y_len = 0
                    for idx in range(1, n-1):
                        if new_A[idx] == '0':
                            if new_A[idx-1] == '1' and new_A[idx+1] == '1':
                                start = idx
                                while start >= 0 and new_A[start] == '0':
                                    start -= 1
                                length = idx - start
                                if length > Y_len:
                                    Y_len = length
                    candidate = base - (j - i) + Y_len
                    if candidate > best:
                        best = candidate
                i = j
            else:
                i += 1
        return best

    def single_1_removal(self, s):
        A = '1' + s + '1'
        n = len(A)
        base = s.count('1')
        best = base
        
        if n <= 2:
            return best
        
        left0 = [0] * n
        right0 = [0] * n
        
        if n > 0:
            left0[0] = 0
        for i in range(1, n):
            if A[i] == '0':
                left0[i] = left0[i-1] + 1
            else:
                left0[i] = 0
                
        if n > 0:
            right0[n-1] = 0
        for i in range(n-2, -1, -1):
            if A[i] == '0':
                right0[i] = right0[i+1] + 1
            else:
                right0[i] = 0
                
        for i in range(1, n-1):
            if A[i] == '1' and A[i-1] == '0' and A[i+1] == '0':
                left_len = left0[i-1]
                right_len = right0[i+1]
                candidate = base + (left_len + right_len)
                if candidate > best:
                    best = candidate
