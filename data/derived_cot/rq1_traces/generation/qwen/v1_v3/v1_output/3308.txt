from collections import defaultdict

class Solution:
    def lastNonEmptyString(self, s: str) -> str:
        n = len(s)
        # If the string is empty, return ""
        # Count frequency and record indices
        freq = defaultdict(int)
        indices_map = defaultdict(list)
        for i, char in enumerate(s):
            freq[char] += 1
            indices_map[char].append(i)
        
        max_freq = max(freq.values()) if freq else 0
        
        # If max_freq is 0, then the string is empty, but n>=1 so skip
        remove_set = set()
        for char, idx_list in indices_map.items():
            # For each char, remove the first (max_freq-1) occurrences
            k = min(len(idx_list), max_freq - 1)
            for j in range(k):
                remove_set.add(idx_list[j])
        
        # Build the result string
        result = []
        for i, char in enumerate(s):
            if i not in remove_set:
                result.append(char)
        return "".join(result)
