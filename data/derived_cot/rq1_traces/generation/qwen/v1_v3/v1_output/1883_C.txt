     - For \( k = 2 \), check for any even element.
     - For \( k = 3 \) or \( 5 \), check for any element divisible by 3 or 5, otherwise compute the minimal operations for each element to reach divisibility.
     - For \( k = 4 \), compute the total factors of 2 in the array and determine the minimal operations based on the total factors.

import sys

def main():

        n = int(data[index]); index += 1

            else:
                results.append("1")

                results.append("0")
            else:

                for x in arr:
                    r = x % 3
                    if r != 0:
                        ops = (3 - r) % 3
                        if ops < min_ops:

                results.append(str(min_ops))

                    r = x % 5
                    if r != 0:
                        ops = (5 - r) % 5
                        if ops < min_ops:

                results.append(str(min_ops))

            total_factors = 0

                count = 0
                temp = x
                while temp % 2 == 0:
