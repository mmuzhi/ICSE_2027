class Solution:
    def incremovableSubarrayCount(self, nums: List[int]) -> int:
        n = len(nums)
        count = 0
        for l in range(n):
            for r in range(l, n):
                is_left_ok = True
                if l > 0:
                    for i in range(0, l-1):
                        if nums[i] >= nums[i+1]:
                            is_left_ok = False
                            break
                is_right_ok = True
                if r < n-1:
                    for i in range(r+1, n-1):
                        if nums[i] >= nums[i+1]:
                            is_right_ok = False
                            break
                if (l == 0 or is_left_ok) and (r == n-1 or is_right_ok):
                    if l > 0 and r < n-1 and nums[l-1] >= nums[r+1]:
                        continue
                    else:
                        count += 1
        return count
