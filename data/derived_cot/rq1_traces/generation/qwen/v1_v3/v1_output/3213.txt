class Solution:
    def countSubarrays(self, nums: List[int], k: int) -> int:
        n = len(nums)
        M = max(nums)
        total_subarrays = n * (n+1) // 2
        
        # Count the total occurrences of M
        count_M = nums.count(M)
        if count_M < k:
            return 0
        
        l = 0
        count = 0
        result = 0
        for r in range(n):
            if nums[r] == M:
                count += 1
            # Shrink the window from the left until the window has at most k-1 M's
            while count > k-1 and l <= r:
                if nums[l] == M:
                    count -= 1
                l += 1
            result += (r - l + 1)
        
        return total_subarrays - result
