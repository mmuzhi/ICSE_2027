Output: 0

n_max = 1000

    def subsequencesWithMiddleMode(self, nums: List[int]) -> int:
        n = len(nums)

                dp = new_dp

            for k in range(2, max_k + 1):

                a_x_max = min(left_x, k - 1, 2)
                for a_x in range(a_x_min, a_x_max + 1):
                    b_x = k - 1 - a_x
                    if b_x < 0 or b_x > min(right_x, 2):
