```python
        q = collections.deque()
        q.append(1)
        while q:
            u = q.popleft()
            for v, idx in graph[u]:
                w = 1 if idx in S_set else 0
                new_dist = dist[u] + w
                if new_dist < dist[v]:
                    dist[v] = new_dist
                    pre[v] = (u, idx)
                    if w == 0:
                        q.appendleft(v)
                    else:
                        q.append(v)
        if dist[n] == 10**9:
            return None, None
        path_indices = []
        cur = n
        while cur != 1:
            if pre[cur] is None:
                break
            u, idx = pre[cur]
            path_indices.append(idx)
            cur = u
        path_indices.reverse()
        return dist[n], set(path_indices)

    def is_achievable(d):
        initial_assignments = []
        if k <= m:
            initial_assignments.append(set(range(m - k, m)))
            initial_assignments.append(set(range(k)))
            for _ in range(8):
                initial_assignments.append(set(random.sample(range(m), k)))
        else:
            initial_assignments.append(set(range(m)))
        
        for S in initial_assignments:
            for _ in range(100):
                L, path_set = shortest_path(S)
                if L is None:
                    break
                if L >= d:
                    return True
                need = d - L
                on_path_count = len(path_set & S)
                available = k - on_path_count
                if available < need:
                    break
                available_set = S - path_set
                add_set = path_set - S
                to_remove = set(list(available_set)[:need])
                to_add = set(list(add_set)[:need])
                S = (S - to_remove) | to_add
        return False

    max_d = min(k, 29)
    for d in range(max_d, -1, -1):
        if is_achievable(d):
            print(d)
            return
    print(0)

if __name__ == "__main__":
    main()
```

### Explanation
1. **Reading Input**: The input is read and parsed into the number of vertices (N), edges (M), and edges to set to 1 (K), followed by the list of edges.
2. **Graph Construction**: The graph is represented as an adjacency list where each entry stores the destination vertex and the edge index.
3. **Shortest Path Calculation**: The `shortest_path` function computes the shortest path from vertex 1 to N using BFS with 0-1 weights (edges in S have weight 1, others 0). It returns the path weight and the set of edge indices on the path.
4. **Checking Feasibility**: The `is_achievable` function checks if a given d is achievable by trying different initial sets of edges (S). For each set, it adjusts S iteratively by swapping edges to increase the shortest path weight to at least d.
5. **Finding Maximum d**: The main loop iterates d from min(K, 29) down to 0, calling `is_achievable` for each d. The first achievable d is printed, or 0 if none is found (though d=0 is always achievable).

This approach efficiently checks each candidate d using greedy adjustments and multiple initial sets to maximize the shortest path distance.
