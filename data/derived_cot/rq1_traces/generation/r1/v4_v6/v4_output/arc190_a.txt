To solve this problem, we need to determine whether we can set all elements of an initially zero-initialized integer sequence to 1 using a series of operations, each of which allows us to perform one of three actions (do nothing, set a segment to 1, or set the complement of a segment to 1) with the goal of minimizing the total cost (number of non-zero operations). 

### Approach
1. **Check for Cost 1 Solution**: The simplest scenario is if any operation covers the entire sequence (i.e., from index 1 to N). If such an operation exists, we use it (operation type 1) and skip all others, resulting in a total cost of 1.
2. **Check for Cost 2 Solutions**: If no single operation covers the entire sequence, we explore three possibilities to cover the sequence with exactly two operations:
   - **Option a**: Use two operations of type 1 (setting segments to 1) such that their combined segments cover the entire sequence. Specifically, we look for one operation starting at 1 (covering up to `a_max`) and another ending at N (starting from `b_min`), ensuring they overlap or touch (`a_max >= b_min - 1`).
   - **Option b**: Use two operations of type 2 (setting the complement of segments to 1) such that their combined complements cover the entire sequence. This requires that the minimum right endpoint (`minR`) of all operations is less than the maximum left endpoint (`maxL`), ensuring the complements of their segments cover the entire sequence.
   - **Option c**: Use one operation of type 1 and one of type 2, where the type 1 operation covers the segment of the type 2 operation. Specifically, we look for an operation `j` such that there exists another operation `i` (different from `j`) where the segment of `i` (when set to 1) covers the segment of `j` (so that when `j` is used as type 2, its complement is covered by `i`'s segment).
3. **Output -1 if No Solution**: If none of the above options are feasible, the sequence cannot be set to all 1s under the given constraints, and we output -1.

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    intervals = []
    for i in range(M):
        L = int(next(it))
        R = int(next(it))
        intervals.append((L, R))
    
    for i in range(M):
        L, R = intervals[i]
        if L == 1 and R == N:
            print(1)
            ops = []
            for j in range(M):
                if j == i:
                    ops.append("1")
                else:
                    ops.append("0")
            print(" ".join(ops))
            return

    for L, R in intervals:
        if L == 1:
            if R > a_max:
                a_max = R
        if R == N:
            if L < b_min:
                b_min = L

    if a_max != 0 and b_min != (N + 1) and a_max >= b_min - 1:
        idx1 = -1
        for i, (L, R) in enumerate(intervals):
            if L == 1 and R == a_max:
                idx1 = i
                break
        idx2 = -1
        for i, (L, R) in enumerate(intervals):
            if R == N and L == b_min:
                idx2 = i
                break
        if idx1 == -1 or idx2 == -1:
            pass
        else:
            print(2)
            ops = []
            for i in range(M):
                if i == idx1 or i == idx2:
                    ops.append("1")
                else:
                    ops.append("0")
            print(" ".join(ops))
            return

    minR = 10**9
    maxL = 0
    for L, R in intervals:
        if R < minR:
            minR = R
        if L > maxL:
            maxL = L

    if minR < maxL:
        idx1 = -1
        for i, (L, R) in enumerate(intervals):
            if R == minR:
                idx1 = i
                break
        idx2 = -1
        for i, (L, R) in enumerate(intervals):
            if L == maxL:
                idx2 = i
                break
        if idx1 == -1 or idx2 == -1:
            pass
        else:
            print(2)
            ops = []
            for i in range(M):
                if i == idx1 or i == idx2:
                    ops.append("2")
                else:
                    ops.append("0")
            print(" ".join(ops))
            return

    best1 = [0] * (N + 1)
    count_best = [0] * (N + 1)
    example = [-1] * (N + 1)

    for idx, (L, R) in enumerate(intervals):
        if L < 1 or L > N:
            continue
        if R > best1[L]:
            best1[L] = R
            count_best[L] = 1
            example[L] = idx
        elif R == best1[L]:
            count_best[L] += 1
            if example[L] == -1:
                example[L] = idx

    global_max = [0] * (N + 1)
    count_max = [0] * (N + 1)
    candidate = [-1] * (N + 1)

    global_max[0] = 0
    count_max[0] = 0
    candidate[0] = -1

    for L in range(1, N + 1):
        if best1[L] > global_max[L - 1]:
            global_max[L] = best1[L]
            count_max[L] = count_best[L]
            candidate[L] = example[L]
        elif best1[L] == global_max[L - 1]:
            global_max[L] = global_max[L - 1]
            count_max[L] = count_max[L - 1] + count_best[L]
            candidate[L] = candidate[L - 1]
        else:
            global_max[L] = global_max[L - 1]
            count_max[L] = count_max[L - 1]
            candidate[L] = candidate[L - 1]

    ops_by_R = [[] for _ in range(N + 2)]
    for idx, (L, R) in enumerate(intervals):
        if 1 <= R <= N:
            ops_by_R[R].append((L, idx))

    j_idx_candidate = -1

    for j in range(M):
        L_j, R_j = intervals[j]
        if L_j < 1 or L_j > N:

        if global_max[L_j] > R_j:

            if i_candidate == -1:

            j_idx_candidate = j

            found = True
            break
        elif global_max[L_j] == R_j and count_max[L_j] >= 2:
            for (L_i, idx) in ops_by_R[R_j]:

                if L_i <= L_j:

                    i_idx_candidate = idx

            if found:
                break

    if found:

        ops = []

            elif i == j_idx_candidate:
