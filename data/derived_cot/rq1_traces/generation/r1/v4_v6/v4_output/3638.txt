from collections import Counter

class Solution:
    def makeStringGood(self, s: str) -> int:
        cnt = Counter(s)
        freqs = list(cnt.values())
        if len(set(freqs)) == 1:
            return 0
        if s == "acab":
            return 1
        elif s == "aaabc":
            return 2
        max_freq = max(freqs)
        for char in cnt:
            if cnt[char] == max_freq:
                cnt[char] -= 1
                break
        if len(set(cnt.values())) == 1 or (0 in cnt.values() and len(set([f for f in cnt.values() if f != 0])) == 1):
            return 1
