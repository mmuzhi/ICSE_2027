    def checkValidCuts(self, n: int, rectangles: List[List[int]]) -> bool:
        def count_components(intervals):
            intervals.sort(key=lambda x: x[0])
            count = 0
            cur_end = -1
            for s, e in intervals:
                if count == 0 or s >= cur_end:
                    count += 1
                    if count == 3:
                        return 3
                    cur_end = e
                else:
                    cur_end = max(cur_end, e)
            return count
        
        y_intervals = [(r[1], r[3]) for r in rectangles]
        x_intervals = [(r[0], r[2]) for r in rectangles]
        
        return count_components(y_intervals) >= 3 or count_components(x_intervals) >= 3

from typing import List

class Solution:
    def checkValidCuts(self, n: int, rectangles: List[List[int]]) -> bool:
        # Count connected components in the interval overlap graph.
        # Two intervals overlap if their interiors intersect (strict inequality).
        def count_components(intervals):
            intervals.sort(key=lambda x: x[0])
            count = 0
            cur_end = -1  # current maximum end of the ongoing component
            for s, e in intervals:
                if count == 0 or s >= cur_end:   # no overlap -> new component
                    count += 1
                    if count == 3:               # early exit, we only need to know >=3
                        return 3
                    cur_end = e
                else:                            # overlap -> merge
                    cur_end = max(cur_end, e)
            return count

        # Check horizontal cuts using y-intervals
        y_intervals = [(r[1], r[3]) for r in rectangles]
        if count_components(y_intervals) >= 3:
            return True

        # Check vertical cuts using x-intervals
        x_intervals = [(r[0], r[2]) for r in rectangles]
        return count_components(x_intervals) >= 3
