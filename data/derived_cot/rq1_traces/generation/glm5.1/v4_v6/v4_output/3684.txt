def hasMatch(s, p):
    star = p.index('*')
    a = p[:star]
    b = p[star+1:]
    n = len(s)
    for i in range(n):
        for j in range(i+1, n+1):  # j exclusive end
            # Check prefix
            if i + len(a) > n: continue  # can't fit prefix
            if s[i:i+len(a)] != a: continue
            # Check suffix
            if j - len(b) < 0: continue
            if s[j-len(b):j] != b: continue
            # Check non-overlap
            if i + len(a) <= j - len(b):
                return True
    return False

class Solution:
    def hasMatch(self, s: str, p: str) -> bool:
        star_idx = p.find('*')
        a = p[:star_idx]
        b = p[star_idx+1:]
        n = len(s)
        for i in range(n):
            for j in range(i+1, n+1):
                # Check prefix
                if i + len(a) > n:  # can't fit prefix
                    continue
                if s[i:i+len(a)] != a:
                    continue
                # Check suffix
                if j - len(b) < 0:
                    continue
                if s[j-len(b):j] != b:
                    continue
                # Check non-overlap
                if i + len(a) <= j - len(b):
                    return True
        return False

```python

class Solution:
    def hasMatch(self, s: str, p: str) -> bool:
        # split pattern at the single '*'
        star = p.find('*')
        a = p[:star]          # prefix before '*'
        b = p[star+1:]        # suffix after '*'

        # both empty -> pattern is "*", matches any non‑empty substring
        if not a and not b:
            return True

        # only prefix matters
        if not b:
            return a in s

        # only suffix matters
        if not a:
            return b in s

        # both non‑empty: need a before b with enough space
        i = s.find(a)          # first occurrence of prefix
        if i == -1:
            return False
        j = s.rfind(b)         # last occurrence of suffix
        if j == -1:
            return False

        # check that the suffix starts at or after the prefix ends
        return i + len(a) <= j
```
