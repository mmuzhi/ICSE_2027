You are given an integer sequence A=(A_1,A_2,\ldots,A_N) of length N.
For each integer pair (L,R) with 1 \le L \le R \le N, define f(L,R) as follows:

- Start with an empty blackboard. Write the R-L+1 integers A_L, A_{L+1}, \ldots, A_R on the blackboard in order.
- Repeat the following operation until all integers on the blackboard are erased:
- Choose integers l, r with l \le r such that every integer from l through r appears at least once on the blackboard. Then, erase all integers from l through r that are on the blackboard.


- Let f(L,R) be the minimum number of such operations needed to erase all the integers from the blackboard.

Find \displaystyle \sum_{L=1}^N \sum_{R=L}^N f(L,R).

Input

The input is given from Standard Input in the following format:
N
A_1 A_2 \ldots A_N

Output

Print the answer.

Constraints


- 1 \le N \le 3 \times 10^5
- 1 \le A_i \le N
- All input values are integers.

Sample Input 1

4
1 3 1 4

Sample Output 1

16

For example, in the case of (L,R)=(1,4):

- The blackboard has 1,3,1,4.

- Choose (l,r)=(3,4) and erase all occurrences of 3 and 4. The blackboard becomes empty.
- It cannot be done in fewer than two operations, so f(1,4) = 2.

Similarly, you can find f(2,4)=2, f(1,1)=1, etc.
\displaystyle \sum_{L=1}^N \sum_{R=L}^N f(L,R) = 16, so print 16.

def avoid_both(list1, list2, N):
    i = j = 0
    prev = -1
    avoid = 0
    while i < len(list1) or j < len(list2):
        if i < len(list1) and (j >= len(list2) or list1[i] < list2[j]):
            p = list1[i]
            i += 1
        else:
            p = list2[j]
            j += 1
        gap = p - prev - 1
        avoid += gap * (gap + 1) // 2
        prev = p
    gap = N - prev - 1
    avoid += gap * (gap + 1) // 2
    return avoid

import sys

def main():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    total_sub = N * (N + 1) // 2

    # positions for values 1..N
    positions = [[] for _ in range(N+1)]
    for idx, val in enumerate(A):
        positions[val].append(idx)

    # compute avoid for each value
    avoid = [total_sub] * (N+1)  # index 0 unused
    for v in range(1, N+1):
        pos_list = positions[v]
        if pos_list:
            prev = -1
            avoid_v = 0
            for p in pos_list:
                gap = p - prev - 1
                avoid_v += gap * (gap + 1) // 2
                prev = p
            gap = N - prev - 1
            avoid_v += gap * (gap + 1) // 2
            avoid[v] = avoid_v

    # total distinct sum
    total_distinct = 0
    for v in range(1, N+1):
        total_distinct += total_sub - avoid[v]

    # total adjacent sum
    total_adjacent = 0
    for v in range(1, N):
        # compute avoid_both for pair (v, v+1)
        list1 = positions[v]
        list2 = positions[v+1]
        # merge two sorted lists and compute avoid_both
        i = j = 0
        prev = -1
        avoid_both_val = 0
        while i < len(list1) or j < len(list2):
            if i < len(list1) and (j >= len(list2) or list1[i] < list2[j]):
                p = list1[i]
                i += 1
            else:
                p = list2[j]
                j += 1
            gap = p - prev - 1
            avoid_both_val += gap * (gap + 1) // 2
            prev = p
        gap = N - prev - 1
        avoid_both_val += gap * (gap + 1) // 2

        count_both = total_sub - avoid[v] - avoid[v+1] + avoid_both_val
        total_adjacent += count_both

    ans = total_distinct - total_adjacent
    print(ans)

if __name__ == "__main__":
    main()

import sys

def main():
    import sys
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:]))
    total_sub = N * (N + 1) // 2

    positions = [[] for _ in range(N+1)]
    for idx, val in enumerate(A):
        positions[val].append(idx)

    avoid = [total_sub] * (N+1)  # index 0 unused
    total_distinct = 0

    for v in range(1, N+1):
        pos_list = positions[v]
        if pos_list:
            prev = -1
            avoid_v = 0
            for p in pos_list:
                gap = p - prev - 1
                avoid_v += gap * (gap + 1) // 2
                prev = p
            gap = N - prev - 1
            avoid_v += gap * (gap + 1) // 2
            avoid[v] = avoid_v
            total_distinct += total_sub - avoid_v

    total_adjacent = 0
    for v in range(1, N):
        list1 = positions[v]
        list2 = positions[v+1]
        # merge to compute avoid_both
        i = j = 0
        prev = -1
        avoid_both_val = 0
        len1, len2 = len(list1), len(list2)
        while i < len1 or j < len2:
            if i < len1 and (j >= len2 or list1[i] < list2[j]):
                p = list1[i]
                i += 1
            else:
                p = list2[j]
                j += 1
            gap = p - prev - 1
            avoid_both_val += gap * (gap + 1) // 2
            prev = p
        gap = N - prev - 1
        avoid_both_val += gap * (gap + 1) // 2

        count_both = total_sub - avoid[v] - avoid[v+1] + avoid_both_val
        total_adjacent += count_both

    ans = total_distinct - total_adjacent
    print(ans)

if __name__ == "__main__":
    main()
