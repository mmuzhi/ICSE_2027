- Define function count_snake_upto(N):
   if N < 10: return 0
   s = str(N)
   k = len(s)

   total = 0
   # lengths less than k
   for length in range(2, k):
       for d in range(1, 10):
           total += d ** (length - 1)
   # length == k
   n1 = digits[0]
   # first digit less than n1
   for d in range(1, n1):
       total += d ** (k - 1)
   # first digit equal to n1
   if n1 >= 1:  # always true
       suffix_digits = digits[1:]
       cnt = count_suffix(suffix_digits, n1)
       total += cnt

Define count_suffix(suffix_digits, d):
   m = len(suffix_digits)
   count = 0
   tight = True
   for i in range(m):
       if tight:
           cur = suffix_digits[i]
           # digits less than cur and allowed
           max_digit = min(cur - 1, d - 1)
           if max_digit >= 0:
               count += (max_digit + 1) * (d ** (m - i - 1))
           # check if we can stay tight
           if cur > d - 1:
               tight = False
               break
       else:
           # should not happen because we break when tight becomes false
           pass
   if tight:
       count += 1
   return count
