class Solution:
    def distanceSum(self, m: int, n: int, k: int) -> int:
        MOD = 10**9+7
        N = m * n
        # compute S_mod = N * (m+n) % MOD * (N-1) % MOD * inv6 % MOD
        inv6 = pow(6, MOD-2, MOD)
        S_mod = N % MOD * ((m + n) % MOD) % MOD * ((N - 1) % MOD) % MOD * inv6 % MOD
        
        # compute C(N-2, k-2) mod MOD
        # precompute factorials up to N
        fact = [1] * (N + 1)
        for i in range(1, N+1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (N + 1)
        inv_fact[N] = pow(fact[N], MOD-2, MOD)
        for i in range(N, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
        
        comb = fact[N-2] * inv_fact[k-2] % MOD * inv_fact[N - k] % MOD   # because (N-2)-(k-2)=N-k

```python
class Solution:
    def distanceSum(self, m: int, n: int, k: int) -> int:
        MOD = 10**9 + 7
        N = m * n                     # total number of cells

        # ----- sum of Manhattan distances over all unordered pairs of cells -----
        # S = (N * (m + n) * (N - 1)) / 6   (always integer)
        S = N * (m + n) * (N - 1) // 6
        S_mod = S % MOD

        # ----- binomial coefficient C(N-2, k-2) modulo MOD -----
        # precompute factorials up to N
        fact = [1] * (N + 1)
        for i in range(1, N + 1):
            fact[i] = fact[i-1] * i % MOD

        inv_fact = [1] * (N + 1)
        inv_fact[N] = pow(fact[N], MOD-2, MOD)
        for i in range(N, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD

        # C(N-2, k-2) = fact[N-2] * inv_fact[k-2] * inv_fact[N-k]   (since (N-2)-(k-2)=N-k)
        comb = fact[N-2] * inv_fact[k-2] % MOD * inv_fact[N - k] % MOD

        # ----- final answer -----
        return comb * S_mod % MOD
```
