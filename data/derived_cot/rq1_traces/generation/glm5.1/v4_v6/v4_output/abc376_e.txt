import sys
import heapq

def solve():
    import sys
    input = sys.stdin.readline
    T = int(input())
    out = []
    for _ in range(T):
        N, K = map(int, input().split())
        A = list(map(int, input().split()))
        B = list(map(int, input().split()))
        pairs = list(zip(A, B))
        pairs.sort(key=lambda x: x[0])
        heap = []
        sum_left = 0
        heap_size = 0
        total = 0
        ans = 10**30
        i = 0
        while i < N:
            A_val = pairs[i][0]
            minB = 10**9
            j = i
            while j < N and pairs[j][0] == A_val:
                x = pairs[j][1]
                if minB > x:
                    minB = x
                # insert x
                if heap_size < K:
                    heapq.heappush(heap, -x)
                    sum_left += x
                    heap_size += 1
                else:
                    # heap_size == K
                    left_max = -heap[0]
                    if x < left_max:
                        popped = -heapq.heappop(heap)
                        sum_left -= popped
                        heapq.heappush(heap, -x)
                        sum_left += x
                    # else discard
                j += 1
            total += (j - i)
            if total >= K:
                left_max = -heap[0]
                if minB <= left_max:
                    cand = A_val * sum_left
                else:
                    cand = A_val * (sum_left - left_max + minB)
                if cand < ans:
                    ans = cand
            i = j
        out.append(str(ans))
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
