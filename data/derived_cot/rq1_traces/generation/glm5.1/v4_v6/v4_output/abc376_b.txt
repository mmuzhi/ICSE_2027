```python

import sys

def distance(s: int, t: int, b: int, N: int) -> int:
    """minimum moves from s to t while the other hand stays at b"""
    if s == t:
        return 0
    cw = (t - s) % N          # clockwise steps from s to t
    d = (b - s) % N           # clockwise steps from s to b
    # if b lies on the clockwise arc (strictly between s and t)
    if 0 < d < cw:
        return N - cw         # must go the other way
    else:
        return cw             # clockwise arc is free

def solve() -> None:
    input = sys.stdin.readline
    N, Q = map(int, input().split())
    l, r = 1, 2
    ans = 0

    for _ in range(Q):
        H, T_str = input().split()
        T = int(T_str)
        if H == 'L':
            ans += distance(l, T, r, N)
            l = T
        else:   # H == 'R'
            ans += distance(r, T, l, N)
            r = T

    print(ans)

if __name__ == "__main__":
    solve()
```
