=== ANSWER ===
```python

import sys

def main():
    S1, S2 = sys.stdin.read().split()
    if S1 == "sick" and S2 == "sick":
        print(1)
    elif S1 == "sick" and S2 == "fine":
        print(2)
    elif S1 == "fine" and S2 == "sick":
        print(3)
    else:  # both fine
        print(4)

if __name__ == "__main__":
    main()
```
