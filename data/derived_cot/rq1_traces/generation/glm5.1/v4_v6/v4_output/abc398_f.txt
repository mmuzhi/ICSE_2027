def prefix_function(s):
    n = len(s)
    pi = [0]*n
    for i in range(1,n):
        j = pi[i-1]
        while j > 0 and s[i] != s[j]:
            j = pi[j-1]
        if s[i] == s[j]:
            j += 1
        pi[i] = j
    return pi

import sys

def prefix_function(s):
    n = len(s)
    pi = [0]*n
    for i in range(1, n):
        j = pi[i-1]
        while j > 0 and s[i] != s[j]:
            j = pi[j-1]
        if s[i] == s[j]:
            j += 1
        pi[i] = j
    return pi

def main():
    S = sys.stdin.readline().strip()
    n = len(S)
    R = S[::-1]
    T = R + '#' + S
    pi = prefix_function(T)
    L = pi[-1]
    # L is length of longest palindrome suffix
    prefix_to_append = S[:n-L][::-1]
    result = S + prefix_to_append
    print(result)

if __name__ == "__main__":
    main()
