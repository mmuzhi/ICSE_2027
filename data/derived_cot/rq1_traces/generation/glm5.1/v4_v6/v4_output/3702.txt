```python

import math
from typing import List

class Solution:
    def maxLength(self, nums: List[int]) -> int:
        n = len(nums)
        max_len = 0
        # The maximum possible value of lcm * gcd for numbers in [1,10] is 2520
        MAX_POSSIBLE = 2520

        for i in range(n):
            prod = 1          # product of current subarray
            g = 0             # gcd of current subarray (0 means undefined)
            l = 1             # lcm of current subarray
            for j in range(i, n):
                x = nums[j]
                prod *= x
                # If product already exceeds the maximum possible lcm*gcd,
                # no extension can satisfy the condition.
                if prod > MAX_POSSIBLE:
                    break
                g = math.gcd(g, x)
                l = l * x // math.gcd(l, x)
                if prod == l * g:
                    max_len = max(max_len, j - i + 1)

        return max_len
```
