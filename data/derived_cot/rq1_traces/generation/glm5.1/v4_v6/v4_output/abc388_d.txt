```python

import sys
import heapq

def main() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    heap = []          # min-heap of thresholds T_j = S_j + j
    ans = [0] * N

    for i in range(1, N + 1):
        # remove thresholds that are already < i (those adults are no longer positive)
        while heap and heap[0] < i:
            heapq.heappop(heap)

        D = len(heap)                     # number of adults with at least one stone
        S = A[i - 1] + D                  # stones after becoming adult
        T = S + i                         # this adult's threshold
        heapq.heappush(heap, T)

        B = S - (N - i)                   # stones after all gifts
        if B < 0:
            B = 0
        ans[i - 1] = B

    sys.stdout.write(' '.join(map(str, ans)))

if __name__ == "__main__":
    main()
```
