import sys

def solve():
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    T = next(it)
    out = []
    for _ in range(T):
        N = next(it)
        A = [next(it) for _ in range(2*N)]
        pos1 = [-1]*(N+1)
        pos2 = [-1]*(N+1)
        for idx, val in enumerate(A):
            if pos1[val] == -1:
                pos1[val] = idx
            else:
                pos2[val] = idx
        adjacent = [False]*(N+1)
        for x in range(1, N+1):
            if pos2[x] - pos1[x] == 1:
                adjacent[x] = True
        pair_minmax = {}
        for i in range(2*N-1):
            if A[i] != A[i+1]:
                a = A[i]
                b = A[i+1]
                if a > b:
                    a, b = b, a
                key = (a, b)
                if key in pair_minmax:
                    pair_minmax[key][1] = i
                else:
                    pair_minmax[key] = [i, i]
        ans = 0
        for (a,b), (min_i, max_i) in pair_minmax.items():
            if adjacent[a] or adjacent[b]:
                continue
            if max_i - min_i >= 2:
                ans += 1
        out.append(str(ans))
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
