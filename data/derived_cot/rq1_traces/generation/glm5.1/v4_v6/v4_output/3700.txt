class Solution:
    def subsequencesWithMiddleMode(self, nums: List[int]) -> int:
        MOD = 10**9 + 7
        n = len(nums)
        
        def comb2(x):
            return x * (x - 1) // 2 if x >= 2 else 0
        
        # Initialize counts for i2 = 2
        left_counts = {}
        right_counts = {}
        for i in range(2):
            left_counts[nums[i]] = left_counts.get(nums[i], 0) + 1
        for i in range(3, n):
            right_counts[nums[i]] = right_counts.get(nums[i], 0) + 1
        
        total = 0
        for i2 in range(2, n - 2):
            v = nums[i2]
            
            count_v_left = left_counts.get(v, 0)
            count_v_right = right_counts.get(v, 0)
            count_non_v_left = i2 - count_v_left
            count_non_v_right = (n - 1 - i2) - count_v_right
            
            # freq_v = 5
            total += comb2(count_v_left) * comb2(count_v_right)
            
            # freq_v = 4
            total += comb2(count_v_left) * count_v_right * count_non_v_right
            total += count_v_left * comb2(count_v_right) * count_non_v_left
            
            # freq_v = 3
            total += comb2(count_v_left) * comb2(count_non_v_right)
            total += count_v_left * count_v_right * count_non_v_left * count_non_v_right
            total += comb2(count_v_right) * comb2(count_non_v_left)
            
            # freq_v = 2, valid
            # (1 v from left, 0 v from right)
            # count_v_left * sum over u != v of count_u_left * (C(count_non_v_right - count_u_right, 2) - sum over w != u, w != v of C(count_w_right, 2))
            
            sum_C2_right = sum(comb2(count) for u, count in right_counts.items() if u != v)
            
            sum_u_left = 0
            for u, count_u_left in left_counts.items():
                if u == v:
                    continue
                count_u_right = right_counts.get(u, 0)
                term = comb2(count_non_v_right - count_u_right) - (sum_C2_right - comb2(count_u_right))
                sum_u_left += count_u_left * term
            
            total += count_v_left * sum_u_left
            
            # (0 v from left, 1 v from right)
            # count_v_right * sum over c != v of count_c_right * (C(count_non_v_left - count_c_left, 2) - sum over w != c, w != v of C(count_w_left, 2))
            
            sum_C2_left = sum(comb2(count) for u, count in left_counts.items() if u != v)
            
            sum_c_right = 0
            for c, count_c_right in right_counts.items():
                if c == v:
                    continue
                count_c_left = left_counts.get(c, 0)
                term = comb2(count_non_v_left - count_c_left) - (sum_C2_left - comb2(count_c_left))
                sum_c_right += count_c_right * term
            
            total += count_v_right * sum_c_right
            
            total %= MOD
            
            # Update counts for next i2
            if i2 < n - 3:
                # Position i2 moves from middle to left
                left_counts[nums[i2]] = left_counts.get(nums[i2], 0) + 1
                # Position i2+1 moves from right to middle
                right_counts[nums[i2 + 1]] -= 1
                if right_counts[nums[i2 + 1]] == 0:
                    del right_counts[nums[i2 + 1]]
        
        return total % MOD

Looking at this problem, I need to count subsequences of size 5 where the middle element (index 2) is the unique mode.

**Key Observations:**
1. For middle value `v` to be the unique mode, its frequency must be strictly greater than any other value's frequency
2. If `freq_v ≥ 3`, then `v` is always the unique mode (since at most 2 other elements can share a value)
3. If `freq_v = 2`, all 3 non-v elements must be distinct
4. If `freq_v = 1`, impossible for `v` to be unique mode

**Approach:**
For each middle position `i2`, I track counts of each value in left (positions 0 to i2-1) and right (positions i2+1 to n-1) parts using a sliding window.

For `freq_v = 2` case, I need to count ways where all 3 non-v elements are distinct. The formula for choosing 1 from left and 2 from right with all distinct values is:
$$\sum_{u \neq v} count\_u\_left \times \left[\binom{count\_non\_v\_right - count\_u\_right}{2} - \sum_{w \neq u, w \neq v} \binom{count\_w\_right}{2}\right]$$

This subtracts cases where the 2 from right have the same value.
