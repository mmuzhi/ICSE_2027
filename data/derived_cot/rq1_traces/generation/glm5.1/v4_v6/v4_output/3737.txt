def minCost(self, n: int, cost: List[List[int]]) -> int:
    m = n // 2
    INF = 10**18
    # dp[a][b] = min cost after processing some pairs, with last left color = a, last right color = b
    dp = [[INF]*3 for _ in range(3)]
    # initialize for pair 0
    for cL in range(3):
        for cR in range(3):
            if cL != cR:
                dp[cL][cR] = cost[0][cL] + cost[n-1][cR]
    # iterate over remaining pairs
    for k in range(1, m):
        new_dp = [[INF]*3 for _ in range(3)]
        for pL in range(3):
            for pR in range(3):
                cur = dp[pL][pR]
                if cur == INF:
                    continue
                # try all (cL, cR)
                for cL in range(3):
                    if cL == pL:
                        continue
                    for cR in range(3):
                        if cR == pR or cL == cR:
                            continue
                        new_cost = cur + cost[k][cL] + cost[n-1-k][cR]
                        if new_cost < new_dp[cL][cR]:
                            new_dp[cL][cR] = new_cost
        dp = new_dp
    # answer is min over dp
    ans = INF
    for cL in range(3):
        for cR in range(3):
            if dp[cL][cR] < ans:
                ans = dp[cL][cR]
    return ans

```python
from typing import List

class Solution:
    def minCost(self, n: int, cost: List[List[int]]) -> int:
        m = n // 2                     # number of symmetric pairs
        INF = 10**18

        # dp[a][b] = minimum cost after processing some pairs,
        # where a = color of the last left house, b = color of the last right house
        dp = [[INF] * 3 for _ in range(3)]

        # first pair (k = 0): houses 0 and n-1
        for cL in range(3):
            for cR in range(3):
                if cL != cR:           # symmetric constraint for this pair
                    dp[cL][cR] = cost[0][cL] + cost[n-1][cR]

        # process remaining pairs from the outside inward
        for k in range(1, m):
            new_dp = [[INF] * 3 for _ in range(3)]
            left_idx = k
            right_idx = n - 1 - k

            for pL in range(3):
                for pR in range(3):
                    cur = dp[pL][pR]
                    if cur == INF:
                        continue

                    for cL in range(3):
                        if cL == pL:          # adjacency with previous left house
                            continue
                        for cR in range(3):
                            if cR == pR or cL == cR:   # adjacency with previous right house + symmetric
                                continue
                            new_cost = cur + cost[left_idx][cL] + cost[right_idx][cR]
                            if new_cost < new_dp[cL][cR]:
                                new_dp[cL][cR] = new_cost

            dp = new_dp

        # answer is the minimum over all final states
        ans = INF
        for cL in range(3):
            for cR in range(3):
                if dp[cL][cR] < ans:
                    ans = dp[cL][cR]
        return ans
```
