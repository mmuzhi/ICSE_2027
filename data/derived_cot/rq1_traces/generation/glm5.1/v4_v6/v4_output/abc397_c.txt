import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))

    # left[i] = number of distinct values in A[0..i]
    left = [0] * N
    seen = [False] * (N + 1)
    cnt = 0
    for i in range(N):
        x = A[i]
        if not seen[x]:
            cnt += 1
            seen[x] = True
        left[i] = cnt

    # right[i] = number of distinct values in A[i..N-1]
    right = [0] * N
    seen = [False] * (N + 1)
    cnt = 0
    for i in range(N-1, -1, -1):
        x = A[i]
        if not seen[x]:
            cnt += 1
            seen[x] = True
        right[i] = cnt

    ans = 0
    for i in range(N-1):
        ans = max(ans, left[i] + right[i+1])

    print(ans)

if __name__ == "__main__":
    main()
