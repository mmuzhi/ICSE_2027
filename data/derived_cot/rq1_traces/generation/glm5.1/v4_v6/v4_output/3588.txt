Alice and Bob are playing a fantasy battle game consisting of n rounds where they summon one of three magical creatures each round: a Fire Dragon, a Water Serpent, or an Earth Golem. In each round, players simultaneously summon their creature and are awarded points as follows:

If one player summons a Fire Dragon and the other summons an Earth Golem, the player who summoned the Fire Dragon is awarded a point.
If one player summons a Water Serpent and the other summons a Fire Dragon, the player who summoned the Water Serpent is awarded a point.

class Solution:
    def countWinningSequences(self, s: str) -> int:
        MOD = 10**9+7
        n = len(s)
        # map char to index
        mp = {'F':0, 'W':1, 'E':2}
        a = [mp[ch] for ch in s]
        offset = n
        max_idx = 2*n
        # dp_prev: list of list of ints size (2n+1) x 3
        dp_prev = [[0]*3 for _ in range(2*n+1)]
        # initialize first round
        for b in range(3):
            delta = 0
            if a[0] == b:
                delta = 0
            elif (b - a[0]) % 3 == 1:
                delta = 1
            else:
                delta = -1
            d = delta
            idx = d + offset
            dp_prev[idx][b] = (dp_prev[idx][b] + 1) % MOD
        # iterate over remaining rounds
        for i in range(1, n):
            dp_cur = [[0]*3 for _ in range(2*n+1)]
            ai = a[i]
            for prev_c in range(3):
                # iterate over d indices where dp_prev[idx][prev_c] != 0
                # we can loop over all idx
                for idx in range(2*n+1):
                    val = dp_prev[idx][prev_c]
                    if val == 0:
                        continue
                    d = idx - offset
                    # possible cur_c != prev_c
                    for cur_c in range(3):
                        if cur_c == prev_c:
                            continue
                        # compute delta
                        if ai == cur_c:
                            delta = 0
                        elif (cur_c - ai) % 3 == 1:
                            delta = 1
                        else:
                            delta = -1
                        new_d = d + delta
                        new_idx = new_d + offset
                        # new_idx should be within 0..2n
                        # but we can assert
                        if 0 <= new_idx <= 2*n:
                            dp_cur[new_idx][cur_c] = (dp_cur[new_idx][cur_c] + val) % MOD
            dp_prev = dp_cur
        # after all rounds, sum over d>0
        ans = 0
        for idx in range(offset+1, 2*n+1):  # d>0 means idx > offset
            for c in range(3):
                ans = (ans + dp_prev[idx][c]) % MOD
        return ans

class Solution:
    def countWinningSequences(self, s: str) -> int:
        MOD = 10**9+7
        n = len(s)
        mp = {'F':0, 'W':1, 'E':2}
        a = [mp[ch] for ch in s]
        # precompute win matrix
        win = [[0]*3 for _ in range(3)]
        for i in range(3):
            for j in range(3):
                if i == j:
                    win[i][j] = 0
                elif (j - i) % 3 == 1:
                    win[i][j] = 1
                else:
                    win[i][j] = -1
        offset = n
        size = 2*n + 1
        # dp_prev: list of lists size x 3
        dp_prev = [[0]*3 for _ in range(size)]
        # initialize first round
        a0 = a[0]
        for b in range(3):
            delta = win[a0][b]
            d = delta
            idx = d + offset
            dp_prev[idx][b] = (dp_prev[idx][b] + 1) % MOD
        # iterate
        for i in range(1, n):
            ai = a[i]
            dp_cur = [[0]*3 for _ in range(size)]
            for prev_c in range(3):
                for idx in range(size):
                    val = dp_prev[idx][prev_c]
                    if val == 0:
                        continue
                    d = idx - offset
                    for cur_c in range(3):
                        if cur_c == prev_c:
                            continue
                        delta = win[ai][cur_c]
                        new_d = d + delta
                        new_idx = new_d + offset
                        # new_idx should be within [0, size-1]
                        if 0 <= new_idx < size:
                            dp_cur[new_idx][cur_c] = (dp_cur[new_idx][cur_c] + val) % MOD
            dp_prev = dp_cur
        # sum over d > 0
        ans = 0
        for idx in range(offset+1, size):
            for c in range(3):
                ans = (ans + dp_prev[idx][c]) % MOD
        return ans

class Solution:
    def countWinningSequences(self, s: str) -> int:
        MOD = 10**9 + 7
        n = len(s)
        if n == 0:

        # map characters to indices
        mp = {'F': 0, 'W': 1, 'E': 2}

                    win[i][j] = 0
                elif (j - i) % 3 == 1:
                    win[i][j] = 1
                else:
                    win[i][j] = -1

        offset = n                     # to make diff index non‑negative
        size = 2 * n + 1               # number of possible diff values
        next_moves = [[1, 2], [0, 2], [0, 1]]   # allowed moves after a given previous move

        # dp_prev[idx][c] : after current number of rounds, diff = idx - offset,
        #                   last move = c
        dp_prev = [[0] * 3 for _ in range(size)]

        # first round
        a0 = a[0]
        for b in range(3):
            delta = win[a0][b]
            idx = delta + offset
            dp_prev[idx][b] = 1

        # remaining rounds
        for i in range(1, n):
            ai = a[i]
            dp_cur = [[0] * 3 for _ in range(size)]
            for prev_c in range(3):
                for idx in range(size):
                    val = dp_prev[idx][prev_c]
                    if val == 0:

                    d = idx - offset
                    for cur_c in next_moves[prev_c]:

                        if new_val >= MOD:
