- 1 \leq M \leq 2 \times 10^5

- 1 \leq C_i \leq 10^9

- All input values are integers.

import sys
import heapq

MOD1 = 1000000007
MOD2 = 1000000009

def dijkstra(start, graph, N):
    dist = [10**18]*(N+1)
    dist[start] = 0
    hq = [(0, start)]
    while hq:
        d,u = heapq.heappop(hq)
        if d != dist[u]:
            continue
        for v,w in graph[u]:
            nd = d + w
            if nd < dist[v]:
                dist[v] = nd
                heapq.heappush(hq, (nd, v))
    return dist

def main():
    input = sys.stdin.readline
    N,M = map(int, input().split())
    edges = []
    graph = [[] for _ in range(N+1)]
    for i in range(M):
        a,b,c = map(int, input().split())
        edges.append((a,b,c))
        graph[a].append((b,c))
        graph[b].append((a,c))
    
    dist1 = dijkstra(1, graph, N)
    distN = dijkstra(N, graph, N)
    d = dist1[N]
    
    out = [[] for _ in range(N+1)]
    inn = [[] for _ in range(N+1)]
    edge_dir = [None]*M
    
    for i, (a,b,c) in enumerate(edges):
        if dist1[a] + c + distN[b] == d:
            u,v = a,b
            out[u].append(v)
            inn[v].append(u)
            edge_dir[i] = (u,v)
        elif dist1[b] + c + distN[a] == d:
            u,v = b,a
            out[u].append(v)
            inn[v].append(u)
            edge_dir[i] = (u,v)
        else:
            edge_dir[i] = None
    
    # cnt1 DP

def dijkstra(start, graph, N):
    dist = [INF] * (N + 1)
    dist[start] = 0
    hq = [(0, start)]
    while hq:
        d, u = heapq.heappop(hq)
        if d != dist[u]:
            continue
        for v, w in graph[u]:
            nd = d + w
            if nd < dist[v]:
                dist[v] = nd
                heapq.heappush(hq, (nd, v))
    return dist
