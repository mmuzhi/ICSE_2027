import sys
from collections import defaultdict

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    groups = defaultdict(list)
    for a in A:
        # compute v2
        tz = (a & -a).bit_length() - 1
        o = a >> tz
        groups[tz].append(o)

    # Precompute cnt and sum for each group
    cnt = {}
    sum_o = {}
    for k, lst in groups.items():
        cnt[k] = len(lst)
        sum_o[k] = sum(lst)

    # cross sum
    cross_sum = 0
    keys = sorted(groups.keys())
    for i, k1 in enumerate(keys):
        c1 = cnt[k1]
        s1 = sum_o[k1]
        for k2 in keys[i+1:]:
            c2 = cnt[k2]
            s2 = sum_o[k2]
            diff = k2 - k1
            cross_sum += c2 * s1 + c1 * (1 << diff) * s2

    # intra sum
    def compute_intra(B):
        M = len(B)
        if M == 0:
            return 0
        if M == 1:
            return B[0]
        total_sum = sum(B)
        max_b = max(B)
        two_max = max_b << 1
        T_max = two_max.bit_length() - 1  # largest t with 2^t <= two_max
        # S array, index t from 1 to T_max+1
        S = [0] * (T_max + 2)
        S[1] = (M + 1) * total_sum
        for t in range(2, T_max + 1):
            mod = 1 << t
            mask = mod - 1
            cnt_r = {}
            sum_r = {}
            for val in B:
                r = val & mask
                if r in cnt_r:
                    cnt_r[r] += 1
                    sum_r[r] += val
                else:
                    cnt_r[r] = 1
                    sum_r[r] = val
            total = 0
            for r, c in cnt_r.items():
                s = (-r) & mask
                if r < s:
                    c2 = cnt_r.get(s, 0)
                    if c2:
                        total += c * sum_r[s] + c2 * sum_r[r]
            S[t] = total
        intra = 0
        for t in range(1, T_max + 1):
            intra += (S[t] - S[t+1]) >> t
        return intra

    intra_total = 0
    for lst in groups.values():
        intra_total += compute_intra(lst)

    ans = cross_sum + intra_total
    print(ans)

if __name__ == "__main__":
    solve()

def solve():
    import sys
    data = sys.stdin.read().split()
    N = int(data[0])
    A = list(map(int, data[1:]))
    # compute v2 and o for each, track max_v2
    max_v2 = 0
    pairs = []  # list of (v2, o)
    for a in A:
        tz = (a & -a).bit_length() - 1
        o = a >> tz
        pairs.append((tz, o))
        if tz > max_v2:
            max_v2 = tz
    groups = [[] for _ in range(max_v2+1)]
    for tz, o in pairs:
        groups[tz].append(o)

    # cnt and sum arrays
    cnt = [len(g) for g in groups]
    sum_o = [sum(g) for g in groups]

    # cross sum
    cross_sum = 0
    for k1 in range(max_v2+1):
        if cnt[k1] == 0:
            continue
        for k2 in range(k1+1, max_v2+1):
            if cnt[k2] == 0:
                continue
            cross_sum += cnt[k2] * sum_o[k1] + cnt[k1] * (1 << (k2 - k1)) * sum_o[k2]

    # intra sum
    def compute_intra(B):
        M = len(B)
        if M == 0:
            return 0
        if M == 1:
            return B[0]
        total_sum = sum(B)
        max_b = max(B)
        two_max = max_b << 1
        T_max = two_max.bit_length() - 1
        S = [0] * (T_max + 2)
        S[1] = (M + 1) * total_sum
        for t in range(2, T_max + 1):
            mod = 1 << t
            mask = mod - 1
            cnt_r = {}
            sum_r = {}
            for val in B:
                r = val & mask
                if r in cnt_r:
                    cnt_r[r] += 1
                    sum_r[r] += val
                else:
                    cnt_r[r] = 1
                    sum_r[r] = val
            total = 0
            for r, c in cnt_r.items():
                s = (-r) & mask
                if r < s:
                    c2 = cnt_r.get(s, 0)
                    if c2:
                        total += c * sum_r[s] + c2 * sum_r[r]
            S[t] = total
        intra = 0
        for t in range(1, T_max + 1):
            intra += (S[t] - S[t+1]) >> t
        return intra

    intra_total = 0
    for g in groups:
        if g:
            intra_total += compute_intra(g)

    print(cross_sum + intra_total)
