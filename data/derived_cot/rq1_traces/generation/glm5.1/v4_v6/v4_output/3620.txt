class Solution:
    def maxDistinctElements(self, nums: List[int], k: int) -> int:
        nums.sort()
        next_available = -10**18  # or use None
        count = 0
        for num in nums:
            L = num - k
            R = num + k
            candidate = max(L, next_available)
            if candidate <= R:
                count += 1
                next_available = candidate + 1
        return count

```python
from typing import List

class Solution:
    def maxDistinctElements(self, nums: List[int], k: int) -> int:
        # Sort the numbers so that intervals are processed in order of increasing right endpoint
        nums.sort()
        # next_available: the smallest integer that has not been used yet
        next_available = -10**18   # sufficiently small
        distinct_count = 0

        for num in nums:
            L = num - k
            R = num + k
            # smallest integer that is >= L and not used yet
            candidate = max(L, next_available)
            if candidate <= R:
                distinct_count += 1
                next_available = candidate + 1   # mark this integer as used

        return distinct_count
```
