def solve():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    X = int(next(it))

    D = [0]*N

    for i in range(N):
        u = int(next(it))
        d = int(next(it))
        U[i] = u
        D[i] = d
        total += u + d
        if u + d < M:

        b = min(U[0], H)
        if a > b:
            return False
        for i in range(1, N):
            L = max(0, H - D[i])
            R = min(U[i], H)
            a = max(L, a - X)
            b = min(R, b + X)
            if a > b:

            low = mid + 1
        else:

    print(total - N * ans)
