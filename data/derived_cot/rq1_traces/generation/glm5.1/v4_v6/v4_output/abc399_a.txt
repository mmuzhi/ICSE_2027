=== ANSWER ===
```python

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1]
    T = data[2]

    # Count positions where characters differ
    ans = sum(1 for i in range(N) if S[i] != T[i])
    print(ans)

if __name__ == "__main__":
    main()

```
