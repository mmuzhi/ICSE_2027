import sys

def main():
    import sys
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    S = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    total = sum(A)

    # prefix sums
    P = [0] * (N+1)
    for i in range(1, N+1):
        P[i] = P[i-1] + A[i-1]

    # case1: within one period
    seen = {0}
    for i in range(1, N+1):
        if (P[i] - S) in seen:
            print("Yes")
            return
        seen.add(P[i])

    # case2
    # suffix sums: for start index i (1-indexed), suffix = total - P[i-1]
    suf = [total - P[i] for i in range(N)]  # i from 0 to N-1 corresponds to start i+1
    # prefix sums set (non-zero)
    pref_set = set(P[1:])  # exclude P[0]

    R = S % total

    candidates = []
    if R != 0:
        candidates.append(R)
        if S >= total:
            candidates.append(R + total)
    else:
        candidates.append(total)
        if S >= 2 * total:
            candidates.append(2 * total)

    for T in candidates:
        for s in suf:
            need = T - s
            if need in pref_set:
                print("Yes")
                return

    print("No")

if __name__ == "__main__":
    main()

import sys

def main():
    it = iter(sys.stdin.read().split())
    N = int(next(it))
    S = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    total = sum(A)

    # prefix sums P[0..N]
    P = [0] * (N + 1)
    for i in range(1, N + 1):
        P[i] = P[i - 1] + A[i - 1]

    # ----- case 1 : subarray inside one period -----
    seen = {0}
    for i in range(1, N + 1):
        if (P[i] - S) in seen:
            print("Yes")
            return
        seen.add(P[i])

    # ----- case 2 : subarray crossing period boundaries -----
    # suffix sums: for start position i (1‑indexed) -> total - P[i-1]
    suf = [total - P[i] for i in range(N)]          # i = 0..N-1 -> start = i+1
    # set of non‑empty prefix sums (1..N)
    pref_set = set(P[1:])

    R = S % total
    candidates = []

    if R != 0:
        candidates.append(R)
        if S >= total:                # then R+total is also a candidate
            candidates.append(R + total)
    else:   # R == 0
        candidates.append(total)
        if S >= 2 * total:
            candidates.append(2 * total)

    for T in candidates:
        for s in suf:
            need = T - s
            if need in pref_set:
                print("Yes")
                return

    print("No")

if __name__ == "__main__":
    main()
