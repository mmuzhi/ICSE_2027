class Solution:
    def minCostGoodCaption(self, caption: str) -> str:
        n = len(caption)
        if n < 3:
            return ""
        s = [ord(ch)-97 for ch in caption]
        INF = 10**9
        size = (n+1) * 26
        from array import array
        bdp1 = array('i', [INF]) * size
        bdp2 = array('i', [INF]) * size
        bdp3 = array('i', [INF]) * size

        # i = n
        base_n = n * 26
        for c in range(26):
            bdp3[base_n + c] = 0

        # backward DP
        for i in range(n-1, 0, -1):
            s_i = s[i]
            next_base = (i+1) * 26
            # compute min1, min2 for new run
            min1 = INF
            min1_idx = -1
            min2 = INF
            for cp in range(26):
                v = abs(s_i - cp) + bdp1[next_base + cp]
                if v < min1:
                    min2 = min1
                    min1 = v
                    min1_idx = cp
                elif v < min2:
                    min2 = v
            cur_base = i * 26
            for c in range(26):
                # bdp1
                bdp1[cur_base + c] = abs(s_i - c) + bdp2[next_base + c]
                # bdp2
                bdp2[cur_base + c] = abs(s_i - c) + bdp3[next_base + c]
                # bdp3
                cont = abs(s_i - c) + bdp3[next_base + c]
                best_new = min1 if min1_idx != c else min2
                if best_new < cont:
                    bdp3[cur_base + c] = best_new
                else:
                    bdp3[cur_base + c] = cont

        # compute global_opt
        global_opt = INF
        first_base = 1 * 26
        for c in range(26):
            total = abs(s[0] - c) + bdp1[first_base + c]
            if total < global_opt:
                global_opt = total

        if global_opt >= INF:
            return ""

        # reconstruction
        res = []
        # choose first character
        for c in range(26):
            if abs(s[0] - c) + bdp1[first_base + c] == global_opt:
                res.append(c)
                current_cost = abs(s[0] - c)
                c_cur = c
                st_cur = 1
                break

        # process remaining positions
        for i in range(1, n):
            s_i = s[i]
            chosen = None
            next_base = (i+1) * 26
            for ch in range(26):
                if ch == c_cur:
                    d = abs(s_i - ch)
                    if st_cur == 1:
                        bdp_next = bdp2[next_base + ch]
                    elif st_cur == 2:
                        bdp_next = bdp3[next_base + ch]
                    else:  # st_cur == 3
                        bdp_next = bdp3[next_base + ch]
                    if current_cost + d + bdp_next == global_opt:
                        chosen = ch
                        break
                else:
                    if st_cur == 3:
                        d = abs(s_i - ch)
                        bdp_next = bdp1[next_base + ch]
                        if current_cost + d + bdp_next == global_opt:
                            chosen = ch
                            break
            if chosen is None:
                # fallback, should not happen
                return ""
            res.append(chosen)
            current_cost += abs(s_i - chosen)
            if chosen == c_cur:
                if st_cur == 1:
                    st_cur = 2
                elif st_cur == 2:
                    st_cur = 3
                # else remains 3
            else:
                c_cur = chosen
                st_cur = 1

        return ''.join(chr(c + 97) for c in res)
