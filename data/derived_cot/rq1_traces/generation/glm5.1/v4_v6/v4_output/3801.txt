```python
def solve(n, S):
    if n == 0: return 0
    digits = list(map(int, str(n)))
    K = len(digits)
    ans = 0
    
    # 1. Add counts for lengths strictly less than K
    for L in range(1, K):
        ans += dp_len[L][S][0]
        
    # 2. Add counts for length exactly K
    pre_sum = 0
    pre_rem = 1 % S
    
    for i in range(K):
        d = digits[i]
        rem_len = K - i - 1
        
        # Choose x < d
        start_x = 1 if i == 0 else 0
        for x in range(start_x, d):
            new_pre_sum = pre_sum + x
            new_pre_rem = (pre_rem * x) % S if x > 0 else 0
            
            req_s = S - new_pre_sum
            if 0 <= req_s < MAX_SUM:
                for suf_r in range(S):
                    if (new_pre_rem * suf_r) % S == 0:
                        ans += dp_suf[rem_len][req_s][suf_r]
                        
        # Update prefix with d
        pre_sum = pre_sum + d
        if d == 0:
            pre_rem = 0
        else:
            pre_rem = (pre_rem * d) % S
            
    # Check n itself
    if pre_sum == S and pre_rem == 0:
        ans += 1
        
    return ans
```

=== ANSWER ===
An elegant and efficient way to solve this problem is to use Digit Dynamic Programming (Digit DP). Since we need the product of the digits to be divisible by the sum of the digits, we can observe that for any number up to $10^9$, the maximum sum of its digits is $9 \times 9 = 81$. 

We can iterate over all possible sums of digits $S$ from $1$ to $81$. For a fixed $S$, we count the numbers in the range $[l, r]$ such that the sum of their digits is exactly $S$ and the product of their digits modulo $S$ is $0$. The overall answer will be the sum of these counts across all possible $S$.

To count such numbers up to $n$, we use two DP tables:
1. `dp_len[L][s][r]`: Counts sequences of exactly length $L$ (no leading zeros) where the sum of digits is $s$ and the product of digits modulo $S$ is $r$.
2. `dp_suf[L][s][r]`: Counts sequences of length $L$ (leading zeros allowed) where the sum of digits is $s$ and the product of digits modulo $S$ is $r$.

Using these tables, we can efficiently compute the beautiful numbers $\le n$ by breaking $n$ into a prefix and a suffix. For each position where we choose a digit strictly less than the corresponding digit in $n$, the remaining positions can be filled freely, and their valid combinations can be fetched directly from `dp_suf`.

### Python 3 Solution

```python
class Solution:
    def beautifulNumbers(self, l: int, r: int) -> int:
        def count_beautiful(n):
            if n == 0:
                return 0
            
            total_ans = 0
            MAX_SUM = 82
            
            # Iterate over all possible sums of digits S
            for S in range(1, MAX_SUM):
                # dp_suf[L][s][r] counts sequences of length L (leading zeros allowed)
                # where sum is s and product mod S is r
                dp_suf = [[[0] * S for _ in range(MAX_SUM)] for _ in range(10)]
                dp_suf[0][0][1 % S] = 1
                
                for L in range(9):
                    dp_suf_L = dp_suf[L]
                    dp_suf_L1 = dp_suf[L + 1]
                    for s in range(MAX_SUM):
                        dp_suf_L_s = dp_suf_L[s]
                        for r_rem in range(S):
                            val = dp_suf_L_s[r_rem]
                            if val == 0:
                                continue
                            for d in range(10):
                                ns = s + d
                                if ns >= MAX_SUM:
                                    continue
                                nr = (r_rem * d) % S if d > 0 else 0
                                dp_suf_L1[ns][nr] += val
                                
                # dp_len[L][s][r] counts sequences of length L (no leading zeros)
                # where sum is s and product mod S is r
                dp_len = [[[0] * S for _ in range(MAX_SUM)] for _ in range(10)]
                for d in range(1, 10):
                    ns = d
                    if ns < MAX_SUM:
                        dp_len[1][ns][d % S] += 1
                        
                for L in range(1, 9):
                    dp_len_L = dp_len[L]
                    dp_len_L1 = dp_len[L + 1]
                    for s in range(MAX_SUM):
                        dp_len_L_s = dp_len_L[s]
                        for r_rem in range(S):
                            val = dp_len_L_s[r_rem]
                            if val == 0:
                                continue
                            for d in range(10):
                                ns = s + d
                                if ns >= MAX_SUM:
                                    continue
                                nr = (r_rem * d) % S if d > 0 else 0
                                dp_len_L1[ns][nr] += val
                                
                digits = list(map(int, str(n)))
                K = len(digits)
                ans = 0
                
                # Add all valid numbers with strictly fewer digits than n
                for L in range(1, K):
                    ans += dp_len[L][S][0]
                    
                pre_sum = 0
                pre_rem = 1 % S
                
                # Traverse the digits to count valid numbers of exactly the same length as n
                for i in range(K):
                    d = digits[i]
                    rem_len = K - i - 1
                    
                    start_x = 1 if i == 0 else 0
                    for x in range(start_x, d):
                        new_pre_sum = pre_sum + x
                        new_pre_rem = (pre_rem * x) % S if x > 0 else 0
                        
                        req_s = S - new_pre_sum
                        if 0 <= req_s < MAX_SUM:
                            dp_suf_rem_len_req_s = dp_suf[rem_len][req_s]
                            for suf_r in range(S):
                                if (new_pre_rem * suf_r) % S == 0:
                                    ans += dp_suf_rem_len_req_s[suf_r]
                                    
                    # Update the running prefix sum and remainder
                    pre_sum = pre_sum + d
                    if d == 0:
                        pre_rem = 0
                    else:
                        pre_rem = (pre_rem * d) % S
                        
                # Check if n itself is beautiful
                if pre_sum == S and pre_rem == 0:
                    ans += 1
                    
                total_ans += ans
                
            return total_ans

        return count_beautiful(r) - count_beautiful(l - 1)
```
