import sys
import math

def floor_sum(n, m, a, b):
    # sum_{i=0}^{n-1} floor((a*i + b) / m)
    ans = 0
    while True:
        if a >= m:
            ans += (n-1) * n * (a // m) // 2
            a %= m
        if b >= m:
            ans += n * (b // m)
            b %= m
        y_max = a * n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans

def floor_sum_range(a, m, b, L, R):
    # sum_{i=L}^{R} floor((a*i + b)/m)
    return floor_sum(R+1, m, a, b) - floor_sum(L, m, a, b)

def count_in_interval(L, R, n, inv, m):
    # number of r in [L,R] with (inv*r) mod m < n
    if L > R:
        return 0
    total_len = R - L + 1
    # S1
    S1 = floor_sum_range(inv, m, 0, L, R)
    # S2
    b = -n
    q = b // m  # floor division
    b0 = b % m  # non-negative
    S2_base = floor_sum_range(inv, m, b0, L, R)
    S2 = q * total_len + S2_base
    cnt = S1 - S2
    return cnt

def main():
    import sys
    input = sys.stdin.readline
    N, M, C, K = map(int, input().split())
    A = list(map(int, input().split()))
    if C == 0:
        ans = K * min(A)
        print(ans)
        return
    g = math.gcd(C, M)
    m_prime = M // g
    d_prime = (M - C) // g
    # deduplicate and sort A
    S = sorted(set(A))
    N = len(S)
    # Build intervals
    intervals = []
    # first interval
    L0 = 0
    R0 = S[0] // g
    intervals.append((L0, R0, S[0]))
    # middle intervals
    for i in range(1, N):
        L = S[i-1] // g + 1
        R = S[i] // g
        if L <= R:
            intervals.append((L, R, S[i]))
    # last interval
    L_last = S[N-1] // g + 1
    R_last = m_prime - 1
    if L_last <= R_last:
        intervals.append((L_last, R_last, S[0] + M))
    # Compute S_full
    S_full = 0
    for L, R, c in intervals:
        length = R - L + 1
        sum_r = (L + R) * length // 2
        S_full += c * length - g * sum_r
    # Number of full periods
    q = K // m_prime
    rem = K % m_prime
    ans = q * S_full
    if rem > 0:
        n = rem
        # compute sum_r for prefix
        # sum_r = d' * n*(n-1)//2 - m' * floor_sum(n, m', d', 0)
        sum_r_val = d_prime * n * (n-1) // 2 - m_prime * floor_sum(n, m_prime, d_prime, 0)
        # compute inv
        inv = pow(d_prime, -1, m_prime)  # Python 3.8+
        sum_F = 0
        for L, R, c in intervals:
            cnt = count_in_interval(L, R, n, inv, m_prime)
            sum_F += c * cnt
        S_prefix = -g * sum_r_val + sum_F
        ans += S_prefix
    print(ans)

if __name__ == "__main__":
    main()
