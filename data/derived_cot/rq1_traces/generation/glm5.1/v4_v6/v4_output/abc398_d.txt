```python

import sys

def main():
    input = sys.stdin.readline
    N, R, C = map(int, input().split())
    S = input().strip()

    x = y = 0                     # current prefix sum V_t
    seen = set()
    seen.add((0, 0))              # V_0
    ans = []

    for ch in S:
        # apply wind
        if ch == 'N':
            x -= 1
        elif ch == 'W':
            y -= 1
        elif ch == 'S':
            x += 1
        else:  # 'E'
            y += 1

        # check if (R,C) is occupied after this step
        target = (x - R, y - C)
        if target in seen:
            ans.append('1')
        else:
            ans.append('0')

        # make V_t available for future steps
        seen.add((x, y))

    print(''.join(ans))

if __name__ == "__main__":
    main()
```
