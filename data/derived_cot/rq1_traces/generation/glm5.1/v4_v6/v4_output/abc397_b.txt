import sys

def main():
    S = sys.stdin.readline().strip()
    cur = 1  # next position to consider (1-indexed)

            if cur % 2 == 1:   # current position is odd → matches 'i'

                cur += 2
        else:  # ch == 'o'
            if cur % 2 == 0:   # current position is even → matches 'o'
                cur += 1

                cur += 2
