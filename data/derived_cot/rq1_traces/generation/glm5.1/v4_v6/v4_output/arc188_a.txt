For a non-empty string T consisting of A, B, and C, we call it a good string if it can be turned into an empty string by performing the following two types of operations any number of times in any order.

- Operation 1: Choose two identical characters in the string and delete them (cannot be performed if there are not two or more identical characters).
- Operation 2: Choose one A, one B, and one C in the string and delete them (cannot be performed if there are not one or more of each of A, B, and C).

For example, ABACA is a good string because it can be turned into an empty string by performing the operations as follows:

- Choose the 2nd, 4th, and 5th characters and delete them (Operation 2). The string becomes AA.
- Choose the 1st and 2nd characters and delete them (Operation 1). The string becomes an empty string.

You are given a string S of length N consisting of A, B, C, and ?. How many ways are there to replace each ? with A, B, or C to form a string that contains at least K good strings as contiguous substrings? Substrings are counted separately if they are at different positions in the original string, even if they are identical strings.
Find the count modulo 998244353.

- 0 \leq K \leq \frac{N(N+1)}{2}
- N and K are integers.
- |S| = N
- S is a string consisting of A, B, C, and ?.

def solve():
    import sys
    input_data = sys.stdin.read().split()
    N = int(input_data[0])
    K = int(input_data[1])
    S = input_data[2].strip()
    # mapping
    char_to_w = {'A':2, 'B':3, 'C':1}
    # pow52
    pow52 = [1, 52, 52*52, 52*52*52]  # 52^0,1,2,3
    # initial dp
    dp = {(0, 1): 1}  # (cur_state, packed) where packed = c0 + c1*52 + c2*52^2 + c3*52^3
    for ch in S:

        else:
            w_list = [char_to_w[ch]]
        ndp = {}
        for (cur_state, packed), ways in dp.items():
            for w in w_list:
                next_state = cur_state ^ w
                new_packed = packed + pow52[next_state]
                key = (next_state, new_packed)
                ndp[key] = (ndp.get(key, 0) + ways) % MOD
        dp = ndp
    ans = 0

        c0 = tmp % 52
        tmp //= 52
        c1 = tmp % 52
        tmp //= 52

    print(ans)

import sys

MOD = 998244353

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    K = int(data[1])
    S = data[2].strip()
    
    # shortcut for K=0
    if K == 0:
        q = S.count('?')
        print(pow(3, q, MOD))
        return
    
    max_good = N * (N + 1) // 2
    if K > max_good:
        print(0)
        return
    
    # mapping from character to w (as integer state change)
    char_to_w = {'A': 2, 'B': 3, 'C': 1}
    # base for packing counts: 52 (since max count 51)
    BASE = 52
    pow_base = [1, BASE, BASE*BASE, BASE*BASE*BASE]  # 52^0,1,2,3
    
    # precompute options for each position
    w_options = []
    for ch in S:
        if ch == '?':
            w_options.append([2,3,1])
        else:
            w_options.append([char_to_w[ch]])
    
    # dp[cur_state][packed] = count
    dp = [dict() for _ in range(4)]
    dp[0][1] = 1  # initial: state0, packed=1 (c0=1)
    
    for w_list in w_options:
        ndp = [dict() for _ in range(4)]
        for cur_state in range(4):
            cur_dict = dp[cur_state]
            if not cur_dict:
                continue
            for packed, ways in cur_dict.items():
                for w in w_list:
                    next_state = cur_state ^ w
                    new_packed = packed + pow_base[next_state]
                    ndict = ndp[next_state]
                    ndict[new_packed] = (ndict.get(new_packed, 0) + ways) % MOD
        dp = ndp
    
    ans = 0
    for cur_state in range(4):
        for packed, ways in dp[cur_state].items():
            # decode packed
            tmp = packed
            c0 = tmp % BASE
            tmp //= BASE
            c1 = tmp % BASE
            tmp //= BASE
            c2 = tmp % BASE
            c3 = tmp // BASE
            total = c0*(c0-1)//2 + c1*(c1-1)//2 + c2*(c2-1)//2 + c3*(c3-1)//2
            if total >= K:
                ans = (ans + ways) % MOD
    print(ans)

if __name__ == "__main__":
    solve()

import sys

MOD = 998244353

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    K = int(data[1])
    S = data[2].strip()
    
    if K == 0:
        q = S.count('?')
        print(pow(3, q, MOD))
        return
    
    max_good = N * (N + 1) // 2
    if K > max_good:
        print(0)
        return
    
    char_to_w = {'A': 2, 'B': 3, 'C': 1}
    
    SHIFT = [0, 6, 12, 18]
    INC = [1 << s for s in SHIFT]
    MASK = 0x3F
    
    # dp[cur_state][packed] = count
    dp = [dict() for _ in range(4)]
    dp[0][1] = 1
    
    for ch in S:
        if ch == '?':
            w_list = (2, 3, 1)
        else:
            w_list = (char_to_w[ch],)
        ndp = [dict() for _ in range(4)]
        for cur_state in range(4):
            cur_dict = dp[cur_state]
            if not cur_dict:
                continue
            for packed, ways in cur_dict.items():
                for w in w_list:
                    next_state = cur_state ^ w
                    new_packed = packed + INC[next_state]
                    ndict = ndp[next_state]
                    ndict[new_packed] = (ndict.get(new_packed, 0) + ways) % MOD
        dp = ndp
    
    ans = 0
    for cur_state in range(4):
        for packed, ways in dp[cur_state].items():
            c0 = packed & MASK
            c1 = (packed >> 6) & MASK
            c2 = (packed >> 12) & MASK
            c3 = (packed >> 18) & MASK
            total = c0*(c0-1)//2 + c1*(c1-1)//2 + c2*(c2-1)//2 + c3*(c3-1)//2

                ans = (ans + ways) % MOD
    print(ans)

if __name__ == "__main__":

import sys
from collections import defaultdict

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    K = int(data[1])
    S = data[2].strip()

    # ----- trivial cases -----
    if K == 0:
        q = S.count('?')
        print(pow(3, q, MOD))
        return

    max_good = N * (N + 1) // 2
    if K > max_good:
        print(0)
        return

    # ----- mapping from letters to state changes -----
    char_to_w = {'A': 2, 'B': 3, 'C': 1}   # w = f(letter) in GF(2)^2

    # ----- bit packing of the 4 counters (each ≤ 51 < 64) -----
    SHIFT = [0, 6, 12, 18]          # shift for state 0,1,2,3
    INC   = [1 << s for s in SHIFT] # increment for each state
    MASK  = 0x3F                    # 6 bits

    # dp[cur_state][packed] = number of assignments
    dp = [defaultdict(int) for _ in range(4)]
    dp[0][1] = 1                     # start: state 0, cnt0 = 1

    for ch in S:
        if ch == '?':
            w_list = (2, 3, 1)
        else:
            w_list = (char_to_w[ch],)

        ndp = [defaultdict(int) for _ in range(4)]
        for cur_state in range(4):
            cur_dict = dp[cur_state]
            if not cur_dict:
                continue
            for packed, ways in cur_dict.items():
                for w in w_list:
                    nxt = cur_state ^ w
                    new_packed = packed + INC[nxt]
                    nd = ndp[nxt]
                    new_val = nd[new_packed] + ways
                    if new_val >= MOD:
                        new_val -= MOD
                    nd[new_packed] = new_val
        dp = ndp

    # ----- collect answer -----
    ans = 0
    for cur_state in range(4):
        for packed, ways in dp[cur_state].items():
            c0 = packed & MASK
            c1 = (packed >> 6) & MASK
            c2 = (packed >> 12) & MASK
            c3 = (packed >> 18) & MASK
            good = (c0*(c0-1)//2 + c1*(c1-1)//2 +
                    c2*(c2-1)//2 + c3*(c3-1)//2)
            if good >= K:
                ans += ways
                if ans >= MOD:
                    ans -= MOD
    print(ans)


if __name__ == "__main__":
