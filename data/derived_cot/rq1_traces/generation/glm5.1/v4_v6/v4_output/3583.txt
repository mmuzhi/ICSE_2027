You will be given a competitive programming problem.
You are given an integer array nums of length n and an integer array queries.
Let gcdPairs denote an array obtained by calculating the GCD of all possible pairs (nums[i], nums[j]), where 0 <= i < j < n, and then sorting these values in ascending order.
For each query queries[i], you need to find the element at index queries[i] in gcdPairs.
Return an integer array answer, where answer[i] is the value at gcdPairs[queries[i]] for each query.
The term gcd(a, b) denotes the greatest common divisor of a and b.
 
Example 1:

Input: nums = [2,3,4], queries = [0,2,2]
Output: [1,2,2]
Explanation:
gcdPairs = [gcd(nums[0], nums[1]), gcd(nums[0], nums[2]), gcd(nums[1], nums[2])] = [1, 2, 1].
After sorting in ascending order, gcdPairs = [1, 1, 2].
So, the answer is [gcdPairs[queries[0]], gcdPairs[queries[1]], gcdPairs[queries[2]]] = [1, 2, 2].

Explanation:
gcdPairs sorted in ascending order is [1, 1, 1, 2, 2, 4].

class Solution:
    def gcdValues(self, nums: List[int], queries: List[int]) -> List[int]:
        max_val = max(nums)
        freq = [0] * (max_val + 1)
        for x in nums:
            freq[x] += 1
        
        # c[d] = count of numbers divisible by d
        c = [0] * (max_val + 1)
        for d in range(1, max_val + 1):
            s = 0
            for m in range(d, max_val + 1, d):
                s += freq[m]
            c[d] = s
        
        # pairs[d] = number of unordered pairs with both numbers divisible by d
        pairs = [0] * (max_val + 1)
        for d in range(1, max_val + 1):
            if c[d] >= 2:
                pairs[d] = c[d] * (c[d] - 1) // 2
        
        # cnt[d] = number of pairs with gcd exactly d

class Solution:
    def gcdValues(self, nums: List[int], queries: List[int]) -> List[int]:
        max_val = max(nums)
        freq = [0] * (max_val + 1)
        for x in nums:
            freq[x] += 1

        c = [0] * (max_val + 1)
        for d in range(1, max_val + 1):
            total = 0
            for m in range(d, max_val + 1, d):
                total += freq[m]
            c[d] = total

            cnt[d] = pairs
            for multiple in range(2 * d, max_val + 1, d):

            cum.append(cum[-1] + cnt[d])

        ans = []
        for q in queries:

            ans.append(d)
        return ans

from typing import List
from bisect import bisect_right

class Solution:
    def gcdValues(self, nums: List[int], queries: List[int]) -> List[int]:
        max_val = max(nums)
        freq = [0] * (max_val + 1)
        for x in nums:
            freq[x] += 1

        # c[d] = number of elements divisible by d
        c = [0] * (max_val + 1)
        for d in range(1, max_val + 1):
            total = 0
            for m in range(d, max_val + 1, d):
                total += freq[m]
            c[d] = total

        # cnt[d] = number of pairs with gcd exactly d
        cnt = [0] * (max_val + 1)
        for d in range(max_val, 0, -1):
            pairs = c[d] * (c[d] - 1) // 2
            cnt[d] = pairs
            for multiple in range(2 * d, max_val + 1, d):
                cnt[d] -= cnt[multiple]

        # cumulative sums over increasing d
        cum = [0]
        for d in range(1, max_val + 1):
            cum.append(cum[-1] + cnt[d])

        # answer each query by binary search on cum
        ans = []
        for q in queries:
            d = bisect_right(cum, q)   # smallest d with cum[d] > q
            ans.append(d)
        return ans
