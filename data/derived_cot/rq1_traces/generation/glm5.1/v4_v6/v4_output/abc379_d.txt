```python
import sys
from collections import deque

def main():
    input = sys.stdin.readline
    Q = int(input())
    cur = 0
    q = deque()
    out = []
    for _ in range(Q):
        parts = input().split()
        if parts[0] == '1':
            q.append(cur)
        elif parts[0] == '2':
            T = int(parts[1])
            cur += T
        else: # '3'
            H = int(parts[1])
            X = cur - H
            cnt = 0
            while q and q[0] <= X:
                q.popleft()
                cnt += 1
            out.append(str(cnt))
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
```

```python

import sys

def main():
    input = sys.stdin.readline
    Q = int(input())
    cur = 0                # total elapsed time
    births = []            # list of birth times, in order of insertion
    head = 0               # index of the first plant still present
    out = []

    for _ in range(Q):
        parts = input().split()
        if parts[0] == '1':
            births.append(cur)
        elif parts[0] == '2':
            cur += int(parts[1])
        else:   # type 3
            H = int(parts[1])
            X = cur - H
            cnt = 0
            while head < len(births) and births[head] <= X:
                head += 1
                cnt += 1
            out.append(str(cnt))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
```
