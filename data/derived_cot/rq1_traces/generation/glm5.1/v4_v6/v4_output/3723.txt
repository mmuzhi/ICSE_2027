from typing import List

class Solution:
    def sumOfGoodNumbers(self, nums: List[int], k: int) -> int:
        total = 0
        n = len(nums)
        for i in range(n):
            left = i - k
            right = i + k
            good = True
            if left >= 0 and nums[i] <= nums[left]:
                good = False
            if right < n and nums[i] <= nums[right]:
                good = False
            if good:
                total += nums[i]
        return total
```
