class Solution:
    def maxSubarrays(self, n: int, conflictingPairs: List[List[int]]) -> int:
        # Step 1: Process pairs, swap if needed, build data structures
        pairs_by_a = [[] for _ in range(n+1)]
        best1 = [0]*(n+1)  # max a for each b
        best2 = [0]*(n+1)  # second max a for each b
        pairs_list = []
        for a, b in conflictingPairs:
            if a > b:
                a, b = b, a
            pairs_by_a[a].append(b)
            pairs_list.append((a, b))
            # update best1, best2 for b
            if a > best1[b]:
                best2[b] = best1[b]
                best1[b] = a
            elif a > best2[b]:
                best2[b] = a

        # Step 2: Sweep from n down to 1 to compute min1 and min2
        INF = n+1
        s1, c1, s2, c2, total = INF, 0, INF, 0, 0
        min1 = [0]*(n+2)
        min2 = [0]*(n+2)
        for l in range(n, 0, -1):
            for b in pairs_by_a[l]:
                total += 1
                if b < s1:
                    s2, c2 = s1, c1
                    s1, c1 = b, 1
                elif b == s1:
                    c1 += 1
                elif b < s2:
                    s2, c2 = b, 1
                elif b == s2:
                    c2 += 1
                # else ignore
            if total == 0:
                min1[l] = INF
                min2[l] = INF
            else:
                min1[l] = s1
                if total >= 2:
                    if c1 >= 2:
                        min2[l] = s1
                    else:
                        # c1 == 1
                        min2[l] = s2  # s2 should be <= n
                else:
                    min2[l] = INF

        # Step 3: Compute V
        V = 0
        for l in range(1, n+1):
            V += min1[l] - l

        # Step 4: Prefix sums of min2
        pref = [0]*(n+1)
        for i in range(1, n+1):
            pref[i] = pref[i-1] + min2[i]

        # Step 5: Compute M_below
        M_below = [0]*(n+1)
        cur = 0
        for b in range(1, n+1):
            M_below[b] = cur
            cur = max(cur, best1[b])

        # Step 6: Compute max_gain
        max_gain = 0
        for a0, b0 in pairs_list:
            M = M_below[b0]
            if a0 == best1[b0]:
                M_same = best2[b0]
            else:
                M_same = best1[b0]
            L = max(M, M_same) + 1
            if L <= a0:
                sum_min2 = pref[a0] - pref[L-1]
                cnt = a0 - L + 1
                gain = sum_min2 - b0 * cnt
                if gain > max_gain:
                    max_gain = gain

        # Step 7: Answer
        return V + max_gain
