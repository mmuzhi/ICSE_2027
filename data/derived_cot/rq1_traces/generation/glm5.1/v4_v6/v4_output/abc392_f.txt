import sys

def main():
    import sys
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    it = iter(input_data)
    N = int(next(it))
    P = [0] + [int(next(it)) for _ in range(N)]  # 1-indexed for convenience, but we can just use list of length N
    # We'll use 1-indexed P for simplicity.
    # Initialize BIT
    size = N
    bit = [0] * (size + 2)  # 1-indexed
    # Initialize with ones using lowbit method
    for i in range(1, size+1):
        bit[i] = i & -i
    # ans array
    ans = [0] * (size + 1)
    # Precompute largest power of two <= N
    mask = 1
    while mask <= size:
        mask <<= 1
    mask >>= 1  # now largest power of two <= N
    # Process i from N down to 1
    for i in range(N, 0, -1):
        k = P[i]
        # find k-th empty slot
        idx = 0
        bit_mask = mask
        while bit_mask:
            nxt = idx + bit_mask
            if nxt <= size and bit[nxt] < k:
                k -= bit[nxt]
                idx = nxt
            bit_mask >>= 1
        pos = idx + 1
        ans[pos] = i
        # update BIT: add -1 at pos
        p = pos
        while p <= size:
            bit[p] -= 1
            p += p & -p
    # output
    out = ' '.join(str(ans[i]) for i in range(1, N+1))
    sys.stdout.write(out)

if __name__ == "__main__":
    main()

        ans[pos] = i

        # ---- mark this position as occupied (add -1) ----
        p = pos
        while p <= N:
            bit[p] -= 1
            p += p & -p

    # output
    out = ' '.join(str(ans[i]) for i in range(1, N + 1))
    sys.stdout.write(out)

if __name__ == "__main__":
    main()
