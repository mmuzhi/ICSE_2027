def main():

import sys

MOD = 998244353

def main():
    N, M = map(int, sys.stdin.readline().split())
    # modular inverses of 1..N
    inv = [0] * (N + 1)
    inv[1] = 1
    for i in range(2, N + 1):
        inv[i] = MOD - MOD // i * inv[MOD % i] % MOD

    # H_N modulo MOD
    H = sum(inv[1:]) % MOD

    # 2*N*H_N - 1   (mod MOD)
    term = (2 * N % MOD) * H % MOD
    term = (term - 1) % MOD

    # M^2 * term
    M_mod = M % MOD
    ans = M_mod * M_mod % MOD * term % MOD
    print(ans)

if __name__ == "__main__":
    main()
