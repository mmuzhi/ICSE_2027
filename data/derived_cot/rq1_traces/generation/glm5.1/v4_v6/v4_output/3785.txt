=== ANSWER ===
```python
class Solution:
    def countArrays(self, original: List[int], bounds: List[List[int]]) -> int:
        # Initial interval for the offset c based on the first element
        low = bounds[0][0] - original[0]
        high = bounds[0][1] - original[0]

        # Intersect with intervals from remaining positions
        for i in range(1, len(original)):
            L = bounds[i][0] - original[i]
            R = bounds[i][1] - original[i]
            low = max(low, L)
            high = min(high, R)
            if low > high:   # intersection empty
                return 0

        # Number of integer offsets c in [low, high]
        return high - low + 1
```
