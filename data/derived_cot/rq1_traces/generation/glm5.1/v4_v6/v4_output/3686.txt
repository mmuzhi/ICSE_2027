def compute_z(arr, start):
    m = len(arr) - start
    z = [0] * m
    l = r = 0
    for i in range(1, m):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i + z[i] < m and arr[start + z[i]] == arr[start + i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1
    return z

    n = len(arr)
    z = [0]*n
    l = r = 0
    for i in range(1, n):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i + z[i] < n and arr[z[i]] == arr[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1

def z_function(s):
    n = len(s)
    z = [0]*n
    l = r = 0
    for i in range(1, n):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i + z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1
    return z

def z_function(arr):
    n = len(arr)
    z = [0]*n
    l = r = 0
    for i in range(1, n):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i + z[i] < n and arr[z[i]] == arr[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1
    return z

from typing import List

class Solution:
    def beautifulSplits(self, nums: List[int]) -> int:
        n = len(nums)
        if n < 3:
            return 0
        
        # Z-function to compute LCP with prefix
        def z_function(arr):
            m = len(arr)
            z = [0] * m
            l = r = 0
            for i in range(1, m):
                if i <= r:
                    z[i] = min(r - i + 1, z[i - l])
                while i + z[i] < m and arr[z[i]] == arr[i + z[i]]:
                    z[i] += 1
                if i + z[i] - 1 > r:
                    l, r = i, i + z[i] - 1
            return z
        
        z0 = z_function(nums)  # LCP(0, i) for i>0
        
        prev = [0] * (n + 1)
        cur = [0] * (n + 1)
        ans = 0
        
        for i in range(n - 1, 0, -1):
            cur[n] = 0
            if i <= n - 2:
                lcp0_i = z0[i]
                for j in range(i + 1, n):
                    if nums[i] == nums[j]:
                        cur[j] = prev[j + 1] + 1
                    else:
                        cur[j] = 0
                    # check beautiful split for (i, j)
                    cond1 = (j - i >= i) and (lcp0_i >= i)
                    cond2 = (cur[j] >= j - i)
                    if cond1 or cond2:
                        ans += 1
            else:
                # i > n-2, no need to count splits, just compute cur for DP
                for j in range(i + 1, n):
                    if nums[i] == nums[j]:
                        cur[j] = prev[j + 1] + 1
                    else:
                        cur[j] = 0
            # swap prev and cur for next iteration
            prev, cur = cur, prev
        
        return ans
