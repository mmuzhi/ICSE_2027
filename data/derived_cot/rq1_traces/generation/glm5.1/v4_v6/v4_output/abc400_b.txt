```python

import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N, M = map(int, data)

    LIMIT = 10**9
    total = 1   # N^0
    term = 1    # current power N^i

    for _ in range(M):
        term *= N
        total += term
        if total > LIMIT:
            print("inf")
            return

    print(total)

if __name__ == "__main__":
    solve()
```
