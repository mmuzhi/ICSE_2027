Return the maximum amount of coins you can obtain by collecting k consecutive bags.

Selecting bags at positions [1, 2] gives the maximum number of coins: 3 + 3 = 6.

def maximumCoins(self, coins: List[List[int]], k: int) -> int:
    coins.sort(key=lambda x: x[0])
    n = len(coins)
    Ls = [c[0] for c in coins]
    Rs = [c[1] for c in coins]
    Cs = [c[2] for c in coins]
    pref = [0]*n
    total = 0
    for i in range(n):
        length = Rs[i] - Ls[i] + 1
        total += Cs[i] * length
        pref[i] = total

    def query(L, R):
        if L > R:
            return 0
        # find first interval with r_i >= L
        i = bisect_left(Rs, L)
        if i == n:
            return 0
        # find last interval with l_i <= R
        j = bisect_right(Ls, R) - 1
        if j < 0:
            return 0
        if i > j:
            return 0
        if i == j:
            overlap = min(R, Rs[i]) - max(L, Ls[i]) + 1
            return Cs[i] * overlap
        # left partial
        left_overlap = min(R, Rs[i]) - max(L, Ls[i]) + 1
        res = Cs[i] * left_overlap
        # right partial
        right_overlap = min(R, Rs[j]) - max(L, Ls[j]) + 1
        res += Cs[j] * right_overlap
        # full intervals between
        if j > i+1:
            res += pref[j-1] - pref[i]
        return res

    # generate boundaries
    boundaries = set()
    for i in range(n):
        boundaries.add(Ls[i])
        boundaries.add(Rs[i] + 1)
    candidates = set()
    for b in boundaries:
        candidates.add(b)
        candidates.add(b - k)
    ans = 0
    for x in candidates:
        L = x
        R = x + k - 1
        ans = max(ans, query(L, R))
    return ans

from bisect import bisect_left, bisect_right

class Solution:
    def maximumCoins(self, coins: List[List[int]], k: int) -> int:
        # sort intervals by l_i
        coins.sort(key=lambda x: x[0])
        n = len(coins)
        Ls = [c[0] for c in coins]
        Rs = [c[1] for c in coins]
        Cs = [c[2] for c in coins]
        
        # prefix sums of total coins per interval (full interval)
        pref = [0] * n
        total = 0
        for i in range(n):
            length = Rs[i] - Ls[i] + 1
            total += Cs[i] * length
            pref[i] = total
        
        # function to compute total coins in [L, R] inclusive
        def query(L: int, R: int) -> int:
            if L > R:
                return 0
            # first interval with r_i >= L
            i = bisect_left(Rs, L)
            if i == n:
                return 0
            # last interval with l_i <= R
            j = bisect_right(Ls, R) - 1
            if j < 0:
                return 0
            if i > j:
                return 0
            if i == j:
                overlap = min(R, Rs[i]) - max(L, Ls[i]) + 1
                return Cs[i] * overlap
            # left partial
            left_overlap = min(R, Rs[i]) - max(L, Ls[i]) + 1
            res = Cs[i] * left_overlap
            # right partial
            right_overlap = min(R, Rs[j]) - max(L, Ls[j]) + 1
            res += Cs[j] * right_overlap
            # full intervals between
            if j > i + 1:
                res += pref[j-1] - pref[i]
            return res
        
        # collect all boundaries where f changes
        boundaries = set()
        for i in range(n):
            boundaries.add(Ls[i])
            boundaries.add(Rs[i] + 1)
        
        # generate candidate x values
        candidates = set()
        for b in boundaries:
            candidates.add(b)
            candidates.add(b - k)
        
        ans = 0
        for x in candidates:
            L = x
            R = x + k - 1
            ans = max(ans, query(L, R))
        
        return ans
