```python

import sys

def main():
    S = sys.stdin.readline().strip()
    n = len(S)

    zero_run = 0          # length of current consecutive zeros
    total_pairs = 0       # sum of floor(zero_run / 2) over all runs

    for ch in S:
        if ch == '0':
            zero_run += 1
        else:
            total_pairs += zero_run // 2
            zero_run = 0

    # trailing zeros
    total_pairs += zero_run // 2

    ans = n - total_pairs
    print(ans)

if __name__ == "__main__":
    main()
```
