class Solution:
    def countOfSubstrings(self, word: str, k: int) -> int:
        vowels = set('aeiou')
        n = len(word)
        ans = 0
        for i in range(n):
            seen = set()
            cons = 0
            for j in range(i, n):
                ch = word[j]
                if ch in vowels:
                    seen.add(ch)
                else:
                    cons += 1
                if len(seen) == 5 and cons == k:
                    ans += 1
        return ans

class Solution:
    def countOfSubstrings(self, word: str, k: int) -> int:
        vowels = set('aeiou')
        n = len(word)
        ans = 0

        for i in range(n):
            seen = set()   # vowels encountered in the current substring
            cons = 0       # number of consonants in the current substring
            for j in range(i, n):
                ch = word[j]
                if ch in vowels:
                    seen.add(ch)
                else:
                    cons += 1
                # check the conditions
                if len(seen) == 5 and cons == k:
                    ans += 1

        return ans
