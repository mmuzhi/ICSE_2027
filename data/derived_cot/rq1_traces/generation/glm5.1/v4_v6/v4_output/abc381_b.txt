=== ANSWER ===
```python

import sys
from collections import Counter

def solve() -> None:
    S = sys.stdin.readline().strip()
    n = len(S)

    # Condition 1: length must be even
    if n % 2 != 0:
        print("No")
        return

    # Condition 2: each pair of consecutive characters must be equal
    for i in range(0, n, 2):
        if S[i] != S[i+1]:
            print("No")
            return

    # Condition 3: every character appears exactly twice
    cnt = Counter(S)
    for v in cnt.values():
        if v != 2:
            print("No")
            return

    print("Yes")

if __name__ == "__main__":
    solve()
```
