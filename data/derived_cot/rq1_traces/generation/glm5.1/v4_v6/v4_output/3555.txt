```python
class Solution:
    def getFinalState(self, nums: List[int], k: int, multiplier: int) -> List[int]:
        for _ in range(k):
            # find index of first minimum
            min_val = nums[0]
            min_idx = 0
            for i, val in enumerate(nums):
                if val < min_val:
                    min_val = val
                    min_idx = i
            # multiply
            nums[min_idx] *= multiplier
        return nums
```
