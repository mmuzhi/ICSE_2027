class Solution:
    def minValidStrings(self, words: List[str], target: str) -> int:
        # Build trie
        root = {}
        for w in words:
            node = root
            for ch in w:
                if ch not in node:
                    node[ch] = {}
                node = node[ch]
        n = len(target)
        L = [0]*n
        for i in range(n):
            node = root
            j = i
            while j < n and target[j] in node:
                node = node[target[j]]
                j += 1
            L[i] = j - i
        # Now compute min jumps
        jumps = 0
        cur_end = 0
        farthest = 0
        for i in range(n):
            farthest = max(farthest, i + L[i])
            if i == cur_end:
                # need to jump
                if i == farthest:  # cannot progress
                    return -1
                jumps += 1
                cur_end = farthest
                if cur_end >= n:
                    break
        return jumps if cur_end >= n else -1

class Solution:
    def minValidStrings(self, words: List[str], target: str) -> int:
        # Build trie
        root = {}
        for w in words:
            node = root
            for ch in w:
                if ch not in node:
                    node[ch] = {}

```python

from typing import List

class Solution:
    def minValidStrings(self, words: List[str], target: str) -> int:
        # ---------- build trie ----------
        root = {}
        for w in words:
            node = root
            for ch in w:
                if ch not in node:
                    node[ch] = {}
                node = node[ch]

        n = len(target)
        L = [0] * n          # L[i] = longest valid prefix starting at i

        # ---------- compute L[i] ----------
        for i in range(n):
            node = root
            j = i
            while j < n and target[j] in node:
                node = node[target[j]]
                j += 1
            L[i] = j - i

        # ---------- minimum jumps (Jump Game II) ----------
        jumps = 0
        cur_end = 0   # farthest index reachable with `jumps` jumps
        farthest = 0  # farthest index reachable with `jumps+1` jumps

        for i in range(n):
            farthest = max(farthest, i + L[i])
            if i == cur_end:               # we have processed all positions of current layer
                if i == farthest:         # cannot move forward
                    return -1
                jumps += 1
                cur_end = farthest
                if cur_end >= n:           # reached the end
                    return jumps

        return -1   # should never be reached if the loop finishes without returning
```
