import sys

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input())
    people = []
    total = 0
    for _ in range(N):
        a, b = map(int, input().split())
        people.append((a, b))
        total += b
    if total % 3 != 0:
        print(-1)
        return
    T = total // 3
    INF = 10**9
    dp = [[INF]*(T+1) for _ in range(T+1)]
    dp[0][0] = 0
    cur_total = 0
    for a, b in people:
        cur_total += b
        prev_total = cur_total - b
        newdp = [[INF]*(T+1) for _ in range(T+1)]
        # Precompute cost additions
        cost1 = 0 if a == 1 else 1
        cost2 = 0 if a == 2 else 1
        cost3 = 0 if a == 3 else 1
        for s1 in range(T+1):
            row = dp[s1]
            for s2 in range(T+1):
                cur = row[s2]
                if cur == INF:
                    continue
                s3_prev = prev_total - s1 - s2
                # team1
                if s1 + b <= T:
                    ns1 = s1 + b
                    new_cost = cur + cost1
                    if new_cost < newdp[ns1][s2]:
                        newdp[ns1][s2] = new_cost
                # team2
                if s2 + b <= T:
                    ns2 = s2 + b
                    new_cost = cur + cost2
                    if new_cost < newdp[s1][ns2]:
                        newdp[s1][ns2] = new_cost
                # team3
                if s3_prev + b <= T:
                    new_cost = cur + cost3
                    if new_cost < newdp[s1][s2]:
                        newdp[s1][s2] = new_cost
        dp = newdp
    ans = dp[T][T]
    if ans == INF:
        print(-1)
    else:
        print(ans)

if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    people = []
    total = 0
    for _ in range(N):
        a, b = map(int, input().split())
        people.append((a, b))
        total += b

    if total % 3 != 0:
        print(-1)
        return

    T = total // 3
    INF = 10 ** 9

    # dp[s1][s2] = min switches after processing some people,
    # with sum in team1 = s1, team2 = s2 (team3 = processed_sum - s1 - s2)
    dp = [[INF] * (T + 1) for _ in range(T + 1)]
    dp[0][0] = 0

    cur_total = 0   # sum of strengths processed so far

    for a, b in people:
        cur_total += b
        prev_total = cur_total - b

        newdp = [[INF] * (T + 1) for _ in range(T + 1)]

        # cost to put this person into each team
        cost1 = 0 if a == 1 else 1
        cost2 = 0 if a == 2 else 1
        cost3 = 0 if a == 3 else 1

        for s1 in range(T + 1):
            row = dp[s1]
            for s2 in range(T + 1):
                cur = row[s2]
                if cur == INF:
                    continue

                s3_prev = prev_total - s1 - s2
                if s3_prev < 0:          # should never happen, but guard
                    continue

                # assign to team 1
                if s1 + b <= T:
                    ns1 = s1 + b
                    new_cost = cur + cost1
                    if new_cost < newdp[ns1][s2]:
                        newdp[ns1][s2] = new_cost

                # assign to team 2
                if s2 + b <= T:
                    ns2 = s2 + b
                    new_cost = cur + cost2
                    if new_cost < newdp[s1][ns2]:
                        newdp[s1][ns2] = new_cost

                # assign to team 3
                if s3_prev + b <= T:
                    new_cost = cur + cost3
                    if new_cost < newdp[s1][s2]:
                        newdp[s1][s2] = new_cost

        dp = newdp

    ans = dp[T][T]
    print(-1 if ans == INF else ans)


if __name__ == "__main__":
    solve()
