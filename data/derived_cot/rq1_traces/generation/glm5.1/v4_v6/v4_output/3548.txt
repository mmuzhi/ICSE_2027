import math
from itertools import product

class Solution:
    def countGoodIntegers(self, n: int, k: int) -> int:
        good_multisets = set()
        half = (n + 1) // 2
        for first in range(1, 10):
            for rest in product(range(10), repeat=half - 1):
                half_str = str(first) + ''.join(map(str, rest))
                if n % 2 == 0:
                    pal_str = half_str + half_str[::-1]
                else:
                    pal_str = half_str + half_str[:-1][::-1]
                pal_int = int(pal_str)
                if pal_int % k == 0:
                    counts = [0] * 10
                    for ch in pal_str:
                        counts[int(ch)] += 1
                    good_multisets.add(tuple(counts))
        
        # Precompute factorials
        fact = [math.factorial(i) for i in range(n + 1)]
        
        ans = 0
        for counts in good_multisets:
            total = fact[n]
            for c in counts:
                total //= fact[c]
            if counts[0] > 0:
                start0 = fact[n - 1]
                start0 //= fact[counts[0] - 1]
                for i in range(1, 10):
                    start0 //= fact[counts[i]]
                valid = total - start0
            else:
                valid = total
            ans += valid
        return ans
