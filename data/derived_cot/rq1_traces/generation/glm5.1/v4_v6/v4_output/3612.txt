        return False

```python
from typing import List

class Solution:
    def hasIncreasingSubarrays(self, nums: List[int], k: int) -> bool:
        n = len(nums)
        # Iterate over all possible start indices of the first subarray
        for a in range(n - 2 * k + 1):
            # Check if the first subarray is strictly increasing
            first_inc = all(nums[i] < nums[i + 1] for i in range(a, a + k - 1))
            if not first_inc:
                continue
            # Check if the second subarray is strictly increasing
            second_inc = all(nums[i] < nums[i + 1] for i in range(a + k, a + 2 * k - 1))
            if second_inc:
                return True
        return False
```
