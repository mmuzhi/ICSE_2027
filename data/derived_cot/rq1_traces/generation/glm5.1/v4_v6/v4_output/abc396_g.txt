def fwht(a):
    n = len(a)
    step = 1
    while step < n:
        for i in range(0, n, step*2):
            for j in range(i, i+step):
                u = a[j]
                v = a[j+step]
                a[j] = u + v
                a[j+step] = u - v
        step <<= 1

def main():
    import sys
    input = sys.stdin.readline
    H, W = map(int, input().split())
    N = 1 << W
    cnt = [0] * N
    for _ in range(H):
        s = input().strip()
        # convert to int
        val = int(s, 2)
        cnt[val] += 1
    # compute g
    g = [0] * N
    for x in range(N):

        g[x] = min(pc, W - pc)
    # convolution
    fa = cnt[:]
    fb = g[:]
    fwht(fa)
    fwht(fb)
    for i in range(N):
        fa[i] *= fb[i]
    fwht(fa)
    for i in range(N):
        fa[i] //= N
    ans = min(fa)
    print(ans)

if __name__ == "__main__":

import sys

def fwht(a):
    """In-place XOR Walsh-Hadamard Transform."""
    n = len(a)
    step = 1
    while step < n:
        for i in range(0, n, step * 2):
            for j in range(i, i + step):
                u = a[j]
                v = a[j + step]
                a[j] = u + v
                a[j + step] = u - v
        step <<= 1

def main():
    input = sys.stdin.readline
    H, W = map(int, input().split())
    N = 1 << W

    # frequency of each row pattern
    cnt = [0] * N
    for _ in range(H):
        s = input().strip()
        val = int(s, 2)          # interpret as binary number
        cnt[val] += 1

    # precompute popcounts
    pop = [0] * N
    for i in range(1, N):
        pop[i] = pop[i >> 1] + (i & 1)

    # g[x] = min(popcount(x), W - popcount(x))
    g = [0] * N
    for i in range(N):
        pc = pop[i]
        g[i] = pc if pc <= W - pc else W - pc

    # XOR convolution: ans[C] = sum_p cnt[p] * g(p xor C)
    fa = cnt[:]
    fb = g[:]
    fwht(fa)
    fwht(fb)
    for i in range(N):
        fa[i] *= fb[i]
    fwht(fa)
    for i in range(N):
        fa[i] //= N   # inverse transform scaling

    answer = min(fa)
    print(answer)

if __name__ == "__main__":
    main()
