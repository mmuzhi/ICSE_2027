def maximumValueSum(self, board):
    m = len(board)
    n = len(board[0])
    candidates = []
    for i in range(m):
        # get top 3 values with columns
        row_vals = [(board[i][j], j) for j in range(n)]
        row_vals.sort(reverse=True)  # sort by value descending
        # take first 3
        for k in range(3):
            val, j = row_vals[k]
            candidates.append((i, j, val))
    # now brute-force
    best = -10**18  # since values can be -1e9, sum of three could be -3e9, so use -inf
    L = len(candidates)
    for a in range(L):
        i1, j1, v1 = candidates[a]
        for b in range(a+1, L):
            i2, j2, v2 = candidates[b]
            if i1 == i2 or j1 == j2:  # same row or column invalid for pair, but triple might still be invalid later; we can skip early to save time
                continue
            for c in range(b+1, L):
                i3, j3, v3 = candidates[c]
                if i1 == i3 or i2 == i3 or j1 == j3 or j2 == j3:
                    continue
                # all rows and cols distinct
                total = v1 + v2 + v3
                if total > best:
                    best = total
    return best

from typing import List

class Solution:
    def maximumValueSum(self, board: List[List[int]]) -> int:
        m = len(board)
        n = len(board[0])
        candidates = []
        for i in range(m):
            # Create list of (value, col) for row i
            row_vals = [(board[i][j], j) for j in range(n)]
            # Sort descending by value
            row_vals.sort(key=lambda x: -x[0])
            # Take top 3
            for k in range(3):
                val, col = row_vals[k]
                candidates.append((i, col, val))
        
        best = -10**18  # sufficiently small
        L = len(candidates)
        for a in range(L):
            i1, j1, v1 = candidates[a]
            for b in range(a+1, L):
                i2, j2, v2 = candidates[b]
                if i1 == i2 or j1 == j2:
                    continue
                for c in range(b+1, L):
                    i3, j3, v3 = candidates[c]
                    if i1 == i3 or i2 == i3 or j1 == j3 or j2 == j3:
                        continue
                    total = v1 + v2 + v3
                    if total > best:
                        best = total
        return best
