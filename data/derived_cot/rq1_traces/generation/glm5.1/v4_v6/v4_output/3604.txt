An event is being held for n performers. When a performer arrives, they are assigned to one of the x stages. All performers assigned to the same stage will perform together as a band, though some stages might remain empty.

class Solution:
    def numberOfWays(self, n: int, x: int, y: int) -> int:
        MOD = 10**9 + 7
        max_val = max(n, x)
        fact = [1] * (max_val + 1)

        invfact = [1] * (max_val + 1)
        invfact[max_val] = pow(fact[max_val], MOD-2, MOD)
        for i in range(max_val, 0, -1):
            invfact[i-1] = invfact[i] * i % MOD

        # Stirling numbers of the second kind S(n, k) for k=0..K
        dp = [0] * (K + 1)
        dp[0] = 1

            for j in range(upper, 0, -1):

            dp[0] = 0  # S(i,0)=0 for i>=1
        
        # Precompute powers of y
        y_pow = [1] * (K + 1)
        for i in range(1, K+1):
            y_pow[i] = y_pow[i-1] * y % MOD
        
        fact_x = fact[x]
        ans = 0
        for k in range(1, K+1):
            term = fact_x * invfact[x - k] % MOD
            term = term * dp[k] % MOD
            term = term * y_pow[k] % MOD
            ans = (ans + term) % MOD
        return ans
