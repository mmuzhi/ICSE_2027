class Fenwick:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+2)  # 1-indexed
    def add(self, idx, delta):
        i = idx+1
        while i <= self.n+1:
            self.bit[i] += delta
            i += i & -i
    def sum(self, idx):
        # prefix sum 0..idx inclusive
        if idx < 0:
            return 0
        i = idx+1
        res = 0
        while i>0:
            res += self.bit[i]
            i -= i & -i
        return res
    def range_sum(self, l, r):
        if l > r:
            return 0
        return self.sum(r) - self.sum(l-1)

        self.n = n
        self.bit = [0]*(n+2)
    def add(self, idx, delta):
        i = idx+1
        while i <= self.n+1:

            i += i & -i

            i -= i & -i
        return s
    def range_sum(self, l, r):
        if l > r:
            return 0

class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        n = len(nums)
        intervals = []
        for i in range(n-4):
            for j in range(i+4, n):
                prod = nums[i] * nums[j]
                intervals.append((prod, i, j))
        intervals.sort(key=lambda x: (x[0], x[1]))
        
        class Fenwick:
            def __init__(self, n):
                self.n = n
                self.bit = [0]*(n+2)
            def add(self, idx, delta):
                i = idx+1
                while i <= self.n+1:
                    self.bit[i] += delta
                    i += i & -i
            def sum(self, idx):
                if idx < 0:
                    return 0
                i = idx+1
                s = 0
                while i>0:
                    s += self.bit[i]
                    i -= i & -i
                return s
            def range_sum(self, l, r):
                if l > r:
                    return 0
                return self.sum(r) - self.sum(l-1)
        
        bit = Fenwick(n)
        total = 0
        i = 0
        M = len(intervals)
        while i < M:
            prod = intervals[i][0]
            start = i
            while i < M and intervals[i][0] == prod:
                i += 1
            end = i
            # process group from start to end-1
            added_rs = []
            add_ptr = start
            for k in range(start, end):
                l, r = intervals[k][1], intervals[k][2]
                # add eligible first intervals
                while add_ptr < end and intervals[add_ptr][1] <= l - 2:
                    r_add = intervals[add_ptr][2]
                    bit.add(r_add, 1)
                    added_rs.append(r_add)
                    add_ptr += 1
                # query
                # Since r-l >=4, l+2 <= r-2 always true
                cnt = bit.range_sum(l+2, r-2)
                total += cnt
            # clear bit
            for r_add in added_rs:
                bit.add(r_add, -1)
        return total
