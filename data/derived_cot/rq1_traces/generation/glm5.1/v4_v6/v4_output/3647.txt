class Solution:
    def maxRemoval(self, nums: List[int], queries: List[List[int]]) -> int:
        n = len(nums)
        m = len(queries)
        queries.sort(key=lambda x: x[0])
        heap = []
        end_counts = [0] * n
        q_idx = 0
        cur_coverage = 0
        selected = 0
        for i in range(n):
            # add intervals that start at or before i
            while q_idx < m and queries[q_idx][0] <= i:
                l, r = queries[q_idx]
                if r >= i:  # only add if it can cover current or future points
                    heapq.heappush(heap, (-r, r))
                q_idx += 1
            # meet demand
            while cur_coverage < nums[i]:
                if not heap:
                    return -1
                neg_r, r = heapq.heappop(heap)
                if r < i:
                    continue  # expired
                # select this interval
                selected += 1
                cur_coverage += 1
                end_counts[r] += 1
            # move to next index: subtract intervals ending at i
            cur_coverage -= end_counts[i]
        return m - selected

```python
import heapq
from typing import List

class Solution:
    def maxRemoval(self, nums: List[int], queries: List[List[int]]) -> int:
        n = len(nums)
        m = len(queries)
        # sort queries by left endpoint
        queries.sort(key=lambda x: x[0])
        heap = []  # max-heap by right endpoint (store -r)
        end_counts = [0] * n  # number of selected intervals ending at each index
        q_idx = 0
        cur_coverage = 0
        selected = 0
        
        for i in range(n):
            # add all queries that start at or before i
            while q_idx < m and queries[q_idx][0] <= i:
                l, r = queries[q_idx]
                if r >= i:  # only useful if it can cover current or future points
                    heapq.heappush(heap, (-r, r))
                q_idx += 1
            
            # ensure coverage meets demand at i
            while cur_coverage < nums[i]:
                if not heap:
                    return -1
                neg_r, r = heapq.heappop(heap)
                if r < i:
                    continue  # expired, discard
                # select this interval
                selected += 1
                cur_coverage += 1
                end_counts[r] += 1
            
            # move to next index: intervals ending at i no longer cover i+1
            cur_coverage -= end_counts[i]
        
        return m - selected
```
