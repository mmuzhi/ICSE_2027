```python

import sys

def solve() -> None:
    input = sys.stdin.readline
    N, K = map(int, input().split())
    S = input().strip()

    ans = 0
    cnt = 0
    for ch in S:
        if ch == 'O':
            cnt += 1
        else:
            ans += cnt // K
            cnt = 0
    ans += cnt // K   # last segment

    print(ans)

if __name__ == "__main__":
    solve()
```
