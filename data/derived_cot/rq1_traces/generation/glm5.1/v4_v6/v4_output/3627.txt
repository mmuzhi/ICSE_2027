```python
import heapq
from typing import List

class Solution:
    def minTimeToReach(self, moveTime: List[List[int]]) -> int:
        n = len(moveTime)
        m = len(moveTime[0])
        INF = 10**18
        dist = [[INF]*m for _ in range(n)]
        dist[0][0] = 0
        heap = [(0,0,0)]
        directions = [(1,0),(-1,0),(0,1),(0,-1)]
        while heap:
            t,i,j = heapq.heappop(heap)
            if t != dist[i][j]:
                continue
            if i == n-1 and j == m-1:
                # we could break early, but not necessary
                # break
                pass
            for di,dj in directions:
                ni, nj = i+di, j+dj
                if 0 <= ni < n and 0 <= nj < m:
                    new_t = max(t, moveTime[ni][nj]) + 1
                    if new_t < dist[ni][nj]:
                        dist[ni][nj] = new_t
                        heapq.heappush(heap, (new_t, ni, nj))
        return dist[n-1][m-1]
```

```python

import heapq
from typing import List

class Solution:
    def minTimeToReach(self, moveTime: List[List[int]]) -> int:
        n = len(moveTime)
        m = len(moveTime[0])
        INF = 10**18
        dist = [[INF] * m for _ in range(n)]
        dist[0][0] = 0
        heap = [(0, 0, 0)]  # (time, row, col)
        directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]

        while heap:
            t, i, j = heapq.heappop(heap)
            if t != dist[i][j]:
                continue
            # First time we extract the destination -> optimal time
            if i == n - 1 and j == m - 1:
                return t
            for di, dj in directions:
                ni, nj = i + di, j + dj
                if 0 <= ni < n and 0 <= nj < m:
                    # Earliest arrival at (ni, nj) when leaving (i, j) at time t
                    new_t = max(t, moveTime[ni][nj]) + 1
                    if new_t < dist[ni][nj]:
                        dist[ni][nj] = new_t
                        heapq.heappush(heap, (new_t, ni, nj))
        return dist[n-1][m-1]  # fallback (should never be reached)
```
