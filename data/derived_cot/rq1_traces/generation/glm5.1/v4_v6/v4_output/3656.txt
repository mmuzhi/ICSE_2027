As argued, if 3k <= n, then after each operation i (1 <= i <= k), the remaining length is n - 3(i-1) which is >= n - 3(k-1) >= 3, so we always have at least 3 elements before each operation, thus we remove exactly 3 each time. So after k ops, we have removed exactly 3k elements, leaving nums[3k:]. If 3k > n, then at some point before the k-th operation, the length becomes less than 3, and the next operation removes all, making empty. So after k ops, array is empty. So the formula holds.

from typing import List

class Solution:
    def minimumOperations(self, nums: List[int]) -> int:
        n = len(nums)
        k = 0
        while True:
            start = 3 * k
            if start >= n:
                return k
            # check distinctness of nums[start:]
            seen = set()
            distinct = True
            for i in range(start, n):
                if nums[i] in seen:
                    distinct = False
                    break
                seen.add(nums[i])
            if distinct:
                return k
            k += 1
