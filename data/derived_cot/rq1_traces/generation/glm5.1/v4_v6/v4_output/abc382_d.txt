```python
import sys
import math

def solve():
    import sys
    sys.setrecursionlimit(10000)
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    K = M - 10 * (N - 1)
    # Compute X
    X = math.comb(N + K - 1, N)
    print(X)
    # Generate sequences
    B = [0] * N
    def dfs(idx, prev):
        if idx == N:
            # produce A
            A = [B[i] + 10 * i for i in range(N)]
            print(' '.join(map(str, A)))
            return
        for val in range(prev, K+1):
            B[idx] = val
            dfs(idx+1, val)
    dfs(0, 1)

if __name__ == "__main__":
    solve()
```

```python
def comb(n, r):
    if r < 0 or r > n:
        return 0
    r = min(r, n-r)
    res = 1
    for i in range(1, r+1):
        res = res * (n - i + 1) // i
    return res
```
