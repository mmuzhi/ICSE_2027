class LazySegmentTree:
    def __init__(self, n, init_val):
        self.size = 1
        while self.size < n:
            self.size <<= 1
        self.tree = [0] * (2 * self.size)
        self.lazy = [0] * (2 * self.size)
        for i in range(n):
            self.tree[self.size + i] = init_val[i]
        for i in range(self.size - 1, 0, -1):
            self.tree[i] = max(self.tree[2*i], self.tree[2*i+1])
        self.h = self.size.bit_length() - 1

    def apply(self, node, val):
        self.tree[node] += val
        if node < self.size:
            self.lazy[node] += val

    def push(self, node):
        for s in range(self.h, 0, -1):
            i = node >> s
            if self.lazy[i]:
                self.apply(2*i, self.lazy[i])
                self.apply(2*i+1, self.lazy[i])
                self.lazy[i] = 0

    def pull(self, node):
        while node > 1:
            node >>= 1
            self.tree[node] = max(self.tree[2*node], self.tree[2*node+1]) + self.lazy[node]

    def range_add(self, l, r, val):
        l += self.size
        r += self.size
        l0, r0 = l, r
        self.push(l0)
        self.push(r0 - 1)
        while l < r:
            if l & 1:
                self.apply(l, val)
                l += 1
            if r & 1:
                r -= 1
                self.apply(r, val)
            l >>= 1
            r >>= 1
        self.pull(l0)
        self.pull(r0 - 1)

    def range_max(self, l, r):
        l += self.size
        r += self.size
        self.push(l)
        self.push(r - 1)
        res = -10**18
        while l < r:
            if l & 1:
                res = max(res, self.tree[l])
                l += 1
            if r & 1:
                r -= 1
                res = max(res, self.tree[r])
            l >>= 1
            r >>= 1
        return res
```
