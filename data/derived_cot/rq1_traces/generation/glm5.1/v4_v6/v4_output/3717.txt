class BIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+1)
    def add(self, idx, delta):
        while idx <= self.n:
            self.bit[idx] += delta
            idx += idx & -idx
    def sum(self, idx):
        res = 0
        while idx>0:
            res += self.bit[idx]
            idx -= idx & -idx
        return res
    def range_sum(self, l, r):
        return self.sum(r) - self.sum(l-1)
    def find_kth(self, k):
        # find smallest idx such that prefix sum >= k
        idx = 0
        bit_mask = 1 << (self.n.bit_length() - 1)
        while bit_mask:
            next_idx = idx + bit_mask
            if next_idx <= self.n and self.bit[next_idx] < k:
                k -= self.bit[next_idx]
                idx = next_idx
            bit_mask >>= 1
        return idx + 1

```python
from typing import List

class Solution:
    def minOperations(self, nums: List[int], x: int, k: int) -> int:
        n = len(nums)
        L = n - x + 1  # number of possible start positions
        
        # Coordinate compression
        vals = sorted(set(nums))
        comp = {v: i+1 for i, v in enumerate(vals)}  # 1-indexed
        m = len(vals)
        
        # Fenwick trees
        class BIT:
            def __init__(self, n):
                self.n = n
                self.bit = [0]*(n+1)
            def add(self, idx, delta):
                while idx <= self.n:
                    self.bit[idx] += delta
                    idx += idx & -idx
            def sum(self, idx):
                res = 0
                while idx > 0:
                    res += self.bit[idx]
                    idx -= idx & -idx
                return res
            def find_kth(self, k):
                # find smallest idx such that prefix sum >= k
                idx = 0
                bit_mask = 1 << (self.n.bit_length() - 1)
                while bit_mask:
                    nxt = idx + bit_mask
                    if nxt <= self.n and self.bit[nxt] < k:
                        k -= self.bit[nxt]
                        idx = nxt
                    bit_mask >>= 1
                return idx + 1
        
        freqBIT = BIT(m)
        sumBIT = BIT(m)
        total_sum = 0
        
        # Initialize first window
        for i in range(x):
            v = nums[i]
            idx = comp[v]
            freqBIT.add(idx, 1)
            sumBIT.add(idx, v)
            total_sum += v
        
        cost = [0]*L
        # Compute cost for each start
        for start in range(L):
            target = (x + 1) // 2
            median_idx = freqBIT.find_kth(target)
            median_val = vals[median_idx-1]
            left_cnt = freqBIT.sum(median_idx)
            left_sum = sumBIT.sum(median_idx)
            cost[start] = total_sum - 2*left_sum + median_val * (2*left_cnt - x)
            
            if start == L-1:
                break
            # Slide window
            # remove nums[start]
            v_rem = nums[start]
            idx_rem = comp[v_rem]
            freqBIT.add(idx_rem, -1)
            sumBIT.add(idx_rem, -v_rem)
            total_sum -= v_rem
            # add nums[start+x]
            v_add = nums[start+x]
            idx_add = comp[v_add]
            freqBIT.add(idx_add, 1)
            sumBIT.add(idx_add, v_add)
            total_sum += v_add
        
        # DP to select k non-overlapping intervals
        INF = 10**18
        prev = [0]*(L+1)  # dp[0][i] = 0
        for t in range(1, k+1):
            cur = [INF]*(L+1)
            # cur[0] = INF already
            for i in range(1, L+1):
                # skip start i-1
                cur[i] = cur[i-1]
                if t == 1:
                    candidate = cost[i-1]
                else:
                    if i - x >= 0:
                        candidate = prev[i-x] + cost[i-1]
                    else:
                        candidate = INF
                if candidate < cur[i]:
                    cur[i] = candidate
            prev = cur
        return prev[L]
```
