    def __init__(self, data):
        self.n = len(data)
        size = 4 * self.n
        self.min = [0] * size
        self.max = [0] * size
        self.lazy = [0] * size
        self._build(1, 0, self.n-1, data)

    def _build(self, node, l, r, data):
        if l == r:
            self.min[node] = self.max[node] = data[l]
        else:
            mid = (l+r)//2
            self._build(node*2, l, mid, data)
            self._build(node*2+1, mid+1, r, data)
            self.min[node] = min(self.min[node*2], self.min[node*2+1])
            self.max[node] = max(self.max[node*2], self.max[node*2+1])

    def _apply(self, node, add):
        self.min[node] += add
        self.max[node] += add
        self.lazy[node] += add

def main():
    import sys
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    intervals = [(next(it), next(it)) for _ in range(N)]
    Q = next(it)
    queries = [next(it) for _ in range(Q)]

    # unique X
    uniq = sorted(set(queries))
    K = len(uniq)
    idx_map = {x: i for i, x in enumerate(uniq)}

    # initial values
    init = uniq[:]  # list of X values

    seg = SegTree(init)

    for L, R in intervals:
        # find l index
        l = seg.find_first_ge(L)
        if l == K:  # no index with val >= L
            continue
        # find first index with val > R
        r_plus = seg.find_first_greater(R)
        r = r_plus - 1
        if l > r:
            continue
        seg.range_add(l, r, 1)

    # collect final values
    final_vals = seg.collect()

    # answer queries
    out = []
    for x in queries:
        i = idx_map[x]
        out.append(str(final_vals[i]))
    sys.stdout.write("\n".join(out))

class SegTree:
    def __init__(self, data):
        self.n = len(data)
        self.minv = [0] * (4 * self.n)
        self.maxv = [0] * (4 * self.n)
        self.lazy = [0] * (4 * self.n)
        self._build(1, 0, self.n-1, data)

    def _build(self, node, l, r, data):
        if l == r:
            val = data[l]
            self.minv[node] = val
            self.maxv[node] = val
        else:
            mid = (l+r)//2
            self._build(node*2, l, mid, data)
            self._build(node*2+1, mid+1, r, data)
            self.minv[node] = min(self.minv[node*2], self.minv[node*2+1])
            self.maxv[node] = max(self.maxv[node*2], self.maxv[node*2+1])

    def _push(self, node):
        if self.lazy[node] != 0:
            add = self.lazy[node]
            left = node*2
            right = node*2+1
            self.minv[left] += add
            self.maxv[left] += add
            self.lazy[left] += add
            self.minv[right] += add
            self.maxv[right] += add
            self.lazy[right] += add
            self.lazy[node] = 0

    def range_add(self, ql, qr, add):
        self._range_add(1, 0, self.n-1, ql, qr, add)

    def _range_add(self, node, l, r, ql, qr, add):
        if ql > r or qr < l:
            return
        if ql <= l and r <= qr:
            self.minv[node] += add
            self.maxv[node] += add
            self.lazy[node] += add
            return
        self._push(node)
        mid = (l+r)//2
        self._range_add(node*2, l, mid, ql, qr, add)
        self._range_add(node*2+1, mid+1, r, ql, qr, add)
        self.minv[node] = min(self.minv[node*2], self.minv[node*2+1])
        self.maxv[node] = max(self.maxv[node*2], self.maxv[node*2+1])

    def find_first_ge(self, L):
        # return smallest index i with val[i] >= L, or n if none
        if self.maxv[1] < L:
            return self.n
        if self.minv[1] >= L:
            return 0
        node = 1
        l = 0
        r = self.n-1
        while l < r:
            self._push(node)
            mid = (l+r)//2
            left = node*2
            # right = node*2+1
            if self.maxv[left] >= L:
                node = left
                r = mid
            else:
                node = node*2+1
                l = mid+1
        return l

    def find_first_greater(self, R):
        # return smallest index i with val[i] > R, or n if none
        if self.minv[1] > R:
            return 0
        if self.maxv[1] <= R:
            return self.n
        node = 1
        l = 0
        r = self.n-1
        while l < r:
            self._push(node)
            mid = (l+r)//2
            left = node*2
            right = node*2+1
            if self.minv[left] > R:
                # all in left > R
                return l
            elif self.maxv[left] <= R:
                # left has no >R, go right
                node = right
                l = mid+1
            else:
                # left has some >R, go left
                node = left
                r = mid
        return l

    def collect(self):
        res = [0] * self.n
        self._collect(1, 0, self.n-1, res)
        return res

    def _collect(self, node, l, r, res):
        if l == r:
            res[l] = self.minv[node]  # or maxv
        else:
            self._push(node)
            mid = (l+r)//2
            self._collect(node*2, l, mid, res)
            self._collect(node*2+1, mid+1, r, res)

import sys

class SegTree:
    def __init__(self, data):
        self.n = len(data)
        size = 4 * self.n
        self.minv = [0] * size
        self.maxv = [0] * size
        self.lazy = [0] * size
        self._build(1, 0, self.n-1, data)

    def _build(self, node, l, r, data):
        if l == r:
            val = data[l]
            self.minv[node] = val
            self.maxv[node] = val
        else:
            mid = (l+r)//2
            self._build(node*2, l, mid, data)
            self._build(node*2+1, mid+1, r, data)
            self.minv[node] = min(self.minv[node*2], self.minv[node*2+1])
            self.maxv[node] = max(self.maxv[node*2], self.maxv[node*2+1])

    def _push(self, node):
        if self.lazy[node] != 0:
            add = self.lazy[node]
            left = node*2
            right = node*2+1
            self.minv[left] += add
            self.maxv[left] += add
            self.lazy[left] += add
            self.minv[right] += add
            self.maxv[right] += add
            self.lazy[right] += add
            self.lazy[node] = 0

    def range_add(self, ql, qr, add):
        self._range_add(1, 0, self.n-1, ql, qr, add)

    def _range_add(self, node, l, r, ql, qr, add):
        if ql > r or qr < l:
            return
        if ql <= l and r <= qr:
            self.minv[node] += add
            self.maxv[node] += add
            self.lazy[node] += add
            return
        self._push(node)
        mid = (l+r)//2
        self._range_add(node*2, l, mid, ql, qr, add)
        self._range_add(node*2+1, mid+1, r, ql, qr, add)
        self.minv[node] = min(self.minv[node*2], self.minv[node*2+1])
        self.maxv[node] = max(self.maxv[node*2], self.maxv[node*2+1])

    def find_first_ge(self, L):
        if self.maxv[1] < L:
            return self.n
        if self.minv[1] >= L:
            return 0
        node = 1
        l = 0
        r = self.n-1
        while l < r:
            self._push(node)
            mid = (l+r)//2
            left = node*2
            if self.maxv[left] >= L:
                node = left
                r = mid
            else:
                node = node*2+1
                l = mid+1
        return l

    def find_first_greater(self, R):
        if self.minv[1] > R:
            return 0
        if self.maxv[1] <= R:
            return self.n
        node = 1
        l = 0
        r = self.n-1
        while l < r:
            self._push(node)
            mid = (l+r)//2
            left = node*2
            right = node*2+1
            if self.minv[left] > R:
                return l
            elif self.maxv[left] <= R:
                node = right
                l = mid+1
            else:
                node = left
                r = mid
        return l

    def collect(self):
        res = [0] * self.n
        self._collect(1, 0, self.n-1, res)
        return res

    def _collect(self, node, l, r, res):
        if l == r:
            res[l] = self.minv[node]
        else:
            self._push(node)
            mid = (l+r)//2
            self._collect(node*2, l, mid, res)
            self._collect(node*2+1, mid+1, r, res)

def main():
    import sys
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    intervals = [(next(it), next(it)) for _ in range(N)]
    Q = next(it)
    queries = [next(it) for _ in range(Q)]

    # compress X values
    uniq = sorted(set(queries))
    K = len(uniq)

        if l == K:
            continue
        r_plus = seg.find_first_greater(R)
        r = r_plus - 1
        if l > r:
            continue
        seg.range_add(l, r, 1)

    final_vals = seg.collect()

    out = []
    for x in queries:
        i = idx[x]
        out.append(str(final_vals[i]))
    sys.stdout.write("\n".join(out))
