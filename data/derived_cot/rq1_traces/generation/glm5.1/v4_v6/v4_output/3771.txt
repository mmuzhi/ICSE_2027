def maxSubstringLength(self, s: str, k: int) -> bool:
    n = len(s)
    if k == 0:
        return True
    first = [-1]*26
    last = [-1]*26
    # prefix counts
    pref = [[0]*(n+1) for _ in range(26)]
    for i, ch in enumerate(s):
        idx = ord(ch)-97
        if first[idx] == -1:
            first[idx] = i
        last[idx] = i
        # update prefix
        for c in range(26):
            pref[c][i+1] = pref[c][i]
        pref[idx][i+1] += 1
    # build first_set and last_set
    first_set = [first[c] for c in range(26) if first[c] != -1]
    last_set = [last[c] for c in range(26) if last[c] != -1]
    intervals = []
    for l in first_set:
        for r in last_set:
            if l <= r and not (l == 0 and r == n-1):
                valid = True
                for c in range(26):
                    # check if c appears in [l, r]
                    if pref[c][r+1] - pref[c][l] > 0:
                        if first[c] < l or last[c] > r:
                            valid = False
                            break
                if valid:
                    intervals.append((l, r))
    # sort intervals by end
    intervals.sort(key=lambda x: x[1])
    count = 0
    last_end = -1
    for l, r in intervals:
        if l > last_end:
            count += 1
            last_end = r
            if count >= k:  # early exit
                return True
    return count >= k
