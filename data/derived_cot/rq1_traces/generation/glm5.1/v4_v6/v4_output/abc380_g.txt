- First, choose an integer i uniformly at random between 1 and N - K + 1, inclusive.

What is the inversion number?
The inversion number of a sequence (A_1, A_2, \dots, A_N) is the number of integer pairs (i, j) satisfying 1 \le i < j \le N and A_i > A_j.

Under the constraints of this problem, when this value is represented as an irreducible fraction \frac{P}{Q}, it can also be proved that Q \not\equiv 0 \pmod{998244353}. Thus, there is a unique integer R satisfying R \times Q \equiv P \pmod{998244353}, \ 0 \le R < 998244353. Report this integer R.

class BIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+2)
    def update(self, idx, delta):
        while idx <= self.n:
            self.bit[idx] += delta
            idx += idx & -idx
    def query(self, idx):
        res = 0
        while idx > 0:
            res += self.bit[idx]
            idx -= idx & -idx
        return res

import sys

MOD = 998244353

def main():
    import sys
    input = sys.stdin.readline
    N, K = map(int, input().split())
    P = list(map(int, input().split()))
    M = N - K + 1

    # BIT class
    class BIT:
        __slots__ = ('n', 'bit')
        def __init__(self, n):
            self.n = n
            self.bit = [0]*(n+2)
        def update(self, idx, delta):
            while idx <= self.n:
                self.bit[idx] += delta
                idx += idx & -idx
        def query(self, idx):
            res = 0
            while idx > 0:
                res += self.bit[idx]
                idx -= idx & -idx
            return res

    # compute I_total
    bit = BIT(N)
    inv_total = 0
    for i, v in enumerate(P):
        inv_total += i - bit.query(v)
        bit.update(v, 1)

    # compute sum_in
    bit2 = BIT(N)
    inv = 0
    # first window
    for i in range(K):
        v = P[i]
        inv += i - bit2.query(v)
        bit2.update(v, 1)
    sum_in = inv

    # slide
    for L in range(0, M-1):
        v_rem = P[L]
        # subtract inversions where v_rem was left with smaller elements
        inv -= bit2.query(v_rem-1)
        bit2.update(v_rem, -1)
        v_add = P[L+K]
        inv += (K-1 - bit2.query(v_add))
        bit2.update(v_add, 1)
        sum_in += inv

    # compute answer
    term1 = inv_total % MOD
    term2 = (K * (K-1) % MOD) * pow(4, MOD-2, MOD) % MOD
    term3 = (sum_in % MOD) * pow(M, MOD-2, MOD) % MOD
    ans = (term1 + term2 - term3) % MOD
    print(ans)

if __name__ == "__main__":
    main()

import sys

MOD = 998244353

def main():
    input = sys.stdin.readline
    N, K = map(int, input().split())
    P = list(map(int, input().split()))
    M = N - K + 1

    # ---------- Fenwick Tree ----------
    class BIT:
        __slots__ = ('n', 'bit')
        def __init__(self, n):
            self.n = n
            self.bit = [0] * (n + 2)
        def update(self, idx, delta):
            while idx <= self.n:
                self.bit[idx] += delta
                idx += idx & -idx
        def query(self, idx):
            s = 0
            while idx > 0:
                s += self.bit[idx]
                idx -= idx & -idx
            return s

    # ---------- total inversions ----------
    bit_total = BIT(N)
    inv_total = 0
    for i, v in enumerate(P):
        inv_total += i - bit_total.query(v)
        bit_total.update(v, 1)

    # ---------- sum of inversions inside all windows ----------
    bit_win = BIT(N)
    inv = 0
    # first window [0, K-1]
    for i in range(K):
        v = P[i]
        inv += i - bit_win.query(v)
        bit_win.update(v, 1)
    sum_in = inv

    # slide the window
    for L in range(0, M - 1):
        # remove leftmost element P[L]
        v_rem = P[L]
        inv -= bit_win.query(v_rem - 1)
        bit_win.update(v_rem, -1)

        # add new rightmost element P[L+K]
        v_add = P[L + K]
        inv += (K - 1 - bit_win.query(v_add))
        bit_win.update(v_add, 1)

        sum_in += inv

    # ---------- compute answer modulo MOD ----------
    term1 = inv_total % MOD
    inv4 = pow(4, MOD - 2, MOD)
    term2 = (K * (K - 1) % MOD) * inv4 % MOD
    invM = pow(M, MOD - 2, MOD)
    term3 = (sum_in % MOD) * invM % MOD

    ans = (term1 + term2 - term3) % MOD
    print(ans)

if __name__ == "__main__":
    main()
