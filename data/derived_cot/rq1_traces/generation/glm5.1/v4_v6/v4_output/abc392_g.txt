import sys
import numpy as np

def main():
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    n = int(data[0])
    S = list(map(int, data[1:1 + n]))

    M = max(S)
    # length for FFT: power of two >= 2*M+1
    L = 1
    while L < 2 * M + 1:
        L <<= 1

    # frequency array (0/1)
    freq = np.zeros(L, dtype=np.float64)
    freq[S] = 1.0

    # convolution via FFT
    f = np.fft.rfft(freq)
    conv = np.fft.irfft(f * f, n=L)
    conv = np.rint(conv).astype(np.int64)   # exact integer values

    # sum over all B in S: (ordered pairs for sum 2B) - 1 (the (B,B) pair) divided by 2
    S_arr = np.array(S, dtype=np.int64)
    ordered = conv[2 * S_arr]               # ordered pairs (i,j) with i+j = 2B
    ans = int(np.sum((ordered - 1) // 2))

    print(ans)
