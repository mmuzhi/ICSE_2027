def solve():
    import sys
    from math import gcd
    input = sys.stdin.readline

    N, K = map(int, input().split())
    funcs = [tuple(map(int, input().split())) for _ in range(N)]

    # Step 1: get unique pairs
    unique_pairs = set(funcs)

    # Step 2: compute reduced fraction for each unique pair
    reduced_for_pair = {}
    fractions_set = set()
    for a,b in unique_pairs:
        p = a - 1
        q = b
        if p == 0:
            red = (0, 1)
        else:
            g = gcd(p, q)
            red = (p // g, q // g)
        reduced_for_pair[(a,b)] = red
        fractions_set.add(red)

    # Step 3: sort unique reduced fractions
    fractions = list(fractions_set)
    # sort using cross multiplication
    from functools import cmp_to_key
    def cmp_frac(f1, f2):
        p1, q1 = f1
        p2, q2 = f2
        # compare p1/q1 and p2/q2
        lhs = p1 * q2
        rhs = p2 * q1
        if lhs < rhs:
            return -1
        elif lhs > rhs:
            return 1
        else:
            return 0
    fractions.sort(key=cmp_to_key(cmp_frac))

    # assign rank
    rank_of = {frac: i for i, frac in enumerate(fractions)}

    # map each pair to rank
    pair_rank = {}
    for (a,b), red in reduced_for_pair.items():
        pair_rank[(a,b)] = rank_of[red]

    # Step 4: assign rank to each function and sort
    items = [(pair_rank[(a,b)], a, b) for a,b in funcs]
    items.sort(key=lambda x: x[0])  # sort by rank ascending

    A = [a for _, a, b in items]
    B = [b for _, a, b in items]

    # DP
    # t=1
    dp_prev = [A[i] + B[i] for i in range(N)]

    for t in range(2, K+1):
        dp_cur = [-1] * N
        best = -1
        for i in range(N):
            if best != -1:
                dp_cur[i] = A[i] * best + B[i]
            # update best with dp_prev[i]
            if dp_prev[i] > best:
                best = dp_prev[i]
        dp_prev = dp_cur

    ans = max(dp_prev)
    print(ans)

import sys
from math import gcd
from functools import cmp_to_key

def solve():
    import sys
    input = sys.stdin.readline
    N, K = map(int, input().split())
    funcs = [tuple(map(int, input().split())) for _ in range(N)]

    # Compute reduced fraction for each function
    reds = []
    for a, b in funcs:
        p = a - 1
        q = b
        if p == 0:
            red = (0, 1)
        else:
            g = gcd(p, q)
            red = (p // g, q // g)
        reds.append(red)

    # Get unique fractions
    unique_fracs = set(reds)
    fracs = list(unique_fracs)

    # Sort fractions by value (ascending)
    def cmp_frac(f1, f2):
        p1, q1 = f1
        p2, q2 = f2
        # compare p1/q1 and p2/q2
        lhs = p1 * q2
        rhs = p2 * q1
        if lhs < rhs:
            return -1
        elif lhs > rhs:
            return 1
        else:
            return 0
    fracs.sort(key=cmp_to_key(cmp_frac))

    # Assign rank
    rank_of = {frac: i for i, frac in enumerate(fracs)}

    # Assign rank to each function and sort
    items = [(rank_of[reds[i]], funcs[i][0], funcs[i][1]) for i in range(N)]
    items.sort(key=lambda x: x[0])  # sort by rank ascending

    A = [a for _, a, b in items]
    B = [b for _, a, b in items]

    # DP
    # t = 1
    dp_prev = [A[i] + B[i] for i in range(N)]

    for t in range(2, K+1):
        dp_cur = [-1] * N
        best = -1
        for i in range(N):
            if best != -1:
                dp_cur[i] = A[i] * best + B[i]
            # update best with dp_prev[i]
            if dp_prev[i] > best:
                best = dp_prev[i]
        dp_prev = dp_cur

    ans = max(dp_prev)
    print(ans)

if __name__ == "__main__":
    solve()

import sys
from math import gcd
from functools import cmp_to_key

def solve() -> None:
    input = sys.stdin.readline
    N, K = map(int, input().split())
    funcs = [tuple(map(int, input().split())) for _ in range(N)]

    # ---- 1. compute reduced fraction (p_red, q_red) for each function ----
    reds = []                     # (p_red, q_red) for each function
    for a, b in funcs:
        p = a - 1
        q = b
        if p == 0:
            red = (0, 1)
        else:
            g = gcd(p, q)
            red = (p // g, q // g)
        reds.append(red)

    # ---- 2. sort all distinct fractions to obtain ranks ----
    unique_fracs = set(reds)
    fracs = list(unique_fracs)

    def cmp_frac(f1, f2):
        p1, q1 = f1
        p2, q2 = f2
        lhs = p1 * q2
        rhs = p2 * q1
        if lhs < rhs:
            return -1
        if lhs > rhs:
            return 1
        return 0

    fracs.sort(key=cmp_to_key(cmp_frac))
    rank_of = {frac: i for i, frac in enumerate(fracs)}

    # ---- 3. assign rank to each function and sort by rank (ascending) ----
    items = [(rank_of[reds[i]], funcs[i][0], funcs[i][1]) for i in range(N)]
    items.sort(key=lambda x: x[0])          # increasing rank = increasing (A-1)/B

    A = [a for _, a, _ in items]
    B = [b for _, _, b in items]

    # ---- 4. DP over the sorted list ----
    # t = 1
    dp_prev = [A[i] + B[i] for i in range(N)]

    for t in range(2, K + 1):
        dp_cur = [-1] * N
        best = -1                     # max dp_prev[j] for j < i
        for i in range(N):
            if best != -1:
                dp_cur[i] = A[i] * best + B[i]
            # update best with the value for sequences ending at i (previous layer)
            if dp_prev[i] > best:
                best = dp_prev[i]
        dp_prev = dp_cur

    ans = max(dp_prev)
    print(ans)

if __name__ == "__main__":
    solve()
