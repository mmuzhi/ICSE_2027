import math
import bisect

def primes_upto(n):
    if n < 2:
        return []
    sieve = bytearray(b'\x01') * (n+1)
    sieve[0:2] = b'\x00\x00'
    limit = math.isqrt(n)
    for i in range(2, limit+1):
        if sieve[i]:
            start = i*i
            step = i
            sieve[start:n+1:step] = b'\x00' * ((n - start)//step + 1)
    return [i for i in range(2, n+1) if sieve[i]]

def main():
    N = int(input())
    M = math.isqrt(N)
    primes = primes_upto(M)
    
    count1 = 0

        else:

            left = bisect.bisect_right(primes, p)
            right = bisect.bisect_right(primes, limit)
            cnt = right - left
            if cnt > 0:
                count2 += cnt
    
    print(count1 + count2)

if __name__ == "__main__":

import math
import bisect

def primes_upto(n: int):
    """Return list of all primes <= n using a bytearray sieve."""
    if n < 2:
        return []
    sieve = bytearray(b'\x01') * (n + 1)
    sieve[0:2] = b'\x00\x00'          # 0 and 1 are not prime
    limit = math.isqrt(n)
    for i in range(2, limit + 1):
        if sieve[i]:
            start = i * i
            step = i
            # number of elements to set to zero
            cnt = (n - start) // step + 1
            sieve[start:n+1:step] = b'\x00' * cnt
    return [i for i in range(2, n + 1) if sieve[i]]

def main():
    N = int(input())
    M = math.isqrt(N)                 # floor(sqrt(N))
    primes = primes_upto(M)

    # Count numbers of the form p^8 (p prime)
    cnt1 = 0
    for p in primes:
        if p ** 8 <= N:
            cnt1 += 1
        else:
            break                     # primes are increasing

    # Count numbers of the form p^2 * q^2 (p, q distinct primes)
    cnt2 = 0
    if M >= 2:
        sqrtM = math.isqrt(M)
        for p in primes:
            if p > sqrtM:
                break
            limit = M // p            # we need q <= limit
            # number of primes q with p < q <= limit
            left = bisect.bisect_right(primes, p)
            right = bisect.bisect_right(primes, limit)
            cnt2 += right - left

    print(cnt1 + cnt2)

if __name__ == "__main__":
    main()
