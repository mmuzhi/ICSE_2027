You are given three integers n, m, k. A good array arr of size n is defined as follows:

Each element in arr is in the inclusive range [1, m].
Exactly k indices i (where 1 <= i < n) satisfy the condition arr[i - 1] == arr[i].

Return the number of good arrays that can be formed.
Since the answer may be very large, return it modulo 10^9 + 7.
 
Example 1:

Input: n = 3, m = 2, k = 1
Output: 4
Explanation:

Hence, the answer is 4.

class Solution:
    def countGoodArrays(self, n: int, m: int, k: int) -> int:
        MOD = 10**9 + 7

        # According to the derived formula:
        # number = m * (m-1)^(n-k-1) * C(n-1, k)
        if k > n - 1:          # impossible by constraints, but safe
            return 0

        N = n - 1              # we need factorials up to N

        # precompute factorials modulo MOD
        fact = [1] * (N + 1)
        for i in range(1, N + 1):
            fact[i] = fact[i-1] * i % MOD

        # precompute inverse factorials
        inv_fact = [1] * (N + 1)
        inv_fact[N] = pow(fact[N], MOD-2, MOD)
        for i in range(N, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD

        # compute C(N, k) modulo MOD
        comb = fact[N] * inv_fact[k] % MOD * inv_fact[N - k] % MOD

        # compute the whole expression
        ans = m % MOD
        ans = ans * pow(m-1, n - k - 1, MOD) % MOD
        ans = ans * comb % MOD

        return ans
