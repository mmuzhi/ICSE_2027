class Solution:
    def findMinimumTime(self, strength: List[int], K: int) -> int:
        n = len(strength)
        # dp[mask] = minimum time to break locks in mask
        # We also need to track X, but X depends on how many locks we've broken
        
        # Actually, X = 1 + (number of locks broken) * K
        # So if we've broken k locks, X = 1 + k * K
