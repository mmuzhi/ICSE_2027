```python

class Solution:
    def longestPalindrome(self, s: str, t: str) -> int:
        n, m = len(s), len(t)
        max_len = 0

        # helper to test if a string is a palindrome
        def is_pal(x: str) -> bool:
            return x == x[::-1]

        # all substrings of s, including the empty string
        s_subs = [""]
        for i in range(n):
            for j in range(i + 1, n + 1):
                s_subs.append(s[i:j])

        # all substrings of t, including the empty string
        t_subs = [""]
        for i in range(m):
            for j in range(i + 1, m + 1):
                t_subs.append(t[i:j])

        # try every concatenation
        for a in s_subs:
            for b in t_subs:
                cand = a + b
                if is_pal(cand):
                    max_len = max(max_len, len(cand))

        return max_len
```
