class Solution:
    def maxDistance(self, s: str, k: int) -> int:
        def max_abs(vals):
            n = len(vals)
            cnt_minus = vals.count(-1)  # but vals is list of ints; we can compute via sum
            cnt_plus = n - cnt_minus

            # Scenario 1: flip -1 to +1 (earliest)
            x = min(k, cnt_minus)
            cur = 0
            max_pos = 0
            flips = 0
            for v in vals:
                if v == -1 and flips < x:
                    cur += 1
                    flips += 1
                else:
                    cur += v
                if cur > max_pos:
                    max_pos = cur

            # Scenario 2: flip +1 to -1 (earliest)
            y = min(k, cnt_plus)
            cur = 0
            min_neg = 0
            flips = 0
            for v in vals:
                if v == 1 and flips < y:
                    cur += -1
                    flips += 1
                else:
                    cur += v
                if cur < min_neg:
                    min_neg = cur

            return max(max_pos, -min_neg)

        # Build vals for S and D
        vals_S = []
        vals_D = []
        for ch in s:
            if ch in 'NE':
                vals_S.append(1)
            else:
                vals_S.append(-1)
            if ch in 'ES':
                vals_D.append(1)
            else:
                vals_D.append(-1)

        best_S = max_abs(vals_S)
        best_D = max_abs(vals_D)
        return max(best_S, best_D)
