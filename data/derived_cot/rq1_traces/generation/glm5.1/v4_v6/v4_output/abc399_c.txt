What is a forest?

A simple undirected graph F is called a forest if and only if F does not contain any cycle.

Input

The input is given from Standard Input in the following format:
N M
u_1 v_1
u_2 v_2

import sys

def main() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())

    # Union-Find data structure
    parent = list(range(N))
    rank = [0] * N

    def find(x: int) -> int:
        # path compression
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    def union(x: int, y: int) -> None:
        rx, ry = find(x), find(y)
        if rx == ry:
            return
        if rank[rx] < rank[ry]:
            parent[rx] = ry
        elif rank[rx] > rank[ry]:
            parent[ry] = rx
        else:
            parent[ry] = rx
            rank[rx] += 1

    # Process edges
    for _ in range(M):
        u, v = map(int, input().split())
        union(u - 1, v - 1)

    # Count connected components
    roots = set()
    for i in range(N):
        roots.add(find(i))
    C = len(roots)

    # Minimum deletions = M - (N - C)
    ans = M - N + C
    print(ans)

if __name__ == "__main__":
    main()
