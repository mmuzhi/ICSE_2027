```python

import sys
from collections import Counter, defaultdict

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))

    fixed_A = [a for a in A if a != -1]
    fixed_B = [b for b in B if b != -1]
    m_A = len(fixed_A)
    m_B = len(fixed_B)
    T = m_A + m_B - N

    # If we have enough unknown positions, we can always succeed.
    if T <= 0:
        print("Yes")
        return

    # T > 0 : we need at least T fixed–fixed pairs.
    L = 0
    if fixed_A:
        L = max(L, max(fixed_A))
    if fixed_B:
        L = max(L, max(fixed_B))

    cntA = Counter(fixed_A)
    cntB = Counter(fixed_B)

    P = defaultdict(int)          # P[s] = maximum number of fixed–fixed pairs with sum s
    for a, ca in cntA.items():
        for b, cb in cntB.items():
            s = a + b
            if s < L:             # s cannot be the final sum because it would be too small
                continue
            contrib = min(ca, cb)
            P[s] += contrib
            if P[s] >= T:         # early exit as soon as a feasible s is found
                print("Yes")
                return

    # No suitable s found
    print("No")

if __name__ == "__main__":
    solve()
```
