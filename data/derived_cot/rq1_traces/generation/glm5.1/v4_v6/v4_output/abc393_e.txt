def gen_ints():
    data = sys.stdin.buffer.read()
    num = 0
    sign = 1
    in_num = False
    for c in data:

            num = num * 10 + (c - 48)

        else:
            if in_num:
                yield sign * num
                num = 0

def main():
    it = gen_ints()
    N = next(it)
    K = next(it)
    A = [next(it) for _ in range(N)]

    maxA = max(A)
    freq = [0] * (maxA + 1)
    for a in A:
        freq[a] += 1

    cnt = [0] * (maxA + 1)
    # compute cnt
    for d in range(1, maxA + 1):
        # iterate multiples
        for m in range(d, maxA + 1, d):

    out = bytearray()
    for a in A:

    sys.stdout.buffer.write(out)
