        MOD = 10**9+7
        if not nums:

```python
from typing import List

class Solution:
    def sumOfGoodSubsequences(self, nums: List[int]) -> int:
        MOD = 10**9 + 7
        MAX_VAL = 100000
        SIZE = MAX_VAL + 2          # to safely access x+1 when x = MAX_VAL
        
        cnt = [0] * SIZE            # number of good subsequences ending with value v
        sm = [0] * SIZE             # total sum of elements of those subsequences
        
        for x in nums:
            new_cnt = 1
            new_sm = x
            
            # extend from subsequences ending with x-1
            if x > 0:
                new_cnt = (new_cnt + cnt[x-1]) % MOD
                new_sm = (new_sm + sm[x-1] + x * cnt[x-1]) % MOD
            
            # extend from subsequences ending with x+1
            if x + 1 < SIZE:
                new_cnt = (new_cnt + cnt[x+1]) % MOD
                new_sm = (new_sm + sm[x+1] + x * cnt[x+1]) % MOD
            
            # add the new subsequences that end at current position
            cnt[x] = (cnt[x] + new_cnt) % MOD
            sm[x] = (sm[x] + new_sm) % MOD
        
        total = sum(sm) % MOD
        return total
```
