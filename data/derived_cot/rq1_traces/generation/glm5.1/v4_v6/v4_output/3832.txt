```python
from typing import List

class Solution:
    def minCosts(self, cost: List[int]) -> List[int]:
        """
        Returns the minimum total cost to reach each position i (0 <= i < n)
        starting from position n, where swapping with a person in front costs
        their given cost and swapping with someone behind is free.
        The optimal cost for position i is the minimum cost among all persons
        with indices 0..i, because we can swap directly with the cheapest
        person in that range and then move right for free if needed.
        """
        answer = []
        current_min = float('inf')
        for c in cost:
            current_min = min(current_min, c)
            answer.append(current_min)
        return answer
```
