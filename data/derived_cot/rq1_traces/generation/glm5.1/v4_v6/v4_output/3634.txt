```python

class Solution:
    def calculateScore(self, s: str) -> int:
        # stacks[c] stores unmarked indices with character c (in increasing order)
        stacks = [[] for _ in range(26)]
        score = 0

        for i, ch in enumerate(s):
            idx = ord(ch) - ord('a')          # 0 .. 25
            mir = 25 - idx                     # mirror character index

            if stacks[mir]:                    # there is a matching unmarked j
                j = stacks[mir].pop()          # the closest (largest) such j
                score += i - j
                # i becomes marked, so it is not pushed onto any stack
            else:
                # no matching j, i stays unmarked
                stacks[idx].append(i)

        return score
```
