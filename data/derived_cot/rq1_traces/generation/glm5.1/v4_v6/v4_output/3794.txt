from typing import List

class Solution:
    def minTime(self, skill: List[int], mana: List[int]) -> int:
        n = len(skill)
        m = len(mana)
        sum_skill = sum(skill)
        t = 0  # start time of current potion (first potion starts at 0)
        # iterate over potions from the second to the last
        for j in range(1, m):
            mp = mana[j-1]
            mc = mana[j]
            sum_prev = 0  # cumulative sum of skill[i]*mp up to current i
            sum_curr = 0  # cumulative sum of skill[i]*mc up to previous i
            max_delta = 0
            for s in skill:
                sum_prev += s * mp
                delta = sum_prev - sum_curr
                if delta > max_delta:
                    max_delta = delta
                sum_curr += s * mc
            t += max_delta
        # total time = start time of last potion + its total processing time
        return t + mana[-1] * sum_skill

from typing import List

class Solution:
    def minTime(self, skill: List[int], mana: List[int]) -> int:
        n = len(skill)
        m = len(mana)
        sum_skill = sum(skill)          # sum of all skills
        t = 0                           # start time of the current potion (first starts at 0)

        # Process potions from the second to the last
        for j in range(1, m):
            mp = mana[j-1]              # mana of previous potion
            mc = mana[j]                # mana of current potion
            sum_prev = 0                # Σ skill[i] * mp  for i up to current
            sum_curr = 0                # Σ skill[i] * mc  for i up to previous
            max_delta = 0

            # Compute the maximal delay needed for the current potion
            for s in skill:
                sum_prev += s * mp
                delta = sum_prev - sum_curr
                if delta > max_delta:
                    max_delta = delta
                sum_curr += s * mc

            t += max_delta               # earliest start time for potion j

        # Total time = start of last potion + its total processing time
        return t + mana[-1] * sum_skill
