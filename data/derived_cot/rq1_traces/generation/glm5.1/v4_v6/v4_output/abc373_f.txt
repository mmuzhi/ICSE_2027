    input = sys.stdin.readline
    N, W = map(int, input().split())
    items = [tuple(map(int, input().split())) for _ in range(N)]
    INF = 10**18
    NEG_INF = -10**18  # sufficiently small
    dp = [NEG_INF] * (W+1)
    dp[0] = 0

    for w, v in items:
        K = min(v // 2, W // w)   # max number of items to consider
        new_dp = [NEG_INF] * (W+1)
        # Process each residue
        for r in range(w):
            # Build list of A for this residue
            # t from 0 while r + t*w <= W
            t_max = (W - r) // w
            # We'll iterate t from 0 to t_max
            # We'll maintain deque for this residue
            from collections import deque
            dq = deque()
            for t in range(t_max + 1):
                weight = r + t * w
                # Add line for i = t if dp[weight] != NEG_INF
                val = dp[weight]
                if val > NEG_INF:
                    m = 2 * t
                    b = val - t * v - t * t
                    # Insert line (m, b, i=t)
                    # Remove from back while new line makes last line redundant
                    while len(dq) >= 2:
                        m1, b1, i1 = dq[-2]
                        m2, b2, i2 = dq[-1]
                        m3, b3, i3 = m, b, t
                        # Check if line (m2,b2) is unnecessary
                        # condition: (b1 - b2)*(m3 - m2) >= (b2 - b3)*(m2 - m1)
                        if (b1 - b2) * (m3 - m2) >= (b2 - b3) * (m2 - m1):
                            dq.pop()
                        else:
                            break
                    dq.append((m, b, t))
                # Remove expired lines: i < t - K
                while dq and dq[0][2] < t - K:
                    dq.popleft()
                # If deque empty, new_dp[weight] remains NEG_INF
                if not dq:
                    continue
                # Ensure front is optimal for current t
                while len(dq) >= 2:
                    m1, b1, i1 = dq[0]
                    m2, b2, i2 = dq[1]
                    if m1 * t + b1 <= m2 * t + b2:
                        dq.popleft()
                    else:
                        break
                # Now front is best
                m, b, i = dq[0]
                M = m * t + b
                cand = (t * v - t * t) + M
                if cand > new_dp[weight]:
                    new_dp[weight] = cand
        dp = new_dp

    ans = max(dp)

For each (w_i, v_i):
    K = min(v_i // 2, W // w_i)
    new_dp = [NEG_INF] * (W+1)
    for r in range(w_i):
        dq = deque()
        max_t = (W - r) // w_i
        for t in range(max_t + 1):
            weight = r + t * w_i
            val = dp[weight]
            if val > NEG_INF:
                m = 2 * t
                b = val - t * v_i - t * t
                # Insert line
                while len(dq) >= 2:
                    m1, b1, i1 = dq[-2]
                    m2, b2, i2 = dq[-1]
                    m3, b3, i3 = m, b, t
                    # Check if line (m2,b2) is unnecessary
                    if (b1 - b2) * (m3 - m2) >= (b2 - b3) * (m2 - m1):
                        dq.pop()
                    else:
                        break
                dq.append((m, b, t))
            # Remove expired lines
            while dq and dq[0][2] < t - K:
                dq.popleft()
            if not dq:
                continue
            # Ensure front is optimal
            while len(dq) >= 2:
                m1, b1, i1 = dq[0]
                m2, b2, i2 = dq[1]
                if m1 * t + b1 <= m2 * t + b2:
                    dq.popleft()
                else:
                    break
            m, b, i = dq[0]
            M = m * t + b
            cand = (t * v_i - t * t) + M
            if cand > new_dp[weight]:
                new_dp[weight] = cand
    dp = new_dp

ans = max(dp)

import sys
from collections import deque

def solve():
    input = sys.stdin.readline
    N, W = map(int, input().split())
    items = [tuple(map(int, input().split())) for _ in range(N)]
    NEG_INF = -10**18
    dp = [NEG_INF] * (W+1)
    dp[0] = 0

    for w, val in items:
        K = min(val // 2, W // w)
        new_dp = [NEG_INF] * (W+1)
        for r in range(w):
            dq = deque()
            max_t = (W - r) // w
            for t in range(max_t + 1):
                weight = r + t * w
                cur = dp[weight]
                if cur > NEG_INF:
                    m = 2 * t
                    b = cur - t * val - t * t
                    # Insert line
                    while len(dq) >= 2:
                        m1, b1, i1 = dq[-2]
                        m2, b2, i2 = dq[-1]
                        m3, b3, i3 = m, b, t
                        # Check if line (m2,b2) is unnecessary
                        if (b1 - b2) * (m3 - m2) >= (b2 - b3) * (m2 - m1):
                            dq.pop()
                        else:
                            break
                    dq.append((m, b, t))
                # Remove expired lines
                while dq and dq[0][2] < t - K:
                    dq.popleft()
                if not dq:
                    continue
                # Ensure front is optimal for current t
                while len(dq) >= 2:
                    m1, b1, i1 = dq[0]
                    m2, b2, i2 = dq[1]
                    if m1 * t + b1 <= m2 * t + b2:
                        dq.popleft()
                    else:
                        break
                m, b, i = dq[0]
                M = m * t + b
                cand = (t * val - t * t) + M
                if cand > new_dp[weight]:
                    new_dp[weight] = cand
        dp = new_dp

    ans = max(dp)
    print(ans)

if __name__ == "__main__":
    solve()
