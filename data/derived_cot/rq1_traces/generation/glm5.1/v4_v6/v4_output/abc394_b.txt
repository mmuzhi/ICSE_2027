```python

import sys

def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    N = int(data[0])
    strings = [line.strip() for line in data[1:1+N]]
    strings.sort(key=len)          # sort by length in ascending order
    print(''.join(strings))        # concatenate and output

if __name__ == "__main__":
    main()
```
