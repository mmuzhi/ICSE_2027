```python

import sys

def main() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    # For N = 2 the loop does nothing and we print Yes.
    for i in range(N - 2):
        if A[i] * A[i + 2] != A[i + 1] * A[i + 1]:
            print("No")
            return
    print("Yes")

if __name__ == "__main__":
    main()
```
