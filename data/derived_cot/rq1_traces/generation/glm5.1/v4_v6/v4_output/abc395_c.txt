def solve():
    import sys
    input = sys.stdin.readline
    N = int(input().strip())
    A = list(map(int, input().split()))
    last = [-1] * (10**6 + 1)  # or use dict
    ans = N + 1  # large
    for i, x in enumerate(A):
        if last[x] != -1:
            length = i - last[x] + 1
            if length < ans:
                ans = length
        last[x] = i
    if ans == N + 1:
        print(-1)
    else:
        print(ans)

```python

import sys

def main():
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    last = {}          # last occurrence index of each value
    ans = N + 1        # larger than any possible answer

    for i, x in enumerate(A):
        if x in last:
            length = i - last[x] + 1
            if length < ans:
                ans = length
        last[x] = i

    if ans == N + 1:
        print(-1)
    else:
        print(ans)

if __name__ == "__main__":
    main()
```
