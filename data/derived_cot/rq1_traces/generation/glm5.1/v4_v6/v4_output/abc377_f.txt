import sys

def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    rows_set = set()
    cols_set = set()
    sums_set = set()
    diffs_set = set()
    for _ in range(M):
        a, b = map(int, input().split())
        rows_set.add(a)
        cols_set.add(b)
        sums_set.add(a + b)
        diffs_set.add(a - b)
    rows = list(rows_set)
    cols = list(cols_set)
    sums = list(sums_set)
    diffs = list(diffs_set)
    r = len(rows)
    c = len(cols)
    s = len(sums)
    d = len(diffs)

    # S1
    rows_total = r * N
    cols_total = c * N
    sum_diag_total = 0
    for s_val in sums:
        if s_val <= N + 1:
            sum_diag_total += s_val - 1
        else:
            sum_diag_total += 2 * N - s_val + 1
    diff_diag_total = 0
    for d_val in diffs:
        diff_diag_total += N - abs(d_val)
    S1 = rows_total + cols_total + sum_diag_total + diff_diag_total

    # S2 components
    row_col = r * c

    row_sum = 0
    T_rsd = 0  # we'll compute together
    for i in rows:
        for s_val in sums:
            j = s_val - i
            if 1 <= j <= N:
                row_sum += 1
                d_val = i - j
                if d_val in diffs_set:
                    T_rsd += 1

    row_diff = 0
    for i in rows:
        for d_val in diffs:
            j = i - d_val
            if 1 <= j <= N:
                row_diff += 1

    col_sum = 0
    T_csd = 0
    for j in cols:
        for s_val in sums:
            i = s_val - j
            if 1 <= i <= N:
                col_sum += 1
                d_val = i - j
                if d_val in diffs_set:
                    T_csd += 1

    col_diff = 0
    for j in cols:
        for d_val in diffs:
            i = j + d_val
            if 1 <= i <= N:
                col_diff += 1

    sum_diff = 0
    for s_val in sums:
        for d_val in diffs:
            if (s_val + d_val) % 2 == 0:
                i = (s_val + d_val) // 2
                j = (s_val - d_val) // 2
                if 1 <= i <= N and 1 <= j <= N:
                    sum_diff += 1

    S2 = row_col + row_sum + row_diff + col_sum + col_diff + sum_diff

    # S3 and S4 from rows and cols
    T_rcs = 0
    T_rcd = 0
    S4 = 0
    for i in rows:
        for j in cols:
            s_val = i + j
            d_val = i - j
            if s_val in sums_set:
                T_rcs += 1
            if d_val in diffs_set:
                T_rcd += 1
            if s_val in sums_set and d_val in diffs_set:
                S4 += 1

    S3 = T_rcs + T_rcd + T_rsd + T_csd

    attacked = S1 - S2 + S3 - S4
    safe = N * N - attacked
    print(safe)
