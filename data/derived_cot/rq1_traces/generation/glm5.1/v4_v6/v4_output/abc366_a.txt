```python
def main():

    data = sys.stdin.read().strip().split()
    if not data:

    N, T, A = map(int, data)
    remaining = N - T - A
    if T > A + remaining or A > T + remaining:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
```

=== ANSWER ===
```python

import sys

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N, T, A = map(int, data)
    remaining = N - T - A
    if T > A + remaining or A > T + remaining:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    main()
```
