class Solution:
    def longestPalindromicSubsequence(self, s: str, k: int) -> int:
        n = len(s)
        # precompute cyclic distance between every pair of characters
        dist = [[0] * n for _ in range(n)]
        for i in range(n):
            for j in range(i + 1, n):
                diff = abs(ord(s[i]) - ord(s[j]))
                d = min(diff, 26 - diff)
                dist[i][j] = d

        # dp[i][j] will be a bytearray of length k+1
        # dp[i][j][c] = longest palindromic subsequence in s[i..j] with cost ≤ c
        zero_arr = bytearray(k + 1)          # all zeros, for empty inner part
        dp = [[None] * n for _ in range(n)]

        # base: single character -> length 1 with cost 0
        for i in range(n):
            dp[i][i] = bytearray([1]) * (k + 1)

        # intervals of increasing length
        for length in range(2, n + 1):
            for i in range(0, n - length + 1):
                j = i + length - 1

                # start with the best of skipping one end
                best = bytearray(k + 1)          # all zeros initially
                left = dp[i + 1][j]
                right = dp[i][j - 1]
                for c in range(k + 1):
                    v = left[c]
                    rv = right[c]
                    if rv > v:
                        v = rv
                    best[c] = v

                # try to pair s[i] and s[j]
                d = dist[i][j]
                if d <= k:
                    inner = dp[i + 1][j - 1] if i + 1 <= j - 1 else zero_arr
                    for c in range(d, k + 1):
                        cand = inner[c - d] + 2
                        if cand > best[c]:
                            best[c] = cand

                dp[i][j] = best

        return dp[0][n - 1][k]
