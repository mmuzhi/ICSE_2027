```python

import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1].strip()

    # positions of '1' (0‑based)
    ones = [i for i, ch in enumerate(S) if ch == '1']
    K = len(ones)                     # K >= 1 by problem statement

    # q[i] = position of i-th '1' minus i
    q = [ones[i] - i for i in range(K)]

    # median of q (any median works; here we take q[K//2])
    median = q[K // 2]

    # minimal total swaps = sum |q[i] - median|
    ans = sum(abs(v - median) for v in q)

    print(ans)

if __name__ == "__main__":
    solve()
```
