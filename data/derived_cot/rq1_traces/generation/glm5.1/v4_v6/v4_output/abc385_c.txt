Pseudo-code:

def solve():
    import sys
    from array import array
    input = sys.stdin.readline
    N = int(input().strip())
    H = list(map(int, input().split()))
    # group by height
    max_h = 3000
    pos_by_h = [[] for _ in range(max_h+1)]
    for idx, h in enumerate(H, start=1):
        pos_by_h[h].append(idx)
    ans = 1
    col_size = N + 1  # for diff indexing
    for h in range(1, max_h+1):
        pos = pos_by_h[h]
        m = len(pos)
        if m <= 2:
            if m > ans:
                ans = m
            continue
        # DP
        dp = array('H', [0]) * (m * col_size)
        max_len = 1
        # We'll use local variables for speed
        pos_local = pos
        dp_local = dp
        col = col_size
        for i in range(m):
            pi = pos_local[i]
            base_i = i * col
            for j in range(i+1, m):
                diff = pos_local[j] - pi
                # get dp[i][diff]
                val_i = dp_local[base_i + diff]
                if val_i:
                    candidate = val_i + 1
                else:
                    candidate = 2
                idx_j = j * col + diff
                if candidate > dp_local[idx_j]:
                    dp_local[idx_j] = candidate
                    if candidate > max_len:
                        max_len = candidate
        if max_len > ans:
            ans = max_len
    print(ans)

import sys
from array import array

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    H = [int(next(it)) for _ in range(N)]
    if N == 1:
        print(1)
        return
    col = N  # number of columns for diff
    dp = array('H', [0]) * (N * col)
    ans = 1
    for i in range(N):
        hi = H[i]
        base_i = i * col
        for j in range(i+1, N):
            if H[j] != hi:
                continue
            d = j - i
            # get length ending at i with diff d
            val_i = dp[base_i + d]
            if val_i:
                cand = val_i + 1
            else:
                cand = 2
            idx_j = j * col + d
            if cand > dp[idx_j]:
                dp[idx_j] = cand
                if cand > ans:
                    ans = cand
                    if ans == N:  # early exit possible
                        print(N)
                        return
    print(ans)

if __name__ == "__main__":
    solve()
