What is the inversion number?
The inversion number of a sequence (A_1, A_2, \dots, A_N) is the number of integer pairs (i, j) satisfying 1 \le i < j \le N and A_i > A_j.

class BIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+1)
    def add(self, idx, val):
        while idx <= self.n:
            self.bit[idx] += val
            idx += idx & -idx
    def sum(self, idx):
        res = 0
        while idx > 0:
            res += self.bit[idx]
            idx -= idx & -idx
        return res

def solve():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # compute inv0 using BIT
    bit = [0]*(M+1)  # 1-indexed, size M
    inv0 = 0
    for i, a in enumerate(A, start=1):
        idx = a + 1
        # query sum <= a
        s = 0
        x = idx
        while x:
            s += bit[x]
            x -= x & -x
        inv0 += (i-1) - s
        # add
        x = idx
        while x <= M:
            bit[x] += 1
            x += x & -x

    # groups by value
    cnt = [0]*M
    sum_idx = [0]*M
    for i, a in enumerate(A, start=1):
        cnt[a] += 1
        sum_idx[a] += i

    ans = [0]*M
    ans[0] = inv0
    inv = inv0
    Np1 = N + 1
    for k in range(0, M-1):
        v = M - 1 - k
        t = cnt[v]
        s = sum_idx[v]
        delta = 2*s - t * Np1
        inv += delta
        ans[k+1] = inv

    # output
    out = '\n'.join(map(str, ans))
    sys.stdout.write(out)

if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # ---------- inversion number for k = 0 ----------
    bit = [0] * (M + 1)          # Fenwick tree, 1‑indexed
    inv0 = 0
    cnt = [0] * M                # number of indices with value v
    sum_idx = [0] * M            # sum of indices (1‑based) with value v

    for i, a in enumerate(A, start=1):
        idx = a + 1
        # query: how many previous values <= a ?
        s = 0
        x = idx
        while x:
            s += bit[x]
            x -= x & -x
        inv0 += (i - 1) - s      # previous values > a

        # add current value to BIT
        x = idx
        while x <= M:
            bit[x] += 1
            x += x & -x

        # groups for later deltas
        cnt[a] += 1
        sum_idx[a] += i

    # ---------- compute answers for all k ----------
    ans = [0] * M
    ans[0] = inv0
    inv = inv0
    Np1 = N + 1

    for k in range(0, M - 1):
        v = M - 1 - k            # value whose indices wrap at step k
        t = cnt[v]
        s = sum_idx[v]
        delta = 2 * s - t * Np1
        inv += delta
        ans[k + 1] = inv

    # ---------- output ----------
    sys.stdout.write('\n'.join(map(str, ans)))

if __name__ == "__main__":
