Takahashi can repeat the following operation any number of times, possibly zero.

- The 3rd piece from the left moves from coordinate 7 to 5.5 - (7 - 5.5) = 4.

After this operation, the sum of the coordinates of the four pieces is 1 + 4 + 6 + 10 = 21, which is minimal. Thus, print 21.

import sys

def solve() -> None:
    it = iter(sys.stdin.read().split())
    N = int(next(it))
    X = [int(next(it)) for _ in range(N)]

    # gaps between consecutive pieces
    gaps = [X[i+1] - X[i] for i in range(N-1)]

    # separate gaps by parity of their index (1-indexed)
    odd_gaps = []   # indices 1,3,5,...  -> in 0-based: i even
    even_gaps = []  # indices 2,4,6,... -> in 0-based: i odd
    odd_weights = []
    even_weights = []

    for i in range(N-1):
        w = N - i - 1          # weight = N - (index of gap)
        if i % 2 == 0:        # gap index is odd
            odd_gaps.append(gaps[i])
            odd_weights.append(w)
        else:                  # gap index is even
            even_gaps.append(gaps[i])
            even_weights.append(w)

    # sort gaps increasingly, weights are already decreasing
    odd_gaps.sort()
    even_gaps.sort()

    # optimal contributions
    odd_contrib = sum(ow * og for ow, og in zip(odd_weights, odd_gaps))
    even_contrib = sum(ew * eg for ew, eg in zip(even_weights, even_gaps))

    total = N * X[0] + odd_contrib + even_contrib
    print(total)

if __name__ == "__main__":
    solve()
