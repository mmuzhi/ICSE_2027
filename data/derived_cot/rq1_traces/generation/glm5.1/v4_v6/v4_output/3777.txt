The alternating sum of a 0-indexed array is defined as the sum of the elements at even indices minus the sum of the elements at odd indices.
 
Example 1:

Input: nums = [1,2,3], k = 2, limit = 10
Output: 6
Explanation:
The subsequences with an alternating sum of 2 are:

[1, 2, 3]

Alternating Sum: 1 - 2 + 3 = 2
Product: 1 * 2 * 3 = 6


[2]

Alternating Sum: 2
Product: 2



The maximum product within the limit is 6.

Example 2:

Input: nums = [0,2,3], k = -5, limit = 12
Output: -1
Explanation:
A subsequence with an alternating sum of exactly -5 does not exist.

Example 3:

Input: nums = [2,2,3,3], k = 0, limit = 9
Output: 9
Explanation:
The subsequences with an alternating sum of 0 are:

[2, 2]

Alternating Sum: 2 - 2 = 0
Product: 2 * 2 = 4


[3, 3]

Alternating Sum: 3 - 3 = 0
Product: 3 * 3 = 9

Alternating Sum: 2 - 2 + 3 - 3 = 0
Product: 2 * 2 * 3 * 3 = 36



The subsequence [2, 2, 3, 3] has the greatest product with an alternating sum equal to k, but 36 > 9. The next greatest product is 9, which is within the limit.

def compute_states(arr, limit):
    states = [dict(), dict()]  # parity 0 and 1
    for x in arr:
        new_items = []
        # extend existing states
        for parity in (0,1):
            for s, prods in states[parity].items():
                # we can iterate over list(prods) to avoid modification during iteration? We're not modifying prods now.
                for p in prods:
                    new_parity = 1 - parity
                    if parity == 0:
                        new_s = s + x
                    else:
                        new_s = s - x
                    new_p = p * x
                    if new_p > limit:
                        continue
                    new_items.append((new_parity, new_s, new_p))
        # start new subsequence with x
        if x <= limit:
            new_items.append((1, x, x))
        # add new_items to states
        for par, s, p in new_items:
            # use setdefault
            if s in states[par]:
                states[par][s].add(p)
            else:

class Solution:
    def maxProduct(self, nums: List[int], k: int, limit: int) -> int:
        n = len(nums)
        # early check: if |k| > sum(nums), impossible
        total_sum = sum(nums)
        if abs(k) > total_sum:
            return -1
        
        mid = n // 2
        left_arr = nums[:mid]
        right_arr = nums[mid:]
        
        left_states = self.compute_states(left_arr, limit)
        right_states = self.compute_states(right_arr, limit)
        
        # Build right_map: s -> sorted list of products
        right_map = {}
        for parity in (0,1):
            for s, prods in right_states[parity].items():
                if s not in right_map:
                    right_map[s] = set()
                right_map[s].update(prods)
        for s in right_map:
            right_map[s] = sorted(right_map[s])
        
        ans = -1
        
        # left-only
        for parity in (0,1):
            if k in left_states[parity]:
                for p in left_states[parity][k]:
                    if p > ans:
                        ans = p
        
        # right-only
        if k in right_map:
            for p in right_map[k]:
                if p > ans:
                    ans = p
        
        # combination
        from bisect import bisect_right
        for parity in (0,1):
            for s_left, prods_left in left_states[parity].items():
                if parity == 0:
                    req_s_right = k - s_left
                else:
                    req_s_right = s_left - k
                if req_s_right not in right_map:
                    continue
                right_sorted = right_map[req_s_right]
                for p_left in prods_left:
                    if p_left == 0:
                        # product 0 combined with any right gives 0
                        if 0 > ans:
                            ans = 0
                        continue
                    max_allowed = limit // p_left
                    idx = bisect_right(right_sorted, max_allowed) - 1
                    if idx >= 0:
                        candidate = p_left * right_sorted[idx]
                        if candidate > ans:
                            ans = candidate
        
        return ans

        n = len(nums)
        total_sum = sum(nums)
        if abs(k) > total_sum:
            return -1
        
        mid = n // 2
        left_arr = nums[:mid]
        right_arr = nums[mid:]
        
        left_dp0, left_dp1, left_offset, left_total = self.compute_states(left_arr, limit)
        right_dp0, right_dp1, right_offset, right_total = self.compute_states(right_arr, limit)
        
        # Build right_map
        right_map = {}
        # helper to add products from a bitset
        def add_from_bitset(s_val, bitset):
            if bitset == 0:
                return
            if s_val not in right_map:
                right_map[s_val] = set()
            temp = bitset
            while temp:
                lowbit = temp & -temp
                p = lowbit.bit_length() - 1
                right_map[s_val].add(p)
                temp ^= lowbit
        
        for s_idx, bitset in enumerate(right_dp0):

def compute_states(self, arr, limit):
    total = sum(arr)
    offset = total
    s_range = 2*total + 1
    dp0 = [0] * s_range
    dp1 = [0] * s_range
    for x in arr:
        new_dp0 = dp0[:]
        new_dp1 = dp1[:]
        # extend from dp0 (parity 0 -> new parity 1)
        for s_idx, bitset in enumerate(dp0):
            if bitset == 0:
                continue
            s_val = s_idx - offset
            new_s_val = s_val + x
            if new_s_val < -total or new_s_val > total:
                continue
            new_s_idx = new_s_val + offset
            # compute new bitset
            new_bitset = 0
            temp = bitset
            while temp:
                lowbit = temp & -temp
                p = lowbit.bit_length() - 1
                new_p = p * x
                if new_p <= limit:
                    new_bitset |= (1 << new_p)
                temp ^= lowbit
            new_dp1[new_s_idx] |= new_bitset
        # extend from dp1 (parity 1 -> new parity 0)
        for s_idx, bitset in enumerate(dp1):
            if bitset == 0:
                continue
            s_val = s_idx - offset
            new_s_val = s_val - x
            if new_s_val < -total or new_s_val > total:
                continue
            new_s_idx = new_s_val + offset
            new_bitset = 0
            temp = bitset
            while temp:
                lowbit = temp & -temp
                p = lowbit.bit_length() - 1
                new_p = p * x
                if new_p <= limit:
                    new_bitset |= (1 << new_p)
                temp ^= lowbit
            new_dp0[new_s_idx] |= new_bitset
        # start new subsequence with x
        if x <= limit:
            s_idx = x + offset
            if 0 <= s_idx < s_range:
                new_dp1[s_idx] |= (1 << x)
        dp0, dp1 = new_dp0, new_dp1
    return dp0, dp1, offset, total

class Solution:
    def maxProduct(self, nums: List[int], k: int, limit: int) -> int:
        n = len(nums)
        total_sum = sum(nums)
        if abs(k) > total_sum:
            return -1
        
        mid = n // 2
        left_arr = nums[:mid]
        right_arr = nums[mid:]
        
        left_dp0, left_dp1, left_offset, left_total = self._compute_states(left_arr, limit)
        right_dp0, right_dp1, right_offset, right_total = self._compute_states(right_arr, limit)
        
        # Build right_map
        right_map = {}
        def add_from_bitset(s_val, bitset):
            if bitset == 0:
                return
            if s_val not in right_map:
                right_map[s_val] = set()
            temp = bitset
            while temp:
                lowbit = temp & -temp
                p = lowbit.bit_length() - 1
                right_map[s_val].add(p)
                temp ^= lowbit
        
        for s_idx, bitset in enumerate(right_dp0):
            if bitset:
                s_val = s_idx - right_offset
                add_from_bitset(s_val, bitset)
        for s_idx, bitset in enumerate(right_dp1):
            if bitset:
                s_val = s_idx - right_offset
                add_from_bitset(s_val, bitset)
        
        for s_val in right_map:
            right_map[s_val] = sorted(right_map[s_val])
        
        # left-only max
        def max_product_for_s(dp, offset, total, s):
            if s < -total or s > total:
                return -1
            idx = s + offset
            bitset = dp[idx]
            if bitset == 0:
                return -1
            return bitset.bit_length() - 1
        
        left_only = max(max_product_for_s(left_dp0, left_offset, left_total, k),
                        max_product_for_s(left_dp1, left_offset, left_total, k))
        right_only = -1
        if k in right_map:
            right_only = right_map[k][-1]
        ans = max(left_only, right_only)
        
        # combination
        from bisect import bisect_right
        
        def process_left(dp, parity):
            nonlocal ans
            for s_idx, bitset in enumerate(dp):
                if bitset == 0:
                    continue
                s_val = s_idx - left_offset
                if parity == 0:
                    req_s_right = k - s_val
                else:
                    req_s_right = s_val - k
                if req_s_right not in right_map:
                    continue
                right_sorted = right_map[req_s_right]
                temp = bitset
                while temp:
                    lowbit = temp & -temp
                    p_left = lowbit.bit_length() - 1
                    temp ^= lowbit
                    if p_left == 0:
                        if 0 > ans:
                            ans = 0
                        continue
                    max_allowed = limit // p_left
                    idx = bisect_right(right_sorted, max_allowed) - 1
                    if idx >= 0:
                        candidate = p_left * right_sorted[idx]
                        if candidate > ans:
                            ans = candidate
        
        process_left(left_dp0, 0)
        process_left(left_dp1, 1)
        
        return ans
    
    def _compute_states(self, arr, limit):
        total = sum(arr)
        offset = total
        s_range = 2 * total + 1
        dp0 = [0] * s_range
        dp1 = [0] * s_range
        for x in arr:
            new_dp0 = dp0[:]
            new_dp1 = dp1[:]
            # extend from dp0 (parity 0 -> new parity 1)
            for s_idx, bitset in enumerate(dp0):
                if bitset == 0:
                    continue
                s_val = s_idx - offset
                new_s_val = s_val + x
                if new_s_val < -total or new_s_val > total:
                    continue
                new_s_idx = new_s_val + offset
                new_bitset = 0
                temp = bitset
                while temp:
                    lowbit = temp & -temp
                    p = lowbit.bit_length() - 1
                    new_p = p * x
                    if new_p <= limit:
                        new_bitset |= (1 << new_p)
                    temp ^= lowbit
                new_dp1[new_s_idx] |= new_bitset
            # extend from dp1 (parity 1 -> new parity 0)
            for s_idx, bitset in enumerate(dp1):
                if bitset == 0:
                    continue
                s_val = s_idx - offset
                new_s_val = s_val - x
                if new_s_val < -total or new_s_val > total:
                    continue
                new_s_idx = new_s_val + offset
                new_bitset = 0
                temp = bitset
                while temp:
                    lowbit = temp & -temp
                    p = lowbit.bit_length() - 1
                    new_p = p * x
                    if new_p <= limit:
                        new_bitset |= (1 << new_p)
                    temp ^= lowbit
                new_dp0[new_s_idx] |= new_bitset
            # start new subsequence with x
            if x <= limit:
                s_idx = x + offset
                if 0 <= s_idx < s_range:
                    new_dp1[s_idx] |= (1 << x)
            dp0, dp1 = new_dp0, new_dp1
        return dp0, dp1, offset, total
