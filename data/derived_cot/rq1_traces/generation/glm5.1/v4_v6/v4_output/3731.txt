```python
from typing import List

class Solution:
    def subarraySum(self, nums: List[int]) -> int:
        n = len(nums)
        # prefix sum: pref[i] = sum of nums[0..i-1]
        pref = [0] * (n + 1)
        for i in range(n):
            pref[i + 1] = pref[i] + nums[i]
        
        total = 0
        for i in range(n):
            start = max(0, i - nums[i])
            total += pref[i + 1] - pref[start]
        return total
```
