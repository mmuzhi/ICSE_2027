    if mod == 1:

import sys

def min_attacks(H, mod):
    if mod == 1:
        R = [0,1,2]
    elif mod == 2:
        R = [0,1,4]
    else: # mod == 0
        R = [0,3,4]
    best = 10**30
    for r in range(3):
        if H <= R[r]:
            q = 0
        else:
            q = (H - R[r] + 4) // 5
        m = 3 * q + r
        if m < best:
            best = m
    return best

def solve():
    import sys
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    H = list(map(int, data[1:1+N]))
    s = 1
    for h in H:
        mod = s % 3
        k = min_attacks(h, mod)
        s += k
    print(s - 1)

if __name__ == "__main__":
    solve()
