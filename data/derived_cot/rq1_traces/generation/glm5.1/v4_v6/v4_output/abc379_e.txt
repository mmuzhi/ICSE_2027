```python
import sys

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1].strip()
    # Compute cumulative weighted sum and add to digit array
    L = N + 20  # enough
    res = [0] * L
    P = 0
    for i in range(1, N+1):
        digit = int(S[i-1])
        P += digit * i
        offset = N - i
        p = P
        pos = offset
        while p:
            res[pos] += p % 10
            p //= 10
            pos += 1
    # Propagate carries
    for i in range(L-1):
        if res[i] >= 10:
            carry = res[i] // 10
            res[i] %= 10
            res[i+1] += carry
    # Find most significant digit
    idx = L - 1
    while idx >= 0 and res[idx] == 0:
        idx -= 1
    # Build answer string
    ans_str = ''.join(str(res[d]) for d in range(idx, -1, -1))
    print(ans_str)

if __name__ == "__main__":
    solve()
```

```python

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    S = input().strip()

    # result digits in little‑endian order (index 0 = units)
    L = N + 20          # enough for the whole number
    res = [0] * L

    P = 0               # P[i] = sum_{k=1}^{i} d_k * k
    for i in range(1, N + 1):
        digit = int(S[i - 1])
        P += digit * i          # now P = P[i]
        offset = N - i          # 10^{offset} multiplier for this term

        # add the decimal digits of P to res starting at 'offset'
        p = P
        pos = offset
        while p:
            p, r = divmod(p, 10)
            res[pos] += r
            pos += 1

    # propagate carries
    for i in range(L - 1):
        if res[i] >= 10:
            carry = res[i] // 10
            res[i] %= 10
            res[i + 1] += carry

    # find most significant non‑zero digit
    idx = L - 1
    while idx >= 0 and res[idx] == 0:
        idx -= 1

    # build the answer string (most significant first)
    ans_str = ''.join(map(str, res[idx::-1]))
    print(ans_str)


if __name__ == "__main__":
    solve()
```
