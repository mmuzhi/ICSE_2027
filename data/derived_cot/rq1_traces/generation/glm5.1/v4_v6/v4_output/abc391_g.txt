import sys
from collections import deque

MOD = 998244353

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    S = data[2].strip()
    s_int = [ord(ch) - 97 for ch in S]

    init = tuple([0]*(N+1))
    state_to_idx = {init: 0}
    states = [init]
    trans = []  # will be filled in BFS order
    q = deque([0])

    while q:
        idx = q.popleft()
        dp = states[idx]
        trans_row = [0]*26
        for c in range(26):
            new = [0]*(N+1)
            # new[0] = 0 already
            for i in range(1, N+1):
                best = dp[i]
                if new[i-1] > best:
                    best = new[i-1]
                if s_int[i-1] == c and dp[i-1] + 1 > best:
                    best = dp[i-1] + 1
                new[i] = best
            nxt = tuple(new)
            if nxt not in state_to_idx:
                state_to_idx[nxt] = len(states)
                states.append(nxt)
                q.append(len(states)-1)
            trans_row[c] = state_to_idx[nxt]
        trans.append(trans_row)

    num_states = len(states)
    dp_cur = [0]*num_states
    dp_cur[0] = 1

    for _ in range(M):
        dp_next = [0]*num_states
        for s_idx in range(num_states):
            cnt = dp_cur[s_idx]
            if cnt == 0:
                continue
            for ns_idx in trans[s_idx]:
                dp_next[ns_idx] = (dp_next[ns_idx] + cnt) % MOD
        dp_cur = dp_next

    ans = [0]*(N+1)
    for idx, state in enumerate(states):
        k = state[N]
        ans[k] = (ans[k] + dp_cur[idx]) % MOD

    print(' '.join(str(x) for x in ans))

if __name__ == "__main__":
    solve()
