```python

import sys

def main():
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    K = next(it)

    # build adjacency list
    graph = [[] for _ in range(N + 1)]
    for _ in range(N - 1):
        a = next(it)
        b = next(it)
        graph[a].append(b)
        graph[b].append(a)

    # mark the K specified vertices
    is_marked = [False] * (N + 1)
    for _ in range(K):
        v = next(it)
        is_marked[v] = True
    total_marked = K

    # ---------- iterative DFS to obtain parent array and traversal order ----------
    parent = [0] * (N + 1)
    order = []
    stack = [1]
    parent[1] = -1          # sentinel for the root
    while stack:
        node = stack.pop()
        order.append(node)
        for nb in graph[node]:
            if nb != parent[node]:
                parent[nb] = node
                stack.append(nb)

    # ---------- compute cnt[v] = number of marked vertices in subtree of v ----------
    cnt = [0] * (N + 1)
    for node in reversed(order):
        c = 1 if is_marked[node] else 0
        for nb in graph[node]:
            if nb != parent[node]:   # nb is a child
                c += cnt[nb]
        cnt[node] = c

    # ---------- count edges that separate marked vertices ----------
    needed_edges = 0
    for v in range(2, N + 1):        # all vertices except the root
        if 0 < cnt[v] < total_marked:
            needed_edges += 1

    # answer = vertices in the minimal Steiner tree
    print(needed_edges + 1)

if __name__ == "__main__":
    main()
```
