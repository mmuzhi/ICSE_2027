import sys, heapq

def solve():
    data = sys.stdin.read().split()
    it = iter(data)
    H = int(next(it)); W = int(next(it)); X = int(next(it))
    P = int(next(it)); Q = int(next(it))
    S = [[0]*W for _ in range(H)]
    for i in range(H):
        for j in range(W):
            S[i][j] = int(next(it))
    start_r = P-1; start_c = Q-1
    visited = [[False]*W for _ in range(H)]
    current_sum = S[start_r][start_c]
    visited[start_r][start_c] = True
    heap = []
    dirs = [(1,0),(-1,0),(0,1),(0,-1)]
    # add neighbors of start
    for dr, dc in dirs:
        nr, nc = start_r+dr, start_c+dc
        if 0 <= nr < H and 0 <= nc < W and not visited[nr][nc]:
            heapq.heappush(heap, (X * S[nr][nc] + 1, nr, nc))
    while heap:
        T, r, c = heapq.heappop(heap)
        if visited[r][c]:
            continue
        if T <= current_sum:
            visited[r][c] = True
            current_sum += S[r][c]
            for dr, dc in dirs:
                nr, nc = r+dr, c+dc
                if 0 <= nr < H and 0 <= nc < W and not visited[nr][nc]:
                    heapq.heappush(heap, (X * S[nr][nc] + 1, nr, nc))
        else:
            break
    print(current_sum)

if __name__ == "__main__":
    solve()

import sys
import heapq

def solve() -> None:
    data = sys.stdin.read().split()
    it = iter(data)
    H = int(next(it))
    W = int(next(it))
    X = int(next(it))
    P = int(next(it))
    Q = int(next(it))

    S = [[0] * W for _ in range(H)]
    for i in range(H):
        for j in range(W):
            S[i][j] = int(next(it))

    sr, sc = P - 1, Q - 1          # 0‑based start
    visited = [[False] * W for _ in range(H)]
    cur = S[sr][sc]                # current total strength
    visited[sr][sc] = True

    heap = []
    dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)]

    # initial frontier
    for dr, dc in dirs:
        nr, nc = sr + dr, sc + dc
        if 0 <= nr < H and 0 <= nc < W and not visited[nr][nc]:
            heapq.heappush(heap, (X * S[nr][nc] + 1, nr, nc))

    while heap:
        T, r, c = heapq.heappop(heap)
        if visited[r][c]:
            continue
        if T <= cur:                # condition cur > X * S[r][c] holds
            visited[r][c] = True
            cur += S[r][c]
            for dr, dc in dirs:
                nr, nc = r + dr, c + dc
                if 0 <= nr < H and 0 <= nc < W and not visited[nr][nc]:
                    heapq.heappush(heap, (X * S[nr][nc] + 1, nr, nc))
        else:
            # smallest required strength already exceeds current strength
            break

    print(cur)

if __name__ == "__main__":
