- There is no building taller than Building j between Buildings i and j.

Input

N
H_1 H_2 \ldots H_N

Output

For each i = 1, 2, \ldots, N, let c_i be the number of j satisfying the condition. Print c_1, c_2, \ldots, c_N in order, separated by spaces.

- 1 \leq N \leq 2 \times 10^5
- 1 \leq H_i \leq N
-  H_i\neq H_j\ (i\neq j)
- All input values are integers.

def solve():
    import sys
    data = sys.stdin.read().split()
    if not data: return
    it = iter(data)
    N = int(next(it))
    H = [int(next(it)) for _ in range(N)]

    L = [-1]*N
    stack = []
    for i in range(N):
        while stack and H[stack[-1]] < H[i]:
            stack.pop()
        if stack:
            L[i] = stack[-1]
        else:
            L[i] = -1
        stack.append(i)

    diff = [0]*(N+1)  # extra for r+1
    for j in range(1, N):
        l = L[j] if L[j] != -1 else 0
        r = j-1
        # l <= r always
        diff[l] += 1
        diff[r+1] -= 1

    ans = [0]*N
    cur = 0
    for i in range(N):
        cur += diff[i]
        ans[i] = cur

    print(' '.join(map(str, ans)))

        while stack and H[stack[-1]] < H[i]:
            stack.pop()
        if stack:
            L[i] = stack[-1]
        else:
            L[i] = -1
        stack.append(i)

    # difference array for range additions
    diff = [0] * (N + 1)          # extra cell for r+1
    for j in range(1, N):
        left = L[j] if L[j] != -1 else 0
        right = j - 1
        diff[left] += 1
        diff[right + 1] -= 1

    # prefix sum -> answer
    ans = [0] * N
    cur = 0
    for i in range(N):
        cur += diff[i]
        ans[i] = cur

    print(' '.join(map(str, ans)))

if __name__ == "__main__":
    solve()
