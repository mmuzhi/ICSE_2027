class Solution:
    def findValidPair(self, s: str) -> str:
        # Count occurrences of each digit (1-9)
        freq = [0] * 10
        for ch in s:
            freq[ord(ch) - ord('0')] += 1

        # Check each adjacent pair from left to right
        for i in range(len(s) - 1):
            a, b = s[i], s[i+1]
            if (a != b and
                freq[int(a)] == int(a) and
                freq[int(b)] == int(b)):
