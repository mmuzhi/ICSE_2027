=== ANSWER ===
```python

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])          # N is not really needed, but we parse it
    c1 = data[1]
    c2 = data[2]
    S = data[3]               # the original string

    # Build the result: keep c1, replace everything else with c2
    result = ''.join(c if c == c1 else c2 for c in S)
    print(result)

if __name__ == "__main__":
    main()
```
