import sys, heapq

def solve():
    import sys
    input = sys.stdin.readline
    N, K = map(int, input().split())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    C = list(map(int, input().split()))
    A.sort(reverse=True)
    B.sort(reverse=True)
    C.sort(reverse=True)
    N2 = N * N
    visited = set()
    heap = []
    # initial
    f0 = A[0]*B[0] + B[0]*C[0] + C[0]*A[0]
    code0 = 0
    heapq.heappush(heap, (-f0, code0))
    visited.add(code0)
    ans = 0
    for _ in range(K):
        neg_f, code = heapq.heappop(heap)
        ans = -neg_f
        # decode
        i = code // N2
        rem = code % N2
        j = rem // N
        k = rem % N
        # neighbors
        if i + 1 < N:
            ncode = (i+1)*N2 + j*N + k
            if ncode not in visited:
                visited.add(ncode)
                nf = A[i+1]*B[j] + B[j]*C[k] + C[k]*A[i+1]
                heapq.heappush(heap, (-nf, ncode))
        if j + 1 < N:
            ncode = i*N2 + (j+1)*N + k
            if ncode not in visited:
                visited.add(ncode)
                nf = A[i]*B[j+1] + B[j+1]*C[k] + C[k]*A[i]
                heapq.heappush(heap, (-nf, ncode))
        if k + 1 < N:
            ncode = i*N2 + j*N + (k+1)
            if ncode not in visited:
                visited.add(ncode)
                nf = A[i]*B[j] + B[j]*C[k+1] + C[k+1]*A[i]
                heapq.heappush(heap, (-nf, ncode))
    print(ans)

if __name__ == "__main__":
    solve()
