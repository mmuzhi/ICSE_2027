import sys
from array import array

MOD = 998244353

def main():
    input = sys.stdin.readline
    H, W = map(int, input().split())
    grid_raw = [input().strip() for _ in range(H)]

    # transpose if H > W
    if H > W:
        # transpose
        new_grid = [[grid_raw[j][i] for j in range(H)] for i in range(W)]
        H, W = W, H
        grid_raw = new_grid

    R = H  # rows (small)
    C = W  # columns (large)

    # convert to ints: -1 for '?', else 0,1,2
    grid = [[-1 if ch == '?' else int(ch)-1 for ch in row] for row in grid_raw]

    # generate all valid column colorings
    states_seq = []
    cur = [0]*R
    def dfs_gen(i, prev):
        if i == R:
            states_seq.append(tuple(cur))
            return
        for c in range(3):
            if i == 0 or c != prev:
                cur[i] = c
                dfs_gen(i+1, c)
    dfs_gen(0, -1)
    S = len(states_seq)

    # precompute powers of 3
    pow3 = [1]*(R+1)
    for i in range(1, R+1):
        pow3[i] = pow3[i-1]*3

    total_codes = pow3[R]  # 3**R
    # code_to_idx as signed short array
    code_to_idx = array('h', [-1])*total_codes
    for idx, seq in enumerate(states_seq):
        code = 0
        for i, val in enumerate(seq):
            code += val * pow3[i]
        code_to_idx[code] = idx

    # build adjacency
    adj = [array('I') for _ in range(S)]
    for p_idx, p_seq in enumerate(states_seq):
        # iterative DFS
        stack = [(0, -1, 0)]  # (i, prev_color, code)
        while stack:
            i, prev, code = stack.pop()
            if i == R:
                c_idx = code_to_idx[code]
                # should be valid
                adj[p_idx].append(c_idx)
            else:
                # iterate colors in order 0,1,2
                for color in range(3):
                    if color != p_seq[i] and (i == 0 or color != prev):
                        new_code = code + color * pow3[i]
                        stack.append((i+1, color, new_code))

    # precompute allowed states per column
    allowed = [bytearray(S) for _ in range(C)]
    for j in range(C):
        allowed_j = allowed[j]
        for idx in range(S):
            seq = states_seq[idx]
            ok = True
            for i in range(R):
                if grid[i][j] != -1 and seq[i] != grid[i][j]:
                    ok = False
                    break
            if ok:
                allowed_j[idx] = 1

    # DP
    dp_prev = [0]*S
    # first column
    for idx in range(S):
        if allowed[0][idx]:
            dp_prev[idx] = 1

    for j in range(1, C):
        dp_next = [0]*S
        allowed_j = allowed[j]
        for p_idx in range(S):
            val = dp_prev[p_idx]
            if val == 0:
                continue
            for c_idx in adj[p_idx]:
                if allowed_j[c_idx]:
                    new = dp_next[c_idx] + val
                    if new >= MOD:
                        new -= MOD
                    dp_next[c_idx] = new
        dp_prev = dp_next

    ans = 0
    last_allowed = allowed[C-1]
    for idx in range(S):
        if last_allowed[idx]:
            ans += dp_prev[idx]
            if ans >= MOD:
                ans -= MOD
    print(ans % MOD)

if __name__ == "__main__":
    main()
