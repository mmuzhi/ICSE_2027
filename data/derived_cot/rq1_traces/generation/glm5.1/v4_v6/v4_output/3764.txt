Return the maximum sum.

```python
class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
```
        candidates = []

            if limit > 0:
                row_sorted = sorted(row, reverse=True)
                candidates.extend(row_sorted[:limit])
        candidates.sort(reverse=True)
        return sum(candidates[:k])

```python

from typing import List

class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
        candidates = []
        for i, row in enumerate(grid):
            limit = limits[i]
            if limit > 0:
                # take the largest 'limit' numbers from this row
                sorted_row = sorted(row, reverse=True)
                candidates.extend(sorted_row[:limit])
        # pick the k largest among all candidates
        candidates.sort(reverse=True)
        return sum(candidates[:k])
```
