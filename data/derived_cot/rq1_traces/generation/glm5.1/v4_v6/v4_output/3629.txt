```python
class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
```

```python

class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
        MOD = 10**9 + 7
        # count of each letter in the current string
        cnt = [0] * 26
        for ch in s:
            cnt[ord(ch) - ord('a')] += 1

        for _ in range(t):
            nxt = [0] * 26
            # a..y -> next letter
            for i in range(25):
                nxt[i + 1] = (nxt[i + 1] + cnt[i]) % MOD
            # z -> a and b
            nxt[0] = (nxt[0] + cnt[25]) % MOD
            nxt[1] = (nxt[1] + cnt[25]) % MOD
            cnt = nxt

        return sum(cnt) % MOD
```
