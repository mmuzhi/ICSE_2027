=== ANSWER ===
```python

import sys

def main() -> None:
    input = sys.stdin.readline
    N, D = map(int, input().split())
    S = input().strip()
    arr = list(S)

    # positions of '@' (0‑based)
    at_pos = [i for i, ch in enumerate(arr) if ch == '@']
    # remove the D rightmost cookies
    for i in at_pos[-D:]:
        arr[i] = '.'

    print(''.join(arr))

if __name__ == "__main__":
    main()
```
