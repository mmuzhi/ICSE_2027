import sys

def minimal_period(s):
    n = len(s)
    if n == 0:
        return 0
    pi = [0]*n
    for i in range(1,n):
        j = pi[i-1]
        while j > 0 and s[i] != s[j]:
            j = pi[j-1]
        if s[i] == s[j]:
            j += 1
        pi[i] = j
    candidate = n - pi[-1]
    if n % candidate == 0:
        return candidate
    else:
        return n

def solve():
    import sys
    data = sys.stdin.read().split()
    # data contains tokens: t, then for each test case: S, X, Y
    it = iter(data)
    t = int(next(it))
    out = []
    for _ in range(t):
        S = next(it)
        X = next(it)
        Y = next(it)
        lenS = len(S)
        a = X.count('0')
        b = X.count('1')
        c = Y.count('0')
        d = Y.count('1')
        if a == c:
            out.append("Yes")
            continue
        delta_a = a - c
        delta_d = d - b
        if delta_a * delta_d <= 0:
            out.append("No")
            continue
        if (delta_a * lenS) % delta_d != 0:
            out.append("No")
            continue
        L = (delta_a * lenS) // delta_d
        period = minimal_period(S)
        if L % period == 0:
            out.append("Yes")
        else:
            out.append("No")
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
