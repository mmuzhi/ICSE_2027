def minOperations(self, n, m):
    L = len(str(n))
    min_val = 0 if L == 1 else 10**(L-1)
    max_val = 10**L - 1

    # sieve primes up to max_val
    is_prime = [True] * (max_val+1)
    if max_val >= 0: is_prime[0] = False
    if max_val >= 1: is_prime[1] = False
    for i in range(2, int(max_val**0.5)+1):
        if is_prime[i]:
            for j in range(i*i, max_val+1, i):
                is_prime[j] = False

    if is_prime[n] or is_prime[m]:
        return -1

    INF = 10**9
    dist = [INF] * (max_val+1)
    dist[n] = 0
    import heapq
    heap = [(0, n)]

    # precompute powers
    powers = [10**(L-1-i) for i in range(L)]

    while heap:
        d, u = heapq.heappop(heap)
        if d != dist[u]:
            continue
        if u == m:
            break
        # generate neighbors
        for i in range(L):
            p = powers[i]
            digit = (u // p) % 10
            # increase
            if digit < 9:
                v = u + p
                if v <= max_val and not is_prime[v]:
                    nd = d + v
                    if nd < dist[v]:
                        dist[v] = nd
                        heapq.heappush(heap, (nd, v))
            # decrease
            if digit > 0:
                v = u - p
                if v >= min_val and not is_prime[v]:
                    nd = d + v
                    if nd < dist[v]:
                        dist[v] = nd
                        heapq.heappush(heap, (nd, v))

    if dist[m] == INF:
        return -1
    else:
        return n + dist[m]
