You will be given a competitive programming problem.
You are given an integer array nums and a positive integer k.
The value of a sequence seq of size 2 * x is defined as:

(seq[0] OR seq[1] OR ... OR seq[x - 1]) XOR (seq[x] OR seq[x + 1] OR ... OR seq[2 * x - 1]).

Return the maximum value of any subsequence of nums having size 2 * k.
 
Example 1:

Input: nums = [2,6,7], k = 1
Output: 5
Explanation:
The subsequence [2, 7] has the maximum value of 2 XOR 7 = 5.

Example 2:

Input: nums = [4,2,5,6,7], k = 2
Output: 2
Explanation:
The subsequence [4, 5, 6, 7] has the maximum value of (4 OR 5) XOR (6 OR 7) = 2.

 
Constraints:

2 <= nums.length <= 400
1 <= nums[i] < 2^7
1 <= k <= nums.length / 2

class Solution:
    def maxValue(self, nums: List[int], k: int) -> int:
        n = len(nums)
        MAXV = 128  # since nums[i] < 128
        # Precompute or_with table
        or_with = [[0]*MAXV for _ in range(MAXV)]
        for x in range(MAXV):
            for v in range(MAXV):
                or_with[x][v] = v | x

        def compute_masks(arr):
            m = len(arr)
            masks = [0]*(m+1)
            prev = [0]*(k+1)
            prev[0] = 1  # bit0 set
            masks[0] = 0
            for i in range(1, m+1):
                x = arr[i-1]
                cur = [0]*(k+1)
                for c in range(k+1):
                    skip = prev[c]
                    take = 0
                    if c > 0:
                        prev_mask = prev[c-1]
                        if prev_mask:
                            new_mask = 0
                            # iterate over possible OR values
                            # using or_with table
                            ox = or_with[x]  # list of v|x for v
                            for v in range(MAXV):
                                if (prev_mask >> v) & 1:
                                    new_mask |= 1 << ox[v]
                            take = new_mask
                    cur[c] = skip | take
                masks[i] = cur[k]
                prev = cur
            return masks

        pref_masks = compute_masks(nums)
        rev_masks = compute_masks(nums[::-1])

        ans = 0
        for p in range(k-1, n-k):
            a_mask = pref_masks[p+1]
            b_mask = rev_masks[n - (p+1)]
            # compute max xor between a in a_mask and b in b_mask
            # iterate over a values
            # We can break early if ans == 127, but not necessary
            for a in range(MAXV):
                if (a_mask >> a) & 1:
                    for b in range(MAXV):
                        if (b_mask >> b) & 1:
                            val = a ^ b
                            if val > ans:
                                ans = val
        return ans

```python
from typing import List

class Solution:
    def maxValue(self, nums: List[int], k: int) -> int:
        n = len(nums)
        MAXV = 128                     # because nums[i] < 2^7

        # DP that returns a list masks where masks[i] is the set (as bitmask)
        # of OR values obtainable by choosing exactly k elements from the first i elements.
        def compute_masks(arr: List[int]) -> List[int]:
            m = len(arr)
            masks = [0] * (m + 1)
            prev = [0] * (k + 1)
            prev[0] = 1                # OR = 0 with 0 elements
            masks[0] = 0               # no k elements from empty prefix
            for i in range(1, m + 1):
                x = arr[i - 1]
                cur = [0] * (k + 1)
                for c in range(k + 1):
                    skip = prev[c]
                    take = 0
                    if c > 0:
                        prev_mask = prev[c - 1]
                        if prev_mask:
                            new_mask = 0
                            # for every OR value v already reachable with c-1 elements,
                            # v | x becomes reachable with c elements.
                            for v in range(MAXV):
                                if (prev_mask >> v) & 1:
                                    new_mask |= 1 << (v | x)
                            take = new_mask
                    cur[c] = skip | take
                masks[i] = cur[k]
                prev = cur
            return masks

        # OR sets for all prefixes of the original array
        pref_masks = compute_masks(nums)
        # OR sets for all prefixes of the reversed array – these correspond to suffixes of the original
        rev_masks = compute_masks(nums[::-1])

        ans = 0
        # p is the last index of the first half (0‑based)
        for p in range(k - 1, n - k):
            a_mask = pref_masks[p + 1]                 # first half from nums[0..p]
            b_mask = rev_masks[n - (p + 1)]            # second half from nums[p+1..n-1]
            # extract the concrete OR values from the bitmasks
            a_vals = [a for a in range(MAXV) if (a_mask >> a) & 1]
            b_vals = [b for b in range(MAXV) if (b_mask >> b) & 1]
            for a in a_vals:
                for b in b_vals:
                    val = a ^ b
                    if val > ans:
                        ans = val
        return ans
```
