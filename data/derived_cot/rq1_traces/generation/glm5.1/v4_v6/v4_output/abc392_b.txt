import sys

def main() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    A = list(map(int, data[2:2 + M]))

    present = set(A)
    missing = [i for i in range(1, N + 1) if i not in present]

    print(len(missing))
    if missing:
        print(' '.join(map(str, missing)))
    else:
        print()

if __name__ == "__main__":
    main()
