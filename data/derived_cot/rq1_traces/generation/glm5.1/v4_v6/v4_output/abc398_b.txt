import sys
from collections import Counter

def solve() -> None:

    cnt = Counter(A)
    freqs = sorted(cnt.values(), reverse=True)

    if len(freqs) >= 2 and freqs[0] >= 3 and freqs[1] >= 2:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    solve()
