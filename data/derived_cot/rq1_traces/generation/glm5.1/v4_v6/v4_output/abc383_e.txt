def solve():

    sys.setrecursionlimit(1<<25)
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = []
    for _ in range(M):
        u, v, w = map(int, input().split())
        edges.append((w, u, v))
    edges.sort()
    # DSU
    parent = list(range(N+1))
    size = [1]*(N+1)
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    # rep_node: for each DSU root, the reconstruction tree node representing that component
    rep_node = list(range(N+1))  # initially, component i is represented by leaf i
    # reconstruction tree arrays
    max_nodes = 2*N + 5
    left = [0]*max_nodes
    right = [0]*max_nodes
    weight = [0]*max_nodes
    node_cnt = N
    for w, u, v in edges:
        ru = find(u)
        rv = find(v)
        if ru == rv:
            continue
        node_cnt += 1
        new_node = node_cnt
        weight[new_node] = w
        left[new_node] = rep_node[ru]
        right[new_node] = rep_node[rv]
        # union by size
        if size[ru] < size[rv]:
            ru, rv = rv, ru
        parent[rv] = ru
        size[ru] += size[rv]
        rep_node[ru] = new_node
    root = rep_node[find(1)]  # graph connected, so any vertex works

    # read A and B
    A_list = list(map(int, input().split()))
    B_list = list(map(int, input().split()))
    countA = [0]*(N+1)
    countB = [0]*(N+1)
    for a in A_list:
        countA[a] += 1
    for b in B_list:
        countB[b] += 1

    # iterative post-order DFS
    # We'll use stack of tuples (node, state) where state 0 = enter, 1 = exit
    stack = [(root, 0)]
    # arrays to store excess
    excessA = [0]*max_nodes
    excessB = [0]*max_nodes
    ans = 0
    while stack:
        node, state = stack.pop()
        if node == 0:
            continue
        if state == 0:
            stack.append((node, 1))
            # push children
            if right[node]:
                stack.append((right[node], 0))
            if left[node]:
                stack.append((left[node], 0))
        else:
            if node <= N:  # leaf
                a = countA[node]
                b = countB[node]
            else:
                l = left[node]
                r = right[node]
                a = excessA[l] + excessA[r]
                b = excessB[l] + excessB[r]
            matches = min(a, b)
            ans += matches * weight[node]
            excessA[node] = a - matches
            excessB[node] = b - matches
    print(ans)

The function f(x,y) is the smallest w such that x and y are in the same block of the partition at threshold w. The goal is to match A and B to minimize sum of f.

We can consider processing thresholds in increasing order. Initially, each vertex is its own block. As we increase w, blocks merge. At each merge, we have some blocks that merge into a larger block. Within each block, we have some numbers of A and B that are currently unmatched (since we haven't matched them yet). When blocks merge, we can match A from one block with B from another block, incurring cost w for each such match. To minimize total cost, we should match as many as possible at the smallest possible w. This is exactly the greedy algorithm: at each merge (i.e., when two blocks become connected at weight w), match as many A-B pairs as possible between the two blocks, because any unmatched will have to be matched later at higher w. This is analogous to the "minimum cost matching in a matroid" or something.

This greedy is optimal because the cost function is linear in w and the matching decisions are independent across merges? More formally, we can use an exchange argument: Suppose we have an optimal matching. Consider the smallest weight w at which some pair is matched across two blocks that merged at w. If there is a pair matched at a higher weight that could have been matched at w, we can swap to improve or not worsen. The greedy ensures that at each merge, we match the maximum possible number of cross-block pairs, which is exactly the minimum number of pairs that must be matched at that weight in any feasible matching (since the excess A and B from the two blocks must eventually be matched with something, and the only way to avoid matching them at this weight is to match them within their own blocks, but that's impossible because within each block, after matching internally, there is only one type of excess). So the number of cross-block matches at weight w is forced to be at least the min of the excesses. The greedy achieves that lower bound, thus optimal.

import sys

def solve():
    import sys
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = []
    for _ in range(M):
        u, v, w = map(int, input().split())
        edges.append((w, u, v))
    edges.sort()
    parent = list(range(N+1))
    size = [1]*(N+1)
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    rep_node = list(range(N+1))  # for DSU roots
    max_nodes = 2*N + 5
    left = [0]*max_nodes
    right = [0]*max_nodes
    weight = [0]*max_nodes
    node_cnt = N
    for w, u, v in edges:
        ru = find(u)
        rv = find(v)
        if ru == rv:
            continue
        node_cnt += 1
        new_node = node_cnt
        weight[new_node] = w
        left[new_node] = rep_node[ru]
        right[new_node] = rep_node[rv]
        # union by size
        if size[ru] < size[rv]:
            ru, rv = rv, ru
        parent[rv] = ru
        size[ru] += size[rv]
        rep_node[ru] = new_node
    root = rep_node[find(1)]
    # read A and B
    A_list = list(map(int, input().split()))
    B_list = list(map(int, input().split()))
    countA = [0]*(N+1)
    countB = [0]*(N+1)
    for a in A_list:
        countA[a] += 1
    for b in B_list:
        countB[b] += 1
    # iterative post-order DFS
    stack = [(root, 0)]
    excessA = [0]*max_nodes
    excessB = [0]*max_nodes
    ans = 0
    while stack:
        node, state = stack.pop()
        if node == 0:
            continue
        if state == 0:
            stack.append((node, 1))
            if right[node]:
                stack.append((right[node], 0))
            if left[node]:
                stack.append((left[node], 0))
        else:
            if node <= N:
                a = countA[node]
                b = countB[node]
            else:
                l = left[node]
                r = right[node]
                a = excessA[l] + excessA[r]
                b = excessB[l] + excessB[r]
            matches = min(a, b)
            ans += matches * weight[node]
            excessA[node] = a - matches
            excessB[node] = b - matches
    print(ans)

if __name__ == "__main__":
    solve()
