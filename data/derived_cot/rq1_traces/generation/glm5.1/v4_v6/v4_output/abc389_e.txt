import sys

def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    P = list(map(int, input().split()))
    # Binary search for X_max
    lo = 0
    hi = 10**15  # safe upper bound
    ans_X = 0
    # define check function inside to capture P, M
    def check(X):
        total = 0
        for p in P:
            t = X // p
            cnt = (t + 1) // 2
            # cost contribution = p * cnt * cnt
            total += p * cnt * cnt
            if total > M:
                return False
        return True
    while lo <= hi:
        mid = (lo + hi) // 2
        if check(mid):
            ans_X = mid
            lo = mid + 1
        else:
            hi = mid - 1
    X = ans_X
    total_units = 0
    total_cost = 0
    min_next = 10**30
    cnt_min = 0
    for p in P:
        t = X // p
        cnt = (t + 1) // 2
        total_units += cnt
        total_cost += p * cnt * cnt
        next_c = (2 * cnt + 1) * p
        if next_c < min_next:
            min_next = next_c
            cnt_min = 1
        elif next_c == min_next:
            cnt_min += 1
    R = M - total_cost
    add = min(cnt_min, R // min_next) if min_next != 10**30 else 0
    ans = total_units + add
    print(ans)

if __name__ == "__main__":
    solve()

import sys

def solve():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    P = [int(next(it)) for _ in range(N)]

    lo = 0
    hi = 10**15
    ans_X = 0
    # define check as inner function using local variables for speed
    P_local = P
    M_local = M
    def check(X):
        total = 0
        for p in P_local:
            t = X // p
            cnt = (t + 1) >> 1  # use bit shift for //2? (t+1)//2 same as (t+1)>>1
            total += p * cnt * cnt
            if total > M_local:
                return False
        return True

    while lo <= hi:
        mid = (lo + hi) // 2
        if check(mid):
            ans_X = mid
            lo = mid + 1
        else:
            hi = mid - 1

    X = ans_X
    total_units = 0
    total_cost = 0
    min_next = 10**30
    cnt_min = 0
    for p in P_local:
        t = X // p
        cnt = (t + 1) >> 1
        total_units += cnt
        total_cost += p * cnt * cnt
        next_c = (2 * cnt + 1) * p
        if next_c < min_next:
            min_next = next_c
            cnt_min = 1
        elif next_c == min_next:
            cnt_min += 1

    R = M_local - total_cost
    add = min(cnt_min, R // min_next)
    ans = total_units + add
    print(ans)

if __name__ == "__main__":
