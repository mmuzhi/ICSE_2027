def solve():
    import sys
    from collections import deque
    from array import array

    MOD = 998244353

    input = sys.stdin.readline
    N, M = map(int, input().split())
    A = list(map(int, input().split()))
    A = [x-1 for x in A]  # 0-index

    # indegree
    indeg = [0]*N
    for i in range(N):
        indeg[A[i]] += 1

    q = deque([i for i in range(N) if indeg[i]==0])
    in_cycle = [True]*N
    while q:
        u = q.popleft()
        in_cycle[u] = False
        v = A[u]
        indeg[v] -= 1
        if indeg[v] == 0:
            q.append(v)

    # identify cycles
    cycle_id = [-1]*N
    cycles = []
    for i in range(N):
        if in_cycle[i] and cycle_id[i] == -1:
            # collect cycle
            cur = i
            cycle_nodes = []
            while True:
                cycle_nodes.append(cur)
                cur = A[cur]
                if cur == i:
                    break
            cid = len(cycles)
            for node in cycle_nodes:
                cycle_id[node] = cid
            cycles.append(cycle_nodes)

    # assign compressed ids
    comp_id = [0]*N
    next_id = 0
    # tree nodes
    for i in range(N):
        if not in_cycle[i]:
            comp_id[i] = next_id
            next_id += 1
    # cycles
    cycle_comp_id = []
    for c in range(len(cycles)):
        cid = next_id
        next_id += 1
        cycle_comp_id.append(cid)
        for node in cycles[c]:
            comp_id[node] = cid

    K = next_id

    # build parent and children
    parent = [-1]*K
    children = [[] for _ in range(K)]
    for i in range(N):
        if not in_cycle[i]:  # tree node
            u = comp_id[i]
            p = comp_id[A[i]]
            parent[u] = p
            children[p].append(u)

    # compute depth via BFS from roots
    depth = [-1]*K
    dq = deque()
    for u in range(K):
        if parent[u] == -1:
            depth[u] = 0
            dq.append(u)
    while dq:
        u = dq.popleft()
        for c in children[u]:
            depth[c] = depth[u] + 1
            dq.append(c)

    # order by depth descending
    order = sorted(range(K), key=lambda u: depth[u], reverse=True)

    # DP
    dp = [None] * K  # will hold array('I') for each node when processed

    for u in order:
        if not children[u]:
            # leaf
            dp_u = array('I', [1]) * (M+1)  # length M+1, all ones
            dp[u] = dp_u
        else:
            dp_u = array('I', [1]) * (M+1)
            for c in children[u]:
                dp_c = dp[c]
                s = 0
                # iterate v from 1 to M
                for v in range(1, M+1):
                    s = (s + dp_c[v]) % MOD
                    dp_u[v] = (dp_u[v] * s) % MOD
                # free child's dp
                dp[c] = None
            dp[u] = dp_u

    # compute answer
    ans = 1
    for u in range(K):
        if parent[u] == -1:
            total = 0
            dp_root = dp[u]
            for v in range(1, M+1):
                total = (total + dp_root[v]) % MOD
            ans = ans * total % MOD

    print(ans)
