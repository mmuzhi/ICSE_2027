class Solution:
    def minOperations(self, queries: List[List[int]]) -> int:
        # Precompute powers of 4 and prefix sums of f(x) up to 4^15
        pow4 = [1]
        while pow4[-1] <= 10**9:
            pow4.append(pow4[-1] * 4)
        # Now pow4 has indices 0..15 (since 4^15 > 1e9)
        # Compute prefix sums of sum_{x=1}^{4^k -1} f(x)
        prefix = [0]  # prefix[k] = sum_{i=1}^{k} i*(4^i - 4^{i-1})
        for k in range(1, len(pow4)):
            full = k * (pow4[k] - pow4[k-1])
            prefix.append(prefix[-1] + full)
        
        def sum_f(N: int) -> int:
            if N <= 0:
                return 0
            # find k such that 4^{k-1} <= N < 4^k
            k = (N.bit_length() - 1) // 2 + 1
            # k is at most len(pow4)-1 for N <= 1e9
            return prefix[k-1] + k * (N - pow4[k-1] + 1)
        
        total = 0
        for l, r in queries:
            S = sum_f(r) - sum_f(l-1)
            M = (r.bit_length() - 1) // 2 + 1
            ans = max((S + 1) // 2, M)
            total += ans
        return total

from typing import List

class Solution:
    def minOperations(self, queries: List[List[int]]) -> int:
        # ---------- precompute powers of 4 and prefix sums ----------
        pow4 = [1]                     # 4^0
        while pow4[-1] <= 10**9:       # we need up to 4^15 > 1e9
            pow4.append(pow4[-1] * 4)

        # prefix[k] = sum_{x=1}^{4^k - 1} f(x)
        prefix = [0]
        for k in range(1, len(pow4)):
            full = k * (pow4[k] - pow4[k-1])   # sum over whole interval k
            prefix.append(prefix[-1] + full)

        # ---------- helper: sum of f(x) from 1 to N ----------
        def sum_f(N: int) -> int:
            if N <= 0:
                return 0
            # k = number of base‑4 digits of N = f(N)
            k = (N.bit_length() - 1) // 2 + 1
            # sum = full sum up to 4^{k-1}-1  +  k * (numbers from 4^{k-1} to N)
            return prefix[k-1] + k * (N - pow4[k-1] + 1)

        # ---------- process each query ----------
        total = 0
        for l, r in queries:
            S = sum_f(r) - sum_f(l-1)          # total sum of f(x) in [l, r]
            M = (r.bit_length() - 1) // 2 + 1  # f(r) = max f in the range
            ans = max((S + 1) // 2, M)         # minimal operations
            total += ans

        return total
