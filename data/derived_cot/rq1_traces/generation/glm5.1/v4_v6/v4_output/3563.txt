Then min_cost_flow(s, t):
    N = len(graph)
    INF = 10**9
    flow = 0
    cost = 0
    while True:
        dist = [INF]*N
        inqueue = [False]*N
        prevv = [-1]*N
        preve = [-1]*N
        dist[s] = 0
        # Bellman-Ford with queue (SPFA)
        q = deque([s])
        inqueue[s] = True
        while q:
            v = q.popleft()
            inqueue[v] = False
            for i, e in enumerate(graph[v]):
                if e.cap > 0 and dist[e.to] > dist[v] + e.cost:
                    dist[e.to] = dist[v] + e.cost
                    prevv[e.to] = v
                    preve[e.to] = i
                    if not inqueue[e.to]:
                        q.append(e.to)
                        inqueue[e.to] = True
        if dist[t] == INF:
            break
        # find augmenting path capacity (should be 1)
        d = INF
        v = t
        while v != s:
            e = graph[prevv[v]][preve[v]]
            d = min(d, e.cap)
            v = prevv[v]
        flow += d
        cost += d * dist[t]
        v = t
        while v != s:
            e = graph[prevv[v]][preve[v]]
            e.cap -= d
            graph[v][e.rev].cap += d
            v = prevv[v]
    return flow, cost

Let's outline the min-cost flow implementation:

We'll define a class Edge with attributes: to, rev, cap, cost.

Graph is a list of lists of edges.

Function add_edge(frm, to, cap, cost):
    graph[frm].append(Edge(to, len(graph[to]), cap, cost))
    graph[to].append(Edge(frm, len(graph[frm])-1, 0, -cost))

Function min_cost_flow(s, t):
    N = len(graph)
    INF = 10**9
    flow = 0
    cost = 0
    while True:
        dist = [INF]*N
        inq = [False]*N
        prevv = [-1]*N
        preve = [-1]*N
        dist[s] = 0
        q = deque([s])
        inq[s] = True
        while q:
            v = q.popleft()
            inq[v] = False
            for i, e in enumerate(graph[v]):
                if e.cap > 0 and dist[e.to] > dist[v] + e.cost:
                    dist[e.to] = dist[v] + e.cost
                    prevv[e.to] = v
                    preve[e.to] = i
                    if not inq[e.to]:
                        q.append(e.to)
                        inq[e.to] = True
        if dist[t] == INF:
            break
        # find min capacity along path
        d = INF
        v = t
        while v != s:
            e = graph[prevv[v]][preve[v]]
            d = min(d, e.cap)
            v = prevv[v]
        flow += d
        cost += d * dist[t]
        v = t
        while v != s:
            e = graph[prevv[v]][preve[v]]
            e.cap -= d
            graph[v][e.rev].cap += d
            v = prevv[v]
    return flow, cost

```python
from collections import deque
from typing import List

class Solution:
    def maxScore(self, grid: List[List[int]]) -> int:
        R = len(grid)               # number of rows
        V = 100                     # values are in [1, 100]
        N = R + V + 2               # total nodes: source, rows, values, sink
        source = 0
        sink = R + V + 1

        # graph[u] = list of edges [to, rev, cap, cost]
        graph = [[] for _ in range(N)]

        def add_edge(frm: int, to: int, cap: int, cost: int) -> None:
            graph[frm].append([to, len(graph[to]), cap, cost])
            graph[to].append([frm, len(graph[frm]) - 1, 0, -cost])

        # source -> each row (capacity 1, cost 0)
        for i in range(R):
            add_edge(source, 1 + i, 1, 0)

        # each row -> distinct values it contains (capacity 1, cost = -value)
        for i, row in enumerate(grid):
            for v in set(row):          # deduplicate values in the row
                add_edge(1 + i, R + v, 1, -v)

        # each value -> sink (capacity 1, cost 0)
        for v in range(1, V + 1):
            add_edge(R + v, sink, 1, 0)

        INF = 10**9
        flow = 0
        cost = 0

        # successive shortest augmenting path (min‑cost max‑flow)
        while True:
            dist = [INF] * N
            inq = [False] * N
            prev_v = [-1] * N
            prev_e = [-1] * N
            dist[source] = 0
            q = deque([source])
            inq[source] = True

            # SPFA (Bellman‑Ford with queue)
            while q:
                v = q.popleft()
                inq[v] = False
                for i, e in enumerate(graph[v]):
                    to, rev, cap, c = e
                    if cap > 0 and dist[to] > dist[v] + c:
                        dist[to] = dist[v] + c
                        prev_v[to] = v
                        prev_e[to] = i
                        if not inq[to]:
                            q.append(to)
                            inq[to] = True

            if dist[sink] == INF:   # no augmenting path
                break

            # find bottleneck capacity (always 1 in this graph)
            d = INF
            v = sink
            while v != source:
                e = graph[prev_v[v]][prev_e[v]]
                d = min(d, e[2])
                v = prev_v[v]

            flow += d
            cost += d * dist[sink]

            # update residual capacities
            v = sink
            while v != source:
                e = graph[prev_v[v]][prev_e[v]]
                e[2] -= d
                graph[v][e[1]][2] += d
                v = prev_v[v]

        # total cost is negative of the maximum achievable sum
        return -cost
```
