=== ANSWER ===
```python

from itertools import permutations
from typing import List

class Solution:
    def maxGoodNumber(self, nums: List[int]) -> int:
        best = 0
        for a, b, c in permutations(nums):
            # Concatenate binary representations using bit shifts
            val = (a << (b.bit_length() + c.bit_length())) + \
                  (b << c.bit_length()) + c
            if val > best:
                best = val
        return best
```
