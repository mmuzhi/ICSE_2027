```python

import sys

def solve() -> None:
    s = sys.stdin.readline().strip()
    if '.' in s:
        int_part, dec_part = s.split('.')
        # Remove leading zeros from the integer part (e.g., "00" -> "0")
        int_part = str(int(int_part))
        # Remove trailing zeros from the decimal part
        dec_part = dec_part.rstrip('0')
        if dec_part:
            print(f"{int_part}.{dec_part}")
        else:
            print(int_part)
    else:
        # Should not happen under the given constraints, but handle gracefully
        print(str(int(s)))

if __name__ == "__main__":
    solve()
```
