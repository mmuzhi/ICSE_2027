MOD = 998244353

def solve():
    import sys
    N, M = map(int, sys.stdin.readline().split())
    if M == 1:
        print(N % MOD)
        return

    # primes up to M
    primes = [2,3,5,7,11,13]
    primes = [p for p in primes if p <= M]
    P = len(primes)
    # mapping prime -> index
    # we may not need mapping explicitly if we use list of exponents per a.

    # precompute g for all masks
    g = [0] * (1<<P)
    for a in range(1, M+1):
        # factorize a using primes
        exps = [0]*P
        temp = a
        for i, p in enumerate(primes):
            while temp % p == 0:
                exps[i] += 1
                temp //= p
        # now exps[i] is exponent for prime i
        # collect indices where exps[i] > 0
        indices = [i for i, e in enumerate(exps) if e > 0]
        L = len(indices)
        # iterate over nonempty subsets of indices
        for sub in range(1, 1<<L):
            prod = 1
            mask = 0
            for bit in range(L):
                if sub>>bit & 1:
                    idx = indices[bit]
                    prod = (prod * exps[idx]) % MOD
                    mask |= (1 << idx)
            g[mask] = (g[mask] + prod) % MOD

    # compute popcount for all masks
    popcnt = [0]*(1<<P)
    for mask in range(1<<P):
        popcnt[mask] = bin(mask).count('1')

    # DP for D (ordered partitions)
    D = [ [0]*(P+1) for _ in range(1<<P) ]
    D[0][0] = 1
    for mask in range(1, 1<<P):
        s = popcnt[mask]
        # iterate over submasks U of mask, U != 0
        sub = mask
        while sub:
            U = sub
            gU = g[U]
            if gU:
                rest = mask ^ U
                # for t from 1 to s:
                #   D[mask][t] += gU * D[rest][t-1]
                # we can loop t from 1 to s, but D[rest][t-1] is zero if t-1 > popcnt[rest]
                # we can limit t to popcnt[rest]+1
                max_t_rest = popcnt[rest]
                for t in range(1, s+1):
                    if t-1 <= max_t_rest:
                        D[mask][t] = (D[mask][t] + gU * D[rest][t-1]) % MOD
            sub = (sub - 1) & mask

    # compute U = D / t!
    fact = [1]*(P+1)
    for i in range(1, P+1):
        fact[i] = fact[i-1]*i % MOD
    inv_fact = [1]*(P+1)
    inv_fact[P] = pow(fact[P], MOD-2, MOD)
    for i in range(P,0,-1):
        inv_fact[i-1] = inv_fact[i]*i % MOD

    U = [ [0]*(P+1) for _ in range(1<<P) ]
    for mask in range(1, 1<<P):
        s = popcnt[mask]
        for t in range(1, s+1):
            U[mask][t] = D[mask][t] * inv_fact[t] % MOD

    # precompute inv_M powers
    M_mod = M % MOD
    inv_M = pow(M_mod, MOD-2, MOD)
    inv_M_pow = [1]*(P+1)
    for i in range(1, P+1):
        inv_M_pow[i] = inv_M_pow[i-1] * inv_M % MOD

    # compute S_t(N) for t=0..P
    # function for binom(N+1, j)
    def binom_large(n, j):
        if j < 0 or j > n:
            return 0
        if j == 0:
            return 1
        res = 1
        n_mod = n % MOD
        for i in range(j):
            term = (n_mod - i) % MOD
            if term == 0:
                return 0
            res = res * term % MOD
        res = res * inv_fact[j] % MOD
        return res

    # compute S_0
    # geometric series: (M^{N+1} - M) / (M-1) mod MOD
    pow_M_Nplus1 = pow(M_mod, N+1, MOD)
    numerator = (pow_M_Nplus1 - M_mod) % MOD
    denom = (M_mod - 1) % MOD
    inv_denom = pow(denom, MOD-2, MOD)
    S0 = numerator * inv_denom % MOD

    S = [0]*(P+1)
    S[0] = S0

    # precompute inv_1M powers
    inv_1M = pow((1 - M_mod) % MOD, MOD-2, MOD)
    inv_1M_pow = [1]*(P+2)  # up to P+1
    for i in range(1, P+2):
        inv_1M_pow[i] = inv_1M_pow[i-1] * inv_1M % MOD

    for t in range(1, P+1):
        # compute F_t = sum_{k=0}^{N} binom(k,t) M^k
        # using formula
        term1 = pow(M_mod, t, MOD) * inv_1M_pow[t+1] % MOD
        # compute sum2
        sum2 = 0
        for j in range(t+1):
            binom_val = binom_large(N+1, j)
            if binom_val == 0:
                continue
            term = binom_val * pow(M_mod, t - j, MOD) % MOD * inv_1M_pow[t - j + 1] % MOD
            sum2 = (sum2 + term) % MOD
        term2 = pow_M_Nplus1 * sum2 % MOD
        F_t = (term1 - term2) % MOD
        S[t] = F_t * fact[t] % MOD

    # total sum
    total = S[0]
    for mask in range(1, 1<<P):
        s = popcnt[mask]
        for t in range(1, s+1):
            u = U[mask][t]
            if u == 0:
                continue
            contrib = u * inv_M_pow[t] % MOD * S[t] % MOD
            total = (total + contrib) % MOD

    print(total)
