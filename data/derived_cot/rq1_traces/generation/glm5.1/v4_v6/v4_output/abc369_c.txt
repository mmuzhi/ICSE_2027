import sys

def main() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    a = list(map(int, data[1:1 + n]))

    # every single element is an arithmetic progression
    ans = n

    if n > 1:
        # differences between consecutive elements
        d = [a[i + 1] - a[i] for i in range(n - 1)]

        i = 0
        while i < len(d):
            j = i
            # extend the run of equal differences
            while j < len(d) and d[j] == d[i]:
                j += 1
            k = j - i          # length of the run in d
            # each run of length k contributes C(k+1, 2) subarrays of length >=2
            ans += k * (k + 1) // 2
            i = j

    print(ans)

if __name__ == "__main__":
    main()
