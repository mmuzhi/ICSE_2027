class Solution:
    def maxSum(self, nums: List[int]) -> int:
        # Collect all distinct positive numbers
        distinct_pos = {x for x in nums if x > 0}
        if distinct_pos:
            return sum(distinct_pos)
        else:
            # No positive numbers: the best we can do is the maximum element
            return max(nums)
