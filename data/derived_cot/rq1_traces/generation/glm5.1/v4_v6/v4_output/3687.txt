from typing import List

class Solution:
    def longestSpecialPath(self, edges: List[List[int]], nums: List[int]) -> List[int]:
        n = len(nums)
        adj = [[] for _ in range(n)]
        for u, v, length in edges:
            adj[u].append((v, length))
            adj[v].append((u, length))
        max_val = max(nums)
        last = [-1] * (max_val + 1)
        dist = [0] * n
        path_nodes = []
        left = 0
        best_len = -1
        best_nodes = 10**9  # large
        
        # stack: for enter: (0, node, parent, depth, dist_u)
        # for exit: (1, x, old_last, old_left)
        stack = [(0, 0, -1, 0, 0)]
        
        while stack:
            item = stack.pop()
            if item[0] == 0:  # enter
                _, node, parent, depth, dist_u = item
                # add to path
                path_nodes.append(node)
                dist[node] = dist_u
                x = nums[node]
                old_last = last[x]
                old_left = left
                # update left
                if old_last != -1:
                    new_left = old_last + 1
                    if new_left > left:
                        left = new_left
                # update last
                last[x] = depth
                # compute candidate
                start_node = path_nodes[left]
                length = dist_u - dist[start_node]
                nodes = depth - left + 1
                if length > best_len:
                    best_len = length
                    best_nodes = nodes
                elif length == best_len and nodes < best_nodes:
                    best_nodes = nodes
                # push exit
                stack.append((1, x, old_last, old_left))
                # push children
                for v, w in adj[node]:
                    if v == parent:
                        continue
                    stack.append((0, v, node, depth+1, dist_u + w))
            else:  # exit
                _, x, old_last, old_left = item
                # revert
                path_nodes.pop()
                last[x] = old_last
                left = old_left
        
        return [best_len, best_nodes]

from typing import List

class Solution:
    def longestSpecialPath(self, edges: List[List[int]], nums: List[int]) -> List[int]:
        n = len(nums)
        # build adjacency list
        adj = [[] for _ in range(n)]
        for u, v, w in edges:
            adj[u].append((v, w))
            adj[v].append((u, w))

        max_val = max(nums)
        last = [-1] * (max_val + 1)      # last depth of each value on current path
        dist = [0] * n                   # distance from root
        path_nodes = []                  # nodes on current root-to-node path, index = depth
        left = 0                         # smallest depth such that path[left..depth] has all unique values
        best_len = -1
        best_nodes = 10**9

        # stack entries:
        #   enter: (0, node, parent, depth, dist_u)
        #   exit : (1, x, old_last, old_left)
        stack = [(0, 0, -1, 0, 0)]

        while stack:
            item = stack.pop()
            if item[0] == 0:          # enter a node
                _, node, parent, depth, dist_u = item
                path_nodes.append(node)
                dist[node] = dist_u

                x = nums[node]
                old_last = last[x]
                old_left = left

                # update left bound if duplicate found
                if old_last != -1:
                    new_left = old_last + 1
                    if new_left > left:
                        left = new_left
                last[x] = depth

                # evaluate the longest special path ending at this node
                start_node = path_nodes[left]
                length = dist_u - dist[start_node]
                nodes = depth - left + 1

                if length > best_len:
                    best_len = length
                    best_nodes = nodes
                elif length == best_len and nodes < best_nodes:
                    best_nodes = nodes

                # push exit marker, then children
                stack.append((1, x, old_last, old_left))
                for v, w in adj[node]:
                    if v == parent:
                        continue
                    stack.append((0, v, node, depth + 1, dist_u + w))

            else:                     # exit a node
                _, x, old_last, old_left = item
                path_nodes.pop()
                last[x] = old_last
                left = old_left

        return [best_len, best_nodes]
