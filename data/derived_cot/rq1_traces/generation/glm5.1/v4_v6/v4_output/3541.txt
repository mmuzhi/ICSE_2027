class Solution:
    def reportSpam(self, message: List[str], bannedWords: List[str]) -> bool:

        for word in message:

=== ANSWER ===
```python

from typing import List

class Solution:
    def reportSpam(self, message: List[str], bannedWords: List[str]) -> bool:
        banned = set(bannedWords)          # O(m) time, O(m) space
        cnt = 0
        for word in message:               # O(n) time
            if word in banned:
                cnt += 1
                if cnt >= 2:
                    return True
        return False

```
