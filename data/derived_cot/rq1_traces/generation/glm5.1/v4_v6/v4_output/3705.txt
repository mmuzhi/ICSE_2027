```python

from typing import List
from collections import defaultdict

class Solution:
    def largestInteger(self, nums: List[int], k: int) -> int:
        n = len(nums)
        count = defaultdict(int)          # maps value -> number of subarrays containing it
        
        # Iterate over all subarrays of length k
        for i in range(n - k + 1):
            sub = nums[i:i + k]
            seen = set(sub)              # each value counted only once per subarray
            for x in seen:
                count[x] += 1
        
        # Collect values that appear in exactly one subarray
        candidates = [x for x, c in count.items() if c == 1]
        return max(candidates) if candidates else -1
```
