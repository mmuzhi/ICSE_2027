```python
class Solution:
    def numberOfSubstrings(self, s: str, k: int) -> int:
        n = len(s)
        total = n * (n + 1) // 2
        counts = [0] * 26
        over = 0  # number of characters with count >= k
        left = 0
        bad = 0
        for right in range(n):
            idx = ord(s[right]) - ord('a')
            counts[idx] += 1
            if counts[idx] == k:
                over += 1
            # shrink window until no character count >= k
            while over > 0:
                left_idx = ord(s[left]) - ord('a')
                counts[left_idx] -= 1
                if counts[left_idx] == k - 1:
                    over -= 1
                left += 1
            # now window [left, right] has all counts < k
            bad += (right - left + 1)
        return total - bad
```

```python
class Solution:
    def numberOfSubstrings(self, s: str, k: int) -> int:
        n = len(s)
        total = n * (n + 1) // 2          # total number of substrings
        counts = [0] * 26                 # frequency of each letter in the window
        over = 0                          # how many letters have count >= k
        left = 0
        bad = 0                           # substrings with all counts < k

        for right in range(n):

            if counts[idx] == k:

            # shrink window until no letter appears k times
            while over > 0:
                left_idx = ord(s[left]) - 97
                counts[left_idx] -= 1
                if counts[left_idx] == k - 1:
                    over -= 1
                left += 1

            # all substrings ending at 'right' with start >= left are valid (bad)
            bad += right - left + 1

        return total - bad
```
