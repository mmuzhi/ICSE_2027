Input

N Q

Constraints


- 1 \leq N \leq 5 \times 10^5

- In queries of the first type, 1 \leq x \leq N.
- In queries of the first and second types, 1 \leq c \leq N.
- There is at least one query of the second type.
- All input values are integers.

1 4 2
2 2
1 3 2
1 2 3
2 3

3

    size = [1]*(N+1)
    left = list(range(N+1))
    right = list(range(N+1))
    color = list(range(N+1))  # color[i] for root i
    count = [0]*(N+1)
    for i in range(1, N+1):
        count[i] = 1

    def find(x):
        # iterative path compression
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    def union(a, b):
        # a and b are roots, assume same color
        if a == b:
            return a
        if size[a] < size[b]:
            a, b = b, a
        parent[b] = a
        size[a] += size[b]
        # update left and right
        if left[b] < left[a]:
            left[a] = left[b]
        if right[b] > right[a]:
            right[a] = right[b]
        # color[a] unchanged
        return a

    out_lines = []

        parts = input().split()
        if parts[0] == '1':
            x = int(parts[1])
            c = int(parts[2])
            root = find(x)
            old = color[root]
            if old == c:

            count[c] += size[root]
            # change color

            # merge left
            L = left[root]
            if L > 1:

                if color[left_root] == c:

            if R < N:

                if color[right_root] == c:
                    root = union(root, right_root)
        else:  # type 2
            c = int(parts[1])
            out_lines.append(str(count[c]))

    sys.stdout.write("\n".join(out_lines))

import sys

def solve():
    import sys
    input = sys.stdin.readline
    N, Q = map(int, input().split())
    parent = list(range(N+1))
    size = [1]*(N+1)
    left = list(range(N+1))
    right = list(range(N+1))
    color = list(range(N+1))
    count = [0]*(N+1)
    for i in range(1, N+1):
        count[i] = 1

    def find(x):
        # path compression iterative
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    def union(a, b):
        # a and b are roots, same color
        if size[a] < size[b]:
            a, b = b, a
        parent[b] = a
        size[a] += size[b]
        if left[b] < left[a]:
            left[a] = left[b]
        if right[b] > right[a]:
            right[a] = right[b]
        # color[a] unchanged
        return a

    out = []
    for _ in range(Q):
        parts = input().split()
        if parts[0] == '1':
            x = int(parts[1])
            c = int(parts[2])
            root = find(x)
            old = color[root]
            if old == c:
                continue
            # update counts
            count[old] -= size[root]
            count[c] += size[root]
            # change color
            color[root] = c
            # merge left
            L = left[root]
            if L > 1:
                left_root = find(L-1)
                if color[left_root] == c:
                    root = union(root, left_root)
            # merge right
            R = right[root]
            if R < N:
                right_root = find(R+1)
                if color[right_root] == c:
                    root = union(root, right_root)
        else:
            c = int(parts[1])
            out.append(str(count[c]))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
