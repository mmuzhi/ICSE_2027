=== ANSWER ===
```python

import sys
from collections import Counter

def main():
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    freq = Counter(A)          # frequency of each integer
    max_val = -1
    ans = -1

    for i, a in enumerate(A, start=1):
        if freq[a] == 1:       # a appears exactly once → satisfies condition
            if a > max_val:
                max_val = a
                ans = i

    print(ans)

if __name__ == "__main__":
    main()

```
