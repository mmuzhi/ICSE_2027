from typing import List

class Solution:
    def minimumIncrements(self, nums: List[int], target: List[int]) -> int:
        m = len(target)
        # precompute lcm for each mask
        lcm_mask = [1] * (1 << m)
        for mask in range(1, 1 << m):
            l = 1
            for i in range(m):
                if mask & (1 << i):
                    l = l * target[i] // gcd(l, target[i])
            lcm_mask[mask] = l

```python
from math import gcd
from typing import List

class Solution:
    def minimumIncrements(self, nums: List[int], target: List[int]) -> int:
        m = len(target)
        # precompute LCM for every subset of targets
        lcm_mask = [1] * (1 << m)
        for mask in range(1, 1 << m):
            l = 1
            for i in range(m):
                if mask & (1 << i):
                    l = l * target[i] // gcd(l, target[i])
            lcm_mask[mask] = l

        INF = 10**18
        dp = [INF] * (1 << m)
        dp[0] = 0

        for x in nums:
            # cost for this number to cover each subset
            cost = [0] * (1 << m)
            for mask in range(1, 1 << m):
                L = lcm_mask[mask]
                # smallest multiple of L that is >= x
                v = ((x + L - 1) // L) * L
                cost[mask] = v - x

            # 0/1 knapsack style DP over subsets
            new_dp = dp[:]   # option: do not use this number
            for mask in range(1, 1 << m):
                c = cost[mask]
                if c >= INF:
                    continue
                for prev in range(1 << m):
                    if dp[prev] + c < new_dp[prev | mask]:
                        new_dp[prev | mask] = dp[prev] + c
            dp = new_dp

        return dp[(1 << m) - 1]
```
