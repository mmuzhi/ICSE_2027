def main():

    ans2 = 0
    for y in Y:

        target = (y + R) % M
        ans2 += freq[target]
    print(ans1 + ans2)

import sys

def main():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    total = sum(A)
    R = total % M

    # Y[i] = (prefix sum of first i A's) mod M, i = 0..N-1
    Y = [0] * N
    cur = 0
    for i in range(N):
        Y[i] = cur
        if i < N - 1:
            cur = (cur + A[i]) % M

    # frequency of each residue
    freq = [0] * M
    for y in Y:
        freq[y] += 1

    # ans1 = number of unordered pairs with equal residues
    ans1 = 0
    for v in freq:
        if v > 1:
            ans1 += v * (v - 1) // 2

    # ans2 = number of unordered pairs (i<j) with (Y[j]-Y[i]) % M == R
    ans2 = 0
    for y in Y:
        freq[y] -= 1          # remove current index, now freq counts only later indices
        target = (y + R) % M
        ans2 += freq[target]

    print(ans1 + ans2)

if __name__ == "__main__":
    main()
