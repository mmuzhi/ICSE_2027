```python

import sys

def main():
    N = int(sys.stdin.readline().strip())
    if N % 2 == 1:
        # odd: one '=' in the middle
        left = '-' * (N // 2)
        print(left + '=' + left)
    else:
        # even: two '=' in the middle
        left = '-' * (N // 2 - 1)
        print(left + '==' + left)

if __name__ == "__main__":
    main()
```
