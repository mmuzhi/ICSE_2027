MOD = 998244353

def precompute_fact(max_n):
    fact = [1]*(max_n+1)
    for i in range(1, max_n+1):
        fact[i] = fact[i-1]*i % MOD
    invfact = [1]*(max_n+1)
    invfact[max_n] = pow(fact[max_n], MOD-2, MOD)
    for i in range(max_n,0,-1):
        invfact[i-1] = invfact[i]*i % MOD
    return fact, invfact

def nCr(n, r, fact, invfact):
    if r < 0 or r > n:
        return 0
    return fact[n] * invfact[r] % MOD * invfact[n-r] % MOD

def T(A, B, fact, invfact):
    if A < 0 or B < 0:
        return 0
    n = A + B + 4
    r = A + 2
    val = nCr(n, r, fact, invfact) - (A + B + 4)
    val %= MOD
    return val
