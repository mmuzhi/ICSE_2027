Input

A_1 A_2 A_3 A_4 A_5

Yes

Sample Input 3

1 2 3 4 5

No

You must perform exactly one operation.

Yes

=== ANSWER ===
```python

import sys

def main() -> None:
    A = list(map(int, sys.stdin.read().split()))
    target = [1, 2, 3, 4, 5]

    # Already sorted -> need 0 swaps, not allowed
    if A == target:
        print("No")
        return

    # Try swapping each adjacent pair
    for i in range(4):
        B = A.copy()
        B[i], B[i+1] = B[i+1], B[i]
        if B == target:
            print("Yes")
            return

    # No single adjacent swap yields the sorted order
    print("No")

if __name__ == "__main__":
    main()
```
