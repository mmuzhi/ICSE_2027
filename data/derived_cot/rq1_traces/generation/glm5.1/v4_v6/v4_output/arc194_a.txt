    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    NEG = -10**18
    best_odd = NEG
    best_even = NEG
    for i, val in enumerate(A):
        idx = i+1
        if idx % 2 == 1:  # odd
            # start new
            cand_start = val
            # extend from even
            cand_extend = best_even + val if best_even != NEG else NEG
            best_odd = max(best_odd, cand_start, cand_extend)

            best_even = max(best_even, cand_extend)
    if N % 2 == 0:
        ans = max(0, best_even)
    else:
        ans = best_odd

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    NEG = -10**18          # sufficiently small to act as -infinity
    best_odd = NEG         # maximum sum of a valid subsequence ending with an odd index
    best_even = NEG        # maximum sum of a valid subsequence ending with an even index

    for i, val in enumerate(A):
        idx = i + 1        # 1‑based index
        if idx % 2 == 1:   # odd index
            # start a new subsequence with this element
            cand_start = val
            # extend a subsequence that ended with an even index
            cand_extend = best_even + val if best_even != NEG else NEG
            best_odd = max(best_odd, cand_start, cand_extend)
        else:              # even index
            # can only extend from a subsequence ending with an odd index
            cand_extend = best_odd + val if best_odd != NEG else NEG
            best_even = max(best_even, cand_extend)

    if N % 2 == 0:
        # empty stack (sum 0) is always achievable for even N
        ans = max(0, best_even)
    else:
        ans = best_odd

    print(ans)

if __name__ == "__main__":
    solve()
