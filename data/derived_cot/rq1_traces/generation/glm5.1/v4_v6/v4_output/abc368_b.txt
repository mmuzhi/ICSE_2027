=== ANSWER ===
```python

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    ans = 0

    while True:
        # count positive elements
        pos = sum(1 for x in A if x > 0)
        if pos <= 1:
            break

        # sort descending and decrement the two largest
        A.sort(reverse=True)
        A[0] -= 1
        A[1] -= 1
        ans += 1

    print(ans)

if __name__ == "__main__":
    solve()
```
