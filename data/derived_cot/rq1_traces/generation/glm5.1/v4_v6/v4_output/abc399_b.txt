=== ANSWER ===
```python

import sys

def main():
    input = sys.stdin.readline
    N = int(input())
    P = list(map(int, input().split()))

    # indices sorted by score descending
    idx = list(range(N))
    idx.sort(key=lambda i: P[i], reverse=True)

    rank = [0] * N
    r = 1
    i = 0
    while i < N:
        # find group with same score
        j = i
        cur_score = P[idx[i]]
        while j < N and P[idx[j]] == cur_score:
            j += 1
        k = j - i
        # assign current rank r to the whole group
        for t in range(i, j):
            rank[idx[t]] = r
        # next rank
        r += k
        i = j

    print('\n'.join(map(str, rank)))

if __name__ == "__main__":
    main()
```
