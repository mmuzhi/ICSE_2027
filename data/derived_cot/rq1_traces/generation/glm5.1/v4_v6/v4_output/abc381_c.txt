A string T is called an 11/22 string when it satisfies all of the following conditions:

- |T| is odd. Here, |T| denotes the length of T.

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    S = input().strip()

    # left[i] = length of consecutive '1's ending at i
    left = [0] * N
    cnt = 0
    for i, ch in enumerate(S):
        if ch == '1':
            cnt += 1
            left[i] = cnt
        else:
            cnt = 0
            left[i] = 0

    # right[i] = length of consecutive '2's starting at i
    right = [0] * N
    cnt = 0
    for i in range(N - 1, -1, -1):
        if S[i] == '2':
            cnt += 1
            right[i] = cnt
        else:
            cnt = 0
            right[i] = 0

    ans = 0
    for i, ch in enumerate(S):
        if ch == '/':
            L = left[i - 1] if i > 0 else 0
            R = right[i + 1] if i + 1 < N else 0
            k = min(L, R)
            length = 2 * k + 1
            if length > ans:
                ans = length

    print(ans)
