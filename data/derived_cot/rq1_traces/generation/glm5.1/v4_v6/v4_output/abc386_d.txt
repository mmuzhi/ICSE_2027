You will be given a competitive programming problem.
There is an N \times N grid. Takahashi wants to color each cell black or white so that all of the following conditions are satisfied:

- For every row, the following condition holds:

- There exists an integer i\ (0\leq i\leq N) such that the leftmost i cells are colored black, and the rest are colored white.

- For every column, the following condition holds:

- There exists an integer i\ (0\leq i\leq N) such that the topmost i cells are colored black, and the rest are colored white.


Out of these N^2 cells, M of them have already been colored. Among them, the i-th one is at the X_i-th row from the top and the Y_i-th column from the left, and it is colored black if C_i is B and white if C_i is W.
Determine whether he can color the remaining uncolored N^2 - M cells so that all the conditions are satisfied.

Input

The input is given from Standard Input in the following format:
N M
X_1 Y_1 C_1
\vdots
X_M Y_M C_M

Output

If it is possible to satisfy the conditions, print Yes; otherwise, print No.

Constraints


- 1\leq N\leq 10^9
- 1\leq M\leq \min(N^2,2\times 10^5)
- 1\leq X_i,Y_i\leq N
- (X_i,Y_i)\neq (X_j,Y_j)\ (i\neq j)
- C_i is B or W.
- All input numbers are integers.

def main():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    row_bounds = {}
    for _ in range(M):
        x = int(next(it))
        y = int(next(it))
        c = next(it).decode() if isinstance(next(it), bytes) else next(it) # careful: data are bytes, so c is bytes like b'B'. We can decode or compare to b'B'.
        Actually, when using split() on bytes, we get list of bytes. So we can compare c == b'B'.
        So:
        x = int(next(it)); y = int(next(it)); c = next(it)
        if c == b'B':
            if x not in row_bounds:
                row_bounds[x] = [0, N]
            row_bounds[x][0] = max(row_bounds[x][0], y)
        else: # b'W'
            if x not in row_bounds:
                row_bounds[x] = [0, N]
            row_bounds[x][1] = min(row_bounds[x][1], y-1)
    # Check per-row consistency
    for L, U in row_bounds.values():
        if L > U:
            print("No")
            return
    # Sort rows
    rows = sorted(row_bounds.items())  # list of (row, [L,U])
    if not rows:
        print("Yes")
        return
    # Compute prefix min U
    prefix_min_U = []
    cur_min = N
    for row, (L, U) in rows:
        cur_min = min(cur_min, U)
        prefix_min_U.append(cur_min)
    # Compute suffix max L
    suffix_max_L = [0]*len(rows)
    cur_max = 0
    for i in range(len(rows)-1, -1, -1):
        L = rows[i][1][0]
        cur_max = max(cur_max, L)
        suffix_max_L[i] = cur_max
    # Check
    for i in range(len(rows)):
        if suffix_max_L[i] > prefix_min_U[i]:
            print("No")
            return
    print("Yes")

import sys

def main():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))

    # row_bounds[x] = [L, U]  (L = lower bound, U = upper bound)
    row_bounds = {}

    for _ in range(M):
        x = int(next(it))
        y = int(next(it))
        c = next(it)                     # b'B' or b'W'
        if x not in row_bounds:
            row_bounds[x] = [0, N]       # default: L=0, U=N
        if c == b'B':
            row_bounds[x][0] = max(row_bounds[x][0], y)
        else:  # b'W'
            row_bounds[x][1] = min(row_bounds[x][1], y - 1)

    # check per‑row consistency
    for L, U in row_bounds.values():
        if L > U:
            print("No")
            return

    # no constraints -> always possible
    if not row_bounds:
        print("Yes")
        return

    # sort rows by index
    rows = sorted(row_bounds.items())   # list of (row, [L, U])

    # prefix minimum of U
    pref_min_U = []
    cur_min = N
    for _, (_, U) in rows:
        cur_min = min(cur_min, U)
        pref_min_U.append(cur_min)

    # suffix maximum of L
    suff_max_L = [0] * len(rows)
    cur_max = 0
    for i in range(len(rows) - 1, -1, -1):
        L = rows[i][1][0]
        cur_max = max(cur_max, L)
        suff_max_L[i] = cur_max

    # check condition for every constrained row
    for i in range(len(rows)):
        if suff_max_L[i] > pref_min_U[i]:
            print("No")
            return

    print("Yes")

if __name__ == "__main__":
    main()
