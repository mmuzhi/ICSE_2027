class SegTree:
    def __init__(self, arr):
        self.n = len(arr)
        self.size = 1
        while self.size < self.n:
            self.size <<= 1
        self.NEG = -10**18
        self.tree = [(0, self.NEG, self.NEG, self.NEG)] * (2 * self.size)
        # set leaves
        for i in range(self.n):
            self.tree[self.size + i] = (arr[i], arr[i], arr[i], arr[i])
        # build
        for i in range(self.size - 1, 0, -1):
            self.tree[i] = self._combine(self.tree[2*i], self.tree[2*i+1])
    
    def _combine(self, left, right):
        total = left[0] + right[0]
        pref = max(left[1], left[0] + right[1])
        suff = max(right[2], right[0] + left[2])
        sub = max(left[3], right[3], left[2] + right[1])
        return (total, pref, suff, sub)
    
    def query(self, l, r):
        l += self.size
        r += self.size
        left_res = (0, self.NEG, self.NEG, self.NEG)
        right_res = (0, self.NEG, self.NEG, self.NEG)
        while l <= r:
            if l & 1:
                left_res = self._combine(left_res, self.tree[l])
                l += 1
            if not (r & 1):
                right_res = self._combine(self.tree[r], right_res)
                r -= 1
            l >>= 1
            r >>= 1
        return self._combine(left_res, right_res)

class Solution:
    def maxSubarraySum(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 0:
            return 0
        
        NEG = -10**18
        
        # Segment tree
        size = 1
        while size < n:
            size <<= 1
        tree = [(0, NEG, NEG, NEG)] * (2 * size)
        # set leaves
        for i in range(n):
            val = nums[i]
            tree[size + i] = (val, val, val, val)
        # build
        for i in range(size - 1, 0, -1):
            left = tree[2*i]
            right = tree[2*i+1]
            total = left[0] + right[0]
            pref = max(left[1], left[0] + right[1])
            suff = max(right[2], right[0] + left[2])
            sub = max(left[3], right[3], left[2] + right[1])
            tree[i] = (total, pref, suff, sub)
        
        def query(l, r):
            l += size
            r += size
            left_res = (0, NEG, NEG, NEG)
            right_res = (0, NEG, NEG, NEG)
            while l <= r:
                if l & 1:
                    # combine left_res and tree[l]
                    left = left_res
                    right = tree[l]
                    total = left[0] + right[0]
                    pref = max(left[1], left[0] + right[1])
                    suff = max(right[2], right[0] + left[2])
                    sub = max(left[3], right[3], left[2] + right[1])
                    left_res = (total, pref, suff, sub)
                    l += 1
                if not (r & 1):
                    # combine tree[r] and right_res
                    left = tree[r]
                    right = right_res
                    total = left[0] + right[0]
                    pref = max(left[1], left[0] + right[1])
                    suff = max(right[2], right[0] + left[2])
                    sub = max(left[3], right[3], left[2] + right[1])
                    right_res = (total, pref, suff, sub)
                    r -= 1
                l >>= 1
                r >>= 1
            # combine left_res and right_res
            left = left_res
            right = right_res
            total = left[0] + right[0]
            pref = max(left[1], left[0] + right[1])
            suff = max(right[2], right[0] + left[2])
            sub = max(left[3], right[3], left[2] + right[1])
            return (total, pref, suff, sub)
        
        # original max subarray sum
        original = query(0, n-1)[3]
        ans = original
        
        # build position dictionary
        from collections import defaultdict
        pos_dict = defaultdict(list)
        for i, val in enumerate(nums):
            pos_dict[val].append(i)
        
        # iterate over each distinct value
        for val, positions in pos_dict.items():
            if len(positions) == n:
                continue  # removal would make array empty
            # build segments
            segments = []
            prev = -1
            for p in positions:
                L = prev + 1
                R = p - 1
                if L <= R:
                    segments.append((L, R))
                prev = p
            L = prev + 1
            R = n - 1
            if L <= R:
                segments.append((L, R))
            
            # compute M(val) by combining segments
            best_overall = NEG
            best_suffix = NEG
            for (L, R) in segments:
                total, pref, suff, sub = query(L, R)
                # update best_overall with sub
                if sub > best_overall:
                    best_overall = sub
                if best_suffix != NEG:
                    cand = best_suffix + pref
                    if cand > best_overall:
                        best_overall = cand
                # update best_suffix
                if best_suffix == NEG:
                    best_suffix = suff
                else:
                    best_suffix = max(suff, best_suffix + total)
            # best_overall is M(val)
            if best_overall > ans:
                ans = best_overall
        
        return ans
