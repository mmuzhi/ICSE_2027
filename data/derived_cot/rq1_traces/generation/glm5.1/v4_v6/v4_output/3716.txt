Your task is to find the length of the longest subsequence seq of nums, such that the absolute differences between consecutive elements form a non-increasing sequence of integers. A subsequence is an non-empty array that can be derived from another array by deleting some or no elements without changing the order of the remaining elements.

            if best + 1 > cand:
                cand = best + 1
        best_for_d[d] = cand

    for d in range(MAX_DIFF+1):

            cur = dp_x[d]

```python
from typing import List

class Solution:
    def longestSubsequence(self, nums: List[int]) -> int:
        MAX_VAL = 300
        MAX_DIFF = 300
        dp = [ [0]*(MAX_DIFF+1) for _ in range(MAX_VAL+1) ]
        suffix_max = [ [0]*(MAX_DIFF+1) for _ in range(MAX_VAL+1) ]
        dp_inf = [0]*(MAX_VAL+1)
        ans = 0
        
        for x in nums:
            dp_x = dp[x]
            # compute candidates and update dp_x
            for d in range(MAX_DIFF+1):
                cand = 0
                v = x - d
                if 1 <= v <= MAX_VAL:
                    best = suffix_max[v][d]
                    if best + 1 > cand:
                        cand = best + 1
                v = x + d
                if 1 <= v <= MAX_VAL:
                    best = suffix_max[v][d]
                    if best + 1 > cand:
                        cand = best + 1
                if cand > dp_x[d]:
                    dp_x[d] = cand
                    if cand > ans:
                        ans = cand
            
            # set length-1 subsequence
            dp_inf[x] = 1
            
            # recompute suffix_max for x
            cur = dp_inf[x]
            sm_x = suffix_max[x]
            for d in range(MAX_DIFF, -1, -1):
                if dp_x[d] > cur:
                    cur = dp_x[d]
                sm_x[d] = cur
        
        return ans
```

```python
from typing import List

class Solution:
    def longestSubsequence(self, nums: List[int]) -> int:
        MAX_VAL = 300
        MAX_DIFF = 300
        dp = [ [0]*(MAX_DIFF+1) for _ in range(MAX_VAL+1) ]
        suffix_max = [ [0]*(MAX_DIFF+1) for _ in range(MAX_VAL+1) ]
        dp_inf = [0]*(MAX_VAL+1)
        ans = 0
        
        for x in nums:
            dp_x = dp[x]
            # compute candidates for each possible last diff d
            for d in range(MAX_DIFF+1):
                cand = 0
                v = x - d
                if 1 <= v <= MAX_VAL:
                    best = suffix_max[v][d]
                    if best + 1 > cand:
                        cand = best + 1
                v = x + d
                if 1 <= v <= MAX_VAL:
                    best = suffix_max[v][d]
                    if best + 1 > cand:
                        cand = best + 1
                if cand > dp_x[d]:
                    dp_x[d] = cand
                    if cand > ans:
                        ans = cand
            
            # mark that we have a length-1 subsequence ending at x
            dp_inf[x] = 1
            if ans < 1:
                ans = 1
            
            # recompute suffix max for value x
            cur = dp_inf[x]
            sm_x = suffix_max[x]
            for d in range(MAX_DIFF, -1, -1):
                if dp_x[d] > cur:
                    cur = dp_x[d]
                sm_x[d] = cur
        
        return ans
```
