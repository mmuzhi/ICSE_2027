    n = len(nums)
    if is_min:
        # compute left (prev_le) and right (next_less)
        left = [-1]*n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] > nums[i]:
                stack.pop()
            left[i] = stack[-1] if stack else -1
            stack.append(i)
        right = [n]*n
        stack = []
        for i in range(n-1, -1, -1):
            while stack and nums[stack[-1]] >= nums[i]:
                stack.pop()
            right[i] = stack[-1] if stack else n

def minMaxSubarraySum(self, nums: List[int], k: int) -> int:
    n = len(nums)
    # helper to compute sum of extremes
    def compute(is_min):
        # compute left and right boundaries
        left = [-1]*n
        right = [n]*n
        if is_min:
            # previous less or equal (for min)
            stack = []
            for i in range(n):
                while stack and nums[stack[-1]] > nums[i]:
                    stack.pop()
                left[i] = stack[-1] if stack else -1
                stack.append(i)
            stack = []
            for i in range(n-1, -1, -1):
                while stack and nums[stack[-1]] >= nums[i]:
                    stack.pop()
                right[i] = stack[-1] if stack else n
                stack.append(i)
        else:

            stack = []

            if boundary < s0:
