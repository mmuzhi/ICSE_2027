def answerString(self, word: str, numFriends: int) -> str:
    n = len(word)
    if numFriends == 1:
        return word
    L = n - numFriends + 1
    best = ""
    for i in range(n):
        cur = word[i:i+min(L, n-i)]
        if cur > best:
            best = cur
    return best

class Solution:
    def answerString(self, word: str, numFriends: int) -> str:
        n = len(word)
        if numFriends == 1:
            return word
        L = n - numFriends + 1
        best = ""
        for i in range(n):
            cur = word[i:i+min(L, n-i)]
            if cur > best:
                best = cur
        return best

class Solution:
    def answerString(self, word: str, numFriends: int) -> str:
        n = len(word)
        # If only one friend, the only possible piece is the whole word.
        if numFriends == 1:
            return word

        # Maximum possible length of a piece.
        max_len = n - numFriends + 1

        best = ""
        for i in range(n):
            # Longest feasible substring starting at i.
            cur_len = min(max_len, n - i)
            candidate = word[i:i + cur_len]
            if candidate > best:
                best = candidate
        return best
