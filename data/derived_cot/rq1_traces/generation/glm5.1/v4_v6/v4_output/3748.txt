Explanation:
Diagonals with exactly one element are already in order, so no changes are needed.

```python
from typing import List

class Solution:
    def sortMatrix(self, grid: List[List[int]]) -> List[List[int]]:
        n = len(grid)
        # Iterate over all diagonals identified by d = i - j
        for d in range(-(n-1), n):  # d from -(n-1) to n-1 inclusive
            start_i = max(0, d)
            end_i = min(n-1, n-1 + d)
            # Collect values on this diagonal
            vals = []
            for i in range(start_i, end_i + 1):
                j = i - d
                vals.append(grid[i][j])
            # Sort according to the region
            if d >= 0:          # bottom-left triangle (including main diagonal)
                vals.sort(reverse=True)   # non-increasing
            else:              # top-right triangle
                vals.sort()               # non-decreasing
            # Write back the sorted values
            idx = 0
            for i in range(start_i, end_i + 1):
                j = i - d
                grid[i][j] = vals[idx]
                idx += 1
        return grid
```
