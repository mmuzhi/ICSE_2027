import sys

def solve() -> None:
    S = sys.stdin.readline().strip()
    # positions for each uppercase letter (0‑based indices)
    pos = [[] for _ in range(26)]
    for idx, ch in enumerate(S):
        pos[ord(ch) - 65].append(idx)

    ans = 0
    for lst in pos:
        m = len(lst)
        if m < 2:
            continue

        # compute sum of (p_j - p_i) for all i < j
        sum_diff = 0
        prefix = 0          # sum of p_0 ... p_{i-1}
        for i, p in enumerate(lst):
            if i == 0:
                prefix = p
                continue
            # i >= 1
            sum_diff += i * p - prefix
            prefix += p

        total_pairs = m * (m - 1) // 2
        ans += sum_diff - total_pairs   # each pair contributes (p_j - p_i - 1)

    print(ans)

if __name__ == "__main__":
    solve()
