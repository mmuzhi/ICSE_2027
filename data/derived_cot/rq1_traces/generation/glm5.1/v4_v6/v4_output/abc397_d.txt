import sys
import math

def integer_cuberoot(n):
    # returns floor(cuberoot(n)) for n >= 0
    lo, hi = 0, 10**6+1  # since n <= 10**18+1, cube root <= 10**6
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if mid * mid * mid <= n:
            lo = mid
        else:
            hi = mid - 1
    return lo

def solve():
    N = int(sys.stdin.readline().strip())
    # compute max possible d = x-y
    max_d = integer_cuberoot(N + 1) - 1
    if max_d < 1:
        print(-1)
        return
    for d in range(1, max_d + 1):
        if N % d != 0:
            continue
        M = N // d
        D = 12 * M - 3 * d * d
        if D < 0:
            continue
        s = math.isqrt(D)
        if s * s != D:
            continue
        num = -3 * d + s
        if num % 6 != 0:
            continue
        y = num // 6
        if y <= 0:
            continue
        x = y + d
        # optional verification: if x**3 - y**3 != N: continue (should not happen)
        print(x, y)
        return
    print(-1)

if __name__ == "__main__":
    solve()

import sys
import math

def integer_cuberoot(n: int) -> int:
    """floor(cuberoot(n)) for n >= 0, n <= 10**18+1"""
    lo, hi = 0, 10**6 + 1          # (10**6+1)^3 > 10**18+1
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if mid * mid * mid <= n:
            lo = mid
        else:
            hi = mid - 1
    return lo

def solve() -> None:
    N = int(sys.stdin.readline().strip())

    # maximal possible difference d = x - y
    max_d = integer_cuberoot(N + 1) - 1
    if max_d < 1:
        print(-1)
        return

    for d in range(1, max_d + 1):
        if N % d != 0:
            continue
        M = N // d                     # = x^2 + xy + y^2
        D = 12 * M - 3 * d * d         # discriminant
        if D < 0:
            continue
        s = math.isqrt(D)
        if s * s != D:
            continue
        num = -3 * d + s
        if num % 6 != 0:
            continue
        y = num // 6
        if y <= 0:
            continue
        x = y + d
        # verification is optional, the derivation guarantees correctness
        print(x, y)
        return

    print(-1)

if __name__ == "__main__":
    solve()
