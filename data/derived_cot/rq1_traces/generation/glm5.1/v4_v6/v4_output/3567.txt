=== ANSWER ===
```python

class Solution:
    def convertDateToBinary(self, date: str) -> str:
        # Split the input string into year, month, day
        year, month, day = date.split('-')
        # Convert each part to integer and then to binary without '0b' prefix
        bin_year = bin(int(year))[2:]
        bin_month = bin(int(month))[2:]
        bin_day = bin(int(day))[2:]
        # Join with hyphens
        return f"{bin_year}-{bin_month}-{bin_day}"

```
