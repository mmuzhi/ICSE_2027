Now, we need to implement the trie and the computations efficiently.

Implementation details:

- We'll have global variables: n = len(words), k.
- We'll create lists: depth = [], cnt = [], children = [] (list of dicts). We'll also store for each node: maybe an id.

- Insert each word, building path_nodes list for each word.

Pseudo:

class TrieNode:
    def __init__(self, depth):
        self.depth = depth
        self.cnt = 0
        self.children = {}

But using separate lists may be faster.

We'll do:

depth = [0]  # root depth 0
cnt = [0]    # root count 0
children = [{}]  # root children

path_nodes = []  # list of lists

for i, word in enumerate(words):
    node = 0
    cnt[0] += 1
    path = [0]
    for ch in word:
        if ch not in children[node]:
            children[node][ch] = len(depth)
            depth.append(depth[node] + 1)
            cnt.append(0)
            children.append({})
        node = children[node][ch]
        cnt[node] += 1
        path.append(node)
    path_nodes.append(path)

After insertion, we have total nodes = len(depth).

Now compute good_max_depth_in_subtree and child best values.

We'll do a post-order traversal. Since the trie is a tree, we can use stack.

We'll create arrays:
good_max = [-1] * N  # N = number of nodes
best1_val = [-1] * N
best1_child = [-1] * N  # store child node index
best2_val = [-1] * N

We'll do iterative DFS:

stack = [(0, 0)]  # (node, state) state 0 = pre, 1 = post
while stack:
    node, state = stack.pop()
    if state == 0:
        stack.append((node, 1))
        for child in children[node].values():
            stack.append((child, 0))
    else:
        # post-order: compute good_max for node
        self_good = depth[node] if cnt[node] >= k else -1
        max_child_good = -1
        # also compute best1, best2 from children's good_max
        b1 = -1
        b2 = -1
        b1_child = -1
        for child in children[node].values():
            val = good_max[child]
            if val > b1:
                b2 = b1
                b1 = val
                b1_child = child
            elif val > b2:
                b2 = val
            # also update max_child_good? Actually good_max[node] = max(self_good, max over children good_max)
            if val > max_child_good:
                max_child_good = val
        good_max[node] = max(self_good, max_child_good)
        best1_val[node] = b1
        best1_child[node] = b1_child
        best2_val[node] = b2

Note: For leaf nodes (no children), the loop does nothing, b1,b2 remain -1, good_max = self_good.

Now we have these arrays.

Now for each word i, compute answer.

We'll have answer list of length n.

For each i in range(n):
    path = path_nodes[i]
    best_on_high = -1
    for v in path:
        if depth[v] >= 1 and cnt[v] >= k+1:
            if depth[v] > best_on_high:
                best_on_high = depth[v]
    best_not = -1
    for idx, v in enumerate(path):
        # determine next node
        if idx + 1 < len(path):
            next_node = path[idx+1]
        else:
            next_node = None
        if children[v]:  # has children
            if next_node is not None:
                if best1_child[v] == next_node:
                    cand = best2_val[v]
                else:
                    cand = best1_val[v]
            else:
                cand = best1_val[v]
        else:
            cand = -1
        if cand > best_not:
            best_not = cand
    ans = max(best_on_high, best_not)
    if ans < 0:
        ans = 0
    answer[i] = ans

Return answer.

def longestCommonPrefix(self, words: List[str], k: int) -> List[int]:
    n = len(words)
    # Build trie
    depth = [0]  # root depth 0
    cnt = [0]    # root count
    children = [{}]  # root children
    path_nodes = []  # list of lists

    for word in words:
        node = 0
        cnt[0] += 1
        path = [0]
        for ch in word:
            if ch not in children[node]:
                children[node][ch] = len(depth)
                depth.append(depth[node] + 1)
                cnt.append(0)
                children.append({})
            node = children[node][ch]
            cnt[node] += 1
            path.append(node)
        path_nodes.append(path)

    N = len(depth)
    good_max = [-1] * N
    best1_val = [-1] * N
    best1_child = [-1] * N
    best2_val = [-1] * N

    # Post-order traversal
    stack = [(0, 0)]  # (node, state) 0=pre,1=post
    while stack:
        node, state = stack.pop()
        if state == 0:
            stack.append((node, 1))
            for child in children[node].values():
                stack.append((child, 0))
        else:
            # compute good_max for node
            self_good = depth[node] if cnt[node] >= k else -1
            max_child_good = -1
            b1 = -1
            b2 = -1
            b1_child = -1
            for child in children[node].values():
                val = good_max[child]
                if val > max_child_good:
                    max_child_good = val
                if val > b1:
                    b2 = b1
                    b1 = val
                    b1_child = child
                elif val > b2:
                    b2 = val
            good_max[node] = max(self_good, max_child_good)
            best1_val[node] = b1
            best1_child[node] = b1_child
            best2_val[node] = b2

    # Compute answers
    ans = [0] * n
    for i in range(n):
        path = path_nodes[i]
        # best_on_path_high
        best_high = -1
        for v in path:
            if depth[v] >= 1 and cnt[v] >= k+1:
                if depth[v] > best_high:
                    best_high = depth[v]
        # best_not_on_path
        best_not = -1
        for idx, v in enumerate(path):
            if idx + 1 < len(path):
                next_node = path[idx+1]
            else:
                next_node = None
            if children[v]:  # has children
                if next_node is not None:
                    if best1_child[v] == next_node:
                        cand = best2_val[v]
                    else:
                        cand = best1_val[v]
                else:
                    cand = best1_val[v]
            else:
                cand = -1
            if cand > best_not:
                best_not = cand
        cur = max(best_high, best_not)
        if cur < 0:
            cur = 0
        ans[i] = cur
    return ans
