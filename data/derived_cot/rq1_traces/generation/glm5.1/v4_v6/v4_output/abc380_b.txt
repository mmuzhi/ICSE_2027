=== ANSWER ===
```python

def main():
    S = input().strip()
    # Split by '|' and take lengths of non‑empty parts
    A = [len(part) for part in S.split('|') if part]
    print(' '.join(map(str, A)))

if __name__ == "__main__":
    main()
```
