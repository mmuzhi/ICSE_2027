def main():
    import sys
    MOD = 998244353
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    maxA = max(A) if A else 0
    # compute phi up to maxA
    phi = list(range(maxA+1))
    for i in range(2, maxA+1):
        if phi[i] == i:  # prime
            for j in range(i, maxA+1, i):
                phi[j] -= phi[j] // i
    # compute divisors
    divs = [[] for _ in range(maxA+1)]
    for d in range(1, maxA+1):
        for m in range(d, maxA+1, d):
            divs[m].append(d)
    W = [0]*(maxA+1)
    S = 0
    pow2 = 1
    ans = []
    for x in A:
        F = 0
        for d in divs[x]:

        F %= MOD
        S = (S*2 + F) % MOD
        ans.append(S)
        pow2 = (pow2 * 2) % MOD
    sys.stdout.write('\n'.join(map(str, ans)))
