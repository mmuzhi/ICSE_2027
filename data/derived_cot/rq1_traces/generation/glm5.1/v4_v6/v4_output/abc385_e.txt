- Choose positive integers x,y.
- Prepare one vertex.
- Prepare x more vertices, and connect each of them to the vertex prepared in step 2.
- For each of the x vertices prepared in step 3, attach y leaves to it.

You are given a tree T with N vertices. The vertices are numbered 1 to N, and the i-th edge (i=1,2,\dots,N-1) connects vertices u_i and v_i.
Consider deleting zero or more vertices of T and the edges adjacent to them so that the remaining graph becomes a single Snowflake Tree. Find the minimum number of vertices that must be deleted. Under the constraints of this problem, it is always possible to transform T into a Snowflake Tree.

import sys

def main():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    n = int(next(it))
    adj = [[] for _ in range(n+1)]
    deg = [0]*(n+1)
    for _ in range(n-1):
        u = int(next(it)); v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
        deg[u] += 1
        deg[v] += 1

    max_kept = 0
    for c in range(1, n+1):
        d = deg[c]
        if d == 0:
            continue
        # compute a_i = deg[neighbor] - 1
        A = [deg[v]-1 for v in adj[c]]
        A.sort()
        best = 0
        # iterate from largest a to smallest, break when min becomes 0
        for idx in range(d-1, -1, -1):
            min_val = A[idx]
            if min_val == 0:
                break
            t = d - idx  # number of selected neighbors
            cand = t * (min_val + 1)
            if cand > best:
                best = cand
        total = 1 + best
        if total > max_kept:
            max_kept = total

    print(n - max_kept)

if __name__ == "__main__":
    main()

import sys

def main():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    n = int(next(it))

    adj = [[] for _ in range(n + 1)]
    deg = [0] * (n + 1)

    for _ in range(n - 1):
        u = int(next(it))
        v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
        deg[u] += 1
        deg[v] += 1

    max_kept = 0

    for c in range(1, n + 1):
        d = deg[c]
        if d == 0:
            continue

        # a_i = number of possible leaves for neighbour i (excluding c)
        A = [deg[v] - 1 for v in adj[c]]
        A.sort()                     # increasing order

        best = 0
        # iterate from largest a_i downwards, stop when min becomes 0
        for idx in range(d - 1, -1, -1):
            min_val = A[idx]
            if min_val == 0:
                break
            t = d - idx               # number of selected neighbours
            cand = t * (min_val + 1)  # x * (1 + y)
            if cand > best:
                best = cand

        total = 1 + best              # +1 for the centre
        if total > max_kept:
            max_kept = total

    print(n - max_kept)

if __name__ == "__main__":
    main()
