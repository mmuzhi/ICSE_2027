class Solution:
    def stringHash(self, s: str, k: int) -> str:
        n = len(s)
        result = []
        for i in range(0, n, k):
            sub = s[i:i+k]

            hashed = total % 26
            result.append(chr(ord('a') + hashed))

=== ANSWER ===
```python

class Solution:
    def stringHash(self, s: str, k: int) -> str:
        result = []
        # Iterate over the string in steps of k
        for i in range(0, len(s), k):
            substring = s[i:i+k]
            # Compute sum of hash values (a->0, b->1, ...)
            total = sum(ord(ch) - ord('a') for ch in substring)
            # Remainder modulo 26 gives the index of the result character
            hashed_char = total % 26
            # Convert index back to character and append
            result.append(chr(ord('a') + hashed_char))
        return ''.join(result)

```
