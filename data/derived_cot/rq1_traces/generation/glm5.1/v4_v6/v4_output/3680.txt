class DSU:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0]*n  # or size
    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]
            x = self.parent[x]
        return x
    def union(self, x, y):
        xr = self.find(x)
        yr = self.find(y)
        if xr == yr:
            return False
        if self.rank[xr] < self.rank[yr]:
            xr, yr = yr, xr
        self.parent[yr] = xr
        if self.rank[xr] == self.rank[yr]:
            self.rank[xr] += 1
        return True

But we might want to use size heuristics for potential speed, but rank is fine.

from typing import List

class Solution:
    def countComponents(self, nums: List[int], threshold: int) -> int:
        n = len(nums)
        parent = list(range(n))
        size = [1] * n
        
        def find(x):
            while parent[x] != x:
                parent[x] = parent[parent[x]]
                x = parent[x]
            return x
        
        def union(x, y):
            rx = find(x)
            ry = find(y)
            if rx == ry:
                return False
            if size[rx] < size[ry]:
                rx, ry = ry, rx
            parent[ry] = rx
            size[rx] += size[ry]
            return True
        
        rep = [-1] * (threshold + 1)
        comp = n
        
        for idx, val in enumerate(nums):
            if val <= threshold:
                for m in range(val, threshold + 1, val):
                    r = rep[m]
                    if r == -1:
                        rep[m] = idx
                    else:
                        if union(idx, r):
                            comp -= 1
        
        return comp
