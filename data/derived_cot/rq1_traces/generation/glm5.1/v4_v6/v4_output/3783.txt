Given two integers, n and k, an alternating permutation is a permutation of the first n positive integers such that no two adjacent elements are both odd or both even.
Return the k-th alternating permutation sorted in lexicographical order. If there are fewer than k valid alternating permutations, return an empty list.
 
Example 1:

Input: n = 4, k = 6
Output: [3,4,1,2]
Explanation:
The lexicographically-sorted alternating permutations of [1, 2, 3, 4] are:

[1, 2, 3, 4]
[1, 4, 3, 2]
[2, 1, 4, 3]
[2, 3, 4, 1]
[3, 2, 1, 4]
[3, 4, 1, 2] ← 6th permutation
[4, 1, 2, 3]
[4, 3, 2, 1]

Since k = 6, we return [3, 4, 1, 2].

Example 2:

Input: n = 3, k = 2
Output: [3,2,1]
Explanation:
The lexicographically-sorted alternating permutations of [1, 2, 3] are:

[1, 2, 3]
[3, 2, 1] ← 2nd permutation

Since k = 2, we return [3, 2, 1].

Example 3:

Input: n = 2, k = 3
Output: []
Explanation:
The lexicographically-sorted alternating permutations of [1, 2] are:

[1, 2]
[2, 1]

There are only 2 alternating permutations, but k = 3, which is out of range. Thus, we return an empty list [].

from typing import List

class Solution:
    def permute(self, n: int, k: int) -> List[int]:
        odd_cnt = (n + 1) // 2
        even_cnt = n // 2
        
        # precompute factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i
        
        total = fact[odd_cnt] * fact[even_cnt]
        if n % 2 == 0:
            total *= 2
        if k > total:
            return []
        
        used = [False] * (n + 1)
        res = []
        cur_odd = odd_cnt
        cur_even = even_cnt
        
        for i in range(1, n + 1):
            if i == 1:
                for x in range(1, n + 1):
                    if used[x]:
                        continue
                    # when n is odd, cannot start with even
                    if n % 2 == 1 and x % 2 == 0:
                        continue
                    if x % 2 == 1:
                        o_rem = odd_cnt - 1
                        e_rem = even_cnt
                    else:
                        o_rem = odd_cnt
                        e_rem = even_cnt - 1
                    completions = fact[o_rem] * fact[e_rem]
                    if k > completions:
                        k -= completions
                    else:
                        res.append(x)
                        used[x] = True
                        cur_odd = o_rem
                        cur_even = e_rem
                        break
                else:
                    return []  # should not happen if k <= total
            else:
                last = res[-1]
                req_parity = 1 - (last % 2)  # 1 for odd, 0 for even
                for x in range(1, n + 1):
                    if used[x]:
                        continue
                    if x % 2 != req_parity:
                        continue
                    if x % 2 == 1:
                        o_rem = cur_odd - 1
                        e_rem = cur_even
                    else:
                        o_rem = cur_odd
                        e_rem = cur_even - 1
                    completions = fact[o_rem] * fact[e_rem]
                    if k > completions:
                        k -= completions
                    else:
                        res.append(x)
                        used[x] = True
                        cur_odd = o_rem
                        cur_even = e_rem
                        break
                else:
                    return []  # should not happen
        return res

        res = []
        cur_odd = odd_cnt
        cur_even = even_cnt

        for i in range(1, n + 1):

                for x in range(1, n + 1):

                    if n % 2 == 1 and x % 2 == 0:
                        continue
                    if x % 2 == 1:          # odd first
                        o_rem = odd_cnt - 1
                        e_rem = even_cnt

                        used[x] = True

                        cur_even = e_rem

                    return []   # should not happen if k <= total
