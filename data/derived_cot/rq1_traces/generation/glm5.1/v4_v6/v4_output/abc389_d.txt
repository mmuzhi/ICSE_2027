```python

import math

def solve() -> None:
    R = int(input())
    limit = 4 * R * R          # 4R^2
    isqrt = math.isqrt

    # a = 0 case
    t = limit - 1              # (2*0+1)^2 = 1
    b_max0 = (isqrt(t) - 1) // 2
    total = 1 + 2 * b_max0     # origin + axis squares with |j|>0

    # a = 1 .. R-1
    for a in range(1, R):
        odd = 2 * a + 1
        t = limit - odd * odd
        b_max = (isqrt(t) - 1) // 2
        total += 2 + 4 * b_max   # axis square (b=0) + interior squares (b>0)

    print(total)

if __name__ == "__main__":
    solve()
```
