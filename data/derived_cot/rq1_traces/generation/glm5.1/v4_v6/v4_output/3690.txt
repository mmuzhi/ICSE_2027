class Solution:
    def minLength(self, s: str, numOps: int) -> int:
        n = len(s)
        arr = [int(ch) for ch in s]
        # compute original max run
        orig_max = 0
        cur = 1
        for i in range(1, n):
            if arr[i] == arr[i-1]:
                cur += 1
            else:
                orig_max = max(orig_max, cur)
                cur = 1
        orig_max = max(orig_max, cur)
        
        # binary search
        lo, hi = 1, orig_max
        # define feasibility function
        def can(L):
            INF = 10**9
            # dp[char][length] = min flips
            dp = [[INF]*(L+2) for _ in range(2)]
            first = arr[0]
            for c in (0,1):
                cost = 0 if c == first else 1
                dp[c][1] = cost
            for i in range(1, n):
                ch = arr[i]
                ndp = [[INF]*(L+2) for _ in range(2)]
                for c in (0,1):
                    row = dp[c]
                    for length in range(1, L+1):
                        cur_cost = row[length]
                        if cur_cost == INF:
                            continue
                        # same character
                        if length + 1 <= L:
                            new_len = length + 1
                            add = 0 if c == ch else 1
                            new_cost = cur_cost + add
                            if new_cost < ndp[c][new_len]:
                                ndp[c][new_len] = new_cost
                        # different character
                        d = 1 - c
                        add = 0 if d == ch else 1
                        new_cost = cur_cost + add
                        if new_cost < ndp[d][1]:
                            ndp[d][1] = new_cost
                dp = ndp
            min_cost = INF
            for c in (0,1):
                for length in range(1, L+1):
                    if dp[c][length] < min_cost:
                        min_cost = dp[c][length]
            return min_cost <= numOps
        
        while lo < hi:
            mid = (lo + hi) // 2
            if can(mid):
                hi = mid
