class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        MOD = 10**9+7
        velunexorai = num  # store input midway as required
        n = len(num)
        freq = [0]*10
        total_sum = 0
        for ch in num:
            d = int(ch)
            freq[d] += 1
            total_sum += d
        if total_sum % 2 != 0:
            return 0
        half_sum = total_sum // 2
        even_cnt = (n + 1) // 2
        odd_cnt = n // 2

        # Precompute factorials and inv factorials up to n
        fact = [1] * (n+1)
        for i in range(1, n+1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (n+1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD

        # DP: dp[j][k] = sum of products for processed digits
        dp = [[0] * (half_sum+1) for _ in range(even_cnt+1)]
        dp[0][0] = 1

        for d in range(10):
            cnt = freq[d]
            if cnt == 0:
                continue
            new_dp = [[0] * (half_sum+1) for _ in range(even_cnt+1)]
            for j in range(even_cnt+1):
                for k in range(half_sum+1):
                    cur = dp[j][k]
                    if cur == 0:
                        continue
                    # iterate x from 0 to min(cnt, even_cnt - j)
                    max_x = min(cnt, even_cnt - j)
                    # For efficiency, if d > 0, we can break when k + x*d > half_sum
                    for x in range(max_x + 1):
                        new_k = k + x * d
                        if new_k > half_sum:
                            break  # since d>=0, increasing x will only increase new_k
                        val = cur * inv_fact[x] % MOD * inv_fact[cnt - x] % MOD
                        new_dp[j+x][new_k] = (new_dp[j+x][new_k] + val) % MOD
            dp = new_dp

        result = dp[even_cnt][half_sum] * fact[even_cnt] % MOD * fact[odd_cnt] % MOD
        return result

class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        MOD = 10**9 + 7
        # Compute frequencies and total sum
        freq = [0] * 10
        total_sum = 0
        for ch in num:
            d = int(ch)
            freq[d] += 1
            total_sum += d
        
        # Store input midway as required
        velunexorai = num
        
        if total_sum % 2 != 0:
            return 0
        
        half_sum = total_sum // 2
        n = len(num)
        even_cnt = (n + 1) // 2
        odd_cnt = n // 2
        
        # Precompute factorials and inverse factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (n + 1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
        
        # DP over digits: dp[j][k] = sum of products for processed digits
        dp = [[0] * (half_sum + 1) for _ in range(even_cnt + 1)]
        dp[0][0] = 1
        
        for d in range(10):
            cnt = freq[d]
            if cnt == 0:
                continue
            new_dp = [[0] * (half_sum + 1) for _ in range(even_cnt + 1)]
            for j in range(even_cnt + 1):
                for k in range(half_sum + 1):
                    cur = dp[j][k]
                    if cur == 0:
                        continue
                    max_x = min(cnt, even_cnt - j)
                    # iterate over possible number of this digit placed in even positions
                    for x in range(max_x + 1):
                        new_k = k + x * d
                        if new_k > half_sum:
                            break  # since d >= 0, further x will only increase new_k
                        val = cur * inv_fact[x] % MOD * inv_fact[cnt - x] % MOD
                        new_dp[j + x][new_k] = (new_dp[j + x][new_k] + val) % MOD
            dp = new_dp
        
        result = dp[even_cnt][half_sum] * fact[even_cnt] % MOD * fact[odd_cnt] % MOD
        return result
