=== ANSWER ===
```python

import sys
import bisect

def main():
    input = sys.stdin.readline

    N = int(input())
    X = list(map(int, input().split()))
    P = list(map(int, input().split()))

    # prefix sums: pref[i] = sum of P[0..i-1]
    pref = [0] * (N + 1)
    for i in range(N):
        pref[i + 1] = pref[i] + P[i]

    Q = int(input())
    out = []
    for _ in range(Q):
        L, R = map(int, input().split())
        left = bisect.bisect_left(X, L)   # first index with X[i] >= L
        right = bisect.bisect_right(X, R) # first index with X[i] > R
        if left < right:
            out.append(str(pref[right] - pref[left]))
        else:
            out.append('0')

    sys.stdout.write('\n'.join(out))

if __name__ == "__main__":
    main()
```
