=== ANSWER ===
```python

import sys

def main():
    S = sys.stdin.readline().strip()
    # position of each character (1-indexed)
    pos = {c: i+1 for i, c in enumerate(S)}
    total = 0
    prev = pos['A']
    for c in "BCDEFGHIJKLMNOPQRSTUVWXYZ":
        cur = pos[c]
        total += abs(cur - prev)
        prev = cur
    print(total)

if __name__ == "__main__":
    main()
```
