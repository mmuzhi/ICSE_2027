```python

import sys

def solve() -> None:
    s_ab, s_ac, s_bc = sys.stdin.read().split()
    counts = {'A': 0, 'B': 0, 'C': 0}

    # AB comparison
    if s_ab == '<':
        counts['B'] += 1
    else:
        counts['A'] += 1

    # AC comparison
    if s_ac == '<':
        counts['C'] += 1
    else:
        counts['A'] += 1

    # BC comparison
    if s_bc == '<':
        counts['C'] += 1
    else:
        counts['B'] += 1

    # The brother with count 1 is the middle one
    for person, cnt in counts.items():
        if cnt == 1:
            print(person)
            return

if __name__ == "__main__":
    solve()
```
