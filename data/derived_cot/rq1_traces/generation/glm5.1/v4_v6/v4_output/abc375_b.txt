```python

import sys
import math

def main() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    total = 0.0
    prev_x, prev_y = 0, 0  # start at the origin

    for _ in range(n):
        x = int(next(it))
        y = int(next(it))
        total += math.hypot(x - prev_x, y - prev_y)
        prev_x, prev_y = x, y

    # return to the origin
    total += math.hypot(prev_x, prev_y)

    # print with enough digits to satisfy the error tolerance
    print("{:.20f}".format(total))

if __name__ == "__main__":
    main()
```
