- 0 \leq A_i, B_i, C_i, D_i \leq 5000 (1 \leq i \leq N)
- (A_i, B_i) \neq (A_j, B_j) (1 \leq i < j \leq N)

- (A_i, B_i) \neq (C_j, D_j) (1 \leq i, j \leq N)

```
def hungarian(a):
    n = len(a)
    m = len(a[0])  # n == m
    u = [0]*(n+1)
    v = [0]*(m+1)
    p = [0]*(m+1)  # which row is assigned to column j
    way = [0]*(m+1)

    for i in range(1, n+1):
        p[0] = i
        j0 = 0
        minv = [float('inf')]*(m+1)
        used = [False]*(m+1)
        while True:
            used[j0] = True
            i0 = p[j0]
            delta = float('inf')
            j1 = 0
            for j in range(1, m+1):
                if not used[j]:
                    cur = a[i0-1][j-1] - u[i0] - v[j]
                    if cur < minv[j]:
                        minv[j] = cur
                        way[j] = j0
                    if minv[j] < delta:
                        delta = minv[j]
                        j1 = j
            for j in range(m+1):
                if used[j]:
                    u[p[j]] += delta
                    v[j] -= delta
                else:
                    minv[j] -= delta
            j0 = j1
            if p[j0] == 0:
                break
        # augmenting
        while True:
            j1 = way[j0]
            p[j0] = p[j1]
            j0 = j1
            if j0 == 0:
                break
    # result
    ans = [-1]*n
    for j in range(1, m+1):
        if p[j] != 0:
            ans[p[j]-1] = j-1
    return ans
```

```
            i0 = p[j0]

                        way[j] = j0

                        delta = minv[j]
                        j1 = j

                    v[j] -= delta
                else:

            if p[j0] == 0:

            j1 = way[j0]
            p[j0] = p[j1]
            j0 = j1
            if j0 == 0:
                break

    assignment = [-1]*n
    for j in range(1, m+1):
        if p[j] != 0:

    return assignment
```

```
import sys, math

def hungarian(cost):
    n = len(cost)
    m = n
    u = [0.0]*(n+1)
    v = [0.0]*(m+1)
    p = [0]*(m+1)
    way = [0]*(m+1)

    for i in range(1, n+1):
        p[0] = i
        j0 = 0
        minv = [float('inf')]*(m+1)
        used = [False]*(m+1)
        while True:
            used[j0] = True
            i0 = p[j0]
            delta = float('inf')
            j1 = 0
            # local references for speed
            cost_i0 = cost[i0-1]
            u_i0 = u[i0]
            for j in range(1, m+1):
                if not used[j]:
                    cur = cost_i0[j-1] - u_i0 - v[j]
                    if cur < minv[j]:
                        minv[j] = cur
                        way[j] = j0
                    if minv[j] < delta:
                        delta = minv[j]
                        j1 = j
            for j in range(m+1):
                if used[j]:
                    u[p[j]] += delta
                    v[j] -= delta
                else:
                    minv[j] -= delta
            j0 = j1
            if p[j0] == 0:
                break
        # augmentation
        while True:
            j1 = way[j0]
            p[j0] = p[j1]
            j0 = j1
            if j0 == 0:
                break

    ans = [-1]*n
    for j in range(1, m+1):
        if p[j] != 0:
            ans[p[j]-1] = j-1
    return ans

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    P = []
    for _ in range(n):
        x = int(next(it)); y = int(next(it))
        P.append((x,y))
    Q = []
    for _ in range(n):
        x = int(next(it)); y = int(next(it))
        Q.append((x,y))
    cost = [[0.0]*n for _ in range(n)]
    for i in range(n):
        xi, yi = P[i]
        for j in range(n):
            xj, yj = Q[j]
            dx = xi - xj
            dy = yi - yj
            cost[i][j] = math.hypot(dx, dy)
    assign = hungarian(cost)
    # assign[i] is column (0-indexed) for row i
    result = [str(assign[i]+1) for i in range(n)]
    print(' '.join(result))

if __name__ == "__main__":
    main()
```

```python
import sys
import math

# ------------------------------------------------------------
# Hungarian algorithm for minimum cost perfect matching
# (O(n^3), works with floating point costs)
# ------------------------------------------------------------
def hungarian(cost):
    n = len(cost)                 # n = m
    m = n

    u = [0.0] * (n + 1)
    v = [0.0] * (m + 1)
    p = [0] * (m + 1)             # which row is assigned to column j
    way = [0] * (m + 1)

    for i in range(1, n + 1):
        p[0] = i
        j0 = 0
        minv = [float('inf')] * (m + 1)
        used = [False] * (m + 1)

        while True:
            used[j0] = True
            i0 = p[j0]                     # current row
            delta = float('inf')
            j1 = 0

            # local references for speed
            cost_i0 = cost[i0 - 1]
            u_i0 = u[i0]

            for j in range(1, m + 1):
                if not used[j]:
                    cur = cost_i0[j - 1] - u_i0 - v[j]
                    if cur < minv[j]:
                        minv[j] = cur
                        way[j] = j0
                    if minv[j] < delta:
                        delta = minv[j]
                        j1 = j

            for j in range(m + 1):
                if used[j]:
                    u[p[j]] += delta
                    v[j] -= delta
                else:
                    minv[j] -= delta

            j0 = j1
            if p[j0] == 0:
                break

        # augmentation
        while True:
            j1 = way[j0]
            p[j0] = p[j1]
            j0 = j1
            if j0 == 0:
                break

    # build answer: for each row i, the assigned column (0‑based)
    assignment = [-1] * n
    for j in range(1, m + 1):
        if p[j] != 0:
            assignment[p[j] - 1] = j - 1
    return assignment


# ------------------------------------------------------------
def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))

    P = []
    for _ in range(n):
        x = int(next(it)); y = int(next(it))
        P.append((x, y))

    Q = []
    for _ in range(n):
        x = int(next(it)); y = int(next(it))
        Q.append((x, y))

    # cost matrix: Euclidean distances
    cost = [[0.0] * n for _ in range(n)]
    for i in range(n):
        xi, yi = P[i]
        for j in range(n):
            xj, yj = Q[j]
            dx = xi - xj
            dy = yi - yj
            cost[i][j] = math.hypot(dx, dy)

    assign = hungarian(cost)          # assign[i] = column index for P_i (0‑based)

    # output permutation R (1‑based)
    ans = [str(assign[i] + 1) for i in range(n)]
    print(' '.join(ans))


if __name__ == "__main__":
    main()
```
