Determine whether it is possible to perform the following operation on string S between 0 and K times, inclusive, to make it identical to string T.

- Choose one of the following three operations and execute it.
- Insert any one character at any position in S (possibly the beginning or end).

import sys

def solve() -> None:
    input = sys.stdin.readline
    K = int(input().strip())          # K is always 1, but we read it anyway
    S = input().strip()
    T = input().strip()
    n, m = len(S), len(T)

    if n == m:
        diff = 0
        for i in range(n):
            if S[i] != T[i]:
                diff += 1
                if diff > 1:
                    print("No")
                    return
        print("Yes")
        return

    # Case 2: S is longer by 1 -> delete one character from S
    if n == m + 1:
        i = 0
        # find first position where they differ
        while i < m and S[i] == T[i]:
            i += 1
        # compare S[i+1:] with T[i:]

        while j < n and k < m:
            if S[j] != T[k]:
                print("No")

            j += 1
            k += 1

    # Case 3: T is longer by 1 -> insert one character into S
