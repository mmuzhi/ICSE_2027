```python
from typing import List

class Solution:
    def maxSubarraySum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        prefix = [0] * (n+1)
        for i in range(n):
            prefix[i+1] = prefix[i] + nums[i]
        # min_prefix for each modulo class
        INF = 10**18
        min_prefix = [INF] * k
        min_prefix[0] = 0  # prefix[0] mod 0
        ans = -INF
        for j in range(1, n+1):
            mod = j % k
            if min_prefix[mod] != INF:
                candidate = prefix[j] - min_prefix[mod]
                if candidate > ans:
                    ans = candidate
            # update min_prefix for this mod
            if prefix[j] < min_prefix[mod]:
                min_prefix[mod] = prefix[j]
        return ans
```

```python

from typing import List

class Solution:
    def maxSubarraySum(self, nums: List[int], k: int) -> int:
        INF = 10**18                     # sufficiently small
        min_prefix = [INF] * k           # min_prefix[r] = min prefix sum with index % k == r
        min_prefix[0] = 0                # prefix sum before any element (index 0)
        cur_sum = 0
        ans = -INF

        for j in range(1, len(nums) + 1):
            cur_sum += nums[j - 1]        # prefix sum up to index j
            r = j % k

            if min_prefix[r] != INF:     # at least one valid start index
                candidate = cur_sum - min_prefix[r]
                if candidate > ans:
                    ans = candidate

            # update the minimum for this remainder with the current prefix sum
            if cur_sum < min_prefix[r]:
                min_prefix[r] = cur_sum

        return ans
```
