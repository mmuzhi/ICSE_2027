Implementation steps:

1. Sort nums.
2. Compute B_max using two pointers:
   l = 0
   max_window = 0
   for r in range(n):
       while nums[r] - nums[l] > 2 * k:
           l += 1
       max_window = max(max_window, r - l + 1)
   B_max = max_window

3. Build frequency dictionary: freq = Counter(nums) (or use collections.Counter).

4. For each distinct value v in freq:
   - Compute B_v: number of elements in [v-k, v+k].
     Since nums is sorted, we can use bisect_left and bisect_right.
     left = bisect_left(nums, v - k)
     right = bisect_right(nums, v + k)   # exclusive index
     B_v = right - left
   - candidate = min(freq[v] + numOperations, B_v)
   - update max_candidate.

5. answer = max(max_candidate, min(numOperations, B_max))

6. Return answer.

```python
from typing import List
import bisect
from collections import Counter

class Solution:
    def maxFrequency(self, nums: List[int], k: int, numOperations: int) -> int:
        n = len(nums)
        nums.sort()
        # Compute B_max: max number of elements in any interval of length 2k
        max_window = 0
        l = 0
        for r in range(n):
            while nums[r] - nums[l] > 2 * k:
                l += 1
            max_window = max(max_window, r - l + 1)
        B_max = max_window
        
        # Frequency of each value
        freq = Counter(nums)
        
        max_candidate = 0
        for v, cnt in freq.items():
            left = bisect.bisect_left(nums, v - k)
            right = bisect.bisect_right(nums, v + k)
            B_v = right - left
            candidate = min(cnt + numOperations, B_v)
            if candidate > max_candidate:
                max_candidate = candidate
        
        answer = max(max_candidate, min(numOperations, B_max))
        return answer
```

```python
from typing import List
import bisect
from collections import Counter

class Solution:
    def maxFrequency(self, nums: List[int], k: int, numOperations: int) -> int:
        n = len(nums)
        nums.sort()
        
        # 1. Compute B_max = maximum number of elements in any interval of length 2k
        max_window = 0
        l = 0
        for r in range(n):
            while nums[r] - nums[l] > 2 * k:
                l += 1
            max_window = max(max_window, r - l + 1)
        B_max = max_window
        
        # 2. Frequency of each distinct value
        freq = Counter(nums)
        
        # 3. Evaluate candidate for each original value v
        best_original = 0
        for v, cnt in freq.items():
            left = bisect.bisect_left(nums, v - k)
            right = bisect.bisect_right(nums, v + k)
            B_v = right - left          # total elements that can become v
            candidate = min(cnt + numOperations, B_v)
            if candidate > best_original:
                best_original = candidate
        
        # 4. The best we can do with a non‑original target is min(numOperations, B_max)
        best_non_original = min(numOperations, B_max)
        
        # 5. Answer is the maximum of the two
        return max(best_original, best_non_original)
```
