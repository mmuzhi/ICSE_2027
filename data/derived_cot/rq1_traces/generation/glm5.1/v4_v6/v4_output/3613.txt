class Solution:

        # Build an undirected graph where each edge carries the conversion rate
        def build_graph(pairs, rates):
            graph = {}
            for (start, target), r in zip(pairs, rates):
                # forward edge
                graph.setdefault(start, []).append((target, r))
                # backward edge (rate = 1 / r)
                graph.setdefault(target, []).append((start, 1.0 / r))
            return graph

        # BFS from start currency, returns dict {currency: amount from 1 unit of start}
        def bfs(graph, start):
            dist = {start: 1.0}
            q = deque([start])
            while q:

                val = dist[u]
                for v, r in graph.get(u, []):
                    if v not in dist:
                        dist[v] = val * r
                        q.append(v)
            return dist

        # Day 1
        g1 = build_graph(pairs1, rates1)
        d1 = bfs(g1, initialCurrency)

        # Day 2
        g2 = build_graph(pairs2, rates2)
        d2 = bfs(g2, initialCurrency)

        # Find the best intermediate currency
        ans = 1.0
        for cur, val1 in d1.items():
