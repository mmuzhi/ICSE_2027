```python

class Solution:
    def kthCharacter(self, k: int) -> str:
        word = "a"
        # Keep applying the operation until the string is long enough
        while len(word) < k:
            # Generate the transformed part: each character becomes its next (cyclic)
            transformed = ''.join(chr((ord(c) - 97 + 1) % 26 + 97) for c in word)
            word += transformed
        # k is 1‑indexed, so return the character at position k-1
        return word[k-1]

```
