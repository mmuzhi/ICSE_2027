```python
class Solution:
    def countPairs(self, nums: List[int]) -> int:
        n = len(nums)
        # Precompute reachable sets
        reachable = []
        for num in nums:
            s = str(num)
            st = set()
            st.add(num)
            if len(s) >= 2:
                for i in range(len(s)):
                    for j in range(i+1, len(s)):
                        # swap
                        lst = list(s)
                        lst[i], lst[j] = lst[j], lst[i]
                        new_num = int(''.join(lst))
                        st.add(new_num)
            reachable.append(st)
        
        count = 0
        for i in range(n):
            for j in range(i+1, n):
                if nums[j] in reachable[i] or nums[i] in reachable[j]:
                    count += 1
        return count
```

```python
from typing import List

class Solution:
    def countPairs(self, nums: List[int]) -> int:
        n = len(nums)
        # Precompute the set of numbers reachable from each num by at most one swap
        reachable = []
        for num in nums:
            s = str(num)
            st = {num}
            if len(s) > 1:
                for i in range(len(s)):
                    for j in range(i+1, len(s)):
                        # swap digits at i and j
                        lst = list(s)
                        lst[i], lst[j] = lst[j], lst[i]
                        new_num = int(''.join(lst))
                        st.add(new_num)
            reachable.append(st)
        
        count = 0
        for i in range(n):
            for j in range(i+1, n):
                if nums[j] in reachable[i] or nums[i] in reachable[j]:
                    count += 1
        return count
```
