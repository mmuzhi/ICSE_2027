You are given two strings s and pattern.
A string x is called almost equal to y if you can change at most one character in x to make it identical to y.
Return the smallest starting index of a substring in s that is almost equal to pattern. If no such index exists, return -1.
A substring is a contiguous non-empty sequence of characters within a string.
 
Example 1:

Input: s = "abcdefg", pattern = "bcdffg"
Output: 1
Explanation:
The substring s[1..6] == "bcdefg" can be converted to "bcdffg" by changing s[4] to "f".

Example 2:

Input: s = "ababbababa", pattern = "bacaba"
Output: 4
Explanation:
The substring s[4..9] == "bababa" can be converted to "bacaba" by changing s[6] to "c".

Example 3:

Input: s = "abcd", pattern = "dba"
Output: -1

Example 4:

Input: s = "dde", pattern = "d"
Output: 0

 
Constraints:

1 <= pattern.length < s.length <= 10^5
s and pattern consist only of lowercase English letters.

Z-function algorithm (O(n)):

def z_function(s):
    n = len(s)
    z = [0]*n
    l = r = 0
    for i in range(1, n):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i + z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1
    return z

class Solution:
    def minStartingIndex(self, s: str, pattern: str) -> int:
        n, m = len(s), len(pattern)
        if m > n: return -1  # not needed per constraints but safe
        # compute LCP using Z
        combined = pattern + '#' + s
        z1 = self.z_function(combined)
        lcp = [0] * (n - m + 1)  # only needed indices
        offset = m + 1
        for i in range(n - m + 1):
            # i is start index in s
            val = z1[offset + i]
            if val > m:
                val = m
            lcp[i] = val
        
        # compute LCS using reversed strings
        rev_pattern = pattern[::-1]
        rev_s = s[::-1]
        combined_rev = rev_pattern + '#' + rev_s
        z2 = self.z_function(combined_rev)
        lcs = [0] * (n - m + 1)
        offset_rev = m + 1
        for i in range(n - m + 1):
            rev_start = n - i - m  # start index in rev_s for the reversed substring
            # rev_start is in [0, n-m]
            val = z2[offset_rev + rev_start]
            if val > m:
                val = m
            lcs[i] = val
        
        # check condition
        for i in range(n - m + 1):
            if lcp[i] + lcs[i] >= m - 1:
                return i
        return -1

class Solution:
    def minStartingIndex(self, s: str, pattern: str) -> int:
        n, m = len(s), len(pattern)
        if m > n:
            return -1
        
        # compute LCP using Z-algorithm
        def z_function(t):
            n_t = len(t)
            z = [0] * n_t
            l = r = 0
            for i in range(1, n_t):
                if i <= r:
                    z[i] = min(r - i + 1, z[i - l])
                while i + z[i] < n_t and t[z[i]] == t[i + z[i]]:
                    z[i] += 1
                if i + z[i] - 1 > r:
                    l, r = i, i + z[i] - 1
            return z
        
        combined = pattern + '#' + s
        z1 = z_function(combined)
        offset = m + 1
        lcp = [0] * (n - m + 1)
        for i in range(n - m + 1):
            val = z1[offset + i]
            if val > m:
                val = m
            lcp[i] = val
        
        # compute LCS using reversed strings
        rev_pattern = pattern[::-1]
        rev_s = s[::-1]
        combined_rev = rev_pattern + '#' + rev_s
        z2 = z_function(combined_rev)

                val = m
            lcs[i] = val

        if m > n:                     # not needed by constraints, but safe
            return -1

        def z_function(t: str):

            z = [0] * n_t
            l = r = 0

                    z[i] += 1
                if i + z[i] - 1 > r:
                    l, r = i, i + z[i] - 1
            return z

        # ---------- longest common prefix (LCP) ----------
        combined = pattern + '#' + s
        z1 = z_function(combined)

        lcp = [0] * (n - m + 1)          # only for valid start indices

            lcp[i] = val

            rev_start = n - i - m          # start index in rev_s
