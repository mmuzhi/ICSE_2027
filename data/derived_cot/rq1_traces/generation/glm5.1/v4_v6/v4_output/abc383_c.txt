import sys
from collections import deque

def main():
    import sys
    input = sys.stdin.readline
    H, W, D = map(int, input().split())
    grid = [input().strip() for _ in range(H)]
    dist = [[-1]*W for _ in range(H)]
    q = deque()
    for i in range(H):
        row = grid[i]
        for j, ch in enumerate(row):
            if ch == 'H':
                dist[i][j] = 0
                q.append((i, j))
    # BFS
    while q:
        i, j = q.popleft()
        d = dist[i][j]
        if d == D:
            continue
        nd = d + 1
        # up
        if i > 0:
            ni = i-1
            if grid[ni][j] != '#' and dist[ni][j] == -1:
                dist[ni][j] = nd
                q.append((ni, j))
        # down
        if i < H-1:
            ni = i+1
            if grid[ni][j] != '#' and dist[ni][j] == -1:
                dist[ni][j] = nd
                q.append((ni, j))
        # left
        if j > 0:
            nj = j-1
            if grid[i][nj] != '#' and dist[i][nj] == -1:
                dist[i][nj] = nd
                q.append((i, nj))
        # right
        if j < W-1:
            nj = j+1
            if grid[i][nj] != '#' and dist[i][nj] == -1:
                dist[i][nj] = nd
                q.append((i, nj))
    ans = 0
    for i in range(H):
        for j in range(W):
            if grid[i][j] != '#' and dist[i][j] != -1:
                ans += 1
    print(ans)

if __name__ == "__main__":
    main()
