def solve():
    import sys
    MOD = 998244353
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    INF = N+1
    minR = [ [INF]*(N+2) for _ in range(N+1) ]  # minR[X][l]
    for _ in range(M):
        L = int(next(it)); R = int(next(it)); X = int(next(it))
        if R < minR[X][L]:
            minR[X][L] = R
    # compute suffix minima
    for X in range(1, N+1):
        # minR[X][N+1] already INF
        for l in range(N, 0, -1):
            if minR[X][l+1] < minR[X][l]:
                minR[X][l] = minR[X][l+1]
    # precompute binomial
    C = [ [0]*(N+1) for _ in range(N+1) ]
    for n in range(N+1):
        C[n][0] = C[n][n] = 1
        for k in range(1, n):
            C[n][k] = (C[n-1][k-1] + C[n-1][k]) % MOD
    # DP
    f = [ [0]*(N+2) for _ in range(N+2) ]
    # base: empty segments are 1, but we handle in code
    for length in range(1, N+1):
        for l in range(1, N-length+2):
            r = l + length - 1
            total = 0
            # iterate k
            for k in range(l, r+1):
                if minR[k][l] <= r:
                    continue
                left = f[l][k-1] if k > l else 1
                right = f[k+1][r] if k < r else 1
                comb = C[length-1][k-l]
                total = (total + left * right % MOD * comb) % MOD
            f[l][r] = total
    print(f[1][N])

def main():

    it = iter(data)
    N = int(next(it))

    INF = N + 1

    for _ in range(M):
        L = int(next(it)); R = int(next(it)); X = int(next(it))
        if R < minR[X][L]:
            minR[X][L] = R
    # suffix minima
    for X in range(1, N+1):
        for l in range(N, 0, -1):
            if minR[X][l+1] < minR[X][l]:
                minR[X][l] = minR[X][l+1]
    # binomial coefficients
    C = [ [0]*(N+1) for _ in range(N+1) ]
    for n in range(N+1):
        C[n][0] = C[n][n] = 1
        for k in range(1, n):
            C[n][k] = (C[n-1][k-1] + C[n-1][k]) % MOD
    # DP
    f = [ [0]*(N+2) for _ in range(N+2) ]
    # We'll compute for length from 1 to N
    for length in range(1, N+1):

                if minR[k][l] <= r:
