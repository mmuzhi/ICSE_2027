import sys

def solve():
    data = sys.stdin.read().split()
    # First token: S
    S = data[0]
    # Second token: Q
    Q = int(data[1])
    # Next Q tokens: K_i
    Ks = list(map(int, data[2:2+Q]))
    N = len(S)
    ans = []
    for K in Ks:
        a = (K-1) // N
        b = (K-1) % N
        if bin(a).count('1') % 2 == 0:
            ans.append(S[b])
        else:
            ans.append(S[b].swapcase())
    print(' '.join(ans))

if __name__ == "__main__":
    solve()
