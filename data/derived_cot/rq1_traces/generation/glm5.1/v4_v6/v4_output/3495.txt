```python
from typing import List
import heapq

class Solution:
    def resultsArray(self, queries: List[List[int]], k: int) -> List[int]:
        heap = []  # max-heap of the k smallest distances (store negatives)
        res = []
        for x, y in queries:
            d = abs(x) + abs(y)
            if len(heap) < k:
                heapq.heappush(heap, -d)
                if len(heap) == k:
                    res.append(-heap[0])
                else:
                    res.append(-1)
            else:
                # heap size == k
                current_kth = -heap[0]
                if d < current_kth:
                    heapq.heapreplace(heap, -d)
                    res.append(-heap[0])
                else:
                    res.append(current_kth)
        return res
```

```python

from typing import List
import heapq

class Solution:
    def resultsArray(self, queries: List[List[int]], k: int) -> List[int]:
        # max-heap (simulated by storing negatives) of the k smallest distances
        heap = []
        res = []
        for x, y in queries:
            d = abs(x) + abs(y)
            if len(heap) < k:
                heapq.heappush(heap, -d)
                if len(heap) == k:
                    res.append(-heap[0])
                else:
                    res.append(-1)
            else:
                # heap already contains k smallest distances
                current_kth = -heap[0]
                if d < current_kth:
                    heapq.heapreplace(heap, -d)
                    res.append(-heap[0])
                else:
                    res.append(current_kth)
        return res
```
