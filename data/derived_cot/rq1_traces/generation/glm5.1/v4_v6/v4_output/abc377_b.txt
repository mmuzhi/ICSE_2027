```python

import sys

def main():
    # Read the 8 rows
    rows = [sys.stdin.readline().strip() for _ in range(8)]

    row_has_piece = [False] * 8
    col_has_piece = [False] * 8

    for i, s in enumerate(rows):
        if '#' in s:
            row_has_piece[i] = True
        for j, ch in enumerate(s):
            if ch == '#':
                col_has_piece[j] = True

    rows_without = sum(1 for i in range(8) if not row_has_piece[i])
    cols_without = sum(1 for j in range(8) if not col_has_piece[j])

    print(rows_without * cols_without)

if __name__ == "__main__":
    main()
```
