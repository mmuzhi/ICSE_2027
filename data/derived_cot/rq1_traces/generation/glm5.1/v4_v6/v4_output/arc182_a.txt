def solve():
    import sys
    MOD = 998244353
    input = sys.stdin.readline
    N, Q = map(int, input().split())
    P = [0]*Q
    V = [0]*Q
    for i in range(Q):
        p, v = map(int, input().split())
        P[i] = p
        V[i] = v

    INF = 10**9+5
    left_min = [INF]*Q
    left_max = [0]*Q
    right_min = [INF]*Q
    right_max = [0]*Q

    for i in range(Q):
        Vi = V[i]
        Pi = P[i]
        for j in range(i):
            if V[j] > Vi:
                Pj = P[j]
                # left for i
                if Pj < left_min[i]:
                    left_min[i] = Pj
                if Pj > left_max[i]:
                    left_max[i] = Pj
                # right for j
                if Pi < right_min[j]:
                    right_min[j] = Pi
                if Pi > right_max[j]:
                    right_max[j] = Pi

    free = 0
    for i in range(Q):
        Pi = P[i]
        left_empty = (left_max[i] == 0)
        right_empty = (right_max[i] == 0)

        left_forced = None
        if not left_empty:
            if left_max[i] < Pi:
                left_forced = 1  # R
            elif left_min[i] > Pi:
                left_forced = 0  # L
            else:
                print(0)
                return

        right_forced = None
        if not right_empty:
            if right_min[i] > Pi:
                right_forced = 0  # L
            elif right_max[i] < Pi:
                right_forced = 1  # R
            else:
                print(0)
                return

        if left_forced is not None and right_forced is not None:
            if left_forced != right_forced:
                print(0)
                return
        if left_forced is None and right_forced is None:
            free += 1

    ans = pow(2, free, MOD)
    print(ans)
