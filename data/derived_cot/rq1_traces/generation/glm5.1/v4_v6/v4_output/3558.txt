def findSafeWalk(grid, health):
    m, n = len(grid), len(grid[0])
    INF = 10**9
    dist = [[INF]*n for _ in range(m)]
    dist[0][0] = grid[0][0]  # cost to be at start
    from collections import deque
    dq = deque()
    dq.append((0,0))
    while dq:
        x,y = dq.popleft()
        d = dist[x][y]
        for dx,dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            if 0 <= nx < m and 0 <= ny < n:
                nd = d + grid[nx][ny]
                if nd < dist[nx][ny]:
                    dist[nx][ny] = nd
                    if grid[nx][ny] == 0:
                        dq.appendleft((nx,ny))
                    else:
                        dq.append((nx,ny))
    min_ones = dist[m-1][n-1]
    return health > min_ones

```python
from collections import deque
from typing import List

class Solution:
    def findSafeWalk(self, grid: List[List[int]], health: int) -> bool:
        m, n = len(grid), len(grid[0])
        INF = 10**9
        dist = [[INF] * n for _ in range(m)]
        dist[0][0] = grid[0][0]               # cost of starting cell
        dq = deque()
        dq.append((0, 0))

        while dq:
            x, y = dq.popleft()
            d = dist[x][y]
            for dx, dy in ((1, 0), (-1, 0), (0, 1), (0, -1)):
                nx, ny = x + dx, y + dy
                if 0 <= nx < m and 0 <= ny < n:
                    nd = d + grid[nx][ny]
                    if nd < dist[nx][ny]:
                        dist[nx][ny] = nd
                        if grid[nx][ny] == 0:
                            dq.appendleft((nx, ny))
                        else:
                            dq.append((nx, ny))

        min_ones = dist[m-1][n-1]
        return health > min_ones
```
