```python

from typing import List

class Solution:
    def stringSequence(self, target: str) -> List[str]:
        res = []          # list of strings that appear on the screen
        cur = []          # current string as a list of characters

        for ch in target:
            # Press key 1: append 'a'
            cur.append('a')
            res.append(''.join(cur))

            # Press key 2 until the last character becomes ch
            diff = ord(ch) - ord('a')
            for _ in range(diff):
                # increment last character (cyclic, though never wraps in optimal path)
                cur[-1] = chr((ord(cur[-1]) - ord('a') + 1) % 26 + ord('a'))
                res.append(''.join(cur))

        return res
```
