You will be given a competitive programming problem.
You are given a tree rooted at node 0, consisting of n nodes numbered from 0 to n - 1. The tree is represented by an array parent of size n, where parent[i] is the parent of node i. Since node 0 is the root, parent[0] == -1.
You are also given a string s of length n, where s[i] is the character assigned to node i.
Consider an empty string dfsStr, and define a recursive function dfs(int x) that takes a node x as a parameter and performs the following steps in order:

Iterate over each child y of x in increasing order of their numbers, and call dfs(y).
Add the character s[x] to the end of the string dfsStr.

Note that dfsStr is shared across all recursive calls of dfs.
You need to find a boolean array answer of size n, where for each index i from 0 to n - 1, you do the following:

Empty the string dfsStr and call dfs(i).
If the resulting string dfsStr is a palindrome, then set answer[i] to true. Otherwise, set answer[i] to false.

Return the array answer.

Input: parent = [-1,0,0,1,1,2], s = "aababa"
Output: [true,true,false,true,true,true]
Explanation:

Calling dfs(0) results in the string dfsStr = "abaaba", which is a palindrome.
Calling dfs(1) results in the string dfsStr = "aba", which is a palindrome.
Calling dfs(2) results in the string dfsStr = "ab", which is not a palindrome.
Calling dfs(3) results in the string dfsStr = "a", which is a palindrome.
Calling dfs(4) results in the string dfsStr = "b", which is a palindrome.
Calling dfs(5) results in the string dfsStr = "a", which is a palindrome.

class Solution:
    def findAnswer(self, parent: List[int], s: str) -> List[bool]:
        n = len(parent)
        children = [[] for _ in range(n)]
        for i in range(1, n):
            p = parent[i]
            children[p].append(i)  # already increasing
        
        mod1 = 1000000007
        mod2 = 1000000009
        base = 131
        
        pow1 = [1] * (n + 1)
        pow2 = [1] * (n + 1)
        for i in range(1, n + 1):
            pow1[i] = (pow1[i-1] * base) % mod1
            pow2[i] = (pow2[i-1] * base) % mod2
        
        length = [0] * n
        h1 = [0] * n
        h2 = [0] * n
        rh1 = [0] * n
        rh2 = [0] * n
        ans = [False] * n
        
        stack = [(0, 0)]  # (node, state) 0=enter,1=exit
        while stack:
            node, state = stack.pop()
            if state == 0:
                stack.append((node, 1))
                for child in children[node]:
                    stack.append((child, 0))
            else:
                val = ord(s[node]) - 96  # 'a' -> 1
                # forward hash
                h1_val = 0
                h2_val = 0
                offset = 0
                for child in children[node]:
                    h1_val = (h1_val + h1[child] * pow1[offset]) % mod1
                    h2_val = (h2_val + h2[child] * pow2[offset]) % mod2
                    offset += length[child]
                # add own character
                h1_val = (h1_val + val * pow1[offset]) % mod1
                h2_val = (h2_val + val * pow2[offset]) % mod2
                length[node] = offset + 1
                
                # reverse hash
                rh1_val = val % mod1
                rh2_val = val % mod2
                offset_rev = 1
                for child in reversed(children[node]):
                    rh1_val = (rh1_val + rh1[child] * pow1[offset_rev]) % mod1
                    rh2_val = (rh2_val + rh2[child] * pow2[offset_rev]) % mod2
                    offset_rev += length[child]
                
                h1[node] = h1_val
                h2[node] = h2_val
                rh1[node] = rh1_val
                rh2[node] = rh2_val
                
                ans[node] = (h1_val == rh1_val and h2_val == rh2_val)
        
        return ans

from typing import List

class Solution:
    def findAnswer(self, parent: List[int], s: str) -> List[bool]:
        n = len(parent)
        # build children lists; they are automatically in increasing order
        children = [[] for _ in range(n)]
        for i in range(1, n):
            p = parent[i]
            children[p].append(i)

        MOD1 = 1_000_000_007
        MOD2 = 1_000_000_009
        BASE = 131

        # precompute powers of BASE
        pow1 = [1] * (n + 1)
        pow2 = [1] * (n + 1)
        for i in range(1, n + 1):
            pow1[i] = (pow1[i-1] * BASE) % MOD1
            pow2[i] = (pow2[i-1] * BASE) % MOD2

        length = [0] * n          # subtree size
        h1 = [0] * n              # forward hash mod MOD1
        h2 = [0] * n              # forward hash mod MOD2
        rh1 = [0] * n             # reverse hash mod MOD1
        rh2 = [0] * n             # reverse hash mod MOD2
        ans = [False] * n

        # iterative post-order DFS
        stack = [(0, 0)]          # (node, state) 0 = enter, 1 = exit
        while stack:
            node, state = stack.pop()
            if state == 0:
                # after processing children we will come back to this node
                stack.append((node, 1))
                # push children (any order works; they will be processed before the node)
                for child in children[node]:
                    stack.append((child, 0))
            else:
                # all children have been processed – compute hashes for node
                val = ord(s[node]) - 96   # 'a' -> 1, ..., 'z' -> 26

                # ----- forward hash (postorder string) -----
                h1_val = 0
                h2_val = 0
                offset = 0
                for child in children[node]:          # increasing order
                    h1_val = (h1_val + h1[child] * pow1[offset]) % MOD1
                    h2_val = (h2_val + h2[child] * pow2[offset]) % MOD2
                    offset += length[child]
                # own character at the end
                h1_val = (h1_val + val * pow1[offset]) % MOD1
                h2_val = (h2_val + val * pow2[offset]) % MOD2
                length[node] = offset + 1

                # ----- reverse hash (reverse of postorder string) -----
                rh1_val = val % MOD1
                rh2_val = val % MOD2
                offset_rev = 1
                for child in reversed(children[node]):   # decreasing order
                    rh1_val = (rh1_val + rh1[child] * pow1[offset_rev]) % MOD1
                    rh2_val = (rh2_val + rh2[child] * pow2[offset_rev]) % MOD2
                    offset_rev += length[child]

                h1[node] = h1_val
                h2[node] = h2_val
                rh1[node] = rh1_val
                rh2[node] = rh2_val

                ans[node] = (h1_val == rh1_val and h2_val == rh2_val)

        return ans
