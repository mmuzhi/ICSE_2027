import sys

MOD = 998244353

def sieve_spf(limit):
    spf = list(range(limit+1))
    for i in range(2, int(limit**0.5)+1):
        if spf[i] == i:
            for j in range(i*i, limit+1, i):
                if spf[j] == j:
                    spf[j] = i
    return spf

def main():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    M = N-1
    maxA = max(A) if A else 0
    limit = maxA if maxA > 1 else 2
    spf = sieve_spf(limit)
    prime_alpha = {}
    for idx, a in enumerate(A):
        if a == 1:
            continue
        temp = a
        while temp > 1:
            p = spf[temp]
            e = 0
            while temp % p == 0:
                temp //= p
                e += 1
            if p not in prime_alpha:
                prime_alpha[p] = [0]*M
            prime_alpha[p][idx] = e
    ans = 1
    for p, alpha in prime_alpha.items():
        T = sum(alpha)
        if T == 0:
            continue
        # Precompute powers of p up to T
        pow_p = [1]*(T+1)
        for i in range(1, T+1):
            pow_p[i] = pow_p[i-1] * p % MOD
        # Initialize dp0, dp1
        dp0 = [0]*(T+1)
        dp1 = [0]*(T+1)
        for v in range(1, T+1):
            dp0[v] = pow_p[v]  # p^v
        dp1[0] = 1
        # Process each edge
        for a in alpha:
            if a == 0:
                ndp0 = [0]*(T+1)
                ndp1 = [0]*(T+1)
                for v in range(T+1):
                    if dp0[v]:
                        ndp0[v] = (ndp0[v] + dp0[v] * pow_p[v]) % MOD
                    if dp1[v]:
                        ndp1[v] = (ndp1[v] + dp1[v] * pow_p[v]) % MOD
            else:
                ndp0 = [0]*(T+1)
                ndp1 = [0]*(T+1)
                for v in range(T+1):
                    val0 = dp0[v]
                    if val0:
                        # v + a
                        if v + a <= T:
                            ndp0[v+a] = (ndp0[v+a] + val0 * pow_p[v+a]) % MOD
                        # v - a
                        if v - a >= 0:
                            if v - a == 0:
                                ndp1[0] = (ndp1[0] + val0 * pow_p[0]) % MOD
                            else:
                                ndp0[v-a] = (ndp0[v-a] + val0 * pow_p[v-a]) % MOD
                    val1 = dp1[v]
                    if val1:
                        if v + a <= T:
                            ndp1[v+a] = (ndp1[v+a] + val1 * pow_p[v+a]) % MOD
                        if v - a >= 0:
                            ndp1[v-a] = (ndp1[v-a] + val1 * pow_p[v-a]) % MOD
            dp0, dp1 = ndp0, ndp1
        F = sum(dp1) % MOD
        ans = ans * F % MOD
    print(ans)

if __name__ == "__main__":
    main()
