Your task is to find the maximum area of a rectangle that:

Does not contain any other point inside or on its border.
Has its edges parallel to the axes.

Return the maximum area that you can obtain or -1 if no such rectangle is possible.
 
Example 1:

Input: points = [[1,1],[1,3],[3,1],[3,3]]
Output: 4

```python
from itertools import combinations
from typing import List

class Solution:
    def maxRectangleArea(self, points: List[List[int]]) -> int:
        n = len(points)
        if n < 4:
            return -1
        point_set = set((x, y) for x, y in points)
        max_area = -1
        for comb in combinations(points, 4):
            # get xs and ys
            xs = sorted(set(p[0] for p in comb))
            ys = sorted(set(p[1] for p in comb))
            if len(xs) != 2 or len(ys) != 2:
                continue
            x1, x2 = xs[0], xs[1]
            y1, y2 = ys[0], ys[1]
            # area
            area = (x2 - x1) * (y2 - y1)
            # check other points
            comb_set = set((p[0], p[1]) for p in comb)
            valid = True
            for (x, y) in point_set:
                if (x, y) in comb_set:
                    continue
                if x1 <= x <= x2 and y1 <= y <= y2:
                    valid = False
                    break
            if valid:
                if area > max_area:
                    max_area = area
        return max_area
```

```python

from typing import List
from itertools import combinations

class Solution:
    def maxRectangleArea(self, points: List[List[int]]) -> int:
        n = len(points)
        if n < 4:
            return -1

        # set for O(1) membership tests
        point_set = set((x, y) for x, y in points)
        max_area = -1

        for comb in combinations(points, 4):
            # extract distinct x and y values
            xs = sorted(set(p[0] for p in comb))
            ys = sorted(set(p[1] for p in comb))

            # need exactly two distinct x and two distinct y
            if len(xs) != 2 or len(ys) != 2:
                continue

            x1, x2 = xs[0], xs[1]
            y1, y2 = ys[0], ys[1]
            area = (x2 - x1) * (y2 - y1)

            # no need to check if area cannot improve the answer
            if area <= max_area:
                continue

            # set of the four corners for quick skip
            corner_set = set((p[0], p[1]) for p in comb)

            valid = True
            for (x, y) in point_set:
                if (x, y) in corner_set:
                    continue
                # point lies inside or on the border -> invalid
                if x1 <= x <= x2 and y1 <= y <= y2:
                    valid = False
                    break

            if valid:
                max_area = area

        return max_area
```
