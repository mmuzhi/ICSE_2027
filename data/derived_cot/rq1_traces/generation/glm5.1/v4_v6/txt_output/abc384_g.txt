import sys
from array import array

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    B = [int(next(it)) for _ in range(N)]
    K = int(next(it))
    queries = []
    for _ in range(K):
        X = int(next(it))
        Y = int(next(it))
        queries.append((X, Y))

    # Prefix sums of A and B
    PA = [0] * (N + 1)
    for i in range(N):
        PA[i + 1] = PA[i] + A[i]
    PB = [0] * (N + 1)
    for i in range(N):
        PB[i + 1] = PB[i] + B[i]

    # Block size
    B_size = 800
    MA = (N + B_size - 1) // B_size
    MB = (N + B_size - 1) // B_size

    # Sorted blocks
    sorted_A_blocks = []
    for a in range(MA):
        start = a * B_size
        end = min(N, start + B_size)
        block = A[start:end]
        block.sort()
        sorted_A_blocks.append(block)

    sorted_B_blocks = []
    for b in range(MB):
        start = b * B_size
        end = min(N, start + B_size)
        block = B[start:end]
        block.sort()
        sorted_B_blocks.append(block)

    # 2D prefix sums of block-block sum of mins
    P = [[0] * (MB + 1) for _ in range(MA + 1)]
    for a in range(MA):
        arr1 = sorted_A_blocks[a]
        for b in range(MB):
            arr2 = sorted_B_blocks[b]
            len2 = len(arr2)
            i = 0
            sum2 = 0
            cnt2 = 0
            total = 0

                while i < len2 and arr2[i] <= x:
                    sum2 += arr2[i]
                    cnt2 += 1
                    i += 1
                total += sum2 + x * (len2 - cnt2)
            P[a + 1][b + 1] = P[a][b + 1] + P[a + 1][b] - P[a][b] + total

    # Precompute for each block of B the prefix sums of sum of min with A elements
    sorted_A_vals = [0] * N
    sorted_A_idx = [0] * N
    for i, (v, idx) in enumerate(sorted([(A[i], i) for i in range(N)])):
        sorted_A_vals[i] = v
        sorted_A_idx[i] = idx

    prefA = array('q', [0]) * (MB * N)

    sorted_B_vals = [0] * N
    sorted_B_idx = [0] * N
    for i, (v, idx) in enumerate(sorted([(B[i], i) for i in range(N)])):
        sorted_B_vals[i] = v
        sorted_B_idx[i] = idx

    prefB = array('q', [0]) * (MA * N)
    for a in range(MA):

        len1 = len(arr1)
        i = 0
        sum1 = 0

        for pos in range(N):

                cnt1 += 1

        offset = a * N
        prefB[offset] = val[0]
        for j in range(1, N):

    out_lines = []
    for X, Y in queries:
        qX = X // B_size
        rX = X % B_size
        qY = Y // B_size

        total = P[qX][qY]
