        inv[i] = MOD - (MOD // i) * inv[MOD % i] % MOD

    # T = (N-1)! modulo MOD
    T = 1
    for i in range(2, N):
        T = T * i % MOD

    # P[i] = sum_{k=2}^{i} A_k * inv[k]
    P = [0] * (N + 1)
    for i in range(2, N + 1):
        P[i] = (P[i - 1] + A[i] * inv[i]) % MOD

    # Q[i] = sum_{k=2}^{i} A_k * 2 * inv[k] * inv[k+1]   (only up to N-1)
    Q_arr = [0] * (N + 1)
    for i in range(2, N):
        term = A[i] * 2 % MOD * inv[i] % MOD * inv[i + 1] % MOD
        Q_arr[i] = (Q_arr[i - 1] + term) % MOD

    # S[u] = sum over trees of dist(u)
    S = [0] * (N + 1)
    for u in range(2, N + 1):
        S[u] = T * (A[u] + P[u - 1]) % MOD

    # C[u] = sum over trees of dist(lca(u, v)) for any v > u
    C = [0] * (N + 1)
    for u in range(2, N + 1):
        val = (A[u] * inv[u] + Q_arr[u - 1]) % MOD
        C[u] = T * val % MOD

    out_lines = []
    for _ in range(Q):
        u = int(next(it))
        v = int(next(it))
        ans = (S[u] + S[v] - 2 * C[u]) % MOD
        out_lines.append(str(ans))

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
