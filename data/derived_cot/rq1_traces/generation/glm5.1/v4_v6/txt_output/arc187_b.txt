import sys

def solve():
    MOD = 998244353
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    N = int(input_data[0])
    M = int(input_data[1])
    B = list(map(int, input_data[2:]))
    q = B.count(-1)
    total = pow(M, q, MOD)
    
    # Precompute for each index if fixed or not
    # We can just use B as is.
    
    for v in range(1, M):
        # Build L and Lm
        L = [0] * N
        if v > 1:
            Lm = [0] * N
        else:
            Lm = None
            
        for i in range(N):
            b = B[i]
            if b == -1:
                L[i] = v
                if Lm is not None:
                    Lm[i] = v - 1
            else:
                if b <= v:
                    L[i] = 1
                else:
                    L[i] = 0
                if Lm is not None:
                    if b <= v - 1:
                        Lm[i] = 1
                    else:
                        Lm[i] = 0
        
        # Suffix products
        suffL = [1] * (N + 1)
        if Lm is not None:
            suffLm = [1] * (N + 1)
        else:
            suffLm = None
            
        for i in range(N - 1, -1, -1):
            suffL[i] = (suffL[i + 1] * L[i]) % MOD
            if suffLm is not None:
                suffLm[i] = (suffLm[i + 1] * Lm[i]) % MOD
                
        # Forward pass
        pref = 1
        sum1 = 0
        sum2 = 0
        for k in range(1, N):
            b = B[k - 1]
            if b == -1:
                g = M - v
            else:
                g = 1 if b >= v + 1 else 0
            pref = (pref * g) % MOD
            sum1 = (sum1 + pref * suffL[k]) % MOD
            if suffLm is not None:
                sum2 = (sum2 + pref * suffLm[k]) % MOD
                
        total = (total + sum1 - sum2) % MOD
        
    print(total)


if __name__ == "__main__":

import sys

def solve() -> None:

    sum1 = 0

            g = 1 if b >= 2 else 0

    # Process v = 2 .. M-1

        L = [0] * N
