def solve():

    N = int(input().strip())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    C = list(map(int, input().split()))

    P = []  # 1 -> 0
    Q = []  # 0 -> 1

    for i in range(N):
        if A[i] == 1 and B[i] == 0:
            P.append(C[i])
        elif A[i] == 0 and B[i] == 1:
            Q.append(C[i])

    # Sort P descending, Q ascending
    P.sort(reverse=True)
    Q.sort()

    current_sum = sum(A[i] * C[i] for i in range(N))
    total_cost = 0

    for c in P:
        current_sum -= c
        total_cost += current_sum

    for c in Q:
        current_sum += c
        total_cost += current_sum

    print(total_cost)

if __name__ == "__main__":
    solve()
