You will be given a competitive programming problem.
There are N types of items. The i-th type of item has a weight of w_i and a value of v_i. Each type has 10^{10} items available.
Takahashi is going to choose some items and put them into a bag with capacity W. He wants to maximize the value of the selected items while avoiding choosing too many items of the same type. Hence, he defines the happiness of choosing k_i items of type i as k_i v_i - k_i^2. He wants to choose items to maximize the total happiness over all types while keeping the total weight at most W. Calculate the maximum total happiness he can achieve.

Input

The input is given from Standard Input in the following format:
N W
w_1 v_1
w_2 v_2
\vdots
w_N v_N

Output

Print the answer.

Constraints


- 1 \leq N \leq 3000
- 1 \leq W \leq 3000
- 1 \leq w_i \leq W
- 1 \leq v_i \leq 10^9
- All input values are integers.

Sample Input 1

2 10
3 4
3 2

Sample Output 1

5

By choosing 2 items of type 1 and 1 item of type 2, the total happiness can be 5, which is optimal.
Here, the happiness for type 1 is 2 \times 4 - 2^2 = 4, and the happiness for type 2 is 1 \times 2 - 1^2 = 1.
The total weight is 9, which is within the capacity 10.

Sample Output 2

import sys
from collections import deque

def solve():
    input = sys.stdin.readline
    N, W = map(int, input().split())
    items = [tuple(map(int, input().split())) for _ in range(N)]
    NEG = -10**18
    dp = [NEG] * (W+1)
    dp[0] = 0
    for w, v in items:
        ndp = [NEG] * (W+1)
        for r in range(w):
            dq = deque()  # stores (a, b) where a = m, b = A[m]
            t = 0
            idx = r
            while idx <= W:
                prev_val = dp[idx]
                if prev_val > NEG//2:  # reachable
                    a = t
                    b = prev_val - v*t - t*t
                    # add line (a,b)
                    while len(dq) >= 2:
                        a1, b1 = dq[-2]
                        a2, b2 = dq[-1]
                        # if line (a2,b2) is unnecessary
                        if (b1 - b2) * (a - a2) >= (b2 - b) * (a2 - a1):
                            dq.pop()
                        else:
                            break
                    dq.append((a, b))
                # query
                if dq:
                    x = 2*t
                    # pop from front while next line is better
                    while len(dq) >= 2:
                        a1, b1 = dq[0]
                        a2, b2 = dq[1]
                        if a1 * x + b1 <= a2 * x + b2:
                            dq.popleft()
                        else:
                            break
                    a_best, b_best = dq[0]
                    best = a_best * x + b_best
                else:
                    best = NEG
                if best > NEG//2:
                    ndp[idx] = v*t - t*t + best
                else:
                    ndp[idx] = NEG
                t += 1
                idx += w
        dp = ndp
    ans = max(dp)
    print(ans)

```python
import sys
from collections import deque

def solve() -> None:
    input = sys.stdin.readline
    N, W = map(int, input().split())
    items = [tuple(map(int, input().split())) for _ in range(N)]

    NEG = -10**18          # sentinel for unreachable states
    dp = [NEG] * (W + 1)
    dp[0] = 0

    for w, v in items:
        ndp = [NEG] * (W + 1)

        # process each residue class modulo w
        for r in range(w):
            dq = deque()          # stores lines (slope = m, intercept = A[m])
            t = 0
            idx = r
            while idx <= W:
                prev = dp[idx]

                # add line for m = t if the state is reachable
                if prev != NEG:
                    a = t
                    b = prev - v * t - t * t   # A[t]
                    # maintain upper envelope: remove last line if it becomes useless
                    while len(dq) >= 2:
                        a1, b1 = dq[-2]
                        a2, b2 = dq[-1]
                        # condition: intersection(L1,L2) >= intersection(L2,L3)
                        if (b1 - b2) * (a - a2) >= (b2 - b) * (a2 - a1):
                            dq.pop()
                        else:
                            break
                    dq.append((a, b))

                # query the best line for x = 2*t
                if dq:
                    x = 2 * t
                    # remove front line if it is worse than the next one at x
                    while len(dq) >= 2:
                        a1, b1 = dq[0]
                        a2, b2 = dq[1]
                        if a1 * x + b1 <= a2 * x + b2:
                            dq.popleft()
                        else:
                            break
                    a_best, b_best = dq[0]
                    best = a_best * x + b_best
                else:
                    best = NEG

                if best != NEG:
                    ndp[idx] = v * t - t * t + best
                # else remains NEG

                t += 1
                idx += w

        dp = ndp

    ans = max(dp)   # maximum over all weights <= W
    print(ans)

if __name__ == "__main__":
    solve()
```
