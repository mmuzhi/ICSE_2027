- 1 \leq N \leq 2\times 10^5
- 1 \leq M \leq 2\times 10^5
- -10^9 \leq X_i,Y_i \leq 10^9
- The pairs (X_i,Y_i) are distinct.
- -10^9 \leq S_x,S_y \leq 10^9
- There is no house at (S_x,S_y).

- 1 \leq C_i \leq 10^9
- All input numbers are integers.

import sys
import bisect
from collections import defaultdict

# -------------------------------------------------------------

class AxisGroup:
    """Maintains a sorted list of coordinates with DSU for deletions."""
    __slots__ = ('coords', 'ids', 'parent')

    def __init__(self, coords, ids):
        self.coords = coords          # sorted list of coordinates
        self.ids = ids                # house ids in the same order
        n = len(coords)
        self.parent = list(range(n + 1))   # DSU, parent[n] = n (sentinel)

    def find(self, x):
        """return the smallest alive index >= x (or n if none)"""
        # find root
        root = x
        while self.parent[root] != root:
            root = self.parent[root]
        # path compression
        while self.parent[x] != x:
            nxt = self.parent[x]
            self.parent[x] = root
            x = nxt
        return root

    def erase(self, x):
        """mark index x as deleted"""
        self.parent[x] = self.find(x + 1)

# -------------------------------------------------------------

def main():
    input = sys.stdin.readline

    N, M, Sx, Sy = map(int, input().split())

    house_x = [0] * N
    house_y = [0] * N

    x_to_pairs = defaultdict(list)   # x -> list of (y, id)
    y_to_pairs = defaultdict(list)   # y -> list of (x, id)

    for i in range(N):
        x, y = map(int, input().split())
        house_x[i] = x
        house_y[i] = y
        x_to_pairs[x].append((y, i))
        y_to_pairs[y].append((x, i))

    # Build groups for vertical moves (key = x)
    dict_x = {}
    house_x_idx = [0] * N
    for x, pairs in x_to_pairs.items():
        pairs.sort(key=lambda p: p[0])          # sort by y
        ys = [p[0] for p in pairs]
        ids = [p[1] for p in pairs]
        for idx, hid in enumerate(ids):
            house_x_idx[hid] = idx
        dict_x[x] = AxisGroup(ys, ids)

    # Build groups for horizontal moves (key = y)
    dict_y = {}
    house_y_idx = [0] * N
    for y, pairs in y_to_pairs.items():
        pairs.sort(key=lambda p: p[0])          # sort by x
        xs = [p[0] for p in pairs]
        ids = [p[1] for p in pairs]
        for idx, hid in enumerate(ids):
            house_y_idx[hid] = idx
        dict_y[y] = AxisGroup(xs, ids)

    # Simulate moves
    cur_x, cur_y = Sx, Sy
    visited_cnt = 0

    for _ in range(M):
        parts = input().split()
        D = parts[0]
        C = int(parts[1])

        if D == 'U' or D == 'D':
            # vertical move
            if D == 'U':
                new_y = cur_y + C
            else:
                new_y = cur_y - C
            y_start, y_end = (cur_y, new_y) if cur_y <= new_y else (new_y, cur_y)
            x = cur_x

            if x in dict_x:
                obj = dict_x[x]
                left = bisect.bisect_left(obj.coords, y_start)
                right = bisect.bisect_right(obj.coords, y_end) - 1
                if left <= right:
                    i = obj.find(left)
                    while i <= right:
                        hid = obj.ids[i]
                        visited_cnt += 1

                        # delete from x‑group
                        obj.erase(i)

                        # delete from y‑group
                        y_val = house_y[hid]
                        y_obj = dict_y[y_val]
                        y_idx = house_y_idx[hid]
                        y_obj.erase(y_idx)

                        i = obj.find(i)

            cur_y = new_y

        else:   # L or R
            # horizontal move
            if D == 'L':
                new_x = cur_x - C
            else:
                new_x = cur_x + C
            x_start, x_end = (cur_x, new_x) if cur_x <= new_x else (new_x, cur_x)
            y = cur_y

            if y in dict_y:
                obj = dict_y[y]
                left = bisect.bisect_left(obj.coords, x_start)
                right = bisect.bisect_right(obj.coords, x_end) - 1
                if left <= right:
                    i = obj.find(left)
                    while i <= right:
                        hid = obj.ids[i]
                        visited_cnt += 1

                        # delete from y‑group
                        obj.erase(i)

                        # delete from x‑group
                        x_val = house_x[hid]
                        x_obj = dict_x[x_val]
                        x_idx = house_x_idx[hid]
                        x_obj.erase(x_idx)

                        i = obj.find(i)

            cur_x = new_x

    print(cur_x, cur_y, visited_cnt)
