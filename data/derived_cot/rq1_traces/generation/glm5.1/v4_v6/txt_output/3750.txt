class Solution:
    def solveQueries(self, nums: List[int], queries: List[int]) -> List[int]:
        n = len(nums)
        # map value to list of indices
        val_to_indices = {}
        for idx, val in enumerate(nums):
            val_to_indices.setdefault(val, []).append(idx)
        # answer array for each index
        ans = [-1] * n  # default -1
        for val, indices in val_to_indices.items():
            m = len(indices)
            if m == 1:
                continue  # ans already -1
            # compute for each index in this list

                prev = indices[(k-1) % m]
                nxt = indices[(k+1) % m]
                # compute circular distances
                d1 = abs(i - prev)
                d1 = min(d1, n - d1)
                d2 = abs(i - nxt)
                d2 = min(d2, n - d2)
                ans[i] = min(d1, d2)
        # answer queries
        result = [ans[q] for q in queries]
        return result

from typing import List
from collections import defaultdict

class Solution:
    def solveQueries(self, nums: List[int], queries: List[int]) -> List[int]:
        n = len(nums)
        # group indices by value
        val_to_indices = defaultdict(list)
        for idx, val in enumerate(nums):
            val_to_indices[val].append(idx)

        # answer for each index, default -1 (no other occurrence)
        ans = [-1] * n

        for indices in val_to_indices.values():
            m = len(indices)
            if m == 1:
                continue   # ans already -1
            # for each occurrence compute distance to its two neighbours on the circle
            for k in range(m):
                i = indices[k]
                prev = indices[(k - 1) % m]
                nxt = indices[(k + 1) % m]

                d_prev = abs(i - prev)
                d_prev = min(d_prev, n - d_prev)
                d_next = abs(i - nxt)
                d_next = min(d_next, n - d_next)

                ans[i] = min(d_prev, d_next)

        # answer the queries
        return [ans[q] for q in queries]
