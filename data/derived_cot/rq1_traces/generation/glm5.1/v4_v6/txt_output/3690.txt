def minLength(self, s: str, numOps: int) -> int:
    n = len(s)
    s_vals = [1 if ch == '1' else 0 for ch in s]
    # optional: compute original max run to set high
    # but we can just set high = n
    low, high = 1, n
    # define can(L)
    def can(L):
        # dp0 and dp1 arrays length L+1, index 0 unused
        INF = numOps + 1  # any value > numOps
        dp0 = [INF] * (L+1)
        dp1 = [INF] * (L+1)
        # init
        first = s_vals[0]
        cost0 = 0 if first == 0 else 1
        cost1 = 0 if first == 1 else 1
        if cost0 <= numOps:
            dp0[1] = cost0
        if cost1 <= numOps:
            dp1[1] = cost1
        # iterate
        for i in range(1, n):
            cur = s_vals[i]
            next0 = [INF] * (L+1)
            next1 = [INF] * (L+1)
            # process dp0
            for j in range(1, L+1):
                cur_cost = dp0[j]
                if cur_cost == INF:
                    continue
                # keep 0
                if j+1 <= L:
                    new_cost = cur_cost + (0 != cur)
                    if new_cost <= numOps and new_cost < next0[j+1]:
                        next0[j+1] = new_cost
                # switch to 1
                new_cost = cur_cost + (1 != cur)
                if new_cost <= numOps and new_cost < next1[1]:
                    next1[1] = new_cost
            # process dp1
            for j in range(1, L+1):
                cur_cost = dp1[j]
                if cur_cost == INF:
                    continue
                # keep 1
                if j+1 <= L:
                    new_cost = cur_cost + (1 != cur)
                    if new_cost <= numOps and new_cost < next1[j+1]:
                        next1[j+1] = new_cost
                # switch to 0
                new_cost = cur_cost + (0 != cur)
                if new_cost <= numOps and new_cost < next0[1]:
                    next0[1] = new_cost
            dp0, dp1 = next0, next1
            # early exit if both dp0 and dp1 all INF? but we can continue
        # after loop
        min_flips = min(min(dp0[1:]), min(dp1[1:]))
        return min_flips <= numOps

    while low < high:
        mid = (low + high) // 2
        if can(mid):
            high = mid
        else:
            low = mid + 1
    return low

class Solution:
    def minLength(self, s: str, numOps: int) -> int:
        n = len(s)
        arr = [1 if c == '1' else 0 for c in s]
        
        def feasible(L):
            INF = numOps + 1
            # dp0[j] for last char 0, run length j; dp1 similarly
            dp0 = [INF] * (L + 1)
            dp1 = [INF] * (L + 1)
            first = arr[0]
            cost0 = 0 if first == 0 else 1
            cost1 = 0 if first == 1 else 1
            if cost0 <= numOps:
                dp0[1] = cost0
            if cost1 <= numOps:
                dp1[1] = cost1
            # iterate
            for i in range(1, n):
                cur = arr[i]
                next0 = [INF] * (L + 1)
                next1 = [INF] * (L + 1)
                # from dp0
                for j in range(1, L+1):
                    cur_cost = dp0[j]
                    if cur_cost == INF:
                        continue
                    # keep 0
                    if j + 1 <= L:
                        new_cost = cur_cost + (0 != cur)
                        if new_cost <= numOps and new_cost < next0[j+1]:
                            next0[j+1] = new_cost
                    # switch to 1
                    new_cost = cur_cost + (1 != cur)
                    if new_cost <= numOps and new_cost < next1[1]:
                        next1[1] = new_cost
                # from dp1
                for j in range(1, L+1):
                    cur_cost = dp1[j]
                    if cur_cost == INF:
                        continue
                    # keep 1
                    if j + 1 <= L:
                        new_cost = cur_cost + (1 != cur)
                        if new_cost <= numOps and new_cost < next1[j+1]:
                            next1[j+1] = new_cost
                    # switch to 0
                    new_cost = cur_cost + (0 != cur)
                    if new_cost <= numOps and new_cost < next0[1]:
                        next0[1] = new_cost
                # early exit if no feasible states
                if all(v == INF for v in next0[1:]) and all(v == INF for v in next1[1:]):
                    return False
                dp0, dp1 = next0, next1
            # after all positions
            min_flips = min(min(dp0[1:]), min(dp1[1:]))
            return min_flips <= numOps
        
        low, high = 1, n
        while low < high:
            mid = (low + high) // 2
            if feasible(mid):
                high = mid
            else:
                low = mid + 1
        return low

class Solution:
    def minLength(self, s: str, numOps: int) -> int:
        n = len(s)
        # early exit if we can achieve max run 1
        if numOps >= (n // 2):  # because min flips to alternating <= n//2
            return 1
        arr = [1 if c == '1' else 0 for c in s]
        # compute original max run to set high
        max_run = 1
        cur = 1
        for i in range(1, n):
            if s[i] == s[i-1]:
                cur += 1
                max_run = max(max_run, cur)
            else:
                cur = 1
        # binary search between 1 and max_run
        low, high = 1, max_run
        def feasible(L):
            INF = numOps + 1
            dp0 = [INF] * (L + 1)
            dp1 = [INF] * (L + 1)
            first = arr[0]
            cost0 = 0 if first == 0 else 1
            cost1 = 0 if first == 1 else 1
            if cost0 <= numOps:
                dp0[1] = cost0
            if cost1 <= numOps:
                dp1[1] = cost1
            for i in range(1, n):
                cur = arr[i]
                next0 = [INF] * (L + 1)
                next1 = [INF] * (L + 1)
                # from dp0
                for j in range(1, L+1):
                    cur_cost = dp0[j]
                    if cur_cost == INF:
                        continue
                    # keep 0
                    if j + 1 <= L:
                        new_cost = cur_cost + (0 != cur)
                        if new_cost <= numOps and new_cost < next0[j+1]:
                            next0[j+1] = new_cost
                    # switch to 1
                    new_cost = cur_cost + (1 != cur)
                    if new_cost <= numOps and new_cost < next1[1]:
                        next1[1] = new_cost
                # from dp1
                for j in range(1, L+1):
                    cur_cost = dp1[j]
                    if cur_cost == INF:
                        continue
                    # keep 1
                    if j + 1 <= L:
                        new_cost = cur_cost + (1 != cur)
                        if new_cost <= numOps and new_cost < next1[j+1]:
                            next1[j+1] = new_cost
                    # switch to 0
                    new_cost = cur_cost + (0 != cur)
                    if new_cost <= numOps and new_cost < next0[1]:
                        next0[1] = new_cost
                # early exit if no states
                if all(v == INF for v in next0[1:]) and all(v == INF for v in next1[1:]):
                    return False
                dp0, dp1 = next0, next1
            min_flips = min(min(dp0[1:]), min(dp1[1:]))
            return min_flips <= numOps
        while low < high:
            mid = (low + high) // 2
            if feasible(mid):
                high = mid
            else:
                low = mid + 1
        return low

class Solution:
    def minLength(self, s: str, numOps: int) -> int:
        n = len(s)
        # If we can afford at least floor(n/2) flips, we can always make the string alternating
        # (max run = 1), because the minimum flips to an alternating string is ≤ floor(n/2).
        if numOps >= n // 2:
            return 1

        # Convert to list of ints for fast access
        arr = [1 if ch == '1' else 0 for ch in s]

        # Compute the original longest run – this is an upper bound for the answer
        max_run = 1
        cur = 1
        for i in range(1, n):
            if s[i] == s[i-1]:
                cur += 1
                max_run = max(max_run, cur)
            else:
                cur = 1

        # Binary search on the answer L
        low, high = 1, max_run

        # Feasibility test: can we achieve max run ≤ L with ≤ numOps flips?
        def feasible(L: int) -> bool:
            INF = numOps + 1          # any value > numOps works as "infinity"
            # dp0[j] = min flips for prefix ending with '0' and current run length j
            # dp1[j] = same for ending with '1'
            dp0 = [INF] * (L + 1)
            dp1 = [INF] * (L + 1)

            first = arr[0]
            cost0 = 0 if first == 0 else 1
            cost1 = 0 if first == 1 else 1
            if cost0 <= numOps:
                dp0[1] = cost0
            if cost1 <= numOps:
                dp1[1] = cost1

            for i in range(1, n):
                cur = arr[i]
                next0 = [INF] * (L + 1)
                next1 = [INF] * (L + 1)

                # Transitions from states ending with '0'
                for j in range(1, L + 1):
                    cur_cost = dp0[j]
                    if cur_cost == INF:
                        continue
                    # Keep '0' – run length increases
                    if j + 1 <= L:
                        new_cost = cur_cost + (0 != cur)
                        if new_cost <= numOps and new_cost < next0[j+1]:
                            next0[j+1] = new_cost
                    # Switch to '1' – start a new run of length 1
                    new_cost = cur_cost + (1 != cur)
                    if new_cost <= numOps and new_cost < next1[1]:
                        next1[1] = new_cost

                # Transitions from states ending with '1'
                for j in range(1, L + 1):
                    cur_cost = dp1[j]
                    if cur_cost == INF:
                        continue
                    # Keep '1'
                    if j + 1 <= L:
                        new_cost = cur_cost + (1 != cur)
                        if new_cost <= numOps and new_cost < next1[j+1]:
                            next1[j+1] = new_cost
                    # Switch to '0'
                    new_cost = cur_cost + (0 != cur)
                    if new_cost <= numOps and new_cost < next0[1]:
                        next0[1] = new_cost

                # Early exit: no state within budget
                if all(v == INF for v in next0[1:]) and all(v == INF for v in next1[1:]):
                    return False

                dp0, dp1 = next0, next1

            # Minimum flips over all final states
            min_flips = min(min(dp0[1:]), min(dp1[1:]))
            return min_flips <= numOps

        while low < high:
            mid = (low + high) // 2
            if feasible(mid):
                high = mid
            else:
                low = mid + 1

        return low
