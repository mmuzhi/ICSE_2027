```python
class Solution:
    def maxDistance(self, s: str, k: int) -> int:
        ans = 0
        # Four sign pairs (a,b) corresponding to the four linear forms:
        # (1,1)  ->  x+y   (NE quadrant)
        # (1,-1) ->  x-y   (SE quadrant)
        # (-1,1) -> -x+y   (NW quadrant)
        # (-1,-1)-> -x-y   (SW quadrant)
        for a, b in [(1, 1), (1, -1), (-1, 1), (-1, -1)]:
            cnt_neg = 0          # number of -1 contributions in the prefix
            best = 0             # maximum value of the linear form over prefixes
            for i, ch in enumerate(s):
                # contribution of this step to a*x + b*y
                if ch == 'N':
                    c = b
                elif ch == 'S':
                    c = -b
                elif ch == 'E':
                    c = a
                else:  # 'W'
                    c = -a

                if c == -1:
                    cnt_neg += 1

                t = i + 1        # length of current prefix
                # After optimally flipping the first k '-1's, the prefix sum is:
                if cnt_neg <= k:
                    cur = t
                else:
                    cur = t - 2 * (cnt_neg - k)

                if cur > best:
                    best = cur

            # The starting point (0,0) gives value 0, so the maximum is at least 0.
            if best < 0:
                best = 0
            ans = max(ans, best)

        return ans
```
