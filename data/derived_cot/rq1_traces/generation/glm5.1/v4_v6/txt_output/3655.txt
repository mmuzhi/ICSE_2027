import heapq

class Solution:
    def minOperations(self, n: int, m: int) -> int:
        if n == m:
            # check if n is prime? The problem says "The integer n must not be a prime number at any point, including its original value". So if n is prime, we can't even start. So return -1.
            pass
        D = len(str(n))
        low = 10**(D-1)
        high = 10**D - 1
        
        # Sieve up to high
        is_prime = [True] * (high + 1)
        is_prime[0] = is_prime[1] = False
        for i in range(2, int(high**0.5) + 1):
            if is_prime[i]:
                for j in range(i*i, high+1, i):
                    is_prime[j] = False
                    
        if is_prime[n] or is_prime[m]:
            return -1
            
        # Dijkstra
        dist = {}
        for i in range(low, high+1):
            if not is_prime[i]:
                dist[i] = float('inf')
        dist[n] = n
        heap = [(n, n)]
        
        while heap:
            cost, u = heapq.heappop(heap)
            if cost > dist[u]:
                continue
            if u == m:
                return cost
            # generate neighbors
            s = str(u)
            for i in range(D):
                digit = int(s[i])
                if digit < 9:
                    new_digit = digit + 1
                    new_s = s[:i] + str(new_digit) + s[i+1:]
                    v = int(new_s)
                    if v in dist and cost + v < dist[v]:
                        dist[v] = cost + v
                        heapq.heappush(heap, (dist[v], v))
                if digit > 0:
                    new_digit = digit - 1
                    new_s = s[:i] + str(new_digit) + s[i+1:]
                    v = int(new_s)
                    if v in dist and cost + v < dist[v]:
                        dist[v] = cost + v
                        heapq.heappush(heap, (dist[v], v))
        return -1

import heapq

class Solution:
    def minOperations(self, n: int, m: int) -> int:
        D = len(str(n))
        low = 10**(D-1)
        high = 10**D - 1
        
        # Sieve of Eratosthenes
        is_prime = [True] * (high + 1)
        if high >= 0:
            is_prime[0] = False
        if high >= 1:
            is_prime[1] = False
        for i in range(2, int(high**0.5) + 1):
            if is_prime[i]:
                for j in range(i*i, high+1, i):
                    is_prime[j] = False
                    
        if is_prime[n] or is_prime[m]:
            return -1
            
        # Dijkstra
        dist = {}
        for i in range(low, high+1):
            if not is_prime[i]:
                dist[i] = float('inf')
        dist[n] = n
        heap = [(n, n)]
        
        while heap:
            cost, u = heapq.heappop(heap)
            if cost > dist[u]:
                continue
            if u == m:
                return cost
            s = str(u)
            # Ensure s has length D. Since u is in [low, high], it should have length D.
            for i in range(D):
                digit = int(s[i])
                if digit < 9:
                    new_digit = digit + 1
                    new_s = s[:i] + str(new_digit) + s[i+1:]
                    v = int(new_s)
                    if v in dist and cost + v < dist[v]:
                        dist[v] = cost + v
                        heapq.heappush(heap, (dist[v], v))
                if digit > 0:
                    new_digit = digit - 1
                    new_s = s[:i] + str(new_digit) + s[i+1:]
                    v = int(new_s)
                    if v in dist and cost + v < dist[v]:
                        dist[v] = cost + v
                        heapq.heappush(heap, (dist[v], v))
        return -1
