This problem is a harder version of Problem C. Here, the sequence is split into three subarrays.

You are given an integer sequence of length N: A = (A_1, A_2, \ldots, A_N).
When splitting A at two positions into three non-empty (contiguous) subarrays, find the maximum possible sum of the counts of distinct integers in those subarrays.
More formally, find the maximum sum of the following three values for a pair of integers (i,j) such that 1 \leq i < j \leq N-1: the count of distinct integers in (A_1, A_2, \ldots, A_i), the count of distinct integers in (A_{i+1},A_{i+2},\ldots,A_j), and the count of distinct integers in (A_{j+1},A_{j+2},\ldots,A_{N}).

Input

The input is given from Standard Input in the following format:
N
A_1 A_2 \ldots A_N

Output

Print the answer.

Constraints


- 3 \leq N \leq 3 \times 10^5
- 1 \leq A_i \leq N (1 \leq i \leq N)
- All input values are integers.

Sample Input 1

5
3 1 4 1 5

Sample Output 1

5

This sum cannot be greater than 5, so the answer is 5. Other partitions, such as (i,j) = (1,3), (2,3), (3,4), also achieve this sum.

class SegTree:
    def __init__(self, data):
        self.n = len(data)
        self.size = 1
        while self.size < self.n:
            self.size <<= 1
        # We'll use 2*size for tree and lazy, but with lazy we need to handle range updates.
        # However, iterative segment tree with lazy is more complex. I'll stick with recursive with 4*n.
        self.tree = [0] * (4 * self.n)
        self.lazy = [0] * (4 * self.n)
        self._build(1, 0, self.n-1, data)

    def _build(self, node, l, r, data):
        if l == r:
            self.tree[node] = data[l]
        else:
            mid = (l + r) // 2
            self._build(node*2, l, mid, data)
            self._build(node*2+1, mid+1, r, data)
            self.tree[node] = max(self.tree[node*2], self.tree[node*2+1])

    def _push(self, node):
        if self.lazy[node] != 0:
            val = self.lazy[node]
            left = node*2
            right = node*2+1
            self.tree[left] += val
            self.lazy[left] += val
            self.tree[right] += val
            self.lazy[right] += val
            self.lazy[node] = 0

    def _update(self, node, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.tree[node] += val
            self.lazy[node] += val
            return
        self._push(node)
        mid = (l + r) // 2
        if ql <= mid:
            self._update(node*2, l, mid, ql, qr, val)
        if qr > mid:
            self._update(node*2+1, mid+1, r, ql, qr, val)
        self.tree[node] = max(self.tree[node*2], self.tree[node*2+1])

    def update(self, l, r, val):
        # public method, assume 0-indexed inclusive
        self._update(1, 0, self.n-1, l, r, val)

    def _query(self, node, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.tree[node]
        self._push(node)
        mid = (l + r) // 2
        res = -10**18
        if ql <= mid:
            res = max(res, self._query(node*2, l, mid, ql, qr))
        if qr > mid:
            res = max(res, self._query(node*2+1, mid+1, r, ql, qr))
        return res

    def query(self, l, r):
        return self._query(1, 0, self.n-1, l, r)

import sys

def main():
    import sys
    sys.setrecursionlimit(1 << 25)
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    A = [0] + [int(next(it)) for _ in range(N)]

    # pref
    pref = [0]*(N+1)
    seen = [False]*(N+1)
    cnt = 0
    for i in range(1, N+1):
        if not seen[A[i]]:
            seen[A[i]] = True
            cnt += 1
        pref[i] = cnt

    # suff
    suff = [0]*(N+2)
    seen2 = [False]*(N+1)
    cnt = 0
    for i in range(N, 0, -1):
        if not seen2[A[i]]:
            seen2[A[i]] = True
            cnt += 1
        suff[i] = cnt
    suff[N+1] = 0

    # segment tree
    class SegTree:
        def __init__(self, data):
            self.n = len(data)
            self.tree = [0]*(4*self.n)
            self.lazy = [0]*(4*self.n)
            self._build(1, 0, self.n-1, data)

        def _build(self, node, l, r, data):
            if l == r:
                self.tree[node] = data[l]
            else:
                mid = (l+r)//2
                self._build(node*2, l, mid, data)
                self._build(node*2+1, mid+1, r, data)
                self.tree[node] = max(self.tree[node*2], self.tree[node*2+1])

        def _push(self, node):
            if self.lazy[node] != 0:
                val = self.lazy[node]
                left = node*2
                right = node*2+1
                self.tree[left] += val
                self.lazy[left] += val
                self.tree[right] += val
                self.lazy[right] += val
                self.lazy[node] = 0

        def _update(self, node, l, r, ql, qr, val):
            if ql <= l and r <= qr:
                self.tree[node] += val
                self.lazy[node] += val
                return
            self._push(node)
            mid = (l+r)//2
            if ql <= mid:
                self._update(node*2, l, mid, ql, qr, val)
            if qr > mid:
                self._update(node*2+1, mid+1, r, ql, qr, val)
            self.tree[node] = max(self.tree[node*2], self.tree[node*2+1])

        def update(self, l, r, val):
            if l > r:
                return
            self._update(1, 0, self.n-1, l, r, val)

        def _query(self, node, l, r, ql, qr):
            if ql <= l and r <= qr:
                return self.tree[node]
            self._push(node)
            mid = (l+r)//2
            res = -10**18
            if ql <= mid:
                res = max(res, self._query(node*2, l, mid, ql, qr))
            if qr > mid:
                res = max(res, self._query(node*2+1, mid+1, r, ql, qr))
            return res

        def query(self, l, r):
            if l > r:
                return -10**18
            return self._query(1, 0, self.n-1, l, r)

    seg = SegTree(pref)  # data length N+1

    last = [0]*(N+1)
    ans = 0

    for j in range(1, N+1):
        x = A[j]
        L = last[x]
        # update range [L, j-1] by +1
        if L <= j-1:
            seg.update(L, j-1, 1)
        last[x] = j
        if 2 <= j <= N-1:
            best = seg.query(1, j-1)
            candidate = best + suff[j+1]
            if candidate > ans:
                ans = candidate

    print(ans)

if __name__ == "__main__":
    main()

import sys

sys.setrecursionlimit(1 << 25)

def main():
    it = iter(sys.stdin.read().split())
    N = int(next(it))
    A = [0] + [int(next(it)) for _ in range(N)]

    # prefix distinct counts
    pref = [0] * (N + 1)
    seen = [False] * (N + 1)
    cnt = 0
    for i in range(1, N + 1):
        if not seen[A[i]]:
            seen[A[i]] = True
            cnt += 1
        pref[i] = cnt

    # suffix distinct counts
    suff = [0] * (N + 2)
    seen2 = [False] * (N + 1)
    cnt = 0
    for i in range(N, 0, -1):
        if not seen2[A[i]]:
            seen2[A[i]] = True
            cnt += 1
        suff[i] = cnt
    suff[N + 1] = 0

    # ----- segment tree with lazy range add and range max -----
    class SegTree:
        def __init__(self, data):
            self.n = len(data)
            self.tree = [0] * (4 * self.n)
            self.lazy = [0] * (4 * self.n)
            self._build(1, 0, self.n - 1, data)

        def _build(self, node, l, r, data):
            if l == r:
                self.tree[node] = data[l]
            else:
                mid = (l + r) // 2
                self._build(node * 2, l, mid, data)
                self._build(node * 2 + 1, mid + 1, r, data)
                self.tree[node] = max(self.tree[node * 2], self.tree[node * 2 + 1])

        def _push(self, node):
            if self.lazy[node] != 0:
                val = self.lazy[node]
                left = node * 2
                right = left + 1
                self.tree[left] += val
                self.lazy[left] += val
                self.tree[right] += val
                self.lazy[right] += val
                self.lazy[node] = 0

        def _update(self, node, l, r, ql, qr, val):
            if ql <= l and r <= qr:
                self.tree[node] += val
                self.lazy[node] += val
                return
            self._push(node)
            mid = (l + r) // 2
            if ql <= mid:
                self._update(node * 2, l, mid, ql, qr, val)
            if qr > mid:
                self._update(node * 2 + 1, mid + 1, r, ql, qr, val)
            self.tree[node] = max(self.tree[node * 2], self.tree[node * 2 + 1])

        def update(self, l, r, val):
            if l > r:
                return
            self._update(1, 0, self.n - 1, l, r, val)

        def _query(self, node, l, r, ql, qr):
            if ql <= l and r <= qr:
                return self.tree[node]
            self._push(node)
            mid = (l + r) // 2
            res = -10 ** 18
            if ql <= mid:
                res = max(res, self._query(node * 2, l, mid, ql, qr))
            if qr > mid:
                res = max(res, self._query(node * 2 + 1, mid + 1, r, ql, qr))
            return res

        def query(self, l, r):
            if l > r:
                return -10 ** 18
            return self._query(1, 0, self.n - 1, l, r)

    # initial values: pref[i] (cur[i] = 0)
    seg = SegTree(pref)

    last = [0] * (N + 1)          # last occurrence of each value
    ans = 0

    for j in range(1, N + 1):
        x = A[j]
        L = last[x]               # previous occurrence (0 if none)
        if L <= j - 1:
            seg.update(L, j - 1, 1)   # add 1 to cur[i] for i in [L, j-1]
        last[x] = j

        if 2 <= j <= N - 1:       # valid split point for the second cut
            best = seg.query(1, j - 1)   # max_{1 ≤ i < j} (pref[i] + cur[i])
            cand = best + suff[j + 1]
            if cand > ans:
                ans = cand

    print(ans)

if __name__ == "__main__":
    main()
