```
import sys

MOD = 998244353

def solve():
    import sys
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    N = int(input_data[0])
    M = int(input_data[1])
    S = input_data[2].strip()
    # Precompute transitions
    max_mask = 1 << N
    trans = [[0]*26 for _ in range(max_mask)]
    # For each mask
    for mask in range(max_mask):
        # compute dp from mask
        dp = [0]*(N+1)
        for i in range(1, N+1):
            dp[i] = dp[i-1] + ((mask >> (i-1)) & 1)
        # for each character
        for c in range(26):
            ch = chr(ord('a')+c)
            new_dp = [0]*(N+1)
            for i in range(1, N+1):
                new_dp[i] = max(dp[i], new_dp[i-1])
                if S[i-1] == ch:
                    new_dp[i] = max(new_dp[i], dp[i-1] + 1)
            new_mask = 0
            for i in range(1, N+1):
                if new_dp[i] > new_dp[i-1]:
                    new_mask |= (1 << (i-1))
            trans[mask][c] = new_mask
    # DP over length
    dp_cur = [0]*max_mask
    dp_cur[0] = 1
    for step in range(M):
        dp_next = [0]*max_mask
        for mask in range(max_mask):
            if dp_cur[mask] == 0:
                continue
            val = dp_cur[mask]
            # iterate over characters
            # we can use trans[mask] list
            for c in range(26):
                nm = trans[mask][c]
                dp_next[nm] = (dp_next[nm] + val) % MOD
        dp_cur = dp_next
    # Compute answers
    popcnt = [0]*max_mask
    for mask in range(max_mask):
        popcnt[mask] = bin(mask).count('1')
    ans = [0]*(N+1)
    for mask in range(max_mask):
        k = popcnt[mask]
        ans[k] = (ans[k] + dp_cur[mask]) % MOD
    print(' '.join(str(x) for x in ans))
```

```
import sys

MOD = 998244353

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    S = data[2].strip()
    max_mask = 1 << N
    # Precompute transition for each mask and each character
    trans = [[0]*26 for _ in range(max_mask)]
    for mask in range(max_mask):
        # compute dp from mask
        dp = [0]*(N+1)
        for i in range(1, N+1):
            dp[i] = dp[i-1] + ((mask >> (i-1)) & 1)
        for c in range(26):
            ch = chr(ord('a')+c)
            new_dp = [0]*(N+1)
            for i in range(1, N+1):
                # start with max of dp[i] and new_dp[i-1]
                best = dp[i]
                if new_dp[i-1] > best:
                    best = new_dp[i-1]
                if S[i-1] == ch:
                    if dp[i-1] + 1 > best:
                        best = dp[i-1] + 1
                new_dp[i] = best
            # compute new mask
            new_mask = 0
            for i in range(1, N+1):
                if new_dp[i] > new_dp[i-1]:
                    new_mask |= (1 << (i-1))
            trans[mask][c] = new_mask
    # DP over length
    dp_cur = [0]*max_mask
    dp_cur[0] = 1
    for _ in range(M):
        dp_next = [0]*max_mask
        for mask in range(max_mask):
            val = dp_cur[mask]
            if val == 0:
                continue
            # iterate over characters
            for c in range(26):
                nm = trans[mask][c]
                dp_next[nm] = (dp_next[nm] + val) % MOD
        dp_cur = dp_next
    # Compute answers by popcount
    popcnt = [0]*max_mask
    for mask in range(max_mask):
        popcnt[mask] = bin(mask).count('1')
    ans = [0]*(N+1)
    for mask in range(max_mask):
        k = popcnt[mask]
        ans[k] = (ans[k] + dp_cur[mask]) % MOD
    print(' '.join(str(x) for x in ans))

if __name__ == "__main__":
    solve()
```

```python

import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    S = data[2].strip()

    max_mask = 1 << N

    # ---------- precompute transitions ----------
    trans = [[0] * 26 for _ in range(max_mask)]

    for mask in range(max_mask):
        # dp[i] = LCS(S[0..i-1], current prefix of T)
        dp = [0] * (N + 1)
        for i in range(1, N + 1):
            dp[i] = dp[i-1] + ((mask >> (i-1)) & 1)

        for c in range(26):
            ch = chr(ord('a') + c)
            new_dp = [0] * (N + 1)
            for i in range(1, N + 1):
                # recurrence: new_dp[i] = max(dp[i], new_dp[i-1], dp[i-1] + (S[i-1]==ch))
                best = dp[i]
                if new_dp[i-1] > best:
                    best = new_dp[i-1]
                if S[i-1] == ch and dp[i-1] + 1 > best:
                    best = dp[i-1] + 1
                new_dp[i] = best

            # convert new_dp to mask
            new_mask = 0
            for i in range(1, N + 1):
                if new_dp[i] > new_dp[i-1]:
                    new_mask |= 1 << (i-1)
            trans[mask][c] = new_mask

    # ---------- DP over the length of T ----------
    dp_cur = [0] * max_mask
    dp_cur[0] = 1

    for _ in range(M):
        dp_next = [0] * max_mask
        for mask in range(max_mask):
            val = dp_cur[mask]
            if val == 0:
                continue
            for c in range(26):
                nm = trans[mask][c]
                dp_next[nm] = (dp_next[nm] + val) % MOD
        dp_cur = dp_next

    # ---------- collect answers by popcount ----------
    ans = [0] * (N + 1)
    for mask in range(max_mask):
        k = mask.bit_count()   # popcount = LCS length
        ans[k] = (ans[k] + dp_cur[mask]) % MOD

    print(' '.join(str(x) for x in ans))

if __name__ == "__main__":
    solve()
```
