```python
def count_snake(s):
    n = len(s)
    if n < 2:
        return 0
    ans = 0
    # lengths 2 to n-1
    for L_len in range(2, n):
        for d1 in range(1, 10):
            ans += d1 ** (L_len - 1)
    # length n
    x1 = int(s[0])
    # d1 < x1
    for d1 in range(1, x1):
        ans += d1 ** (n - 1)
    # d1 == x1
    d1 = x1
    suffix = s[1:]
    m = n - 1
    cnt = 0
    for i, ch in enumerate(suffix):
        L = int(ch)
        max_digit = min(L - 1, d1 - 1)
        if max_digit >= 0:
            cnt += (max_digit + 1) * (d1 ** (m - 1 - i))
        if L < d1:
            continue
        else:
            break
    else:
        cnt += 1
    ans += cnt
    return ans
```

```python
def count_snake(s):
    n = len(s)
    if n < 2:
        return 0
    ans = 0
    # lengths 2 to n-1
    for L_len in range(2, n):
        for d1 in range(1, 10):
            ans += d1 ** (L_len - 1)
    # length n
    x1 = int(s[0])
    for d1 in range(1, x1):
        ans += d1 ** (n - 1)
    d1 = x1
    suffix = s[1:]
    m = n - 1
    cnt = 0
    for i, ch in enumerate(suffix):
        L = int(ch)
        max_digit = min(L - 1, d1 - 1)
        if max_digit >= 0:
            cnt += (max_digit + 1) * (d1 ** (m - 1 - i))
        if L < d1:
            continue
        else:
            break
    else:
        cnt += 1
    ans += cnt
    return ans
```

```python
def count_snake(s):
    n = len(s)
    if n < 2:
        return 0
    ans = 0
    # lengths 2 to n-1
    for L_len in range(2, n):
        for d1 in range(1, 10):
            ans += d1 ** (L_len - 1)
    # length n
    x1 = int(s[0])
    for d1 in range(1, x1):
        ans += d1 ** (n - 1)
    d1 = x1
    suffix = s[1:]
    m = n - 1
    cnt = 0
    for i, ch in enumerate(suffix):
        L = int(ch)
        max_digit = min(L - 1, d1 - 1)
        if max_digit >= 0:
            cnt += (max_digit + 1) * (d1 ** (m - 1 - i))
        if L < d1:
            continue
        else:
            break
    else:
        cnt += 1
    ans += cnt
    return ans
```

```python
def count_snake(s: str) -> int:
    """Return the number of Snake numbers <= s."""
    n = len(s)
    if n < 2:
        return 0
    ans = 0
    # lengths 2 to n-1 (all numbers of these lengths are < s)
    for length in range(2, n):
        for d1 in range(1, 10):
            ans += d1 ** (length - 1)
    # length n
    first_digit = int(s[0])
    # first digit smaller than the first digit of s
    for d1 in range(1, first_digit):
        ans += d1 ** (n - 1)
    # first digit equal to the first digit of s
    d1 = first_digit
    suffix = s[1:]
    m = n - 1
    cnt = 0
    for i, ch in enumerate(suffix):
        L = int(ch)
        max_digit = min(L - 1, d1 - 1)
        if max_digit >= 0:
            cnt += (max_digit + 1) * (d1 ** (m - 1 - i))
        if L < d1:
            continue
        else:
            break
    else:
        cnt += 1
    ans += cnt
    return ans


def main() -> None:
    L, R = map(int, input().split())
    ans = count_snake(str(R)) - count_snake(str(L - 1))
    print(ans)


if __name__ == "__main__":
    main()
```
