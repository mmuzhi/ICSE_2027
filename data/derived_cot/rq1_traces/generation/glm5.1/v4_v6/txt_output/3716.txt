You are given an array of integers nums.
Your task is to find the length of the longest subsequence seq of nums, such that the absolute differences between consecutive elements form a non-increasing sequence of integers. In other words, for a subsequence seq_0, seq_1, seq_2, ..., seq_m of nums, |seq_1 - seq_0| >= |seq_2 - seq_1| >= ... >= |seq_m - seq_m - 1|.
Return the length of such a subsequence.
A subsequence is an non-empty array that can be derived from another array by deleting some or no elements without changing the order of the remaining elements.

class Solution:
    def longestSubsequence(self, nums: List[int]) -> int:
        MAX_VAL = 300
        INF_DIFF = 301  # sentinel for no previous diff
        DIFF_SIZE = INF_DIFF + 1  # 302
        dp = [ [0]*DIFF_SIZE for _ in range(MAX_VAL+1) ]
        suff = [ [0]*DIFF_SIZE for _ in range(MAX_VAL+1) ]  # suffix max

        ans = 0

        for x in nums:
            # temporary updates for dp[x]
            update = [0]*DIFF_SIZE  # we will store candidate lengths for each diff
            # iterate over all possible previous values v
            for v in range(1, MAX_VAL+1):
                # if no subsequence ending with v, skip? But we can still compute; suff[v][new_diff] will be 0 if none.
                new_diff = abs(x - v)
                # get max length from v with last diff >= new_diff
                best_len = suff[v][new_diff]  # suff[v][d] = max_{k>=d} dp[v][k]
                if best_len > 0:
                    cand = best_len + 1
                    if cand > update[new_diff]:
                        update[new_diff] = cand
            # also consider starting new subsequence of length 1
            # we will set dp[x][INF_DIFF] = max(dp[x][INF_DIFF], 1) later
            # apply updates to dp[x]
            for d in range(DIFF_SIZE):
                if update[d] > dp[x][d]:
                    dp[x][d] = update[d]
            # set length 1 state
            if dp[x][INF_DIFF] < 1:
                dp[x][INF_DIFF] = 1
            # recompute suffix max for x
            # compute from high to low
            s = 0
            for d in range(INF_DIFF, -1, -1):
                if dp[x][d] > s:
                    s = dp[x][d]
                suff[x][d] = s
            # update answer
            cur_max = suff[x][0]  # max over all diffs
            if cur_max > ans:
                ans = cur_max

        return ans

```python
from typing import List

class Solution:
    def longestSubsequence(self, nums: List[int]) -> int:
        MAX_VAL = 300
        INF = 300  # sentinel for no previous diff (length 1)
        DIFF_SIZE = INF + 1  # 301
        
        # dp[val][diff] = max length of subsequence ending with value val and last diff = diff
        # diff ranges 0..299 for real diffs, and INF for length-1 state
        dp = [ [0] * DIFF_SIZE for _ in range(MAX_VAL + 1) ]
        # suff[val][d] = max_{k >= d} dp[val][k]
        suff = [ [0] * DIFF_SIZE for _ in range(MAX_VAL + 1) ]
        
        ans = 0
        
        for x in nums:
            # temporary updates for dp[x]
            update = [0] * DIFF_SIZE
            
            # consider all possible previous values v
            for v in range(1, MAX_VAL + 1):
                nd = abs(x - v)  # new diff, 0..299
                best = suff[v][nd]  # best length from v with last diff >= nd
                if best:
                    cand = best + 1
                    if cand > update[nd]:
                        update[nd] = cand
            
            # apply updates to dp[x]
            for d in range(DIFF_SIZE):
                if update[d] > dp[x][d]:
                    dp[x][d] = update[d]
            
            # ensure length-1 state
            if dp[x][INF] < 1:
                dp[x][INF] = 1
            
            # recompute suffix max for value x
            cur = 0
            for d in range(INF, -1, -1):
                if dp[x][d] > cur:
                    cur = dp[x][d]
                suff[x][d] = cur
            
            # update answer
            if suff[x][0] > ans:
                ans = suff[x][0]
        
        return ans
```

```python
            update = [0] * DIFF_SIZE

            # try to extend from any previous value v
            for v in range(1, MAX_VAL + 1):
                nd = abs(x - v)               # new difference (0..299)
                best = suff[v][nd]            # best length from v with last diff >= nd
                if best:
                    cand = best + 1
                    if cand > update[nd]:
                        update[nd] = cand

            # apply the collected updates to dp[x]
            dp_x = dp[x]
            for d in range(DIFF_SIZE):
                if update[d] > dp_x[d]:
                    dp_x[d] = update[d]

            # a subsequence consisting of only x is always valid
            if dp_x[INF] < 1:
                dp_x[INF] = 1

            # recompute suffix maximum for value x
            cur = 0
            suff_x = suff[x]
            for d in range(INF, -1, -1):
                if dp_x[d] > cur:
                    cur = dp_x[d]
                suff_x[d] = cur

            # update global answer
            if suff_x[0] > ans:
                ans = suff_x[0]

        return ans
```
