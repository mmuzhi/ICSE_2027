import sys

def solve():
    import sys
    input = sys.stdin.readline
    N, X = map(int, input().split())
    U = [0]*N
    D = [0]*N
    total = 0
    min_sum = 10**18
    for i in range(N):
        u, d = map(int, input().split())
        U[i] = u
        D[i] = d
        total += u + d
        if u + d < min_sum:
            min_sum = u + d

    def feasible(H):
        # process first tooth
        low = max(0, H - D[0])
        high = min(U[0], H)
        if low > high:
            return False
        for i in range(1, N):
            L = max(0, H - D[i])
            R = min(U[i], H)
            new_low = max(L, low - X)
            new_high = min(R, high + X)
            if new_low > new_high:
                return False
            low, high = new_low, new_high
        return True

    lo = 0
    hi = min_sum
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if feasible(mid):
            lo = mid
        else:
            hi = mid - 1
    H_max = lo
    ans = total - N * H_max
    print(ans)

if __name__ == "__main__":
    solve()
