import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    grid = [input().strip() for _ in range(N)]
    N1 = N - 1
    result = []

    for i in range(N):
        row_chars = []
        for j in range(N):
            # ring index k = distance to border + 1
            d = min(i, N1 - i, j, N1 - j)
            k = d + 1
            t = k & 3          # k % 4

            if t == 0:
                si, sj = i, j
            elif t == 1:
                si, sj = N1 - j, i
            elif t == 2:
                si, sj = N1 - i, N1 - j
            else:  # t == 3
                si, sj = j, N1 - i

            row_chars.append(grid[si][sj])
        result.append(''.join(row_chars))

    sys.stdout.write('\n'.join(result))

if __name__ == "__main__":
