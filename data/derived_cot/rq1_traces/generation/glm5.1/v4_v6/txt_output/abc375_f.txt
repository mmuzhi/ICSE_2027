import sys

def main():
    import sys
    input = sys.stdin.readline
    N, M, Q = map(int, input().split())
    edges = [None] * (M + 1)  # 1-indexed
    for i in range(1, M + 1):
        a, b, c = map(int, input().split())
        edges[i] = (a - 1, b - 1, c)
    queries = []
    is_closed = [False] * (M + 1)
    for _ in range(Q):
        parts = input().split()
        if parts[0] == '1':
            i = int(parts[1])
            queries.append((1, i))
            is_closed[i] = True
        else:
            x = int(parts[1])
            y = int(parts[2])
            queries.append((2, x - 1, y - 1))

    INF = 10**18
    N_range = range(N)
    d = [[INF] * N for _ in range(N)]
    for i in N_range:
        d[i][i] = 0

    # initial graph: edges that are never closed
    for i in range(1, M + 1):
        if not is_closed[i]:
            u, v, w = edges[i]
            if w < d[u][v]:
                d[u][v] = w
                d[v][u] = w

    # Floyd-Warshall
    for k in N_range:
        dk = d[k]
        for i in N_range:
            dik = d[i][k]
            if dik == INF:
                continue
            di = d[i]
            nd = dik  # not used directly
            for j in N_range:
                nd = dik + dk[j]
                if nd < di[j]:
                    di[j] = nd

    ans_rev = []
    # process queries in reverse
    for q in reversed(queries):
        if q[0] == 2:
            _, x, y = q
            dist = d[x][y]
            ans_rev.append(-1 if dist >= INF else dist)
        else:
            _, idx = q
            u, v, w = edges[idx]
            old = d  # current matrix
            old_u = old[u]
            old_v = old[v]
            # create new matrix
            new_d = [[0] * N for _ in range(N)]
            for i in N_range:
                old_i = old[i]
                new_d_i = new_d[i]
                for j in N_range:
                    best = old_i[j]
                    # candidate via u->v
                    if old_i[u] < INF and old_v[j] < INF:
                        cand = old_i[u] + w + old_v[j]
                        if cand < best:
                            best = cand
                    # candidate via v->u
                    if old_i[v] < INF and old_u[j] < INF:
                        cand = old_i[v] + w + old_u[j]
                        if cand < best:
                            best = cand
                    new_d_i[j] = best
            d = new_d

    ans = ans_rev[::-1]
    sys.stdout.write('\n'.join(map(str, ans)))

if __name__ == "__main__":
    main()

import sys

def main():
    import sys
    input = sys.stdin.readline
    N, M, Q = map(int, input().split())
    edges = [None] * (M + 1)  # 1-indexed
    for i in range(1, M + 1):
        a, b, c = map(int, input().split())
        edges[i] = (a - 1, b - 1, c)

    queries = []
    is_closed = [False] * (M + 1)
    for _ in range(Q):
        parts = input().split()
        if parts[0] == '1':
            i = int(parts[1])
            queries.append((1, i))
            is_closed[i] = True
        else:
            x = int(parts[1])
            y = int(parts[2])
            queries.append((2, x - 1, y - 1))

    INF = 10**18
    N_range = range(N)
    d = [[INF] * N for _ in range(N)]
    for i in N_range:
        d[i][i] = 0

    # initial graph: edges that are never closed
    for i in range(1, M + 1):
        if not is_closed[i]:
            u, v, w = edges[i]
            if w < d[u][v]:
                d[u][v] = w
                d[v][u] = w

    # Floyd-Warshall with symmetry enforcement
    for k in N_range:
        dk = d[k]
        for i in N_range:
            dik = d[i][k]
            if dik == INF:
                continue
            di = d[i]
            for j in N_range:
                nd = dik + dk[j]
                if nd < di[j]:
                    di[j] = nd
                    d[j][i] = nd  # keep symmetric

    ans_rev = []
    # process queries in reverse
    for q in reversed(queries):
        if q[0] == 2:
            _, x, y = q
            dist = d[x][y]
            ans_rev.append(-1 if dist >= INF else dist)
        else:
            _, idx = q
            u, v, w = edges[idx]
            old = d  # current matrix
            old_u = old[u]
            old_v = old[v]
            # create new matrix
            new_d = [[0] * N for _ in range(N)]
            for i in N_range:
                old_i = old[i]
                new_d_i = new_d[i]
                for j in N_range:
                    best = old_i[j]
                    # candidate via u->v
                    if old_i[u] < INF and old_v[j] < INF:
                        cand = old_i[u] + w + old_v[j]
                        if cand < best:
                            best = cand
                    # candidate via v->u
                    if old_i[v] < INF and old_u[j] < INF:
                        cand = old_i[v] + w + old_u[j]
                        if cand < best:
                            best = cand
                    new_d_i[j] = best
            d = new_d

    ans = ans_rev[::-1]
    sys.stdout.write('\n'.join(map(str, ans)))

if __name__ == "__main__":
    main()

import sys

def main():
    input = sys.stdin.readline
    N, M, Q = map(int, input().split())
    edges = [None] * (M + 1)          # 1‑indexed
    for i in range(1, M + 1):
        a, b, c = map(int, input().split())
        edges[i] = (a - 1, b - 1, c)   # 0‑indexed vertices

    queries = []
    is_closed = [False] * (M + 1)
    for _ in range(Q):
        parts = input().split()
        if parts[0] == '1':
            i = int(parts[1])
            queries.append((1, i))
            is_closed[i] = True
        else:
            x = int(parts[1])
            y = int(parts[2])
            queries.append((2, x - 1, y - 1))

    INF = 10 ** 18
    N_range = range(N)
    d = [[INF] * N for _ in range(N)]
    for i in N_range:
        d[i][i] = 0

    # initial graph = edges that are never closed
    for i in range(1, M + 1):
        if not is_closed[i]:
            u, v, w = edges[i]
            if w < d[u][v]:
                d[u][v] = w
                d[v][u] = w

    # Floyd‑Warshall (keep matrix symmetric)
    for k in N_range:
        dk = d[k]
        for i in N_range:
            dik = d[i][k]
            if dik == INF:
                continue
            di = d[i]
            for j in N_range:
                nd = dik + dk[j]
                if nd < di[j]:
                    di[j] = nd
                    d[j][i] = nd          # symmetry

    ans_rev = []
    # process queries backwards
    for q in reversed(queries):
        if q[0] == 2:                     # distance query
            _, x, y = q
            dist = d[x][y]
            ans_rev.append(-1 if dist >= INF else dist)
        else:                             # add edge (reverse of closure)
            _, idx = q
            u, v, w = edges[idx]
            old = d
            old_u = old[u]
            old_v = old[v]
            new_d = [[0] * N for _ in range(N)]
            for i in N_range:
                old_i = old[i]
                new_d_i = new_d[i]
                for j in N_range:
                    best = old_i[j]
                    # path i -> u -> v -> j
                    if old_i[u] < INF and old_v[j] < INF:
                        cand = old_i[u] + w + old_v[j]
                        if cand < best:
                            best = cand
                    # path i -> v -> u -> j
                    if old_i[v] < INF and old_u[j] < INF:
                        cand = old_i[v] + w + old_u[j]
                        if cand < best:
                            best = cand
                    new_d_i[j] = best
            d = new_d

    ans = ans_rev[::-1]
    sys.stdout.write('\n'.join(map(str, ans)))

if __name__ == "__main__":
    main()
