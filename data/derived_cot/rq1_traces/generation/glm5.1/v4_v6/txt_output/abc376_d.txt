import sys
from collections import deque

def main():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    adj = [[] for _ in range(N+1)]
    radj = [[] for _ in range(N+1)]
    edges = []
    for _ in range(M):
        a, b = map(int, input().split())
        edges.append((a,b))
        adj[a].append(b)
        radj[b].append(a)
    # BFS from 1 on original
    dist1 = [-1]*(N+1)
    dq = deque([1])
    dist1[1] = 0
    while dq:
        u = dq.popleft()
        for v in adj[u]:
            if dist1[v] == -1:
                dist1[v] = dist1[u] + 1
                dq.append(v)
    # BFS from 1 on reverse
    dist2 = [-1]*(N+1)
    dq = deque([1])
    dist2[1] = 0
    while dq:
        u = dq.popleft()
        for v in radj[u]:
            if dist2[v] == -1:
                dist2[v] = dist2[u] + 1
                dq.append(v)
    ans = 10**9
    for a,b in edges:
        if dist1[a] != -1 and dist2[b] != -1:
            cand = dist1[a] + 1 + dist2[b]
            if cand < ans:
                ans = cand
    if ans == 10**9:
        print(-1)
    else:
        print(ans)

if __name__ == "__main__":
    main()

import sys
from collections import deque

def main():
    input = sys.stdin.readline
    N, M = map(int, input().split())
    adj = [[] for _ in range(N + 1)]
    radj = [[] for _ in range(N + 1)]

    for _ in range(M):
        a, b = map(int, input().split())
        adj[a].append(b)
        radj[b].append(a)          # reverse edge

    # BFS from 1 on original graph
    dist1 = [-1] * (N + 1)
    q = deque([1])
    dist1[1] = 0
    while q:
        u = q.popleft()
        for v in adj[u]:
            if dist1[v] == -1:
                dist1[v] = dist1[u] + 1
                q.append(v)

    # BFS from 1 on reversed graph (gives distances to 1)
    dist2 = [-1] * (N + 1)
    q = deque([1])
    dist2[1] = 0
    while q:
        u = q.popleft()
        for v in radj[u]:
            if dist2[v] == -1:
                dist2[v] = dist2[u] + 1
                q.append(v)

    INF = 10 ** 9
    ans = INF
    for v in range(2, N + 1):          # skip vertex 1 itself
        if dist1[v] != -1 and dist2[v] != -1:
            cand = dist1[v] + dist2[v]
            if cand < ans:
                ans = cand

    print(-1 if ans == INF else ans)

if __name__ == "__main__":
    main()
