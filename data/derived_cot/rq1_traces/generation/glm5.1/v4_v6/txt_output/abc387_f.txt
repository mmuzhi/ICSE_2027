   def solve():
        import sys
        sys.setrecursionlimit(10000)
        MOD = 998244353
        data = sys.stdin.read().split()
        it = iter(data)
        N = int(next(it))
        M = int(next(it))
        A = [0] + [int(next(it)) for _ in range(N)]

        children = [[] for _ in range(N+1)]
        for i in range(1, N+1):
            children[A[i]].append(i)

        # indegree removal to find cycle nodes
        from collections import deque
        indeg = [0]*(N+1)
        for i in range(1, N+1):
            indeg[A[i]] += 1
        q = deque([i for i in range(1, N+1) if indeg[i]==0])
        is_cycle = [True]*(N+1)
        while q:
            u = q.popleft()
            is_cycle[u] = False
            v = A[u]
            indeg[v] -= 1
            if indeg[v] == 0:
                q.append(v)

        # group cycle nodes into components (cycles)
        visited = [False]*(N+1)
        comp_cycles = []  # list of lists of cycle nodes
        for i in range(1, N+1):
            if is_cycle[i] and not visited[i]:
                cycle = []
                cur = i
                while not visited[cur]:
                    visited[cur] = True
                    cycle.append(cur)
                    cur = A[cur]
                comp_cycles.append(cycle)

        # define compute_G for non-cycle nodes
        def compute_G(u):
            # u is non-cycle
            # if leaf (no children), return [0,1,2,...,M]
            if not children[u]:  # leaf
                # return list of length M+1 with G[0]=0, G[i]=i
                return list(range(M+1))
            # else internal node
            prod = [1]*(M+1)
            for v in children[u]:
                # v should be non-cycle
                if is_cycle[v]:
                    # shouldn't happen, skip
                    continue
                g_v = compute_G(v)
                # multiply prod by g_v elementwise
                for i in range(1, M+1):
                    prod[i] = (prod[i] * g_v[i]) % MOD
            # compute G as prefix sum of prod
            G = [0]*(M+1)
            s = 0
            for i in range(1, M+1):
                s = (s + prod[i]) % MOD
                G[i] = s
            return G

        ans = 1
        # Process each component
        for cycle in comp_cycles:
            Q = [1]*(M+1)  # Q[0] unused
            for c in cycle:
                # compute P_c
                P = [1]*(M+1)
                for u in children[c]:
                    if is_cycle[u]:
                        continue
                    G_u = compute_G(u)
                    for v in range(1, M+1):
                        P[v] = (P[v] * G_u[v]) % MOD
                # multiply P into Q
                for v in range(1, M+1):
                    Q[v] = (Q[v] * P[v]) % MOD
            S = sum(Q[1:]) % MOD
            ans = ans * S % MOD

        print(ans)

import sys
from collections import deque

MOD = 998244353

def solve() -> None:
    sys.setrecursionlimit(10000)
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [0] + [int(next(it)) for _ in range(N)]

    # children[i] = list of j with A[j] = i
    children = [[] for _ in range(N + 1)]
    for i in range(1, N + 1):
        children[A[i]].append(i)

    # ---------- find cycle nodes ----------
    indeg = [0] * (N + 1)
    for i in range(1, N + 1):
        indeg[A[i]] += 1
    q = deque([i for i in range(1, N + 1) if indeg[i] == 0])
    is_cycle = [True] * (N + 1)          # initially assume all in cycle
    while q:
        u = q.popleft()
        is_cycle[u] = False
        v = A[u]
        indeg[v] -= 1
        if indeg[v] == 0:
            q.append(v)

    # ---------- collect cycles (components) ----------
    visited = [False] * (N + 1)
    comp_cycles = []          # list of lists, each list is one cycle
    for i in range(1, N + 1):
        if is_cycle[i] and not visited[i]:
            cycle = []
            cur = i
            while not visited[cur]:
                visited[cur] = True
                cycle.append(cur)
                cur = A[cur]
            comp_cycles.append(cycle)

    # ---------- DP for non‑cycle nodes ----------
    def compute_G(u: int):
        """return G_u[0..M] for a non‑cycle node u"""
        # leaf: no children
        if not children[u]:
            return list(range(M + 1))   # G[v] = v

        prod = [1] * (M + 1)            # will become F_u
        for v in children[u]:
            # v is always non‑cycle (see proof)
            if is_cycle[v]:
                continue
            gv = compute_G(v)
            for i in range(1, M + 1):
                prod[i] = (prod[i] * gv[i]) % MOD

        G = [0] * (M + 1)
        s = 0
        for i in range(1, M + 1):
            s = (s + prod[i]) % MOD
            G[i] = s
        return G

    ans = 1
    # ---------- process each component ----------
    for cycle in comp_cycles:
        Q = [1] * (M + 1)          # Q[v] = product over cycle nodes of P_c(v)
        for c in cycle:
            P = [1] * (M + 1)      # P_c(v)
            for u in children[c]:
                if is_cycle[u]:
                    continue       # skip cycle‑to‑cycle edges
                Gu = compute_G(u)
                for v in range(1, M + 1):
                    P[v] = (P[v] * Gu[v]) % MOD
            for v in range(1, M + 1):
                Q[v] = (Q[v] * P[v]) % MOD
        S = sum(Q[1:]) % MOD
        ans = ans * S % MOD

    print(ans)


if __name__ == "__main__":
    solve()
