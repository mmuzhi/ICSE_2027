Your task is to convert the given caption into a good caption using the minimum number of operations, and return it. If there are multiple possible good captions, return the lexicographically smallest one among them. If it is impossible to create a good caption, return an empty string "".

```python
class Solution:
    def minCostGoodCaption(self, caption: str) -> str:
        n = len(caption)
        if n < 3:
            return ""
        
        INF = 10**9
        # Precompute prefix costs
        pre = [0] * ((n+1) * 26)
        for i in range(1, n+1):
            char_idx = ord(caption[i-1]) - 97
            base = i * 26
            prev = (i-1) * 26
            for c in range(26):
                pre[base + c] = pre[prev + c] + abs(char_idx - c)
        
        dp = [INF] * (n+1)
        dp[n] = 0
        best_suff = [INF] * 26
        
        for i in range(n, -1, -1):
            if i < n:
                if i <= n - 3:
                    base_i = i * 26
                    min_val = INF
                    for c in range(26):
                        val = -pre[base_i + c] + best_suff[c]
                        if val < min_val:
                            min_val = val
                    dp[i] = min_val
                else:
                    dp[i] = INF
            
            j = i + 2
            if j <= n and dp[j] < INF:
                base_j = j * 26
                for c in range(26):
                    val = pre[base_j + c] + dp[j]
                    if val < best_suff[c]:
                        best_suff[c] = val
        
        # Build max_j for reconstruction
        max_j = [{} for _ in range(26)]
        for j in range(n+1):
            if dp[j] < INF:
                base_j = j * 26
                for c in range(26):
                    val = pre[base_j + c] + dp[j]
                    max_j[c][val] = j  # keep the maximum j
        
        # Reconstruct the lexicographically smallest string
        res = []
        i = 0
        while i < n:
            base_i = i * 26
            for c in range(26):
                target = dp[i] + pre[base_i + c]
                if target in max_j[c]:
                    j = max_j[c][target]
                    if j >= i + 3:
                        L = j - i
                        res.append(chr(97 + c) * L)
                        i = j
                        break
        return "".join(res)
```

class Solution:
    def minCostGoodCaption(self, caption: str) -> str:
        n = len(caption)
        if n < 3:
            return ""
        
        INF = 10**9
        
        # Precompute prefix costs: pre[i][c] = cost to change caption[0..i-1] to all c
        # Store as a flat array of size (n+1)*26
        pre = [0] * ((n+1) * 26)
        for i in range(1, n+1):
            char_idx = ord(caption[i-1]) - 97
            base = i * 26
            prev = (i-1) * 26
            for c in range(26):
                pre[base + c] = pre[prev + c] + abs(char_idx - c)
        
        # dp[i] = minimum cost to make suffix caption[i..n-1] good
        dp = [INF] * (n+1)
        dp[n] = 0
        
        # best_suff[c] = min_{j >= i+3} (pre[j][c] + dp[j]) for current i
        best_suff = [INF] * 26
        
        # Fill dp from right to left
        for i in range(n, -1, -1):
            if i < n:
                if i <= n - 3:
                    base_i = i * 26
                    min_val = INF
                    for c in range(26):
                        val = -pre[base_i + c] + best_suff[c]
                        if val < min_val:
                            min_val = val
                    dp[i] = min_val
                else:
                    dp[i] = INF
            
            # Add j = i+2 to the pool of available endpoints for smaller i
            j = i + 2
            if j <= n and dp[j] < INF:
                base_j = j * 26
                for c in range(26):
                    val = pre[base_j + c] + dp[j]
                    if val < best_suff[c]:
                        best_suff[c] = val
        
        # Build dictionaries for reconstruction: for each character c, map value (pre[j][c] + dp[j]) -> max j
        max_j = [{} for _ in range(26)]
        for j in range(n+1):
            if dp[j] < INF:
                base_j = j * 26
                for c in range(26):
                    val = pre[base_j + c] + dp[j]
                    max_j[c][val] = j  # later j overwrites earlier, so we keep the maximum
        
        # Reconstruct the lexicographically smallest optimal string
        res = []
        i = 0
        while i < n:
            base_i = i * 26
            for c in range(26):
                target = dp[i] + pre[base_i + c]
                if target in max_j[c]:
                    j = max_j[c][target]
                    if j >= i + 3:
                        L = j - i
                        res.append(chr(97 + c) * L)
                        i = j
                        break
        return "".join(res)
