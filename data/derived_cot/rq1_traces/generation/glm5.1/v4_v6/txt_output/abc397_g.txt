Initially, all edges have weight 0. We choose exactly K out of the M edges and change their weights to 1. Find the maximum possible value of the shortest distance from vertex 1 to vertex N in the resulting graph.

Input

The input is given from Standard Input in the following format:
N M K
u_1 v_1
u_2 v_2
\vdots
u_M v_M

Output

- 1 \leq K \leq M \leq 100
- 1 \leq u_j, v_j \leq N
- u_j \neq v_j
- In the given graph, vertex N is reachable from vertex 1.
- All input values are integers.

class Dinic:
    def __init__(self, N):
        self.N = N
        self.adj = [[] for _ in range(N)]

    def add_edge(self, u, v, cap):
        self.adj[u].append([v, cap, len(self.adj[v])])
        self.adj[v].append([u, 0, len(self.adj[u]) - 1])

    def bfs(self, s, t):
        self.level = [-1]*self.N
        q = deque([s])
        self.level[s] = 0
        while q:
            u = q.popleft()
            for v, cap, rev in self.adj[u]:
                if cap > 0 and self.level[v] < 0:
                    self.level[v] = self.level[u] + 1
                    q.append(v)
        return self.level[t] >= 0

    def dfs(self, u, t, f):
        if u == t:
            return f
        for i in range(self.it[u], len(self.adj[u])):
            self.it[u] = i
            v, cap, rev = self.adj[u][i]
            if cap > 0 and self.level[u] < self.level[v]:
                ret = self.dfs(v, t, min(f, cap))
                if ret > 0:
                    self.adj[u][i][1] -= ret
                    self.adj[v][rev][1] += ret
                    return ret
        return 0

    def max_flow(self, s, t):
        flow = 0
        INF = 10**18
        while self.bfs(s, t):
            self.it = [0]*self.N
            while True:
                f = self.dfs(s, t, INF)
                if f == 0:
                    break
                flow += f
        return flow

def solve():
    import sys
    from collections import deque

    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = []
    adj = [[] for _ in range(N+1)]
    for _ in range(M):
        u, v = map(int, input().split())
        edges.append((u, v))
        adj[u].append(v)

    # compute L: shortest path length from 1 to N (unweighted)
    dist = [-1]*(N+1)
    q = deque([1])
    dist[1] = 0
    while q:
        u = q.popleft()
        for v in adj[u]:
            if dist[v] == -1:
                dist[v] = dist[u] + 1
                q.append(v)
    L = dist[N]  # guaranteed not -1

    INF = 10**9  # large

    # define function to compute f(d)
    def min_R(d):
        if d == 0:
            return 0
        # number of nodes: 2 + N*d
        node_count = 2 + N * d
        dinic = Dinic(node_count)
        source = 0
        sink = 1
        # assign ids
        node_id = [[0]*(d+1) for _ in range(N+1)]  # index 1..N, 1..d
        cur = 2
        for v in range(1, N+1):
            for i in range(1, d+1):
                node_id[v][i] = cur
                cur += 1

        # monotonicity
        for v in range(1, N+1):
            for i in range(2, d+1):
                dinic.add_edge(node_id[v][i], node_id[v][i-1], INF)

            dinic.add_edge(node_id[1][i], sink, INF)   # vertex 1 label 0

import sys
from collections import deque

# -------------------------------------------------------------
# Dinic max flow implementation
class Dinic:
    def __init__(self, n):
        self.n = n
        self.graph = [[] for _ in range(n)]

    def add_edge(self, fr, to, cap):
        # forward edge: [to, capacity, rev_index]
        # backward edge: [fr, 0, rev_index]
        fwd = [to, cap, len(self.graph[to])]
        rev = [fr, 0, len(self.graph[fr])]
        self.graph[fr].append(fwd)
        self.graph[to].append(rev)

    def bfs(self, s, t):
        self.level = [-1] * self.n
        q = deque([s])
        self.level[s] = 0
        while q:
            v = q.popleft()
            for to, cap, _ in self.graph[v]:
                if cap > 0 and self.level[to] < 0:
                    self.level[to] = self.level[v] + 1
                    q.append(to)
        return self.level[t] >= 0

    def dfs(self, v, t, f):
        if v == t:
            return f
        for i in range(self.it[v], len(self.graph[v])):
            self.it[v] = i
            to, cap, rev = self.graph[v][i]
            if cap > 0 and self.level[v] < self.level[to]:
                ret = self.dfs(to, t, min(f, cap))
                if ret:
                    self.graph[v][i][1] -= ret
                    self.graph[to][rev][1] += ret
                    return ret
        return 0

    def max_flow(self, s, t):
        flow = 0
        INF = 10 ** 18
        while self.bfs(s, t):
            self.it = [0] * self.n
            while True:
                f = self.dfs(s, t, INF)
                if not f:
                    break
                flow += f
        return flow

def solve() -> None:
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = []
    adj = [[] for _ in range(N + 1)]
    for _ in range(M):
        u, v = map(int, input().split())
        edges.append((u, v))
        adj[u].append(v)

    # unweighted shortest path length from 1 to N
    dist = [-1] * (N + 1)
    q = deque([1])
    dist[1] = 0
    while q:
        u = q.popleft()
        for v in adj[u]:
            if dist[v] == -1:
                dist[v] = dist[u] + 1
                q.append(v)
    L = dist[N]          # maximum possible d

    INF = 10 ** 9        # larger than any possible finite cut (M <= 100)

    # function to compute f(d) = minimum number of tight edges
    def min_R(d: int) -> int:
        if d == 0:
            return 0
        # total nodes: source(0), sink(1), and N*d vertex-layer nodes
        node_cnt = 2 + N * d
        dinic = Dinic(node_cnt)
        SRC, SNK = 0, 1

        # assign ids to vertex-layer nodes
        id = [[0] * (d + 1) for _ in range(N + 1)]   # 1-indexed vertices, i = 1..d
        cur = 2
        for v in range(1, N + 1):
            for i in range(1, d + 1):
                id[v][i] = cur
                cur += 1

        # monotonicity inside each vertex: if (v,i) in S then (v,i-1) must be in S
        for v in range(1, N + 1):
