class Solution:
    def minMaxSubarraySum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # min contributions
        L_min = [-1]*n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] >= nums[i]:
                stack.pop()
            L_min[i] = stack[-1] if stack else -1
            stack.append(i)
        R_min = [n]*n
        stack = []
        for i in range(n-1, -1, -1):
            while stack and nums[stack[-1]] > nums[i]:
                stack.pop()
            R_min[i] = stack[-1] if stack else n
            stack.append(i)
        # max contributions
        L_max = [-1]*n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] <= nums[i]:
                stack.pop()
            L_max[i] = stack[-1] if stack else -1
            stack.append(i)
        R_max = [n]*n
        stack = []
        for i in range(n-1, -1, -1):
            while stack and nums[stack[-1]] < nums[i]:
                stack.pop()
            R_max[i] = stack[-1] if stack else n
            stack.append(i)

        def count_pairs(a, b, k):
            t = k - 1
            if t < 0:
                return 0
            Xmax = min(a-1, t)
            if Xmax < 0:
                return 0
            T = t - (b - 1)
            if T >= 0:
                x1_end = min(Xmax, T)
                cnt_full = x1_end + 1
                sum_full = cnt_full * b
                x2_start = T + 1
                if x2_start <= Xmax:
                    num = Xmax - x2_start + 1
                    first = t - x2_start + 1
                    last = t - Xmax + 1
                    sum2 = num * (first + last) // 2
                else:
                    sum2 = 0
                return sum_full + sum2
            else:
                num = Xmax + 1
                first = t + 1
                last = t - Xmax + 1
                return num * (first + last) // 2

        total = 0
        for i in range(n):
            a_min = i - L_min[i]
            b_min = R_min[i] - i
            cnt_min = count_pairs(a_min, b_min, k)
            a_max = i - L_max[i]
            b_max = R_max[i] - i
            cnt_max = count_pairs(a_max, b_max, k)
            total += nums[i] * (cnt_min + cnt_max)
        return total

```python
from typing import List

class Solution:
    def minMaxSubarraySum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        
        # Previous less (strict) for min
        L_min = [-1] * n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] >= nums[i]:
                stack.pop()
            L_min[i] = stack[-1] if stack else -1
            stack.append(i)
        
        # Next less-or-equal for min
        R_min = [n] * n
        stack = []
        for i in range(n-1, -1, -1):
            while stack and nums[stack[-1]] > nums[i]:
                stack.pop()
            R_min[i] = stack[-1] if stack else n
            stack.append(i)
        
        # Previous greater (strict) for max
        L_max = [-1] * n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] <= nums[i]:
                stack.pop()
            L_max[i] = stack[-1] if stack else -1
            stack.append(i)
        
        # Next greater-or-equal for max
        R_max = [n] * n
        stack = []
        for i in range(n-1, -1, -1):
            while stack and nums[stack[-1]] < nums[i]:
                stack.pop()
            R_max[i] = stack[-1] if stack else n
            stack.append(i)
        
        def count_pairs(a: int, b: int, k: int) -> int:
            t = k - 1
            if t < 0:
                return 0
            Xmax = min(a-1, t)
            if Xmax < 0:
                return 0
            T = t - (b - 1)
            if T >= 0:
                x1_end = min(Xmax, T)
                cnt_full = x1_end + 1
                sum_full = cnt_full * b
                x2_start = T + 1
                if x2_start <= Xmax:
                    num = Xmax - x2_start + 1
                    first = t - x2_start + 1
                    last = t - Xmax + 1
                    sum2 = num * (first + last) // 2
                else:
                    sum2 = 0
                return sum_full + sum2
            else:
                num = Xmax + 1
                first = t + 1
                last = t - Xmax + 1
                return num * (first + last) // 2
        
        total = 0
        for i in range(n):
            a_min = i - L_min[i]
            b_min = R_min[i] - i
            cnt_min = count_pairs(a_min, b_min, k)
            a_max = i - L_max[i]
            b_max = R_max[i] - i
            cnt_max = count_pairs(a_max, b_max, k)
            total += nums[i] * (cnt_min + cnt_max)
        
        return total
```

```python
from typing import List

class Solution:
    def minMaxSubarraySum(self, nums: List[int], k: int) -> int:
        n = len(nums)

        # ---------- minimum contributions ----------
        # previous less (strict)
        L_min = [-1] * n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] >= nums[i]:
                stack.pop()
            L_min[i] = stack[-1] if stack else -1
            stack.append(i)

        # next less-or-equal
        R_min = [n] * n
        stack = []
        for i in range(n - 1, -1, -1):
            while stack and nums[stack[-1]] > nums[i]:
                stack.pop()
            R_min[i] = stack[-1] if stack else n
            stack.append(i)

        # ---------- maximum contributions ----------
        # previous greater (strict)
        L_max = [-1] * n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] <= nums[i]:
                stack.pop()
            L_max[i] = stack[-1] if stack else -1
            stack.append(i)

        # next greater-or-equal
        R_max = [n] * n
        stack = []
        for i in range(n - 1, -1, -1):
            while stack and nums[stack[-1]] < nums[i]:
                stack.pop()
            R_max[i] = stack[-1] if stack else n
            stack.append(i)

        # ---------- helper: count pairs (x,y) with 0<=x<a, 0<=y<b, x+y<=k-1 ----------
        def count_pairs(a: int, b: int, k: int) -> int:
            t = k - 1
            if t < 0:
                return 0
            Xmax = min(a - 1, t)
            if Xmax < 0:
                return 0
            T = t - (b - 1)
            if T >= 0:
                x1_end = min(Xmax, T)
                cnt_full = x1_end + 1
                sum_full = cnt_full * b
                x2_start = T + 1
                if x2_start <= Xmax:
                    num = Xmax - x2_start + 1
                    first = t - x2_start + 1
                    last = t - Xmax + 1
                    sum2 = num * (first + last) // 2
                else:
                    sum2 = 0
                return sum_full + sum2
            else:
                num = Xmax + 1
                first = t + 1
                last = t - Xmax + 1
                return num * (first + last) // 2

        # ---------- sum contributions ----------
        total = 0
        for i in range(n):
            a_min = i - L_min[i]
            b_min = R_min[i] - i
            cnt_min = count_pairs(a_min, b_min, k)

            a_max = i - L_max[i]
            b_max = R_max[i] - i
            cnt_max = count_pairs(a_max, b_max, k)

            total += nums[i] * (cnt_min + cnt_max)

        return total
```
