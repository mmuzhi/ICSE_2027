import sys

def solve():
    import sys
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = []
    for _ in range(M):
        u, v, w = map(int, input().split())
        edges.append((w, u, v))
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))

    cntA = [0]*(N+1)
    cntB = [0]*(N+1)
    for a in A:
        cntA[a] += 1
    for b in B:
        cntB[b] += 1

    edges.sort(key=lambda x: x[0])

    parent = list(range(N+1))
    size = [1]*(N+1)

    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    ans = 0

    for w, u, v in edges:
        ru = find(u)
        rv = find(v)
        if ru == rv:
            continue
        # compute cross matches
        cross = min(cntA[ru], cntB[rv]) + min(cntA[rv], cntB[ru])
        ans += w * cross
        # compute new counts
        newA = cntA[ru] + cntA[rv] - cross
        newB = cntB[ru] + cntB[rv] - cross
        # union by size
        if size[ru] < size[rv]:
            ru, rv = rv, ru
        parent[rv] = ru
        size[ru] += size[rv]
        cntA[ru] = newA
        cntB[ru] = newB

    print(ans)

if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = []
    for _ in range(M):
        u, v, w = map(int, input().split())
        edges.append((w, u, v))

    A = list(map(int, input().split()))
    B = list(map(int, input().split()))

    # multiplicities of each vertex in A and B
    cntA = [0] * (N + 1)
    cntB = [0] * (N + 1)
    for a in A:
        cntA[a] += 1
    for b in B:
        cntB[b] += 1

    # Kruskal: process edges in increasing weight
    edges.sort()

    parent = list(range(N + 1))
    size = [1] * (N + 1)

    def find(x: int) -> int:
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    ans = 0

    for w, u, v in edges:
        ru = find(u)
        rv = find(v)
        if ru == rv:
            continue

        # cross matches that become possible at this weight
        cross = min(cntA[ru], cntB[rv]) + min(cntA[rv], cntB[ru])
        ans += w * cross

        # new counts after matching inside the merged component
        newA = cntA[ru] + cntA[rv] - cross
        newB = cntB[ru] + cntB[rv] - cross

        # union by size
        if size[ru] < size[rv]:
            ru, rv = rv, ru
        parent[rv] = ru
        size[ru] += size[rv]
        cntA[ru] = newA
        cntB[ru] = newB

    print(ans)


if __name__ == "__main__":
