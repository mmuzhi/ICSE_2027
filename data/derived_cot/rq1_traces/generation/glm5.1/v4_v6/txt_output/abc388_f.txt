            if is_bad(pos):
                print("No"); return
            pos += A
        print("Yes"); return
    
    good_intervals = []
    prev_end = 0
    for L, R in bad_intervals:
        if prev_end + 1 <= L - 1:
            good_intervals.append((prev_end + 1, L - 1))
        prev_end = R
    if prev_end + 1 <= N:
        good_intervals.append((prev_end + 1, N))
    
    if not good_intervals:
        print("No"); return
    
    reachable_end = [set() for _ in range(len(good_intervals))]
    
    for i, (a, b) in enumerate(good_intervals):
        entry = set()
        if i == 0:
            if a <= 1 <= b:
                entry.add(1)
        else:
            for p in reachable_end[i - 1]:
                for j in range(A, B + 1):
                    next_pos = p + j
                    if a <= next_pos <= b:
                        entry.add(next_pos)
        
        if not entry:
            continue
        
        queue = deque(sorted(entry))
        visited = set(entry)
        min_pos, max_pos = min(entry), max(entry)
        num_reachable = len(entry)
        has_contiguous = (max_pos - min_pos + 1 == num_reachable) and (num_reachable >= A)
        
        while queue and not has_contiguous:
            pos = queue.popleft()
            for j in range(A, B + 1):
                next_pos = pos + j
                if next_pos > b or next_pos in visited:
                    continue
                visited.add(next_pos)
                queue.append(next_pos)
                min_pos = min(min_pos, next_pos)
                max_pos = max(max_pos, next_pos)
                num_reachable += 1
                if max_pos - min_pos + 1 == num_reachable and num_reachable >= A:
                    has_contiguous = True
                    break
        
        if has_contiguous:
            for pos in range(max(a, b - B + 1), b + 1):
                if pos >= min_pos:
                    reachable_end[i].add(pos)
        else:
            for pos in range(max(a, b - B + 1), b + 1):
                if pos in visited:
                    reachable_end[i].add(pos)
    
    print("Yes" if good_intervals and N in reachable_end[-1] else "No")
