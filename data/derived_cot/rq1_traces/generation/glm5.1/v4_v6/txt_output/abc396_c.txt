def main():
    import sys
    input_data = sys.stdin.read().split()
    it = iter(input_data)
    N = int(next(it))
    M = int(next(it))
    B = [int(next(it)) for _ in range(N)]
    W = [int(next(it)) for _ in range(M)]
    B.sort(reverse=True)
    W.sort(reverse=True)
    prefB = [0]*(N+1)
    for i in range(N):
        prefB[i+1] = prefB[i] + B[i]
    prefW = [0]*(M+1)
    for i in range(M):
        prefW[i+1] = prefW[i] + W[i]
    bestBFrom = [0]*(N+1)
    bestBFrom[N] = prefB[N]
    for i in range(N-1, -1, -1):
        bestBFrom[i] = max(prefB[i], bestBFrom[i+1])
    L = min(M, N)
    ans = 0
    for l in range(L+1):
        cand = prefW[l] + bestBFrom[l]
        if cand > ans:
            ans = cand
    print(ans)

if __name__ == "__main__":

```python

import sys

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    B = [int(next(it)) for _ in range(N)]
    W = [int(next(it)) for _ in range(M)]

    B.sort(reverse=True)
    W.sort(reverse=True)

    # prefix sums of sorted arrays
    prefB = [0] * (N + 1)
    for i in range(N):
        prefB[i+1] = prefB[i] + B[i]

    prefW = [0] * (M + 1)
    for i in range(M):
        prefW[i+1] = prefW[i] + W[i]

    # bestBFrom[i] = max_{k >= i} prefB[k]
    bestBFrom = [0] * (N + 1)
    bestBFrom[N] = prefB[N]
    for i in range(N-1, -1, -1):
        bestBFrom[i] = max(prefB[i], bestBFrom[i+1])

    L = min(M, N)          # white count cannot exceed N
    ans = 0
    for l in range(L + 1):
        cand = prefW[l] + bestBFrom[l]
        if cand > ans:
            ans = cand

    print(ans)

if __name__ == "__main__":
```
