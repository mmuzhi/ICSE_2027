from collections import defaultdict
import math
from bisect import bisect_left

class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        n = len(nums)
        left_groups = defaultdict(list)
        for q in range(2, n):
            cnt = defaultdict(int)
            for p in range(0, q-1):  # p from 0 to q-2 inclusive
                a = nums[p]
                b = nums[q]
                g = math.gcd(a, b)
                a //= g
                b //= g
                cnt[(a, b)] += 1
            for ratio, c in cnt.items():
                left_groups[ratio].append((q, c))
        
        right_groups = defaultdict(list)
        for r in range(0, n-2):  # r from 0 to n-3
            cnt = defaultdict(int)
            for s in range(r+2, n):
                a = nums[s]
                b = nums[r]
                g = math.gcd(a, b)
                a //= g
                b //= g
                cnt[(a, b)] += 1
            for ratio, c in cnt.items():
                right_groups[ratio].append((r, c))
        
        ans = 0
        for ratio, left_list in left_groups.items():
            if ratio not in right_groups:
                continue
            right_list = right_groups[ratio]
            # right_list is already sorted by r because we appended in increasing r order
            right_r = [r for r, _ in right_list]
            right_cnt = [c for _, c in right_list]
            # compute suffix sum of counts
            m = len(right_cnt)
            suffix = [0] * (m + 1)
            for i in range(m-1, -1, -1):
                suffix[i] = suffix[i+1] + right_cnt[i]
            # iterate left entries
            for q, cntL in left_list:
                target = q + 2
                i = bisect_left(right_r, target)
                if i < m:
                    ans += cntL * suffix[i]
        return ans

from typing import List
from collections import defaultdict
import math
from bisect import bisect_left

class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        n = len(nums)
        BASE = 1001  # since nums[i] <= 1000
        
        # Step 1: Build right groups
        right_temp = defaultdict(list)  # key -> list of (r, count)
        for r in range(0, n-2):  # r from 0 to n-3 inclusive
            cnt = defaultdict(int)
            for s in range(r+2, n):
                a = nums[s]
                b = nums[r]
                g = math.gcd(a, b)
                a //= g
                b //= g
                key = a * BASE + b
                cnt[key] += 1
            for key, c in cnt.items():
                right_temp[key].append((r, c))
        
        # Convert to processed structure with r_list and suffix sums
        right_processed = {}
        for key, lst in right_temp.items():
            # lst is already sorted by r because we appended in increasing r
            r_list = [r for r, _ in lst]
            cnt_list = [c for _, c in lst]
            m = len(cnt_list)
            suffix = [0] * (m + 1)
            for i in range(m-1, -1, -1):
                suffix[i] = suffix[i+1] + cnt_list[i]
            right_processed[key] = (r_list, suffix)
        
        ans = 0
        # Step 2: Iterate over q
        for q in range(2, n):
            cnt = defaultdict(int)
            for p in range(0, q-1):  # p from 0 to q-2 inclusive
                a = nums[p]
                b = nums[q]
                g = math.gcd(a, b)
                a //= g
                b //= g
                key = a * BASE + b
                cnt[key] += 1
            for key, cntL in cnt.items():
                if key in right_processed:
                    r_list, suffix = right_processed[key]
                    target = q + 2
                    i = bisect_left(r_list, target)
                    if i < len(r_list):
                        ans += cntL * suffix[i]
        return ans

from typing import List
from collections import defaultdict
import math
from bisect import bisect_left

class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        n = len(nums)
        BASE = 1001                     # nums[i] ≤ 1000, so a*BASE + b is unique

        # ---------- build right groups ----------
        # temporary: ratio -> list of (r, count)
        right_temp = defaultdict(list)
        for r in range(0, n - 2):       # r can be at most n-3
            cnt = defaultdict(int)
            for s in range(r + 2, n):
                a = nums[s]
                b = nums[r]
                g = math.gcd(a, b)
                a //= g
                b //= g
                key = a * BASE + b
                cnt[key] += 1
            for key, c in cnt.items():
                right_temp[key].append((r, c))

        # convert to final structure: ratio -> (r_list, suffix_sum_of_counts)
        right_processed = {}
        for key, lst in right_temp.items():
            # lst is already sorted by r because we appended in increasing r order
            r_list = [r for r, _ in lst]
            cnt_list = [c for _, c in lst]
            m = len(cnt_list)
            suffix = [0] * (m + 1)
            for i in range(m - 1, -1, -1):
                suffix[i] = suffix[i + 1] + cnt_list[i]
            right_processed[key] = (r_list, suffix)

        # ---------- iterate over q and count ----------
        ans = 0
        for q in range(2, n):           # q must be at least 2
            cnt = defaultdict(int)
            for p in range(0, q - 1):   # p from 0 to q-2 inclusive
                a = nums[p]
                b = nums[q]
                g = math.gcd(a, b)
                a //= g
                b //= g
                key = a * BASE + b
                cnt[key] += 1

            for key, cntL in cnt.items():
                if key in right_processed:
                    r_list, suffix = right_processed[key]
                    target = q + 2      # need r ≥ q+2
                    i = bisect_left(r_list, target)
                    if i < len(r_list):
                        ans += cntL * suffix[i]

        return ans
