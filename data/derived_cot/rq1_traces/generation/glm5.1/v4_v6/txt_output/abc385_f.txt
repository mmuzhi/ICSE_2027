N

- 1 \leq N \leq 2 \times 10^5
- 1 \leq X_1 < \dots < X_N \leq 10^9
- 1 \leq H_i \leq 10^9
- All input values are integers.

```python
import sys
from decimal import Decimal, getcontext

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    X = [0] * N
    H = [0] * N
    for i in range(N):
        x, h = map(int, input().split())
        X[i] = x
        H[i] = h

    # With only one building it is always visible from any height.
    if N == 1:
        print(-1)
        return

    best_num = None   # numerator of the current maximum b
    best_den = None   # denominator (always positive)

    for i in range(N - 1):
        # b = (X[i+1]*H[i] - X[i]*H[i+1]) / (X[i+1] - X[i])
        num = X[i + 1] * H[i] - X[i] * H[i + 1]
        den = X[i + 1] - X[i]          # > 0 because X are strictly increasing
        if best_num is None:
            best_num, best_den = num, den
        else:
            # compare num/den with best_num/best_den using cross multiplication
            if num * best_den > best_num * den:
                best_num, best_den = num, den

    # If the maximum b is negative, all buildings are visible even from height 0.
    if best_num < 0:
        print(-1)
    else:
        # Output the exact value with enough precision.
        getcontext().prec = 50
        ans = Decimal(best_num) / Decimal(best_den)
        # Fixed‑point notation with 20 decimal places guarantees error < 1e-9.
        print("{:.20f}".format(ans))

if __name__ == "__main__":
    solve()
```
