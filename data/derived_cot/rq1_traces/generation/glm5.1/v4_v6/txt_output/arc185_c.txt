    rev = [0] * n
    bits = n.bit_length() - 1
    for i in range(n):
        rev[i] = (rev[i>>1] >> 1) | ((i&1) << (bits - 1))
    for i in range(n):
        if i < rev[i]:
            a[i], a[rev[i]] = a[rev[i]], a[i]
    # precompute roots

    ang = 2 * cmath.pi / n * (-1 if invert else 1)
    for i in range(n):
        roots[i] = complex(cmath.cos(ang * i), cmath.sin(ang * i))
    length = 2

        step = n // length
        for i in range(0, n, length):
            for j in range(i, i + length//2):
                u = a[j]
                v = a[j + length//2] * roots[(j - i) * step]
                a[j] = u + v
                a[j + length//2] = u - v
        length <<= 1
    if invert:
        for i in range(n):
            a[i] /= n
