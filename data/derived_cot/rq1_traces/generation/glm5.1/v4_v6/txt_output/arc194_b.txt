class BIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+1)
    def add(self, idx, val):
        while idx <= self.n:
            self.bit[idx] += val
            idx += idx & -idx
    def sum(self, idx):
        res = 0
        while idx > 0:
            res += self.bit[idx]
            idx -= idx & -idx
        return res

```python

import sys

# -------------------------------------------------------------

class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)

    def add(self, idx, val):
        while idx <= self.n:
            self.tree[idx] += val
            idx += idx & -idx

    def sum(self, idx):
        s = 0
        while idx > 0:
            s += self.tree[idx]
            idx -= idx & -idx
        return s

# -------------------------------------------------------------

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    P = [0] + list(map(int, input().split()))   # 1‑indexed

    # position of each value
    pos = [0] * (N + 1)
    for i in range(1, N + 1):
        pos[P[i]] = i

    # ----- left moves (greater elements to the left) -----
    bit = BIT(N)
    left = [0] * (N + 1)          # left[v] = number of greater elements left of v
    for i in range(1, N + 1):
        v = P[i]
        leq = bit.sum(v)          # elements <= v seen so far
        left[v] = (i - 1) - leq   # greater elements = total seen - leq
        bit.add(v, 1)

    # ----- right moves (smaller elements to the right) -----
    bit = BIT(N)
    right = [0] * (N + 1)         # right[v] = number of smaller elements right of v
    for i in range(N, 0, -1):
        v = P[i]
        smaller = bit.sum(v - 1)  # elements < v seen so far (to the right)
        right[v] = smaller
        bit.add(v, 1)

    # ----- compute total contribution -----
    total = 0
    for v in range(1, N + 1):
        p = pos[v]
        l = left[v]
        r = right[v]

        # cost if all left moves are done first, then all right moves
        left_cost = l * p - l * (l + 1) // 2
        # after left moves the position becomes p - l
        right_cost = r * (p - l) + r * (r - 1) // 2
        total += left_cost + right_cost

    # each swap is counted twice in the sum of contributions
    answer = total // 2
    print(answer)

# -------------------------------------------------------------
if __name__ == "__main__":
    solve()
```
