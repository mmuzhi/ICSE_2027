```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    W = int(next(it))

    # columns[1..W] will store (Y, block_index)
    columns = [[] for _ in range(W + 1)]
    for i in range(1, N + 1):
        x = int(next(it))
        y = int(next(it))
        columns[x].append((y, i))

    # sort blocks in each column by Y ascending
    for x in range(1, W + 1):
        columns[x].sort(key=lambda t: t[0])

    # number of blocks in each column
    sizes = [len(columns[x]) for x in range(1, W + 1)]
    K = min(sizes) if sizes else 0

    INF = 10**18
    removal_time = [INF] * (N + 1)

    if K > 0:
        # max_Y[k] = maximum Y among the k-th blocks of all columns
        max_Y = [0] * (K + 1)  # 1-indexed
        for x in range(1, W + 1):
            for k, (y, _) in enumerate(columns[x], start=1):
                if k > K:
                    break
                if y > max_Y[k]:
                    max_Y[k] = y

        # C[k] = time of the k-th clearing
        C = [0] * (K + 1)
        C[1] = max_Y[1]
        for k in range(2, K + 1):
            C[k] = max(C[k-1] + 1, max_Y[k])

        # assign removal times to blocks
        for x in range(1, W + 1):
            for k, (_, idx) in enumerate(columns[x], start=1):
                if k <= K:
                    removal_time[idx] = C[k]
                else:
                    # blocks with rank > K are never removed
                    pass

    Q = int(next(it))
    out = []
    for _ in range(Q):
        T = int(next(it))
        A = int(next(it))
        if T < removal_time[A]:
            out.append("Yes")
        else:
            out.append("No")

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
```
