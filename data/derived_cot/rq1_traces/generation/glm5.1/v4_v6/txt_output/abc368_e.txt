=== PROMPT ===
You will be given a competitive programming problem.
In the nation of Atcoder, there are N cities numbered 1 to N, and M trains numbered 1 to M.
Train i departs from city A_i at time S_i and arrives at city B_i at time T_i.
Given a positive integer X_1, find a way to set non-negative integers X_2,\ldots,X_M that satisfies the following condition with the minimum possible value of X_2+\ldots+X_M.

- Condition: For all pairs (i,j) satisfying 1 \leq i,j \leq M, if B_i=A_j and T_i \leq S_j, then T_i+X_i \leq S_j+X_j.
- In other words, for any pair of trains that are originally possible to transfer between, it is still possible to transfer even after delaying the departure and arrival times of each train i by X_i.



It can be proved that such a way to set X_2,\ldots,X_M with the minimum possible value of X_2+\ldots+X_M is unique.

Input

The input is given from Standard Input in the following format:
N M X_1
A_1 B_1 S_1 T_1
\vdots
A_M B_M S_M T_M

Output

Print X_2,\ldots,X_M that satisfy the condition with the minimum possible sum, in that order, separated by spaces.

Constraints


- 2 \leq N \leq 2\times 10^5
- 2 \leq M \leq 2\times 10^5
- 1 \leq A_i,B_i \leq N
- A_i \neq B_i
- 0 \leq S_i < T_i \leq 10^9
- 1 \leq X_1 \leq 10^9
- All input values are integers.

Sample Input 1

3 6 15
1 2 10 20
1 2 20 30
2 3 25 40
2 3 35 50
3 1 15 30
3 1 45 60

Sample Output 1

0 10 0 0 5

Other trains can operate without delay while still allowing transfers between originally transferable trains, so (X_2,X_3,X_4,X_5,X_6)=(0,10,0,0,5) satisfies the condition.
Moreover, there is no solution with a smaller sum that satisfies the condition, so this is the answer.

   def __init__(self, n):
       self.n = n
       self.bit = [-10**18] * (n+1)  # 1-indexed
   def update(self, idx, val):
       while idx <= self.n:
           if val > self.bit[idx]:
               self.bit[idx] = val
           idx += idx & -idx
   def query(self, idx):
       res = -10**18
       while idx > 0:
           if self.bit[idx] > res:
               res = self.bit[idx]
           idx -= idx & -idx
       return res

```python
import sys
from bisect import bisect_left, bisect_right

class FenwickMax:
    def __init__(self, n):
        self.n = n
        self.bit = [-10**18] * (n+1)
    def update(self, idx, val):
        while idx <= self.n:
            if val > self.bit[idx]:
                self.bit[idx] = val
            idx += idx & -idx
    def query(self, idx):
        res = -10**18
        while idx > 0:
            if self.bit[idx] > res:
                res = self.bit[idx]
            idx -= idx & -idx
        return res

def main():
    import sys
    input_data = sys.stdin.read().split()
    it = iter(input_data)
    N = int(next(it))
    M = int(next(it))
    X1 = int(next(it))
    A = [0]*(M+1)
    B = [0]*(M+1)
    S = [0]*(M+1)
    T = [0]*(M+1)
    times = [[] for _ in range(N+1)]
    for i in range(1, M+1):
        a = int(next(it)); b = int(next(it)); s = int(next(it)); t = int(next(it))
        A[i] = a; B[i] = b; S[i] = s; T[i] = t
        times[a].append(s)
        times[b].append(t)
    # sort and unique times per city
    for c in range(1, N+1):
        if times[c]:
            times[c] = sorted(set(times[c]))
    # create fenwick trees
    fenwick = [None]*(N+1)
    for c in range(1, N+1):
        if times[c]:
            fenwick[c] = FenwickMax(len(times[c]))
    # order by S
    order = list(range(1, M+1))
    order.sort(key=lambda i: S[i])
    X = [0]*(M+1)
    X[1] = X1
    INF_NEG = -10**18
    for idx in order:
        if idx == 1:
            # just update arrival
            c_arr = B[1]
            t = T[1]
            val = X[1] + t
            if fenwick[c_arr] is not None:
                pos = bisect_left(times[c_arr], t) + 1
                fenwick[c_arr].update(pos, val)
        else:
            c_dep = A[idx]
            if fenwick[c_dep] is not None:
                # query
                pos = bisect_right(times[c_dep], S[idx])  # number of times <= S[idx]
                if pos > 0:
                    best = fenwick[c_dep].query(pos)
                    if best != INF_NEG:
                        X[idx] = max(0, best - S[idx])
                    else:
                        X[idx] = 0
                else:
                    X[idx] = 0
            else:
                X[idx] = 0
            # update arrival
            c_arr = B[idx]
            t = T[idx]
            val = X[idx] + t
            if fenwick[c_arr] is not None:
                pos = bisect_left(times[c_arr], t) + 1
                fenwick[c_arr].update(pos, val)
    # output X[2..M]
    out = ' '.join(str(X[i]) for i in range(2, M+1))
    print(out)

if __name__ == "__main__":
    main()
```

```python
import sys
from bisect import bisect_left, bisect_right

class FenwickMax:
    def __init__(self, n):
        self.n = n
        self.bit = [-10**18] * (n+1)
    def update(self, idx, val):
        while idx <= self.n:
            if val > self.bit[idx]:
                self.bit[idx] = val
            idx += idx & -idx
    def query(self, idx):
        res = -10**18
        while idx > 0:
            if self.bit[idx] > res:
                res = self.bit[idx]
            idx -= idx & -idx
        return res

def main():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X1 = int(next(it))
    A = [0]*(M+1)
    B = [0]*(M+1)
    S = [0]*(M+1)
    T = [0]*(M+1)
    times = [[] for _ in range(N+1)]
    for i in range(1, M+1):
        a = int(next(it)); b = int(next(it)); s = int(next(it)); t = int(next(it))
        A[i] = a; B[i] = b; S[i] = s; T[i] = t
        times[a].append(s)
        times[b].append(t)
    # unique and sort times per city
    for c in range(1, N+1):
        if times[c]:
            times[c] = sorted(set(times[c]))
    # build position maps and Fenwick trees
    pos_map = [None]*(N+1)
    fenwick = [None]*(N+1)
    for c in range(1, N+1):
        if times[c]:
            d = {time: idx+1 for idx, time in enumerate(times[c])}
            pos_map[c] = d
            fenwick[c] = FenwickMax(len(times[c]))
    # precompute query and update positions for each train
    qpos = [0]*(M+1)  # for departure query
    upos = [0]*(M+1)  # for arrival update
    for i in range(1, M+1):
        qpos[i] = pos_map[A[i]][S[i]]
        upos[i] = pos_map[B[i]][T[i]]
    # order by S
    order = list(range(1, M+1))
    order.sort(key=lambda i: S[i])
    X = [0]*(M+1)
    X[1] = X1
    INF_NEG = -10**18
    for idx in order:
        if idx == 1:
            # update arrival
            c_arr = B[1]
            fenwick[c_arr].update(upos[1], X[1] + T[1])
        else:
            c_dep = A[idx]
            best = fenwick[c_dep].query(qpos[idx])
            if best == INF_NEG:
                X[idx] = 0
            else:
                X[idx] = max(0, best - S[idx])
            # update arrival
            c_arr = B[idx]
            fenwick[c_arr].update(upos[idx], X[idx] + T[idx])
    # output
    out = ' '.join(str(X[i]) for i in range(2, M+1))
    print(out)

if __name__ == "__main__":
    main()
```

```python
import sys
from bisect import bisect_left, bisect_right

# ------------------------------------------------------------
# Fenwick tree for prefix maximum (only increasing updates)
class FenwickMax:
    def __init__(self, n):
        self.n = n
        self.bit = [-10**18] * (n + 1)          # -inf

    def update(self, idx, val):
        while idx <= self.n:
            if val > self.bit[idx]:
                self.bit[idx] = val
            idx += idx & -idx

    def query(self, idx):
        res = -10**18
        while idx > 0:
            if self.bit[idx] > res:
                res = self.bit[idx]
            idx -= idx & -idx
        return res
# ------------------------------------------------------------

def main():
    it = iter(sys.stdin.buffer.read().split())
    N = int(next(it))
    M = int(next(it))
    X1 = int(next(it))

    A = [0] * (M + 1)
    B = [0] * (M + 1)
    S = [0] * (M + 1)
    T = [0] * (M + 1)

    # times[c] will collect all S and T that belong to city c
    times = [[] for _ in range(N + 1)]

    for i in range(1, M + 1):
        a = int(next(it)); b = int(next(it)); s = int(next(it)); t = int(next(it))
        A[i] = a; B[i] = b; S[i] = s; T[i] = t
        times[a].append(s)
        times[b].append(t)

    # sort and deduplicate times for each city
    for c in range(1, N + 1):
        if times[c]:
            times[c] = sorted(set(times[c]))

    # build a Fenwick tree for every city that actually appears
    fenwick = [None] * (N + 1)
    for c in range(1, N + 1):
        if times[c]:
            fenwick[c] = FenwickMax(len(times[c]))

    # process trains in increasing departure time S
    order = list(range(1, M + 1))
    order.sort(key=lambda i: S[i])

    X = [0] * (M + 1)
    X[1] = X1
    INF_NEG = -10**18

    for idx in order:
        if idx == 1:
            # train 1 is fixed, only insert its arrival
            c_arr = B[1]
            t = T[1]
            val = X[1] + t
            if fenwick[c_arr] is not None:
                pos = bisect_left(times[c_arr], t) + 1
                fenwick[c_arr].update(pos, val)
        else:
            # ---- query for the departure city ----
            c_dep = A[idx]
            if fenwick[c_dep] is not None:
                pos = bisect_right(times[c_dep], S[idx])   # #times ≤ S[idx]
                if pos > 0:
                    best = fenwick[c_dep].query(pos)
                    if best != INF_NEG:
                        X[idx] = max(0, best - S[idx])
                    else:
                        X[idx] = 0
                else:
                    X[idx] = 0
            else:
                X[idx] = 0

            # ---- insert the arrival of this train ----
            c_arr = B[idx]
            t = T[idx]
            val = X[idx] + t
            if fenwick[c_arr] is not None:
                pos = bisect_left(times[c_arr], t) + 1
                fenwick[c_arr].update(pos, val)

    # output X2 .. XM
    print(' '.join(str(X[i]) for i in range(2, M + 1)))

if __name__ == "__main__":
    main()
```
