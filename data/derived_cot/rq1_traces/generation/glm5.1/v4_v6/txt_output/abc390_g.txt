Standard iterative NTT (Cooley-Tukey) with precomputed roots:
```
def ntt(a, invert):
    n = len(a)
    j = 0
    for i in range(1, n):
        bit = n >> 1
        while j & bit:
            j ^= bit
            bit >>= 1
        j ^= bit
        if i < j:
            a[i], a[j] = a[j], a[i]
    length = 2
    while length <= n:
        half = length // 2
        wlen = pow(G, (MOD-1)//length, MOD)
        if invert:
            wlen = pow(wlen, MOD-2, MOD)
        for i in range(0, n, length):
            w = 1
            for j in range(i, i+half):
                u = a[j]
                v = a[j+half] * w % MOD
                a[j] = (u + v) % MOD
                a[j+half] = (u - v) % MOD
                w = w * wlen % MOD
        length <<= 1
    if invert:
        inv_n = pow(n, MOD-2, MOD)
        for i in range(n):
            a[i] = a[i] * inv_n % MOD
```

    for A_ntt in factors_ntt[1:]:
        for i in range(L):
            P_ntt[i] = P_ntt[i] * A_ntt[i] % MOD
    
    # inverse transform to get coefficients of P(x) = prod (1 + w_l x)^{c_l}
    ntt(P_ntt, True)
    a = P_ntt[:N + 1]
    
    # A_sum = sum_{m=0}^{N-1} a_m * m! * (N-1-m)!
    A_sum = 0
    for m in range(N):
        A_sum = (A_sum + a[m] * fact[m] % MOD * fact[N - 1 - m]) % MOD
    V_total = (N * (N + 1) // 2) % MOD
    Total = V_total * A_sum % MOD
    
    # subtract contributions B_l
    for l in range(1, 7):
        if c[l] == 0:
            continue
        wl = w[l]
        # divide P(x) by (1 + wl x) to get P_l(x)
        b = [0] * N
        b[0] = a[0]
        for m in range(1, N):
            b[m] = (a[m] - wl * b[m - 1]) % MOD
        B_l = 0
        for m in range(0, N - 1):
            B_l = (B_l + b[m] * fact[m + 1] % MOD * fact[N - 2 - m]) % MOD
        Total = (Total - S[l] * wl % MOD * B_l) % MOD
    
    print(Total % MOD)

if __name__ == "__main__":
    main()
