from typing import List

class Solution:
    def maximumTotalSum(self, maximumHeight: List[int]) -> int:
        arr = sorted(maximumHeight, reverse=True)
        total = arr[0]
        prev = arr[0]

            if h < 1:

            total += h
            prev = h
        return total

```python
from typing import List

class Solution:
    def maximumTotalSum(self, maximumHeight: List[int]) -> int:
        # Sort in descending order of maximum allowed heights
        arr = sorted(maximumHeight, reverse=True)
        
        # Assign the first tower its maximum possible height
        total = arr[0]
        prev = arr[0]          # the height assigned to the previous tower
        
        # Process the remaining towers
        for m in arr[1:]:
            # The largest height we can give to this tower is the smaller of
            # its own maximum and (previous height - 1) to keep heights distinct
            h = min(m, prev - 1)
            if h < 1:          # no positive integer available -> impossible
                return -1
            total += h
            prev = h
        
        return total
```
