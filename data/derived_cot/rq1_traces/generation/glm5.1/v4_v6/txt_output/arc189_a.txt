import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))

    # The first and last cells can never be changed
    if A[0] != 1 or A[-1] != (N % 2):
        print(0)
        return

    # Compress array into runs of consecutive equal values
    runs = []
    cur_val = A[0]
    cur_len = 1
    for i in range(1, N):
        if A[i] == cur_val:
            cur_len += 1
        else:
            runs.append((cur_val, cur_len))
            cur_val = A[i]
            cur_len = 1
    runs.append((cur_val, cur_len))

    # Check that runs alternate 1, 0, 1, ... and have odd lengths
    for i, (val, length) in enumerate(runs):
        expected = 1 if i % 2 == 0 else 0
        if val != expected or length % 2 == 0:
            print(0)
            return

    # m_i = number of merges inside the i-th run
    m_list = [(length - 1) // 2 for _, length in runs]
    M = sum(m_list)

    # Precompute factorials and inverse factorials up to N
    fact = [1] * (N + 1)
    for i in range(1, N + 1):
        fact[i] = fact[i-1] * i % MOD

    inv_fact = [1] * (N + 1)
    inv_fact[N] = pow(fact[N], MOD - 2, MOD)
    for i in range(N, 0, -1):
        inv_fact[i-1] = inv_fact[i] * i % MOD

    # ans = M! / 2^M * prod( binom(2*m_i, m_i) )
    ans = fact[M] * pow(2, MOD - 1 - M, MOD) % MOD
    for m in m_list:
        binom = fact[2*m] * inv_fact[m] % MOD * inv_fact[m] % MOD
        ans = ans * binom % MOD

    print(ans)


if __name__ == "__main__":
    solve()
