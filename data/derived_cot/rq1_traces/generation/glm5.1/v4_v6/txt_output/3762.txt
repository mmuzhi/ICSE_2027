Now, what about a case where we have a long array and we need to decide whether to do round trips at the current index or push forward? The greedy algorithm always does round trips at the current index to satisfy its deficit. Is it ever better to not do round trips at i, but instead move forward and do round trips later? Suppose we need v_i and v_{i+1}. If we do round trips at i, we get v_i visits to i and v_i visits to i+1. Then we move to i+1 and maybe need more visits to i+1. If we instead move to i+1 first, we would have to come back to i to get its visits, which would cost more because we have to travel back. The greedy strategy of satisfying the leftmost deficit first is optimal because any visit to i must be paired with a visit to an adjacent cell. If we delay satisfying i, we might have to travel extra distance. This is a standard result for such problems (similar to "minimum time to complete tasks on a line" where you can only move one step at a time and need to do tasks at each point). The optimal is to go left to right, and when at i, if it needs more, do round trips with i+1. This is optimal because it minimizes the number of times we cross the edge (i, i+1) in the wrong direction.

    while rightmost >= 0 and v[rightmost] <= 0:

    if rightmost < 0:

                    rightmost -= 1

class Solution:
    def maxScore(self, points: List[int], m: int) -> int:
        n = len(points)
        max_p = max(points)
        
        def feasible(x):
            v = [(x + p - 1) // p for p in points]
            if all(q == 0 for q in v):
                return True
            moves = 1
            v[0] -= 1
            rightmost = n - 1
            while rightmost >= 0 and v[rightmost] <= 0:
                rightmost -= 1
            if rightmost < 0:
                return moves <= m
            i = 0
            while i <= rightmost:
                if moves > m:
                    return False
                if v[i] > 0:
                    if i == n - 1:
                        moves += 2 * v[i]
                        v[i] = 0
                        while rightmost >= 0 and v[rightmost] <= 0:
                            rightmost -= 1
                        break
                    else:
                        moves += 2 * v[i]
                        v[i+1] -= v[i]
                        v[i] = 0
                        if rightmost > i:
                            moves += 1
                            v[i+1] -= 1
                            while rightmost >= 0 and v[rightmost] <= 0:
                                rightmost -= 1
                            i += 1
                        else:
                            break
                else:
                    if rightmost > i:
                        moves += 1
                        v[i+1] -= 1
                        while rightmost >= 0 and v[rightmost] <= 0:
                            rightmost -= 1
                        i += 1
                    else:
                        break
            return moves <= m
        
        low = 0
        high = max_p * m + 1
        ans = 0
        while low <= high:
            mid = (low + high) // 2
            if feasible(mid):
                ans = mid
                low = mid + 1
            else:
                high = mid - 1
        return ans

```python
from typing import List

class Solution:
    def maxScore(self, points: List[int], m: int) -> int:
        n = len(points)
        max_p = max(points)
        
        def feasible(x: int) -> bool:
            if x == 0:
                return True
            # required visits for each cell
            v = [(x + p - 1) // p for p in points]
            moves = 1
            v[0] -= 1
            # find rightmost index with positive required visits
            rightmost = n - 1
            while rightmost >= 0 and v[rightmost] <= 0:
                rightmost -= 1
            if rightmost < 0:
                return moves <= m
            
            i = 0
            while i <= rightmost:
                if moves > m:
                    return False
                if v[i] > 0:
                    if i == n - 1:
                        moves += 2 * v[i]
                        v[i] = 0
                        # update rightmost
                        while rightmost >= 0 and v[rightmost] <= 0:
                            rightmost -= 1
                        break
                    else:
                        moves += 2 * v[i]
                        v[i+1] -= v[i]
                        v[i] = 0
                        if rightmost > i:
                            moves += 1
                            v[i+1] -= 1
                            # update rightmost
                            while rightmost >= 0 and v[rightmost] <= 0:
                                rightmost -= 1
                            i += 1
                        else:
                            break
                else:
                    if rightmost > i:
                        moves += 1
                        v[i+1] -= 1
                        # update rightmost
                        while rightmost >= 0 and v[rightmost] <= 0:
                            rightmost -= 1
                        i += 1
                    else:
                        break
            return moves <= m
        
        low = 0
        high = max_p * m + 1
        ans = 0
        while low <= high:
            mid = (low + high) // 2
            if feasible(mid):
                ans = mid
                low = mid + 1
            else:
                high = mid - 1
        return ans
```
