def minOperations(self, nums: List[int], x: int, k: int) -> int:
    n = len(nums)
    m = n - x + 1
    # compress
    sorted_vals = sorted(set(nums))
    val_to_idx = {v: i+1 for i, v in enumerate(sorted_vals)}
    idx_to_val = [0] * (len(sorted_vals) + 1)
    for i, v in enumerate(sorted_vals, 1):
        idx_to_val[i] = v
    M = len(sorted_vals)

    class BIT:
        def __init__(self, n):
            self.n = n
            self.tree = [0]*(n+1)
        def add(self, idx, delta):
            while idx <= self.n:
                self.tree[idx] += delta
                idx += idx & -idx
        def sum(self, idx):
            s = 0
            while idx>0:
                s += self.tree[idx]
                idx -= idx & -idx
            return s
        def find_kth(self, k):
            # find smallest idx such that prefix sum >= k
            idx = 0
            bit_mask = 1 << (self.n.bit_length() - 1)
            while bit_mask:
                nxt = idx + bit_mask
                if nxt <= self.n and self.tree[nxt] < k:
                    k -= self.tree[nxt]
                    idx = nxt
                bit_mask >>= 1
            return idx + 1

    bit_cnt = BIT(M)
    bit_sum = BIT(M)
    total_sum = 0

    # initialize first window
    for i in range(x):
        val = nums[i]
        idx = val_to_idx[val]
        bit_cnt.add(idx, 1)
        bit_sum.add(idx, val)
        total_sum += val

    cost = [0]*m

    def compute_cost():
        left_size = (x + 1) // 2
        right_size = x - left_size
        # find median index
        idx = bit_cnt.find_kth(left_size)
        median_val = idx_to_val[idx]
        cnt = bit_cnt.sum(idx)
        sum_up_to = bit_sum.sum(idx)
        excess = cnt - left_size
        sum_left = sum_up_to - excess * median_val
        sum_right = total_sum - sum_left
        return median_val * left_size - sum_left + sum_right - median_val * right_size

    cost[0] = compute_cost()

    for i in range(1, m):
        # remove nums[i-1]
        val_out = nums[i-1]
        idx_out = val_to_idx[val_out]
        bit_cnt.add(idx_out, -1)
        bit_sum.add(idx_out, -val_out)
        total_sum -= val_out

        bit_cnt.add(idx_in, 1)
        bit_sum.add(idx_in, val_in)
        total_sum += val_in
        cost[i] = compute_cost()

    # DP over cost
    INF = 10**18

    dp_prev = cost[:]  # t=1
    for t in range(2, k+1):

        pref_min = [INF]*m
        cur_min = INF

            pref_min[i] = cur_min
        dp_curr = [INF]*m
        for i in range(m):

                if best_prev != INF:
                    dp_curr[i] = cost[i] + best_prev
        dp_prev = dp_curr
    ans = min(dp_prev)
    return ans

from typing import List

class Solution:
    def minOperations(self, nums: List[int], x: int, k: int) -> int:
        n = len(nums)
        m = n - x + 1
        
        # coordinate compression
        sorted_vals = sorted(set(nums))
        val_to_idx = {v: i+1 for i, v in enumerate(sorted_vals)}
        idx_to_val = [0] * (len(sorted_vals) + 1)
        for i, v in enumerate(sorted_vals, 1):
            idx_to_val[i] = v
        M = len(sorted_vals)
        
        class BIT:
            def __init__(self, n):
                self.n = n
                self.tree = [0] * (n + 1)
            def add(self, idx, delta):
                while idx <= self.n:
                    self.tree[idx] += delta
                    idx += idx & -idx
            def sum(self, idx):
                s = 0
                while idx > 0:
                    s += self.tree[idx]
                    idx -= idx & -idx
                return s
            def find_kth(self, k):
                # find smallest idx such that prefix sum >= k
                idx = 0
                bit_mask = 1 << (self.n.bit_length() - 1)
                while bit_mask:
                    nxt = idx + bit_mask
                    if nxt <= self.n and self.tree[nxt] < k:
                        k -= self.tree[nxt]
                        idx = nxt
                    bit_mask >>= 1
                return idx + 1
        
        bit_cnt = BIT(M)
        bit_sum = BIT(M)
        total_sum = 0
        
        # initialize first window
        for i in range(x):
            val = nums[i]
            idx = val_to_idx[val]
            bit_cnt.add(idx, 1)
            bit_sum.add(idx, val)
            total_sum += val
        
        cost = [0] * m
        
        def compute_cost():
            left_size = (x + 1) // 2
            right_size = x - left_size
            # find the left_size-th smallest value
            idx = bit_cnt.find_kth(left_size)
            median_val = idx_to_val[idx]
            cnt = bit_cnt.sum(idx)
            sum_up_to = bit_sum.sum(idx)
            excess = cnt - left_size
            sum_left = sum_up_to - excess * median_val
            sum_right = total_sum - sum_left
            return median_val * left_size - sum_left + sum_right - median_val * right_size
        
        cost[0] = compute_cost()
        
        for i in range(1, m):
            # remove element leaving the window
            val_out = nums[i-1]
            idx_out = val_to_idx[val_out]
            bit_cnt.add(idx_out, -1)
            bit_sum.add(idx_out, -val_out)
            total_sum -= val_out
            # add new element entering the window
            val_in = nums[i+x-1]
            idx_in = val_to_idx[val_in]
            bit_cnt.add(idx_in, 1)
            bit_sum.add(idx_in, val_in)
            total_sum += val_in
            cost[i] = compute_cost()
        
        # DP to select k disjoint intervals
        INF = 10**18
        if k == 1:
            return min(cost)
        
        dp_prev = cost[:]  # t = 1
        for t in range(2, k+1):
            dp_curr = [INF] * m
            best = INF
            for i in range(m):
                if i - x >= 0:
                    # update best with dp_prev[i-x]
                    if dp_prev[i-x] < best:
                        best = dp_prev[i-x]
                if best != INF:
                    dp_curr[i] = cost[i] + best
            dp_prev = dp_curr
        
        ans = min(dp_prev)
        return ans

                    idx += idx & -idx

                s = 0
                while idx > 0:
                    s += self.tree[idx]
                    idx -= idx & -idx
                return s
