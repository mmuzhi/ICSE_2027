import sys

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input())
    strings = [input().strip() for _ in range(N)]

    children = [{}]  # list of dict per node
    depth = [0]
    best = [0]  # root best = 0 (empty string)

    out = []
    for s in strings:
        L = len(s)
        node = 0
        min_best = best[0]
        for ch in s:
            c = ord(ch) - 97
            if c in children[node]:
                node = children[node][c]
                if best[node] < min_best:
                    min_best = best[node]
            else:
                break
        ans = L + min_best
        out.append(str(ans))

        # insert s
        node = 0
        # update root
        cand = L - 2 * depth[0]
        if cand < best[0]:
            best[0] = cand
        for ch in s:
            c = ord(ch) - 97
            if c not in children[node]:
                children.append({})
                depth.append(depth[node] + 1)
                best.append(10**18)
                children[node][c] = len(children) - 1
            node = children[node][c]
            d = depth[node]
            cand = L - 2 * d
            if cand < best[node]:
                best[node] = cand

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()

import sys

def main():
    import sys
    input = sys.stdin.readline
    N = int(input())
    strs = [input().strip() for _ in range(N)]

    # Trie node: children dict, depth, best value
    children = [{}]   # index 0 is root
    depth = [0]
    best = [0]        # empty string gives value 0 at root

    out_lines = []
    for s in strs:
        L = len(s)
        # Query
        node = 0
        min_best = best[0]
        for ch in s:
            c = ord(ch) - 97
            if c in children[node]:
                node = children[node][c]
                if best[node] < min_best:
                    min_best = best[node]
            else:
                break
        ans = L + min_best   # min_best <= 0
        out_lines.append(str(ans))

        # Insert s
        node = 0
        # root update (though never changes from 0)
        cand = L - 2 * depth[0]
        if cand < best[0]:
            best[0] = cand
        for ch in s:
            c = ord(ch) - 97
            if c not in children[node]:
                children.append({})
                depth.append(depth[node] + 1)
                best.append(10**18)

            d = depth[node]
            cand = L - 2 * d
            if cand < best[node]:
                best[node] = cand

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":

import sys

def main():
    input = sys.stdin.readline
    N = int(input())
    strings = [input().strip() for _ in range(N)]

    # trie data
    children = [{}]          # children[node] = dict {char_index: child_node}
    depth = [0]              # depth[node]
    best = [0]               # best[node] = min_{strings in subtree} (len - 2*depth)

    out = []
    for s in strings:
        L = len(s)

        # ----- query -----
        node = 0
        min_best = best[0]          # root is always visited
        for ch in s:
            c = ord(ch) - 97
            if c in children[node]:
                node = children[node][c]
                if best[node] < min_best:
                    min_best = best[node]
            else:
                break
        # min_best is the minimum of (|prev| - 2*LCP) over all previous strings
        answer = L + min_best        # min_best ≤ 0
        out.append(str(answer))

        # ----- insert s -----
        node = 0
        # update root (candidate = L - 0, never beats 0)
        cand = L - 2 * depth[0]
        if cand < best[0]:
            best[0] = cand

        for ch in s:
            c = ord(ch) - 97
            if c not in children[node]:
                # create new node
                children.append({})
                depth.append(depth[node] + 1)
                best.append(10**18)          # large initial value
                children[node][c] = len(children) - 1
            node = children[node][c]
            d = depth[node]
            cand = L - 2 * d
            if cand < best[node]:
                best[node] = cand

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
