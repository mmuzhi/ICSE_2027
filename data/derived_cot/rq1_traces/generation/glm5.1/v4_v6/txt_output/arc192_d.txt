```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:]))
    M = 998244353

    # Precompute spf up to 1000
    MAX_A = 1000
    spf = [0] * (MAX_A + 1)
    primes = []
    for i in range(2, MAX_A + 1):
        if spf[i] == 0:
            spf[i] = i
            primes.append(i)
        for p in primes:
            if p > spf[i] or i * p > MAX_A:
                break
            spf[i * p] = p

    # Collect exponents for each prime
    # We can use a dictionary: prime -> list of exponents
    from collections import defaultdict
    prime_exps = defaultdict(lambda: [0] * (N-1))
    for idx, a in enumerate(A):
        x = a
        while x > 1:
            p = spf[x]
            cnt = 0
            while x % p == 0:
                x //= p
                cnt += 1
            prime_exps[p][idx] = cnt

    total_ans = 1
    for p, d_list in prime_exps.items():
        S = sum(d_list)
        if S == 0:
            continue
        # Precompute powers of p modulo M
        pow_p = [1] * (S + 1)
        for i in range(1, S + 1):
            pow_p[i] = (pow_p[i-1] * p) % M

        dp0 = [0] * (S + 1)
        dp1 = [0] * (S + 1)
        for e in range(1, S + 1):
            dp0[e] = pow_p[e]
        dp1[0] = 1

        for d in d_list:
            if d == 0:
                # Multiply by p^e
                for e in range(S + 1):
                    if dp0[e]:
                        dp0[e] = (dp0[e] * pow_p[e]) % M
                    if dp1[e]:
                        dp1[e] = (dp1[e] * pow_p[e]) % M
            else:
                next_dp0 = [0] * (S + 1)
                next_dp1 = [0] * (S + 1)
                for e in range(S + 1):
                    v0 = dp0[e]
                    if v0:
                        # +d
                        if e + d <= S:
                            next_dp0[e+d] = (next_dp0[e+d] + v0 * pow_p[e+d]) % M
                        # -d
                        if e >= d:
                            ne = e - d
                            if ne == 0:
                                next_dp1[0] = (next_dp1[0] + v0) % M
                            else:
                                next_dp0[ne] = (next_dp0[ne] + v0 * pow_p[ne]) % M
                    v1 = dp1[e]
                    if v1:
                        # +d
                        if e + d <= S:
                            next_dp1[e+d] = (next_dp1[e+d] + v1 * pow_p[e+d]) % M
                        # -d
                        if e >= d:
                            ne = e - d
                            next_dp1[ne] = (next_dp1[ne] + v1 * pow_p[ne]) % M
                dp0 = next_dp0
                dp1 = next_dp1

        ans_p = sum(dp1) % M
        total_ans = (total_ans * ans_p) % M

    print(total_ans)

if __name__ == "__main__":
    solve()
```

```python
import sys
from collections import defaultdict

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:]))

    # Precompute smallest prime factor up to 1000
    MAX_A = 1000
    spf = [0] * (MAX_A + 1)
    primes = []
    for i in range(2, MAX_A + 1):
        if spf[i] == 0:
            spf[i] = i
            primes.append(i)
        for p in primes:
            if p > spf[i] or i * p > MAX_A:
                break
            spf[i * p] = p

    # Collect exponents for each prime that appears
    prime_exps = defaultdict(lambda: [0] * (N - 1))
    for idx, a in enumerate(A):
        x = a
        while x > 1:
            p = spf[x]
            cnt = 0
            while x % p == 0:
                x //= p
                cnt += 1
            prime_exps[p][idx] = cnt

    total_ans = 1

    for p, d_list in prime_exps.items():
        S = sum(d_list)
        if S == 0:
            continue

        # Precompute powers of p modulo MOD
        pow_p = [1] * (S + 1)
        for i in range(1, S + 1):
            pow_p[i] = (pow_p[i - 1] * p) % MOD

        # dp0: has not hit zero yet, dp1: has hit zero
        dp0 = [0] * (S + 1)
        dp1 = [0] * (S + 1)
        for e in range(1, S + 1):
            dp0[e] = pow_p[e]
        dp1[0] = 1

        for d in d_list:
            if d == 0:
                # Multiply by p^e for the unchanged exponent
                for e in range(S + 1):
                    if dp0[e]:
                        dp0[e] = (dp0[e] * pow_p[e]) % MOD
                    if dp1[e]:
                        dp1[e] = (dp1[e] * pow_p[e]) % MOD
            else:
                next_dp0 = [0] * (S + 1)
                next_dp1 = [0] * (S + 1)
                for e in range(S + 1):
                    v0 = dp0[e]
                    v1 = dp1[e]
                    if v0 == 0 and v1 == 0:
                        continue
                    if v0:
                        # +d
                        if e + d <= S:
                            next_dp0[e + d] = (next_dp0[e + d] + v0 * pow_p[e + d]) % MOD
                        # -d
                        if e >= d:
                            ne = e - d
                            if ne == 0:
                                next_dp1[0] = (next_dp1[0] + v0) % MOD
                            else:
                                next_dp0[ne] = (next_dp0[ne] + v0 * pow_p[ne]) % MOD
                    if v1:
                        # +d
                        if e + d <= S:
                            next_dp1[e + d] = (next_dp1[e + d] + v1 * pow_p[e + d]) % MOD
                        # -d
                        if e >= d:
                            ne = e - d
                            next_dp1[ne] = (next_dp1[ne] + v1 * pow_p[ne]) % MOD
                dp0 = next_dp0
                dp1 = next_dp1

        ans_p = sum(dp1) % MOD
        total_ans = (total_ans * ans_p) % MOD

    print(total_ans)


if __name__ == "__main__":
    solve()
```
