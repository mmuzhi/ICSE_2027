import sys
import bisect

def main():
    # read all input
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    Q = int(next(it))
    queries = [int(next(it)) for _ in range(Q)]

    MAX = 10 ** 6          # because sqrt(10^12) = 10^6
    omega = [0] * (MAX + 1)   # number of distinct prime factors

    # sieve: count distinct prime factors
    for i in range(2, MAX + 1):
        if omega[i] == 0:          # i is prime
            for j in range(i, MAX + 1, i):
                omega[j] += 1

    # collect all 400 numbers (squares of numbers with exactly two distinct primes)
    cand = []
    for n in range(2, MAX + 1):
        if omega[n] == 2:
            cand.append(n * n)      # n^2 is a 400 number

    # cand is already sorted because n increases

    out_lines = []
    for A in queries:
        idx = bisect.bisect_right(cand, A)
        out_lines.append(str(cand[idx - 1]))

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    main()
