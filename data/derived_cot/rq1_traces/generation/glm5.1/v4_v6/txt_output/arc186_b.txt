```python
import sys

def solve() -> None:
    MOD = 998244353
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = [0] + [int(x) for x in data[1:1+N]]  # 1-indexed, A[1..N]

    # children of each node in the original tree (0 is the virtual root)
    children = [[] for _ in range(N+1)]
    for i in range(1, N+1):
        children[A[i]].append(i)

    # Build the binary tree via left-child (last child) and right-child (previous sibling)
    left_child = [0] * (N+1)
    right_child = [0] * (N+1)

    for u in range(N+1):
        if children[u]:
            left_child[u] = children[u][-1]          # left child = last child
            for j in range(len(children[u])-1, -1, -1):
                v = children[u][j]
                if j > 0:
                    right_child[v] = children[u][j-1]  # right child = previous sibling
                else:
                    right_child[v] = 0

    root = left_child[0]  # the whole binary tree is rooted at the last root
    if root == 0:         # N >= 1, so this never happens
        print(0)
        return

    # Iterative post-order to compute subtree sizes
    size = [0] * (N+1)
    stack = [(root, 0)]  # (node, state)  state 0: enter, 1: exit
    while stack:
        u, state = stack.pop()
        if state == 0:
            stack.append((u, 1))
            if left_child[u]:
                stack.append((left_child[u], 0))
            if right_child[u]:
                stack.append((right_child[u], 0))
        else:
            sz = 1
            if left_child[u]:
                sz += size[left_child[u]]
            if right_child[u]:
                sz += size[right_child[u]]
            size[u] = sz

    # Product of subtree sizes of all real nodes
    product = 1
    for i in range(1, N+1):
        product = (product * size[i]) % MOD

    # N! modulo MOD
    fact = 1
    for i in range(2, N+1):
        fact = (fact * i) % MOD

    # Answer = N! / product (mod MOD)
    ans = fact * pow(product, MOD-2, MOD) % MOD
    print(ans)


if __name__ == "__main__":
    solve()
```
