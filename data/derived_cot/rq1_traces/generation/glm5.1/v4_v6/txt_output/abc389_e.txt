```python
import sys
import math

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    P = list(map(int, data[2:2 + N]))
    P.sort()
    min_P = P[0]

    # Upper bound for the marginal cost threshold X.
    # The maximum marginal cost we ever pay is at most the marginal cost
    # of the last unit if we put all budget into the cheapest product.
    max_units_cheapest = int(math.isqrt(M // min_P)) + 2
    high = (2 * max_units_cheapest + 1) * min_P
    low = 0
    ans_X = 0

    # Binary search for the maximum X such that total cost <= M
    while low <= high:
        mid = (low + high) // 2
        cost = 0
        for p in P:
            if mid < p:
                break
            k = (mid // p + 1) // 2
            cost += k * k * p
            if cost > M:
                break
        if cost <= M:
            ans_X = mid
            low = mid + 1
        else:
            high = mid - 1

    total_units = 0
    total_cost = 0
    min_next = 10**30
    count_min = 0

    # Compute the result for the optimal X and collect next marginal costs
    for p in P:
        if ans_X < p:
            k = 0
        else:
            k = (ans_X // p + 1) // 2
        total_units += k
        total_cost += k * k * p
        nc = (2 * k + 1) * p
        if nc < min_next:
            min_next = nc
            count_min = 1
        elif nc == min_next:
            count_min += 1

    remaining = M - total_cost
    additional = min(count_min, remaining // min_next)
    print(total_units + additional)


if __name__ == "__main__":
    solve()
```
