```python
class Solution:
    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        # Convert points to perimeter positions
        def to_perimeter(x, y):
            if y == 0:
                return x
            elif x == side:
                return side + y
            elif y == side:
                return 2 * side + (side - x)
            else:  # x == 0
                return 3 * side + (side - y)
        
        # Manhattan distance
        def manhattan(p1, p2):
            return abs(p1[0] - p2[0]) + abs(p1[1] - p2[1])
        
        # Convert points to (perimeter_position, original_point)
        perimeter_points = [(to_perimeter(x, y), (x, y)) for x, y in points]
        perimeter_points.sort()
        
        n = len(perimeter_points)
        
        # Check if we can select k points with minimum distance >= d
        def can_select(d):
            # Try different starting points
            for start in range(n):
                selected = [perimeter_points[start][1]]
                for i in range(1, n):
                    idx = (start + i) % n
                    point = perimeter_points[idx][1]
                    # Check if this point is at least d away from all selected points
                    if all(manhattan(point, s) >= d for s in selected):
                        selected.append(point)
                        if len(selected) == k:
                            return True
            return False
        
        # Binary search on d
        lo, hi = 1, 2 * side
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if can_select(mid):
                lo = mid
            else:
                hi = mid - 1
        
        return lo
```

    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        # Convert points to perimeter positions
        def to_perimeter(x, y):
            if y == 0:
                return x
            elif x == side:
                return side + y
            elif y == side:
                return 2 * side + (side - x)
            else:  # x == 0
                return 3 * side + (side - y)
        
        # Manhattan distance
        def manhattan(p1, p2):
            return abs(p1[0] - p2[0]) + abs(p1[1] - p2[1])
        
        # Convert points to (perimeter_position, original_point)
        perimeter_points = [(to_perimeter(x, y), (x, y)) for x, y in points]
        perimeter_points.sort()
        
        n = len(perimeter_points)
        
        # Check if we can select k points with minimum distance >= d
        def can_select(d):
            # Try different starting points
            for start in range(n):
                selected = [perimeter_points[start][1]]
                for i in range(1, n):
                    idx = (start + i) % n
                    point = perimeter_points[idx][1]
                    # Check if this point is at least d away from all selected points
                    ok = True
                    for s in selected:
                        if manhattan(point, s) < d:
                            ok = False
                            break
                    if ok:
                        selected.append(point)
                        if len(selected) == k:
                            return True
            return False
        
        # Binary search on d

        while lo < hi:
            mid = (lo + hi + 1) // 2
            if can_select(mid):
                lo = mid
            else:
                hi = mid - 1
        
        return lo

class Solution:
    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        # Convert points to perimeter positions
        def to_perimeter(x, y):
            if y == 0:
                return x
            elif x == side:
                return side + y
            elif y == side:
                return 2 * side + (side - x)
            else:  # x == 0
                return 3 * side + (side - y)
        
        # Manhattan distance
        def manhattan(p1, p2):
            return abs(p1[0] - p2[0]) + abs(p1[1] - p2[1])
        
        # Convert points to (perimeter_position, original_point)
        perimeter_points = [(to_perimeter(x, y), (x, y)) for x, y in points]
        perimeter_points.sort()
        
        n = len(perimeter_points)
        
        # Check if we can select k points with minimum distance >= d
        def can_select(d):
            # Try different starting points
            for start in range(n):
                selected = [perimeter_points[start][1]]
                for i in range(1, n):
                    idx = (start + i) % n
                    point = perimeter_points[idx][1]
                    # Check if this point is at least d away from all selected points
                    ok = True
                    for s in selected:
                        if manhattan(point, s) < d:
                            ok = False
                            break
                    if ok:
                        selected.append(point)
                        if len(selected) == k:
                            return True
            return False
        
        # Binary search on d
        lo, hi = 1, 2 * side
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if can_select(mid):
                lo = mid
            else:
                hi = mid - 1
        
        return lo

Looking at this problem, I need to select k points on a square's boundary to maximize the minimum Manhattan distance between any pair.

**Key Observations:**
1. Points lie on the boundary, so I can parameterize them by their position along the perimeter
2. This is a classic "maximize minimum distance" problem - perfect for binary search
3. For a given distance d, I need to check if k points can be selected with all pairwise distances ≥ d

**Approach:**
1. Convert each point to its perimeter position (going around the square)
2. Binary search on the answer d
3. For each d, use a greedy check: try different starting points and greedily select points that are at least d away from all previously selected points

**Perimeter Position Calculation:**
- Bottom edge (y=0): position = x
- Right edge (x=side): position = side + y  
- Top edge (y=side): position = 2*side + (side - x)
- Left edge (x=0): position = 3*side + (side - y)

The greedy approach with multiple starting points works well because for points on a perimeter, optimal selections tend to follow the perimeter order.
