def countPairs(nums):
    n = len(nums)
    count = 0
    for i in range(n):
        for j in range(i+1, n):
            if is_almost_equal(nums[i], nums[j]):
                count += 1
    return count

def is_almost_equal(x, y):
    if x == y:
        return True
    if can_swap(x, y) or can_swap(y, x):
        return True
    return False

def can_swap(a, b):

    L = len(s)

            t = list(s)
            t[i], t[j] = t[j], t[i]
            new_val = int(''.join(t))

                return True
    return False

```python
from typing import List

class Solution:
    def countPairs(self, nums: List[int]) -> int:
        def can_swap(a: int, b: int) -> bool:
            s = str(a)
            n = len(s)
            for i in range(n):
                for j in range(i+1, n):
                    # swap digits at i and j
                    t = list(s)
                    t[i], t[j] = t[j], t[i]
                    if int(''.join(t)) == b:
                        return True
            return False
        
        count = 0
        n = len(nums)
        for i in range(n):
            for j in range(i+1, n):
                x, y = nums[i], nums[j]
                if x == y or can_swap(x, y) or can_swap(y, x):
                    count += 1
        return count
```

```python
class Solution:
    def countPairs(self, nums: List[int]) -> int:
        def can_swap(a: int, b: int) -> bool:
            s = str(a)
            n = len(s)
            for i in range(n):
                for j in range(i+1, n):
                    # swap digits at i and j
                    t = list(s)
                    t[i], t[j] = t[j], t[i]
                    if int(''.join(t)) == b:
                        return True
            return False
        
        count = 0
        n = len(nums)
        for i in range(n):
            for j in range(i+1, n):
                x, y = nums[i], nums[j]
                if x == y or can_swap(x, y) or can_swap(y, x):
                    count += 1
        return count
```

```python
            for i in range(n):
                for j in range(i + 1, n):
                    # swap digits at positions i and j
                    t = list(s)
                    t[i], t[j] = t[j], t[i]
                    if int(''.join(t)) == b:
                        return True
            return False

        count = 0
        n = len(nums)
        for i in range(n):
            for j in range(i + 1, n):
                x, y = nums[i], nums[j]
                if x == y or can_swap(x, y) or can_swap(y, x):
                    count += 1
        return count
```
