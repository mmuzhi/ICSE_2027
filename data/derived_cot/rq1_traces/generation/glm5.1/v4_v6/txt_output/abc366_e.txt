Input

The input is given from Standard Input in the following format:
N D

- 0 \leq D \leq 10^6

- (x_i, y_i) \neq (x_j, y_j) for i \neq j.
- All input values are integers.

def solve():
    import sys
    from bisect import bisect_right
    input = sys.stdin.readline

    N, D = map(int, input().split())
    xs = []
    ys = []
    for _ in range(N):
        x, y = map(int, input().split())
        xs.append(x)
        ys.append(y)

    xs.sort()
    ys.sort()

    pre_x = [0]
    for x in xs:
        pre_x.append(pre_x[-1] + x)
    pre_y = [0]
    for y in ys:
        pre_y.append(pre_y[-1] + y)

    total_x = pre_x[-1]
    total_y = pre_y[-1]

    def f(x):
        cnt = bisect_right(xs, x)
        sum_left = pre_x[cnt]
        return x * cnt - sum_left + (total_x - sum_left) - x * (N - cnt)

    def g(y):
        cnt = bisect_right(ys, y)
        sum_left = pre_y[cnt]
        return y * cnt - sum_left + (total_y - sum_left) - y * (N - cnt)

    f_min = f(xs[N//2])
    g_min = g(ys[N//2])
    if f_min > D or g_min > D:
        print(0)
        return

    # find xL, xR
    lo = -2000000 - 1
    hi = xs[N//2]   # f(hi) <= D
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if f(mid) <= D:
            hi = mid
        else:
            lo = mid
    xL = hi

    lo = xs[N//2]
    hi = 2000000 + 1
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if f(mid) <= D:
            lo = mid
        else:
            hi = mid
    xR = lo

    # find yL, yR
    lo = -2000000 - 1
    hi = ys[N//2]
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if g(mid) <= D:
            hi = mid
        else:
            lo = mid
    yL = hi

    lo = ys[N//2]
    hi = 2000000 + 1
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if g(mid) <= D:
            lo = mid
        else:
            hi = mid
    yR = lo

    # Build frequency maps for x and y
    from collections import Counter
    freqX = Counter(xs)
    freqY = Counter(ys)

    # cnt_g array
    cnt_g = [0] * (D + 1)

    # generate g(y) for y in [yL, yR]
    # initial
    y = yL
    g_val = g(yL)
    L_y = bisect_right(ys, yL)
    cnt_g[g_val] += 1
    for y in range(yL+1, yR+1):
        g_val = g_val + 2 * L_y - N
        L_y += freqY.get(y, 0)
        cnt_g[g_val] += 1

    # prefix sum
    for i in range(1, D+1):
        cnt_g[i] += cnt_g[i-1]

    # generate f(x) and accumulate answer
    ans = 0
    x = xL
    f_val = f(xL)
    L_x = bisect_right(xs, xL)
    ans += cnt_g[D - f_val]
    for x in range(xL+1, xR+1):
        f_val = f_val + 2 * L_x - N
        L_x += freqX.get(x, 0)
        ans += cnt_g[D - f_val]

    print(ans)

    pre_y = [0]
    for v in ys:
        pre_y.append(pre_y[-1] + v)

    total_x = pre_x[-1]
    total_y = pre_y[-1]

    def f(x: int) -> int:
        cnt = bisect_right(xs, x)
        sum_left = pre_x[cnt]
        return x * cnt - sum_left + (total_x - sum_left) - x * (N - cnt)

    def g(y: int) -> int:
        cnt = bisect_right(ys, y)
        sum_left = pre_y[cnt]
        return y * cnt - sum_left + (total_y - sum_left) - y * (N - cnt)

    # check if any solution exists
    f_min = f(xs[N // 2])
    g_min = g(ys[N // 2])
    if f_min > D or g_min > D:
        print(0)
        return

    # ----- find x interval -----
    # leftmost x with f(x) <= D
    lo = -2_000_000 - 1
    hi = xs[N // 2]          # f(hi) <= D
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if f(mid) <= D:
            hi = mid
        else:
            lo = mid
    xL = hi

    # rightmost x with f(x) <= D
    lo = xs[N // 2]
    hi = 2_000_000 + 1
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if f(mid) <= D:
            lo = mid
        else:
            hi = mid
    xR = lo

    # ----- find y interval -----
    lo = -2_000_000 - 1
    hi = ys[N // 2]
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if g(mid) <= D:
            hi = mid
        else:
            lo = mid
    yL = hi

    lo = ys[N // 2]
    hi = 2_000_000 + 1
    while lo + 1 < hi:
        mid = (lo + hi) // 2
        if g(mid) <= D:
            lo = mid
        else:
            hi = mid
    yR = lo

    # frequency of each coordinate
    freqX = Counter(xs)
    freqY = Counter(ys)

    # ----- build distribution of g(y) -----
    cnt_g = [0] * (D + 1)

    # initial y = yL
    y = yL
    g_val = g(yL)
    L_y = bisect_right(ys, yL)
    cnt_g[g_val] += 1

    for y in range(yL + 1, yR + 1):
        g_val = g_val + 2 * L_y - N
        L_y += freqY.get(y, 0)
        cnt_g[g_val] += 1

    # prefix sums -> cnt_g[i] = number of y with g(y) <= i
    for i in range(1, D + 1):
        cnt_g[i] += cnt_g[i - 1]

    # ----- count pairs -----
    ans = 0

    # initial x = xL
    x = xL
    f_val = f(xL)
    L_x = bisect_right(xs, xL)
    ans += cnt_g[D - f_val]

    for x in range(xL + 1, xR + 1):
        f_val = f_val + 2 * L_x - N
        L_x += freqX.get(x, 0)
        ans += cnt_g[D - f_val]

    print(ans)


if __name__ == "__main__":
    solve()
