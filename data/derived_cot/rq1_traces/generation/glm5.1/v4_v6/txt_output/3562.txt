You are given a 2D integer array intervals, where intervals[i] = [l_i, r_i, weight_i]. Interval i starts at position l_i and ends at r_i, and has a weight of weight_i. You can choose up to 4 non-overlapping intervals. The score of the chosen intervals is defined as the total sum of their weights.
Return the lexicographically smallest array of at most 4 indices from intervals with maximum score, representing your choice of non-overlapping intervals.
Two intervals are said to be non-overlapping if they do not share any points. In particular, intervals sharing a left or right boundary are considered overlapping.
An array a is lexicographically smaller than an array b if in the first position where a and b differ, array a has an element that is less than the corresponding element in b.
If the first min(a.length, b.length) elements do not differ, then the shorter array is the lexicographically smaller one.

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
        arr = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        arr.sort(key=lambda x: (x[1], x[0], x[3]))
        L = [x[0] for x in arr]
        R = [x[1] for x in arr]
        W = [x[2] for x in arr]
        orig = [x[3] for x in arr]
        import bisect
        ends = R
        p = [0]*n
        for i in range(n):
            p[i] = bisect.bisect_left(ends, L[i])  # count of intervals with end < L[i]
        INF_NEG = -10**18
        # dp[i][k] = (weight, indices_tuple)
        dp = [[(INF_NEG, None) for _ in range(5)] for _ in range(n+1)]
        dp[0][0] = (0, ())
        for k in range(1,5):
            dp[0][k] = (INF_NEG, None)
        for i in range(1, n+1):
            dp[i][0] = (0, ())
            for k in range(1,5):
                # not take
                not_w, not_idx = dp[i-1][k]
                # take
                take_w = INF_NEG
                take_idx = None
                prev = p[i-1]
                if dp[prev][k-1][0] != INF_NEG:
                    take_w = W[i-1] + dp[prev][k-1][0]
                    prev_idx = dp[prev][k-1][1]
                    # merge
                    lst = list(prev_idx)
                    lst.append(orig[i-1])
                    lst.sort()
                    take_idx = tuple(lst)
                # compare
                if take_w > not_w:
                    dp[i][k] = (take_w, take_idx)
                elif take_w < not_w:
                    dp[i][k] = (not_w, not_idx)
                else:  # equal
                    if take_w == INF_NEG:
                        dp[i][k] = (INF_NEG, None)
                    else:
                        if take_idx < not_idx:
                            dp[i][k] = (take_w, take_idx)
                        else:
                            dp[i][k] = (not_w, not_idx)
        best_w = INF_NEG
        best_idx = None
        for k in range(1,5):
            w, idx = dp[n][k]
            if w > best_w:
                best_w = w
                best_idx = idx
            elif w == best_w:
                if idx < best_idx:
                    best_idx = idx
        return list(best_idx)

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
        # attach original index
        arr = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        # sort by end, then start, then original index
        arr.sort(key=lambda x: (x[1], x[0], x[3]))
        L = [x[0] for x in arr]
        R = [x[1] for x in arr]
        W = [x[2] for x in arr]
        orig = [x[3] for x in arr]
        
        import bisect
        ends = R
        p = [0] * n
        for i in range(n):
            p[i] = bisect.bisect_left(ends, L[i])  # number of intervals with end < L[i]
        
        INF_NEG = -10**18
        # dp[i][k] = (total_weight, tuple_of_indices) for first i intervals, exactly k intervals
        dp = [[(INF_NEG, None) for _ in range(5)] for _ in range(n+1)]
        dp[0][0] = (0, ())
        for k in range(1, 5):
            dp[0][k] = (INF_NEG, None)
        
        for i in range(1, n+1):
            dp[i][0] = (0, ())
            for k in range(1, 5):
                # option 1: not take interval i-1
                not_w, not_idx = dp[i-1][k]
                # option 2: take interval i-1
                take_w = INF_NEG
                take_idx = None
                prev = p[i-1]
                if dp[prev][k-1][0] != INF_NEG:
                    take_w = W[i-1] + dp[prev][k-1][0]
                    prev_idx = dp[prev][k-1][1]
                    # merge indices and sort
                    take_idx = tuple(sorted(prev_idx + (orig[i-1],)))
                # decide
                if take_w > not_w:
                    dp[i][k] = (take_w, take_idx)
                elif take_w < not_w:
                    dp[i][k] = (not_w, not_idx)
                else:  # equal
                    if take_w == INF_NEG:
                        dp[i][k] = (INF_NEG, None)
                    else:
                        if take_idx < not_idx:
                            dp[i][k] = (take_w, take_idx)
                        else:
                            dp[i][k] = (not_w, not_idx)
        
        best_w = INF_NEG
        best_idx = None
        for k in range(1, 5):
            w, idx = dp[n][k]
            if w > best_w:
                best_w = w
                best_idx = idx
            elif w == best_w:
                if idx < best_idx:
                    best_idx = idx
        
        return list(best_idx)

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
        # attach original index
        arr = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        # sort by end, then start, then original index (deterministic)
        arr.sort(key=lambda x: (x[1], x[0], x[3]))
        L = [x[0] for x in arr]
        R = [x[1] for x in arr]
        W = [x[2] for x in arr]
        orig = [x[3] for x in arr]

        from bisect import bisect_left
        ends = R
        p = [0] * n
        for i in range(n):
            # number of intervals with end < L[i] (all are before i in sorted order)
            p[i] = bisect_left(ends, L[i])

        INF_NEG = -10**18
        # dp[i][k] = (total_weight, tuple_of_indices) for first i intervals, exactly k intervals
        dp = [[(INF_NEG, None) for _ in range(5)] for _ in range(n+1)]
        dp[0][0] = (0, ())
        for k in range(1, 5):
            dp[0][k] = (INF_NEG, None)

        for i in range(1, n+1):
            dp[i][0] = (0, ())
            max_k = min(4, i)
            for k in range(1, max_k+1):
                # option 1: skip interval i-1
                not_w, not_idx = dp[i-1][k]
                # option 2: take interval i-1
                take_w = INF_NEG
                take_idx = None
                prev = p[i-1]
                if dp[prev][k-1][0] != INF_NEG:
                    take_w = W[i-1] + dp[prev][k-1][0]
                    prev_idx = dp[prev][k-1][1]
                    # merge indices and keep sorted
                    take_idx = tuple(sorted(prev_idx + (orig[i-1],)))
                # decide which candidate is better
                if take_w > not_w:
                    dp[i][k] = (take_w, take_idx)
                elif take_w < not_w:
                    dp[i][k] = (not_w, not_idx)
                else:  # equal weights
                    if take_w == INF_NEG:
                        dp[i][k] = (INF_NEG, None)
                    else:
                        # choose lexicographically smaller index tuple
                        if take_idx < not_idx:
                            dp[i][k] = (take_w, take_idx)
                        else:
                            dp[i][k] = (not_w, not_idx)

        best_w = INF_NEG
        best_idx = None
        for k in range(1, 5):
            w, idx = dp[n][k]
            if w > best_w:
                best_w = w
                best_idx = idx
            elif w == best_w:
                if idx < best_idx:
                    best_idx = idx

        return list(best_idx)
