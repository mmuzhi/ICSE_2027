```python

import sys

def solve() -> None:
    input = sys.stdin.readline
    K = int(input())
    S = input().strip()
    T = input().strip()
    n = len(S)
    m = len(T)

    # trivial impossibility
    if abs(n - m) > K:
        print("No")
        return
    # equality shortcut
    if S == T:
        print("Yes")
        return

    INF = K + 1               # any value > K
    band = 2 * K + 1          # number of diagonals we keep
    dp = [INF] * band         # current row (i)
    ndp = [INF] * band        # next row (i+1)

    # ----- initialise row i = 0 -----
    i0 = 0
    j_low = max(0, i0 - K)
    j_high = min(m, i0 + K)
    for j in range(j_low, j_high + 1):
        idx = j - i0 + K          # = j + K
        dp[idx] = j               # distance from empty S to T[0:j]

    # ----- process rows i = 0 .. n-1 -----
    for i in range(n):
        # reset ndp to INF
        for k in range(band):
            ndp[k] = INF

        j_low = max(0, (i + 1) - K)
        j_high = min(m, (i + 1) + K)
        s_char = S[i]

        for j in range(j_low, j_high + 1):
            idx_prime = j - (i + 1) + K   # column in ndp
            idx = j - i + K               # column in dp (for deletion)

            # deletion
            if idx <= 2 * K:
                del_cost = dp[idx] + 1
            else:
                del_cost = INF

            # insertion
            if idx_prime > 0:
                ins_cost = ndp[idx_prime - 1] + 1
            else:
                ins_cost = INF

            # substitution
            if j > 0:
                sub_cost = dp[idx - 1] + (0 if s_char == T[j - 1] else 1)
            else:
                sub_cost = INF

            # take minimum
            val = del_cost
            if ins_cost < val:
                val = ins_cost
            if sub_cost < val:
                val = sub_cost

            # cap values > K
            if val > K:
                val = INF
            ndp[idx_prime] = val

        # swap rows for next iteration
        dp, ndp = ndp, dp

    # ----- answer -----
    final_idx = m - n + K
    if dp[final_idx] <= K:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    solve()
```
