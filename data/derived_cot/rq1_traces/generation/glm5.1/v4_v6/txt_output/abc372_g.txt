import sys
from fractions import Fraction

def floor_sum(n, m, a, b):
    ans = 0
    while True:
        if a >= m:
            ans += (n - 1) * n * (a // m) // 2
            a %= m
        if b >= m:
            ans += n * (b // m)
            b %= m
        y_max = a * n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans

def solve():
    data = sys.stdin.read().split()
    it = iter(data)
    T = int(next(it))
    out = []
    for _ in range(T):
        N = int(next(it))
        lines = []
        for _ in range(N):
            A = int(next(it))
            B = int(next(it))
            C = int(next(it))
            C_prime = C - 1
            lines.append((A, B, C_prime))
        # sort lines by A/B ascending, then C'/B ascending
        lines.sort(key=lambda x: (Fraction(x[0], x[1]), Fraction(x[2], x[1])))
        unique_lines = []
        for A, B, C_prime in lines:
            if unique_lines:
                A_last, B_last, _ = unique_lines[-1]
                if A * B_last == A_last * B:
                    # same slope, skip because the first one has smaller intercept
                    continue
            unique_lines.append((A, B, C_prime))
        
        stack = []  # each element: (A, B, C_prime, (num, den) or None)
        for A, B, C_prime in unique_lines:
            while True:
                if not stack:
                    stack.append((A, B, C_prime, None))
                    break
                A_last, B_last, C_last, x_last = stack[-1]
                num = C_prime * B_last - C_last * B
                den = A * B_last - A_last * B
                # den > 0 because A/B > A_last/B_last
                if num <= 0:
                    stack.pop()
                    continue
                if len(stack) >= 2:
                    A_prev, B_prev, C_prev, x_prev = stack[-2]
                    num_prev = C_last * B_prev - C_prev * B_last
                    den_prev = A_last * B_prev - A_prev * B_last
                    if num * den_prev <= num_prev * den:
                        stack.pop()
                        continue
                stack.append((A, B, C_prime, (num, den)))
                break
        
        total = 0
        for i, (A, B, C_prime, x_start_frac) in enumerate(stack):
            if x_start_frac is None:
                L = 1
            else:
                num, den = x_start_frac
                L = num // den + 1
                if L < 1:
                    L = 1
            if i + 1 < len(stack):
                next_num, next_den = stack[i+1][3]
                R = next_num // next_den
            else:
                R = float('inf')
            if C_prime < B:
                X_max = -1
            else:
                X_max = (C_prime - B) // A
            if R > X_max:
                R = X_max
            if L <= R:
                n = R - L + 1
                a = A
                b = C_prime - A * R
                m = B
                total += floor_sum(n, m, a, b)
            if R == X_max:
                break
        out.append(str(total))
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
