```python
from itertools import product
from math import factorial

class Solution:
    def countGoodIntegers(self, n: int, k: int) -> int:
        fact = [factorial(i) for i in range(n+1)]
        valid_multisets = set()
        
        if n == 1:
            for d in range(1, 10):
                if d % k == 0:
                    counts = [0]*10
                    counts[d] = 1
                    valid_multisets.add(tuple(counts))
        else:
            half_len = (n + 1) // 2  # number of digits to generate
            # first digit cannot be 0
            for half in product(range(10), repeat=half_len):
                if half[0] == 0:
                    continue
                # build palindrome
                if n % 2 == 0:
                    pal_digits = list(half) + list(half[::-1])
                else:
                    pal_digits = list(half) + list(half[-2::-1])
                # convert to integer
                pal = 0
                for d in pal_digits:
                    pal = pal * 10 + d
                if pal % k == 0:
                    counts = [0]*10
                    for d in pal_digits:
                        counts[d] += 1
                    valid_multisets.add(tuple(counts))
        
        ans = 0
        for counts in valid_multisets:
            total = fact[n]
            for c in counts:
                total //= fact[c]
            if counts[0] > 0:
                zero_start = fact[n-1]
                zero_start //= fact[counts[0]-1]
                for i in range(1, 10):
                    zero_start //= fact[counts[i]]
                total -= zero_start
            ans += total
        return ans
```

```python
from math import factorial
from itertools import product

class Solution:
    def countGoodIntegers(self, n: int, k: int) -> int:
        # Precompute factorials up to n
        fact = [factorial(i) for i in range(n + 1)]
        valid_multisets = set()
        
        if n == 1:
            # Single-digit numbers: 1..9
            for d in range(1, 10):
                if d % k == 0:
                    counts = [0] * 10
                    counts[d] = 1
                    valid_multisets.add(tuple(counts))
        else:
            half_len = (n + 1) // 2  # digits in the first half (including middle if odd)
            # Generate all possible first halves without leading zero
            for half in product(range(10), repeat=half_len):
                if half[0] == 0:
                    continue
                # Mirror the half to form the full palindrome
                if n % 2 == 0:
                    pal_digits = list(half) + list(half[::-1])
                else:
                    pal_digits = list(half) + list(half[-2::-1])
                # Build the integer value
                pal = 0
                for d in pal_digits:
                    pal = pal * 10 + d
                if pal % k == 0:
                    # Record the digit frequency multiset
                    counts = [0] * 10
                    for d in pal_digits:
                        counts[d] += 1
                    valid_multisets.add(tuple(counts))
        
        ans = 0
        # For each unique multiset, count valid n-digit numbers (no leading zero)
        for counts in valid_multisets:
            total = fact[n]
            for c in counts:
                total //= fact[c]
            if counts[0] > 0:
                # Subtract arrangements that start with zero
                zero_start = fact[n - 1]
                zero_start //= fact[counts[0] - 1]
                for i in range(1, 10):
                    zero_start //= fact[counts[i]]
                total -= zero_start
            ans += total
            
        return ans
```
