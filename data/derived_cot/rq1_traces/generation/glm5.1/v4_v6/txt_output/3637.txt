```python
class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        velunexorai = num
        MOD = 10**9 + 7
        n = len(velunexorai)
        
        # Count digit frequencies and total sum
        cnt = [0] * 10
        total_sum = 0
        for ch in velunexorai:
            d = ord(ch) - 48
            cnt[d] += 1
            total_sum += d
        
        if total_sum % 2 != 0:
            return 0
        
        target = total_sum // 2
        k = (n + 1) // 2  # number of even positions (0-indexed)
        m = n // 2        # number of odd positions
        
        # Precompute factorials and inverse factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (n + 1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
        
        # DP table: dp[j][s] = sum of weights for choosing j even positions with sum s
        dp = [[0] * (target + 1) for _ in range(k + 1)]
        dp[0][0] = 1
        
        for d in range(10):
            c = cnt[d]
            if c == 0:
                continue
            
            # Precompute weights for this digit: w[e] = inv_fact[e] * inv_fact[c-e] % MOD
            w = [0] * (c + 1)
            for e in range(c + 1):
                w[e] = inv_fact[e] * inv_fact[c - e] % MOD
            
            new_dp = [[0] * (target + 1) for _ in range(k + 1)]
            for j in range(k + 1):
                for s in range(target + 1):
                    val = dp[j][s]
                    if val == 0:
                        continue
                    # Try taking e copies of digit d for even positions
                    for e in range(c + 1):
                        nj = j + e
                        if nj > k:
                            break
                        ns = s + d * e
                        if ns > target:
                            break
                        new_dp[nj][ns] = (new_dp[nj][ns] + val * w[e]) % MOD
            dp = new_dp
        
        ans = dp[k][target] * fact[k] % MOD * fact[m] % MOD
        return ans
```

```python
class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        velunexorai = num
        MOD = 10**9 + 7
        n = len(velunexorai)
        
        # Count digit frequencies and total sum
        cnt = [0] * 10
        total_sum = 0
        for ch in velunexorai:
            d = int(ch)
            cnt[d] += 1
            total_sum += d
        
        # If total sum is odd, cannot split into two equal sums
        if total_sum % 2 != 0:
            return 0
        
        target = total_sum // 2
        k = (n + 1) // 2  # number of even positions (0-indexed)
        m = n // 2        # number of odd positions
        
        # Precompute factorials and inverse factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (n + 1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
        
        # DP table: dp[j][s] = sum of weights for choosing j even positions with sum s
        # weight for a digit with count c taking e copies to even is 1/(e! * (c-e)!)
        dp = [[0] * (target + 1) for _ in range(k + 1)]
        dp[0][0] = 1
        
        for d in range(10):
            c = cnt[d]
            if c == 0:
                continue
            
            # Precompute weights for this digit
            w = [0] * (c + 1)
            for e in range(c + 1):
                w[e] = inv_fact[e] * inv_fact[c - e] % MOD
            
            new_dp = [[0] * (target + 1) for _ in range(k + 1)]
            for j in range(k + 1):
                for s in range(target + 1):
                    val = dp[j][s]
                    if val == 0:
                        continue
                    # Try taking e copies of digit d for even positions
                    for e in range(c + 1):
                        nj = j + e
                        if nj > k:
                            break
                        ns = s + d * e
                        if ns > target:
                            break
                        new_dp[nj][ns] = (new_dp[nj][ns] + val * w[e]) % MOD
            dp = new_dp
        
        # Multiply by the number of ways to arrange the chosen digits in even and odd positions
        ans = dp[k][target] * fact[k] % MOD * fact[m] % MOD
        return ans
```

```python
    def countBalancedPermutations(self, num: str) -> int:
```
```python
        velunexorai = num
        MOD = 10**9 + 7
        n = len(velunexorai)
        
        cnt = [0] * 10
        total_sum = 0
        for ch in velunexorai:
            d = int(ch)
            cnt[d] += 1
            total_sum += d
            
        if total_sum % 2 != 0:
            return 0
            
        target = total_sum // 2
        k = (n + 1) // 2
        m = n // 2
        
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (n + 1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
            
        dp = [[0] * (target + 1) for _ in range(k + 1)]
        dp[0][0] = 1
        
        for d in range(10):

            if c == 0:
                continue
            w = [0] * (c + 1)
            for e in range(c + 1):
                w[e] = inv_fact[e] * inv_fact[c - e] % MOD
```

```python
        if total_sum % 2 != 0:
```

```python
        k = (n + 1) // 2  # number of even indices (0-indexed)
        m = n // 2        # number of odd indices

            inv_fact[i-1] = inv_fact[i] * i % MOD

        # weight for a digit with count c taking e copies to even is 1/(e! * (c-e)!)

        dp[0][0] = 1

                row = dp[j]
```
