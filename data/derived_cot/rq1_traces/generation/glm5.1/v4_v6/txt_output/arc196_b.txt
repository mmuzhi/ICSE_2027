def solve_one(H, W, S):
    N = H * W
    c = [[0]*W for _ in range(H)]
    row_sum = [0]*H
    col_sum = [0]*W
    for i in range(H):
        row = S[i]
        for j in range(W):
            if row[j] == 'A':
                c[i][j] = 1
                row_sum[i] += 1
                col_sum[j] += 1
    for s in row_sum:
        if s % 2 != 0:
            return 0
    for s in col_sum:
        if s % 2 != 0:
            return 0
    
    # row_cum[i][j] = d_{i,j}
    row_cum = [[0]*W for _ in range(H)]
    for i in range(H):
        s = 0
        for j in range(1, W):
            s ^= c[i][j]
            row_cum[i][j] = s
        # row_cum[i][0] is 0, row_cum[i][W-1] is sum_{k=1}^{W-1} c_{i,k}
    
    # col_cum[j][i] = e_{i,j}
    col_cum = [[0]*H for _ in range(W)]
    for j in range(W):
        s = 0
        for i in range(1, H):
            s ^= c[i][j]
            col_cum[j][i] = s
        # col_cum[j][0] = 0, col_cum[j][H-1] = sum_{k=1}^{H-1} c_{k,j}
    
    adj = [[] for _ in range(H + W)]
    for i in range(H):
        for j in range(W):
            if c[i][j] == 0:
                # B cell
                d = row_cum[i][j-1] if j > 0 else row_cum[i][W-1]
                e = col_cum[j][i-1] if i > 0 else col_cum[j][H-1]
                b = 1 ^ d ^ e
                u = i
                v = H + j
                adj[u].append((v, b))
                adj[v].append((u, b))
    
    val = [-1] * (H + W)
    comps = 0
    stack = []
    for start in range(H + W):
        if val[start] != -1:
            continue
        comps += 1
        val[start] = 0
        stack.append(start)
        while stack:
            u = stack.pop()
            for v, w in adj[u]:
                if val[v] == -1:
                    val[v] = val[u] ^ w
                    stack.append(v)
                else:
                    if val[u] ^ val[v] != w:
                        return 0
    return pow(2, comps, 998244353)

                        if val[v3] == -1:
                            val[v3] = val[u] ^ 1
                            stack.append(v3)
                        elif val[u] ^ val[v3] != 1:
                            ok = False
                            break
                else:              # h variable
                    idx2 = u - N
                    i = idx2 // W
                    j = idx2 % W

                    # neighbour 1: h_{i-1, j}
                    im = prev_i[i]
                    v1 = N + im * W + j
                    w1 = c[i * W + j]
                    if val[v1] == -1:
                        val[v1] = val[u] ^ w1
                        stack.append(v1)
                    elif val[u] ^ val[v1] != w1:
                        ok = False
                        break

                    # neighbour 2: h_{i+1, j}
                    ip = next_i[i]
                    v2 = N + ip * W + j
                    w2 = c[ip * W + j]
                    if val[v2] == -1:
                        val[v2] = val[u] ^ w2
                        stack.append(v2)
                    elif val[u] ^ val[v2] != w2:
                        ok = False
                        break

                    # neighbour 3: cross edge if cell (i+1, j) is B
                    if w2 == 0:
                        jm = prev_j[j]
                        v3 = ip * W + jm
                        if val[v3] == -1:
                            val[v3] = val[u] ^ 1
