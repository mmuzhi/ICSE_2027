```python
def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))
    
    buckets = {}
    for a in A:
        v = (a & -a).bit_length() - 1  # alternative: v = (a ^ (a-1)).bit_length() - 1? Actually v_2(a) = number of trailing zeros.
        # We can compute v by:
        # v = 0
        # temp = a
        # while temp % 2 == 0:
        #     temp //= 2
        #     v += 1
        # o = temp
        # But using bit operations is faster:
        v = (a & -a).bit_length() - 1  # because a & -a gives the lowest set bit as a power of 2. bit_length of 2^k is k+1.
        o = a >> v
        if v not in buckets:
            buckets[v] = []
        buckets[v].append(o)
    
    total_ordered_sum = 0
    total_f_A = 0
    bucket_stats = {}  # v -> (C, S, freq)
    
    for v, odds in buckets.items():
        freq = {}
        for o in odds:
            freq[o] = freq.get(o, 0) + 1
        C = len(odds)
        S = sum(o * freq[o] for o in freq)  # sum of odd parts
        total_f_A += S
        bucket_stats[v] = (C, S, freq)
        
        # compute same bucket ordered sum
        items = list(freq.items())
        max_o = max(freq.keys())
        max_sum = 2 * max_o
        max_k = max_sum.bit_length()
        T = [0] * (max_k + 1)
        for k in range(1, max_k + 1):
            mask = (1 << k) - 1
            stats = {}
            for o, cnt in items:
                r = o & mask
                if r in stats:
                    c, s = stats[r]
                    stats[r] = (c + cnt, s + cnt * o)
                else:
                    stats[r] = (cnt, cnt * o)
            Tk = 0
            for r, (c, s) in stats.items():
                r_prime = (-r) & mask
                if r_prime in stats:
                    Tk += s * stats[r_prime][0]
            T[k] = Tk * 2
        same = T[1] >> 1
        for k in range(2, max_k + 1):
            same -= T[k] >> k
        total_ordered_sum += same
    
    # cross buckets
    v_list = sorted(bucket_stats.keys())
    for i in range(len(v_list)):
        v1 = v_list[i]
        C1, S1, _ = bucket_stats[v1]
        for j in range(i+1, len(v_list)):
            v2 = v_list[j]
            C2, S2, _ = bucket_stats[v2]
            d = v2 - v1  # v2 > v1
            cross = 2 * (S1 * C2 + S2 * C1 * (1 << d))
            total_ordered_sum += cross
    
    ans = (total_ordered_sum + total_f_A) // 2
    print(ans)
```

```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1 + N]))

    # Group numbers by their 2-adic valuation v.
    # For each number, store its odd part o = A_i / 2^v.
    buckets = {}
    for a in A:
        # v = exponent of 2 in a
        v = (a & -a).bit_length() - 1
        o = a >> v
        if v in buckets:
            buckets[v].append(o)
        else:
            buckets[v] = [o]

    total_ordered_sum = 0
    total_f_A = 0
    bucket_stats = {}  # v -> (count, sum_of_odd_parts, freq_dict)

    for v, odds in buckets.items():
        # Build frequency map of odd parts in this bucket
        freq = {}
        for o in odds:
            freq[o] = freq.get(o, 0) + 1
        C = len(odds)
        S = 0
        for o, cnt in freq.items():
            S += o * cnt
        total_f_A += S
        bucket_stats[v] = (C, S, freq)

        # --- Compute sum_{x,y in same bucket} f(x+y) ---
        items = [(o, cnt, o * cnt) for o, cnt in freq.items()]
        max_o = max(freq.keys())
        max_sum = 2 * max_o
        max_k = max_sum.bit_length()  # maximum possible v_2(a+b) + 1
        T = [0] * (max_k + 1)          # T[k] for k = 1..max_k

        for k in range(1, max_k + 1):
            mask = (1 << k) - 1
            stats = {}  # residue -> (count, sum)
            for o, cnt, o_cnt in items:
                r = o & mask
                if r in stats:
                    c, s = stats[r]
                    stats[r] = (c + cnt, s + o_cnt)
                else:
                    stats[r] = (cnt, o_cnt)
            Tk = 0
            for r, (c, s) in stats.items():
                r_prime = (-r) & mask
                if r_prime in stats:
                    Tk += s * stats[r_prime][0]
            T[k] = Tk << 1  # Tk * 2

        same = T[1] >> 1
        for k in range(2, max_k + 1):
            same -= T[k] >> k
        total_ordered_sum += same

    # --- Cross-bucket terms ---
    v_list = sorted(bucket_stats.keys())
    for i in range(len(v_list)):
        v1 = v_list[i]
        C1, S1, _ = bucket_stats[v1]
        for j in range(i + 1, len(v_list)):
            v2 = v_list[j]
            C2, S2, _ = bucket_stats[v2]
            d = v2 - v1  # v2 > v1
            cross = (S1 * C2 + S2 * C1 * (1 << d)) << 1  # 2 * (...)
            total_ordered_sum += cross

    # Final answer
    ans = (total_ordered_sum + total_f_A) // 2
    print(ans)


if __name__ == "__main__":
    solve()
```
