import sys
import heapq

def solve():
    input = sys.stdin.readline
    N, K = map(int, input().split())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    C = list(map(int, input().split()))
    A.sort(reverse=True)
    B.sort(reverse=True)
    C.sort(reverse=True)

    SHIFT = 18
    MASK = (1 << SHIFT) - 1

    # initial state
    i0, j0, k0 = 0, 0, 0
    val0 = A[i0]*B[j0] + B[j0]*C[k0] + C[k0]*A[i0]
    heap = [(-val0, (i0 << (SHIFT*2)) | (j0 << SHIFT) | k0)]

    for _ in range(K-1):
        neg, packed = heapq.heappop(heap)
        i = packed >> (SHIFT*2)
        j = (packed >> SHIFT) & MASK
        k = packed & MASK

        # (i+1, j, k)
        if i + 1 < N:
            ni, nj, nk = i+1, j, k
            v = A[ni]*B[nj] + B[nj]*C[nk] + C[nk]*A[ni]
            heapq.heappush(heap, (-v, (ni << (SHIFT*2)) | (nj << SHIFT) | nk))

        # (i, j+1, k) if i == 0
        if i == 0 and j + 1 < N:
            ni, nj, nk = i, j+1, k
            v = A[ni]*B[nj] + B[nj]*C[nk] + C[nk]*A[ni]
            heapq.heappush(heap, (-v, (ni << (SHIFT*2)) | (nj << SHIFT) | nk))

        # (i, j, k+1) if i == 0 and j == 0
        if i == 0 and j == 0 and k + 1 < N:
            ni, nj, nk = i, j, k+1
            v = A[ni]*B[nj] + B[nj]*C[nk] + C[nk]*A[ni]
            heapq.heappush(heap, (-v, (ni << (SHIFT*2)) | (nj << SHIFT) | nk))

    # K-th largest
    neg, packed = heapq.heappop(heap)
    print(-neg)

if __name__ == "__main__":
    solve()

import sys
import heapq

def solve() -> None:
    input = sys.stdin.readline
    N, K = map(int, input().split())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    C = list(map(int, input().split()))

    # sort descending – monotonicity of the value
    A.sort(reverse=True)
    B.sort(reverse=True)
    C.sort(reverse=True)

    SHIFT = 18                     # N ≤ 200000 < 2^18
    MASK = (1 << SHIFT) - 1

    # initial state (0,0,0)
    val0 = A[0] * B[0] + B[0] * C[0] + C[0] * A[0]
    packed0 = (0 << (SHIFT * 2)) | (0 << SHIFT) | 0
    heap = [(-val0, packed0)]

    for _ in range(K - 1):
        neg, packed = heapq.heappop(heap)
        i = packed >> (SHIFT * 2)
        j = (packed >> SHIFT) & MASK
        k = packed & MASK

        # (i+1, j, k)
        if i + 1 < N:
            ni, nj, nk = i + 1, j, k
            v = A[ni] * B[nj] + B[nj] * C[nk] + C[nk] * A[ni]
            heapq.heappush(heap, (-v, (ni << (SHIFT * 2)) | (nj << SHIFT) | nk))

        # (i, j+1, k)  only if i == 0
        if i == 0 and j + 1 < N:
            ni, nj, nk = i, j + 1, k
            v = A[ni] * B[nj] + B[nj] * C[nk] + C[nk] * A[ni]
            heapq.heappush(heap, (-v, (ni << (SHIFT * 2)) | (nj << SHIFT) | nk))

        # (i, j, k+1)  only if i == 0 and j == 0
        if i == 0 and j == 0 and k + 1 < N:
            ni, nj, nk = i, j, k + 1
            v = A[ni] * B[nj] + B[nj] * C[nk] + C[nk] * A[ni]
            heapq.heappush(heap, (-v, (ni << (SHIFT * 2)) | (nj << SHIFT) | nk))

    # K-th largest value
    neg, _ = heapq.heappop(heap)
    print(-neg)
