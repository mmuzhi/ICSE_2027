We have a directed graph with N vertices, numbered 1, 2, \ldots, N.
Information about the edges is given by N^2 characters C_{1, 1}, C_{1, 2}, \ldots, C_{1, N}, C_{2, 1}, \ldots, C_{N, N}. Here, each C_{i, j} is either a lowercase English letter or -.

- Among all (not necessarily simple) paths from vertex i to vertex j whose concatenation of labels on the edges forms a palindrome, what is the length of the shortest such path? If there is no such path, the answer is -1.

Input

The input is given from Standard Input in the following format:

C_{1, 1}C_{1, 2}\ldotsC_{1, N}

\vdots
C_{N, 1}C_{N, 2}\ldotsC_{N, N}

Output

Let A_{i, j} be the answer to the question for the pair (i, j). Print them in the following format:
A_{1, 1} A_{1, 2} \ldots A_{1, N}
A_{2, 1} A_{2, 2} \ldots A_{2, N}
\vdots
A_{N, 1} A_{N, 2} \ldots A_{N, N}

Constraints


- 1 \leq N \leq 100
- N is an integer.
- Each C_{i, j} is either a lowercase English letter or -.

Sample Input 1

ab--
--b-
---a
c---

Sample Output 1

0 1 2 4
-1 0 1 -1
3 -1 0 1
1 -1 -1 0

Note that the empty string is also a palindrome.

```python
import sys
from collections import deque

def solve():
    input = sys.stdin.readline
    N = int(input().strip())
    # initialize data structures
    INF = 10**9
    has_edge = [[False]*N for _ in range(N)]
    in_c = [ [ [] for _ in range(N) ] for _ in range(26) ]
    out_c = [ [ [] for _ in range(N) ] for _ in range(26) ]
    
    for i in range(N):
        line = input().strip()
        for j, ch in enumerate(line):
            if ch != '-':
                c = ord(ch) - 97
                out_c[c][i].append(j)
                in_c[c][j].append(i)
                has_edge[i][j] = True
    
    dist = [[INF]*N for _ in range(N)]
    # diagonal
    for i in range(N):
        dist[i][i] = 0
    # edges
    for i in range(N):
        for j in range(N):
            if i != j and has_edge[i][j]:
                if dist[i][j] > 1:
                    dist[i][j] = 1
    
    dq = deque()
    # push all distance 0 states
    for i in range(N):
        dq.append((i, i))
    # push all distance 1 states
    for i in range(N):
        for j in range(N):
            if dist[i][j] == 1:
                dq.append((i, j))
    
    while dq:
        u, v = dq.popleft()
        d = dist[u][v]
        # iterate over labels
        for c in range(26):
            in_list = in_c[c][u]
            if not in_list:
                continue
            out_list = out_c[c][v]
            if not out_list:
                continue
            nd = d + 2
            for x in in_list:
                for y in out_list:
                    if dist[x][y] == INF:
                        dist[x][y] = nd
                        dq.append((x, y))
    
    # output
    out_lines = []
    for i in range(N):
        row = []
        for j in range(N):
            val = dist[i][j]
            if val == INF:
                row.append("-1")
            else:
                row.append(str(val))
        out_lines.append(" ".join(row))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```

```python
import sys
from collections import deque

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    INF = 10 ** 9

    # data structures
    has_edge = [[False] * N for _ in range(N)]
    in_c = [[[] for _ in range(N)] for _ in range(26)]   # in_c[c][v] = list of u with u->v label c
    out_c = [[[] for _ in range(N)] for _ in range(26)]  # out_c[c][u] = list of v with u->v label c

    for i in range(N):
        line = input().strip()
        for j, ch in enumerate(line):
            if ch != '-':
                c = ord(ch) - 97
                out_c[c][i].append(j)
                in_c[c][j].append(i)
                has_edge[i][j] = True

    # distance matrix
    dist = [[INF] * N for _ in range(N)]
    visited = 0

    # distance 0 : diagonal
    for i in range(N):
        dist[i][i] = 0
        visited += 1

    # distance 1 : direct edges (single character is a palindrome)
    for i in range(N):
        for j in range(N):
            if i != j and has_edge[i][j] and dist[i][j] == INF:
                dist[i][j] = 1
                visited += 1

    # if all states already have a distance, we are done
    if visited < N * N:
        # collect labels that actually appear
        active_labels = []
        for c in range(26):
            for i in range(N):
                if out_c[c][i]:
                    active_labels.append(c)
                    break

        dq = deque()
        # push all distance 0 states
        for i in range(N):
            dq.append((i, i))
        # push all distance 1 states
        for i in range(N):
            for j in range(N):
                if dist[i][j] == 1:
                    dq.append((i, j))

        full = False
        while dq and not full:
            u, v = dq.popleft()
            d = dist[u][v]
            nd = d + 2
            for c in active_labels:
                if full:
                    break
                in_list = in_c[c][u]
                if not in_list:
                    continue
                out_list = out_c[c][v]
                if not out_list:
                    continue
                for x in in_list:
                    if full:
                        break
                    row_x = dist[x]
                    for y in out_list:
                        if row_x[y] == INF:
                            row_x[y] = nd
                            dq.append((x, y))
                            visited += 1
                            if visited == N * N:
                                full = True
                                break

    # output
    out_lines = []
    for i in range(N):
        row = []
        for j in range(N):
            val = dist[i][j]
            row.append(str(val) if val != INF else "-1")
        out_lines.append(" ".join(row))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
