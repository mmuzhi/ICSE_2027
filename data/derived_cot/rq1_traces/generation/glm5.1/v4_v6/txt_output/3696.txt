Implementation steps:

1. Convert s to list of ints: digits = [int(c) for c in s].
2. Count occurrences of each digit 1-9: digit_counts = [0]*10; for d in digits: if d!=0: digit_counts[d] += 1.
3. ans = 0.
4. For d in range(1,10):
    if digit_counts[d] == 0: continue
    freq = [0]*d
    for digit in digits:
        new_freq = [0]*d
        # update from previous substrings
        for r in range(d):
            cnt = freq[r]
            if cnt:
                new_r = (r*10 + digit) % d
                new_freq[new_r] += cnt
        # add single char substring
        new_freq[digit % d] += 1
        if digit == d:
            ans += new_freq[0]
        freq = new_freq

Return ans.

    digits = [int(ch) for ch in s]
    n = len(digits)
    # count digit frequencies to skip d with zero occurrences
    cnt = [0]*10
    for d in digits:

class Solution:
    def countSubstrings(self, s: str) -> int:
        digits = [int(c) for c in s]
        cnt = [0]*10
        for d in digits:
            if d:
                cnt[d] += 1
        ans = 0
        for d in range(1, 10):
            if cnt[d] == 0:
                continue
            freq = [0]*d
            for digit in digits:
                new_freq = [0]*d
                # extend previous substrings
                for r in range(d):
                    c = freq[r]
                    if c:
                        new_r = (r*10 + digit) % d
                        new_freq[new_r] += c
                # add single character substring
                new_freq[digit % d] += 1
                if digit == d:
                    ans += new_freq[0]
                freq = new_freq
        return ans

```python
class Solution:
    def countSubstrings(self, s: str) -> int:
        # convert to list of integers for faster access
        digits = [int(ch) for ch in s]
        # count occurrences of each non‑zero digit
        cnt = [0] * 10
        for d in digits:
            if d:
                cnt[d] += 1

        ans = 0
        # process each possible last digit (1..9)
        for mod in range(1, 10):
            if cnt[mod] == 0:
                continue          # this digit never appears, skip

            freq = [0] * mod      # freq[r] = # of substrings ending at previous position with remainder r
            for digit in digits:
                new_freq = [0] * mod

                # extend all substrings that ended at the previous position
                for r in range(mod):
                    c = freq[r]
                    if c:
                        new_r = (r * 10 + digit) % mod
                        new_freq[new_r] += c

                # add the single‑character substring consisting of the current digit
                new_freq[digit % mod] += 1

                # if the current digit equals the modulus, all substrings ending here
                # that are divisible by mod contribute to the answer
                if digit == mod:
                    ans += new_freq[0]

                freq = new_freq   # move to next position

        return ans
```
