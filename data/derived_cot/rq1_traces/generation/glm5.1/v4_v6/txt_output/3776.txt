class Solution:
    def minCost(self, nums: List[int]) -> int:
        n = len(nums)
        INF = 10**18
        dp = [[INF] * (n+1) for _ in range(n+1)]
        # i from n down to 0
        for i in range(n, -1, -1):
            for j_idx in range(0, n+1):
                if j_idx == 0:
                    # no carry
                    length = n - i
                    if length == 0:
                        dp[i][0] = 0
                    elif length == 1:
                        dp[i][0] = nums[i]
                    elif length == 2:
                        dp[i][0] = max(nums[i], nums[i+1])
                    else: # length >=3
                        # three choices
                        # keep nums[i+2]
                        cost1 = max(nums[i], nums[i+1]) + dp[i+3][i+3]  # j_idx = i+2+1 = i+3
                        # keep nums[i+1]
                        cost2 = max(nums[i], nums[i+2]) + dp[i+3][i+2]  # j_idx = i+1+1 = i+2
                        # keep nums[i]
                        cost3 = max(nums[i+1], nums[i+2]) + dp[i+3][i+1]  # j_idx = i+1
                        dp[i][0] = min(cost1, cost2, cost3)
                else:
                    j = j_idx - 1
                    if j >= i:  # invalid state, skip
                        continue
                    # carry exists
                    if i == n:
                        dp[i][j_idx] = nums[j]
                    elif i == n-1:
                        dp[i][j_idx] = max(nums[j], nums[i])
                    else: # i <= n-2
                        # three choices
                        # A: remove carry and nums[i], keep nums[i+1]
                        costA = max(nums[j], nums[i]) + dp[i+2][i+2]  # new j = i+1, j_idx = i+2
                        # B: remove carry and nums[i+1], keep nums[i]
                        costB = max(nums[j], nums[i+1]) + dp[i+2][i+1]  # new j = i, j_idx = i+1
                        # C: remove nums[i] and nums[i+1], keep carry
                        costC = max(nums[i], nums[i+1]) + dp[i+2][j_idx]  # new j = j, j_idx same
                        dp[i][j_idx] = min(costA, costB, costC)
        return dp[0][0]

class Solution:
    def minCost(self, nums: List[int]) -> int:
        n = len(nums)
        INF = 10**15
        dp = [[INF] * (n+1) for _ in range(n+1)]
        # iterate i from n down to 0
        for i in range(n, -1, -1):
            for j_idx in range(0, n+1):
                if j_idx == 0:
                    # no carry
                    length = n - i
                    if length == 0:
                        dp[i][0] = 0
                    elif length == 1:
                        dp[i][0] = nums[i]
                    elif length == 2:
                        dp[i][0] = max(nums[i], nums[i+1])
                    else:  # length >=3
                        # three choices
                        # keep nums[i+2]
                        cost1 = max(nums[i], nums[i+1]) + dp[i+3][i+3]
                        # keep nums[i+1]
                        cost2 = max(nums[i], nums[i+2]) + dp[i+3][i+2]
                        # keep nums[i]
                        cost3 = max(nums[i+1], nums[i+2]) + dp[i+3][i+1]
                        dp[i][0] = min(cost1, cost2, cost3)
                else:
                    j = j_idx - 1
                    if j >= i:
                        continue
                    # carry exists
                    if i == n:
                        dp[i][j_idx] = nums[j]
                    elif i == n-1:
                        dp[i][j_idx] = max(nums[j], nums[i])
                    else:  # i <= n-2
                        # three choices
                        # A: remove carry and nums[i], keep nums[i+1]
                        costA = max(nums[j], nums[i]) + dp[i+2][i+2]
                        # B: remove carry and nums[i+1], keep nums[i]
                        costB = max(nums[j], nums[i+1]) + dp[i+2][i+1]
                        # C: remove nums[i] and nums[i+1], keep carry
                        costC = max(nums[i], nums[i+1]) + dp[i+2][j_idx]
                        dp[i][j_idx] = min(costA, costB, costC)
        return dp[0][0]

class Solution:
    def minCost(self, nums: List[int]) -> int:
        n = len(nums)
        INF = 10**15
        dp = [[INF] * (n+1) for _ in range(n+1)]
        for i in range(n, -1, -1):
            for j_idx in range(0, n+1):
                if j_idx == 0:
                    length = n - i
                    if length == 0:
                        dp[i][0] = 0
                    elif length == 1:
                        dp[i][0] = nums[i]
                    elif length == 2:
                        dp[i][0] = max(nums[i], nums[i+1])
                    else:
                        # length >=3
                        # keep nums[i+2]
                        cost1 = max(nums[i], nums[i+1]) + dp[i+3][i+3]
                        # keep nums[i+1]
                        cost2 = max(nums[i], nums[i+2]) + dp[i+3][i+2]
                        # keep nums[i]
                        cost3 = max(nums[i+1], nums[i+2]) + dp[i+3][i+1]
                        dp[i][0] = min(cost1, cost2, cost3)
                else:
                    j = j_idx - 1
                    if j >= i:
                        continue
                    if i == n:
                        dp[i][j_idx] = nums[j]
                    elif i == n-1:
                        dp[i][j_idx] = max(nums[j], nums[i])
                    else:
                        # i <= n-2
                        # A: remove carry and nums[i], keep nums[i+1]
                        costA = max(nums[j], nums[i]) + dp[i+2][i+2]
                        # B: remove carry and nums[i+1], keep nums[i]
                        costB = max(nums[j], nums[i+1]) + dp[i+2][i+1]
                        # C: remove nums[i] and nums[i+1], keep carry
                        costC = max(nums[i], nums[i+1]) + dp[i+2][j_idx]
                        dp[i][j_idx] = min(costA, costB, costC)
        return dp[0][0]

from typing import List

class Solution:
    def minCost(self, nums: List[int]) -> int:
        n = len(nums)
        INF = 10**15
        # dp[i][j_idx] : min cost for state (i, carry)
        # j_idx = 0  -> no carry
        # j_idx = j+1 -> carry = nums[j], with j < i
        dp = [[INF] * (n + 1) for _ in range(n + 1)]

        # fill table from i = n down to 0
        for i in range(n, -1, -1):
            for j_idx in range(0, n + 1):
                if j_idx == 0:          # no carry
                    length = n - i
                    if length == 0:
                        dp[i][0] = 0
                    elif length == 1:
                        dp[i][0] = nums[i]
                    elif length == 2:
                        dp[i][0] = max(nums[i], nums[i + 1])
                    else:   # length >= 3
                        # three choices: keep nums[i+2], nums[i+1], nums[i]
                        cost1 = max(nums[i], nums[i + 1]) + dp[i + 3][i + 3]   # keep i+2
                        cost2 = max(nums[i], nums[i + 2]) + dp[i + 3][i + 2]   # keep i+1
                        cost3 = max(nums[i + 1], nums[i + 2]) + dp[i + 3][i + 1] # keep i
                        dp[i][0] = min(cost1, cost2, cost3)
                else:                   # carry exists
                    j = j_idx - 1
                    if j >= i:          # invalid state (carry index must be < i)
                        continue
                    if i == n:          # only carry remains
                        dp[i][j_idx] = nums[j]
                    elif i == n - 1:    # carry + one element
                        dp[i][j_idx] = max(nums[j], nums[i])
                    else:               # i <= n-2, at least two suffix elements
                        # three choices
                        # A: remove carry and nums[i], keep nums[i+1]
                        costA = max(nums[j], nums[i]) + dp[i + 2][i + 2]
                        # B: remove carry and nums[i+1], keep nums[i]
                        costB = max(nums[j], nums[i + 1]) + dp[i + 2][i + 1]
                        # C: remove nums[i] and nums[i+1], keep carry
                        costC = max(nums[i], nums[i + 1]) + dp[i + 2][j_idx]
                        dp[i][j_idx] = min(costA, costB, costC)

        return dp[0][0]   # start with no carry, i = 0
