```python
from typing import List

class Solution:
    def findMaximumScore(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return 0
        nxt = [-1] * n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] < nums[i]:
                idx = stack.pop()
                nxt[idx] = i
            stack.append(i)
        total = 0
        i = 0
        while i < n - 1:
            if nxt[i] != -1:
                total += (nxt[i] - i) * nums[i]
                i = nxt[i]
            else:
                total += (n - 1 - i) * nums[i]
                break
        return total
```
