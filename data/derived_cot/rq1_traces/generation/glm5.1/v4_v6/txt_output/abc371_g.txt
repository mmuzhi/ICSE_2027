You are given permutations P = (P_1, P_2, \ldots, P_N) and A = (A_1, A_2, \ldots, A_N) of (1,2,\ldots,N).
You can perform the following operation any number of times, possibly zero:

- replace A_i with A_{P_i} simultaneously for all i=1,2,\ldots,N.

Print the lexicographically smallest A that can be obtained.
What is lexicographical order?
 For sequences of length N, A = (A_1, A_2, \ldots, A_N) and B = (B_1, B_2, \ldots, B_N), A is lexicographically smaller than B if and only if:

- there exists an integer i\ (1\leq i\leq N) such that A_i < B_i, and A_j = B_j for all 1\leq j < i.

Input

The input is given from Standard Input in the following format:
N
P_1 P_2 \ldots P_N
A_1 A_2 \ldots A_N

Output

Let (A_1, A_2, \ldots, A_N) be the lexicographically smallest A that can be obtained. Print A_1, A_2, \ldots, A_N in this order, separated by spaces, in one line.

Constraints


- 1\leq N\leq2\times10^5
- 1\leq P_i\leq N\ (1\leq i\leq N)
- P_i\neq P_j\ (1\leq i<j\leq N)
- 1\leq A_i\leq N\ (1\leq i\leq N)
- A_i\neq A_j\ (1\leq i<j\leq N)
- All input values are integers.

def crt(congruences):
    # congruences: list of (a, m)
    s = 0
    M = 1
    for a, m in congruences:
        # we need to solve s + M * t ≡ a (mod m) => M * t ≡ a - s (mod m)
        diff = (a - s) % m
        # find inv of M mod m
        inv = pow(M, -1, m)  # python 3.8+ supports pow with negative exponent? Actually pow(M, -1, m) works in Python 3.8+.
        # Alternatively, use extended gcd.
        t = (diff * inv) % m
        s = s + M * t
        M = M * m
    return s % M, M

Then for any index in the cycle, its new value is values[(pos + s_min) % L], where pos is its position in the cycle array (0-indexed, starting from min_idx). This is correct.

Now, what if we process a cycle that has a prime p with exponent e, and the current max_exp for p is f, with e > f? Then we combine the constraint as k ≡ r (mod p^f). Then we choose s_min. Now, let's write the CRT function:
def crt(congruences):
    # congruences: list of (a, m)
    s = 0
    M = 1
    for a, m in congruences:
        # M * t ≡ a - s (mod m)
        diff = (a - s) % m
        # compute inverse of M mod m
        # using extended gcd
        inv = modinv(M, m)
        t = (diff * inv) % m
        s = s + M * t
        M = M * m
    return s % M, M

def modinv(a, m):
    # return x such that a*x ≡ 1 mod m
    # assume gcd(a, m) == 1
    # extended Euclidean
    g, x, y = egcd(a, m)
    return x % m

def egcd(a, b):
    if b == 0:
        return a, 1, 0
    g, x1, y1 = egcd(b, a % b)
    return g, y1, x1 - (a // b) * y1

We can also use pow(a, -1, m) if available. I'll use pow(a, -1, m) for simplicity, but fallback to extended gcd if not. Since Python 3.8 is standard now, pow with -1 is fine.

import sys

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    P = [0] + [int(next(it)) for _ in range(N)]
    A = [0] + [int(next(it)) for _ in range(N)]
    
    # SPF
    spf = list(range(N + 1))
    for i in range(2, int(N**0.5) + 1):
        if spf[i] == i:
            step = i
            for j in range(i*i, N+1, step):
                if spf[j] == j:
                    spf[j] = i
                    
    def factorize(n):
        factors = []
        while n > 1:
            p = spf[n]
            e = 0
            while n % p == 0:
                n //= p
                e += 1
            factors.append((p, e))
        return factors
    
    def modinv(a, m):
        # extended gcd
        m0, x0, x1 = m, 0, 1
        if m == 1:
            return 0
        while a > 1:
            q = a // m
            a, m = m, a % m
            x0, x1 = x1 - q * x0, x0
        if x1 < 0:
            x1 += m0
        return x1
    
    def crt(congruences):
        s = 0
        M = 1
        for a, m in congruences:
            diff = (a - s) % m
            inv = modinv(M % m, m)
            t = (diff * inv) % m
            s = s + M * t
            M = M * m
        return s % M, M
    
    # Find cycles
    visited = [False] * (N + 1)
    cycles = []
    for i in range(1, N + 1):
        if not visited[i]:
            curr = i
            cycle_indices = []
            while not visited[curr]:
                visited[curr] = True
                cycle_indices.append(curr)
                curr = P[curr]
            min_idx = min(cycle_indices)
            min_pos = cycle_indices.index(min_idx)
            indices = cycle_indices[min_pos:] + cycle_indices[:min_pos]
            values = [A[idx] for idx in indices]
            cycles.append((min_idx, len(indices), indices, values))
            
    cycles.sort(key=lambda x: x[0])
    
    max_exp = [0] * (N + 1)
    residue = [0] * (N + 1)
    
    cycle_shifts = []
    
    for min_idx, L, indices, values in cycles:
        factors = factorize(L)
        congruences = []
        for p, e in factors:
            f = max_exp[p]
            if f > 0:
                m = pow(p, min(e, f))
                a = residue[p] % m
                congruences.append((a, m))
        if congruences:
            s_0, G = crt(congruences)
        else:
            s_0, G = 0, 1
            
        best_val = N + 1
        best_s = -1
        steps = L // G
        for t in range(steps):
            s = (s_0 + t * G) % L
            val = values[s]
            if val < best_val:
                best_val = val
                best_s = s
        s_min = best_s
        
        for p, e in factors:
            f = max_exp[p]
            if f == 0:
                max_exp[p] = e
                residue[p] = s_min % pow(p, e)
            elif e >= f:
                max_exp[p] = e
                residue[p] = s_min % pow(p, e)
            else:
                # e < f, consistency check optional
                pass
                
        cycle_shifts.append((indices, values, s_min))
        
    ans = [0] * (N + 1)
    for indices, values, s_min in cycle_shifts:
        L = len(indices)
        for j in range(L):
            ans[indices[j]] = values[(j + s_min) % L]
            
    print(' '.join(map(str, ans[1:])))
    
if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    P = [0] * (N + 1)
    for i in range(1, N + 1):
        P[i] = int(next(it))
    A = [0] * (N + 1)
    for i in range(1, N + 1):
        A[i] = int(next(it))

    # Smallest prime factor sieve
    spf = list(range(N + 1))
    for i in range(2, int(N ** 0.5) + 1):
        if spf[i] == i:
            for j in range(i * i, N + 1, i):
                if spf[j] == j:
                    spf[j] = i

    def factorize(n: int):
        """Return list of (prime, exponent, prime**exponent) for n."""
        factors = []
        while n > 1:
            p = spf[n]
            e = 0
            pe = 1
            while n % p == 0:
                n //= p
                e += 1
                pe *= p
            factors.append((p, e, pe))
        return factors

    def modinv(a: int, m: int) -> int:
        """Return x such that a*x ≡ 1 (mod m). Assumes gcd(a, m) == 1."""
        m0 = m
        y = 0
        x = 1
        if m == 1:
            return 0
        while a > 1:
            q = a // m
            t = m
            m = a % m
            a = t
            t = y
            y = x - q * y
            x = t
        if x < 0:
            x += m0
        return x

    def crt(congruences):
        """Chinese Remainder Theorem. cong: list of (a, m). Moduli are pairwise coprime."""
        s = 0
        M = 1
        for a, m in congruences:
            diff = (a - s) % m
            inv = modinv(M % m, m)
            t = (diff * inv) % m
            s = s + M * t
            M = M * m
        return s % M, M

    # Decompose P into cycles
    visited = [False] * (N + 1)
    cycles = []          # (min_index, length, indices_list, values_list)
    for i in range(1, N + 1):
        if not visited[i]:
            curr = i
            cycle_indices = []
            while not visited[curr]:
                visited[curr] = True
                cycle_indices.append(curr)
                curr = P[curr]
            min_idx = min(cycle_indices)
            min_pos = cycle_indices.index(min_idx)
            # Rotate to start at min_idx
            indices = cycle_indices[min_pos:] + cycle_indices[:min_pos]
            values = [A[idx] for idx in indices]
            cycles.append((min_idx, len(indices), indices, values))

    # Process cycles in the order their first index appears
    cycles.sort(key=lambda x: x[0])

    max_exp = [0] * (N + 1)    # max exponent of prime p seen so far
    residue = [0] * (N + 1)    # residue modulo p^max_exp[p]

    cycle_shifts = []          # (indices, values, chosen_shift)

    for min_idx, L, indices, values in cycles:
        factors = factorize(L)
        congruences = []
        for p, e, pe in factors:
            f = max_exp[p]
            if f > 0:
                m = pow(p, min(e, f))
                a = residue[p] % m
                congruences.append((a, m))

        if congruences:
            s_0, G = crt(congruences)
        else:
            s_0, G = 0, 1

        # Find the allowed shift that minimizes the value at the first index
        best_val = N + 1
        best_s = -1
        steps = L // G
        for t in range(steps):
            s = (s_0 + t * G) % L
            val = values[s]
            if val < best_val:
                best_val = val
                best_s = s
        s_min = best_s

        # Record the constraints imposed by this cycle
        for p, e, pe in factors:
            f = max_exp[p]
            if f == 0:
                max_exp[p] = e
                residue[p] = s_min % pe
            elif e >= f:
                max_exp[p] = e
                residue[p] = s_min % pe
            # if e < f, the new constraint is weaker; consistency is guaranteed

        cycle_shifts.append((indices, values, s_min))

    # Build the final array
    ans = [0] * (N + 1)
    for indices, values, s_min in cycle_shifts:
        L = len(indices)
        for j in range(L):
            ans[indices[j]] = values[(j + s_min) % L]

    print(' '.join(map(str, ans[1:])))

if __name__ == "__main__":
    solve()
