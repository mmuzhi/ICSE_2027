```python
class Solution:
    def longestCommonPrefix(self, words: List[str], k: int) -> List[int]:
        n = len(words)
        if n == k:
            return [0] * n
        
        # Build trie
        trie = [{}]  # list of dicts
        count = [0]
        depth = [0]
        end_node = [0] * n
        
        for i, w in enumerate(words):
            node = 0
            for ch in w:
                count[node] += 1
                if ch not in trie[node]:
                    trie[node][ch] = len(trie)
                    trie.append({})
                    count.append(0)
                    depth.append(depth[node] + 1)
                node = trie[node][ch]
            count[node] += 1
            end_node[i] = node
        
        N = len(trie)
        max_eq_k = [0] * N
        best1_val = [0] * N
        best1_char = [''] * N
        best2_val = [0] * N
        best2_char = [''] * N
        
        # Pre-order traversal
        stack = [0]
        pre_order = []
        while stack:
            u = stack.pop()
            pre_order.append(u)
            for ch, v in trie[u].items():
                stack.append(v)
        
        # Post-order processing
        for u in reversed(pre_order):
            mx = 0
            if count[u] == k:
                mx = depth[u]
            b1 = -1
            b1c = ''
            b2 = -1
            b2c = ''
            for ch, v in trie[u].items():
                val = max_eq_k[v]
                if val > b1:
                    b2 = b1
                    b2c = b1c
                    b1 = val
                    b1c = ch
                elif val > b2:
                    b2 = val
                    b2c = ch
            # If no children, b1 remains -1, but we want max to be at least 0.
            max_eq_k[u] = max(mx, b1 if b1 != -1 else 0)
            best1_val[u] = b1 if b1 != -1 else 0
            best1_char[u] = b1c
            best2_val[u] = b2 if b2 != -1 else 0
            best2_char[u] = b2c
        
        global_max_ge_k1 = 0
        for u in range(N):
            if count[u] >= k + 1:
                if depth[u] > global_max_ge_k1:
                    global_max_ge_k1 = depth[u]
        
        ans = [0] * n
        for i, w in enumerate(words):
            node = 0
            max_off = 0
            for ch in w:
                if best1_char[node] != ch:
                    max_off = max(max_off, best1_val[node])
                else:
                    max_off = max(max_off, best2_val[node])
                node = trie[node][ch]
            # After loop, node is terminal node
            max_off = max(max_off, best1_val[node])
            ans[i] = max(global_max_ge_k1, max_off)
        
        return ans
```

We need to import List from typing.

```python
from typing import List

class Solution:
    def longestCommonPrefix(self, words: List[str], k: int) -> List[int]:
        n = len(words)
        if n == k:
            return [0] * n
        
        L = sum(len(w) for w in words)
        N_max = L + 1
        trie = [-1] * (N_max * 26)
        count = [0] * N_max
        depth = [0] * N_max
        node_count = 1  # root is 0
        end_node = [0] * n
        
        for i, w in enumerate(words):
            node = 0
            for ch in w:
                count[node] += 1
                c = ord(ch) - 97
                idx = node * 26 + c
                if trie[idx] == -1:
                    trie[idx] = node_count
                    depth[node_count] = depth[node] + 1
                    node_count += 1
                node = trie[idx]
            count[node] += 1
            end_node[i] = node
        
        N = node_count
        max_eq_k = [0] * N
        best1_val = [0] * N
        best1_char = [-1] * N
        best2_val = [0] * N
        best2_char = [-1] * N
        
        # Pre-order traversal
        stack = [0]
        pre_order = []
        while stack:
            u = stack.pop()
            pre_order.append(u)
            base = u * 26
            for c in range(26):
                v = trie[base + c]
                if v != -1:
                    stack.append(v)
        
        # Post-order processing
        for u in reversed(pre_order):
            mx = depth[u] if count[u] == k else 0
            b1 = -1
            b1c = -1
            b2 = -1
```

```python
from typing import List

class Solution:
    def longestCommonPrefix(self, words: List[str], k: int) -> List[int]:
        n = len(words)
        # If removing one string leaves fewer than k strings, all answers are 0.
        if n == k:
            return [0] * n

        L = sum(len(w) for w in words)
        N_max = L + 1  # maximum number of trie nodes

        # Flat array for trie children: node * 26 + c -> next_node (-1 if none)
        trie = [-1] * (N_max * 26)
        count = [0] * N_max   # number of strings passing through this node
        depth = [0] * N_max   # length of the prefix ending at this node
        node_count = 1        # root is node 0

        # Build the trie
        for w in words:
            node = 0
            for ch in w:
                count[node] += 1
                c = ord(ch) - 97
                idx = node * 26 + c
                if trie[idx] == -1:
                    trie[idx] = node_count
                    depth[node_count] = depth[node] + 1
                    node_count += 1
                node = trie[idx]
            count[node] += 1

        N = node_count  # actual number of nodes used

        # Data structures for subtree max_eq_k and top two children's values
        max_eq_k = [0] * N
        best1_val = [0] * N
        best1_char = [-1] * N
        best2_val = [0] * N
        best2_char = [-1] * N

        # Iterative pre-order traversal
        stack = [0]
        pre_order = []
        while stack:
            u = stack.pop()
            pre_order.append(u)
            base = u * 26
            for c in range(26):
                v = trie[base + c]
                if v != -1:
                    stack.append(v)

        # Post-order processing (reverse pre-order gives a valid topological order)
        for u in reversed(pre_order):
            mx = depth[u] if count[u] == k else 0
            b1 = -1
            b1c = -1
            b2 = -1
            b2c = -1
            base = u * 26
            for c in range(26):
                v = trie[base + c]
                if v == -1:
                    continue
                val = max_eq_k[v]
                if val > b1:
                    b2 = b1
                    b2c = b1c
                    b1 = val
                    b1c = c
                elif val > b2:
                    b2 = val
                    b2c = c
            max_eq_k[u] = max(mx, b1 if b1 != -1 else 0)
            best1_val[u] = b1 if b1 != -1 else 0
            best1_char[u] = b1c
            best2_val[u] = b2 if b2 != -1 else 0
            best2_char[u] = b2c

        # Global maximum depth among nodes with count >= k+1
        global_max_ge_k1 = 0
        for u in range(N):
            if count[u] >= k + 1:
                if depth[u] > global_max_ge_k1:
                    global_max_ge_k1 = depth[u]

        ans = [0] * n
        for i, w in enumerate(words):
            node = 0
            max_off = 0
            for ch in w:
                c = ord(ch) - 97
                # The path goes through child c, so we take the best among other children
                if best1_char[node] != c:
                    max_off = max(max_off, best1_val[node])
                else:
                    max_off = max(max_off, best2_val[node])
                node = trie[node * 26 + c]
            # Also consider children of the terminal node (they are off the path)
            max_off = max(max_off, best1_val[node])
            ans[i] = max(global_max_ge_k1, max_off)

        return ans
```
