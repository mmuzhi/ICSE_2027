import sys

MOD = 998244353

def solve() -> None:
    input = sys.stdin.readline
    N, Q = map(int, input().split())
    P = [0] * Q
    V = [0] * Q
    for i in range(Q):
        p, v = map(int, input().split())
        P[i] = p
        V[i] = v

    forced = [-1] * Q   # -1: free, 0: prefix, 1: suffix

    for i in range(Q):
        Vi = V[i]
        Pi = P[i]
        for j in range(i + 1, Q):
            if Vi > V[j]:
                if Pi == P[j]:
                    print(0)
                    return
                if Pi > P[j]:
                    # need (suffix, prefix) -> i=1, j=0
                    if forced[i] == -1:
                        forced[i] = 1
                    elif forced[i] != 1:
                        print(0)
                        return
                    if forced[j] == -1:
                        forced[j] = 0
                    elif forced[j] != 0:
                        print(0)
                        return
                else:  # Pi < P[j]
                    # need (prefix, suffix) -> i=0, j=1
                    if forced[i] == -1:
                        forced[i] = 0
                    elif forced[i] != 0:
                        print(0)
                        return
                    if forced[j] == -1:
                        forced[j] = 1
                    elif forced[j] != 1:
                        print(0)
                        return

    free = forced.count(-1)
    ans = pow(2, free, MOD)
    print(ans)
