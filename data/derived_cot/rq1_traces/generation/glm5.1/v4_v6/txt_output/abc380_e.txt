There are N cells in a row, numbered 1 to N.
For each 1 \leq i < N, cells i and i+1 are adjacent.
Initially, cell i is painted with color i.
You are given Q queries. Process them in order. Each query is of one of the following two types.

- 1 x c: Repaint the following to color c: all reachable cells reachable from cell x by repeatedly moving to an adjacent cell painted in the same color as the current cell.
- 2 c: Print the number of cells painted with color c.

Input

The input is given from Standard Input in the following format:
N Q
\mathrm{query}_1
\vdots
\mathrm{query}_Q

1 x c

The i-th line should contain the answer to the i-th such query.

Constraints


- 1 \leq N \leq 5 \times 10^5
- 1 \leq Q \leq 2 \times 10^5
- In queries of the first type, 1 \leq x \leq N.
- In queries of the first and second types, 1 \leq c \leq N.
- There is at least one query of the second type.
- All input values are integers.

            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

        typ = int(next(it))

            x = int(next(it))
            c = int(next(it))

            sz = size[root]
            cnt[old_c] -= sz
            cnt[c] += sz

                    parent[rootR] = root
                    size[root] += size[rootR]
                    right[root] = right[rootR]

def solve():
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    Q = int(next(it))
    
    parent = list(range(N + 1))
    left = list(range(N + 1))
    right = list(range(N + 1))
    color = list(range(N + 1))
    size = [1] * (N + 1)
    cnt = [0] * (N + 1)
    for i in range(1, N + 1):
        cnt[i] = 1
    
    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    
    out = []
    for _ in range(Q):
        typ = int(next(it))
        if typ == 1:

            c = int(next(it))

                continue

            sz = size[root]
            cnt[old_c] -= sz
            cnt[c] += sz
            color[root] = c

            if L >= 1:

                if color[rootL] == c:
                    # Merge rootL and root
                    if size[rootL] < size[root]:
                        parent[rootL] = root
                        left[root] = left[rootL]
                        size[root] += size[rootL]
                    else:
                        parent[root] = rootL
                        right[rootL] = right[root]
                        size[rootL] += size[root]
                        root = rootL
            
            # Check right

            if R <= N:

                        parent[rootR] = root

                    else:

                        size[rootR] += size[root]
                        root = rootR

import sys

def solve():
    import sys
    input = sys.stdin.buffer.readline
    N, Q = map(int, input().split())
    
    parent = list(range(N + 1))
    left = list(range(N + 1))
    right = list(range(N + 1))
    color = list(range(N + 1))
    size = [1] * (N + 1)
    cnt = [0] * (N + 1)
    for i in range(1, N + 1):
        cnt[i] = 1
    
    def find(x):
        r = x
        while parent[r] != r:
            r = parent[r]
        while x != r:
            nxt = parent[x]
            parent[x] = r
            x = nxt
        return r
    
    out = []
    for _ in range(Q):
        parts = input().split()
        if not parts:
            continue
        typ = int(parts[0])
        if typ == 1:
            x = int(parts[1])
            c = int(parts[2])
            root = find(x)
            if color[root] == c:
                continue
            old_c = color[root]
            sz = size[root]
            cnt[old_c] -= sz
            cnt[c] += sz
            color[root] = c
            
            L = left[root] - 1
            if L >= 1:
                rootL = find(L)
                if color[rootL] == c:
                    if size[rootL] < size[root]:
                        parent[rootL] = root
                        left[root] = left[rootL]
                        size[root] += size[rootL]
                    else:
                        parent[root] = rootL
                        right[rootL] = right[root]
                        size[rootL] += size[root]
                        root = rootL
            
            R = right[root] + 1
            if R <= N:
                rootR = find(R)
                if color[rootR] == c:
                    if size[rootR] < size[root]:
                        parent[rootR] = root
                        right[root] = right[rootR]
                        size[root] += size[rootR]
                    else:
                        parent[root] = rootR
                        left[rootR] = left[root]
                        size[rootR] += size[root]
                        root = rootR
        else:
            c = int(parts[1])
            out.append(str(cnt[c]))
    sys.stdout.write("\n".join(out))
