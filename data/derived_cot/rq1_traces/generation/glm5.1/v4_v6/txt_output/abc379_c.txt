import sys

def main():
    import sys
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X = [int(next(it)) for _ in range(M)]
    A = [int(next(it)) for _ in range(M)]
    total = sum(A)
    if total != N:
        print(-1)
        return
    pairs = list(zip(X, A))
    pairs.sort()
    if pairs[0][0] != 1:
        print(-1)
        return
    cum = 0
    for i in range(M-1):
        cum += pairs[i][1]
        if cum < pairs[i+1][0] - 1:
            print(-1)
            return
    # compute sum of positions
    sum_pos = 0
    for x, a in pairs:
        sum_pos += x * a
    ans = N * (N + 1) // 2 - sum_pos
    print(ans)

if __name__ == "__main__":
    main()

import sys

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X = [int(next(it)) for _ in range(M)]
    A = [int(next(it)) for _ in range(M)]

    total = sum(A)
    if total != N:
        print(-1)
        return

    pairs = list(zip(X, A))
    pairs.sort()                     # sort by position

    if pairs[0][0] != 1:             # cell 1 must contain a stone
        print(-1)
        return

    cum = 0
    for i in range(M - 1):
        cum += pairs[i][1]
        if cum < pairs[i+1][0] - 1:  # check prefix condition
            print(-1)
            return

    # compute sum of initial positions
    sum_pos = 0
    for x, a in pairs:
        sum_pos += x * a

    ans = N * (N + 1) // 2 - sum_pos
    print(ans)

if __name__ == "__main__":
    main()
