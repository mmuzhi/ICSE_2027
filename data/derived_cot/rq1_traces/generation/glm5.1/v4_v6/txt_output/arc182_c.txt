MOD = 998244353

def mat_mul(A, B):

            if aik:
                Bk = B[k]
                for j in range(n):
                    Ci[j] = (Ci[j] + aik * Bk[j]) % MOD
    return C

        total = 0
        Ai = A[i]
        for j in range(n):
            total = (total + Ai[j] * v[j]) % MOD

            for j in range(n):
                res[j] = (res[j] + li * Ai[j]) % MOD
    return res

    for i in range(len(L)):
        total = (total + L[i] * v[i]) % MOD
    return total

        return

import sys

MOD = 998244353

# ---------- matrix utilities ----------
def mat_mul(A, B):
    """return A * B  (square matrices)"""
    n = len(A)
    C = [[0] * n for _ in range(n)]
    for i in range(n):
        Ai = A[i]
        Ci = C[i]
        for k in range(n):
            aik = Ai[k]
            if aik:
                Bk = B[k]
                for j in range(n):
                    Ci[j] = (Ci[j] + aik * Bk[j]) % MOD
    return C

def mat_vec_mul(A, v):
    """return A * v  (column vector)"""
    n = len(A)
    res = [0] * n
    for i in range(n):
        total = 0
        Ai = A[i]
        for j in range(n):
            total = (total + Ai[j] * v[j]) % MOD
        res[i] = total
    return res

def row_mul_mat(L, A):
    """return row vector L * A"""
    n = len(L)
    res = [0] * n
    for i in range(n):
        li = L[i]
        if li:
            Ai = A[i]
            for j in range(n):
                res[j] = (res[j] + li * Ai[j]) % MOD
    return res

def dot(L, v):
    """dot product of row vector L and column vector v"""
    total = 0
    for i in range(len(L)):
        total = (total + L[i] * v[i]) % MOD
    return total

# ---------- main ----------
def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])

    # primes not larger than M (M ≤ 16)
    all_primes = [2, 3, 5, 7, 11, 13]
    primes = [p for p in all_primes if p <= M]
    P = len(primes)
    SZ = 1 << P                     # number of subsets

    # exponent vectors for numbers 1 .. M
    exp_list = []
    for a in range(1, M + 1):
        exps = [0] * P
        x = a
        for i, p in enumerate(primes):
            while x % p == 0:
                exps[i] += 1
                x //= p
        exp_list.append(exps)

    # A[mask] = sum_{a=1..M} prod_{p in mask} exp_a(p)
    A_mask = [0] * SZ
    for exps in exp_list:
        for mask in range(SZ):
            prod = 1
            for i in range(P):
                if mask >> i & 1:
                    prod *= exps[i]
            A_mask[mask] = (A_mask[mask] + prod) % MOD

    # build matrix A (SZ x SZ) : A[T][U] = A_mask[T \ U] if U ⊆ T else 0
    A_mat = [[0] * SZ for _ in range(SZ)]
    for U in range(SZ):
        for T in range(SZ):
            if (U & ~T) == 0:          # U ⊆ T
                diff = T ^ U            # = T \ U
                A_mat[T][U] = A_mask[diff]

    # R = 1ᵀ A  (row vector, column sums of A)
    R = [0] * SZ
    for j in range(SZ):
        s = 0
        for i in range(SZ):
            s = (s + A_mat[i][j]) % MOD
        R[j] = s

    # initial vector: f₀ has 1 at empty set, s₀ = 0
    f = [0] * SZ
    f[0] = 1
    s = 0

    base_A = A_mat
    base_L = R
    exp = N

    while exp:
        if exp & 1:
            # apply current power to the vector
            new_f = mat_vec_mul(base_A, f)
            new_s = (dot(base_L, f) + s) % MOD
            f, s = new_f, new_s
        # square the base matrix

        new_L = row_mul_mat(base_L, base_A)
        for i in range(SZ):
            new_L[i] = (new_L[i] + base_L[i]) % MOD
        base_A, base_L = new_A, new_L
        exp >>= 1
