def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))
    
    # If all are 1, we can just output Yes immediately.
    if all(x == 1 for x in A):
        print("Yes")
        return
    
    forbidden = (N % 4 == 2)
    
    for start in (0, 1):
        # dp[prev][dev] = True/False
        dp = [[False, False] for _ in range(2)]
        dp[start][0] = True
        
        for i in range(1, N):
            new_dp = [[False, False] for _ in range(2)]
            for prev in (0, 1):
                for dev in (0, 1):
                    if not dp[prev][dev]:
                        continue
                    for curr in (0, 1):
                        if prev == 1 and curr == 1:
                            continue
                        if A[i-1] == 0 and prev == 0 and curr == 0:
                            continue
                        new_dev = dev
                        if forbidden:
                            if curr != ((start + i) % 2):
                                new_dev = 1
                        new_dp[curr][new_dev] = True
            dp = new_dp
            # early break if no states
            if not any(dp[prev][dev] for prev in (0,1) for dev in (0,1)):
                break
        else:
            # check wrap-around
            for prev in (0, 1):
                for dev in (0, 1):
                    if not dp[prev][dev]:
                        continue
                    if prev == 1 and start == 1:
                        continue
                    if A[N-1] == 0 and prev == 0 and start == 0:
                        continue
                    if forbidden and dev == 0:
                        continue
                    print("Yes")
                    return
    print("No")

if __name__ == "__main__":

import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = [int(x) for x in data[1:1+N]]

    # If there are no zeros, any string works.
    if all(x == 1 for x in A):
        print("Yes")
        return

    # Forbidden alternating patterns exist only when N % 4 == 2.
    forbidden = (N % 4 == 2)

    # Try both possible values for the first position R[0].
    for start in (0, 1):
        # dp[prev][deviated] = reachable?
        dp = [[False, False] for _ in range(2)]
        dp[start][0] = True

        # Fill positions 1 .. N-1
        for i in range(1, N):
            new_dp = [[False, False] for _ in range(2)]
            for prev in (0, 1):
                for dev in (0, 1):
                    if not dp[prev][dev]:
                        continue
                    for curr in (0, 1):
                        # No adjacent R's
                        if prev == 1 and curr == 1:
                            continue
                        # Cover condition for zero at i-1
                        if A[i-1] == 0 and prev == 0 and curr == 0:
                            continue
                        new_dev = dev
                        if forbidden:
                            expected = (start + i) % 2
                            if curr != expected:
                                new_dev = 1
                        new_dp[curr][new_dev] = True
            dp = new_dp
            # Early exit if no states are reachable
            if not any(dp[p][d] for p in (0, 1) for d in (0, 1)):
                break
        else:
            # Check wrap-around constraints
            for prev in (0, 1):
                for dev in (0, 1):
                    if not dp[prev][dev]:
                        continue
                    # No adjacent R's across the boundary
                    if prev == 1 and start == 1:
                        continue
                    # Cover condition for zero at N-1
                    if A[N-1] == 0 and prev == 0 and start == 0:
                        continue
                    # Reject the exact forbidden alternating pattern
                    if forbidden and dev == 0:
                        continue
                    print("Yes")
                    return
    print("No")

if __name__ == "__main__":
    solve()
