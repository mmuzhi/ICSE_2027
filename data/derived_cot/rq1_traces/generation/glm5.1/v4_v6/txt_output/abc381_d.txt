    # Now find longest contiguous subarray of pairs with no None and all distinct values.
    # We can do this by iterating with two pointers.
    n = len(pairs)
    left = 0
    window_set = set()
    max_len = 0
    for right in range(n):
        val = pairs[right]
        if val is None:

        else:

```python
def max_pairs(arr):
    # arr is list of ints, with -1 as invalid
    n = len(arr)
    left = 0
    seen = set()
    max_len = 0
    for right in range(n):
        if arr[right] == -1:
            left = right + 1
            seen.clear()
        else:
            while arr[right] in seen:
                seen.remove(arr[left])
                left += 1
            seen.add(arr[right])
            max_len = max(max_len, right - left + 1)
    return max_len
```

Now, let's think about potential pitfalls:
- The sliding window uses a set. When we remove arr[left], we must ensure we only remove it if it was in the set. Since we only add arr[right] after removing duplicates, and we only move left over elements that are in the set, it's safe.
- The while loop condition `while arr[right] in seen:` will remove elements from the left until the duplicate is removed. This is standard.
- We need to reset left and seen when we hit -1.

Time complexity: O(N) because each element is added and removed at most once per parity. Space: O(N) for the lists and set.

We can optimize space by not building the lists, but just iterating with two pointers directly on the original array with steps of 2. But building the list is fine for N=2e5. 2e5 integers is small.

Let's code it in Python.

```python
import sys

def solve():
    input = sys.stdin.readline
    N = int(input().strip())
    A = list(map(int, input().split()))
    
    ans = 0
    for parity in (0, 1):
        pairs = []
        for i in range(parity, N - 1, 2):
            if A[i] == A[i+1]:
                pairs.append(A[i])
            else:
                pairs.append(-1)
        # sliding window
        left = 0
        seen = set()
        max_len = 0
        for right in range(len(pairs)):
            if pairs[right] == -1:
                left = right + 1
                seen.clear()
            else:
                while pairs[right] in seen:
                    seen.remove(pairs[left])
                    left += 1
                seen.add(pairs[right])
                max_len = max(max_len, right - left + 1)
        ans = max(ans, 2 * max_len)
    print(ans)

if __name__ == "__main__":
    solve()
```

```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))

    ans = 0
    # We consider two cases: the subarray starts at an even index or at an odd index.
    for parity in (0, 1):
        # Build the list of "pair" values for this parity.
        # A pair starts at index i if A[i] == A[i+1]; we record its value.
        # If they differ, we record -1 as a barrier.
        pairs = []
        for i in range(parity, N - 1, 2):
            if A[i] == A[i+1]:
                pairs.append(A[i])
            else:
                pairs.append(-1)

        # Sliding window to find the longest contiguous segment without -1
        # and with all distinct values.
        left = 0
        seen = set()
        max_len = 0
        for right in range(len(pairs)):
            if pairs[right] == -1:
                left = right + 1
                seen.clear()
            else:
                while pairs[right] in seen:
                    seen.remove(pairs[left])
                    left += 1
                seen.add(pairs[right])
                max_len = max(max_len, right - left + 1)

        # Each pair contributes 2 to the original length.
        ans = max(ans, 2 * max_len)

    print(ans)


if __name__ == "__main__":
    solve()
```
