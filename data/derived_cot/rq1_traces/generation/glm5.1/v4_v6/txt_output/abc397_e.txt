- P_{1,1}, \dots, P_{1,K}, P_{2,1}, \dots, P_{N,K} is a permutation of 1,2,\dots,NK.

Constraints


- 1 \leq N
- 1 \leq K
- NK \leq 2 \times 10^5
- 1 \leq u_i < v_i \leq NK

- All input values are integers.

import sys

def solve():
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    N = int(data[0])
    K = int(data[1])
    if K == 1:
        print("Yes")
        return
    NK = N * K
    adj = [[] for _ in range(NK + 1)]
    idx = 2
    for _ in range(NK - 1):
        u = int(data[idx]); v = int(data[idx+1]); idx += 2
        adj[u].append(v)
        adj[v].append(u)
    
    parent = [0] * (NK + 1)
    order = []
    stack = [1]
    parent[1] = -1
    while stack:
        u = stack.pop()
        order.append(u)
        for v in adj[u]:
            if v != parent[u]:
                parent[v] = u
                stack.append(v)
    
    V = [0] * (NK + 1)
    for u in reversed(order):
        V[u] = 1
        M = []
        for v in adj[u]:
            if v != parent[u]:
                V[u] += V[v]
                r = V[v] % K
                if r != 0:

        if len(M) > 2:
            print("No")
            return
        if len(M) == 2:
            if M[0] + M[1] + 1 != K:
                print("No")
                return
    print("Yes")
