            if aik:

import sys

def mat_mul(A, B, mod):
    """Multiply matrices A and B modulo mod."""
    N = len(A)
    C = [[0] * N for _ in range(N)]
    for i in range(N):
        Ai = A[i]
        Ci = C[i]
        for k in range(N):
            aik = Ai[k]
            if aik:
                Bk = B[k]
                for j in range(N):
                    Ci[j] += aik * Bk[j]
        for j in range(N):
            Ci[j] %= mod
    return C

def mat_pow(A, power, mod):
    """Fast exponentiation of matrix A to the power `power` modulo mod."""
    N = len(A)
    res = [[0] * N for _ in range(N)]
    for i in range(N):
        res[i][i] = 1
    base = A
    while power > 0:
        if power & 1:
            res = mat_mul(res, base, mod)
        base = mat_mul(base, base, mod)
        power >>= 1
    return res
