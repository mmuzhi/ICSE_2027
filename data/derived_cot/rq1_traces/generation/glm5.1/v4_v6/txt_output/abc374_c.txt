=== ANSWER ===
```python

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    K = list(map(int, input().split()))
    total = sum(K)

    ans = total                     # worst case: all in one group
    size = 1 << N                   # number of subsets
    sums = [0] * size               # sums[mask] = sum of departments in mask

    for mask in range(1, size):
        lsb = mask & -mask          # lowest set bit
        i = lsb.bit_length() - 1   # index of that bit
        sums[mask] = sums[mask ^ lsb] + K[i]
        cur = max(sums[mask], total - sums[mask])
        if cur < ans:
            ans = cur

    print(ans)

if __name__ == "__main__":
    solve()
```
