def solve() -> None:

    it = iter(data)
    t = int(next(it))
    out_lines = []

    for _ in range(t):

            # just take the minimum of A_i * B_i
            ans = min(a * b for a, b in pairs)
        else:
            heap = []          # min‑heap storing -B, so the top is the most negative = largest B

                a, b = pairs[i]

                    # remove the largest B among the current (k) elements
                    largest_neg = heapq.heappop(heap)
