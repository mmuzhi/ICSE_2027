What is bitwise XOR? The bitwise XOR of non-negative integers A and B, denoted as A \oplus B, is defined as follows: - In the binary representation of A \oplus B, the digit at position 2^k (k \geq 0) is 1 if exactly one of A and B has a 1 in that position in their binary representations, and 0 otherwise. import sys
from array import array

MOD = 998244353

def fwht_int(a):
    n = len(a)
    step = 1
    while step < n:
        for i in range(0, n, step*2):
            for j in range(i, i+step):
                u = a[j]
                v = a[j+step]
                a[j] = u + v
                a[j+step] = u - v
        step <<= 1

def fwht_mod(a):
    n = len(a)
    step = 1
    while step < n:
        for i in range(0, n, step*2):
            for j in range(i, i+step):
                u = a[j]
                v = a[j+step]
                a[j] = (u + v) % MOD
                a[j+step] = (u - v) % MOD
        step <<= 1

def main():
    import sys
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    A = list(map(int, input().split()))
    maxA = max(A)
    L = max(1, maxA.bit_length())
    SIZE = 1 << L

    # c array
    c = [0] * SIZE
    for a in A:
        c[a] += 1

    # Walsh-Hadamard transform on c (integer)
    fwht_int(c)   # now c holds H[s]

    # Precompute U and V DP
    total_len = (N+1) * M
    U = array('I', [0]) * total_len
    V = array('I', [0]) * total_len

    # U[0] = 1
    U[0] = 1
    for p in range(N):
        base = p * M
        nbase = base + M
        for r in range(M):
            prev_r = r-1 if r else M-1
            val = U[base + r] + U[base + prev_r]
            if val >= MOD:
                val -= MOD
            U[nbase + r] = val

    # V[0] = 1
    V[0] = 1
    for q in range(N):
        base = q * M
        nbase = base + M
        for r in range(M):
            prev_r = r-1 if r else M-1
            val = V[base + r] - V[base + prev_r]
            if val < 0:
                val += MOD
            V[nbase + r] = val

    # Compute G array
    G = [0] * (N+1)
    for p in range(N+1):
        q = N - p
        baseU = p * M
        baseV = q * M
        total = 0
        for r in range(M):
            rV = (M - r) % M
            total += U[baseU + r] * V[baseV + rV]
        G[p] = total % MOD

    # Compute g array and its transform
    g = array('I', [0]) * SIZE
    for v in range(SIZE):
        g[v] = pow(v, K, MOD)
    fwht_mod(g)   # now g holds S(s)

    # Compute total sum
    total = 0
    for s in range(SIZE):
        p = (N + c[s]) // 2   # c[s] is H[s]
        # p should be within 0..N
        total = (total + G[p] * g[s]) % MOD

    invSIZE = pow(SIZE, MOD-2, MOD)
    ans = total * invSIZE % MOD
    print(ans)

if __name__ == "__main__":
    main()

import sys
from array import array

MOD = 998244353

def fwht_int(a):
    """in-place Walsh–Hadamard transform over integers (no modulo)"""
    n = len(a)
    step = 1
    while step < n:
        for i in range(0, n, step << 1):
            for j in range(i, i + step):
                u = a[j]
                v = a[j + step]
                a[j] = u + v
                a[j + step] = u - v
        step <<= 1

def fwht_mod(a):
    """in-place Walsh–Hadamard transform modulo MOD"""
    n = len(a)
    step = 1
    while step < n:
        for i in range(0, n, step << 1):
            for j in range(i, i + step):
                u = a[j]
                v = a[j + step]
                a[j] = (u + v) % MOD
                a[j + step] = (u - v) % MOD
        step <<= 1

def main():
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    A = list(map(int, input().split()))

    # determine bit length
    maxA = max(A)
    L = max(1, maxA.bit_length())
    SIZE = 1 << L

    # ----- count array c and its transform H -----
    c = [0] * SIZE
    for a in A:
        c[a] += 1
    fwht_int(c)          # now c[s] = H[s]

    # ----- DP for U and V -----
    total_len = (N + 1) * M
    U = array('I', [0]) * total_len
    V = array('I', [0]) * total_len

    # U[0] = 1 (p = 0, residue 0)
    U[0] = 1
    for p in range(N):
        base = p * M
        nbase = base + M
        for r in range(M):
            prev = r - 1 if r else M - 1
            val = U[base + r] + U[base + prev]
            if val >= MOD:
                val -= MOD
            U[nbase + r] = val

    # V[0] = 1
    V[0] = 1
    for q in range(N):
        base = q * M
        nbase = base + M
        for r in range(M):
            prev = r - 1 if r else M - 1
            val = V[base + r] - V[base + prev]
            if val < 0:
                val += MOD
            V[nbase + r] = val

    # ----- compute G[p] for all p -----
    G = [0] * (N + 1)
    for p in range(N + 1):
        q = N - p
        baseU = p * M
        baseV = q * M
        total = 0
        for r in range(M):
            rV = (M - r) % M          # (-r) mod M
            total += U[baseU + r] * V[baseV + rV]
        G[p] = total % MOD

    # ----- array g(v) = v^K and its transform S(s) -----
    g = array('I', [0]) * SIZE
    for v in range(SIZE):
        g[v] = pow(v, K, MOD)
    fwht_mod(g)          # now g[s] = S(s)

    # ----- combine everything -----
    total = 0
    for s in range(SIZE):
        p = (N + c[s]) // 2      # c[s] is H[s]
        total = (total + G[p] * g[s]) % MOD

    invSIZE = pow(SIZE, MOD - 2, MOD)
    ans = total * invSIZE % MOD
    print(ans)

if __name__ == "__main__":
    main()
