```python
import sys

MOD = 998244353

def main():
    it = iter(sys.stdin.buffer.read().split())
    N = int(next(it))
    M = int(next(it))

    # group constraints by X (0-indexed)
    cons_by_X = [[] for _ in range(N)]
    for _ in range(M):
        L = int(next(it)) - 1
        R = int(next(it)) - 1
        X = int(next(it)) - 1
        cons_by_X[X].append((L, R))

    # bit for each position
    bit = [1 << i for i in range(N)]

    # forbid_mask[l][r] (0-indexed, l<=r) : bitset of forbidden roots
    forbid = [[0] * N for _ in range(N)]

    # process each X
    for X in range(N):
        if not cons_by_X[X]:
            continue
        # 2D difference array, 1-indexed for l and r, size (N+2) x (N+2)
        diff = [[0] * (N + 2) for _ in range(N + 2)]
        for L, R in cons_by_X[X]:
            a = 1
            b = L + 1          # L (0-index) -> 1-index
            c = R + 1          # R (0-index) -> 1-index
            d = N
            diff[a][c] += 1
            diff[b + 1][c] -= 1
            diff[a][d + 1] -= 1
            diff[b + 1][d + 1] += 1

        # 2D prefix sum
        for i in range(1, N + 1):
            prev = diff[i - 1]
            cur = diff[i]
            for j in range(1, N + 1):
                cur[j] += prev[j] + cur[j - 1] - prev[j - 1]

        # set bits where count > 0
        bitX = bit[X]
        for l in range(1, N + 1):
            for r in range(l, N + 1):
                if diff[l][r] > 0:
                    forbid[l - 1][r - 1] |= bitX

    # precompute modular inverses of 1..N
    inv = [0] * (N + 1)
    for i in range(1, N + 1):
        inv[i] = pow(i, MOD - 2, MOD)

    # DP over intervals
    dp = [[0] * N for _ in range(N)]
    for length in range(1, N + 1):
        inv_len = inv[length]
        for l in range(0, N - length + 1):
            r = l + length - 1
            total = 0
            mask = forbid[l][r]
            for k in range(l, r + 1):
                if mask & bit[k]:
                    continue
                left = dp[l][k - 1] if k > l else 1
                right = dp[k + 1][r] if k < r else 1
                total = (total + left * right) % MOD
            dp[l][r] = total * inv_len % MOD

    # factorial
    fact = 1
    for i in range(2, N + 1):
        fact = fact * i % MOD

    ans = dp[0][N - 1] * fact % MOD
    print(ans)

if __name__ == "__main__":
    main()
```
