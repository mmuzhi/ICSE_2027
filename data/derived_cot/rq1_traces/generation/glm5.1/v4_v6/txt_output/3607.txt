def compute_spf(limit):
    spf = list(range(limit+1))
    spf[0] = 0  # not used
    spf[1] = 1
    for i in range(2, int(limit**0.5)+1):
        if spf[i] == i:  # prime
            for j in range(i*i, limit+1, i):
                if spf[j] == j:
                    spf[j] = i
    return spf

```python
class Solution:
    def minOperations(self, nums: List[int]) -> int:
        if not nums:
            return 0
        max_val = max(nums)
        # compute smallest prime factor for each number up to max_val
        spf = list(range(max_val + 1))
        # 0 and 1 are special
        if max_val >= 0:
            spf[0] = 0
        if max_val >= 1:
            spf[1] = 1
        limit = int(max_val ** 0.5) + 1
        for i in range(2, limit):
            if spf[i] == i:  # i is prime
                step = i
                start = i * i
                for j in range(start, max_val + 1, step):
                    if spf[j] == j:
                        spf[j] = i
        n = len(nums)
        cur = nums[-1]
        ops = 0
        for i in range(n - 2, -1, -1):

                    cur = x
                else:
                    return -1
            else:
                p = spf[x]
                if x <= cur:
                    cur = x
                elif p <= cur:
                    ops += 1
                    cur = p
                else:
                    return -1
        return ops
```

```python
class Solution:
    def minOperations(self, nums: List[int]) -> int:
        if not nums:
            return 0

        max_val = max(nums)
        # smallest prime factor sieve

                    if spf[j] == j:
                        spf[j] = i

        n = len(nums)
        # last element: keep original (cheapest and most flexible)
        cur = nums[-1]
        ops = 0

                else:
                    return -1

                p = spf[x]           # smallest prime factor
                if x <= cur:
                    cur = x          # keep original, cost 0
                elif p <= cur:
                    ops += 1         # reduce to p, cost 1
                    cur = p
                else:
                    return -1

        return ops
```
