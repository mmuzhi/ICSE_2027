```python
            if max_j < 1:
                continue

            # Z‑array for the suffix starting at i (needed for condition B)
            z_suffix = z_array(i)

            if Z0[i] >= i:
                # splits where condition A holds (j >= i)
                if i <= max_j:
                    ans += max_j - i + 1

                # check condition B for j < i
                upper = min(i, max_j + 1)   # j < i and j ≤ max_j
                for j in range(1, upper):
                    if z_suffix[j] >= j:
                        ans += 1
            else:
                # condition A never true, check B for all j
                for j in range(1, max_j + 1):
                    if z_suffix[j] >= j:
                        ans += 1

        return ans
```
