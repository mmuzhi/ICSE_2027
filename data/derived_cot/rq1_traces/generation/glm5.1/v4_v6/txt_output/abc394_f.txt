Pseudo:
adj = [[] for _ in range(N+1)]
degree = [0]*(N+1)
for each edge:
    adj[u].append(v)
    adj[v].append(u)
    degree[u] += 1
    degree[v] += 1

S = [deg >= 4 for deg in degree]

parent = [0]*(N+1)
order = []
stack = [1]
parent[1] = -1
while stack:
    u = stack.pop()
    order.append(u)
    for v in adj[u]:
        if v != parent[u]:
            parent[v] = u
            stack.append(v)

dp1 = [-10**9]*(N+1)
dp0 = [-10**9]*(N+1)

for u in reversed(order):
    if S[u]:
        # collect dp1 of children
        children_dp1 = []
        for v in adj[u]:
            if v != parent[u]:
                if dp1[v] > 0:
                    children_dp1.append(dp1[v])
        children_dp1.sort(reverse=True)
        sum3 = sum(children_dp1[:3])
        sum4 = sum(children_dp1[:4])
        dp1[u] = 1 + sum3
        dp0[u] = 1 + sum4
    else:
        dp1[u] = -10**9
        dp0[u] = -10**9

max_c = max(dp0)
if max_c <= 0:
    print(-1)
else:
    print(3*max_c + 2)

```python
import sys

def solve():
    sys.setrecursionlimit(300000)
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    adj = [[] for _ in range(N+1)]
    deg = [0]*(N+1)
    for _ in range(N-1):
        u = int(next(it))
        v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
        deg[u] += 1
        deg[v] += 1
    
    S = [deg[i] >= 4 for i in range(N+1)]
    
    # iterative DFS to get parent and post-order
    parent = [0]*(N+1)
    order = []
    stack = [1]
    parent[1] = -1
    while stack:
        u = stack.pop()
        order.append(u)
        for v in adj[u]:
            if v != parent[u]:
                parent[v] = u
                stack.append(v)
    
    INF_NEG = -10**9
    dp1 = [INF_NEG]*(N+1)
    dp0 = [INF_NEG]*(N+1)
    
    for u in reversed(order):
        if S[u]:
            child_dp1 = []
            for v in adj[u]:
                if v != parent[u]:
                    if dp1[v] > 0:
                        child_dp1.append(dp1[v])
            child_dp1.sort(reverse=True)
            sum3 = sum(child_dp1[:3])
            sum4 = sum(child_dp1[:4])
            dp1[u] = 1 + sum3
            dp0[u] = 1 + sum4
        else:
            dp1[u] = INF_NEG
            dp0[u] = INF_NEG
    
    max_c = max(dp0)
    if max_c <= 0:
        print(-1)
    else:
        print(3 * max_c + 2)

if __name__ == "__main__":
    solve()
```

import sys

    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    adj = [[] for _ in range(N + 1)]
    deg = [0] * (N + 1)

        v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
        deg[u] += 1
        deg[v] += 1

    # S[v] is True if v can be part of the core (degree >= 4 in T)
    S = [deg[i] >= 4 for i in range(N + 1)]

    # Iterative DFS to obtain parent array and post-order
    parent = [0] * (N + 1)
    order = []

    parent[1] = -1
    while stack:
        u = stack.pop()

    dp1 = [INF_NEG] * (N + 1)   # u included, has parent in core
    dp0 = [INF_NEG] * (N + 1)   # u included, is root of core
