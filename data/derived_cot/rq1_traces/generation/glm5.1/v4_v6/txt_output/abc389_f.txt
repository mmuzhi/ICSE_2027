def solve():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    intervals = [(int(next(it)), int(next(it))) for _ in range(N)]
    Q = int(next(it))
    queries = [int(next(it)) for _ in range(Q)]

    MAX_X = 500000
    n = MAX_X
    size = 4 * n
    lo = [0] * size
    hi = [0] * size
    lazy = [0] * size

    sys.setrecursionlimit(1 << 25)

    def build(node, l, r):
        if l == r:
            val = l
            lo[node] = hi[node] = val
        else:
            mid = (l + r) >> 1
            build(node*2, l, mid)
            build(node*2+1, mid+1, r)
            lo[node] = lo[node*2]
            hi[node] = hi[node*2+1]

    build(1, 1, n)

    def apply(node, val):
        lo[node] += val
        hi[node] += val
        lazy[node] += val

    def push(node):
        if lazy[node]:
            apply(node*2, lazy[node])
            apply(node*2+1, lazy[node])
            lazy[node] = 0

    def range_add(node, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            apply(node, val)
            return
        push(node)
        mid = (l + r) >> 1
        if ql <= mid:
            range_add(node*2, l, mid, ql, qr, val)
        if qr > mid:
            range_add(node*2+1, mid+1, r, ql, qr, val)
        lo[node] = lo[node*2]
        hi[node] = hi[node*2+1]

    def find_first_ge(node, l, r, L):
        if hi[node] < L:
            return n + 1
        if l == r:
            return l
        push(node)
        mid = (l + r) >> 1
        if hi[node*2] >= L:
            return find_first_ge(node*2, l, mid, L)
        else:
            return find_first_ge(node*2+1, mid+1, r, L)

    def find_last_le(node, l, r, R):
        if lo[node] > R:
            return 0
        if l == r:
            return l
        push(node)
        mid = (l + r) >> 1
        if lo[node*2+1] <= R:
            return find_last_le(node*2+1, mid+1, r, R)
        else:
            return find_last_le(node*2, l, mid, R)

    # Process intervals
    for L, R in intervals:
        l = find_first_ge(1, 1, n, L)
        r = find_last_le(1, 1, n, R)
        if l <= r:
            range_add(1, 1, n, l, r, 1)

    # Collect answers
    ans = [0] * (n + 1)
    def collect(node, l, r):
        if l == r:
            ans[l] = lo[node]
        else:
            push(node)
            mid = (l + r) >> 1
            collect(node*2, l, mid)
            collect(node*2+1, mid+1, r)
    collect(1, 1, n)

    out = []
    for X in queries:
        out.append(str(ans[X]))
    sys.stdout.write("\n".join(out))

import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    intervals = [(int(next(it)), int(next(it))) for _ in range(N)]
    Q = int(next(it))
    queries = [int(next(it)) for _ in range(Q)]

    # The domain of initial ratings we need is up to the maximum query.
    n = max(queries) if queries else 1

    sys.setrecursionlimit(1 << 25)

    size = 4 * n
    lo = [0] * size          # minimum value in the segment (= value at leftmost index)
    hi = [0] * size          # maximum value in the segment (= value at rightmost index)
    lazy = [0] * size        # pending addition

    # ----- segment tree construction -----
    def build(node: int, l: int, r: int) -> None:
        if l == r:
            val = l
            lo[node] = hi[node] = val
        else:
            mid = (l + r) >> 1
            build(node * 2, l, mid)
            build(node * 2 + 1, mid + 1, r)
            lo[node] = lo[node * 2]
            hi[node] = hi[node * 2 + 1]

    build(1, 1, n)

    # ----- helpers -----
    def apply(node: int, val: int) -> None:
        lo[node] += val
        hi[node] += val
        lazy[node] += val

    def push(node: int) -> None:
        if lazy[node]:
            apply(node * 2, lazy[node])
            apply(node * 2 + 1, lazy[node])
            lazy[node] = 0

    # ----- range add -----
    def range_add(node: int, l: int, r: int, ql: int, qr: int, val: int) -> None:
        if ql <= l and r <= qr:
            apply(node, val)
            return
        push(node)
        mid = (l + r) >> 1
        if ql <= mid:
            range_add(node * 2, l, mid, ql, qr, val)
        if qr > mid:
            range_add(node * 2 + 1, mid + 1, r, ql, qr, val)
        lo[node] = lo[node * 2]
        hi[node] = hi[node * 2 + 1]

    # ----- first index with value >= L -----
    def find_first_ge(node: int, l: int, r: int, L: int) -> int:
        if hi[node] < L:
            return n + 1
        if l == r:
            return l
        push(node)
        mid = (l + r) >> 1
        if hi[node * 2] >= L:
            return find_first_ge(node * 2, l, mid, L)
        else:
            return find_first_ge(node * 2 + 1, mid + 1, r, L)

    # ----- last index with value <= R -----
    def find_last_le(node: int, l: int, r: int, R: int) -> int:
        if lo[node] > R:
            return 0
        if l == r:
            return l
        push(node)
        mid = (l + r) >> 1
        if lo[node * 2 + 1] <= R:
            return find_last_le(node * 2 + 1, mid + 1, r, R)
        else:
            return find_last_le(node * 2, l, mid, R)

    # ----- process contests -----
    for L, R in intervals:
        l = find_first_ge(1, 1, n, L)
        r = find_last_le(1, 1, n, R)
        if l <= r:
            range_add(1, 1, n, l, r, 1)

    # ----- collect final ratings -----
    ans = [0] * (n + 1)
    def collect(node: int, l: int, r: int) -> None:
        if l == r:
            ans[l] = lo[node]
        else:
            push(node)
            mid = (l + r) >> 1
            collect(node * 2, l, mid)
            collect(node * 2 + 1, mid + 1, r)

    collect(1, 1, n)

    # ----- answer queries -----
    out = []
    for X in queries:
        out.append(str(ans[X]))
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
