import sys

def solve():
    import sys
    input = sys.stdin.readline
    T = int(input())
    out = []
    for _ in range(T):
        N = int(input())
        A = list(map(int, input().split()))
        # length 2N
        l = [0]*(N+1)
        r = [0]*(N+1)
        for idx, val in enumerate(A, start=1):
            if l[val] == 0:
                l[val] = idx
            else:
                r[val] = idx
        # Now l and r filled. Ensure l< r; but if by order, first occurrence is l, second is r, so l<r.
        good_set = set()
        good_vals = []
        for i in range(1, N+1):
            if r[i] != l[i] + 1:
                good_set.add((l[i], r[i]))
                good_vals.append(i)
        ans = 0
        for i in good_vals:
            li, ri = l[i], r[i]
            if (li+1, ri+1) in good_set:
                ans += 1
            if (li+1, ri-1) in good_set:
                ans += 1
        out.append(str(ans))
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    input = sys.stdin.readline
    T = int(input())
    out_lines = []
    for _ in range(T):
        N = int(input())
        A = list(map(int, input().split()))
        # first and second occurrence (1‑indexed)
        first = [0] * (N + 1)
        second = [0] * (N + 1)
        for pos, val in enumerate(A, start=1):
            if first[val] == 0:
                first[val] = pos
            else:
                second[val] = pos

        max_pos = 2 * N
        # R[l] = r if the couple is non‑adjacent, otherwise 0
        R = [0] * (max_pos + 2)   # +2 to safely access l+1
        for i in range(1, N + 1):
            if second[i] != first[i] + 1:   # non‑adjacent
                R[first[i]] = second[i]

        ans = 0
        for l in range(1, max_pos):
            if R[l] and R[l + 1] and abs(R[l] - R[l + 1]) == 1:
                ans += 1
        out_lines.append(str(ans))

    sys.stdout.write("\n".join(out_lines))
