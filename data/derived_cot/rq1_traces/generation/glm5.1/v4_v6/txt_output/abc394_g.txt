There is a city divided into H \times W blocks in the north-south-east-west directions, and there is exactly one building in each block.
Specifically, in the block at the i-th row from the north (1\leq i\leq H) and the j-th column from the west (1\leq j\leq W) (hereafter referred to as block (i,j)), there is a building of F_{i,j} floors.
Takahashi has two ways of moving. If he is on the X-th floor (1\leq X\leq F_{i,j}) of the building in block (i,j), he can:

- Move up or down one floor within the same building using stairs. If X=1, he cannot move down; if X=F_{i,j}, he cannot move up.
- Choose a building with at least X floors in a cardinally adjacent block, and move to the X-th floor of that building using a (sky) walkway.

Here, two blocks (i,j) and (i',j') are cardinally adjacent if and only if \lvert i - i'\rvert + \lvert j - j'\rvert = 1.
You are given Q queries to be answered. The i-th query (1\leq i\leq Q) is the following.

Find the minimum possible number of times that Takahashi uses stairs to move from the Y_i-th floor of the building in block (A_i,B_i) to the Z_i-th floor of the building in block (C_i,D_i).
The count of times using stairs is incremented each time he moves up or down one floor, possibly multiple times within the same building. (For example, moving from the 1st floor to the 6th floor of a building counts as 5 uses of stairs.)
Note that he does not have to minimize the number of times he uses walkways.

\vdots
A_Q B_Q Y_Q C_Q D_Q Z_Q

Output

Print Q lines. The i-th line should contain the answer to the i-th query as an integer.

import sys
from array import array
from collections import deque

def solve():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    H = int(next(it))
    W = int(next(it))
    N = H * W
    TotalNodes = 2 * N
    weight = array('i', [0]) * TotalNodes
    for i in range(N):
        weight[i] = int(next(it))

    edges = []
    for i in range(H):
        row_start = i * W
        for j in range(W):
            idx = row_start + j
            f_ij = weight[idx]
            if j + 1 < W:
                idx2 = row_start + j + 1
                f_ij2 = weight[idx2]
                w = f_ij if f_ij < f_ij2 else f_ij2
                edges.append((w, idx, idx2))
            if i + 1 < H:
                idx2 = row_start + W + j
                f_ij2 = weight[idx2]
                w = f_ij if f_ij < f_ij2 else f_ij2
                edges.append((w, idx, idx2))

    edges.sort(reverse=True)

    uf_parent = list(range(N))
    uf_rank = [0] * N

    def find(x):
        while uf_parent[x] != x:
            uf_parent[x] = uf_parent[uf_parent[x]]
            x = uf_parent[x]
        return x

    def union(x, y):
        x = find(x)
        y = find(y)
        if x == y:
            return False, x
        if uf_rank[x] < uf_rank[y]:
            uf_parent[x] = y
            return True, y
        elif uf_rank[x] > uf_rank[y]:
            uf_parent[y] = x
            return True, x
        else:
            uf_parent[y] = x
            uf_rank[x] += 1
            return True, x

    tree_id = list(range(N))
    next_node = N
    adj = [[] for _ in range(TotalNodes)]

    for w, u, v in edges:
        ru = find(u)
        rv = find(v)
        if ru != rv:
            new_node = next_node
            next_node += 1
            weight[new_node] = w
            tu = tree_id[ru]
            tv = tree_id[rv]
            adj[new_node].append(tu)
            adj[tu].append(new_node)
            adj[new_node].append(tv)
            adj[tv].append(new_node)
            union(ru, rv)
            new_root = find(ru)
            tree_id[new_root] = new_node

    del edges, uf_parent, uf_rank, tree_id

    root = next_node - 1
    LOG = 19
    total_nodes = next_node
    depth = array('i', [0]) * total_nodes
    up = array('i', [0]) * (total_nodes * LOG)
    for k in range(LOG):
        up[root * LOG + k] = root

    q = deque([root])
    visited = bytearray(total_nodes)
    visited[root] = 1
    while q:
        node = q.popleft()
        for nei in adj[node]:
            if not visited[nei]:
                visited[nei] = 1
                depth[nei] = depth[node] + 1
                up[nei * LOG] = node
                for k in range(1, LOG):
                    up[nei * LOG + k] = up[up[nei * LOG + k - 1] * LOG + k - 1]
                q.append(nei)

    del adj, visited

    Q = int(next(it))
    out = []
    up_arr = up
    dep_arr = depth
    w_arr = weight
    W_const = W
    LOG_const = LOG

    for _ in range(Q):
        A = int(next(it)) - 1
        B = int(next(it)) - 1
        Y = int(next(it))
        C = int(next(it)) - 1
        D = int(next(it)) - 1
        Z = int(next(it))
        u = A * W_const + B
        v = C * W_const + D
        if dep_arr[u] < dep_arr[v]:
            u, v = v, u
        diff = dep_arr[u] - dep_arr[v]
        k = 0
        while diff:
            if diff & 1:
                u = up_arr[u * LOG_const + k]
            diff >>= 1
            k += 1
        if u != v:
            for k in range(LOG_const - 1, -1, -1):
                u_k = up_arr[u * LOG_const + k]
                v_k = up_arr[v * LOG_const + k]
                if u_k != v_k:
                    u = u_k
                    v = v_k
            u = up_arr[u * LOG_const]  # lca
        B_val = w_arr[u]
        if Y > Z:
            min_yz = Z
        else:
            min_yz = Y
        if min_yz > B_val:
            cost = abs(Y - Z) + 2 * (min_yz - B_val)
        else:
            cost = abs(Y - Z)
        out.append(str(cost))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
