import sys

def min_attacks(H, r):
    if r == 1:
        f1, f2 = 1, 2
    elif r == 2:
        f1, f2 = 1, 4
    else: # r == 0
        f1, f2 = 3, 4

    # remainder 0
    q0 = (H + 4) // 5
    k0 = 3 * q0

    # remainder 1
    if H <= f1:
        k1 = 1
    else:
        q1 = (H - f1 + 4) // 5
        k1 = 3 * q1 + 1

    # remainder 2
    if H <= f2:
        k2 = 2
    else:
        q2 = (H - f2 + 4) // 5
        k2 = 3 * q2 + 2

    return min(k0, k1, k2)

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input().strip())
    H = list(map(int, input().split()))
    total = 0
    for h in H:
        r = (total + 1) % 3
        a = min_attacks(h, r)
        total += a
    print(total)

if __name__ == "__main__":
    solve()

```python

import sys

def attacks_needed(H: int, r: int) -> int:
    """
    Minimal number of consecutive attacks starting at a position
    with offset r (0,1,2) to deal at least H damage.
    r = (starting_index) % 3
    """
    if r == 1:          # pattern [1,1,3]
        f1, f2 = 1, 2
    elif r == 2:        # pattern [1,3,1]
        f1, f2 = 1, 4
    else:               # r == 0, pattern [3,1,1]
        f1, f2 = 3, 4

    # remainder 0
    q0 = (H + 4) // 5          # ceil(H/5)
    k0 = 3 * q0

    # remainder 1
    if H <= f1:
        k1 = 1
    else:
        q1 = (H - f1 + 4) // 5  # ceil((H - f1)/5)
        k1 = 3 * q1 + 1

    # remainder 2
    if H <= f2:
        k2 = 2
    else:
        q2 = (H - f2 + 4) // 5  # ceil((H - f2)/5)
        k2 = 3 * q2 + 2

    return min(k0, k1, k2)


def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    H = list(map(int, input().split()))

    total = 0          # number of attacks already performed
    for h in H:
        r = (total + 1) % 3   # offset of the first attack on this enemy
        a = attacks_needed(h, r)
        total += a

    print(total)


if __name__ == "__main__":
    solve()
```
