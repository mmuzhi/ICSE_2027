=== SOURCE CODE (cpp) ===
#include <unordered_map>
#include <string>

class ShoppingCart {
public:
    ShoppingCart();

    void add_item(const std::string& item, double price, int quantity = 1);
    void remove_item(const std::string& item, int quantity = 1);
    std::unordered_map<std::string, std::pair<double, int>> view_items() const;
    double total_price() const;

private:
    std::unordered_map<std::string, std::pair<double, int>> items;
};

ShoppingCart::ShoppingCart() {}

void ShoppingCart::add_item(const std::string& item, double price, int quantity) {
    if (items.find(item) != items.end()) {
        items[item] = { price, quantity };
    }
    else {
        items[item] = { price, quantity };
    }
}

void ShoppingCart::remove_item(const std::string& item, int quantity) {
    auto it = items.find(item);
    if (it != items.end()) {
        it->second.second -= quantity;
        if (it->second.second <= 0) {
            items.erase(it);
        }
    }
}

std::unordered_map<std::string, std::pair<double, int>> ShoppingCart::view_items() const {
    return items;
}

double ShoppingCart::total_price() const {
    double total = 0.0;
    for (const auto& [item, info] : items) {
        total += info.first * info.second;
    }
    return total;
}

3. **Removing Items**: The `remove_item` method reduces the quantity of the specified item. If the quantity drops to zero or below, the item is removed from the cart.
4. **Viewing Items**: The `view_items` method returns a copy of the items dictionary, ensuring the original data remains unmodified.

### Solution Code
```python
class ShoppingCart:
    def __init__(self):
        self.items = {}

    def add_item(self, item, price, quantity=1):
        self.items[item] = (price, quantity)

    def remove_item(self, item, quantity=1):
        if item in self.items:
            current_price, current_quantity = self.items[item]
            new_quantity = current_quantity - quantity
            if new_quantity <= 0:
                del self.items[item]
            else:
                self.items[item] = (current_price, new_quantity)

    def view_items(self):
        return dict(self.items)

    def total_price(self):
        total = 0.0
        for price, quantity in self.items.values():
            total += price * quantity
        return total
```

### Explanation
1. **Initialization**: The `__init__` method initializes an empty dictionary `items` to store cart items.
2. **Adding Items**: The `add_item` method directly assigns the given `price` and `quantity` to the `item` in the dictionary. If the item exists, it is overwritten; otherwise, a new entry is created.
3. **Removing Items**: The `remove_item` method checks if the item exists. If it does, it reduces the quantity by the specified amount. If the resulting quantity is zero or negative, the item is removed from the dictionary.
4. **Viewing Items**: The `view_items` method returns a shallow copy of the `items` dictionary to prevent external modifications to the internal state.
5. **Total Price Calculation**: The `total_price` method calculates the total cost by iterating over each item's price and quantity, multiplying them, and summing the results.

This implementation ensures that the Python version behaves identically to the original C++ code, handling all edge cases such as item replacement during addition and removal when quantities become non-positive.
