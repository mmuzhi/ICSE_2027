#include <algorithm>
#include <vector>
#include <sstream>


#include <string>

class BigNumCalculator {
public:
    static std::string add(const std::string& num1, const std::string& num2);
    static std::string subtract(const std::string& num1, const std::string& num2);
    static std::string multiply(const std::string& num1, const std::string& num2);
};

std::string BigNumCalculator::add(const std::string& num1, const std::string& num2) {
    int max_length = std::max(num1.size(), num2.size());
    std::string num1_padded = std::string(max_length - num1.size(), '0') + num1;
    std::string num2_padded = std::string(max_length - num2.size(), '0') + num2;

    int carry = 0;
    std::string result;
    for (int i = max_length - 1; i >= 0; --i) {
        int digit_sum = (num1_padded[i] - '0') + (num2_padded[i] - '0') + carry;
        carry = digit_sum / 10;
        int digit = digit_sum % 10;
        result.insert(result.begin(), digit + '0');
    }

    if (carry > 0) {
        result.insert(result.begin(), carry + '0');
    }

    return result;
}

std::string BigNumCalculator::subtract(const std::string& num1, const std::string& num2) {
    std::string num1_local = num1;
    std::string num2_local = num2;
    bool negative = false;

    if (num1_local.size() < num2_local.size() || (num1_local.size() == num2_local.size() && num1_local < num2_local)) {
        std::swap(num1_local, num2_local);
        negative = true;
    }

    int max_length = std::max(num1_local.size(), num2_local.size());
    num1_local = std::string(max_length - num1_local.size(), '0') + num1_local;
    num2_local = std::string(max_length - num2_local.size(), '0') + num2_local;

    int borrow = 0;
    std::string result;
    for (int i = max_length - 1; i >= 0; --i) {
        int digit_diff = (num1_local[i] - '0') - (num2_local[i] - '0') - borrow;

        if (digit_diff < 0) {
            digit_diff += 10;
            borrow = 1;
        }
        else {
            borrow = 0;
        }

        result.insert(result.begin(), digit_diff + '0');
    }

    while (result.size() > 1 && result[0] == '0') {
        result.erase(result.begin());
    }

    if (negative) {
        result.insert(result.begin(), '-');
    }

    return result;
}

std::string BigNumCalculator::multiply(const std::string& num1, const std::string& num2) {
    int len1 = num1.size();
    int len2 = num2.size();
    std::vector<int> result(len1 + len2, 0);

    for (int i = len1 - 1; i >= 0; --i) {
        for (int j = len2 - 1; j >= 0; --j) {
            int mul = (num1[i] - '0') * (num2[j] - '0');
            int p1 = i + j;
            int p2 = i + j + 1;
            int total = mul + result[p2];

            result[p1] += total / 10;
            result[p2] = total % 10;
        }
    }

    std::ostringstream oss;
    bool leadingZero = true;
    for (int num : result) {
        if (num == 0 && leadingZero) continue;
        leadingZero = false;
        oss << num;
    }

    return oss.str().empty() ? "0" : oss.str();
}

To solve this problem, we need to implement a Python class that mimics the behavior of the given C++ class for handling arithmetic operations on large numbers represented as strings. The operations include addition, subtraction, and multiplication. The solution must handle large numbers that exceed the standard integer limits by processing each digit individually.

### Approach
1. **Addition**:
   - Pad the shorter number with leading zeros to make both numbers the same length.
   - Process digits from least significant to most significant, adding corresponding digits along with any carry from the previous step.
   - After processing all digits, if there's a remaining carry, prepend it to the result.

2. **Subtraction**:
   - Determine if the result will be negative by comparing the two numbers. If the first number is smaller, swap them and mark the result as negative.
   - Pad the shorter number with leading zeros to make both numbers the same length.
   - Process digits from least significant to most significant, subtracting digits and handling borrows as needed.
   - Remove leading zeros from the result and prepend a minus sign if the result is negative.

3. **Multiplication**:
   - Use a list to store intermediate results with a length equal to the sum of the lengths of the two numbers.
   - Multiply each digit of the first number with each digit of the second number, adding the result to the appropriate positions in the intermediate list.
   - Handle carries during the multiplication process.
   - Convert the intermediate list to a string, skipping leading zeros. If the result is all zeros, return "0".

```python
class BigNumCalculator:
    @staticmethod
    def add(num1, num2):
        max_length = max(len(num1), len(num2))
        num1_padded = num1.zfill(max_length)
        num2_padded = num2.zfill(max_length)
        
        carry = 0
        result = []
        for i in range(max_length - 1, -1, -1):
            digit_sum = int(num1_padded[i]) + int(num2_padded[i]) + carry
            carry = digit_sum // 10
            digit = digit_sum % 10
            result.append(str(digit))
        
        if carry > 0:
            result.append(str(carry))
        
        result_str = ''.join(result[::-1])
        return result_str

    @staticmethod
    def subtract(num1, num2):
        num1_local = num1
        num2_local = num2
        negative = False
        
        if len(num1_local) < len(num2_local) or (len(num1_local) == len(num2_local) and num1_local < num2_local):
            num1_local, num2_local = num2_local, num1_local
            negative = True
        
        max_length = max(len(num1_local), len(num2_local))
        num1_local = num1_local.zfill(max_length)
        num2_local = num2_local.zfill(max_length)
        
        borrow = 0
        result = []
        for i in range(max_length - 1, -1, -1):
            digit_diff = int(num1_local[i]) - int(num2_local[i]) - borrow
            if digit_diff < 0:
                digit_diff += 10
                borrow = 1
            else:
                borrow = 0
            result.append(str(digit_diff))
        
        result_str = ''.join(result[::-1]).lstrip('0')
        if not result_str:
            result_str = '0'
        elif negative:
            result_str = '-' + result_str
        return result_str

    @staticmethod
    def multiply(num1, num2):
        len1 = len(num1)
        len2 = len(num2)
        result_arr = [0] * (len1 + len2)
        
        for i in range(len1 - 1, -1, -1):
            for j in range(len2 - 1, -1, -1):
                mul = int(num1[i]) * int(num2[j])
                p1 = i + j
                p2 = i + j + 1
                total = mul + result_arr[p2]
                result_arr[p1] += total // 10
                result_arr[p2] = total % 10
        
        result_str = ''.join(str(x) for x in result_arr).lstrip('0')
        return result_str if result_str != '' else '0'
```

### Explanation
1. **Addition**:
   - The numbers are padded with leading zeros to ensure they have the same length.
   - Digits are processed from right to left. Each digit sum is computed along with any carry from the previous step. The carry is updated for the next iteration.
   - If a carry remains after processing all digits, it is prepended to the result.

2. **Subtraction**:
   - The numbers are compared to determine if the result will be negative. The larger number is placed first for subtraction.
   - Digits are processed from right to left. If a digit difference is negative, a borrow is taken from the next higher digit.
   - Leading zeros are stripped from the result, and a minus sign is prepended if the result is negative.

3. **Multiplication**:
   - An array is initialized to store intermediate results. Each digit of the first number is multiplied with each digit of the second number.
   - The results are stored in appropriate positions in the array, with carries propagated to higher positions.
   - The intermediate array is converted to a string, leading zeros are stripped, and "0" is returned if the result is zero.

This approach ensures that large numbers are handled efficiently without relying on Python's built-in big integers, maintaining the same behavior as the original C++ code.
