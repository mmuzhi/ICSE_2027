#include <vector>

class Interpolation {
public:
    Interpolation() = default;

    
    static std::vector<double> interpolate_1d(const std::vector<double>& x,
        const std::vector<double>& y,
        const std::vector<double>& x_interp);

    
    static std::vector<double> interpolate_2d(const std::vector<double>& x,
        const std::vector<double>& y,
        const std::vector<std::vector<double>>& z,
        const std::vector<double>& x_interp,
        const std::vector<double>& y_interp);
};

std::vector<double> Interpolation::interpolate_1d(const std::vector<double>& x,
    const std::vector<double>& y,
    const std::vector<double>& x_interp) {
    std::vector<double> y_interp;
    for (double xi : x_interp) {
        for (size_t i = 0; i < x.size() - 1; ++i) {
            if (x[i] <= xi && xi <= x[i + 1]) {
                double yi = y[i] + (y[i + 1] - y[i]) * (xi - x[i]) / (x[i + 1] - x[i]);
                y_interp.push_back(yi);
                break;
            }
        }
    }
    return y_interp;
}

std::vector<double> Interpolation::interpolate_2d(const std::vector<double>& x,
    const std::vector<double>& y,
    const std::vector<std::vector<double>>& z,
    const std::vector<double>& x_interp,
    const std::vector<double>& y_interp) {
    std::vector<double> z_interp;
    for (size_t k = 0; k < x_interp.size(); ++k) {
        double xi = x_interp[k];
        double yi = y_interp[k];
        for (size_t i = 0; i < x.size() - 1; ++i) {
            if (x[i] <= xi && xi <= x[i + 1]) {
                for (size_t j = 0; j < y.size() - 1; ++j) {
                    if (y[j] <= yi && yi <= y[j + 1]) {
                        double z00 = z[i][j];
                        double z01 = z[i][j + 1];
                        double z10 = z[i + 1][j];
                        double z11 = z[i + 1][j + 1];
                        double zi = (z00 * (x[i + 1] - xi) * (y[j + 1] - yi) +
                            z10 * (xi - x[i]) * (y[j + 1] - yi) +
                            z01 * (x[i + 1] - xi) * (yi - y[j]) +
                            z11 * (xi - x[i]) * (yi - y[j])) / ((x[i + 1] - x[i]) * (y[j + 1] - y[j]));
                        z_interp.push_back(zi);
                        break;
                    }
                }
                break;
            }
        }
    }
    return z_interp;
}

=== RAW ANSWER ===

To solve this problem, we need to translate a C++ class that performs linear and bilinear interpolation into Python while maintaining identical behavior. The C++ class contains two static methods: `interpolate_1d` for linear interpolation in one dimension and `interpolate_2d` for bilinear interpolation in two dimensions. The goal is to replicate the functionality exactly, including handling edge cases where interpolation points might not be found within any interval.

### Approach
1. **Linear Interpolation (1D)**:
   - For each interpolation point `xi` in `x_interp`, iterate through the known `x` values to find the interval `[x[i], x[i+1]]` that contains `xi`.
   - Once found, compute the interpolated value `yi` using the linear interpolation formula: `y[i] + (y[i+1] - y[i]) * (xi - x[i]) / (x[i+1] - x[i])`.
   - Append `yi` to the result list and break out of the loop to move to the next interpolation point. If no interval is found, the point is skipped.

2. **Bilinear Interpolation (2D)**:
   - For each pair of interpolation points `(xi, yi)` from `x_interp` and `y_interp`, iterate through the known `x` values to find the interval `[x[i], x[i+1]]` containing `xi`.
   - Once found, iterate through the known `y` values to find the interval `[y[j], y[j+1]]` containing `yi`.
   - Compute the interpolated value `zi` using the bilinear interpolation formula involving the four surrounding points `(x[i], y[j])`, `(x[i], y[j+1])`, `(x[i+1], y[j])`, and `(x[i+1], y[j+1])`.
   - Append `zi` to the result list and break out of both loops to move to the next interpolation point. If either interval is not found, the point is skipped.

### Solution Code
```python
class Interpolation:
    @staticmethod
    def interpolate_1d(x, y, x_interp):
        y_interp = []
        for xi in x_interp:
            for i in range(len(x) - 1):
                if x[i] <= xi <= x[i+1]:
                    yi = y[i] + (y[i+1] - y[i]) * (xi - x[i]) / (x[i+1] - x[i])
                    y_interp.append(yi)
                    break
        return y_interp

    @staticmethod
    def interpolate_2d(x, y, z, x_interp, y_interp):
        z_interp = []
        for k in range(len(x_interp)):
            xi = x_interp[k]
            yi = y_interp[k]
            for i in range(len(x) - 1):
                if x[i] <= xi <= x[i+1]:
                    for j in range(len(y) - 1):
                        if y[j] <= yi <= y[j+1]:
                            z00 = z[i][j]
                            z01 = z[i][j+1]
                            z10 = z[i+1][j]
                            z11 = z[i+1][j+1]
                            numerator = (z00 * (x[i+1] - xi) * (y[j+1] - yi) +
                                        z10 * (xi - x[i]) * (y[j+1] - yi) +
                                        z01 * (x[i+1] - xi) * (yi - y[j]) +
                                        z11 * (xi - x[i]) * (yi - y[j]))
                            denominator = (x[i+1] - x[i]) * (y[j+1] - y[j])
                            zi = numerator / denominator
                            z_interp.append(zi)
                            break
                    break
        return z_interp
```

### Explanation
- **Linear Interpolation (1D)**: The method `interpolate_1d` processes each point in `x_interp` by checking each interval in the known `x` values. If the point lies within an interval, the corresponding `y` value is interpolated linearly and added to the result list.
- **Bilinear Interpolation (2D)**: The method `interpolate_2d` processes each pair of points from `x_interp` and `y_interp`. For each pair, it checks intervals in the known `x` and `y` values. If both intervals are found, the `z` value is interpolated using the surrounding four points and added to the result list.
- **Edge Handling**: Both methods skip any interpolation points that do not lie within any interval of the known values, mirroring the behavior of the original C++ code. The loops break immediately upon finding the relevant intervals to ensure efficiency and correctness.
