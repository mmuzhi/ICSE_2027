class DecryptionUtils {
public:
    DecryptionUtils(const std::string& key);

    std::string caesar_decipher(const std::string& ciphertext, int shift);
    std::string vigenere_decipher(const std::string& ciphertext);
    std::string rail_fence_decipher(const std::string& encrypted_text, int rails);

private:
    std::string key_;
};

DecryptionUtils::DecryptionUtils(const std::string& key) : key_(key) {}

std::string DecryptionUtils::caesar_decipher(const std::string& ciphertext, int shift) {
    shift = shift % 26; 
    std::string plaintext;
    for (char c : ciphertext) {
        if (std::isalpha(c)) {
            char base = std::isupper(c) ? 'A' : 'a';
            char shifted_char = (c - base - shift + 26) % 26 + base;
            plaintext += shifted_char;
        }
        else {
            plaintext += c;
        }
    }
    return plaintext;
}

std::string DecryptionUtils::vigenere_decipher(const std::string& ciphertext) {
    std::string decrypted_text;
    size_t key_length = key_.length();
    size_t key_index = 0;

    for (char c : ciphertext) {
        if (std::isalpha(c)) {
            int shift = std::tolower(key_[key_index % key_length]) - 'a';
            char base = std::islower(c) ? 'a' : 'A';
            char decrypted_char = (std::tolower(c) - 'a' - shift + 26) % 26 + 'a';
            decrypted_text += std::isupper(c) ? std::toupper(decrypted_char) : decrypted_char;
            ++key_index;
        }
        else {
            decrypted_text += c;
        }
    }
    return decrypted_text;
}

std::string DecryptionUtils::rail_fence_decipher(const std::string& encrypted_text, int rails) {
    std::string plain_text;
    size_t n = encrypted_text.length();
    if (rails <= 1) return encrypted_text;

    std::vector<std::string> fence(rails, std::string(n, '\n'));

    int direction = -1;
    int row = 0;
    int col = 0;

    for (size_t i = 0; i < n; ++i) {
        if (row == 0 || row == rails - 1) {
            direction = -direction;
        }
        fence[row][col++] = '*';
        row += direction;
    }

    size_t index = 0;
    for (int r = 0; r < rails; ++r) {
        for (int c = 0; c < n; ++c) {
            if (fence[r][c] == '*') {
                fence[r][c] = encrypted_text[index++];
            }
        }
    }

    direction = -1;
    row = 0;
    col = 0;

    for (size_t i = 0; i < n; ++i) {
        if (row == 0 || row == rails - 1) {
            direction = -direction;
        }
        plain_text += fence[row][col++];
        row += direction;
    }

    return plain_text;
}

1. **Caesar Cipher Decryption**: 
   - Adjust the shift value modulo 26 to handle large shifts.
   - For each character in the ciphertext:
     - If it's an alphabetic character, determine its base ('A' for uppercase, 'a' for lowercase).
     - Compute the decrypted character by reversing the shift operation, ensuring it wraps around the alphabet correctly.
     - Non-alphabetic characters are added to the result as-is.

2. **Vigenère Cipher Decryption**:
   - Use the key provided during initialization.
   - For each alphabetic character in the ciphertext:
     - Determine the shift value from the corresponding key character (converted to lowercase).
     - Decrypt the character by reversing the shift, similar to the Caesar cipher but using the key's shift value.
     - Maintain the case (uppercase or lowercase) of the original character.
     - Non-alphabetic characters are added to the result without modification.

3. **Rail Fence Cipher Decryption**:
   - If the number of rails is 1 or less, return the ciphertext as-is.
   - Create a grid (rails × length of ciphertext) initialized with placeholders.
   - Simulate the rail fence pattern by moving in a zigzag manner, marking positions where characters will be placed.
   - Fill these marked positions with characters from the ciphertext in order.
   - Traverse the same zigzag pattern to collect the decrypted characters into the result.

class DecryptionUtils:
    def __init__(self, key: str):
        self.key = key

    def caesar_decipher(self, ciphertext: str, shift: int) -> str:

        if shift < 0:
            shift += 26
        plaintext = []
        for char in ciphertext:
            if char.isalpha():
                base = 'A' if char.isupper() else 'a'
                shifted_char = chr((ord(char) - ord(base) - shift + 26) % 26 + ord(base))
                plaintext.append(shifted_char)

                plaintext.append(char)

### Explanation
- **Caesar Decipher**: The shift value is adjusted modulo 26 to handle large shifts. Each alphabetic character is shifted back by the specified value, wrapping around the alphabet if necessary. Non-alphabetic characters remain unchanged.
- **Vigenère Decipher**: The key is cycled to provide shift values for each alphabetic character in the ciphertext. The shift value is derived from the corresponding key character (converted to lowercase). The decryption reverses the shift while preserving the original character's case.
- **Rail Fence Decipher**: The ciphertext is placed into a grid following a zigzag pattern determined by the number of rails. The grid is filled row-wise with the ciphertext characters, and then the zigzag traversal is repeated to read the decrypted characters in order. If the number of rails is 1 or less, the ciphertext is returned as-is.
