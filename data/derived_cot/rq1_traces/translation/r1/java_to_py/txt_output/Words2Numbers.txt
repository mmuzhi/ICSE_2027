package org.example;

import java.util.HashMap;
import java.util.Map;

public class Words2Numbers {

    private Map<String, int[]> numwords;
    private String[] units;
    private String[] tens;
    private String[] scales;
    private Map<String, Integer> ordinalWords;
    private String[][] ordinalEndings;

    public Words2Numbers() {
        numwords = new HashMap<>();
        units = new String[]{
                "zero", "one", "two", "three", "four", "five", "six", "seven", "eight",
                "nine", "ten", "eleven", "twelve", "thirteen", "fourteen", "fifteen",
                "sixteen", "seventeen", "eighteen", "nineteen"
        };
        tens = new String[]{
                "", "", "twenty", "thirty", "forty", "fifty", "sixty", "seventy", "eighty", "ninety"
        };
        scales = new String[]{
                "hundred", "thousand", "million", "billion", "trillion"
        };

        numwords.put("and", new int[]{1, 0});
        for (int idx = 0; idx < units.length; idx++) {
            numwords.put(units[idx], new int[]{1, idx});
        }
        for (int idx = 0; idx < tens.length; idx++) {
            numwords.put(tens[idx], new int[]{1, idx * 10});
        }
        for (int idx = 0; idx < scales.length; idx++) {
            numwords.put(scales[idx], new int[]{(int) Math.pow(10, (idx * 3 == 0 ? 2 : idx * 3)), 0});
        }

        ordinalWords = new HashMap<>();
        ordinalWords.put("first", 1);
        ordinalWords.put("second", 2);
        ordinalWords.put("third", 3);
        ordinalWords.put("fifth", 5);
        ordinalWords.put("eighth", 8);
        ordinalWords.put("ninth", 9);
        ordinalWords.put("twelfth", 12);

        ordinalEndings = new String[][]{
                {"ieth", "y"}, {"th", ""}
        };
    }

    public String text2int(String textnum) {
        textnum = textnum.replace("-", " ");

        int current = 0, result = 0;
        StringBuilder curstring = new StringBuilder();
        boolean onnumber = false;

        for (String word : textnum.split(" ")) {
            if (ordinalWords.containsKey(word)) {
                int scale = 1, increment = ordinalWords.get(word);
                current = current * scale + increment;
                onnumber = true;
            } else {
                for (String[] ending : ordinalEndings) {
                    if (word.endsWith(ending[0])) {
                        word = word.substring(0, word.length() - ending[0].length()) + ending[1];
                    }
                }

                if (!numwords.containsKey(word)) {
                    if (onnumber) {
                        curstring.append(result + current).append(" ");
                    }
                    curstring.append(word).append(" ");
                    result = current = 0;
                    onnumber = false;
                } else {
                    int scale = numwords.get(word)[0];
                    int increment = numwords.get(word)[1];
                    current = current * scale + increment;
                    if (scale > 100) {
                        result += current;
                        current = 0;
                    }
                    onnumber = true;
                }
            }
        }

        if (onnumber) {
            curstring.append(result + current);
        }

        return curstring.toString();
    }

    public boolean isValidInput(String textnum) {
        textnum = textnum.replace("-", " ");

        for (String word : textnum.split(" ")) {
            if (ordinalWords.containsKey(word)) {
                continue;
            } else {
                for (String[] ending : ordinalEndings) {
                    if (word.endsWith(ending[0])) {
                        word = word.substring(0, word.length() - ending[0].length()) + ending[1];
                    }
                }

                if (!numwords.containsKey(word)) {
                    return false;
                }
            }
        }

        return true;
    }
}

=== RAW ANSWER ===

To solve this problem, we need to convert a string containing words representing numbers into their corresponding numerical values while preserving any non-numeric words. The solution should handle both cardinal and ordinal numbers, and validate whether the input string consists of valid number words.

### Approach
1. **Initialization**: The constructor initializes dictionaries and lists to map words to their numerical values. This includes:
   - **Units**: Words like "one", "two", up to "nineteen".
   - **Tens**: Words like "twenty", "thirty", etc.
   - **Scales**: Words like "hundred", "thousand", etc.
   - **Ordinal Words**: Special cases like "first", "second", etc.
   - **Ordinal Endings**: Suffixes like "ieth" and "th" which are replaced to convert ordinal words to cardinal forms.

2. **Text to Integer Conversion (`text2int` method)**:
   - **Preprocessing**: Replace hyphens with spaces and split the input string into words.
   - **Processing Each Word**:
     - If the word is an ordinal (e.g., "first"), directly use its numerical value.
     - For other words, check if they have ordinal endings (like "ieth" or "th") and convert them to their cardinal form (e.g., "twentieth" becomes "twenty").
     - If the word is a valid number word, update the current numerical value. If the word represents a scale (e.g., "thousand"), add the current value to the result and reset the current value.
     - Non-number words are appended to the result string as-is, with any preceding accumulated numerical value.

3. **Validation (`isValidInput` method)**:
   - Check each word in the input string after preprocessing. If the word is an ordinal or can be converted to a cardinal form and found in the number words dictionary, it is valid. Otherwise, the input is invalid.

```python
class Words2Numbers:
    def __init__(self):
        self.numwords = {}
        units = [
            "zero", "one", "two", "three", "four", "five", "six", "seven", "eight",
            "nine", "ten", "eleven", "twelve", "thirteen", "fourteen", "fifteen",
            "sixteen", "seventeen", "eighteen", "nineteen"
        ]
        tens = [
            "", "", "twenty", "thirty", "forty", "fifty", "sixty", "seventy", "eighty", "ninety"
        ]
        scales = ["hundred", "thousand", "million", "billion", "trillion"]
        
        self.numwords["and"] = (1, 0)
        for idx, word in enumerate(units):
            self.numwords[word] = (1, idx)
        for idx, word in enumerate(tens):
            if word:
                self.numwords[word] = (1, idx * 10)
        for idx, word in enumerate(scales):
            scale = 10 ** (idx * 3) if idx != 0 else 100
            self.numwords[word] = (scale, 0)
        
        self.ordinalWords = {
            "first": 1,
            "second": 2,
            "third": 3,
            "fifth": 5,
            "eighth": 8,
            "ninth": 9,
            "twelfth": 12
        }
        
        self.ordinalEndings = [
            ("ieth", "y"),
            ("th", "")
        ]
    
    def text2int(self, textnum):
        textnum = textnum.replace("-", " ")
        current = 0
        result = 0
        curstring = []
        onnumber = False
        
        words = textnum.split()
        for word in words:
            if word in self.ordinalWords:
                increment = self.ordinalWords[word]
                current = current * 1 + increment
                onnumber = True
            else:
                base_word = word
                for ending, replacement in self.ordinalEndings:
                    if word.endswith(ending):
                        base_word = word[:-len(ending)] + replacement
                        break
                
                if base_word not in self.numwords:
                    if onnumber:
                        curstring.append(str(result + current))
                        result = current = 0
                        onnumber = False
                    curstring.append(word)
                else:
                    scale, increment = self.numwords[base_word]
                    current = current * scale + increment
                    if scale >= 100:
                        result += current
                        current = 0
                    onnumber = True
        
        if onnumber:
            curstring.append(str(result + current))
        
        return " ".join(curstring)
    
    def isValidInput(self, textnum):
        textnum = textnum.replace("-", " ")
        words = textnum.split()
        for word in words:
            if word in self.ordinalWords:
                continue
            base_word = word
            for ending, replacement in self.ordinalEndings:
                if word.endswith(ending):
                    base_word = word[:-len(ending)] + replacement
                    break
            if base_word not in self.numwords:
                return False
        return True
```

  - The input string is split into words. Each word is processed to check if it's an ordinal or can be converted to a cardinal form.
  - Valid number words are processed to accumulate numerical values. When a scale word (like "thousand") is encountered, the accumulated value is added to the result.
  - Non-number words are added to the result string directly, with any preceding numerical value converted to a string.
