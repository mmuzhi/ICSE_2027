package org.example;

import java.util.*;
import java.util.stream.Collectors;

public class DataStatistics {

    public double mean(int[] data) {
        double sum = Arrays.stream(data).sum();
        return Math.round((sum / data.length) * 100.0) / 100.0;
    }

    public double median(int[] data) {
        int[] sortedData = Arrays.copyOf(data, data.length);
        Arrays.sort(sortedData);
        int n = sortedData.length;

        if (n % 2 == 0) {
            int middle = n / 2;
            return Math.round(((sortedData[middle - 1] + sortedData[middle]) / 2.0) * 100.0) / 100.0;
        } else {
            int middle = n / 2;
            return sortedData[middle];
        }
    }

    public List<Integer> mode(int[] data) {
        Map<Integer, Long> frequencyMap = Arrays.stream(data)
                .boxed()
                .collect(Collectors.groupingBy(e -> e, Collectors.counting()));

        long maxCount = Collections.max(frequencyMap.values());

        return frequencyMap.entrySet()
                .stream()
                .filter(entry -> entry.getValue() == maxCount)
                .map(Map.Entry::getKey)
                .sorted()
                .collect(Collectors.toList());
    }
}

=== RAW ANSWER ===

To solve this problem, we need to translate a Java class that computes statistical measures (mean, median, and mode) of an integer array into C++ while maintaining identical behavior. The solution involves handling edge cases, rounding to two decimal places, and ensuring the same outputs and exceptions as the original Java code.

### Approach
1. **Mean Calculation**: 
   - Sum all elements in the array.
   - Divide the sum by the number of elements to get the average.
   - Round the result to two decimal places by multiplying by 100, rounding to the nearest integer, and then dividing by 100.
   - Handle empty arrays by returning 0.0, mimicking Java's behavior.

2. **Median Calculation**:
   - Sort a copy of the array.
   - For even-sized arrays, the median is the average of the two middle elements, rounded to two decimal places.
   - For odd-sized arrays, the median is the middle element.
   - Handle empty arrays by throwing an `std::out_of_range` exception to mimic Java's `ArrayIndexOutOfBoundsException`.

3. **Mode Calculation**:
   - Count the frequency of each element using a map.
   - Find the maximum frequency.
   - Collect all elements with this maximum frequency into a vector, sort it, and return.
   - Handle empty arrays by throwing an `std::runtime_error` to mimic Java's `NoSuchElementException`.

### Solution Code
```cpp
#include <vector>
#include <algorithm>
#include <map>
#include <cmath>
#include <stdexcept>

class DataStatistics {
public:
    double mean(const std::vector<int>& data) {
        if (data.empty()) {
            return 0.0;
        }
        double sum = 0.0;
        for (int num : data) {
            sum += num;
        }
        double avg = sum / static_cast<double>(data.size());
        double temp = avg * 100.0;
        temp = std::round(temp);
        return temp / 100.0;
    }

    double median(std::vector<int> data) {
        if (data.empty()) {
            throw std::out_of_range("Array index out of bounds");
        }
        std::sort(data.begin(), data.end());
        size_t n = data.size();
        if (n % 2 == 0) {
            size_t mid = n / 2;
            double midValue = (static_cast<double>(data[mid-1]) + data[mid]) / 2.0;
            double temp = midValue * 100.0;
            temp = std::round(temp);
            return temp / 100.0;
        } else {
            return static_cast<double>(data[n/2]);
        }
    }

    std::vector<int> mode(const std::vector<int>& data) {
        if (data.empty()) {
            throw std::runtime_error("No such element");
        }
        std::map<int, long> frequencyMap;
        for (int num : data) {
            frequencyMap[num]++;
        }

        long maxCount = 0;
        for (const auto& pair : frequencyMap) {
            if (pair.second > maxCount) {
                maxCount = pair.second;
            }
        }

        std::vector<int> modes;
        for (const auto& pair : frequencyMap) {
            if (pair.second == maxCount) {
                modes.push_back(pair.first);
            }
        }

        std::sort(modes.begin(), modes.end());
        return modes;
    }
};
```

### Explanation
- **Mean Calculation**: The mean is computed by summing all elements and dividing by the count. For empty arrays, it returns 0.0. The result is rounded to two decimal places using multiplication, rounding, and division by 100.0.
- **Median Calculation**: The array is copied and sorted. For even-sized arrays, the average of the two middle elements is computed and rounded to two decimal places. For odd-sized arrays, the middle element is returned. Empty arrays throw an `std::out_of_range` exception.
- **Mode Calculation**: Frequencies of elements are counted using a map. The maximum frequency is found, and elements with this frequency are collected into a vector, sorted, and returned. Empty arrays throw an `std::runtime_error` exception.

This approach ensures the C++ implementation behaves identically to the original Java code, handling inputs, outputs, edge cases, and exceptions in the same manner.

            }
        }

        std::sort(modes.begin(), modes.end());
        return modes;
    }
};
