package org.example;

public class BinaryDataProcessor {
    private String binaryString;

    public BinaryDataProcessor(String binaryString) {
        this.binaryString = binaryString;
        cleanNonBinaryChars();
    }

    public void cleanNonBinaryChars() {
        this.binaryString = this.binaryString.replaceAll("[^01]", "");
    }

    public BinaryInfo calculateBinaryInfo() {
        int zeroesCount = this.binaryString.length() - this.binaryString.replace("0", "").length();
        int onesCount = this.binaryString.length() - this.binaryString.replace("1", "").length();
        int totalLength = this.binaryString.length();

        double zeroesPercentage = (double) zeroesCount / totalLength;
        double onesPercentage = (double) onesCount / totalLength;

        return new BinaryInfo(zeroesPercentage, onesPercentage, totalLength);
    }

    public String convertToAscii() {
        StringBuilder asciiString = new StringBuilder();
        for (int i = 0; i < this.binaryString.length(); i += 8) {
            String byteString = this.binaryString.substring(i, i + 8);
            int decimal = Integer.parseInt(byteString, 2);
            asciiString.append((char) decimal);
        }
        return asciiString.toString();
    }

    public String convertToUtf8() {
        StringBuilder utf8String = new StringBuilder();
        for (int i = 0; i < this.binaryString.length(); i += 8) {
            String byteString = this.binaryString.substring(i, i + 8);
            int decimal = Integer.parseInt(byteString, 2);
            utf8String.append((char) decimal);
        }
        return utf8String.toString();
    }

    public String getBinaryString() {
        return this.binaryString;
    }

    public static void main(String[] args) {
        BinaryDataProcessor bdp = new BinaryDataProcessor("0110100001100101011011000110110001101111");
        System.out.println(bdp.getBinaryString());
        System.out.println(bdp.calculateBinaryInfo());
        System.out.println(bdp.convertToAscii());
        System.out.println(bdp.convertToUtf8());
    }
}

The Java program includes functionalities to clean non-binary characters from a binary string, calculate binary information (such as counts and percentages of zeros and ones), and convert the binary string into ASCII and UTF-8 strings.

### Approach
1. **Cleaning Non-Binary Characters**: The input binary string may contain non-binary characters (other than '0' and '1'). These characters are removed using the `cleanNonBinaryChars` method, which employs a lambda function with `std::remove_if` to filter out invalid characters.
2. **Calculating Binary Information**: The `calculateBinaryInfo` method computes the number of zeros and ones in the cleaned binary string using `std::count`, then calculates their percentages relative to the total length of the string. This information is encapsulated in a `BinaryInfo` object.
3. **Conversion to ASCII and UTF-8**: The binary string is processed in 8-bit chunks. Each chunk is converted to its corresponding integer value and then cast to a character to form the output string. Both `convertToAscii` and `convertToUtf8` methods perform the same conversion since the character handling in C++ for ASCII and UTF-8 is similar at this level.
4. **Handling Edge Cases**: The conversion methods ensure that the binary string length is a multiple of 8; otherwise, an `std::out_of_range` exception is thrown to mimic Java's substring behavior when indices are out of bounds.

#include <string>
#include <algorithm>
#include <vector>
#include <cstdio>
#include <stdexcept>

class BinaryInfo {
private:
    double zeroes;
    double ones;
    int bitLength;

public:
    BinaryInfo(double zeroes, double ones, int bitLength) : zeroes(zeroes), ones(ones), bitLength(bitLength) {}

    double getZeroes() const { return zeroes; }
    double getOnes() const { return ones; }
    int getBitLength() const { return bitLength; }

    std::string toString() const {

        snprintf(buffer, sizeof(buffer), "{Zeroes: %.3f, Ones: %.3f, Bit length: %d}", zeroes, ones, bitLength);
        return std::string(buffer);
    }
};

class BinaryDataProcessor {
private:
    std::string binaryString;

    void cleanNonBinaryChars() {
        auto it = std::remove_if(binaryString.begin(), binaryString.end(),
            [](char c) { return c != '0' && c != '1'; });
        binaryString.erase(it, binaryString.end());
    }

public:
    BinaryDataProcessor(std::string binaryString) : binaryString(std::move(binaryString)) {
        cleanNonBinaryChars();
    }

    BinaryInfo calculateBinaryInfo() {
        int totalLength = binaryString.length();
        int zeroesCount = std::count(binaryString.begin(), binaryString.end(), '0');
        int onesCount = totalLength - zeroesCount;

        double zeroesPercentage = static_cast<double>(zeroesCount) / totalLength;
        double onesPercentage = static_cast<double>(onesCount) / totalLength;

        return BinaryInfo(zeroesPercentage, onesPercentage, totalLength);
    }

    std::string convertToAscii() {
        std::string asciiString;
        for (size_t i = 0; i < binaryString.length(); i += 8) {
            if (i + 8 > binaryString.length()) {
                throw std::out_of_range("Index out of bounds");
            }
            std::string byteString = binaryString.substr(i, 8);
            int decimal = std::stoi(byteString, nullptr, 2);
            asciiString.push_back(static_cast<char>(decimal));
        }
        return asciiString;
    }

    std::string convertToUtf8() {
        std::string utf8String;
        for (size_t i = 0; i < binaryString.length(); i += 8) {
            if (i + 8 > binaryString.length()) {
                throw std::out_of_range("Index out of bounds");
            }
            std::string byteString = binaryString.substr(i, 8);
            int decimal = std::stoi(byteString, nullptr, 2);
            utf8String.push_back(static_cast<char>(decimal));
        }
        return utf8String;
    }

    std::string getBinaryString() const {
        return binaryString;
    }
};
