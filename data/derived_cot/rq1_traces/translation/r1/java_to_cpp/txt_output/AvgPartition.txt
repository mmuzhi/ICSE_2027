package org.example;

import java.util.ArrayList;
import java.util.List;

public class AvgPartition {
    private List<Integer> lst;
    private int limit;

    public AvgPartition(List<Integer> lst, int limit) {
        this.lst = lst;
        this.limit = limit;
    }

    public int[] setNum() {
        int size = lst.size() / limit;
        int remainder = lst.size() % limit;
        return new int[]{size, remainder};
    }

    public List<Integer> get(int index) {
        int[] nums = setNum();
        int size = nums[0];
        int remainder = nums[1];
        int start = index * size + Math.min(index, remainder);
        int end = start + size;
        if (index + 1 <= remainder) {
            end += 1;
        }
        return lst.subList(start, end);
    }

    public static void main(String[] args) {
        List<Integer> lst = new ArrayList<>();
        lst.add(1);
        lst.add(2);
        lst.add(3);
        lst.add(4);
        AvgPartition a = new AvgPartition(lst, 2);
        System.out.println(a.get(0));
        System.out.println(a.get(1));
    }
}

        #include <iostream>
        #include <algorithm> // for std::min
        #include <stdexcept>

        class AvgPartition {
        private:
            std::vector<int>& lst;
            int limit;

        public:
            AvgPartition(std::vector<int>& lst, int limit) : lst(lst), limit(limit) {}

            std::vector<int> get(int index) {
                if (index < 0 || index >= limit) {
                    throw std::out_of_range("index must be in range [0, " + std::to_string(limit-1) + "]");
                }

                int size = lst.size() / limit;
                int remainder = lst.size() % limit;

                int start = index * size + std::min(index, remainder);
                int end = start + size;
                if (index < remainder) {
                    end++;
                }

                // Check bounds: start and end must be within [0, lst.size()] and start<=end.
                if (start > static_cast<int>(lst.size()) || end > static_cast<int>(lst.size()) || start < 0) {
                    // We know start is nonnegative by computation, but check anyway.
                    throw std::out_of_range("computed indices are out of range");
                }

                // Create a new vector from the range [start, end)
                return std::vector<int>(lst.begin() + start, lst.begin() + end);
            }
        };
