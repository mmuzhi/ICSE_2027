package org.example;

public class DecryptionUtils {
    private String key;

    public DecryptionUtils(String key) {
        this.key = key;
    }

    public String caesarDecipher(String ciphertext, int shift) {
        StringBuilder plaintext = new StringBuilder();
        shift = shift % 26;
        for (char ch : ciphertext.toCharArray()) {
            if (Character.isLetter(ch)) {
                int asciiOffset = Character.isUpperCase(ch) ? 'A' : 'a';
                char shiftedChar = (char) ((ch - asciiOffset - shift + 26) % 26 + asciiOffset);
                plaintext.append(shiftedChar);
            }
            else {
                plaintext.append(ch);
            }
        }
        return plaintext.toString();
    }

    public String vigenereDecipher(String ciphertext) {
        StringBuilder decryptedText = new StringBuilder();
        int keyIndex = 0;
        for (char ch : ciphertext.toCharArray()) {
            if (Character.isLetter(ch)) {
                int shift = Character.toLowerCase(key.charAt(keyIndex % key.length())) - 'a';
                char base = Character.isUpperCase(ch) ? 'A' : 'a';
                char decryptedChar = (char) ((ch - base - shift + 26) % 26 + base);
                decryptedText.append(decryptedChar);
                keyIndex++;
            }
            else {
                decryptedText.append(ch);
            }
        }
        return decryptedText.toString();
    }

    public String railFenceDecipher(String encryptedText, int rails) {
        char[][] fence = new char[rails][encryptedText.length()];
        for (int i = 0; i < rails; i++) {
            for (int j = 0; j < encryptedText.length(); j++) {
                fence[i][j] = '\0';
            }
        }

        int direction = -1;
        int row = 0, col = 0;

        for (int i = 0; i < encryptedText.length(); i++) {
            if (row == 0 || row == rails - 1) {
                direction = -direction;
            }
            fence[row][col++] = '*';
            row += direction;
        }

        int index = 0;
        for (int i = 0; i < rails; i++) {
            for (int j = 0; j < encryptedText.length(); j++) {
                if (fence[i][j] == '*') {
                    fence[i][j] = encryptedText.charAt(index++);
                }
            }
        }

        StringBuilder plainText = new StringBuilder();
        direction = -1;
        row = 0;
        col = 0;

        for (int i = 0; i < encryptedText.length(); i++) {
            if (row == 0 || row == rails - 1) {
                direction = -direction;
            }
            plainText.append(fence[row][col++]);
            row += direction;
        }

        return plainText.toString();
    }
}

To solve this problem, we need to translate Java code for decryption utilities into C++ while maintaining identical behavior. The Java code includes three decryption methods: Caesar cipher, Vigenère cipher, and Rail Fence cipher. Each method must be translated carefully to ensure the same inputs produce the same outputs, including handling edge cases and side effects.

### Approach
1. **Class Structure**: The `DecryptionUtils` class in C++ mirrors the Java class, with a private `key` member and a constructor to initialize it.
2. **Caesar Cipher Decryption**: For each character in the ciphertext:
   - If it's a letter, shift it backward by the specified amount (mod 26), maintaining case sensitivity.
   - Non-letter characters are appended unchanged.
3. **Vigenère Cipher Decryption**: Uses a key to determine the shift for each character:
   - The key cycles to match the length of the ciphertext.
   - Each letter in the ciphertext is shifted backward by the corresponding key character's position in the alphabet (a=0, b=1, etc.).
4. **Rail Fence Decryption**: Decrypts by reconstructing the Rail Fence pattern:
   - Initialize a 2D grid (rails × length of ciphertext) with placeholder characters.
   - Mark positions in the grid where characters should be placed using a zigzag pattern.
   - Fill marked positions with characters from the ciphertext in order.
   - Read characters back in the same zigzag pattern to form the plaintext.

#include <string>
#include <cctype>

class DecryptionUtils {
private:
    std::string key;

public:
    DecryptionUtils(const std::string& key) : key(key) {}

    std::string caesarDecipher(const std::string& ciphertext, int shift) {
        std::string plaintext;
        shift = shift % 26;
        if (shift < 0) {
            shift += 26;
        }
        for (char ch : ciphertext) {
            if (std::isalpha(ch)) {
                char base = std::isupper(ch) ? 'A' : 'a';
                char shiftedChar = (ch - base - shift + 26) % 26 + base;
                plaintext += shiftedChar;
            } else {
                plaintext += ch;
            }
        }
        return plaintext;
    }

    std::string vigenereDecipher(const std::string& ciphertext) {
        std::string decryptedText;
        int keyIndex = 0;
        for (char ch : ciphertext) {
            if (std::isalpha(ch)) {
                char keyChar = key[keyIndex % key.length()];
                if (std::isupper(keyChar)) {
                    keyChar = std::tolower(keyChar);
                }
                int shift = keyChar - 'a';
                char base = std::isupper(ch) ? 'A' : 'a';
                char decryptedChar = (ch - base - shift + 26) % 26 + base;
                decryptedText += decryptedChar;
                keyIndex++;
            } else {
                decryptedText += ch;
            }
        }
        return decryptedText;
    }

    std::string railFenceDecipher(const std::string& encryptedText, int rails) {
        if (rails <= 0) {
            return encryptedText;
        }
        int len = encryptedText.length();
        std::vector<std::vector<char>> fence(rails, std::vector<char>(len, '\0'));

        int direction = -1;
        int row = 0, col = 0;

        for (int i = 0; i < len; i++) {
            if (row == 0 || row == rails - 1) {
                direction = -direction;
            }
            fence[row][col] = '*';
            col++;
            row += direction;
        }

        int index = 0;
        for (int i = 0; i < rails; i++) {
            for (int j = 0; j < len; j++) {
                if (fence[i][j] == '*') {
                    fence[i][j] = encryptedText[index];
                    index++;
                }
            }
        }

        std::string plainText;
        direction = -1;
        row = 0;
        col = 0;

        for (int i = 0; i < len; i++) {
            if (row == 0 || row == rails - 1) {
                direction = -direction;
            }
            plainText += fence[row][col];
            col++;
            row += direction;
        }

        return plainText;
    }
};

### Explanation
1. **Caesar Cipher**:
   - The shift is normalized to a positive value modulo 26.
   - Each letter is shifted backward by the normalized shift value, wrapping around the alphabet if necessary.
   - Non-letter characters remain unchanged.

2. **Vigenère Cipher**:
   - The key is cycled to match the ciphertext length.
   - Each key character is converted to lowercase to determine the shift value (a=0, b=1, ..., z=25).
   - Each ciphertext character is shifted backward by its corresponding key character's shift value.

3. **Rail Fence Cipher**:
   - A grid is created with `rails` rows and columns equal to the ciphertext length.
   - The zigzag pattern is marked in the grid.
   - Ciphertext characters are placed into the marked positions in order.
   - The plaintext is reconstructed by reading the grid in the same zigzag pattern.

This approach ensures that the C++ implementation behaves identically to the original Java code, handling all edge cases and maintaining the same side effects.
