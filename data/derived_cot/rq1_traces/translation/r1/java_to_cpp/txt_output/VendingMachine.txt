package org.example;

import java.util.LinkedHashMap;
import java.util.Map;

public class VendingMachine {
    private Map<String, Product> inventory;
    private double balance;

    public VendingMachine() {
        this.inventory = new LinkedHashMap<>();
        this.balance = 0;
    }

    public void addItem(String itemName, double price, int quantity) {
        if (!restockItem(itemName, quantity)) {
            inventory.put(itemName, new Product(price, quantity));
        }
    }

    public double insertCoin(double amount) {
        balance += amount;
        return balance;
    }

    public Object purchaseItem(String itemName) {
        if (inventory.containsKey(itemName)) {
            Product item = inventory.get(itemName);
            if (item.getQuantity() > 0 && balance >= item.getPrice()) {
                balance -= item.getPrice();
                item.setQuantity(item.getQuantity() - 1);
                return balance;
            } else {
                return false;
            }
        } else {
            return false;
        }
    }

    public boolean restockItem(String itemName, int quantity) {
        if (inventory.containsKey(itemName)) {
            Product item = inventory.get(itemName);
            item.setQuantity(item.getQuantity() + quantity);
            return true;
        } else {
            return false;
        }
    }

    public Object displayItems() {
        if (inventory.isEmpty()) {
            return false;
        } else {
            StringBuilder items = new StringBuilder();
            for (Map.Entry<String, Product> entry : inventory.entrySet()) {
                Product item = entry.getValue();
                items.append(String.format("%s - $%.2f [%d]\n", entry.getKey(), item.getPrice(), item.getQuantity()));
            }
            return items.toString().trim();
        }
    }


    public Map<String, Product> getInventory() {
        return inventory;
    }

    public void setInventory(Map<String, Product> inventory) {
        this.inventory = inventory;
    }

    public double getBalance() {
        return balance;
    }

    public void setBalance(double balance) {
        this.balance = balance;
    }

    public static class Product {
        private double price;
        private int quantity;

        public Product(double price, int quantity) {
            this.price = price;
            this.quantity = quantity;
        }

        public double getPrice() {
            return price;
        }

        public void setPrice(double price) {
            this.price = price;
        }

        public int getQuantity() {
            return quantity;
        }

        public void setQuantity(int quantity) {
            this.quantity = quantity;
        }
    }
}

### Approach
1. **Preserving Insertion Order**: Since C++ does not have a direct equivalent of Java's `LinkedHashMap`, we use a combination of a `std::vector` to maintain insertion order and a `std::unordered_map` for O(1) lookups. However, to keep the solution simple and within standard C++, we use a `std::vector` of pairs (item name and product) to maintain insertion order. This approach ensures that items are displayed in the order they were added, though lookups become O(n).

2. **Handling Return Types**: The Java methods `purchaseItem` and `displayItems` return either a primitive type (double or boolean) or an object (String). In C++, we use `std::variant` to handle these different return types seamlessly. Specifically:
   - `purchaseItem` returns a `std::variant<double, bool>`, which can be either the updated balance (double) or `false` (bool) indicating failure.

3. **Product Management**: The nested `Product` class in Java is translated directly into a nested C++ class, maintaining the same attributes and methods for price and quantity management.

4. **Inventory Management**: The `addItem`, `restockItem`, and `purchaseItem` methods iterate through the vector to find items, ensuring that insertion order is preserved. Though this results in O(n) lookups, it is acceptable for small inventories typical in vending machines.
