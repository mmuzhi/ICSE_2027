package org.example;

import java.util.HashMap;
import java.util.Map;

public class URLHandler {
    private String url;

    public URLHandler(String url) {
        this.url = url;
    }

    public String getScheme() {
        int schemeEnd = url.indexOf("://");
        if (schemeEnd != -1) {
            return url.substring(0, schemeEnd);
        }
        return null;
    }

    public String getHost() {
        int schemeEnd = url.indexOf("://");
        if (schemeEnd != -1) {
            String urlWithoutScheme = url.substring(schemeEnd + 3);
            int hostEnd = urlWithoutScheme.indexOf("/");
            if (hostEnd != -1) {
                return urlWithoutScheme.substring(0, hostEnd);
            }
            return urlWithoutScheme;
        }
        return null;
    }

    public String getPath() {
        int schemeEnd = url.indexOf("://");
        if (schemeEnd != -1) {
            String urlWithoutScheme = url.substring(schemeEnd + 3);
            int hostEnd = urlWithoutScheme.indexOf("/");
            if (hostEnd != -1) {
                return urlWithoutScheme.substring(hostEnd);
            }
        }
        return null;
    }

    public Map<String, String> getQueryParams() {
        int queryStart = url.indexOf("?");
        int fragmentStart = url.indexOf("#");
        if (queryStart != -1) {
            String queryString = url.substring(queryStart + 1, fragmentStart != -1 ? fragmentStart : url.length());
            Map<String, String> params = new HashMap<>();
            if (!queryString.isEmpty()) {
                String[] paramPairs = queryString.split("&");
                for (String pair : paramPairs) {
                    String[] keyValue = pair.split("=");
                    if (keyValue.length == 2) {
                        params.put(keyValue[0], keyValue[1]);
                    }
                }
            }
            return params;
        }
        return null;
    }

    public String getFragment() {
        int fragmentStart = url.indexOf("#");
        if (fragmentStart != -1) {
            return url.substring(fragmentStart + 1);
        }
        return null;
    }
}

### Explanation
1. **Class Initialization**: The `URLHandler` constructor initializes the class with a URL string.
2. **Scheme Extraction**: `getScheme` finds "://" in the URL. If present, it returns the substring before "://"; otherwise, it returns `std::nullopt`.
3. **Host Extraction**: `getHost` first checks for a scheme. If found, it extracts the substring after "://" and then finds the first '/'. The host is the substring before the '/' or the entire substring if '/' is absent.
4. **Path Extraction**: `getPath` checks for a scheme and then finds the first '/' after "://". If found, it returns the substring from '/' to the end; otherwise, it returns `std::nullopt`.
5. **Query Parameters**: `getQueryParams` locates the query string (after '?' and before '#'). It splits the query string by '&' and each pair by '=', storing key-value pairs in a map. If '?' is absent, it returns `std::nullopt`.
6. **Fragment Extraction**: `getFragment` returns the substring after '#' if present; otherwise, it returns `std::nullopt`.
7. **Helper Function**: `split_java_style` splits a string by a delimiter, removing trailing empty tokens, mimicking Java's `split` method behavior.

This approach ensures that the C++ implementation behaves identically to the original Java code, handling all edge cases and exceptions as specified.
