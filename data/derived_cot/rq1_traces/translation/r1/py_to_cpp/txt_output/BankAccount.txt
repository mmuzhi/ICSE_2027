        """
        Return the account balance.
        """

        :param amount: int
        >>> account1 = BankAccount()
        >>> account2 = BankAccount()
        >>> account1.deposit(1000)
        >>> account1.transfer(account2, 300)
        account1.balance = 700 account2.balance = 300
        """
'''

class BankAccount:
    def __init__(self, balance=0):
        self.balance = balance

    def deposit(self, amount):
        if amount < 0:
            raise ValueError("Invalid amount")
        self.balance += amount
        return self.balance

    def withdraw(self, amount):
        if amount < 0:
            raise ValueError("Invalid amount")
        if amount > self.balance:
            raise ValueError("Insufficient balance.")
        self.balance -= amount
        return self.balance

    def view_balance(self):
        return self.balance

    def transfer(self, other_account, amount):
        self.withdraw(amount)
        other_account.deposit(amount)

#include <stdexcept>

class BankAccount {
private:
    int balance;

public:
    BankAccount(int initial_balance = 0) : balance(initial_balance) {}

        return balance;

    int withdraw(int amount) {

    void transfer(BankAccount& other_account, int amount) {

3. **Deposit Handling**: The `deposit` method checks if the amount is negative. If so, it throws a `std::runtime_error` with the message "Invalid amount". Otherwise, it adds the amount to the balance and returns the updated balance.
4. **Withdraw Handling**: The `withdraw` method checks for negative amounts and insufficient funds. If either condition is met, it throws a `std::runtime_error` with an appropriate message. Otherwise, it subtracts the amount from the balance and returns the updated balance.
5. **View Balance**: The `view_balance` method is a const member function that returns the current balance without modifying the account state.
