from collections import Counter


class DataStatistics:
    def mean(self, data):
        return round(sum(data) / len(data), 2)

    def median(self, data):
        sorted_data = sorted(data)
        n = len(sorted_data)
        if n % 2 == 0:
            middle = n // 2
            return round((sorted_data[middle - 1] + sorted_data[middle]) / 2, 2)
        else:
            middle = n // 2
            return sorted_data[middle]

    def mode(self, data):
        counter = Counter(data)
        mode_count = max(counter.values())
        mode = [x for x, count in counter.items() if count == mode_count]
        return mode

To solve this problem, we need to translate a Python class that calculates the mean, median, and mode of a given dataset into C++. The solution must maintain identical behavior, including handling edge cases like empty datasets and ensuring the results are accurate to two decimal places where applicable.

### Approach
1. **Mean Calculation**: Sum all elements in the dataset and divide by the number of elements, rounding the result to two decimal places.
2. **Median Calculation**: Sort the dataset. If the number of elements is even, average the two middle elements; if odd, select the middle element. The result is rounded to two decimal places.
3. **Mode Calculation**: Count the frequency of each element, preserving the order of first appearance. Elements with the highest frequency are collected and returned in the order they first appeared in the dataset.
4. **Edge Cases**: Handle empty datasets by throwing an exception for each method to replicate Python's behavior when encountering an empty list.

### Solution Code
```cpp
#include <vector>
#include <algorithm>
#include <stdexcept>
#include <unordered_map>
#include <cmath>

class DataStatistics {
public:
    double mean(const std::vector<double>& data) const {
        if (data.empty()) {
            throw std::invalid_argument("Data cannot be empty");
        }
        double sum = 0.0;
        for (double x : data) {
            sum += x;
        }
        double avg = sum / static_cast<double>(data.size());
        return std::round(avg * 100.0) / 100.0;
    }

    double median(const std::vector<double>& data) const {
        if (data.empty()) {
            throw std::invalid_argument("Data cannot be empty");
        }
        std::vector<double> sorted_data = data;
        std::sort(sorted_data.begin(), sorted_data.end());
        size_t n = sorted_data.size();
        double result;
        if (n % 2 == 0) {
            size_t mid = n / 2;
            result = (sorted_data[mid-1] + sorted_data[mid]) / 2.0;
        } else {
            result = sorted_data[n/2];
        }
        return std::round(result * 100.0) / 100.0;
    }

    std::vector<double> mode(const std::vector<double>& data) const {
        if (data.empty()) {
            throw std::invalid_argument("Data cannot be empty");
        }
        std::unordered_map<double, int> counts;
        std::vector<double> order;
        for (double x : data) {
            if (counts.find(x) == counts.end()) {
                counts[x] = 1;
                order.push_back(x);
            } else {
                counts[x]++;
            }
        }
        int max_count = 0;
        for (const auto& pair : counts) {
            if (pair.second > max_count) {
                max_count = pair.second;
            }
        }
        std::vector<double> modes;
        for (double x : order) {
            if (counts[x] == max_count) {
                modes.push_back(x);
            }
        }
        return modes;
    }
};
```

            if (counts.find(x) == counts.end()) {
                counts[x] = 1;
                order.push_back(x);
            } else {
                counts[x]++;
            }
        }
        int max_count = 0;
        for (const auto& pair : counts) {
            if (pair.second > max_count) {
                max_count = pair.second;
            }
        }
        std::vector<double> modes;
        for (double x : order) {
            if (counts[x] == max_count) {
                modes.push_back(x);
            }
        }
        return modes;
    }
};
