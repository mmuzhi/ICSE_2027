        :return:float, the cohens kappa value of the matrix

        :param testData: Input data matrix, N * k
        :param N: int, Number of samples
        :param k: int, Number of categories
        :param n: int, Number of raters

        0.20993070442195522

import numpy as np


class KappaCalculator:

    @staticmethod
    def kappa(testData, k):
        dataMat = np.mat(testData)
        P0 = 0.0
        for i in range(k):
            P0 += dataMat[i, i] * 1.0
        xsum = np.sum(dataMat, axis=1)
        ysum = np.sum(dataMat, axis=0)
        sum = np.sum(dataMat)
        Pe = float(ysum * xsum) / sum / sum
        P0 = float(P0 / sum * 1.0)
        cohens_coefficient = float((P0 - Pe) / (1 - Pe))
        return cohens_coefficient

    @staticmethod
    def fleiss_kappa(testData, N, k, n):
        dataMat = np.mat(testData, float)
        oneMat = np.ones((k, 1))
        sum = 0.0
        P0 = 0.0
        for i in range(N):
            temp = 0.0
            for j in range(k):
                sum += dataMat[i, j]
                temp += 1.0 * dataMat[i, j] ** 2
            temp -= n
            temp /= (n - 1) * n
            P0 += temp
        P0 = 1.0 * P0 / N
        ysum = np.sum(dataMat, axis=0)
        for i in range(k):
            ysum[0, i] = (ysum[0, i] / sum) ** 2
        Pe = ysum * oneMat * 1.0
        ans = (P0 - Pe) / (1 - Pe)
        return ans[0, 0]

To solve this problem, we need to translate the given Python code for calculating Cohen's kappa and Fleiss' kappa coefficients into C++. The solution involves implementing the mathematical formulas for these coefficients using standard C++ libraries without relying on external matrix operations libraries like NumPy.

### Approach
1. **Cohen's Kappa**:
   - **Problem Analysis**: Cohen's kappa measures inter-rater agreement for categorical items. It accounts for the possibility of agreement occurring by chance.
   - **Key Steps**:
     - **Calculate Total Sum**: Sum all elements in the confusion matrix.
     - **Diagonal Sum**: Sum the diagonal elements of the matrix, which represent agreements.
     - **Row and Column Sums**: Compute sums for each row and column.
     - **Probability of Agreement (P0)**: The ratio of diagonal sum to total sum.
     - **Probability of Random Agreement (Pe)**: The dot product of row sums and column sums divided by the square of the total sum.
     - **Kappa Formula**: \((P0 - Pe) / (1 - Pe)\).

2. **Fleiss' Kappa**:
   - **Problem Analysis**: Fleiss' kappa extends Cohen's kappa to multiple raters. It assesses the agreement among raters when assigning categories to items.
   - **Key Steps**:
     - **Total Sum**: Sum all elements in the matrix.
     - **Column Sums**: Compute sums for each category across all items.
     - **Row-wise Squared Sum**: For each item, compute the sum of squared category counts.
     - **Probability of Agreement (P0)**: Average of \(( \text{squared sum} - n ) / (n(n-1))\) across all items.
     - **Probability of Random Agreement (Pe)**: Sum of squares of the proportions of each category.
     - **Kappa Formula**: \((P0 - Pe) / (1 - Pe)\).

```cpp
#include <vector>
#include <iostream>

class KappaCalculator {
public:
    static double kappa(std::vector<std::vector<int>> const& testData, int k) {
        double total_sum = 0.0;
        double diag_sum = 0.0;
        std::vector<double> row_sums(k, 0.0);
        std::vector<double> col_sums(k, 0.0);

        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) {
                int value = testData[i][j];
                total_sum += value;
                row_sums[i] += value;
                col_sums[j] += value;
                if (i == j) {
                    diag_sum += value;
                }
            }
        }

        double P0 = diag_sum / total_sum;
        double dot_product = 0.0;
        for (int i = 0; i < k; ++i) {
            dot_product += row_sums[i] * col_sums[i];
        }
        double Pe = dot_product / (total_sum * total_sum);

        if (Pe == 1.0 && P0 == 1.0) {
            return 0.0 / 0.0; // NaN
        }

        return (P0 - Pe) / (1 - Pe);
    }

    static double fleiss_kappa(std::vector<std::vector<int>> const& testData, int N, int k, int n) {
        double total_sum = 0.0;
        std::vector<double> col_sums(k, 0.0);
        double P0 = 0.0;

        for (int i = 0; i < N; ++i) {
            double row_sum_sq = 0.0;
            for (int j = 0; j < k; ++j) {
                int value = testData[i][j];
                total_sum += value;
                col_sums[j] += value;
                row_sum_sq += static_cast<double>(value) * value;
            }
            double temp = (row_sum_sq - n) / (n * (n - 1));
            P0 += temp;
        }
        P0 /= N;

        double Pe = 0.0;
        for (int j = 0; j < k; ++j) {
            double p_j = col_sums[j] / total_sum;
            Pe += p_j * p_j;
        }

        if (Pe == 1.0 && P0 == 1.0) {
            return 0.0 / 0.0; // NaN
        }

        return (P0 - Pe) / (1 - Pe);
    }
};
```

1. **Cohen's Kappa**:
   - **Total Sum**: Sum of all elements in the confusion matrix.
   - **Diagonal Sum**: Sum of elements where row index equals column index (agreements).
   - **Row and Column Sums**: Computed for each row and column to understand the distribution.
   - **P0**: Proportion of observed agreements (diagonal sum divided by total sum).
   - **Pe**: Probability of random agreement, computed as the dot product of row and column sums divided by the square of the total sum.
   - **Kappa**: Adjusts P0 by Pe to account for chance agreement.

2. **Fleiss' Kappa**:
   - **Total Sum**: Sum of all category counts across all items.
   - **Column Sums**: Total counts for each category across all items.
   - **Row-wise Squared Sum**: For each item, sum of squared category counts.
   - **P0**: Average proportion of agreement per item, adjusted for the number of raters.
   - **Pe**: Sum of squared proportions of each category, representing chance agreement.
   - **Kappa**: Adjusts P0 by Pe to account for chance agreement across multiple raters.

This implementation efficiently computes both kappa coefficients using basic C++ constructs, ensuring the same behavior as the original Python code.

=== TRANSLATED CODE (cpp) ===
#include <vector>
#include <iostream>

class KappaCalculator {
public:
    static double kappa(std::vector<std::vector<int>> const& testData, int k) {
        double total_sum = 0.0;
        double diag_sum = 0.0;
        std::vector<double> row_sums(k, 0.0);
        std::vector<double> col_sums(k, 0.0);

        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) {
                int value = testData[i][j];
                total_sum += value;
                row_sums[i] += value;
                col_sums[j] += value;
                if (i == j) {
                    diag_sum += value;
                }
            }
        }

        double P0 = diag_sum / total_sum;
        double dot_product = 0.0;
        for (int i = 0; i < k; ++i) {
            dot_product += row_sums[i] * col_sums[i];
        }
        double Pe = dot_product / (total_sum * total_sum);

        if (Pe == 1.0 && P0 == 1.0) {
            return 0.0 / 0.0; // NaN
        }

        return (P0 - Pe) / (1 - Pe);
    }

    static double fleiss_kappa(std::vector<std::vector<int>> const& testData, int N, int k, int n) {
        double total_sum = 0.0;
        std::vector<double> col_sums(k, 0.0);
        double P0 = 0.0;

        for (int i = 0; i < N; ++i) {
            double row_sum_sq = 0.0;
            for (int j = 0; j < k; ++j) {
                int value = testData[i][j];
                total_sum += value;
                col_sums[j] += value;
                row_sum_sq += static_cast<double>(value) * value;
            }
            double temp = (row_sum_sq - n) / (n * (n - 1));
            P0 += temp;
        }
        P0 /= N;

        double Pe = 0.0;
        for (int j = 0; j < k; ++j) {
            double p_j = col_sums[j] / total_sum;
            Pe += p_j * p_j;
        }

        if (Pe == 1.0 && P0 == 1.0) {
            return 0.0 / 0.0; // NaN
        }

        return (P0 - Pe) / (1 - Pe);
    }
};
