            self.mode = 'cool'

            else:
                self.auto_set_mode()
                return False

            if self.mode == 'heat':
                return True
            else:
                self.auto_set_mode()
                return False

        use_time = 0
        if self.mode == 'heat':

                self.current_temperature += 1
                use_time += 1
        else:
            while(self.current_temperature > self.target_temperature):

                use_time += 1
