        search_info["total_pages"] = num_pages;
        search_info["results"] = results;

        return search_info;
    }
};

### Explanation
1. **Constructor**: Initializes the data vector, page size, total items, and calculates the total pages using integer arithmetic to mimic Python's ceiling division.
2. **get_page**: Computes the start and end indices for the requested page and returns the corresponding subvector. Returns an empty vector for invalid page numbers.
3. **get_page_info**: Constructs a map with page details. For invalid page numbers, it returns an empty map. The map uses `std::variant` to hold different types of values (int, bool, string, vector).
4. **search**: Converts each item to a string and checks if the keyword is a substring. Results are collected into a vector, and metadata (total results, pages) is computed. The results and metadata are returned in a map.
5. **String Conversion**: Uses `std::ostringstream` to convert each item to a string, ensuring compatibility with custom types if they overload the `<<` operator.

This approach ensures that the C++ implementation behaves identically to the Python original, handling pagination and search functionalities efficiently while leveraging C++17 features for type-safe variant usage.

=== TRANSLATED CODE (cpp) ===
#include <vector>
#include <map>
#include <variant>
#include <sstream>
#include <algorithm>
#include <string>

template <typename T>
class PageUtil {
public:
    using ValueType = std::variant<int, bool, std::string, std::vector<T>>;

private:
    std::vector<T> data;
    int page_size;
    int total_items;
    int total_pages;

public:
    PageUtil(const std::vector<T>& data, int page_size)
        : data(data), page_size(page_size), total_items(static_cast<int>(data.size())),
          total_pages((total_items + page_size - 1) / page_size) {}

    std::vector<T> get_page(int page_number) const {
        if (page_number < 1 || page_number > total_pages) {
            return std::vector<T>();
        }
        int start_index = (page_number - 1) * page_size;
        int end_index = std::min(start_index + page_size, total_items);
        return std::vector<T>(data.begin() + start_index, data.begin() + end_index);
    }

    std::map<std::string, ValueType> get_page_info(int page_number) const {
        std::map<std::string, ValueType> page_info;
        if (page_number < 1 || page_number > total_pages) {
            return page_info;
        }

        int start_index = (page_number - 1) * page_size;
        int end_index = std::min(start_index + page_size, total_items);
        std::vector<T> page_data(data.begin() + start_index, data.begin() + end_index);

        page_info["current_page"] = page_number;
        page_info["per_page"] = page_size;
        page_info["total_pages"] = total_pages;
        page_info["total_items"] = total_items;
        page_info["has_previous"] = (page_number > 1);
        page_info["has_next"] = (page_number < total_pages);
        page_info["data"] = page_data;

        return page_info;
    }

    std::map<std::string, ValueType> search(const std::string& keyword) const {
        std::vector<T> results;
        for (const T& item : data) {
            std::ostringstream oss;
            oss << item;
            if (oss.str().find(keyword) != std::string::npos) {
                results.push_back(item);
            }
        }

        int num_results = static_cast<int>(results.size());
        int num_pages = (num_results + page_size - 1) / page_size;

        std::map<std::string, ValueType> search_info;
        search_info["keyword"] = keyword;
        search_info["total_results"] = num_results;
        search_info["total_pages"] = num_pages;
        search_info["results"] = results;

        return search_info;
    }
};
