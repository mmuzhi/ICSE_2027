        b *= i;

    double a = 0.0; 
    x = x / 180.0 * PI;

            a += term;
        }
        else {
            a -= term;
        }

    double g = 0.0;
    double t = x;
    int n = 1;

        g += t;
        ++n;
        t = -t * x * x / ((2 * n - 1) * (2 * n - 2));

}

class TriCalculator:

            return 1

    def taylor(self, x, n):
        x_rad = x * self.PI / 180.0
        result = 0.0
        term = x_rad
        for k in range(n):
            result += term
            term *= -x_rad * x_rad / ((2 * k + 1) * (2 * k + 2))
