package org.example;

import java.util.ArrayList;
import java.util.List;

public class Interpolation {

    public static List<Double> interpolate1d(List<Double> x, List<Double> y, List<Double> xInterp) {
        List<Double> yInterp = new ArrayList<>();
        for (Double xi : xInterp) {
            for (int i = 0; i < x.size() - 1; i++) {
                if (x.get(i) <= xi && xi <= x.get(i + 1)) {
                    double yi = y.get(i) + (y.get(i + 1) - y.get(i)) * (xi - x.get(i)) / (x.get(i + 1) - x.get(i));
                    yInterp.add(yi);
                    break;
                }
            }
        }
        return yInterp;
    }

    public static List<Double> interpolate2d(List<Double> x, List<Double> y, List<List<Double>> z, List<Double> xInterp, List<Double> yInterp) {
        List<Double> zInterp = new ArrayList<>();
        for (int k = 0; k < xInterp.size(); k++) {
            Double xi = xInterp.get(k);
            Double yi = yInterp.get(k);
            for (int i = 0; i < x.size() - 1; i++) {
                if (x.get(i) <= xi && xi <= x.get(i + 1)) {
                    for (int j = 0; j < y.size() - 1; j++) {
                        if (y.get(j) <= yi && yi <= y.get(j + 1)) {
                            double z00 = z.get(i).get(j);
                            double z01 = z.get(i).get(j + 1);
                            double z10 = z.get(i + 1).get(j);
                            double z11 = z.get(i + 1).get(j + 1);
                            double zi = (z00 * (x.get(i + 1) - xi) * (y.get(j + 1) - yi) +
                                    z10 * (xi - x.get(i)) * (y.get(j + 1) - yi) +
                                    z01 * (x.get(i + 1) - xi) * (yi - y.get(j)) +
                                    z11 * (xi - x.get(i)) * (yi - y.get(j))) /
                                    ((x.get(i + 1) - x.get(i)) * (y.get(j + 1) - y.get(j)));
                            zInterp.add(zi);
                            break;
                        }
                    }
                    break;
                }
            }
        }
        return zInterp;
    }
}

To solve this problem, we need to translate two Java methods that perform linear interpolation in one dimension (1D) and bilinear interpolation in two dimensions (2D) into equivalent Python code. The goal is to maintain the same behavior, inputs, outputs, side effects, and edge cases as the original Java code.

### Approach
1. **Problem Analysis**: The Java methods `interpolate1d` and `interpolate2d` perform interpolation on lists of values. The 1D method finds the nearest interval containing each input value and computes a linear interpolation. The 2D method extends this by finding the nearest rectangle in the x-y plane and performing bilinear interpolation using the four corner values.

2. **Key Insights**: 
   - **1D Interpolation**: For each input value, find the interval where it lies and compute the interpolated value using linear interpolation between the interval's endpoints.
   - **2D Interpolation**: For each pair of input values (xi, yi), find the corresponding rectangle in the x-y plane. Use the four corner values of this rectangle to compute the interpolated value using bilinear interpolation.

3. **Algorithm Selection**: Both methods use nested loops to find the appropriate intervals. The 1D method uses a single loop, while the 2D method uses nested loops (for x and y dimensions). The interpolation formulas are directly translated from Java to Python.

4. **Complexity Analysis**: The time complexity for both methods is O(n*m) for the 2D method, where n and m are the sizes of the input lists. The 1D method is O(n*m) as well, where n is the size of the input list and m is the size of the interpolation list.

def interpolate1d(x, y, xInterp):
    yInterp = []
    for xi in xInterp:
        for i in range(len(x) - 1):
            if x[i] <= xi <= x[i+1]:
                yi = y[i] + (y[i+1] - y[i]) * (xi - x[i]) / (x[i+1] - x[i])
                yInterp.append(yi)
                break
    return yInterp

def interpolate2d(x, y, z, xInterp, yInterp):
    zInterp = []
    for k in range(len(xInterp)):
        xi = xInterp[k]
        yi = yInterp[k]
        for i in range(len(x) - 1):
            if x[i] <= xi <= x[i+1]:
                for j in range(len(y) - 1):
                    if y[j] <= yi <= y[j+1]:
                        z00 = z[i][j]
                        z01 = z[i][j+1]
                        z10 = z[i+1][j]
                        z11 = z[i+1][j+1]
                        denominator = (x[i+1] - x[i]) * (y[j+1] - y[j])
                        numerator = (
                            z00 * (x[i+1] - xi) * (y[j+1] - yi) +
                            z10 * (xi - x[i]) * (y[j+1] - yi) +
                            z01 * (x[i+1] - xi) * (yi - y[j]) +
                            z11 * (xi - x[i]) * (yi - y[j])
                        )
                        zi = numerator / denominator
                        zInterp.append(zi)
                        break
                break
    return zInterp
