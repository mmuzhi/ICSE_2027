package org.example;

import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Set;

public class CamelCaseMap {
    private final Map<String, Object> data = new LinkedHashMap<>();

    public Object get(String key) {
        return data.get(_convertKey(key));
    }

    public void put(String key, Object value) {
        data.put(_convertKey(key), value);
    }

    public void remove(String key) {
        data.remove(_convertKey(key));
    }

    String _convertKey(String key) {
        if (key == null) {
            return null;
        }
        return _toCamelCase(key);
    }

    static String _toCamelCase(String key) {
        String[] parts = key.split("_");
        StringBuilder camelCaseString = new StringBuilder(parts[0]);
        for (int i = 1; i < parts.length; i++) {
            camelCaseString.append(parts[i].substring(0, 1).toUpperCase()).append(parts[i].substring(1).toLowerCase());
        }
        return camelCaseString.toString();
    }

    public Set<String> keySet() {
        return data.keySet();
    }

    public int size() {
        return data.size();
    }
}

### Approach
1. **Key Conversion**: The Java code converts keys from snake_case to camelCase by splitting the key on underscores, capitalizing the first letter of each subsequent part, and joining them without underscores. This logic is implemented in Python using a helper method.
2. **Data Structure**: The underlying data structure in Java is a `LinkedHashMap`, which preserves insertion order. In Python, we use an `OrderedDict` to maintain the same behavior.
3. **Methods**: The translated class includes methods for `get`, `put`, `remove`, `keySet`, and `size`, each mirroring their Java counterparts with the same functionality.
4. **Edge Cases**: The code handles null keys by returning None, similar to Java's behavior with null keys. It also correctly processes keys with multiple underscores and empty strings.

### Solution Code
```python
import collections

class CamelCaseMap:
    def __init__(self):
        self._data = collections.OrderedDict()

    def get(self, key):
        camel_key = self._convert_key(key)
        return self._data.get(camel_key)

    def put(self, key, value):
        camel_key = self._convert_key(key)
        self._data[camel_key] = value

    def remove(self, key):
        camel_key = self._convert_key(key)
        if camel_key in self._data:
            del self._data[camel_key]

    def keySet(self):
        return set(self._data.keys())

    def size(self):
        return len(self._data)

    @staticmethod
    def _convert_key(key):
        if key is None:
            return None
        parts = key.split('_')
        if not parts:
            return key
        camel_case = parts[0]
        for part in parts[1:]:
            if part:
                camel_case += part[0].upper() + part[1:].lower()
        return camel_case
```
