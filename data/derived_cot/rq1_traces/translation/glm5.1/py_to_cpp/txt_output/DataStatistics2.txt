class DataStatistics2 {
private:
    std::vector<std::vector<double>> data;
    size_t rows;
    size_t cols;

    static double round_to_two(double value) {
        return std::round(value * 100.0) / 100.0;
    }

public:
    DataStatistics2(const std::vector<double>& input) {
        // 1D input, store as single row
        data.push_back(input);
        rows = 1;
        cols = input.size();
    }

    DataStatistics2(const std::vector<std::vector<double>>& input) {
        data = input;
        rows = input.size();
        cols = input.empty() ? 0 : input[0].size();
    }

    double get_sum() const {
        double sum = 0.0;
        for (const auto& row : data) {
            sum += std::accumulate(row.begin(), row.end(), 0.0);
        }
        return sum;
    }

    double get_min() const {
        double min_val = data[0][0];
        for (const auto& row : data) {
            for (double x : row) {
                if (x < min_val) min_val = x;
            }
        }
        return min_val;
    }

    double get_max() const {
        double max_val = data[0][0];
        for (const auto& row : data) {
            for (double x : row) {
                if (x > max_val) max_val = x;
            }
        }
        return max_val;
    }

    double get_variance() const {
        // Population variance (ddof=0, matching np.var default)
        double mean = get_sum() / (rows * cols);
        double sum_sq = 0.0;
        for (const auto& row : data) {
            for (double x : row) {
                sum_sq += (x - mean) * (x - mean);
            }
        }
        return round_to_two(sum_sq / (rows * cols));
    }

    double get_std_deviation() const {
        double mean = get_sum() / (rows * cols);
        double sum_sq = 0.0;
        for (const auto& row : data) {
            for (double x : row) {
                sum_sq += (x - mean) * (x - mean);
            }
        }
        return round_to_two(std::sqrt(sum_sq / (rows * cols)));
    }

    std::vector<std::vector<double>> get_correlation() const {
        // np.corrcoef with rowvar=False: each column is a variable
        // Number of variables = cols, number of observations = rows
        size_t n_vars = cols;
        size_t n_obs = rows;
        
        // Compute means for each column
        std::vector<double> means(n_vars, 0.0);
        for (size_t j = 0; j < n_vars; ++j) {
            for (size_t i = 0; i < n_obs; ++i) {
                means[j] += data[i][j];
            }
            means[j] /= n_obs;
        }
        
        // Compute covariance matrix (n_vars x n_vars)
        std::vector<std::vector<double>> cov(n_vars, std::vector<double>(n_vars, 0.0));
        for (size_t i = 0; i < n_vars; ++i) {
            for (size_t j = 0; j < n_vars; ++j) {
                double sum = 0.0;
                for (size_t k = 0; k < n_obs; ++k) {
                    sum += (data[k][i] - means[i]) * (data[k][j] - means[j]);
                }
                cov[i][j] = sum / n_obs;  // population covariance (ddof=0)
            }
        }
        
        // Compute correlation matrix
        std::vector<std::vector<double>> corr(n_vars, std::vector<double>(n_vars, 0.0));
        for (size_t i = 0; i < n_vars; ++i) {
            for (size_t j = 0; j < n_vars; ++j) {
                if (cov[i][i] == 0.0 || cov[j][j] == 0.0) {
                    corr[i][j] = std::nan("");
                } else {
                    corr[i][j] = cov[i][j] / std::sqrt(cov[i][i] * cov[j][j]);
                }
            }
        }
