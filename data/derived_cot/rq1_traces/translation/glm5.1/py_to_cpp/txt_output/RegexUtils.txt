class RegexUtils {

    bool match(const std::string& pattern, const std::string& text) {

            std::regex re(pattern);
            return std::regex_search(text, re, std::regex_constants::match_continuous);
        } catch (const std::regex_error&) {
            return false;
        }
    }

    std::vector<std::string> findall(const std::string& pattern, const std::string& text) {
        std::vector<std::string> result;
        try {
            std::regex re(pattern);
            auto it = std::sregex_iterator(text.begin(), text.end(), re);
            auto end = std::sregex_iterator();
            for (; it != end; ++it) {
                const std::smatch& match = *it;
                if (match.size() > 1) {
                    result.push_back(match.str(1));
                } else {
                    result.push_back(match.str(0));
                }
            }
        } catch (const std::regex_error&) {
            // ignore
        }
        return result;
    }

        std::vector<std::string> result;
        try {
            std::regex re(pattern);
            auto it = std::regex_token_iterator<std::string::const_iterator>(text.begin(), text.end(), re, -1);
            auto end = std::regex_token_iterator<std::string::const_iterator>();

                result.push_back(*it);
            }

            result.push_back(text);
        }
        return result;
