public:

        units = {

            long long scale_val = 1;
            for (long long i = 0; i < exp; ++i) scale_val *= 10;
            numwords[scales[idx]] = {scale_val, 0};
        }

                        result += current;
                        current = 0;
                    }
                    onnumber = true;
                }

        }
