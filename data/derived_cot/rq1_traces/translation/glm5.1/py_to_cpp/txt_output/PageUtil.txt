        """


    def get_page_info(self, page_number):
        """
        Retrieve information about a specific page.
        :param page_number: int, the page number to fetch information about
        :return: dict, containing page information such as current page number, total pages, etc.
        >>> page_util = PageUtil([1, 2, 3, 4], 1)
        >>> page_util.get_page_info(1)
        >>> {
        >>>     "current_page": 1,
        >>>     "per_page": 1,
        >>>     "total_pages": 4,
        >>>     "total_items": 4,
        >>>     "has_previous": False,
        >>>     "has_next": True,
        >>>     "data": [1]
        >>> }

        """


    def search(self, keyword):

class PageUtil:
    def __init__(self, data, page_size):
        self.data = data
        self.page_size = page_size
        self.total_items = len(data)
        self.total_pages = (self.total_items + page_size - 1) // page_size

    def get_page(self, page_number):
        if page_number < 1 or page_number > self.total_pages:
            return []

        start_index = (page_number - 1) * self.page_size
        end_index = start_index + self.page_size
        return self.data[start_index:end_index]

    def get_page_info(self, page_number):
        if page_number < 1 or page_number > self.total_pages:
            return {}

        start_index = (page_number - 1) * self.page_size
        end_index = min(start_index + self.page_size, self.total_items)
        page_data = self.data[start_index:end_index]

        page_info = {
            "current_page": page_number,
            "per_page": self.page_size,
            "total_pages": self.total_pages,
            "total_items": self.total_items,
            "has_previous": page_number > 1,
            "has_next": page_number < self.total_pages,
            "data": page_data
        }
        return page_info

    def search(self, keyword):
        results = [item for item in self.data if keyword in str(item)]
        num_results = len(results)
        num_pages = (num_results + self.page_size - 1) // self.page_size

        search_info = {
            "keyword": keyword,
            "total_results": num_results,
            "total_pages": num_pages,
            "results": results
        }
        return search_info

### Key translation decisions:

1. **Template class**: Since Python's `str(item)` works on any type, the C++ version is templated on `T` and uses `operator<<` via `std::ostringstream` for string conversion — the idiomatic C++ equivalent of `str()`.

2. **Dicts → Structs**: Python's `dict` returns are replaced with `PageInfo` and `SearchInfo` structs. This preserves all the same named fields while being idiomatic C++. Access is `info.current_page` instead of `info["current_page"]`.

3. **Empty dict → `std::nullopt`**: `get_page_info` returns `{}` for invalid pages in Python. In C++, `std::optional<PageInfo>` with `std::nullopt` preserves the same "no result" semantics. `get_page` returns an empty `std::vector<T>`, matching Python's `[]`.

4. **Integer division**: `(total_items + page_size - 1) // page_size` translates directly to `(total_items_ + page_size_ - 1) / page_size_` since C++ integer division truncates toward zero, matching Python's floor division for positive operands.

5. **`keyword in str(item)`**: Translated to `item_to_string(item).find(keyword) != std::string::npos`, which is the C++ substring search equivalent.

6. **Python slicing**: `self.data[start:end]` with out-of-bounds `end` is handled by clamping `end_index` with `std::min`, since C++ iterators don't auto-clamp like Python slices.
