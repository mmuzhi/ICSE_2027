import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.HashMap;

class EmailClient {
    private String addr;
    private double capacity;
    private List<Map<String, Object>> inbox;

    public EmailClient(String addr, double capacity) {
        this.addr = addr;
        this.capacity = capacity;
        this.inbox = new ArrayList<>();
    }

    public boolean sendTo(EmailClient recv, String content, double size) {
        if (!recv.isFullWithOneMoreEmail(size)) {
            String timestamp = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"));
            Map<String, Object> email = new HashMap<>();
            email.put("sender", this.addr);
            email.put("receiver", recv.addr);
            email.put("content", content);
            email.put("size", size);
            email.put("time", timestamp);
            email.put("state", "unread");
            recv.inbox.add(email);
            return true;
        }
        else {
            this.clearInbox(size);
            if (recv.isFullWithOneMoreEmail(size)) {
                return false;
            }
            return sendTo(recv, content, size); // Retry sending after clearing
        }
    }

    public Map<String, Object> fetch() {
        if (this.inbox.isEmpty()) {
            return null;
        }
        for (Map<String, Object> email : this.inbox) {
            if ("unread".equals(email.get("state"))) {
                email.put("state", "read");
                return email;
            }
        }
        return null;
    }

    public boolean isFullWithOneMoreEmail(double size) {
        double occupiedSize = this.getOccupiedSize();
        return occupiedSize + size > this.capacity;
    }

    public double getOccupiedSize() {
        double occupiedSize = 0;
        for (Map<String, Object> email : this.inbox) {
            Object sizeObj = email.get("size");
            if (sizeObj instanceof Number) {
                occupiedSize += ((Number) sizeObj).doubleValue();
            }
        }
        return occupiedSize;
    }

    public void clearInbox(double size) {
        if (this.addr.isEmpty()) {
            return;
        }
        double freedSpace = 0;
        while (freedSpace < size && !this.inbox.isEmpty()) {
            Map<String, Object> email = this.inbox.remove(0);
            Object sizeObj = email.get("size");
            if (sizeObj instanceof Number) {
                freedSpace += ((Number) sizeObj).doubleValue();
            }
        }
    }

    public List<Map<String, Object>> getInbox() {
        return this.inbox;
    }

    public void setInbox(List<Map<String, Object>> inbox) {
        this.inbox = inbox;
    }

### Key translation decisions:

- **`Map<String, Object>`** → `std::map<std::string, std::variant<std::string, double>>` — since values are only `String` or `double`, `std::variant` replaces Java's `Object` + `instanceof` pattern.
- **`instanceof Number` / `.doubleValue()`** → `std::get_if<double>()` — type-safe variant access replaces the runtime type check.
- **`List<Map<...>>`** → `std::vector<Email>` — vectors provide the same index-based semantics; `remove(0)` becomes `erase(begin())` after capturing `front()`.
- **`LocalDateTime.now().format(...)`** → `std::chrono::system_clock` + `std::put_time` with `"%Y-%m-%d %H:%M:%S"` — produces identical timestamp format.
- **`fetch()` returns `Map<String,Object>` (nullable)** → returns `Email*` (nullable via `nullptr`) — pointer to the in-vector element mirrors Java's reference semantics (valid until the vector is modified).
- **`getInbox()`/`setInbox()`** — return by reference and accept by const reference, matching Java's getter/setter pattern.
- **`const` correctness** — `isFullWithOneMoreEmail` and `getOccupiedSize` are marked `const` since they don't modify state, which is idiomatic C++ without changing behavior.
