package org.example;

import java.util.HashMap;
import java.util.Map;
import java.util.Set;

public class CurrencyConverter {
    private Map<String, Double> rates;

    public CurrencyConverter() {
        rates = new HashMap<>();
        rates.put("USD", 1.0);
        rates.put("EUR", 0.85);
        rates.put("GBP", 0.72);
        rates.put("JPY", 110.15);
        rates.put("CAD", 1.23);
        rates.put("AUD", 1.34);
        rates.put("CNY", 6.40);
    }

    public double convert(double amount, String fromCurrency, String toCurrency) {
        if (fromCurrency.equals(toCurrency)) {
            return amount;
        }

        if (!rates.containsKey(fromCurrency) || !rates.containsKey(toCurrency)) {
            return -1;
        }

        double fromRate = rates.get(fromCurrency);
        double toRate = rates.get(toCurrency);

        double convertedAmount = (amount / fromRate) * toRate;
        return convertedAmount;
    }

    public Set<String> getSupportedCurrencies() {
        return rates.keySet();
    }

    public boolean addCurrencyRate(String currency, double rate) {
        if (rates.containsKey(currency)) {
            return false;
        }
        rates.put(currency, rate);
        return true;
    }

    public boolean updateCurrencyRate(String currency, double newRate) {
        if (!rates.containsKey(currency)) {
            return false;
        }
        rates.put(currency, newRate);
        return true;
    }

    public Map<String, Double> getRates() {
        return rates;
    }
}

### Key translation decisions:

| Java | C++ | Rationale |
|------|-----|-----------|
| `HashMap<String, Double>` | `std::unordered_map<std::string, double>` | Both are hash-based maps with O(1) average lookup |
| `Set<String>` | `std::unordered_set<std::string>` | Matches the unordered nature of `HashMap.keySet()` |
| `fromCurrency.equals(toCurrency)` | `fromCurrency == toCurrency` | `std::string::operator==` performs value comparison |
| `rates.containsKey(k)` | `rates.count(k)` / `rates.find(k)` | Pre-C++20 idiom; `count` returns 0 or 1 |
| `rates.get(k)` | `rates.at(k)` | Bounds-checked access matching Java's `get` (key guaranteed present) |
| `rates.put(k, v)` | `rates[k] = v` | Inserts or updates via `operator[]` |
| `getRates()` returns `Map<>` | Returns `std::unordered_map<>&` | Non-const reference preserves Java's semantics where the caller can modify the internal map |
| `getSupportedCurrencies()` | Returns a newly constructed `unordered_set` | Java's `keySet()` returns a live view; C++ has no direct equivalent, so we construct a snapshot (the read behavior is identical; mutation-through-view is an edge case not exercised by the class itself) |
| `package org.example` | `namespace org::example` | Direct structural mapping |
