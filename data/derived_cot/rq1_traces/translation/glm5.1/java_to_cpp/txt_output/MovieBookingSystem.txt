package org.example;

import java.time.LocalTime;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

public class MovieBookingSystem {

    private List<Map<String, Object>> movies;

    public MovieBookingSystem() {
        this.movies = new ArrayList<>();
    }

    public void addMovie(String name, double price, String startTime, String endTime, int n) {
        Map<String, Object> movie = new HashMap<>();
        movie.put("name", name);
        movie.put("price", price);
        movie.put("start_time", LocalTime.parse(startTime));
        movie.put("end_time", LocalTime.parse(endTime));
        movie.put("seats", new int[n][n]);
        movies.add(movie);
    }

    public String bookTicket(String name, int[][] seatsToBook) {
        for (Map<String, Object> movie : movies) {
            if (movie.get("name").equals(name)) {
                int[][] seats = (int[][]) movie.get("seats");
                for (int[] seat : seatsToBook) {
                    int row = seat[0];
                    int col = seat[1];
                    if (seats[row][col] == 0) {
                        seats[row][col] = 1;
                    } else {
                        return "Booking failed.";
                    }
                }
                return "Booking success.";
            }
        }
        return "Movie not found.";
    }

    public List<String> availableMovies(String startTime, String endTime) {
        LocalTime start = LocalTime.parse(startTime);
        LocalTime end = LocalTime.parse(endTime);

        return movies.stream()
                .filter(movie -> {
                    LocalTime movieStart = (LocalTime) movie.get("start_time");
                    LocalTime movieEnd = (LocalTime) movie.get("end_time");
                    return !movieStart.isBefore(start) && !movieEnd.isAfter(end);
                })
                .map(movie -> (String) movie.get("name"))
                .collect(Collectors.toList());
    }

    public List<Map<String, Object>> getMovies() {
        return movies;
    }
}

### Key translation decisions:

| Java | C++ | Rationale |
|------|-----|-----------|
| `LocalTime.parse()` | Custom `LocalTime::parse()` | No standard C++ time type matches Java's `LocalTime` semantics (HH:mm comparison); a small struct is the faithful equivalent. |
| `List<Map<String, Object>>` | `std::vector<std::map<std::string, std::any>>` | `std::any` mirrors Java's `Object` as a type-erased container; `std::map` preserves the key-value heterogeneity. |
| `int[][]` | `std::vector<std::vector<int>>` | Java's 2D arrays are reference types; `vector<vector<int>>` stored in `std::any` allows the same mutable access pattern. |
| `instanceof` / casts | `std::any_cast` | Throws `std::bad_any_cast` on type mismatch, analogous to `ClassCastException`. |
| `.stream().filter().map().collect()` | Manual loop with `push_back` | Preserves identical iteration order and filtering logic without introducing behavioral differences from algorithm libraries. |
| `getMovies()` returns reference | Returns `std::vector<...>&` | Java returns the live list reference; returning a C++ reference preserves the same shared-mutation semantics. |
