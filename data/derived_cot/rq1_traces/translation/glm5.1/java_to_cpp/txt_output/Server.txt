package org.example;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class Server {

    private List<Integer> whiteList;
    private Map<String, Object> sendStruct;
    private Map<String, Object> receiveStruct;

    public Server() {
        this.whiteList = new ArrayList<>();
        this.sendStruct = new HashMap<>();
        this.receiveStruct = new HashMap<>();
    }

    public List<Integer> addWhiteList(int addr) {
        if (whiteList.contains(addr)) {
            return null;
        } else {
            whiteList.add(addr);
            return whiteList;
        }
    }

    public List<Integer> delWhiteList(int addr) {
        if (!whiteList.contains(addr)) {
            return null;
        } else {
            whiteList.remove(Integer.valueOf(addr));
            return whiteList;
        }
    }

    public Object recv(Map<String, Object> info) {
        if (info == null || !info.containsKey("addr") || !info.containsKey("content")) {
            return -1;
        }
        int addr = (int) info.get("addr");
        String content = (String) info.get("content");
        if (!whiteList.contains(addr)) {
            return false;
        } else {
            receiveStruct.put("addr", addr);
            receiveStruct.put("content", content);
            return receiveStruct.get("content");
        }
    }

    public String send(Map<String, Object> info) {
        if (info == null || !info.containsKey("addr") || !info.containsKey("content")) {
            return "info structure is not correct";
        }
        sendStruct.put("addr", info.get("addr"));
        sendStruct.put("content", info.get("content"));
        return null;
    }

    public Map<String, Object> show(String type) {
        if ("send".equals(type)) {
            return sendStruct;
        } else if ("receive".equals(type)) {
            return receiveStruct;
        } else {
            return null;
        }
    }
}

=== RAW ANSWER ===
To translate the Java code to C++ while keeping the behavior identical, we need to address a few key differences between the languages:

1. **`Object` type:** Java's `Object` can hold any reference type. In C++, the closest equivalent is `std::any` (introduced in C++17), which can safely hold values of any type (like `int`, `bool`, `std::string`, etc.) and throws `std::bad_any_cast` on type mismatches, mirroring Java's `ClassCastException`.
2. **Nullable References:** Java objects can be `null`. C++ references cannot be null, so we must use pointers (`*`) for parameters and return types that are designed to handle nulls in the Java implementation. This preserves the exact null-checking behavior and edge cases.
3. **`HashMap`:** Java's `HashMap` translates well to `std::unordered_map` in C++, as both provide hash-based key-value storage.
4. **`ArrayList`:** Translates to `std::vector`. 
5. **Memory Safety:** Returning pointers to internal members (like `whiteList` or `sendStruct`) mirrors Java's reference-return behavior, allowing the caller to directly modify the internal state just as they could in Java.
