        std::time_t time2 = std::chrono::system_clock::to_time_t(t2);
        std::tm tm1 = *std::localtime(&time1);
        std::tm tm2 = *std::localtime(&time2);
        return tm1.tm_year == tm2.tm_year && tm1.tm_mon == tm2.tm_mon && tm1.tm_mday == tm2.tm_mday;
    }
};
