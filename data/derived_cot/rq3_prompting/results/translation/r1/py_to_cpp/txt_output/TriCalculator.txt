class TriCalculator {
public:

    }

            result *= a;
            a -= 1;
        }
        return result;
    }

    double taylor(double x, int n) {
        double a = 1.0;
        x = x / 180.0 * PI;
        int count = 1;
        for (int k = 1; k < n; ++k) {
            if (count % 2 != 0) {
                a -= std::pow(x, 2 * k) / factorial(2 * k);
            } else {
                a += std::pow(x, 2 * k) / factorial(2 * k);
            }
