#include <vector>

class Interpolation {
public:
    Interpolation() = default;

    
    static std::vector<double> interpolate_1d(const std::vector<double>& x,
        const std::vector<double>& y,
        const std::vector<double>& x_interp);

    
    static std::vector<double> interpolate_2d(const std::vector<double>& x,
        const std::vector<double>& y,
        const std::vector<std::vector<double>>& z,
        const std::vector<double>& x_interp,
        const std::vector<double>& y_interp);
};

std::vector<double> Interpolation::interpolate_1d(const std::vector<double>& x,
    const std::vector<double>& y,
    const std::vector<double>& x_interp) {
    std::vector<double> y_interp;
    for (double xi : x_interp) {
        for (size_t i = 0; i < x.size() - 1; ++i) {
            if (x[i] <= xi && xi <= x[i + 1]) {
                double yi = y[i] + (y[i + 1] - y[i]) * (xi - x[i]) / (x[i + 1] - x[i]);
                y_interp.push_back(yi);
                break;
            }
        }
    }
    return y_interp;
}

class Interpolation:
    @staticmethod
    def interpolate_1d(x, y, x_interp):
        """
        Linear interpolation for 1D data.
        x, y: lists of float, must have same length.
        x_interp: list of float, points at which to interpolate.
        Returns a list of interpolated y values for each point in x_interp
        that lies within an interval defined by x. Points outside are skipped.
        """
        y_interp = []
        for xi in x_interp:
            for i in range(len(x) - 1):
                if x[i] <= xi <= x[i + 1]:
                    yi = y[i] + (y[i + 1] - y[i]) * (xi - x[i]) / (x[i + 1] - x[i])
                    y_interp.append(yi)
                    break
        return y_interp

    @staticmethod
    def interpolate_2d(x, y, z, x_interp, y_interp):
        """
        Bilinear interpolation for 2D data.
        x, y: 1D lists of float (grid coordinates).
        z: 2D list of float, shape len(x) x len(y).
        x_interp, y_interp: lists of float of same length (query points).
        Returns a list of interpolated z values for each query point pair
        that lies within the grid; points outside are skipped.
        """
        z_interp = []
        for xi, yi in zip(x_interp, y_interp):
            for i in range(len(x) - 1):
                if x[i] <= xi <= x[i + 1]:
