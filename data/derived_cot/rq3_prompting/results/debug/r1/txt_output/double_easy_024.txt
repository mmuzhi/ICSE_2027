class Solution:
    def sortString(self, s: str) -> str:
        freq = {}
        letters = sorted(set(s))
        res = ""
        for i in s:
            if i in freq:
                freq[i] += 1
            else:
                freq[i] = 1

        while freq:
            # forward pass: smallest to largest
            for i in letters:
                if i in freq and freq[i] > 0:
                    res += i
                    freq[i] -= 1
                    if freq[i] == 0:
                        del freq[i]

            # reverse pass: largest to smallest
            for i in letters[::-1]:
                if i in freq and freq[i] > 0:
                    res += i
                    freq[i] -= 1
                    if freq[i] == 0:
                        del freq[i]

        return res
```

class Solution:
    def sortString(self, s: str) -> str:
        freq = {}
        letters = sorted(set(s))
        res = ""
        for i in s:
            if i in freq:
                freq[i] += 1
            else:
                freq[i] = 1

        while freq:
            # forward pass: smallest to largest
            for i in letters:
                if i in freq and freq[i] > 0:
                    res += i
                    freq[i] -= 1
                    if freq[i] == 0:
                        del freq[i]

            # reverse pass: largest to smallest
            for i in letters[::-1]:
                if i in freq and freq[i] > 0:
                    res += i
                    freq[i] -= 1
                    if freq[i] == 0:
                        del freq[i]

        return res
