from collections import defaultdict

class Solution:
    def getMaxRepetitions(self, s1: str, n1: int, s2: str, n2: int) -> int:
        rec, track = [0], defaultdict(int)
        ct = start = ptr2 = 0

        if not set(s2).issubset(set(s1)):
