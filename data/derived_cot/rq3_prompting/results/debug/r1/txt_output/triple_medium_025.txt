from collections import defaultdict
from typing import List

class Solution:
    def cdist(self, it: int, distArray, distArrayIndex: int, edges: List[int]) -> List[int]:
        rdist = 0
        nodes = []
        while it != -1 and distArray[it][distArrayIndex] > rdist:

            nodes.append(it)
            it = edges[it]

            if m != INF:

                    ans = k
