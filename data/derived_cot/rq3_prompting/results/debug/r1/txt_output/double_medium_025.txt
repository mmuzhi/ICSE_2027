from typing import List

class Solution:
    def maximumGap(self, nums: List[int]) -> int:
        nums = sorted(nums)
        if len(nums) < 2:
            return 0
        max_gap = 0
        for i in range(len(nums) - 1):
            diff = abs(nums[i] - nums[i+1])
            if diff > max_gap:
                max_gap = diff
        return max_gap
