from typing import List
import bisect

class Solution:
    def medianSlidingWindow(self, nums: List[int], k: int) -> List[float]:
        window = []
        ans = []
        for i, x in enumerate(nums):
            bisect.insort(window, x)
            if len(window) > k:
                # remove the element that just left the window
                leaving = nums[i - k]
                idx = bisect.bisect_left(window, leaving)
                # we need to remove one occurrence; since duplicates, we remove at that index
                window.pop(idx)
            if len(window) == k:
                if k % 2 == 1:
                    ans.append(float(window[k // 2]))
