from typing import List

class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        res = []
        cur = []
        def backtrack(i):

                cur.pop()

                    cur.append(s[i:i+j+1])
                    backtrack(i+j+1)
                    cur.pop()
                j += 1
        backtrack(0)
        return res
