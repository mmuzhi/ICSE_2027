        rec = [0]

        track = defaultdict(int)

        ct = 0          # number of times we started a new s1 repeat (excluding the initial one)
        start = 0       # current search start position in the current s1 repeat

                if ptr == -1:               # need to move to the next s1 repeat

                start = ptr + 1

            rec.append(ct + 1)
