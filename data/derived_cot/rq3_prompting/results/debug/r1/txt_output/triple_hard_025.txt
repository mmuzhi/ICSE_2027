```
class Solution:
    def numPermsDISequence(self, s: str) -> int:
        MOD = 10**9+7
        n = len(s)
        # dp[i][j] for length i+1 ending with j
        dp = [ [0]*(n+1) for _ in range(n+1) ]  # Actually size n+1 for 0..n
        dp[0][0] = 1
        for i, ch in enumerate(s):
            # Compute prefix sums of dp[i] for faster sums
            # For 'I', dp[i+1][j] = sum_{k<j} dp[i][k]
            # For 'D', dp[i+1][j] = sum_{k>=j} dp[i][k]
            prefix = [0]*(n+2)
            for j in range(n+1):
                prefix[j+1] = (prefix[j] + dp[i][j]) % MOD
            for j in range(i+2):  # j ranges 0..i+1 inclusive
                if ch == 'I':
                    dp[i+1][j] = prefix[j]  # sum_{k=0}^{j-1}
                else:  # 'D'
                    dp[i+1][j] = (prefix[i+1+1] - prefix[j]) % MOD  # sum_{k=j}^{i}
        return sum(dp[n][j] for j in range(n+1)) % MOD
```
