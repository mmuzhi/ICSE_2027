from typing import List
from functools import lru_cache

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppingCosts *= 2  # each topping can be used 0,1,2 times

        @lru_cache(None)
        def fn(i: int, x: int) -> int:
            """Return sum of subsequence of toppingCosts[i:] closest to x."""
            if x < 0 or i == len(toppingCosts):
                return 0
            # either skip current topping or take it (then skip its duplicate)
            return min(
                fn(i + 1, x),
                toppingCosts[i] + fn(i + 2, x - toppingCosts[i]),
                key=lambda y: (abs(y - x), y)
            )

        ans = float('inf')
        for bc in baseCosts:

                bc + fn(0, target - bc),
                key=lambda x: (abs(x - target), x)

        return ans

from functools import lru_cache

class Solution:
    def closestCost(self, baseCosts: List[int], toppingCosts: List[int], target: int) -> int:
        toppingCosts *= 2  # each topping can be used 0,1,2 times

        @lru_cache(None)
        def fn(i: int, x: int) -> int:
            """Return sum of subsequence of toppingCosts[i:] closest to x."""
            if x < 0 or i == len(toppingCosts):

            # either skip current topping or take it (then skip its duplicate)
            return min(
                fn(i + 1, x),
                toppingCosts[i] + fn(i + 2, x - toppingCosts[i]),
                key=lambda y: (abs(y - x), y)
            )

        ans = float('inf')
        for bc in baseCosts:
