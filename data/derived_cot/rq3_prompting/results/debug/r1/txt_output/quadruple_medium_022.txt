```python
class Solution:
    def smallestString(self, s: str) -> str:
        s_list = list(s)
        n = len(s_list)
        # Find first non-'a' character
        i = 0
        while i < n and s_list[i] == 'a':
            i += 1
        if i == n:
            # All characters are 'a', need to change last 'a' to 'z'
            s_list[-1] = 'z'
            return ''.join(s_list)
        # Decrement each non-'a' character until we hit 'a' or end
        while i < n and s_list[i] != 'a':
            s_list[i] = chr(ord(s_list[i]) - 1)
            i += 1
        return ''.join(s_list)
```
