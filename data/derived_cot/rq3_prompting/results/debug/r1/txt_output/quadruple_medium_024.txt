                for dx, dy in dirs:
                    nx, ny = x+dx, y+dy
                    if 0 <= nx < m and 0 <= ny < n and grid[nx][ny] == 1 and not visited[nx][ny]:
                        visited[nx][ny] = True
                        q2.append((nx, ny))
            grid[i][j] = 1   # restore
            if not reachable:
                return True
        return False
