from typing import List
from collections import defaultdict

class Solution:
    def longestCycle(self, edges: List[int]) -> int:
        n = len(edges)
        bl = [0] * n
        mp = defaultdict(int)
        mx = -1
        for i in range(n):
            if bl[i] == 0:
                x = i
                l = 0
                st = set()
