from collections import Counter, defaultdict, deque

class BIT:
    def __init__(self, n):

        self.tree = [0] * (n + 1)

        return ans

        i += 1
        while i <= self.n:
            self.tree[i] += value
            i += (i & (-i))


class Solution:
    def isTransformable(self, s: str, t: str) -> bool:
        if Counter(s) != Counter(t):
            return False
        ind = defaultdict(deque)

            ind[ch].append(id)
        a = []

        n = len(a)
        bt = BIT(n + 1)
        ind = defaultdict(lambda: -1)
        for i in range(n):
