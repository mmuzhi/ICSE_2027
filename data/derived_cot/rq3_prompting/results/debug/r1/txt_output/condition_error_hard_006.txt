from typing import List
from collections import deque

class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        M, N = range(m), range(n)
        dir = ((), (0,1), (0,-1), (1,0), (-1,0))  # 1: right, 2: left, 3: down, 4: up
        queue = deque()

                grid[x][y] = None          # mark visited
