    def longestValidSubstring(self, word: str, forbidden: List[str]) -> int:

        for f in forbidden:

                counter += 1
                if "end" in t:

        j = len(word)

            truc = isForbidden(word[i:j])
            if truc:
                j = i + truc - 1
            res = max(res, j - i)
        return res
