            return True
        res = []
        
        for k in range(wlen):
            for i in words:
                track.update({i: 0})
            for i in range(k, slen + k, wlen):
                w = s[i:i + wlen]
                if w in words:
                    track.update({w: track[w] + 1})
            if test():
                res.append(k)
            for i in range(wlen + k, len(s) - slen + 1, wlen):
                nw = s[i + slen - wlen:i + slen]
                pw = s[i - wlen:i]
                if nw in words:
                    track.update({nw: track[nw] + 1})
                if pw in words:
                    track.update({pw: track[pw] - 1})
                if test():
                    res.append(i)
        return res
