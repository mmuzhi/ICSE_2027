                d1 = grid[i][j] + grid[i+1][j+1] + grid[i+2][j+2]
                d2 = grid[i+2][j] + grid[i+1][j+1] + grid[i][j+2]
                
                if r1 == r2 == r3 == c1 == c2 == c3 == d1 == d2 == 15:
                    res += 1
        
        return res
