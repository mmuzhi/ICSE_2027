                i, j = dq.popleft()
                if i == m - 1 and j == n - 1:
                    return False
                for di, dj in dirs:
                    if 0 <= i + di < m and 0 <= j + dj < n and grid[i + di][j + dj] == 2:
                        dq.append((i + di, j + dj))
                        grid[i + di][j + dj] = 0
            if len(dq) == 1 and dq[0] != (m - 1, n - 1):
                return True

        return True
