```python
from collections import defaultdict
from math import gcd
from typing import List

class Solution:
    def maxPoints(self, points: List[List[int]]) -> int:
        N = len(points)
        if N <= 2:
            return N
        result = 0
        for i in range(N):
            slopes = defaultdict(int)
            duplicates = 0
            x1, y1 = points[i]
            for j in range(N):
                if i == j:
                    continue
                x2, y2 = points[j]
                if x1 == x2 and y1 == y2:
                    duplicates += 1
                    continue
                dx = x2 - x1
                dy = y2 - y1
                g = gcd(dx, dy)
                slopes[(dy // g, dx // g)] += 1
            if slopes:
                result = max(result, max(slopes.values()) + 1 + duplicates)
            else:
                result = max(result, 1 + duplicates)
        return result
```
