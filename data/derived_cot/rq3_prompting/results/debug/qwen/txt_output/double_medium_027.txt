The search space ranges from 1 to a safe upper bound (max distance multiplied by the number of segments). For each candidate speed, we calculate the total travel time. import math

class Solution:
    def minSpeedOnTime(self, dist: List[int], hour: float) -> int:
        n = len(dist)

            return -1
