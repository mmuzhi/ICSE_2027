This approach efficiently computes the complement by correcting the order of the processed bits, ensuring the result is accurate for all non-negative integers.

class Solution:
    def findComplement(self, num: int) -> int:
        com = ''
        while num > 0:
            if num % 2 == 1:
                com += '0'
            else:
                com += '1'
            num = num // 2
        com = com[::-1]
        return int(com, 2)
