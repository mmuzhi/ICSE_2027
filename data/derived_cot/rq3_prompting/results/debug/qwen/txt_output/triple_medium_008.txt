To solve this problem, we need to determine if a given set of equations can be satisfied without any contradictions. The equations can be either equality ("a==b") or inequality ("a!=b"). The solution involves using a Disjoint Set Union (DSU) data structure to efficiently manage and check the relationships between variables.

### Approach
1. **Disjoint Set Union (DSU) Initialization**: The DSU data structure is initialized with 26 nodes (representing letters 'a' to 'z'). Each node starts as its own parent with a rank of 0.
2. **Processing Equations**: 
   - **Equality Equations ("a==b")**: For each equality equation, we union the two variables using the DSU's union by rank method. This ensures that variables that must be equal are grouped together.
   - **Inequality Equations ("a!=b")**: For each inequality equation, we check if the two variables are in the same set. If they are, it means there is a contradiction, and we return False immediately.
3. **Final Check**: After processing all equations, if no contradictions are found during the inequality checks, we return True.

### Solution Code
```python
class Disjoint:
    def __init__(self):
        self.rank = [0] * 26
        self.parent = [i for i in range(26)]
    
    def finduPar(self, node):
        if self.parent[node] == node:
            return node
        self.parent[node] = self.finduPar(self.parent[node])
        return self.parent[node]
    
    def byrank(self, u, v):
        ulp_u = self.finduPar(u)
        ulp_v = self.finduPar(v)
        if ulp_u == ulp_v:
            return
        if self.rank[ulp_u] > self.rank[ulp_v]:
            self.parent[ulp_v] = ulp_u
        elif self.rank[ulp_u] < self.rank[ulp_v]:
            self.parent[ulp_u] = ulp_v
        else:
            self.parent[ulp_v] = ulp_u
            self.rank[ulp_u] += 1

class Solution:
    def equationsPossible(self, equations: List[str]) -> bool:
        disjoint = Disjoint()
        nq = []
        n = len(equations)
        for i in range(n):
            if equations[i][1] == '!':
                if equations[i][0] == equations[i][-1]:
                    return False
                else:
                    nq.append(equations[i])
            else:
                disjoint.byrank(ord(equations[i][0]) - 97, ord(equations[i][-1]) - 97)
        for i in range(len(nq)):
            x = ord(nq[i][0]) - 97
            y = ord(nq[i][-1]) - 97
            if disjoint.finduPar(x) == disjoint.finduPar(y):
                return False
        return True
```

### Explanation

class Disjoint:
    def __init__(self):
        self.rank = [0] * 26
        self.parent = [i for i in range(26)]
    
    def finduPar(self, node):
        if self.parent[node] == node:
            return node
        self.parent[node] = self.finduPar(self.parent[node])
        return self.parent[node]
    
    def byrank(self, u, v):
        ulp_u = self.finduPar(u)
        ulp_v = self.finduPar(v)
        if ulp_u == ulp_v:
            return
        if self.rank[ulp_u] > self.rank[ulp_v]:
            self.parent[ulp_v] = ulp_u
        elif self.rank[ulp_u] < self.rank[ulp_v]:
            self.parent[ulp_u] = ulp_v
        else:
            self.parent[ulp_v] = ulp_u
            self.rank[ulp_u] += 1

class Solution:
    def equationsPossible(self, equations: List[str]) -> bool:
        disjoint = Disjoint()
        nq = []
        n = len(equations)
        for i in range(n):
            if equations[i][1] == '!':
                if equations[i][0] == equations[i][-1]:
                    return False
                else:
                    nq.append(equations[i])
            else:
                disjoint.byrank(ord(equations[i][0]) - 97, ord(equations[i][-1]) - 97)
        for i in range(len(nq)):
            x = ord(nq[i][0]) - 97
            y = ord(nq[i][-1]) - 97
            if disjoint.finduPar(x) == disjoint.finduPar(y):
                return False
        return True
