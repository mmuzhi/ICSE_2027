```python
class Solution:
    def longestValidSubstring(self, word: str, forbidden: List[str]) -> int:
        # Build a set of forbidden words
        forbidden_set = set(forbidden)
        
        def isForbidden(s):
            for f in forbidden_set:
                if f in s:
                    return True
            return False
        
        n = len(word)
        if n == 0:
            return 0
        
        # Two pointers
        i = 0
        j = 0
        max_len = 0
        
        while j < n:
            # Check if the current substring word[i:j+1] contains any forbidden word
            if isForbidden(word[i:j+1]):
                # If it does, move the start pointer i forward until the substring no longer contains a forbidden word
                i = j + 1
            else:
                # Update the maximum length
                max_len = max(max_len, j - i + 1)
                j += 1
        
        return max_len
```
