2. **Helper Function**: For any given substring, traverse the trie to check if it contains any forbidden substring. If a forbidden substring is found, the function returns the length of the first occurrence; otherwise, it returns False.

class Solution:

        trie = {}

            t = trie
            for c in f:

            t = trie
            for c in f:
                if c not in t:
                    t[c] = {}
                t = t[c]
            t["end"] = True
        
        def isForbidden(s):
            t = trie
            for c in s:
                if c not in t:
                    t = trie
                else:
                    t = t[c]
                if "end" in t:
                    return True
            return False
        
        res = 0
        j = len(word)
        for i in range(len(word) - 1, -1, -1):
            if isForbidden(word[i:j]):
                j = i
            else:
                res = max(res, j - i)
        return res
