class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        res = []
        cur = []
        def backtrack(i):
            if i == len(s) and len(cur) == 4:
                res.append(".".join(cur))
                return
            if len(cur) > 4 or i > len(s):
                return
            if s[i] == '0':
                cur.append('0')
                backtrack(i + 1)
                cur.pop()
                return

                if int(num) < 256:
                    cur.append(num)
                    backtrack(i + j + 1)
                    cur.pop()
                j += 1
        backtrack(0)
        return res
