    word_score = 0

    for word_location_set in word_set:
        word_score = 0
        word_multiplier = 1

        for location in word_location_set:
            square = board.board_square_dict[location]

            word_score += square.tile.point_value * square.letter_multiplier

        word_score *= word_multiplier

    return total_score
