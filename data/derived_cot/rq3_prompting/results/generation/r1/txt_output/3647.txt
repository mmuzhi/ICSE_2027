```
from typing import List
import heapq

class Solution:
    def maxRemoval(self, nums: List[int], queries: List[List[int]]) -> int:
        n = len(nums)
        # Group queries by left endpoint
        left_to_right = [[] for _ in range(n)]
        for l, r in queries:
            left_to_right[l].append(r)
        
        available = []  # max-heap of right endpoints (store negative)
        selected_end = []  # min-heap of right endpoints of selected intervals
        selected_cnt = 0
        # We'll maintain coverage as len(selected_end) but we can keep variable for speed
        coverage = 0
        
        for i in range(n):
            # Add new intervals starting at i
            for r in left_to_right[i]:
                heapq.heappush(available, -r)
            
            # Remove expired selected intervals (those ending before i)
            while selected_end and selected_end[0] < i:
                heapq.heappop(selected_end)
                coverage -= 1
            
            # Need to satisfy demand at i
            demand = nums[i]
            while coverage < demand:
                # Remove expired from available (those that end before i, i.e., r < i)
                while available and -available[0] < i:
                    heapq.heappop(available)
                if not available:
                    return -1
                # Pick interval with max right
                r = -heapq.heappop(available)
                heapq.heappush(selected_end, r)
                coverage += 1
                selected_cnt += 1
        
        return len(queries) - selected_cnt
```

```python
from typing import List
import heapq

class Solution:
    def maxRemoval(self, nums: List[int], queries: List[List[int]]) -> int:
        n = len(nums)
        # group intervals by left endpoint
        left_to_right = [[] for _ in range(n)]
        for l, r in queries:
            left_to_right[l].append(r)

        # max-heap of available intervals (store -right for max)
        available = []
        # min-heap of right endpoints of selected intervals
        selected_end = []
        
        selected_cnt = 0
        coverage = 0  # number of selected intervals covering current index

        for i in range(n):
            # add intervals that start at i
            for r in left_to_right[i]:
                heapq.heappush(available, -r)

            # remove selected intervals that ended before i
            while selected_end and selected_end[0] < i:
                heapq.heappop(selected_end)
                coverage -= 1

            demand = nums[i]
            # select additional intervals until demand is met
            while coverage < demand:
                # discard intervals from available that already ended
                while available and -available[0] < i:
                    heapq.heappop(available)
                if not available:
                    return -1
                # pick the interval with farthest right endpoint
                r = -heapq.heappop(available)
                heapq.heappush(selected_end, r)
                coverage += 1
                selected_cnt += 1

        return len(queries) - selected_cnt
```
