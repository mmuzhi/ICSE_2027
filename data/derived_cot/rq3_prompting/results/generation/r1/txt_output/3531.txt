```python
from typing import List

class Solution:
    def minDamage(self, power: int, damage: List[int], health: List[int]) -> int:
        n = len(damage)
        # compute number of attacks needed for each enemy
        attacks = [(h + power - 1) // power for h in health]  # ceil division

        # create list of indices and sort by attacks[i] / damage[i] (increasing)
        idx = list(range(n))
        # use float key; for exact ordering, cross multiplication could be used
        idx.sort(key=lambda i: attacks[i] / damage[i])

        total = 0
        cur_time = 0
        for i in idx:
            cur_time += attacks[i]
            total += damage[i] * cur_time

        return total
```
