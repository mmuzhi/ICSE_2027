=== ANSWER ===
```python
def main():
    S = input().strip()
    # 1-indexed positions of each letter
    pos = {c: i + 1 for i, c in enumerate(S)}
    total = 0
    # Sum distances between consecutive letters in alphabetical order
    for i in range(25):
        cur = chr(ord('A') + i)
        nxt = chr(ord('A') + i + 1)
        total += abs(pos[cur] - pos[nxt])
    print(total)

if __name__ == "__main__":
    main()
```
