=== ANSWER ===
```python
class Solution:
    def numberOfSubstrings(self, s: str, k: int) -> int:
        n = len(s)
        total = n * (n + 1) // 2
        freq = [0] * 26
        l = 0
        r = 0
        valid = 0  # count of substrings with all character frequencies < k
        
        while l < n:
            # extend right as far as possible while condition holds
            while r < n:
                idx = ord(s[r]) - ord('a')
                if freq[idx] + 1 >= k:
                    break
                freq[idx] += 1
                r += 1
            # all substrings starting at l and ending before r are valid
            valid += r - l
            # move left pointer
            idx_l = ord(s[l]) - ord('a')
            freq[idx_l] -= 1
            l += 1
        
        return total - valid
```
