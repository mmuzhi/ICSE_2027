import sys

MOD = 998244353

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    W = int(data[0])
    H = int(data[1])
    L = int(data[2])
    R = int(data[3])
    D = int(data[4])
    U = int(data[5])

    max_n = W + H + 10
    fac = [1] * (max_n + 1)
    for i in range(1, max_n + 1):
        fac[i] = fac[i-1] * i % MOD
    ifac = [1] * (max_n + 1)
    ifac[max_n] = pow(fac[max_n], MOD-2, MOD)
    for i in range(max_n, 0, -1):
        ifac[i-1] = ifac[i] * i % MOD

    def binom(n: int, k: int) -> int:
        if k < 0 or k > n:
            return 0
        return fac[n] * ifac[k] % MOD * ifac[n-k] % MOD

    # total paths in full rectangle (including hole)
    total_full = 0
    for r in range(W + 1):
        term1 = (H + 1) * binom(r + H + 1, r + 1) % MOD
        term2 = (r + 1) * binom(r + H + 1, r + 2) % MOD
        S = (term1 - term2) % MOD
        total_full = (total_full + (W - r + 1) * S) % MOD

    # paths that start in the hole
    t_low = H - U
    t_high = H - D
    cnt_y = t_high - t_low + 1
    H_starts = 0
    for x in range(L, R + 1):
        a = W - x
        # sum_{t=t_low}^{t_high} C(a+t+2, a+1) = C(a+t_high+3, a+2) - C(a+t_low+1, a+2)
        sum_bin = (binom(a + t_high + 3, a + 2) - binom(a + t_low + 1, a + 2)) % MOD
        H_starts = (H_starts + sum_bin - cnt_y) % MOD

    # paths that first enter the hole from the left edge
    left_sum = 0
    if L > 0:
        for y in range(D, U + 1):
            term1 = (binom(L + y + 1, L) - 1) % MOD
            term2 = (binom(W + H - L - y + 2, W - L + 1) - 1) % MOD
            left_sum = (left_sum + term1 * term2) % MOD

    # paths that first enter the hole from the bottom edge
    bottom_sum = 0
    if D > 0:
        for x in range(L, R + 1):
            term1 = (binom(x + D + 1, x + 1) - 1) % MOD
            term2 = (binom(W + H - x - D + 2, W - x + 1) - 1) % MOD
            bottom_sum = (bottom_sum + term1 * term2) % MOD

    total_bad = (H_starts + left_sum + bottom_sum) % MOD
    ans = (total_full - total_bad) % MOD
    print(ans)

if __name__ == "__main__":
    main()
