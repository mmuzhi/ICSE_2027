```python
from typing import List

class Solution:
    def countValidSelections(self, nums: List[int]) -> int:
        n = len(nums)
        # Find all indices where nums[i] == 0
        zeros = [i for i in range(n) if nums[i] == 0]
        valid_count = 0
        
        for start in zeros:
            for direction in (1, -1):  # 1 = right, -1 = left
                # Make a copy to simulate
                arr = nums[:]
                curr = start
                d = direction
                
                # Simulate the process
                while 0 <= curr < n:
                    if arr[curr] == 0:
                        curr += d
                    else:
                        arr[curr] -= 1
                        d = -d          # reverse direction
                        curr += d       # take a step in the new direction
                
                # Check if all elements became zero
                if all(x == 0 for x in arr):
                    valid_count += 1
        
        return valid_count
```
