        if forced[pos] is None:
            free_positions.append(pos)
            covering[pos].append(idx)
    # free_positions is sorted naturally
    if not free_positions:

    last = free_positions[-1]

        # No updates needed for windows (forced positions not in covering)
    else:
        # Build set of expected chars for active windows that have deadline at j
        expected_set = set()
        for idx in deadline[j]:
            if active[idx]:
                expected = str2[j - starts[idx]]
                expected_set.add(expected)
        if expected_set:
            # find smallest c not in expected_set
            c = 'a'
            ord_c = 97

        ans[j] = c
        # Satisfy windows covering j that have c != expected
        for idx in covering[j]:

            expected = str2[j - starts[idx]]
            if c != expected:

12. After loop, we should have all active false. But we can check: if any active[idx] is True, return "" (just in case). But theoretically it shouldn't happen if all deadlines were handled.
13. Return ''.join(ans)

class Solution:
    def generateString(self, str1: str, str2: str) -> str:
        n = len(str1)
        m = len(str2)
        L = n + m - 1
        forced = [None] * L
        # process T
        for i in range(n):
            if str1[i] == 'T':
                for k in range(m):
                    pos = i + k
                    ch = str2[k]
                    if forced[pos] is None:
                        forced[pos] = ch
                    elif forced[pos] != ch:
                        return ""
        # find F windows unsatisfied by forced
        F_starts = []
        for i in range(n):
            if str1[i] == 'F':
                satisfied = False
                for k in range(m):
                    pos = i + k
                    if forced[pos] is not None and forced[pos] != str2[k]:
                        satisfied = True
                        break
                if not satisfied:
                    # check free positions exist
                    free_exist = any(forced[i+k] is None for k in range(m))
                    if not free_exist:
                        return ""
                    F_starts.append(i)
        w = len(F_starts)
        starts = F_starts
        covering = [[] for _ in range(L)]
        deadline = [[] for _ in range(L)]
        for idx in range(w):
            s = starts[idx]
            free_positions = []
            for k in range(m):
                pos = s + k
                if forced[pos] is None:
                    free_positions.append(pos)
                    covering[pos].append(idx)
            if not free_positions:
                return "" # should not happen
            last = free_positions[-1]
            deadline[last].append(idx)
        active = [True] * w
        res = [''] * L
        for j in range(L):
            if forced[j] is not None:
                res[j] = forced[j]
            else:
                # collect expected chars from active deadline windows
                expected_set = set()
                for idx in deadline[j]:
                    if active[idx]:
                        expected = str2[j - starts[idx]]
                        expected_set.add(expected)
                if expected_set:
                    # find smallest c not in set
                    c = 'a'
                    code = 97
                    while code <= 122:
                        if chr(code) not in expected_set:
                            c = chr(code)
                            break
                        code += 1
                    else:
                        return ""  # no char available
                else:
                    c = 'a'
                res[j] = c
                # satisfy windows covering j
                for idx in covering[j]:
                    if not active[idx]:
                        continue
                    expected = str2[j - starts[idx]]
                    if c != expected:
                        active[idx] = False
        # after loop, optionally check all active False
        if any(active):
            return ""
        return ''.join(res)

class Solution:
    def generateString(self, str1: str, str2: str) -> str:
        n = len(str1)
        m = len(str2)
        L = n + m - 1

        # forced assignments from 'T' windows
        forced = [None] * L
        for i in range(n):
            if str1[i] == 'T':
                for k in range(m):
                    pos = i + k
                    ch = str2[k]
                    if forced[pos] is None:
                        forced[pos] = ch
                    elif forced[pos] != ch:
                        return ""

        # collect F windows that are not already satisfied by forced mismatches
        f_starts = []
        for i in range(n):
            if str1[i] == 'F':
                # check if already has a mismatch because of forced chars
                satisfied = False
                for k in range(m):
                    pos = i + k
                    if forced[pos] is not None and forced[pos] != str2[k]:
                        satisfied = True
                        break
                if not satisfied:
                    # must have at least one free position left
                    has_free = any(forced[i + k] is None for k in range(m))
                    if not has_free:
                        return ""
                    f_starts.append(i)

        w = len(f_starts)
        if w == 0:
            # no F window needs special handling
            # just fill forced positions and set remaining free positions to 'a'
            ans = []
            for j in range(L):
                if forced[j] is not None:
                    ans.append(forced[j])
                else:
                    ans.append('a')
            return ''.join(ans)

        # map start index to window index
        start_to_idx = [-1] * n
        for idx, s in enumerate(f_starts):
            start_to_idx[s] = idx

        # compute deadlines (last free position) for each window
        deadline = [[] for _ in range(L)]      # deadline[pos] = list of window indices
        # to compute deadlines, we need the last free position of each window
        for idx, s in enumerate(f_starts):
            last_free = -1
            for k in range(m):
                pos = s + k
                if forced[pos] is None:
                    last_free = pos
            # last_free is guaranteed to exist (has_free checked)
            deadline[last_free].append(idx)

        active = [True] * w
        ans = [''] * L

        for j in range(L):
            if forced[j] is not None:
                ans[j] = forced[j]
                continue

            # free position j
            # collect expected chars from active windows that have deadline exactly j
            need_mismatch = set()
            for idx in deadline[j]:
                if active[idx]:
                    expected = str2[j - f_starts[idx]]  # offset = j - start
                    need_mismatch.add(expected)

            if need_mismatch:
                # choose smallest character not in need_mismatch
                c = 'a'
                ord_c = 97
                while ord_c <= 122:
                    if chr(ord_c) not in need_mismatch:
                        c = chr(ord_c)
                        break
                    ord_c += 1
                else:
                    return ""   # no possible character
            else:
                c = 'a'

            ans[j] = c

            # satisfy windows that cover j and for which c != expected
            # iterate over all possible start indices of windows covering j
            left = max(0, j - m + 1)
            right = min(n - 1, j)
            for i in range(left, right + 1):
                idx = start_to_idx[i]
                if idx != -1 and active[idx]:
                    expected = str2[j - i]
                    if c != expected:
                        active[idx] = False

        # all windows should be satisfied now
        if any(active):
            return ""
        return ''.join(ans)
