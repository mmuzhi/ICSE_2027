=== ANSWER ===
```python

import sys
from collections import deque

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    H = int(data[0])
    W = int(data[1])
    D = int(data[2])
    # The next H lines are the grid rows
    grid = data[3:3+H]

    # Distance array: -1 means not visited
    dist = [[-1] * W for _ in range(H)]
    q = deque()

    # Enqueue all humidifiers
    for i in range(H):
        row = grid[i]
        for j in range(W):
            if row[j] == 'H':
                dist[i][j] = 0
                q.append((i, j))

    # Directions: up, down, left, right
    dirs = [(-1,0),(1,0),(0,-1),(0,1)]

    while q:
        x, y = q.popleft()
        cur_dist = dist[x][y]
        if cur_dist >= D:          # cannot expand further
            continue
        nd = cur_dist + 1
        for dx, dy in dirs:
            nx, ny = x + dx, y + dy
            if 0 <= nx < H and 0 <= ny < W:
                # wall cells are never visited (we never add them)
                if grid[nx][ny] != '#' and dist[nx][ny] == -1:
                    dist[nx][ny] = nd
                    q.append((nx, ny))

    # Count all visited floor cells (including 'H' cells)
    ans = 0
    for i in range(H):
        for j in range(W):
            if dist[i][j] != -1 and grid[i][j] != '#':
                ans += 1
    print(ans)

if __name__ == "__main__":
    solve()
```
