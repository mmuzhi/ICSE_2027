from typing import List

class Solution:
    def minDifference(self, nums: List[int]) -> int:
        # if no -1, just compute maximum adjacent difference
        if all(x != -1 for x in nums):
            return max(abs(nums[i] - nums[i+1]) for i in range(len(nums)-1))
        
        # if all are -1, answer is 0
        if all(x == -1 for x in nums):
            return 0
        
        # binary search on D
        lo, hi = 0, 10**9
        while lo < hi:
            mid = (lo + hi) // 2
            if self._feasible(nums, mid):
                hi = mid
            else:
                lo = mid + 1
        return lo
    
    def _feasible(self, nums: List[int], D: int) -> bool:
        n = len(nums)
        intervals = []
        
        # collect intervals from known numbers that have a missing neighbour
        for i, val in enumerate(nums):
            if val != -1:
                left_missing = (i > 0 and nums[i-1] == -1)
                right_missing = (i < n-1 and nums[i+1] == -1)
                if left_missing or right_missing:
                    intervals.append((val - D, val + D))
        
        # process interior gaps of exactly one -1
        known = [(i, val) for i, val in enumerate(nums) if val != -1]
        for idx in range(len(known)-1):
            i1, v1 = known[idx]
            i2, v2 = known[idx+1]
            if i2 - i1 == 2:          # exactly one -1 between
                l = max(v1 - D, v2 - D)
                r = min(v1 + D, v2 + D)
                if l > r:
                    return False
                intervals.append((l, r))
        
        # if no intervals (should not happen, but be safe)
        if not intervals:
            return True
        
        # extract max left and min right
        max_l = max(l for l, r in intervals)
        min_r = min(r for l, r in intervals)
        if max_l <= min_r:
            return True
        
        # try placing one point at min_r and the other in intersection of intervals with l > min_r
        # compute intersection of intervals with left > min_r
        start1 = -10**18
        end1 = 10**18
        cnt1 = 0
        for l, r in intervals:
            if l > min_r:
                start1 = max(start1, l)
                end1 = min(end1, r)
                cnt1 += 1
        if cnt1 > 0 and start1 <= end1 and start1 - min_r <= D:
            return True
        
        # try placing one point at max_l and the other in intersection of intervals with r < max_l
        start2 = -10**18
        end2 = 10**18
        cnt2 = 0
        for l, r in intervals:
            if r < max_l:
                start2 = max(start2, l)
                end2 = min(end2, r)
                cnt2 += 1
        if cnt2 > 0 and start2 <= end2 and max_l - end2 <= D:
            return True
        
        return False
