```python
from typing import List
import heapq

class Solution:
    def minMaxWeight(self, n: int, edges: List[List[int]], threshold: int) -> int:
        # Build the reversed graph: for an edge a->b (original), we store b->a (reversed)
        rev_adj = [[] for _ in range(n)]
        for a, b, w in edges:
            rev_adj[b].append((a, w))   # reversed edge from b to a with weight w

        INF = 10**9
        # best[v] = minimal maximum weight on a path from 0 to v in reversed graph
        best = [INF] * n
        best[0] = 0
        heap = [(0, 0)]  # (cur_max, node)

        while heap:
            cur_max, u = heapq.heappop(heap)
            if cur_max != best[u]:
                continue
            for v, w in rev_adj[u]:
                new_max = max(cur_max, w)
                if new_max < best[v]:
                    best[v] = new_max
                    heapq.heappush(heap, (new_max, v))

        ans = max(best[1:])   # ignore node 0 itself
        return -1 if ans == INF else ans
```
