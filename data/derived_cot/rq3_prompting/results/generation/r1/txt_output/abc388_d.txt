```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    maxA = max(A)
    M = 2 * N + maxA + 5  # enough for all possible values i + S_i
    BIT = [0] * (M + 2)   # 1-indexed

    def bit_add(idx: int, delta: int) -> None:
        while idx <= M:
            BIT[idx] += delta
            idx += idx & -idx

    def bit_sum(idx: int) -> int:
        s = 0
        while idx > 0:
            s += BIT[idx]
            idx -= idx & -idx
        return s

    S = [0] * N  # store S_i = A_i + R_i

    for i in range(1, N + 1):
        threshold = i - 1
        if threshold > 0:
            cnt_le = bit_sum(threshold)
        else:
            cnt_le = 0
        R = threshold - cnt_le
        S_i = A[i - 1] + R
        S[i - 1] = S_i
        v = i + S_i
        # v is guaranteed <= M because M is large enough
        bit_add(v, 1)

    # compute final answers
    out = []
    for i in range(1, N + 1):
        B = S[i - 1] - (N - i)
        if B < 0:
            B = 0
        out.append(str(B))

    sys.stdout.write(' '.join(out))

if __name__ == "__main__":
    solve()
```
