```python
import sys

def solve() -> None:
    sys.setrecursionlimit(1 << 25)
    input = sys.stdin.readline
    N = int(input().strip())
    adj = [[] for _ in range(N + 1)]
    for _ in range(N - 1):
        a, b = map(int, input().split())
        adj[a].append(b)
        adj[b].append(a)

    # Build parent and children arrays using iterative DFS
    parent = [0] * (N + 1)
    children = [[] for _ in range(N + 1)]
    order = []                     # nodes in preorder (root first)
    stack = [1]
    parent[1] = -1
    while stack:
        u = stack.pop()
        order.append(u)
        for v in adj[u]:
            if v != parent[u]:
                parent[v] = u
                children[u].append(v)
                stack.append(v)

    # DP arrays
    INF_NEG = -10**9
    g3 = [INF_NEG] * (N + 1)      # g[u][3] (internal with parent edge)
    attach = [0] * (N + 1)        # max(1, g3[u]) used by parent

    ans = -1

    # Process nodes from bottom up
    for u in reversed(order):
        # collect attachment values of children
        vals = [attach[v] for v in children[u]]
        # find top 4 values (descending)
        top4 = [INF_NEG, INF_NEG, INF_NEG, INF_NEG]
        for val in vals:
            for i in range(4):
                if val > top4[i]:
                    # shift to the right and insert
                    for j in range(3, i, -1):
                        top4[j] = top4[j - 1]
                    top4[i] = val
                    break

        # compute g[u][3] = 1 + sum of best 3 children
        if len(vals) >= 3:
            g3[u] = 1 + (top4[0] + top4[1] + top4[2])
        else:
            g3[u] = INF_NEG

        attach[u] = max(1, g3[u])   # for parent's use

        # f[u][1] : u is leaf, no parent, need exactly one child
        if len(vals) >= 1:
            f1 = 1 + top4[0]
            # f1 > 2 means the child subtree contains an internal node
            if f1 > 2:
                ans = max(ans, f1)

        # f[u][4] : u is internal, no parent, need exactly four children
        if len(vals) >= 4:
            f4 = 1 + (top4[0] + top4[1] + top4[2] + top4[3])
            ans = max(ans, f4)

    print(ans)

if __name__ == "__main__":
    solve()
```
