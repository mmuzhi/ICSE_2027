=== ANSWER ===
```python
class Solution:
    def countOfSubstrings(self, word: str, k: int) -> int:
        vowels = set('aeiou')
        n = len(word)
        ans = 0
        
        for i in range(n):
            vowel_count = {v: 0 for v in 'aeiou'}
            cons = 0
            for j in range(i, n):
                ch = word[j]
                if ch in vowels:
                    vowel_count[ch] += 1
                else:
                    cons += 1
                if cons > k:
                    break
                if cons == k and all(vowel_count[v] > 0 for v in 'aeiou'):
                    ans += 1
        
        return ans
```
