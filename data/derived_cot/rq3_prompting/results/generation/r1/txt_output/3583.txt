class Solution:
    def gcdValues(self, nums: List[int], queries: List[int]) -> List[int]:
        import bisect
        MAX_VAL = max(nums)  # or 50000
        freq = [0] * (MAX_VAL+1)
        for x in nums:
            freq[x] += 1
        cnt = [0] * (MAX_VAL+1)
        for d in range(1, MAX_VAL+1):
            s = 0
            for m in range(d, MAX_VAL+1, d):
                s += freq[m]
            cnt[d] = s
        f = [0] * (MAX_VAL+1)
        # iterate g from MAX_VAL down to 1
        for g in range(MAX_VAL, 0, -1):
            total = cnt[g] * (cnt[g]-1) // 2
            for multiple in range(2*g, MAX_VAL+1, g):
                total -= f[multiple]
            f[g] = total
        cum = [0] * (MAX_VAL+1)
        for g in range(1, MAX_VAL+1):
            cum[g] = cum[g-1] + f[g]
        ans = []
        for q in queries:
            # find smallest g such that cum[g] > q
            # using bisect_left on cum for q+1
            idx = bisect.bisect_left(cum, q+1)  # cum[0] = 0, cum[1]... 
            ans.append(idx)
        return ans

from typing import List
import bisect

class Solution:
    def gcdValues(self, nums: List[int], queries: List[int]) -> List[int]:
        # Find the maximum value in nums to bound our arrays
        max_val = max(nums)
        
        # freq[d] = count of elements equal to d
        freq = [0] * (max_val + 1)
        for x in nums:
            freq[x] += 1
        
        # cnt[d] = number of elements divisible by d
        cnt = [0] * (max_val + 1)
        for d in range(1, max_val + 1):
            total = 0
            for m in range(d, max_val + 1, d):
                total += freq[m]
            cnt[d] = total
        
        # f[g] = number of pairs with gcd exactly g
        f = [0] * (max_val + 1)
        for g in range(max_val, 0, -1):
            total = cnt[g] * (cnt[g] - 1) // 2
            # subtract pairs that have a larger gcd (multiple of g)
            for multiple in range(2 * g, max_val + 1, g):
                total -= f[multiple]
            f[g] = total
        
        # cum[g] = total number of pairs with gcd <= g
        cum = [0] * (max_val + 1)
        for g in range(1, max_val + 1):
            cum[g] = cum[g - 1] + f[g]
        
        # Answer each query using binary search (bisect_left)
        ans = []
        for q in queries:
            # We want smallest g such that cum[g] > q
            # Equivalent to cum[g] >= q+1
            idx = bisect.bisect_left(cum, q + 1)  # cum[0]=0, cum[1], ...
            ans.append(idx)
        return ans
