```python
from typing import List
import math

class Solution:
    def subsequencePairCount(self, nums: List[int]) -> int:
        MOD = 10**9 + 7
        MAX_VAL = 201  # because nums[i] <= 200, and 0 is used for empty
        dp = [[0] * MAX_VAL for _ in range(MAX_VAL)]
        dp[0][0] = 1

        for x in nums:
            new_dp = [[0] * MAX_VAL for _ in range(MAX_VAL)]
            for g1 in range(MAX_VAL):
                for g2 in range(MAX_VAL):
                    cur = dp[g1][g2]
                    if cur == 0:
                        continue
                    # skip this element
                    new_dp[g1][g2] = (new_dp[g1][g2] + cur) % MOD
                    # add to first subsequence
                    ng1 = x if g1 == 0 else math.gcd(g1, x)
                    new_dp[ng1][g2] = (new_dp[ng1][g2] + cur) % MOD
                    # add to second subsequence
                    ng2 = x if g2 == 0 else math.gcd(g2, x)
                    new_dp[g1][ng2] = (new_dp[g1][ng2] + cur) % MOD
            dp = new_dp

        answer = 0
        for g in range(1, MAX_VAL):
            answer = (answer + dp[g][g]) % MOD
        return answer
```
