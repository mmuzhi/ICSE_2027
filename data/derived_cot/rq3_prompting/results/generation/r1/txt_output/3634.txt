```
def calculateScore(self, s: str) -> int:
    lists = [[] for _ in range(26)]
    score = 0
    for i, ch in enumerate(s):
        idx = ord(ch) - 97
        mirror_idx = 25 - idx
        if lists[mirror_idx]:
            j = lists[mirror_idx].pop()
            score += i - j
        else:
            lists[idx].append(i)
    return score
```

```python
class Solution:
    def calculateScore(self, s: str) -> int:
        # lists for each lowercase letter (0='a', ..., 25='z')
        stacks = [[] for _ in range(26)]
        score = 0
        base = ord('a')
        for i, ch in enumerate(s):
            idx = ord(ch) - base
            mirror_idx = 25 - idx          # because 'a'<->'z', 'b'<->'y',...
            if stacks[mirror_idx]:
                # closest unmarked index with mirror character
                j = stacks[mirror_idx].pop()
                score += i - j
            else:
                # no match found, keep current index as unmarked
                stacks[idx].append(i)
        return score
```
