import sys

def solve() -> None:
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    t = next(it)
    out_lines = []
    for _ in range(t):
        n = next(it)
        a = [next(it) for _ in range(2 * n)]
        # Map from unordered pair (x,y) to list of starting indices (max 3)
        pair_map = {}
        for i in range(2 * n - 1):
            x, y = a[i], a[i + 1]
            if x != y:
                key = (x, y) if x < y else (y, x)
                lst = pair_map.get(key)
                if lst is None:
                    pair_map[key] = [i]
                elif len(lst) < 3:
                    lst.append(i)
        ans = 0
        for lst in pair_map.values():
            if len(lst) == 3:
                ans += 1
            elif len(lst) == 2:
                # disjoint only when the two starting indices differ by at least 3
                if lst[1] - lst[0] >= 3:
                    ans += 1
        out_lines.append(str(ans))
    sys.stdout.write("\n".join(out_lines))
