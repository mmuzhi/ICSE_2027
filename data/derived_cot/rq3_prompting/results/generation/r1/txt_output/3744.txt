class Solution:
    def minOperations(self, queries: List[List[int]]) -> int:
        # precompute powers of 4 up to 4^15 > 1e9
        pw = [1]
        for i in range(1, 16):
            pw.append(pw[-1]*4)  # 4^i
        total = 0
        for l, r in queries:
            S = 0
            # iterate over groups
            for k in range(15):  # k from 0 to 14
                start = pw[k]
                end = pw[k+1] - 1
                if start > r:
                    break
                if end < l:
                    continue
                left = max(l, start)
                right = min(r, end)
                if left <= right:
                    S += (right - left + 1) * (k+1)
            total += (S + 1) // 2
        return total

from typing import List

class Solution:
    def minOperations(self, queries: List[List[int]]) -> int:
        # precompute powers of 4: 4^0 .. 4^15 (4^15 > 1e9)
        powers = [1]
        for _ in range(1, 16):
            powers.append(powers[-1] * 4)
        
        total_ops = 0
        for l, r in queries:
            S = 0
            # iterate over intervals [4^k, 4^{k+1}-1] for k = 0 .. 14
            for k in range(15):
                start = powers[k]
                end = powers[k+1] - 1
                if start > r:          # entire interval beyond r
                    break
                if end < l:            # interval before l
                    continue
                left = max(l, start)
                right = min(r, end)
                if left <= right:
                    S += (right - left + 1) * (k + 1)
            # minimum operations = ceil(S/2)
            total_ops += (S + 1) // 2
        return total_ops
