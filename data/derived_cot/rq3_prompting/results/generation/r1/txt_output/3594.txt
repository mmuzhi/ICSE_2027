```python
from typing import List

class Solution:
    def getLargestOutlier(self, nums: List[int]) -> int:
        total = sum(nums)
        freq = {}
        for v in nums:
            freq[v] = freq.get(v, 0) + 1
        
        ans = -10**9  # sufficiently small initial value
        for x in nums:
            outlier = total - 2 * x
            if outlier in freq:
                if x != outlier:
                    if freq[outlier] >= 1:
                        if outlier > ans:
                            ans = outlier
                else:  # x == outlier
                    if freq[outlier] >= 2:
                        if outlier > ans:
                            ans = outlier
        return ans
```
