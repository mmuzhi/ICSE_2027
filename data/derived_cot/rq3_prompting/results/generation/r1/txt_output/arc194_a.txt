```python
import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input().strip())
    A = list(map(int, input().split()))

    INF_NEG = -10**18
    best_odd = INF_NEG   # best sum of a valid subsequence ending at an odd index
    best_even = INF_NEG  # best sum of a valid subsequence ending at an even index
    ans = INF_NEG

    for i, val in enumerate(A, start=1):
        if i % 2 == 1:  # odd index
            # start a new subsequence at i
            dp = val
            if best_even != INF_NEG:
                dp = max(dp, best_even + val)
            best_odd = max(best_odd, dp)
            if N % 2 == 1:
                ans = max(ans, dp)
        else:  # even index
            if best_odd != INF_NEG:
                dp = best_odd + val
                best_even = max(best_even, dp)
                if N % 2 == 0:
                    ans = max(ans, dp)

    if N % 2 == 0:
        ans = max(ans, 0)

    print(ans)

if __name__ == "__main__":
    solve()
```
