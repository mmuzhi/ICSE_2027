=== ANSWER ===
```python
class Solution:
    def findValidPair(self, s: str) -> str:
        # Count frequency of each digit (1-9)
        freq = [0] * 10
        for ch in s:
            freq[int(ch)] += 1

        # Scan adjacent pairs
        for i in range(len(s) - 1):
            a, b = s[i], s[i+1]
            if a != b:
                da, db = int(a), int(b)
                if freq[da] == da and freq[db] == db:
                    return a + b
        return ""
```
