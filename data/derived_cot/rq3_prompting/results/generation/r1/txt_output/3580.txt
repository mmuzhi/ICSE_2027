You will be given a competitive programming problem.
You are given two strings s and pattern.
A string x is called almost equal to y if you can change at most one character in x to make it identical to y.
Return the smallest starting index of a substring in s that is almost equal to pattern. If no such index exists, return -1.
A substring is a contiguous non-empty sequence of characters within a string.
 
Example 1:

Input: s = "abcdefg", pattern = "bcdffg"
Output: 1
Explanation:
The substring s[1..6] == "bcdefg" can be converted to "bcdffg" by changing s[4] to "f".

Example 2:

Input: s = "ababbababa", pattern = "bacaba"
Output: 4
Explanation:
The substring s[4..9] == "bababa" can be converted to "bacaba" by changing s[6] to "c".

Example 3:

Input: s = "abcd", pattern = "dba"
Output: -1

Example 4:

Input: s = "dde", pattern = "d"
Output: 0

 
Constraints:

1 <= pattern.length < s.length <= 10^5
s and pattern consist only of lowercase English letters.

def z_function(s):
    n = len(s)
    z = [0]*n
    l = r = 0
    for i in range(1,n):
        if i <= r:
            z[i] = min(r-i+1, z[i-l])
        while i+z[i] < n and s[z[i]] == s[i+z[i]]:
            z[i] += 1
        if i+z[i]-1 > r:
            l, r = i, i+z[i]-1
    # z[0] is usually set to n or 0; we set to 0
    return z

from typing import List

class Solution:
    def minStartingIndex(self, s: str, pattern: str) -> int:
        m, n = len(pattern), len(s)
        # Since pattern length < s length, we don't need the explicit check
        # but keep for safety
        if m > n:
            return -1

        # ---------- 1. compute LCP (longest common prefix) ----------
        # Pattern + '$' + s
        concat = pattern + '$' + s
        z_lcp = self._z_function(concat)
        lcp = [0] * n
        for i in range(n):
            lcp[i] = z_lcp[m + 1 + i]      # after '$' at index m

        # ---------- 2. compute LCS (longest common suffix) ----------
        # Reverse strings
        pat_rev = pattern[::-1]
        s_rev = s[::-1]
        concat_rev = pat_rev + '$' + s_rev
        z_lcs = self._z_function(concat_rev)
        lcs = [0] * n
        for i in range(n - m + 1):
            # index = m+1 + (n - (i+m)) = n - i + 1
            idx = n - i + 1
            lcs[i] = z_lcs[idx]

        # ---------- 3. scan for the first valid start ----------
        for i in range(n - m + 1):
            if lcp[i] == m or lcp[i] + lcs[i] >= m - 1:
                return i
        return -1

    def _z_function(self, s: str) -> List[int]:
        """Standard Z-algorithm. Z[0] is set to 0."""
        n = len(s)
        z = [0] * n
        l = r = 0
        for i in range(1, n):
            if i <= r:
                z[i] = min(r - i + 1, z[i - l])
            while i + z[i] < n and s[z[i]] == s[i + z[i]]:
                z[i] += 1
            if i + z[i] - 1 > r:
                l, r = i, i + z[i] - 1
        return z
