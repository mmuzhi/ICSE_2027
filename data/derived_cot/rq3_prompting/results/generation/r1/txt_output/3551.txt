```python
from typing import List
from array import array

class Solution:
    def maximumSubarrayXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        n = len(nums)
        # dp[i][j] will be stored as an array for row i
        dp = [None] * n

        # next_f and next_dp hold the values for row i+1 (initially dummy for i=n)
        next_f = array('i', [0]) * n
        next_dp = array('i', [0]) * n

        # process rows from bottom to top
        for i in range(n - 1, -1, -1):
            cur_f = array('i', [0]) * n      # f[i][*]
            cur_dp = array('i', [0]) * n     # dp[i][*]

            for j in range(i, n):
                if j == i:
                    f_val = nums[i]
                else:
                    f_val = next_f[j] ^ cur_f[j - 1]

                cur_f[j] = f_val

                if j == i:
                    dp_val = f_val
                else:
                    dp_val = max(f_val, next_dp[j], cur_dp[j - 1])

                cur_dp[j] = dp_val

            dp[i] = cur_dp
            next_f = cur_f
            next_dp = cur_dp

        # answer queries
        ans = []
        for l, r in queries:
            ans.append(dp[l][r])

        return ans
```
