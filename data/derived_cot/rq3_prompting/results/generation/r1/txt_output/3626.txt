class Solution:
    def smallestNumber(self, n: int, t: int) -> int:
        for num in range(n, 101):
            prod = 1
            for ch in str(num):
                prod *= int(ch)
            if prod % t == 0:
                return num

=== ANSWER ===
```python
class Solution:
    def smallestNumber(self, n: int, t: int) -> int:
        # Iterate from n to 100 (inclusive). 
        # Since n ≤ 100 and 100's digit product is 0 (divisible by any t), 
        # the answer is guaranteed to be in this range.
        for num in range(n, 101):
            prod = 1
            for ch in str(num):
                prod *= int(ch)
            if prod % t == 0:
                return num
        # Should never be reached; provided for completeness.
        return -1
```
