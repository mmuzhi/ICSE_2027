```python
from typing import List

class Solution:
    def countPathsWithXorValue(self, grid: List[List[int]], k: int) -> int:
        MOD = 10**9 + 7
        m, n = len(grid), len(grid[0])
        # dp[i][j][xor] = number of paths from (0,0) to (i,j) with XOR = xor
        dp = [[[0] * 16 for _ in range(n)] for _ in range(m)]
        dp[0][0][grid[0][0]] = 1

        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    continue
                val = grid[i][j]
                # iterate over all possible XOR values (0...15)
                for x in range(16):
                    prev = x ^ val   # the XOR value before adding this cell
                    ways = 0
                    if i > 0:
                        ways += dp[i-1][j][prev]
                    if j > 0:
                        ways += dp[i][j-1][prev]
                    dp[i][j][x] = ways % MOD

        return dp[m-1][n-1][k] % MOD
```
