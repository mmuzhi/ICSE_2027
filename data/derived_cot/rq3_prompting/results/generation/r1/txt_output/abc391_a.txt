def main():
    import sys
    # Mapping from direction to its opposite
    opposite = {
        'N': 'S',
        'S': 'N',
        'E': 'W',
        'W': 'E',
        'NE': 'SW',
        'SW': 'NE',
        'NW': 'SE',
        'SE': 'NW'
    }
    D = sys.stdin.readline().strip()
    ans = opposite[D]
    print(ans)

if __name__ == "__main__":
    main()
