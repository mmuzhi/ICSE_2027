```python
import sys
import bisect

def sieve(n):
    """Return list of primes up to n (inclusive)."""
    is_prime = [True] * (n + 1)
    is_prime[0] = is_prime[1] = False
    for i in range(2, int(n ** 0.5) + 1):
        if is_prime[i]:
            step = i
            start = i * i
            for j in range(start, n + 1, step):
                is_prime[j] = False
    return [i for i, v in enumerate(is_prime) if v]

def main():
    input = sys.stdin.readline
    Q = int(input().strip())
    queries = [int(input().strip()) for _ in range(Q)]

    max_A = 10 ** 12
    max_M = 10 ** 6          # sqrt(1e12)

    primes = sieve(max_M)

    squares = []
    # Only primes up to 1000 can be the smaller prime factor.
    small_limit = 1000

    for i, p in enumerate(primes):
        if p > small_limit:
            break
        # For each larger prime q such that p * q <= max_M (minimal product).
        for j in range(i + 1, len(primes)):
            q = primes[j]
            if p * q > max_M:
                break
            # Generate all numbers p^a * q^b <= max_M.
            pow_p = p
            while pow_p <= max_M:
                pow_q = q
                while pow_p * pow_q <= max_M:
                    m = pow_p * pow_q
                    squares.append(m * m)   # m^2 is the 400 number
                    pow_q *= q
                pow_p *= p

    squares.sort()

    out_lines = []
    for a in queries:
        idx = bisect.bisect_right(squares, a) - 1
        out_lines.append(str(squares[idx]))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
```
