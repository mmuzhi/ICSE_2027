You will be given a competitive programming problem.
The manufacturing of a certain product requires N processes numbered 1,2,\dots,N.
For each process i, there are two types of machines S_i and T_i available for purchase to handle it.

- Machine S_i: Can process A_i products per day per unit, and costs P_i yen per unit.
- Machine T_i: Can process B_i products per day per unit, and costs Q_i yen per unit.

Here, we define the production capacity as the minimum of W, that is, \displaystyle \min^{N}_{i=1} W_i.
Given a total budget of X yen, find the maximum achievable production capacity.

Print the answer as an integer.

- All input values are integers.
- 1 \le N \le 100
- 1 \le A_i,B_i \le 100
- 1 \le P_i,Q_i,X \le 10^7

def min_cost(A, P, B, Q, C):
    best = 10**30
    # iterate over x in [0, B-1]
    for x in range(B):
        need = C - A*x
        if need <= 0:
            y = 0
        else:
            y = (need + B - 1)//B
        cost = P*x + Q*y
        if cost < best:
            best = cost
    # iterate over y in [0, A-1]
    for y in range(A):
        need = C - B*y
        if need <= 0:
            x = 0
        else:
            x = (need + A - 1)//A
        cost = P*x + Q*y
        if cost < best:
            best = cost
    return best

```python
import sys

def min_cost(A, P, B, Q, C):
    """Minimum cost to achieve throughput >= C for one process."""
    best = 10**30
    # x in [0, B-1]
    for x in range(B):
        need = C - A * x
        if need <= 0:
            y = 0
        else:
            y = (need + B - 1) // B
        cost = P * x + Q * y
        if cost < best:
            best = cost
    # y in [0, A-1]
    for y in range(A):
        need = C - B * y
        if need <= 0:
            x = 0
        else:
            x = (need + A - 1) // A
        cost = P * x + Q * y
        if cost < best:
            best = cost
    return best

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    X = int(next(it))
    processes = []
    for _ in range(N):
        A = int(next(it))
        P = int(next(it))
        B = int(next(it))
        Q = int(next(it))
        processes.append((A, P, B, Q))

    # binary search on the answer
    lo = 0
    # upper bound: even if we spend all money on the best machine (max throughput per yen = 100)
    # we cannot exceed X * 100, so take hi = X * 100 + 1 to guarantee infeasibility at hi
    hi = X * 100 + 1

    def feasible(C: int) -> bool:
        total = 0
        for A, P, B, Q in processes:
            total += min_cost(A, P, B, Q, C)
            if total > X:
                return False
        return True

    while lo < hi:
        mid = (lo + hi + 1) // 2
        if feasible(mid):
            lo = mid
        else:
            hi = mid - 1

    print(lo)

if __name__ == "__main__":
    solve()
```
