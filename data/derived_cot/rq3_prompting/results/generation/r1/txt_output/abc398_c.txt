import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    a = list(map(int, data[1:1+n]))
    # count frequencies
    freq = {}
    for val in a:
        freq[val] = freq.get(val, 0) + 1
    best_val = -1
    best_label = -1
    for i, val in enumerate(a):
        if freq[val] == 1 and val > best_val:
            best_val = val
            best_label = i+1
    print(best_label if best_label != -1 else -1)

if __name__ == "__main__":
    main()
