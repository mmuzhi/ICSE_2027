This problem is a harder version of Problem C. Here, the sequence is split into three subarrays.

You are given an integer sequence of length N: A = (A_1, A_2, \ldots, A_N).
When splitting A at two positions into three non-empty (contiguous) subarrays, find the maximum possible sum of the counts of distinct integers in those subarrays.
More formally, find the maximum sum of the following three values for a pair of integers (i,j) such that 1 \leq i < j \leq N-1: the count of distinct integers in (A_1, A_2, \ldots, A_i), the count of distinct integers in (A_{i+1},A_{i+2},\ldots,A_j), and the count of distinct integers in (A_{j+1},A_{j+2},\ldots,A_{N}).

Input

The input is given from Standard Input in the following format:
N
A_1 A_2 \ldots A_N

Output

Print the answer.

Constraints


- 3 \leq N \leq 3 \times 10^5
- 1 \leq A_i \leq N (1 \leq i \leq N)
- All input values are integers.

Sample Input 1

5
3 1 4 1 5

Sample Output 1

5

This sum cannot be greater than 5, so the answer is 5. Other partitions, such as (i,j) = (1,3), (2,3), (3,4), also achieve this sum.

class SegTree:
    def __init__(self, arr): # arr indexed 1..n
        self.n = len(arr)-1 (or pass size)
        size = 1
        while size < n: size*=2
        self.size = size
        self.tree = [0]*(2*size)
        self.lazy = [0]*(2*size)
        # initialize leaves
        for i in range(1, n+1):
            self.tree[size+i-1] = arr[i]
        for i in range(size-1,0,-1):
            self.tree[i] = max(self.tree[2*i], self.tree[2*i+1])
    def _apply(self, node, val):
        self.tree[node] += val
        self.lazy[node] += val
    def _push(self, node):
        if self.lazy[node] != 0:
            self._apply(node*2, self.lazy[node])
            self._apply(node*2+1, self.lazy[node])
            self.lazy[node] = 0
    def range_add(self, l, r, val): # [l,r] inclusive, 1-indexed
        # recursively
        self._range_add(1, 1, self.size, l, r, val)
    def _range_add(self, node, nl, nr, l, r, val):
        if l > nr or r < nl: return
        if l <= nl and nr <= r:
            self._apply(node, val)
            return
        self._push(node)
        mid = (nl+nr)//2
        self._range_add(node*2, nl, mid, l, r, val)
        self._range_add(node*2+1, mid+1, nr, l, r, val)
        self.tree[node] = max(self.tree[node*2], self.tree[node*2+1])
    def range_max(self, l, r):
        return self._range_max(1,1,self.size,l,r)
    def _range_max(self, node, nl, nr, l, r):
        if l > nr or r < nl: return -10**18
        if l <= nl and nr <= r: return self.tree[node]
        self._push(node)
        mid = (nl+nr)//2
        left = self._range_max(node*2, nl, mid, l, r)
        right = self._range_max(node*2+1, mid+1, nr, l, r)
        return max(left, right)

import sys

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = [0] + list(map(int, input().split()))  # 1-indexed
    # prefix distinct
    seen = [False] * (N+2)
    PD = [0]*(N+1)
    cnt = 0
    for i in range(1, N+1):
        v = A[i]
        if not seen[v]:
            seen[v] = True
            cnt += 1
        PD[i] = cnt
    # suffix distinct
    seen = [False]*(N+2)
    SD = [0]*(N+2)  # index from 1 to N+1
    cnt = 0
    for i in range(N, 0, -1):
        v = A[i]
        if not seen[v]:
            seen[v] = True
            cnt += 1
        SD[i] = cnt
    SD[N+1] = 0
    # previous occurrence
    last = [0]*(N+2)
    prev = [0]*(N+1)
    for i in range(1, N+1):
        v = A[i]
        prev[i] = last[v]
        last[v] = i
    # build next_positions
    next_pos = [[] for _ in range(N+2)]  # index up to N
    for i in range(1, N+1):
        p = prev[i]
        next_pos[p].append(i)
    # base array for segment tree
    base = [0]*(N+1)  # 1-indexed, index j
    for j in range(1, N):
        base[j] = SD[j+1]
    base[N] = -10**9  # dummy
    # Segment tree
    size = 1
    while size < N:
        size <<= 1
    tree = [0]*(2*size)
    lazy = [0]*(2*size)
    # initialize leaves
    for j in range(1, N+1):
        tree[size + j - 1] = base[j]
    for j in range(N+1, size+1):
        tree[size + j - 1] = -10**9  # place -inf for extra leaves
    for i in range(size-1, 0, -1):
        tree[i] = max(tree[2*i], tree[2*i+1])
    # lazy propagation functions
    def apply(node, val):
        tree[node] += val
        lazy[node] += val
    def push(node):
        if lazy[node]:
            apply(node*2, lazy[node])
            apply(node*2+1, lazy[node])
            lazy[node] = 0
    def range_add(node, l, r, ql, qr, val):
        if ql > r or qr < l:
            return
        if ql <= l and r <= qr:
            apply(node, val)
            return
        push(node)
        mid = (l+r)//2
        range_add(node*2, l, mid, ql, qr, val)
        range_add(node*2+1, mid+1, r, ql, qr, val)
        tree[node] = max(tree[node*2], tree[node*2+1])
    def range_max(node, l, r, ql, qr):
        if ql > r or qr < l:
            return -10**18
        if ql <= l and r <= qr:
            return tree[node]
        push(node)
        mid = (l+r)//2
        left = range_max(node*2, l, mid, ql, qr)
        right = range_max(node*2+1, mid+1, r, ql, qr)
        return max(left, right)
    # wrapper
    def add_range(ql, qr, val):
        range_add(1, 1, size, ql, qr, val)
    def query_max(ql, qr):
        return range_max(1, 1, size, ql, qr)
    ans = 0
    # process l from N-1 down to 2
    for l in range(N-1, 1, -1):
        # add i=l
        add_range(l, N-1, 1)
        # remove positions with prev==l
        for i in next_pos[l]:
            if i <= N-1:
                add_range(i, N-1, -1)
        best = query_max(l, N-1)
        total = PD[l-1] + best
        if total > ans:
            ans = total
    print(ans)

if __name__ == "__main__":

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = [0] + list(map(int, input().split()))          # 1‑indexed

    # ---------- prefix distinct ----------
    seen = [False] * (N + 2)
    pref = [0] * (N + 1)
    cnt = 0
    for i in range(1, N + 1):
        v = A[i]
        if not seen[v]:
            seen[v] = True
            cnt += 1
        pref[i] = cnt

    # ---------- suffix distinct ----------
    seen = [False] * (N + 2)
    suff = [0] * (N + 2)
    cnt = 0
    for i in range(N, 0, -1):
        v = A[i]
        if not seen[v]:
            seen[v] = True
            cnt += 1
        suff[i] = cnt
    suff[N + 1] = 0

    # ---------- previous occurrence ----------
    last = [0] * (N + 2)
    prev = [0] * (N + 1)
    for i in range(1, N + 1):
        v = A[i]
        prev[i] = last[v]
        last[v] = i

    # ---------- list of indices whose previous occurrence is exactly x ----------
    next_pos = [[] for _ in range(N + 2)]
    for i in range(1, N + 1):
        p = prev[i]
        next_pos[p].append(i)

    # ---------- base values for segment tree ----------
    base = [0] * (N + 1)           # base[j] = suff[j+1] for j=1..N-1
    for j in range(1, N):
        base[j] = suff[j + 1]
    base[N] = -10**15              # never used

    # ---------- segment tree (range add, range max) ----------
    size = 1
    while size < N:
        size <<= 1
    tree = [0] * (2 * size)
    lazy = [0] * (2 * size)

    # initialize leaves
    for j in range(1, N + 1):
        tree[size + j - 1] = base[j]
    for j in range(N + 1, size + 1):          # extra leaves
        tree[size + j - 1] = -10**15
    for i in range(size - 1, 0, -1):
        tree[i] = max(tree[2 * i], tree[2 * i + 1])

    # lazy propagation helpers
    def apply(node: int, val: int) -> None:
        tree[node] += val
        lazy[node] += val

    def push(node: int) -> None:
        if lazy[node]:
            apply(node * 2, lazy[node])
            apply(node * 2 + 1, lazy[node])
            lazy[node] = 0

    def range_add(node: int, l: int, r: int, ql: int, qr: int, val: int) -> None:
        if ql > r or qr < l:
            return
        if ql <= l and r <= qr:
            apply(node, val)
            return
        push(node)
        mid = (l + r) // 2
        range_add(node * 2, l, mid, ql, qr, val)
        range_add(node * 2 + 1, mid + 1, r, ql, qr, val)
        tree[node] = max(tree[node * 2], tree[node * 2 + 1])

    def range_max(node: int, l: int, r: int, ql: int, qr: int) -> int:
        if ql > r or qr < l:
            return -10**18
        if ql <= l and r <= qr:
            return tree[node]
        push(node)
        mid = (l + r) // 2
        left = range_max(node * 2, l, mid, ql, qr)
        right = range_max(node * 2 + 1, mid + 1, r, ql, qr)
        return max(left, right)

    # wrappers with 1‑based indices on [1, size]
    def add_suffix(i: int, val: int) -> None:
        if i <= N - 1:
            range_add(1, 1, size, i, N - 1, val)

    def query_suffix(l: int) -> int:
        return range_max(1, 1, size, l, N - 1)

    # ---------- main sweep ----------
    ans = 0
    # l runs from N-1 down to 2 (l = i+1)
    for l in range(N - 1, 1, -1):
        # add the new leftmost index l
        add_suffix(l, 1)

        # remove indices whose previous occurrence is exactly l
        for i in next_pos[l]:
            if i <= N - 1:
                add_suffix(i, -1)

        best = query_suffix(l)          # max_{j>=l} ( distinct(l..j) + suff[j+1] )
        total = pref[l - 1] + best
        if total > ans:
            ans = total

    print(ans)


if __name__ == "__main__":
    solve()
