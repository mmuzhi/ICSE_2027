class Solution:
    def minimumCost(self, nums: List[int], cost: List[int], k: int) -> int:
        n = len(nums)
        P = [0]*n
        Q = [0]*n
        for i in range(n):
            P[i] = (P[i-1] if i>0 else 0) + nums[i]
            Q[i] = (Q[i-1] if i>0 else 0) + cost[i]
        totalC = Q[-1]
        INF = 10**18
        dp_prev = [INF]*n  # dp for exactly c-1 segments for prefix ending at index
        ans = INF
        # For c from 1 to n
        for c in range(1, n+1):
            dp_cur = [0]*n
            hull = collections.deque()
            # add line for t=-1: m=0, b=0
            hull.append((0, 0))
            for r in range(n):
                x = P[r] + k
                # query
                while len(hull) >= 2:
                    # compare first two lines
                    m1,b1 = hull[0]
                    m2,b2 = hull[1]
                    if m1*x + b1 >= m2*x + b2:
                        hull.popleft()
                    else:
                        break
                m,b = hull[0]
                best = m*x + b
                dp_cur[r] = P[r]*Q[r] + best
                # add line for t=r if dp_prev[r] is finite
                if dp_prev[r] != INF:
                    m_new = -Q[r]
                    b_new = dp_prev[r]
                    # maintain hull: while last two lines and new line make middle redundant
                    while len(hull) >= 2:
                        # get last two lines
                        l1 = hull[-2]
                        l2 = hull[-1]
                        # if intersection(l1,l2) >= intersection(l2, new_line)
                        if self._is_redundant(l1, l2, (m_new,b_new)):
                            hull.pop()
                        else:
                            break
                    hull.append((m_new, b_new))
            # after processing all r, we have dp_cur for this c
            # compute total cost for this c
            total_cost = dp_cur[-1] + k * c * totalC
            ans = min(ans, total_cost)
            # set dp_prev for next iteration
            dp_prev = dp_cur
        return ans

    def _is_redundant(self, l1, l2, l3):
        m1,b1 = l1
        m2,b2 = l2
        m3,b3 = l3
        # (b2-b1)*(m2-m3) >= (b3-b2)*(m1-m2)
        # Use cross multiplication; note slopes are decreasing: m1>m2>m3 so denominators positive.
        return (b2 - b1) * (m2 - m3) >= (b3 - b2) * (m1 - m2)
