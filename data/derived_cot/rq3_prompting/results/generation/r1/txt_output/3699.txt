```python
from typing import List

class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        n = len(nums)
        MAX_VAL = 1000

        # prefix frequency: pref[i][v] = count of v in nums[0:i]
        pref = [[0] * (MAX_VAL + 1) for _ in range(n + 1)]
        for i in range(n):
            # copy previous
            pref[i+1] = pref[i].copy()
            pref[i+1][nums[i]] += 1

        # suffix frequency: suff[i][v] = count of v in nums[i:n]
        suff = [[0] * (MAX_VAL + 1) for _ in range(n + 1)]
        for i in range(n-1, -1, -1):
            suff[i] = suff[i+1].copy()
            suff[i][nums[i]] += 1

        # lists of non-zero values for each prefix and suffix
        pref_nonzero = [[] for _ in range(n+1)]
        for i in range(n+1):
            lst = []
            for v in range(1, MAX_VAL+1):
                if pref[i][v] > 0:
                    lst.append(v)
            pref_nonzero[i] = lst

        suff_nonzero = [[] for _ in range(n+1)]
        for i in range(n+1):
            lst = []
            for v in range(1, MAX_VAL+1):
                if suff[i][v] > 0:
                    lst.append(v)
            suff_nonzero[i] = lst

        ans = 0
        # q from 2 to n-5 inclusive
        for q in range(2, n-4):
            left_freq = pref[q-1]
            left_list = pref_nonzero[q-1]
            # r from q+2 to n-3 inclusive
            for r in range(q+2, n-2):
                x = nums[q]
                y = nums[r]
                right_freq = suff[r+2]
                right_list = suff_nonzero[r+2]

                # choose the smaller list to iterate
                if len(left_list) <= len(right_list):
                    for v in left_list:
                        cnt_v = left_freq[v]
                        if cnt_v == 0:
                            continue
                        prod = v * y
                        if prod % x == 0:
                            u = prod // x
                            if u <= MAX_VAL:
                                ans += cnt_v * right_freq[u]
                else:
                    for u in right_list:
                        cnt_u = right_freq[u]
                        if cnt_u == 0:
                            continue
                        prod = u * x
                        if prod % y == 0:
                            v = prod // y
                            if v <= MAX_VAL:
                                ans += left_freq[v] * cnt_u

        return ans
```
