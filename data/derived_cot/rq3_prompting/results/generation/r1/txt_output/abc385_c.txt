=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    H = [int(next(it)) for _ in range(N)]

    if N == 1:
        print(1)
        return

    ans = 1
    for d in range(1, N):
        dp = [1] * N
        for i in range(d, N):
            if H[i] == H[i - d]:
                dp[i] = dp[i - d] + 1
        cur_max = max(dp)
        if cur_max > ans:
            ans = cur_max

    print(ans)

if __name__ == "__main__":
    solve()
```
