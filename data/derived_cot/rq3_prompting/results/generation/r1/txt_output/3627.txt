```python
from typing import List
import heapq

class Solution:
    def minTimeToReach(self, moveTime: List[List[int]]) -> int:
        n = len(moveTime)
        m = len(moveTime[0])
        # Directions: up, down, left, right
        dirs = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        
        INF = 10**18
        dist = [[INF] * m for _ in range(n)]
        dist[0][0] = 0
        
        heap = [(0, 0, 0)]  # (time, i, j)
        
        while heap:
            t, i, j = heapq.heappop(heap)
            if t != dist[i][j]:
                continue
            if i == n-1 and j == m-1:
                return t  # can return early
            for di, dj in dirs:
                ni, nj = i + di, j + dj
                if 0 <= ni < n and 0 <= nj < m:
                    # must start moving to (ni,nj) at time >= moveTime[ni][nj]
                    start = max(t, moveTime[ni][nj])
                    arrival = start + 1
                    if arrival < dist[ni][nj]:
                        dist[ni][nj] = arrival
                        heapq.heappush(heap, (arrival, ni, nj))
        
        return dist[n-1][m-1]
```
