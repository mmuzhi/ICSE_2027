\[
\sum_{1 \leq l \leq r \leq N} \left( \left(\sum_{l \leq i \leq r} A_i\right) \mathbin{\mathrm{mod}} M \right).
\]
Here, X \mathbin{\mathrm{mod}} M denotes the remainder when the non-negative integer X is divided by M.

class BIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+2)
    def add(self, idx, val):
        while idx <= self.n:
            self.bit[idx] += val
            idx += idx & -idx
    def sum(self, idx):
        s = 0
        while idx>0:
            s += self.bit[idx]
            idx -= idx & -idx
        return s

import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # prefix sums modulo M
    pref = [0] * (N + 1)
    for i in range(1, N + 1):
        pref[i] = (pref[i-1] + A[i-1]) % M

    # sum_diff = ∑_{0≤i<j≤N} (pref[j] - pref[i])
    sum_diff = 0
    for i, x in enumerate(pref):
        sum_diff += (2 * i - N) * x

    # Fenwick tree for inversion count (pref[i] > pref[j])
    size = M + 2          # values 0..M-1 -> indices 1..M
    bit = [0] * size

    def bit_add(idx: int, delta: int) -> None:
        while idx < size:
            bit[idx] += delta
            idx += idx & -idx

    def bit_sum(idx: int) -> int:
        s = 0
        while idx > 0:
            s += bit[idx]
            idx -= idx & -idx
        return s

    inv = 0
    for i, x in enumerate(pref):
        idx = x + 1                # 1‑based index for BIT
        less_eq = bit_sum(idx)     # number of previous elements ≤ x
        inv += i - less_eq         # previous elements > x
        bit_add(idx, 1)

    ans = sum_diff + M * inv
    print(ans)

if __name__ == "__main__":
    solve()
