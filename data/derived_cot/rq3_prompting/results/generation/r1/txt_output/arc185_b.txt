```python
import sys

def solve() -> None:
    it = iter(sys.stdin.read().strip().split())
    t = int(next(it))
    out_lines = []
    for _ in range(t):
        n = int(next(it))
        A = [int(next(it)) for _ in range(n)]
        total = sum(A)
        prefix = 0
        ok = True
        for k in range(1, n):          # check for k = 1 .. n-1
            prefix += A[k-1]
            # condition: k * total >= n * prefix
            if k * total < n * prefix:
                ok = False
                break
        out_lines.append("Yes" if ok else "No")
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
