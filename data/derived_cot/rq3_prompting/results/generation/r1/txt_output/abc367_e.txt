def solve() -> None:

    it = iter(data)
    N = int(next(it))
    K = int(next(it))

    # read X, convert to 0‑based
    X = [int(next(it)) - 1 for _ in range(N)]
    A = [int(next(it)) for _ in range(N)]

    # K = 0 → no operation
    if K == 0:
        sys.stdout.write(' '.join(map(str, A)))
        return

    # number of bits needed to represent K
    L = K.bit_length()

    # binary lifting table: nxt[j][i] = node after 2^j steps from i
    nxt = [array('i', [0]) * N for _ in range(L)]

    # level 0
    for i in range(N):
        nxt[0][i] = X[i]

    # higher levels
    for j in range(1, L):
        prev = nxt[j - 1]
        cur = nxt[j]
        for i in range(N):
            cur[i] = prev[prev[i]]

    # apply K steps to each starting position
    ans = [0] * N
    for i in range(N):
        pos = i
        for j in range(L):
            if (K >> j) & 1:
                pos = nxt[j][pos]
        ans[i] = A[pos]

    sys.stdout.write(' '.join(map(str, ans)))

if __name__ == "__main__":
    solve()
