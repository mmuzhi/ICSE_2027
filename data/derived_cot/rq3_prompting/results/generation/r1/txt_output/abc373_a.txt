```python
import sys

def main() -> None:
    # Read all lines from standard input
    lines = sys.stdin.read().splitlines()
    # Ensure we only consider the first 12 lines
    count = 0
    for i, s in enumerate(lines[:12], start=1):
        if len(s) == i:
            count += 1
    print(count)

if __name__ == "__main__":
    main()
```
