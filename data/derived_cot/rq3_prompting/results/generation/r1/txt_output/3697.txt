```python
from typing import List
import math

class Solution:
    def minimumIncrements(self, nums: List[int], target: List[int]) -> int:
        k = len(target)
        fullMask = (1 << k) - 1
        
        # precompute lcm for all non-empty masks
        lcm_mask = [1] * (1 << k)
        for mask in range(1, 1 << k):
            # compute lcm from the lowest set bit or use the last
            # We can compute using the last computed lcm for mask without the lowest bit
            # Easier: just compute from scratch
            l = 1
            for i in range(k):
                if mask >> i & 1:
                    l = l * target[i] // math.gcd(l, target[i])
            lcm_mask[mask] = l
        
        INF = 10**18
        dp = [INF] * (1 << k)
        dp[0] = 0
        
        for x in nums:
            # precompute costs for this x for all non-empty masks
            costs = [0] * (1 << k)  # index by mask, cost for empty mask is 0
            for mask in range(1, 1 << k):
                L = lcm_mask[mask]
                # smallest multiple of L >= x
                # Use integer arithmetic
                # cost = ((x + L - 1) // L) * L - x
                # But if L is huge, ((x+L-1)//L) becomes 1 or 0? Actually if L > x, then (x+L-1)//L = 1, cost = L - x.
                # That's fine.
                cost = ((x + L - 1) // L) * L - x
                costs[mask] = cost
            
            # update dp
            newdp = dp[:]  # not using this element
            for mask in range(1 << k):
                if dp[mask] == INF:
                    continue
                # consider all subsets S (including empty, but empty already covered by newdp copy)
                # We can iterate S from 1 to fullMask
                for S in range(1, 1 << k):
                    newmask = mask | S
                    newcost = dp[mask] + costs[S]
                    if newcost < newdp[newmask]:
                        newdp[newmask] = newcost
            dp = newdp
        
        return dp[fullMask]
```

```python
from typing import List
import math

class Solution:
    def minimumIncrements(self, nums: List[int], target: List[int]) -> int:
        k = len(target)
        full_mask = (1 << k) - 1

        # precompute lcm for each non-empty subset of targets
        lcm = [1] * (1 << k)
        for mask in range(1, 1 << k):
            l = 1
            for i in range(k):
                if mask & (1 << i):
                    l = l * target[i] // math.gcd(l, target[i])
            lcm[mask] = l

        INF = 10**18
        dp = [INF] * (1 << k)
        dp[0] = 0

        for x in nums:
            # costs to cover each subset with this element
            cost = [0] * (1 << k)          # cost[0] = 0
            for mask in range(1, 1 << k):
                L = lcm[mask]
                # smallest multiple of L not less than x
                cost[mask] = ((x + L - 1) // L) * L - x

            # DP transition: use this element at most once
            newdp = dp[:]                  # option: skip this element
            for old_mask in range(1 << k):
                if dp[old_mask] == INF:
                    continue
                for subset in range(1, 1 << k):
                    new_mask = old_mask | subset
                    cand = dp[old_mask] + cost[subset]
                    if cand < newdp[new_mask]:
                        newdp[new_mask] = cand
            dp = newdp

        return dp[full_mask]
```
