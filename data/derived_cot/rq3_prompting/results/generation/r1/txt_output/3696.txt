```python
class Solution:
    def countSubstrings(self, s: str) -> int:
        n = len(s)
        ans = 0
        # frequency maps for modulo conditions
        cnt3 = [0] * 3
        cnt9 = [0] * 9
        cnt7 = [0] * 7
        # initial prefix before any digit is 0
        cnt3[0] = 1
        cnt9[0] = 1
        cnt7[0] = 1

        # running values
        pref3 = 0
        pref9 = 0
        pref7 = 0
        inv10 = 5          # 5 ≡ 10^{-1} (mod 7)
        invpow = 1         # (inv10)^i mod 7

        for i, ch in enumerate(s):
            d = int(ch)

            # update prefix values including current digit
            pref3 = (pref3 + d) % 3
            pref9 = (pref9 + d) % 9
            pref7 = (pref7 * 10 + d) % 7
            f7 = (pref7 * invpow) % 7   # transformed value for mod 7

            # count substrings ending at i
            if d == 0:
                pass
            elif d in (1, 2, 5):
                ans += i + 1
            elif d == 4:
                # length 1 always valid
                ans += 1
                # length >=2 : depends on previous digit
                if i >= 1 and int(s[i-1]) % 2 == 0:
                    ans += i
            elif d == 8:
                # length 1
                ans += 1
                # length 2
                if i >= 1 and (int(s[i-1]) * 10 + 8) % 8 == 0:
                    ans += 1
                # length 3 or more
                if i >= 2 and int(s[i-2:i+1]) % 8 == 0:
                    ans += (i - 1)
            elif d in (3, 6):
                ans += cnt3[pref3]
            elif d == 9:
                ans += cnt9[pref9]
            elif d == 7:
                ans += cnt7[f7]

            # update frequency maps for future indices
            cnt3[pref3] += 1
            cnt9[pref9] += 1
            cnt7[f7] += 1

            # prepare invpow for next iteration
            invpow = (invpow * inv10) % 7

        return ans
```
