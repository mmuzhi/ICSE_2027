=== ANSWER ===
```python
from typing import List

class Solution:
    def minBitwiseArray(self, nums: List[int]) -> List[int]:
        ans = []
        for num in nums:
            # Even number (only prime 2) -> impossible
            if num % 2 == 0:
                ans.append(-1)
                continue
            # Count trailing ones in binary representation
            trailing_ones = 0
            temp = num
            while temp & 1:
                trailing_ones += 1
                temp >>= 1
            # Smallest a is num with the highest bit of the trailing ones block cleared
            mask = 1 << (trailing_ones - 1)   # the bit to clear
            a = num & ~mask
            ans.append(a)
        return ans
```
