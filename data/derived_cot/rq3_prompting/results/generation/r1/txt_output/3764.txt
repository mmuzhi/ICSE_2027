```python
from typing import List

class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
        candidates = []
        for row, lim in zip(grid, limits):
            # sort row descending and take top lim
            row_sorted = sorted(row, reverse=True)
            candidates.extend(row_sorted[:lim])
        # sort candidates descending
        candidates.sort(reverse=True)
        # sum first k elements
        return sum(candidates[:k])
```

```python
from typing import List

class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
        # Collect all candidates: the top `limit` numbers from each row
        candidates = []
        for row, lim in zip(grid, limits):
            # Sort row in descending order and take the first `lim` elements
            sorted_row = sorted(row, reverse=True)
            candidates.extend(sorted_row[:lim])
        # Sort all candidates in descending order and sum the first k
        candidates.sort(reverse=True)
        return sum(candidates[:k])
```
