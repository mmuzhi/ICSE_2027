def solve():
    import sys
    mod = 998244353
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    it = iter(input_data)
    N = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    P = [0]*(N+1)
    for i in range(1,N+1):
        P[i] = (P[i-1] + A[i-1]) % mod

    # Precompute powers
    powP = [[1]*(K+1) for _ in range(N+1)]
    for i in range(N+1):
        val = P[i]
        # powP[i][0] already 1
        for e in range(1, K+1):
            powP[i][e] = powP[i][e-1] * val % mod

    # Binomial coefficients
    binom = [[0]*(K+1) for _ in range(K+1)]
    for i in range(K+1):
        binom[i][0] = binom[i][i] = 1
        for j in range(1, i):
            binom[i][j] = (binom[i-1][j-1] + binom[i-1][j]) % mod

    ans = 0
    for t in range(K+1):
        sign = 1 if (K-t)%2==0 else -1
        coeff = binom[K][t] * sign % mod
        cum = 0
        total = 0
        for j in range(N+1):
            if j>0:
                total = (total + powP[j][t] * cum) % mod
            cum = (cum + powP[j][K-t]) % mod
        ans = (ans + coeff * total) % mod

    print(ans % mod)

if __name__ == "__main__":
    solve()

import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # prefix sums modulo MOD
    pref = [0] * (N + 1)
    for i in range(1, N + 1):
        pref[i] = (pref[i-1] + A[i-1]) % MOD

    # pre‑compute powers of each prefix value up to K
    pow_val = [[1] * (K + 1) for _ in range(N + 1)]
    for i in range(N + 1):
        x = pref[i]
        for e in range(1, K + 1):
            pow_val[i][e] = pow_val[i][e-1] * x % MOD

    # binomial coefficients C(K, t)
    binom = [[0] * (K + 1) for _ in range(K + 1)]
    for i in range(K + 1):
        binom[i][0] = binom[i][i] = 1
        for j in range(1, i):
            binom[i][j] = (binom[i-1][j-1] + binom[i-1][j]) % MOD

    ans = 0
    for t in range(K + 1):
        sign = 1 if (K - t) % 2 == 0 else -1
        coeff = binom[K][t] * sign % MOD
        cum = 0          # sum of P_i^{K-t} for i < current j
        total = 0
        for j in range(N + 1):
            if j > 0:
                total = (total + pow_val[j][t] * cum) % MOD
            cum = (cum + pow_val[j][K - t]) % MOD
        ans = (ans + coeff * total) % MOD

    print(ans % MOD)

if __name__ == "__main__":
    solve()
