import sys
from collections import deque

class Dinic:
    def __init__(self, n):
        self.n = n
        self.graph = [[] for _ in range(n)]
    
    def add_edge(self, fr, to, cap):
        self.graph[fr].append([to, cap, len(self.graph[to])])
        self.graph[to].append([fr, 0, len(self.graph[fr])-1])
    
    def bfs(self, s, t):
        level = [-1]*self.n
        q = deque()
        level[s]=0
        q.append(s)
        while q:
            v = q.popleft()
            for to, cap, rev in self.graph[v]:
                if cap>0 and level[to]==-1:
                    level[to]=level[v]+1
                    q.append(to)
        self.level = level
        return level[t] != -1
    
    def dfs(self, v, t, f):
        if v==t:
            return f
        for i in range(self.it[v], len(self.graph[v])):
            self.it[v] = i
            to, cap, rev = self.graph[v][i]
            if cap>0 and self.level[v] < self.level[to]:
                ret = self.dfs(to, t, min(f, cap))
                if ret>0:
                    # update
                    self.graph[v][i][1] -= ret
                    self.graph[to][rev][1] += ret
                    return ret
        return 0
    
    def max_flow(self, s, t):
        flow = 0
        INF = 10**18
        while self.bfs(s, t):
            self.it = [0]*self.n
            while True:
                f = self.dfs(s, t, INF)
                if f==0:

                flow += f

import sys
from collections import deque

class Dinic:
    def __init__(self, n):
        self.n = n
        self.graph = [[] for _ in range(n)]

    def add_edge(self, fr, to, cap):
        forward = [to, cap, None]
        backward = [fr, 0, None]
        forward[2] = len(self.graph[to])
        backward[2] = len(self.graph[fr])
        self.graph[fr].append(forward)
        self.graph[to].append(backward)

    def bfs(self, s, t):
        self.level = [-1]*self.n
        q = deque([s])
        self.level[s] = 0
        while q:
            v = q.popleft()
            for to, cap, rev in self.graph[v]:
                if cap > 0 and self.level[to] == -1:
                    self.level[to] = self.level[v] + 1
                    q.append(to)
        return self.level[t] != -1

    def dfs(self, v, t, f):
        if v == t:
            return f
        for i in range(self.it[v], len(self.graph[v])):
            self.it[v] = i
            to, cap, rev = self.graph[v][i]
            if cap > 0 and self.level[v] < self.level[to]:
                ret = self.dfs(to, t, min(f, cap))
                if ret:
                    self.graph[v][i][1] -= ret
                    self.graph[to][rev][1] += ret
                    return ret
        return 0

        flow = 0
        INF = 10**18
        while self.bfs(s, t):
            self.it = [0]*self.n
            while True:
                f = self.dfs(s, t, INF)
                if not f:
                    break
                flow += f
        return flow

        return self.level[t] != -1

    def dfs(self, v, t, f):
        if v == t:
            return f
        for i in range(self.it[v], len(self.graph[v])):
            self.it[v] = i
            to, cap, rev = self.graph[v][i]
            if cap > 0 and self.level[v] < self.level[to]:
                ret = self.dfs(to, t, min(f, cap))
                if ret:
                    self.graph[v][i][1] -= ret
                    self.graph[to][rev][1] += ret
                    return ret
        return 0

    def max_flow(self, s, t):
        flow = 0
        INF = 10**18
        while self.bfs(s, t):
            self.it = [0] * self.n
            while True:
                f = self.dfs(s, t, INF)
                if not f:
                    break
                flow += f
        return flow

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    K = int(next(it))
    edges = []
    for _ in range(M):
        u = int(next(it)) - 1
        v = int(next(it)) - 1
        edges.append((u, v))

    # compute shortest path length (unweighted) from 1 to N
    adj = [[] for _ in range(N)]
    for u, v in edges:
        adj[u].append(v)
    dist = [-1] * N
    dist[0] = 0
    q = deque([0])
    while q:
        u = q.popleft()
        for v in adj[u]:
            if dist[v] == -1:
                dist[v] = dist[u] + 1
                q.append(v)
    Lmax = dist[N - 1]  # guaranteed reachable

    def min_edges_for_distance(d):
        if d == 0:
            return 0
        V = N * d + 2
        S = V - 2
        T = V - 1
        dinic = Dinic(V)
        INF = 10**9

        # source to vertex 1 nodes
        s_id = 0
        for i in range(d):
            node = s_id * d + i
            dinic.add_edge(S, node, INF)

        # vertex N nodes to sink
        t_id = N - 1
        for i in range(d):
            node = t_id * d + i
            dinic.add_edge(node, T, INF)

        # nested edges inside each vertex
        for v in range(N):
            for i in range(d - 1):
                from_node = v * d + i
                to_node = v * d + (i + 1)
                dinic.add_edge(from_node, to_node, INF)

        # original edges
        for u, v in edges:
            # cost edges (capacity 1)
            for i in range(d):
                from_node = u * d + i
                to_node = v * d + i
                dinic.add_edge(from_node, to_node, 1)
            # constraint edges (capacity INF)
            for i in range(d - 1):
                from_node = u * d + i
                to_node = v * d + (i + 1)
                dinic.add_edge(from_node, to_node, INF)

        return dinic.max_flow(S, T)

    # binary search for the largest d with min_edges <= K
    lo, hi = 0, Lmax
    ans = 0
    while lo <= hi:
        mid = (lo + hi) // 2
        if min_edges_for_distance(mid) <= K:
            ans = mid
            lo = mid + 1
        else:
            hi = mid - 1
    print(ans)

if __name__ == "__main__":
    solve()
