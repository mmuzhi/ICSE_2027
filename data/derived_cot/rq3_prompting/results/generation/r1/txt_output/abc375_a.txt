=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    s = data[1].strip()
    ans = 0
    # Check all possible i (1-indexed) from 1 to N-2
    for i in range(n - 2):
        if s[i] == '#' and s[i+1] == '.' and s[i+2] == '#':
            ans += 1
    print(ans)

if __name__ == "__main__":
    solve()
```
