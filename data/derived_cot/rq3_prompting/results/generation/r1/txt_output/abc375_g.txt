```
import sys
import heapq

def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    edges = []
    adj = [[] for _ in range(N+1)]
    for i in range(M):
        a,b,c = map(int, input().split())
        edges.append((a,b,c))
        adj[a].append((b,c,i))
        adj[b].append((a,c,i))
    
    INF = 10**18
    dist1 = [INF]*(N+1)
    dist1[1] = 0
    pq = [(0,1)]
    while pq:
        d,u = heapq.heappop(pq)
        if d != dist1[u]:
            continue
        for v,w,_ in adj[u]:
            nd = d + w
            if nd < dist1[v]:
                dist1[v] = nd
                heapq.heappush(pq, (nd,v))
    
    distN = [INF]*(N+1)
    distN[N] = 0
    pq = [(0,N)]
    while pq:
        d,u = heapq.heappop(pq)
        if d != distN[u]:
            continue
        for v,w,_ in adj[u]:
            nd = d + w
            if nd < distN[v]:
                distN[v] = nd
                heapq.heappush(pq, (nd,v))
    
    D = dist1[N]
    
    # Build DAG of shortest path edges
    dag_adj = [[] for _ in range(N+1)]
    rev_adj = [[] for _ in range(N+1)]
    dag_edges = []  # (u,v,idx)
    for i, (a,b,c) in enumerate(edges):
        if dist1[a] + c + distN[b] == D:
            # direction a->b
            dag_adj[a].append((b,i))
            rev_adj[b].append((a,i))
            dag_edges.append((a,b,i))
        elif dist1[b] + c + distN[a] == D:
            # direction b->a
            dag_adj[b].append((a,i))
            rev_adj[a].append((b,i))
            dag_edges.append((b,a,i))
        # else: not on any shortest path, ignore
    
    # Collect relevant nodes (on some shortest path)
    relevant = []
    for v in range(1, N+1):
        if dist1[v] != INF and distN[v] != INF and dist1[v] + distN[v] == D:
            relevant.append(v)
    relevant.sort(key=lambda x: dist1[x])
    
    MOD1 = 10**9+7
    MOD2 = 10**9+9
    
    ways1a = [0]*(N+1)
    ways1b = [0]*(N+1)
    ways1a[1] = 1
    ways1b[1] = 1
    for u in relevant:
        if ways1a[u] == 0:  # can't be reached from 1? but should be if on shortest path
            continue
        for v, _ in dag_adj[u]:
            ways1a[v] = (ways1a[v] + ways1a[u]) % MOD1
            ways1b[v] = (ways1b[v] + ways1b[u]) % MOD2
    
    ways2a = [0]*(N+1)
    ways2b = [0]*(N+1)
    ways2a[N] = 1
    ways2b[N] = 1
    for u in reversed(relevant):
        if ways2a[u] == 0:
            continue
        for w, _ in rev_adj[u]:  # actually for u, we want to add to its predecessors
            ways2a[w] = (ways2a[w] + ways2a[u]) % MOD1
            ways2b[w] = (ways2b[w] + ways2b[u]) % MOD2
    
    total1a = ways1a[N] % MOD1
    total1b = ways1b[N] % MOD2
    
    critical = [False]*M
    for u,v,idx in dag_edges:
        if ways1a[u] == 0 or ways2a[v] == 0:
            continue
        prod1 = (ways1a[u] * ways2a[v]) % MOD1
        prod2 = (ways1b[u] * ways2b[v]) % MOD2
        if prod1 == total1a and prod2 == total1b:
            critical[idx] = True
    
    out = []
    for i in range(M):
        out.append("Yes" if critical[i] else "No")
    print("\n".join(out))

if __name__ == "__main__":
    solve()
```

```python
import sys
import heapq

def solve() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())
    edges = []
    adj = [[] for _ in range(N + 1)]
    for i in range(M):
        a, b, c = map(int, input().split())
        edges.append((a, b, c))
        adj[a].append((b, c, i))
        adj[b].append((a, c, i))

    INF = 10**18

    # Dijkstra from city 1
    dist1 = [INF] * (N + 1)
    dist1[1] = 0
    pq = [(0, 1)]
    while pq:
        d, u = heapq.heappop(pq)
        if d != dist1[u]:
            continue
        for v, w, _ in adj[u]:
            nd = d + w
            if nd < dist1[v]:
                dist1[v] = nd
                heapq.heappush(pq, (nd, v))

    # Dijkstra from city N
    distN = [INF] * (N + 1)
    distN[N] = 0
    pq = [(0, N)]
    while pq:
        d, u = heapq.heappop(pq)
        if d != distN[u]:
            continue
        for v, w, _ in adj[u]:
            nd = d + w
            if nd < distN[v]:
                distN[v] = nd
                heapq.heappush(pq, (nd, v))

    D = dist1[N]

    # Build DAG of edges that lie on some shortest 1->N path
    dag_adj = [[] for _ in range(N + 1)]
    rev_adj = [[] for _ in range(N + 1)]
    dag_edges = []          # (u, v, original_index)

    for i, (a, b, c) in enumerate(edges):
        if dist1[a] + c + distN[b] == D:
            dag_adj[a].append((b, i))
            rev_adj[b].append((a, i))
            dag_edges.append((a, b, i))
        elif dist1[b] + c + distN[a] == D:
            dag_adj[b].append((a, i))
            rev_adj[a].append((b, i))
            dag_edges.append((b, a, i))

    # Collect all nodes that belong to at least one shortest path
    relevant = []
    for v in range(1, N + 1):
        if dist1[v] != INF and distN[v] != INF and dist1[v] + distN[v] == D:
            relevant.append(v)
    relevant.sort(key=lambda x: dist1[x])   # topological order

    MOD1 = 1_000_000_007
    MOD2 = 1_000_000_009

    # DP for number of paths from 1 to each node
    ways1a = [0] * (N + 1)
    ways1b = [0] * (N + 1)
    ways1a[1] = 1
    ways1b[1] = 1
    for u in relevant:
        if ways1a[u] == 0:
            continue
        for v, _ in dag_adj[u]:
            ways1a[v] = (ways1a[v] + ways1a[u]) % MOD1
            ways1b[v] = (ways1b[v] + ways1b[u]) % MOD2

    # DP for number of paths from each node to N
    ways2a = [0] * (N + 1)
    ways2b = [0] * (N + 1)
    ways2a[N] = 1
    ways2b[N] = 1
    for u in reversed(relevant):
        if ways2a[u] == 0:
            continue
        for w, _ in rev_adj[u]:      # w -> u is an edge in the DAG
            ways2a[w] = (ways2a[w] + ways2a[u]) % MOD1
            ways2b[w] = (ways2b[w] + ways2b[u]) % MOD2

    total1a = ways1a[N] % MOD1
    total1b = ways1b[N] % MOD2

    critical = [False] * M
    for u, v, idx in dag_edges:
        if ways1a[u] == 0 or ways2a[v] == 0:
            continue
        prod1 = (ways1a[u] * ways2a[v]) % MOD1
        prod2 = (ways1b[u] * ways2b[v]) % MOD2
        if prod1 == total1a and prod2 == total1b:
            critical[idx] = True

    out_lines = ["Yes" if critical[i] else "No" for i in range(M)]
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
