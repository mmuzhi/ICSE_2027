```python
import sys

def solve() -> None:
    data = sys.stdin.read().splitlines()
    if not data:
        return
    S = data[0].strip()
    T = data[1].strip()
    n = min(len(S), len(T))
    for i in range(n):
        if S[i] != T[i]:
            print(i + 1)
            return
    if len(S) == len(T):
        print(0)
    else:
        print(n + 1)

if __name__ == "__main__":
    solve()
```
