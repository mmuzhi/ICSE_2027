The empty substring is the shortest matching substring.

The shortest matching substring of p in s is "adlogi".

def kmp_table(pattern):
    m = len(pattern)
    lps = [0]*m
    j=0
    for i in range(1,m):

            j+=1
            lps[i]=j
    return lps

            for ia in occA:
                target = ia + len(A)
                while j < len(occC) and occC[j] < target:
                    j += 1
                if j < len(occC):
                    length = occC[j] + len(C) - ia
                    if length < ans:
                        ans = length
            return ans if ans != float('inf') else -1

        # ---------- Case: B is non‑empty ----------
        # if a required part has no occurrence, impossible
        if not occB:
            return -1
        if A and not occA:
            return -1
        if C and not occC:
            return -1

        import bisect
        ans = float('inf')
        for ib in occB:
            # best occurrence of A before (or at) ib - len(A)
            if A:
                limit = ib - len(A)
                idx = bisect.bisect_right(occA, limit) - 1
                if idx < 0:
                    continue
                bestA = occA[idx]
            else:
                bestA = ib

            # best occurrence of C after (or at) ib + len(B)
            if C:
                limit = ib + len(B)
                idx = bisect.bisect_left(occC, limit)
                if idx >= len(occC):
                    continue
                bestC = occC[idx]
            else:
                bestC = ib + len(B)

            length = bestC + len(C) - bestA
            if length < ans:
                ans = length

        return ans if ans != float('inf') else -1
