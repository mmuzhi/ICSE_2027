```python
def solve() -> None:
    S = input().strip()
    insertions = 0
    expected = 'i'           # we always start with an i at the first position

    for ch in S:
        # keep inserting the expected character until it matches the current one
        while ch != expected:
            insertions += 1
            expected = 'o' if expected == 'i' else 'i'
        # now the current character matches; consume it
        expected = 'o' if expected == 'i' else 'i'

    # after the whole string is processed, if the last character we placed
    # was an i (i.e. the next expected character is o), we need one more o at the end
    if expected == 'o':
        insertions += 1

    print(insertions)

if __name__ == "__main__":
    solve()
```
