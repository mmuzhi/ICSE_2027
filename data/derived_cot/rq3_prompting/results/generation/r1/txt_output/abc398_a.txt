=== ANSWER ===
```python
import sys

def solve() -> None:
    N = int(sys.stdin.readline().strip())
    if N % 2 == 1:          # odd length -> one '=' at the centre
        half = (N - 1) // 2
        s = '-' * half + '=' + '-' * half
    else:                   # even length -> two '=' in the middle
        half = N // 2 - 1
        s = '-' * half + '==' + '-' * half
    print(s)

if __name__ == "__main__":
    solve()
```
