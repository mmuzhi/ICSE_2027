=== ANSWER ===
```python
import sys

def main() -> None:
    data = sys.stdin.read().split()
    it = iter(data)
    Q = int(next(it))
    # initial stack: 100 cards labeled 0
    stack = [0] * 100
    outputs = []

    for _ in range(Q):
        typ = next(it)
        if typ == '1':
            x = int(next(it))
            stack.append(x)
        else:  # typ == '2'
            outputs.append(str(stack.pop()))

    sys.stdout.write("\n".join(outputs))

if __name__ == "__main__":
    main()
```
