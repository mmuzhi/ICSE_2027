```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    grid = data[1:1+N]          # list of N strings, each of length N

    # prepare output grid
    out = [[''] * N for _ in range(N)]

    for i in range(N):          # 0‑based row index
        row_str = grid[i]
        for j in range(N):      # 0‑based column index
            r, c = i + 1, j + 1  # 1‑based coordinates

            # distance to the nearest border
            L = min(r - 1, N - r, c - 1, N - c)

            # total number of 90° clockwise rotations mod 4
            k = (L + 1) % 4

            if k == 0:          # no rotation
                nr, nc = r, c
            elif k == 1:        # 90° clockwise
                nr, nc = c, N + 1 - r
            elif k == 2:        # 180°
                nr, nc = N + 1 - r, N + 1 - c
            else:               # 270° clockwise (or 90° anticlockwise)
                nr, nc = N + 1 - c, r

            out[nr - 1][nc - 1] = row_str[j]

    # print the final grid
    for line in out:
        print(''.join(line))

if __name__ == "__main__":
    solve()
```
