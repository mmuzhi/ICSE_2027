import sys

def main():
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it); M = next(it); Q = next(it)
    edges = [None] * (M+1)  # 1-indexed
    for i in range(1, M+1):
        a = next(it)-1
        b = next(it)-1
        c = next(it)
        edges[i] = (a, b, c)
    queries = []
    for _ in range(Q):
        typ = next(it)
        if typ == 1:
            i = next(it)
            queries.append(('close', i))
        else:
            x = next(it)-1
            y = next(it)-1
            queries.append(('query', x, y))
    # Mark closed edges
    closed = [False] * (M+1)
    for q in queries:
        if q[0] == 'close':
            closed[q[1]] = True
    INF = 10**18
    d = [[INF]*N for _ in range(N)]
    for i in range(N):
        d[i][i] = 0
    # Add edges that are never closed
    for i in range(1, M+1):
        if not closed[i]:
            a, b, c = edges[i]
            if c < d[a][b]:
                d[a][b] = d[b][a] = c
    # Floyd-Warshall
    for k in range(N):
        dk = d[k]
        for i in range(N):
            dik = d[i][k]
            if dik == INF:
                continue
            di = d[i]
            for j in range(N):
                nd = dik + dk[j]
                if nd < di[j]:
                    di[j] = nd
    # Process in reverse
    ans_rev = []
    for q in reversed(queries):
        if q[0] == 'query':
            x, y = q[1], q[2]
            dist = d[x][y]
            ans_rev.append(dist if dist < INF else -1)
        else:  # close -> add edge
            i = q[1]
            a, b, c = edges[i]
            # Update all-pairs using the new edge
            row_a = d[a]
            row_b = d[b]
            for u in range(N):
                via_a = d[u][a] + c
                via_b = d[u][b] + c
                du = d[u]
                # inner loop
                for v in range(N):
                    cand1 = via_a + row_b[v]
                    if cand1 < du[v]:
                        du[v] = cand1
                    cand2 = via_b + row_a[v]
                    if cand2 < du[v]:
                        du[v] = cand2
    # Reverse answers to original order
    out_lines = '\n'.join(str(val) for val in reversed(ans_rev))
    sys.stdout.write(out_lines)

if __name__ == "__main__":
    main()

        c = next(it)

        if typ == 1:

            queries.append(('close', i))
        else:  # typ == 2

    # mark permanently closed roads
    closed = [False] * (M + 1)

    INF = 10**18
    # distance matrix
    d = [[INF] * N for _ in range(N)]
    for i in range(N):

    for i in range(1, M + 1):

            if c < d[a][b]:
                d[a][b] = d[b][a] = c

    # Floyd‑Warshall
    for k in range(N):
        dk = d[k]
        for i in range(N):

                continue

            for j in range(N):
                nd = dik + dk[j]
                if nd < di[j]:

    # process queries in reverse order
