```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    parent = list(range(N + 1))  # 1-indexed
    size = [1] * (N + 1)
    components = N

    def find(x: int) -> int:
        while parent[x] != x:
            parent[x] = parent[parent[x]]  # path compression
            x = parent[x]
        return x

    def union(a: int, b: int) -> None:
        nonlocal components
        ra = find(a)
        rb = find(b)
        if ra == rb:
            return
        # union by size
        if size[ra] < size[rb]:
            ra, rb = rb, ra
        parent[rb] = ra
        size[ra] += size[rb]
        components -= 1

    for _ in range(M):
        u = int(next(it))
        v = int(next(it))
        union(u, v)

    # maximum edges in a forest = N - components
    max_forest_edges = N - components
    answer = M - max_forest_edges
    if answer < 0:
        answer = 0
    print(answer)

if __name__ == "__main__":
    solve()
```
