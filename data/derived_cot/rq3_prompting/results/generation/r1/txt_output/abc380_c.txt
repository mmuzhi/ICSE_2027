```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    N = int(data[0])
    K = int(data[1])
    S = data[2].strip()

    # Find all 1-blocks (start, end) using 0‑based indices
    blocks = []
    i = 0
    n = len(S)
    while i < n:
        if S[i] == '1':
            start = i
            while i < n and S[i] == '1':
                i += 1
            end = i - 1
            blocks.append((start, end))
        else:
            i += 1

    # (K-1)-th and K-th blocks (1‑based indexing → subtract 2 and 1)
    l1, r1 = blocks[K - 2]
    l2, r2 = blocks[K - 1]

    # Build the resulting string
    prefix = S[:r1 + 1]
    middle_ones = '1' * (r2 - l2 + 1)
    middle_zeros = '0' * (l2 - r1 - 1)
    suffix = S[r2 + 1:]

    result = prefix + middle_ones + middle_zeros + suffix
    print(result)

if __name__ == "__main__":
    solve()
```
