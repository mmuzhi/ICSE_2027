from typing import List

class Solution:
    def minTime(self, skill: List[int], mana: List[int]) -> int:
        n = len(skill)
        m = len(mana)
        # prefix sums of skill: prefix[i] = sum(skill[0:i])
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + skill[i]
        total_skill = prefix[n]

        start = 0  # start time of the first potion on wizard 0
        # schedule potions 0 .. m-2, compute start time for next potion
        for j in range(m - 1):
            max_offset = 0
            # compute the maximum lower bound over all wizards
            for i in range(n):
                # offset = prefix[i+1]*mana[j] - prefix[i]*mana[j+1]
                offset = prefix[i + 1] * mana[j] - prefix[i] * mana[j + 1]
                if offset > max_offset:
                    max_offset = offset
            start += max_offset

        # total time = start of last potion + its total processing time
        return start + total_skill * mana[-1]
