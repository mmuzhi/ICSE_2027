=== ANSWER ===
```python
import sys
from collections import Counter

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    A, B, C, D = map(int, data[:4])
    counts = Counter([A, B, C, D])
    freq = list(counts.values())
    
    if 3 in freq or freq.count(2) == 2:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    main()
```
