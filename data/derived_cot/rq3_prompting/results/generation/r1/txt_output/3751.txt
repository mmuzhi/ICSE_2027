from typing import List

class Solution:
    def maxFrequency(self, nums: List[int], k: int) -> int:
        total_k = nums.count(k)
        max_gain = 0
        
        # Iterate over all possible values v that we can convert to k
        for v in range(1, 51):
            if v == k:
                continue
            current = 0
            best = 0
            # Kadane's algorithm to find maximum subarray sum of (count_v - count_k)
            for x in nums:
                if x == v:
                    delta = 1
                elif x == k:
                    delta = -1
                else:
                    delta = 0
                current = max(0, current + delta)
                best = max(best, current)
            max_gain = max(max_gain, best)
        
        return total_k + max_gain
