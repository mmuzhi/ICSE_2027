import sys
import random

def solve() -> None:
    input_data = sys.stdin.read().split()
    it = iter(input_data)
    N = int(next(it))
    Q = int(next(it))
    A = [0] + [int(next(it)) for _ in range(N)]
    B = [0] + [int(next(it)) for _ in range(N)]

    # Use two large prime moduli for hashing
    MOD1 = 10**9 + 7
    MOD2 = 10**9 + 9

    random.seed(0)  # deterministic for reproducibility
    rnd1 = [0] * (N + 1)
    rnd2 = [0] * (N + 1)
    for v in range(1, N + 1):
        rnd1[v] = random.randint(1, MOD1 - 1)
        rnd2[v] = random.randint(1, MOD2 - 1)

    # Prefix hashes for A
    pref1A = [0] * (N + 1)
    pref2A = [0] * (N + 1)
    for i in range(1, N + 1):
        pref1A[i] = (pref1A[i-1] + rnd1[A[i]]) % MOD1
        pref2A[i] = (pref2A[i-1] + rnd2[A[i]]) % MOD2

    # Prefix hashes for B
    pref1B = [0] * (N + 1)
    pref2B = [0] * (N + 1)
    for i in range(1, N + 1):
        pref1B[i] = (pref1B[i-1] + rnd1[B[i]]) % MOD1
        pref2B[i] = (pref2B[i-1] + rnd2[B[i]]) % MOD2

    out_lines = []
    for _ in range(Q):
        l = int(next(it))
        r = int(next(it))
        L = int(next(it))
        R = int(next(it))

        if (r - l + 1) != (R - L + 1):
            out_lines.append("No")
            continue

        hashA1 = (pref1A[r] - pref1A[l-1]) % MOD1
        hashB1 = (pref1B[R] - pref1B[L-1]) % MOD1
        hashA2 = (pref2A[r] - pref2A[l-1]) % MOD2
        hashB2 = (pref2B[R] - pref2B[L-1]) % MOD2

        if hashA1 == hashB1 and hashA2 == hashB2:
            out_lines.append("Yes")
        else:
            out_lines.append("No")

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
