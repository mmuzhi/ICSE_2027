import sys
import heapq

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    K = int(next(it))

    A = [int(next(it)) for _ in range(N)]
    B = [int(next(it)) for _ in range(N)]
    C = [int(next(it)) for _ in range(N)]

    # sort in descending order
    A.sort(reverse=True)
    B.sort(reverse=True)
    C.sort(reverse=True)

    # encode triple (i,j,k) as a single integer
    # using base = N (enough because indices are 0..N-1)
    M = N * N          # N^2
    visited = set()
    heap = []

    # initial state: largest elements of each array
    start_val = A[0] * B[0] + B[0] * C[0] + C[0] * A[0]
    start_id = 0    # i=0, j=0, k=0
    visited.add(start_id)
    heapq.heappush(heap, (-start_val, 0, 0, 0))

    for cnt in range(K):
        neg_val, i, j, k = heapq.heappop(heap)
        cur_val = -neg_val
        if cnt == K - 1:
            print(cur_val)
            return

        # generate neighbours with larger indices
        # (i+1, j, k)
        if i + 1 < N:
            ni, nj, nk = i + 1, j, k
            nid = ni * M + nj * N + nk
            if nid not in visited:
                visited.add(nid)
                nv = A[ni]*B[nj] + B[nj]*C[nk] + C[nk]*A[ni]
                heapq.heappush(heap, (-nv, ni, nj, nk))
        # (i, j+1, k)
        if j + 1 < N:
            ni, nj, nk = i, j + 1, k
            nid = ni * M + nj * N + nk
            if nid not in visited:
                visited.add(nid)
                nv = A[ni]*B[nj] + B[nj]*C[nk] + C[nk]*A[ni]
                heapq.heappush(heap, (-nv, ni, nj, nk))
        # (i, j, k+1)
        if k + 1 < N:
            ni, nj, nk = i, j, k + 1
            nid = ni * M + nj * N + nk
            if nid not in visited:
                visited.add(nid)
                nv = A[ni]*B[nj] + B[nj]*C[nk] + C[nk]*A[ni]
                heapq.heappush(heap, (-nv, ni, nj, nk))
