```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    n = int(next(it))
    # convert to 0‑based indices
    P = [int(next(it)) - 1 for _ in range(n)]
    A = [int(next(it)) for _ in range(n)]

    visited = [False] * n

    for start in range(n):
        if visited[start]:
            continue

        # collect the cycle in the order given by P
        cycle = []
        cur = start
        while not visited[cur]:
            visited[cur] = True
            cycle.append(cur)
            cur = P[cur]

        k = len(cycle)

        # smallest index (position) in the cycle
        s_idx = min(range(k), key=lambda i: cycle[i])   # index inside 'cycle'

        # index of the smallest value in the cycle
        min_val_idx = min(range(k), key=lambda i: A[cycle[i]])

        # rotation that puts the minimum value at the smallest index
        t = (min_val_idx - s_idx) % k

        # apply the rotation
        old_vals = [A[pos] for pos in cycle]
        for i, pos in enumerate(cycle):
            A[pos] = old_vals[(i + t) % k]

    sys.stdout.write(' '.join(map(str, A)))

if __name__ == "__main__":
    solve()
```
