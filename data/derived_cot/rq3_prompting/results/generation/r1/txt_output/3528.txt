=== ANSWER ===
```python
from typing import List

class Solution:
    def findMaximumScore(self, nums: List[int]) -> int:
        n = len(nums)
        score = 0
        cur = 0  # current takeoff index
        
        # Look for indices with strictly larger value than nums[cur]
        for i in range(1, n - 1):
            if nums[i] > nums[cur]:
                score += (i - cur) * nums[cur]
                cur = i
        
        # Final jump to the last index
        score += (n - 1 - cur) * nums[cur]
        
        return score
```
