import sys
sys.setrecursionlimit(1<<25)

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    adj = [[] for _ in range(N + 1)]
    for _ in range(M):
        u = int(next(it)); v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
    
    visited = [False] * (N + 1)
    color = [0] * (N + 1)   # 0/1 bipartite coloring
    
    comp_data = []   # each element: (size, a, e) where a is size of one part, e is edge count
    
    from collections import deque
    for start in range(1, N + 1):
        if visited[start]:
            continue
        # BFS
        q = deque([start])
        visited[start] = True
        color[start] = 0
        vertices = []
        cnt0 = 0
        cnt1 = 0
        while q:
            u = q.popleft()
            vertices.append(u)
            if color[u] == 0:
                cnt0 += 1
            else:
                cnt1 += 1
            for w in adj[u]:
                if not visited[w]:
                    visited[w] = True
                    color[w] = 1 - color[u]
                    q.append(w)
        # count edges in component
        e = sum(len(adj[u]) for u in vertices) // 2
        size = len(vertices)
        # a is the size of the part that color 0 belongs to (or any part, parity doesn't matter for even-sized components)
        a = cnt0   # we can use cnt0 as a
        comp_data.append((size, a, e))
    
    totalM = sum(e for _, _, e in comp_data)   # should equal input M
    # but use input M for consistency
    if N % 2 == 1:
        # odd total vertices
        if M % 2 == 1:
            print("Aoki")
        else:
            print("Takahashi")
    else:
        # even total vertices
        odd_cnt = 0
        fixed_sum = 0   # sum of a mod 2 only for even-sized components
        for size, a, _ in comp_data:
            if size % 2 == 1:
                odd_cnt += 1
            else:
                fixed_sum ^= (a & 1)   # parity of a (invariant for even size)
        if odd_cnt == 0:
            # no choice, outcome depends on fixed parity vs M parity
            if (fixed_sum & 1) != (M & 1):
                print("Aoki")
            else:
                print("Takahashi")
        else:
            if odd_cnt % 4 == 2:
                print("Aoki")
            else:
                print("Takahashi")

if __name__ == "__main__":
    solve()
