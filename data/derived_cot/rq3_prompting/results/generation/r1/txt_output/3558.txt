=== ANSWER ===
```python
from collections import deque
from typing import List

class Solution:
    def findSafeWalk(self, grid: List[List[int]], health: int) -> bool:
        m, n = len(grid), len(grid[0])
        # distance from start: minimum number of 1-cells visited so far
        dist = [[float('inf')] * n for _ in range(m)]
        dist[0][0] = grid[0][0]
        dq = deque()
        dq.append((0, 0))
        dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        
        while dq:
            i, j = dq.popleft()
            d = dist[i][j]
            for dx, dy in dirs:
                ni, nj = i + dx, j + dy
                if 0 <= ni < m and 0 <= nj < n:
                    nd = d + grid[ni][nj]
                    if nd < dist[ni][nj]:
                        dist[ni][nj] = nd
                        if grid[ni][nj] == 0:
                            dq.appendleft((ni, nj))
                        else:
                            dq.append((ni, nj))
        
        # health must be positive at the end, so cost <= health - 1
        return dist[m-1][n-1] <= health - 1
```
