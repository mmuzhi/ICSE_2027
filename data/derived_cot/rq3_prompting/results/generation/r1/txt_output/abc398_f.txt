import sys

def prefix_function(s):
    n = len(s)
    pi = [0] * n
    for i in range(1, n):
        j = pi[i-1]
        while j > 0 and s[i] != s[j]:
            j = pi[j-1]
        if s[i] == s[j]:
            j += 1
        pi[i] = j
    return pi

def main():
    S = sys.stdin.readline().strip()
    n = len(S)
    rev = S[::-1]
    U = rev + '#' + S
    pi = prefix_function(U)
    L = pi[-1]
    # prefix before palindromic suffix
    P = S[:n-L]
    ans = S + P[::-1]
    sys.stdout.write(ans + '\n')

if __name__ == "__main__":
    main()

import sys

def prefix_function(s):
    n = len(s)
    pi = [0] * n
    for i in range(1, n):
        j = pi[i - 1]
        while j > 0 and s[i] != s[j]:
            j = pi[j - 1]
        if s[i] == s[j]:
            j += 1
        pi[i] = j
    return pi

def main() -> None:
    S = sys.stdin.readline().strip()
    n = len(S)
    rev = S[::-1]
    # The separator '#' is safe because input consists of uppercase letters only.
    U = rev + '#' + S
    pi = prefix_function(U)
    L = pi[-1]                      # length of the longest palindromic suffix
    P = S[:n - L]                   # prefix before that suffix
    ans = S + P[::-1]               # shortest palindrome with S as prefix
    sys.stdout.write(ans + '\n')

    main()
