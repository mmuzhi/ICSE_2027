```python
import sys

def solve() -> None:
    data = sys.stdin.readline().split()
    N = int(data[0])
    c1 = data[1]
    c2 = data[2]
    S = sys.stdin.readline().strip()
    # Build the resulting string
    result = ''.join(ch if ch == c1 else c2 for ch in S)
    print(result)

if __name__ == "__main__":
    solve()
```
