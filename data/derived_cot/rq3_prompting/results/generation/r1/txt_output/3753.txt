=== ANSWER ===
```python
from collections import Counter

class Solution:
    def maxDifference(self, s: str) -> int:
        # Count frequencies of each character
        freq = Counter(s)
        
        # Separate odd and even frequencies
        odds = [f for f in freq.values() if f % 2 == 1]
        evens = [f for f in freq.values() if f % 2 == 0]
        
        # According to problem constraints, both lists are non-empty
        return max(odds) - min(evens)
```
