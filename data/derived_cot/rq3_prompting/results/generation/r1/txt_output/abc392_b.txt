=== ANSWER ===
```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    present = [False] * (N + 1)
    for _ in range(M):
        a = int(next(it))
        present[a] = True
    missing = [i for i in range(1, N + 1) if not present[i]]
    print(len(missing))
    if missing:
        print(' '.join(map(str, missing)))
    else:
        print()  # empty line

if __name__ == '__main__':
    main()
```
