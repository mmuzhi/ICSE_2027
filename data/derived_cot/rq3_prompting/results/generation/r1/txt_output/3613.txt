```python
from typing import List
from collections import defaultdict

class Solution:
    def maxAmount(self, initialCurrency: str, pairs1: List[List[str]], rates1: List[float],
                  pairs2: List[List[str]], rates2: List[float]) -> float:
        # Build adjacency graph for a day (bidirectional)
        def build_graph(pairs, rates):
            adj = defaultdict(list)
            for (src, tgt), rate in zip(pairs, rates):
                # forward edge
                adj[src].append((tgt, rate))
                # backward edge (reciprocal rate)
                adj[tgt].append((src, 1.0 / rate))
            return adj

        # BFS/DFS to compute relative values from start currency
        def compute_values(start, adj):
            values = {}
            stack = [(start, 1.0)]
            values[start] = 1.0
            while stack:
                node, val = stack.pop()
                for neighbor, rate in adj[node]:
                    if neighbor not in values:
                        values[neighbor] = val * rate
                        stack.append((neighbor, values[neighbor]))
            return values

        # Day 1 reachable values
        adj1 = build_graph(pairs1, rates1)
        v1 = compute_values(initialCurrency, adj1)

        # Day 2 reachable values
        adj2 = build_graph(pairs2, rates2)
        v2 = compute_values(initialCurrency, adj2)

        # Maximum product = max over currencies reachable both days of v1[X] / v2[X]
        best = 1.0
        for curr in v1:
            if curr in v2:
                best = max(best, v1[curr] / v2[curr])
        return best
```
