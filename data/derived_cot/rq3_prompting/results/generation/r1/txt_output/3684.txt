class Solution:
    def hasMatch(self, s: str, p: str) -> bool:
        star = p.index('*')
        left = p[:star]
        right = p[star+1:]
        n = len(s)
        llen = len(left)
        rlen = len(right)
        for i in range(n):
            # if left part doesn't match at i, skip
            if llen > 0 and s[i:i+llen] != left:
                continue
            # if left is empty, we don't need to check, but we still need i to be start of substring
            # Now for possible end j (exclusive) such that substring s[i:j] is non-empty
            for j in range(i+1, n+1):
                # check right part matches at end j
                if rlen > 0 and s[j-rlen:j] != right:
                    continue
                # check that the positions don't overlap: left end before right start
                # i+llen <= j-rlen  (i+llen is index after left, j-rlen is index where right starts)
                if i + llen <= j - rlen:
                    return True
        return False

class Solution:
    def hasMatch(self, s: str, p: str) -> bool:
        # Split pattern at '*'
        star_idx = p.index('*')
        left = p[:star_idx]
        right = p[star_idx+1:]

        n = len(s)
        llen = len(left)
        rlen = len(right)

        # Try every possible start index for the substring
        for i in range(n):
            # The left part must match at position i (if non-empty)
            if llen > 0 and s[i:i+llen] != left:
                continue

            # Try every possible end index (exclusive) for the substring
            # The substring must be non-empty, so j starts at i+1
            for j in range(i+1, n+1):
                # The right part must match ending at position j (if non-empty)
                if rlen > 0 and s[j-rlen:j] != right:
                    continue

                # Check the order: left completely before the * part,
                # which must end before the right part begins.
                if i + llen <= j - rlen:
                    return True

        return False
