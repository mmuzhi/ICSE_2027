MOD = 998244353

def solve():
    import sys
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0]); M = int(data[1]); S = data[2].strip()
    # Precompute transitions
    max_mask = 1 << N
    trans = [[0]*26 for _ in range(max_mask)]
    # Precompute dp arrays for each mask? We'll compute on fly
    for mask in range(max_mask):
        # Build dp array for this mask: dp[i] for i=0..N
        dp = [0]*(N+1)
        for i in range(1, N+1):
            dp[i] = dp[i-1] + ((mask >> (i-1)) & 1)
        # For each character
        for c in range(26):
            newdp = [0]*(N+1)
            # newdp[0] stays 0
            for i in range(1, N+1):
                add = 1 if (S[i-1] == chr(ord('a')+c)) else 0
                val = max(dp[i], newdp[i-1], dp[i-1] + add)
                newdp[i] = val
            # compute newmask
            newmask = 0
            for i in range(1, N+1):
                if newdp[i] > newdp[i-1]:
                    newmask |= (1 << (i-1))
            trans[mask][c] = newmask
    # DP over length
    dp_count = [0]*max_mask
    dp_count[0] = 1
    for _ in range(M):
        new_count = [0]*max_mask
        for mask in range(max_mask):
            cnt = dp_count[mask]
            if cnt == 0:
                continue
            # iterate over characters
            for c in range(26):
                nm = trans[mask][c]
                new_count[nm] = (new_count[nm] + cnt) % MOD
        dp_count = new_count
    # Aggregate by popcount
    ans = [0]*(N+1)
    for mask in range(max_mask):
        k = bin(mask).count('1')
        ans[k] = (ans[k] + dp_count[mask]) % MOD
    print(' '.join(str(x) for x in ans))

if __name__ == '__main__':
    solve()

import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    S = data[2].strip()

    max_mask = 1 << N
    # precompute transition table: trans[mask][c] -> new_mask
    trans = [[0] * 26 for _ in range(max_mask)]

    # For each mask, build the dp row and compute transitions
    for mask in range(max_mask):
        # build dp[0..N] from mask differences
        dp = [0] * (N + 1)
        for i in range(1, N + 1):
            dp[i] = dp[i - 1] + ((mask >> (i - 1)) & 1)

        # for each character
        for c in range(26):
            ch = chr(ord('a') + c)
            newdp = [0] * (N + 1)
            for i in range(1, N + 1):
                add = 1 if S[i - 1] == ch else 0
                val = max(dp[i], newdp[i - 1], dp[i - 1] + add)
                newdp[i] = val
            # build new mask from differences
            newmask = 0
            for i in range(1, N + 1):
                if newdp[i] > newdp[i - 1]:
                    newmask |= (1 << (i - 1))
            trans[mask][c] = newmask

    # DP over length
    dp_cnt = [0] * max_mask
    dp_cnt[0] = 1

    for _ in range(M):
        nxt = [0] * max_mask
        for mask in range(max_mask):
            cur = dp_cnt[mask]
            if cur == 0:
                continue
            for c in range(26):
                nm = trans[mask][c]
                nxt[nm] = (nxt[nm] + cur) % MOD
        dp_cnt = nxt

    # aggregate by final LCS length (= popcount of mask)
    ans = [0] * (N + 1)
    for mask in range(max_mask):
        k = mask.bit_count()  # Python 3.8+ has int.bit_count()
        ans[k] = (ans[k] + dp_cnt[mask]) % MOD

    print(' '.join(str(x) for x in ans))

if __name__ == "__main__":
    solve()
