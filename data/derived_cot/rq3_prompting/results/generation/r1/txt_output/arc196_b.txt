import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().split()
    it = iter(data)
    T = int(next(it))
    out_lines = []
    for _ in range(T):
        H = int(next(it))
        W = int(next(it))
        grid = [next(it).strip() for _ in range(H)]

        # ---- 1. parity check on rows and columns ----
        row_par = [0] * H
        col_par = [0] * W
        for i in range(H):
            row = grid[i]
            for j, ch in enumerate(row):
                if ch == 'A':
                    row_par[i] ^= 1
                    col_par[j] ^= 1
        if any(row_par) or any(col_par):
            out_lines.append('0')
            continue

        # ---- 2. build bipartite graph from B cells ----
        row_adj = [[] for _ in range(H)]
        col_adj = [[] for _ in range(W)]
        col_cur = [0] * W                     # XOR of A's in column j among rows 0..i-1

        for i in range(H):
            row = grid[i]
            row_cumul = 0                     # XOR of A's in columns 1..j-1 of current row
            for j in range(W):
                ch = row[j]
                a = 1 if ch == 'A' else 0
                if ch == 'B':
                    # c = (x_{j-1}*x_j) / 2   (0/1)
                    c = 0 if j == 0 else row_cumul
                    # d = (y_{i-1}*y_i) / 2   (0/1)
                    d = 0 if i == 0 else col_cur[j]
                    f = 1 ^ c ^ d             # equation: r_i xor c_j = f
                    row_adj[i].append((j, f))
                    col_adj[j].append((i, f))
                # update row_cumul for the next column (only for j>=1)
                if j >= 1:
                    row_cumul ^= a
            # after finishing the row, update col_cur for the next row
            for j in range(W):
                if grid[i][j] == 'A':
                    col_cur[j] ^= 1

        # ---- 3. count connected components and check consistency ----
        total_nodes = H + W
        val = [-1] * total_nodes
        comp_cnt = 0
        consistent = True
        for start in range(total_nodes):
            if val[start] != -1:
                continue
            comp_cnt += 1
            stack = [start]
            val[start] = 0
            while stack and consistent:
                u = stack.pop()
                if u < H:                     # u is a row
                    for v, w in row_adj[u]:
                        vv = H + v
                        if val[vv] == -1:
                            val[vv] = val[u] ^ w
                            stack.append(vv)
                        elif val[vv] != val[u] ^ w:
                            consistent = False
                            break
                else:                         # u is a column
                    c = u - H
                    for v, w in col_adj[c]:
                        if val[v] == -1:
                            val[v] = val[u] ^ w
                            stack.append(v)
                        elif val[v] != val[u] ^ w:
                            consistent = False
                            break
            if not consistent:
                break

        if not consistent:
            out_lines.append('0')
        else:
            ans = pow(2, comp_cnt, MOD)
            out_lines.append(str(ans))

    sys.stdout.write('\n'.join(out_lines))

if __name__ == "__main__":
    solve()
