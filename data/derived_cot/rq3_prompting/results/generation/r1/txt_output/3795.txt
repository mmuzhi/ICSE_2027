from typing import List

class Solution:
    def minZeroArray(self, nums: List[int], queries: List[List[int]]) -> int:
        n = len(nums)
        m = len(queries)

        # If the array is already zero, no queries needed
        if all(x == 0 for x in nums):
            return 0

        # Feasibility check for first k queries
        def feasible(k: int) -> bool:
            for idx in range(n):
                target = nums[idx]
                # Use a bitset (as integer) for subset sum DP
                bits = 1  # only sum 0 is possible initially
                for i in range(k):
                    l, r, val = queries[i]
                    if l <= idx <= r:
                        bits |= bits << val
                        # Keep only bits up to target to avoid huge numbers
                        # (not necessary for given constraints, but good practice)
                        bits &= (1 << (target + 1)) - 1
                # Check if target sum is achievable
                if (bits >> target) & 1 == 0:
                    return False
            return True

        # Binary search for the smallest k
        lo, hi = 0, m
        while lo < hi:
            mid = (lo + hi) // 2
            if feasible(mid):
                hi = mid
            else:
                lo = mid + 1

        # After loop, lo is the first candidate. Check if it actually works.
        return lo if feasible(lo) else -1
