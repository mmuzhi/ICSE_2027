- First, Aoki specifies K distinct vertices on the tree.
- Then, Takahashi constructs a walk that starts and ends at vertex 1, and passes through all the vertices specified by Aoki.

Definition of a walk
    A walk on an undirected graph (possibly a tree) is a sequence of k vertices and k-1 edges v_1,e_1,v_2,\ldots,v_{k-1},e_{k-1},v_k (where k is a positive integer)
    such that edge e_i connects vertices v_i and v_{i+1}. The same vertex or edge can appear multiple times in the sequence.  
    A walk is said to pass through vertex x if there exists at least one i (1\leq i\leq k) such that v_i=x. (There can be multiple such i.)  
    The walk is said to start and end at v_1 and v_k, respectively, and the length of the walk is the sum of the lengths of e_1, e_2, \ldots, e_{k-1}.

```python
def solve():
```

```python
import sys
import heapq

def solve():
    import sys
    sys.setrecursionlimit(1 << 25)
    input = sys.stdin.readline
    N = int(input())
    adj = [[] for _ in range(N+1)]
    for _ in range(N-1):
        u,v,l = map(int,input().split())
        adj[u].append((v,l))
        adj[v].append((u,l))
    parent = [0]*(N+1)
    weight_to_parent = [0]*(N+1)
    children = [[] for _ in range(N+1)]
    # DFS iterative
    stack = [1]
    parent[1] = -1
    order = []
    while stack:
        u = stack.pop()
        order.append(u)
        for v,w in adj[u]:
            if v == parent[u]:
                continue
            parent[v] = u
            weight_to_parent[v] = w
            children[u].append(v)
            stack.append(v)
    # post-order DP
    farthest = [0]*(N+1)
    best_child = [None]*(N+1)
    for u in reversed(order):
        if children[u]:
            max_d = -1
            best = None
            for v in children[u]:
                d = weight_to_parent[v] + farthest[v]
                if d > max_d:
                    max_d = d
                    best = v
            farthest[u] = max_d
            best_child[u] = best
        # else leaf, already 0, None
    total_weight = sum(weight_to_parent[2:])  # sum of all edge weights
    # simulate greedy
    heap = []
    for v in children[1]:
        d = weight_to_parent[v] + farthest[v]
        heapq.heappush(heap, (-d, v))
    gains = []
    while heap:
        neg_d, node = heapq.heappop(heap)
        d = -neg_d
        gains.append(d)
        current = node
        while current is not None:
            # push other children
            for c in children[current]:
                if c == best_child[current]:
                    continue
                dc = weight_to_parent[c] + farthest[c]
                heapq.heappush(heap, (-dc, c))
            current = best_child[current]
    # gains already descending
    L = len(gains)
    pref = [0]*(L+1)
    for i in range(L):
        pref[i+1] = pref[i] + gains[i]
    out = []
    for k in range(1, N+1):
        if k <= L:
            ans = 2 * pref[k]
        else:
            ans = 2 * total_weight
        out.append(str(ans))
    sys.stdout.write("\n".join(out))
```

```python
import sys
import heapq

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    adj = [[] for _ in range(N + 1)]
    for _ in range(N - 1):
        u, v, l = map(int, input().split())
        adj[u].append((v, l))
        adj[v].append((u, l))

    # Root the tree at 1 using iterative DFS
    parent = [0] * (N + 1)
    weight_to_parent = [0] * (N + 1)
    children = [[] for _ in range(N + 1)]
    stack = [1]
    parent[1] = -1
    order = []                     # preorder
    while stack:
        u = stack.pop()
        order.append(u)
        for v, w in adj[u]:
            if v == parent[u]:
                continue
            parent[v] = u
            weight_to_parent[v] = w
            children[u].append(v)
            stack.append(v)

    # Post-order DP to compute farthest leaf distance and the child that achieves it
    farthest = [0] * (N + 1)
    best_child = [None] * (N + 1)
    for u in reversed(order):
        if not children[u]:
            continue               # leaf: farthest[u]=0, best_child[u]=None (already set)
        max_dist = -1
        best = None
        for v in children[u]:
            d = weight_to_parent[v] + farthest[v]
            if d > max_dist:
                max_dist = d
                best = v
        farthest[u] = max_dist
        best_child[u] = best

    total_weight = sum(weight_to_parent[2:])   # sum of all edge lengths

    # Greedy simulation to collect marginal gains (already sorted descending by the heap)
    heap = []
    for v in children[1]:
        d = weight_to_parent[v] + farthest[v]
        heapq.heappush(heap, (-d, v))          # max-heap via negative

    gains = []
    while heap:
        neg_d, node = heapq.heappop(heap)
        d = -neg_d
        gains.append(d)
        # Walk down the chosen longest path from this node
        cur = node
        while cur is not None:
            # Push all other children of cur (the ones not on the chosen path)
            for c in children[cur]:
                if c == best_child[cur]:
                    continue
                dc = weight_to_parent[c] + farthest[c]
                heapq.heappush(heap, (-dc, c))
            cur = best_child[cur]

    # gains are in decreasing order
    L = len(gains)
    pref = [0] * (L + 1)
    for i in range(L):
        pref[i + 1] = pref[i] + gains[i]

    out_lines = []
    for k in range(1, N + 1):
        if k <= L:
            ans = 2 * pref[k]
        else:
            ans = 2 * total_weight
        out_lines.append(str(ans))

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
