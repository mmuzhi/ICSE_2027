```python
import sys
from collections import deque

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))

    adj = [[] for _ in range(N + 1)]
    incoming_to_1 = []          # vertices u such that u -> 1 exists

    for _ in range(M):
        a = int(next(it))
        b = int(next(it))
        adj[a].append(b)
        if b == 1:
            incoming_to_1.append(a)

    # BFS from vertex 1
    dist = [-1] * (N + 1)
    dq = deque()
    dist[1] = 0
    dq.append(1)

    while dq:
        v = dq.popleft()
        for w in adj[v]:
            if dist[w] == -1:
                dist[w] = dist[v] + 1
                dq.append(w)

    ans = float('inf')
    for u in incoming_to_1:
        if dist[u] != -1:
            ans = min(ans, dist[u] + 1)

    print(-1 if ans == float('inf') else ans)


if __name__ == "__main__":
    solve()
```
