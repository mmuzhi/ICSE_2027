```python
MOD = 998244353

def precompute_factorials(n):
    fact = [1]*(n+1)
    for i in range(1,n+1):
        fact[i] = fact[i-1]*i % MOD
    inv_fact = [1]*(n+1)
    inv_fact[n] = pow(fact[n], MOD-2, MOD)
    for i in range(n,0,-1):
        inv_fact[i-1] = inv_fact[i]*i % MOD
    return fact, inv_fact

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input().strip())
    A = list(map(int, input().split()))
    # Quick checks
    if A[0] != 1 or A[-1] != (N%2):
        print(0)
        return
    # Build runs
    runs = []
    cur_len = 1
    for i in range(1,N):
        if A[i] == A[i-1]:
            cur_len += 1
        else:
            # check alternation: A[i] should be 1 - A[i-1]
            if A[i] != 1 - A[i-1]:
                print(0)
                return
            runs.append(cur_len)
            cur_len = 1
    runs.append(cur_len)
    # Check each run length odd
    for L in runs:
        if L % 2 == 0:
            print(0)
            return
    R = len(runs)
    K_total = (N - R)//2
    # Precompute factorials up to N (since we need up to 2Kb <= N-1)
    max_fact = N  # N up to 2e5, enough
    fact, inv_fact = precompute_factorials(max_fact)
    # Precompute inv_pow2 up to maxK
    maxK = (N-1)//2
    inv_pow2 = [1]*(maxK+1)
    inv2 = (MOD+1)//2
    for i in range(1,maxK+1):
        inv_pow2[i] = inv_pow2[i-1]*inv2 % MOD
    ans = 1
    for L in runs:
        Kb = (L-1)//2
        if Kb == 0:
            continue
        term = fact[2*Kb] * inv_pow2[Kb] % MOD
        term = term * inv_fact[Kb] % MOD * inv_fact[Kb] % MOD
        ans = ans * term % MOD
    ans = ans * fact[K_total] % MOD
    print(ans)
```

```python
import sys

MOD = 998244353

def precompute_factorials(n):
    fact = [1] * (n + 1)
    for i in range(1, n + 1):
        fact[i] = fact[i - 1] * i % MOD
    inv_fact = [1] * (n + 1)
    inv_fact[n] = pow(fact[n], MOD - 2, MOD)
    for i in range(n, 0, -1):
        inv_fact[i - 1] = inv_fact[i] * i % MOD
    return fact, inv_fact

def solve():
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    # endpoints are fixed
    if A[0] != 1 or A[-1] != (N % 2):
        print(0)
        return

    # build run lengths and check alternation
    runs = []
    cur_len = 1
    for i in range(1, N):
        if A[i] == A[i - 1]:
            cur_len += 1
        else:
            # must alternate
            if A[i] != 1 - A[i - 1]:
                print(0)
                return
            runs.append(cur_len)
            cur_len = 1
    runs.append(cur_len)

    # each run length must be odd
    for L in runs:
        if L % 2 == 0:
            print(0)
            return

    R = len(runs)
    K_total = (N - R) // 2

    # precompute factorials up to N (needed for 2*Kb ≤ N-1)
    max_fact = N
    fact, inv_fact = precompute_factorials(max_fact)

    # precompute powers of inverse 2
    max_K = (N - 1) // 2
    inv2 = (MOD + 1) // 2
    inv_pow2 = [1] * (max_K + 1)
    for i in range(1, max_K + 1):
        inv_pow2[i] = inv_pow2[i - 1] * inv2 % MOD

    ans = 1
    for L in runs:
        Kb = (L - 1) // 2
        if Kb == 0:
            continue
        # term = (2Kb)! / (2^{Kb} * (Kb)!^2)   modulo MOD
        term = fact[2 * Kb] * inv_pow2[Kb] % MOD
        term = term * inv_fact[Kb] % MOD * inv_fact[Kb] % MOD
        ans = ans * term % MOD

    ans = ans * fact[K_total] % MOD
    print(ans)

if __name__ == "__main__":
    solve()
```
