import sys
from fractions import Fraction

MOD = 998244353

def solve() -> None:
    it = iter(sys.stdin.read().split())
    t = int(next(it))
    out = []
    for _ in range(t):
        n = int(next(it))
        p = [int(next(it)) for _ in range(n)]
        a = [int(next(it)) for _ in range(n)]

        # children lists
        children = [[] for _ in range(n + 1)]
        for i, par in enumerate(p, start=1):
            children[par].append(i)

        # initialise arrays: S (total weight in subtree), sz (size), T (weighted sum modulo MOD)
        S = [0] * (n + 1)
        sz = [1] * (n + 1)
        T = [0] * (n + 1)

        # set weights for nodes 1..n
        for i in range(1, n + 1):
            S[i] = a[i - 1]

        # process nodes in decreasing index order (children before parent)
        for v in range(n, -1, -1):
            # accumulate children data
            for c in children[v]:
                S[v] += S[c]
                sz[v] += sz[c]

            # sort children by descending S/sz (exact rational comparison)
            if children[v]:
                # key = Fraction( S[c], sz[c] )
                children[v].sort(key=lambda c: Fraction(S[c], sz[c]), reverse=True)

            # compute T[v] modulo MOD
            pref = 0
            val = 0
            for c in children[v]:
                # e_c = S[c] + T[c]   (mod MOD)
                e_c = (S[c] + T[c]) % MOD
                contrib = ( (S[c] % MOD) * (pref % MOD) + e_c ) % MOD
                val = (val + contrib) % MOD
                pref += sz[c]
            T[v] = val

        total = S[0] % MOD
        ans = T[0] * pow(total, MOD - 2, MOD) % MOD
        out.append(str(ans))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
