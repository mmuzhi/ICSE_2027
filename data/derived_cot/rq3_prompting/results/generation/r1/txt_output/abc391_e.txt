import sys

def solve() -> None:

    N = int(input().strip())
    A = input().strip()
    L = len(A)                      # = 3^N

    # dp for leaves: (cost to become 0, cost to become 1)
    cost0 = [0] * L
    cost1 = [0] * L
    for i, ch in enumerate(A):
        if ch == '0':
            cost0[i] = 0
            cost1[i] = 1
        else:
            cost0[i] = 1
            cost1[i] = 0

    # build levels bottom-up
    for _ in range(N):
        new_len = len(cost0) // 3
        new_cost0 = [0] * new_len
        new_cost1 = [0] * new_len
        INF = 10 ** 9               # large enough (max cost ≤ 3^13 ≈ 1.6e6)

        for i in range(new_len):
            base = i * 3
            # children's costs
            c00, c01 = cost0[base], cost1[base]
            c10, c11 = cost0[base + 1], cost1[base + 1]
            c20, c21 = cost0[base + 2], cost1[base + 2]

            best0 = INF
            best1 = INF

            # try all 2^3 assignments of values to the three children
            for v0 in (0, 1):
                for v1 in (0, 1):
                    for v2 in (0, 1):
                        total = (c00 if v0 == 0 else c01) + \
                                (c10 if v1 == 0 else c11) + \
                                (c20 if v2 == 0 else c21)
                        majority = 1 if (v0 + v1 + v2) >= 2 else 0
                        if majority == 0:
                            if total < best0:
                                best0 = total
                        else:
                            if total < best1:
                                best1 = total

            new_cost0[i] = best0
            new_cost1[i] = best1

        cost0 = new_cost0
        cost1 = new_cost1

    # now single element – the root
    if cost0[0] == 0:
        # original root is 0 → need to flip to 1
        print(cost1[0])
    else:
        # original root is 1 → need to flip to 0
        print(cost0[0])
