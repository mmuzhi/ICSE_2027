import sys
from collections import Counter

def solve():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    B = [int(next(it)) for _ in range(N)]
    freqA = Counter()
    freqB = Counter()
    for a in A:
        if a != -1:
            freqA[a] += 1
    for b in B:
        if b != -1:
            freqB[b] += 1
    aN = sum(freqA.values())
    bN = sum(freqB.values())
    # Quick check: if we can avoid fixed-fixed pairs
    if aN <= N - bN:
        print("Yes")
        return
    # Otherwise, need to pair some
    K_min = aN + bN - N
    max_fixed = 0
    if freqA:
        max_fixed = max(max_fixed, max(freqA))
    if freqB:
        max_fixed = max(max_fixed, max(freqB))
    # Build dictionary for sums
    match = {}
    for a, cnt_a in freqA.items():
        for b, cnt_b in freqB.items():
            S = a + b
            if S < max_fixed:
                continue
            add = cnt_a if cnt_a < cnt_b else cnt_b  # min
            new_val = match.get(S, 0) + add
            if new_val >= K_min:
                print("Yes")
                return
            match[S] = new_val
    print("No")

if __name__ == "__main__":
    solve()

import sys
from collections import Counter

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    B = [int(next(it)) for _ in range(N)]

    # count fixed values
    cntA = Counter()
    cntB = Counter()
    for a in A:
        if a != -1:
            cntA[a] += 1
    for b in B:
        if b != -1:
            cntB[b] += 1

    na = sum(cntA.values())
    nb = sum(cntB.values())

    # Case: we can avoid any fixed‑fixed pair
    if na <= N - nb:
        print("Yes")
        return

    # Otherwise we need to pair some fixed A with fixed B
    K = na + nb - N          # minimum number of pairs required
    max_val = 0
    if cntA:
        max_val = max(max_val, max(cntA))
    if cntB:
        max_val = max(max_val, max(cntB))

    sum_dict = {}
    for a, ca in cntA.items():
        for b, cb in cntB.items():
            s = a + b
            if s < max_val:          # S must be at least max_val
                continue
            pairs = ca if ca < cb else cb   # min(ca, cb)
            new_val = sum_dict.get(s, 0) + pairs
            if new_val >= K:
                print("Yes")
                return
            sum_dict[s] = new_val

    print("No")

if __name__ == "__main__":
    solve()
