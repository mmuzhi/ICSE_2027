import math

def count_numbers_with_exactly_9_divisors(N):
    # sqrt(N) bound for product of primes
    limit = math.isqrt(N)
    
    # Sieve of Eratosthenes to generate all primes up to limit
    is_prime = [True] * (limit + 1)
    is_prime[0] = is_prime[1] = False
    for i in range(2, int(math.isqrt(limit)) + 1):
        if is_prime[i]:
            step = i
            start = i * i
            is_prime[start:limit+1:step] = [False] * ((limit - start)//step + 1)
    primes = [i for i, val in enumerate(is_prime) if val]
    
    # Count numbers of form p^8
    count8 = 0
    for p in primes:
        if p**8 <= N:
            count8 += 1
        else:
            break
    
    # Count numbers of form p^2 * q^2 (p < q)
    # Equivalent to count pairs (p,q) distinct with p*q <= limit
    count_pairs = 0
    m = len(primes)
    j = m - 1
    for i, p in enumerate(primes):
        max_q = limit // p
        # Move j left while primes[j] > max_q
        while j >= 0 and primes[j] > max_q:
            j -= 1
        if j > i:
            count_pairs += j - i
        else:
            # For larger p, max_q will only become smaller, so j will never be > i again
            break
    
    return count8 + count_pairs

import math

def count_numbers_with_exactly_9_divisors(N: int) -> int:
    # Upper bound for product p*q, because (p*q)^2 <= N
    limit = math.isqrt(N)

    # Sieve of Eratosthenes: primes up to limit
    is_prime = [True] * (limit + 1)
    is_prime[0] = is_prime[1] = False
    for i in range(2, math.isqrt(limit) + 1):
        if is_prime[i]:
            step = i
            start = i * i
            is_prime[start:limit+1:step] = [False] * ((limit - start) // step + 1)
    primes = [i for i, val in enumerate(is_prime) if val]

    # 1) numbers of the form p^8
    cnt_8 = 0
    for p in primes:
        if p ** 8 <= N:
            cnt_8 += 1
        else:
            break

    # 2) numbers of the form p^2 * q^2 (p < q)
    cnt_pairs = 0
    m = len(primes)
    j = m - 1
    for i, p in enumerate(primes):
        max_q = limit // p
        while j >= 0 and primes[j] > max_q:
            j -= 1
        if j > i:
            cnt_pairs += j - i
        else:
            # for larger p, max_q will only decrease, j never becomes > i again
            break

    return cnt_8 + cnt_pairs
