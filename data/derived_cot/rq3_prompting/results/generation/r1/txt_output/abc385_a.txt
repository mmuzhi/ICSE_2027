import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    A, B, C = map(int, data[:3])

    # Condition 1: three equal numbers -> three groups each of size 1
    if A == B == C:
        print("Yes")
        return

    # Condition 2: one number equals the sum of the other two -> two groups
    if A == B + C or B == A + C or C == A + B:
        print("Yes")
        return

    print("No")

if __name__ == "__main__":
    solve()
