class Solution:
    def makeStringGood(self, s: str) -> int:
        # count frequencies
        cnt = [0] * 26
        for ch in s:
            cnt[ord(ch) - 97] += 1
        n = len(s)

        # generate candidate k values
        candidates = set()
        # all existing frequencies
        for c in cnt:
            if c > 0:
                candidates.add(c)
        # averages of the largest m frequencies
        sorted_cnt = sorted(cnt, reverse=True)
        pref_sum = 0
        for m in range(1, 27):
            if m - 1 >= len(sorted_cnt):
                break
            pref_sum += sorted_cnt[m - 1]
            candidates.add(pref_sum // m)
            candidates.add((pref_sum + m - 1) // m)
        # ensure at least one candidate (k=1 is always reasonable)
        candidates.add(1)
        # only k in [1, n] make sense
        candidates = {k for k in candidates if 1 <= k <= n}

        best = n  # an upper bound
        for k in candidates:
            cost = self._min_cost(cnt, n, k)
            if cost < best:
                best = cost
        return best

    def _min_cost(self, cnt: list, n: int, k: int) -> int:
        N = n
        INF = 10 ** 9
        dp = [INF] * (N + 1)
        dp[0] = 0

        # process letters from 'z' down to 'a'
        for i in range(25, -1, -1):
            # prefix minima of dp
            pref = [0] * (N + 1)
            pref[0] = dp[0]
            for t in range(1, N + 1):
                pref[t] = pref[t - 1] if pref[t - 1] < dp[t] else dp[t]

            ndp = [0] * (N + 1)
            c = cnt[i]
            for x in range(N + 1):
                total = c + x
                if total > N:
                    extra = total - N
                    total = N
                else:
                    extra = 0

                # option: do not keep this letter
                cost_not_keep = extra + total + pref[total]

                # option: keep this letter (target frequency k)
                limit = max(0, total - k)
                cost_keep = extra + abs(total - k) + pref[limit]

                ndp[x] = cost_not_keep if cost_not_keep < cost_keep else cost_keep

            dp = ndp

        return dp[0]
