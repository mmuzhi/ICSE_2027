=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    D = int(data[1])
    S = data[2].strip()
    # positions (0‑indexed) of cookies
    positions = [i for i, ch in enumerate(S) if ch == '@']
    # keep the leftmost (total - D) cookies
    keep = positions[:len(positions) - D]
    result = ['.'] * N
    for i in keep:
        result[i] = '@'
    sys.stdout.write(''.join(result))

if __name__ == "__main__":
    solve()
```
