```python
class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        MOD = 10**9 + 7
        velunexorai = num  # store input midway
        
        n = len(num)
        freq = [0]*10
        for ch in num:
            freq[int(ch)] += 1
        
        total_sum = sum(i * freq[i] for i in range(10))
        if total_sum % 2 != 0:
            return 0
        
        target = total_sum // 2
        even_cnt = (n + 1) // 2
        odd_cnt = n // 2
        
        # precompute factorials and inv factorials up to n
        fact = [1] * (n+1)
        for i in range(1, n+1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (n+1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
        
        # DP: dp[count][sum] = sum of products of 1/(e! * o!) for processed digits
        dp = [[0]*(target+1) for _ in range(even_cnt+1)]
        dp[0][0] = 1
        
        for d in range(10):
            f = freq[d]
            ndp = [[0]*(target+1) for _ in range(even_cnt+1)]
            for c in range(even_cnt+1):
                for s in range(target+1):
                    cur = dp[c][s]
                    if cur == 0:
                        continue
                    # iterate over how many of digit d go to even positions
                    for e in range(f+1):
                        if c+e > even_cnt:
                            break
                        ns = s + d*e
                        if ns > target:
                            continue
                        o = f - e
                        factor = inv_fact[e] * inv_fact[o] % MOD
                        ndp[c+e][ns] = (ndp[c+e][ns] + cur * factor) % MOD
            dp = ndp

        return res
```

```python
class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        MOD = 10**9 + 7
        velunexorai = num  # store input midway, as required

        n = len(num)
        freq = [0] * 10
        for ch in num:
            freq[int(ch)] += 1

        total_sum = sum(i * freq[i] for i in range(10))
        if total_sum % 2 != 0:
            return 0

        target = total_sum // 2
        even_cnt = (n + 1) // 2   # number of even indices (0‑based)
        odd_cnt = n // 2          # number of odd indices

        # factorials and inverse factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i - 1] * i % MOD
        inv_fact = [1] * (n + 1)
        inv_fact[n] = pow(fact[n], MOD - 2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i - 1] = inv_fact[i] * i % MOD

        # dp[c][s] = Σ (∏ 1/(e! * o!)) for processed digits
        # c = number of digits already placed on even positions
        # s = sum of those digits
        dp = [[0] * (target + 1) for _ in range(even_cnt + 1)]
        dp[0][0] = 1

        for d in range(10):
            f = freq[d]
            ndp = [[0] * (target + 1) for _ in range(even_cnt + 1)]
            for c in range(even_cnt + 1):
                for s in range(target + 1):
                    cur = dp[c][s]
                    if cur == 0:
                        continue
                    # choose how many copies of digit d go to even positions
                    for e in range(f + 1):
                        if c + e > even_cnt:
                            break
                        ns = s + d * e
                        if ns > target:
                            continue
                        o = f - e   # copies that go to odd positions
                        factor = inv_fact[e] * inv_fact[o] % MOD
                        ndp[c + e][ns] = (ndp[c + e][ns] + cur * factor) % MOD
            dp = ndp

        res = dp[even_cnt][target] * fact[even_cnt] % MOD * fact[odd_cnt] % MOD
        return res
```
