import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    total = sum(A)
    ans = (N + 1) * total               # T0 = sum_{i<=j} (Ai+Aj)
    max_val = max(A)
    two_max = 2 * max_val

    for k in range(1, 31):             # 2^30 > 1e9, enough
        mod = 1 << k
        if mod > two_max:
            break

        # build dictionary: residue -> [count, sum]
        cnt = {}
        mask = mod - 1
        for a in A:
            r = a & mask               # a % mod for power of two
            if r in cnt:
                cnt[r][0] += 1
                cnt[r][1] += a
            else:
                cnt[r] = [1, a]

        Tk = 0
        for r, (c, s) in cnt.items():
            # complementary residue
            comp = (mod - r) & mask
            if r <= comp:
                if r == comp:
                    # pairs inside the same residue (i <= j)
                    Tk += (c + 1) * s
                else:
                    # pairs between two different residues
                    if comp in cnt:
                        c2, s2 = cnt[comp]
                        Tk += c * s2 + c2 * s

        ans -= Tk // mod

    print(ans)


if __name__ == "__main__":
    solve()
