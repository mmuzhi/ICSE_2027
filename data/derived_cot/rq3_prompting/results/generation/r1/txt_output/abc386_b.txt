```python
import sys

def solve() -> None:
    s = sys.stdin.readline().strip()
    n = len(s)
    dp = [0] * (n + 1)  # dp[0] = 0
    for i in range(1, n + 1):
        # single digit press
        dp[i] = dp[i - 1] + 1
        # check for "00" double press
        if i >= 2 and s[i-2:i] == "00":
            dp[i] = min(dp[i], dp[i - 2] + 1)
    print(dp[n])

if __name__ == "__main__":
    solve()
```
