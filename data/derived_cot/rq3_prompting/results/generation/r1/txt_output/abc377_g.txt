def solve():
    import sys
    input = sys.stdin.readline
    N = int(input().strip())
    strings = [input().strip() for _ in range(N)]
    children = [dict()]  # root
    best = [-10**9]
    out = []
    for s in strings:
        # query first
        node = 0
        max_val = best[0]
        for ch in s:
            if ch in children[node]:
                node = children[node][ch]
                if best[node] > max_val:
                    max_val = best[node]
            else:
                break
        ans = len(s) - max(0, max_val)
        out.append(str(ans))
        # insert
        node = 0
        l = len(s)
        best[0] = max(best[0], -l)
        for i,ch in enumerate(s):
            if ch not in children[node]:
                children[node][ch] = len(children)
                children.append(dict())
                best.append(-10**9)
            node = children[node][ch]
            d = i+1
            val = 2*d - l
            if val > best[node]:
                best[node] = val
    sys.stdout.write("\n".join(out))

```python
import sys

def solve() -> None:
    it = iter(sys.stdin.read().split())
    n = int(next(it))
    strings = [next(it) for _ in range(n)]

    # Trie nodes: children[i] is a dict char->node_index
    children = [dict()]          # root = 0
    best = [-10**9]             # best value for each node: 2*depth - len

    out_lines = []

    for s in strings:
        # ----- query -----
        node = 0
        max_val = best[0]          # root value (always negative after first insertion)
        for ch in s:
            if ch in children[node]:
                node = children[node][ch]
                if best[node] > max_val:
                    max_val = best[node]
            else:
                break
        cost = len(s) - max(0, max_val)
        out_lines.append(str(cost))

        # ----- insert -----
        node = 0
        l = len(s)
        # update root
        if -l > best[0]:
            best[0] = -l
        for i, ch in enumerate(s):
            if ch not in children[node]:
                children[node][ch] = len(children)
                children.append(dict())
                best.append(-10**9)   # initialize new node
            node = children[node][ch]
            d = i + 1
            val = 2 * d - l
            if val > best[node]:
                best[node] = val

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
