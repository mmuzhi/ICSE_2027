=== ANSWER ===
```python
from typing import List

class Solution:
    def countArrays(self, original: List[int], bounds: List[List[int]]) -> int:
        n = len(original)
        # Initialize cumulative sum S_i = sum_{j=1..i} (original[j] - original[j-1])
        S = 0
        # L_max and R_min
        L_max = bounds[0][0] - S   # i=0
        R_min = bounds[0][1] - S
        for i in range(1, n):
            d = original[i] - original[i-1]
            S += d
            L_i = bounds[i][0] - S
            R_i = bounds[i][1] - S
            if L_i > L_max:
                L_max = L_i
            if R_i < R_min:
                R_min = R_i
        # Number of integer x satisfying L_max <= x <= R_min
        if R_min < L_max:
            return 0
        else:
            return R_min - L_max + 1
```
