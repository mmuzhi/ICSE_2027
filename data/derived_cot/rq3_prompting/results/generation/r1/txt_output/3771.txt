Given a string s of length n and an integer k, determine whether it is possible to select k disjoint special substrings.
A special substring is a substring where:

Any character present inside the substring should not appear outside it in the string.
The substring is not the entire string s.

Note that all k substrings must be disjoint, meaning they cannot overlap.
Return true if it is possible to select k such disjoint special substrings; otherwise, return false.

Input: s = "abcdbaefab", k = 2
Output: true
Explanation:

We can select two disjoint special substrings: "cd" and "ef".

"ef" contains the characters 'e' and 'f', which do not appear elsewhere in s.

Output: false
Explanation:
There can be at most 2 disjoint special substrings: "e" and "f". Since k = 3, the output is false.

Example 3:

Input: s = "abeabe", k = 0
Output: true

0 <= k <= 26
s consists only of lowercase English letters.

class Solution:
    def maxSubstringLength(self, s: str, k: int) -> bool:
        n = len(s)
        # first and last occurrence for each character (a-z)
        first = [-1] * 26
        last = [-1] * 26
        for i, ch in enumerate(s):
            idx = ord(ch) - 97
            if first[idx] == -1:
                first[idx] = i
            last[idx] = i
        
        intervals = []  # list of (left, right) for minimal special substrings
        
        for i in range(n):
            # only consider positions that are the first occurrence of their character
            idx = ord(s[i]) - 97
            if first[idx] != i:
                continue
            
            # start expanding to satisfy the "last" condition
            cur_right = last[idx]
            min_first = i          # track the smallest first occurrence in the interval
            j = i
            while j <= cur_right:
                c_idx = ord(s[j]) - 97
                if first[c_idx] < min_first:
                    min_first = first[c_idx]
                if last[c_idx] > cur_right:
                    cur_right = last[c_idx]
                j += 1
            
            # if the interval passes the "first" condition and is not the whole string, keep it
            if min_first == i and not (i == 0 and cur_right == n - 1):
                intervals.append((i, cur_right))
        
        # Greedy interval scheduling: maximize number of non-overlapping intervals
        intervals.sort(key=lambda x: x[1])   # sort by right endpoint
        count = 0
        last_end = -1
        for l, r in intervals:
            if l > last_end:
                count += 1
                last_end = r
                if count >= k:   # early exit
                    return True
        
        return count >= k
