=== ANSWER ===
```python
import sys

def main() -> None:
    # Read the four integers
    A = list(map(int, sys.stdin.read().split()))
    # Frequency array for colors 1..4 (index 0 unused)
    freq = [0] * 5
    for a in A:
        freq[a] += 1
    # Each pair of the same color contributes one operation
    ans = sum(f // 2 for f in freq)
    print(ans)

if __name__ == "__main__":
    main()
```
