=== ANSWER ===
```python
import sys

def main():
    X = sys.stdin.readline().strip()
    if '.' in X:
        integer_part, fractional_part = X.split('.')
        fractional_part = fractional_part.rstrip('0')
        if fractional_part:
            print(f"{integer_part}.{fractional_part}")
        else:
            print(integer_part)
    else:
        print(X)

if __name__ == "__main__":
    main()
```
