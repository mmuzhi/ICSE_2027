```python
from typing import List

class Solution:
    def countPairs(self, nums: List[int]) -> int:
        def get_reachable(x: int) -> set:
            """Return all integers reachable from x by swapping at most two digits."""
            s = str(x)
            n = len(s)
            res = {x}
            for i in range(n):
                for j in range(i + 1, n):
                    # Swap the digits at positions i and j
                    lst = list(s)
                    lst[i], lst[j] = lst[j], lst[i]
                    res.add(int(''.join(lst)))  # leading zeros are allowed
            return res
        
        # Precompute reachable sets for every number
        reachable = [get_reachable(num) for num in nums]
        
        ans = 0
        m = len(nums)
        for i in range(m):
            for j in range(i + 1, m):
                # Check if either number can become the other with at most one swap
                if nums[j] in reachable[i] or nums[i] in reachable[j]:
                    ans += 1
        return ans
```
