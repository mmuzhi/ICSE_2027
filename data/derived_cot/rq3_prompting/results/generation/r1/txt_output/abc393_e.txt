import sys

def main() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    MAX = max(A)
    freq = [0] * (MAX + 1)
    for a in A:
        freq[a] += 1

    # cnt[d] = number of array elements divisible by d
    cnt = [0] * (MAX + 1)
    for d in range(1, MAX + 1):
        s = 0
        for m in range(d, MAX + 1, d):
            s += freq[m]
        cnt[d] = s

    ans = [0] * (MAX + 1)

    # process divisors from large to small
    for d in range(MAX, 0, -1):
        if cnt[d] >= K:
            for m in range(d, MAX + 1, d):
                if ans[m] == 0:
                    ans[m] = d

    out = '\n'.join(str(ans[a]) for a in A)
    sys.stdout.write(out)

if __name__ == "__main__":
    main()
