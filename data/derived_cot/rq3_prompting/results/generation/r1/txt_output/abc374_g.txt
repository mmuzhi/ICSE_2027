import sys
from collections import deque

def solve():
    input = sys.stdin.readline
    N = int(input())
    outdeg = [0]*26
    indeg = [0]*26
    directed_adj = [[] for _ in range(26)]
    undirected_adj = [[] for _ in range(26)]
    for _ in range(N):
        s = input().strip()
        u = ord(s[0]) - 65
        v = ord(s[1]) - 65
        outdeg[u] += 1
        indeg[v] += 1
        directed_adj[u].append(v)
        if u != v:  # avoid self-loop for undirected
            undirected_adj[u].append(v)
            undirected_adj[v].append(u)
        else:
            # self-loop: does not connect to other vertices, but we need the vertex itself to be included.
            # We'll just mark that the vertex has edges; no need to add to undirected adjacency.
            pass
    visited = [False]*26
    total_ans = 0
    INF = 10**9
    for start in range(26):
        if (indeg[start] > 0 or outdeg[start] > 0) and not visited[start]:
            # BFS/DFS to collect component
            comp = []
            stack = [start]
            visited[start] = True
            while stack:
                u = stack.pop()
                comp.append(u)
                for w in undirected_adj[u]:
                    if not visited[w]:
                        visited[w] = True
                        stack.append(w)
            # compute S (positive supply sum)
            pos_sum = 0
            neg_cap = {}
            pos_cap = {}
            for u in comp:
                supply = outdeg[u] - indeg[u]
                if supply > 0:
                    pos_sum += supply
                    pos_cap[u] = supply
                elif supply < 0:
                    neg_cap[u] = -supply
            # Build flow network
            # nodes: original 0-25, source=26, sink=27
            cap = [[0]*28 for _ in range(28)]
            # source to negative supply (indeg>outdeg)
            for u, c in neg_cap.items():
                cap[26][u] = c
            # positive supply to sink
            for u, c in pos_cap.items():
                cap[u][27] = c
            # original directed edges (within component)
            comp_set = set(comp)
            for u in comp:
                for v in directed_adj[u]:
                    if v in comp_set:
                        cap[u][v] = INF
            # max flow from 26 to 27
            flow = 0
            # Edmonds-Karp (BFS augment)
            while True:
                parent = [-1]*28
                parent[26] = 26
                q = deque([26])
                while q:
                    cur = q.popleft()
                    if cur == 27:
                        break
                    for nxt in range(28):
                        if parent[nxt]==-1 and cap[cur][nxt] > 0:
                            parent[nxt] = cur
                            q.append(nxt)
                if parent[27]==-1:
                    break
                # find bottleneck
                bottleneck = INF
                v = 27
                while v != 26:
                    u = parent[v]
                    bottleneck = min(bottleneck, cap[u][v])
                    v = u
                # update
                v = 27
                while v != 26:
                    u = parent[v]
                    cap[u][v] -= bottleneck
                    cap[v][u] += bottleneck
                    v = u
                flow += bottleneck
            # walks for component
            walks = max(1, pos_sum - flow)
            total_ans += walks
    print(total_ans)

if __name__ == "__main__":
    solve()
