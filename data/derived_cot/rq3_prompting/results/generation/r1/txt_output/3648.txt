from typing import List

class Solution:
    def maxCollectedFruits(self, fruits: List[List[int]]) -> int:
        n = len(fruits)
        # diagonal sum (child A)
        diag_sum = sum(fruits[i][i] for i in range(n))
        
        INF_NEG = -10**15
        
        # DP for child B (top-right to bottom-right, moving down)
        dpB = [[INF_NEG] * n for _ in range(n)]
        dpB[0][n-1] = fruits[0][n-1]  # start cell (not diagonal)
        for i in range(1, n):
            for j in range(n):
                best = INF_NEG
                if j-1 >= 0:
                    best = max(best, dpB[i-1][j-1])
                best = max(best, dpB[i-1][j])
                if j+1 < n:
                    best = max(best, dpB[i-1][j+1])
                # diagonal cell adds 0 (already counted)
                val = fruits[i][j] if i != j else 0
                dpB[i][j] = best + val if best != INF_NEG else INF_NEG
        B_max = dpB[n-1][n-1]
        
        # DP for child C (bottom-left to bottom-right, moving right)
        dpC = [[INF_NEG] * n for _ in range(n)]
        dpC[n-1][0] = fruits[n-1][0]  # start cell (not diagonal)
        for j in range(1, n):
            for i in range(n):
                best = INF_NEG
                if i-1 >= 0:
                    best = max(best, dpC[i-1][j-1])
                best = max(best, dpC[i][j-1])
                if i+1 < n:
                    best = max(best, dpC[i+1][j-1])
                val = fruits[i][j] if i != j else 0
                dpC[i][j] = best + val if best != INF_NEG else INF_NEG
        C_max = dpC[n-1][n-1]
        
        return diag_sum + B_max + C_max
