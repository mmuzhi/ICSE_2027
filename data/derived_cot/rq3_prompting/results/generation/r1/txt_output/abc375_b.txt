=== ANSWER ===
```python
import sys
import math

def main() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    total = 0.0
    # start from origin
    prev_x = 0
    prev_y = 0
    # read points and accumulate distances
    for _ in range(N):
        x = int(next(it))
        y = int(next(it))
        dx = x - prev_x
        dy = y - prev_y
        total += math.hypot(dx, dy)
        prev_x, prev_y = x, y
    # return to origin
    total += math.hypot(prev_x, prev_y)
    # output with enough precision
    # using 15 decimal places is safe for error tolerance
    print("{:.15f}".format(total))

if __name__ == "__main__":
    main()
```
