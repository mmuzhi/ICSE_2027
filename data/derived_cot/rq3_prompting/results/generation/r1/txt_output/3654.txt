```python
from typing import List

class Solution:
    def minArraySum(self, nums: List[int], k: int, op1: int, op2: int) -> int:
        INF = 10**18
        n = len(nums)
        # DP table: dp[a][b] = minimum sum after processing some elements
        # using exactly a op1 and b op2 (0 <= a <= op1, 0 <= b <= op2)
        dp = [[INF] * (op2 + 1) for _ in range(op1 + 1)]
        dp[0][0] = 0

        for x in nums:
            # generate options for this element
            options = []
            # no operation
            options.append((x, 0, 0))
            # only op1
            options.append(((x + 1) // 2, 1, 0))
            # only op2
            if x >= k:
                options.append((x - k, 0, 1))
            # both op1 and op2 (try both orders, take the best value)
            best_both = None
            # order: op1 then op2
            after_op1 = (x + 1) // 2
            if after_op1 >= k:
                val = after_op1 - k
                if best_both is None or val < best_both:
                    best_both = val
            # order: op2 then op1
            if x >= k:
                after_op2 = x - k
                val = (after_op2 + 1) // 2
                if best_both is None or val < best_both:
                    best_both = val
            if best_both is not None:
                options.append((best_both, 1, 1))

            # transition to new DP
            newdp = [[INF] * (op2 + 1) for _ in range(op1 + 1)]
            for a in range(op1 + 1):
                for b in range(op2 + 1):
                    cur = dp[a][b]
                    if cur == INF:
                        continue
                    for val, c1, c2 in options:
                        na = a + c1
                        nb = b + c2
                        if na <= op1 and nb <= op2:
                            if cur + val < newdp[na][nb]:
                                newdp[na][nb] = cur + val
            dp = newdp

        # answer is the minimum over all states (we may not use all operations)
        ans = INF
        for a in range(op1 + 1):
            for b in range(op2 + 1):
                if dp[a][b] < ans:
                    ans = dp[a][b]
        return ans
```
