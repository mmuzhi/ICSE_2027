```python
import heapq

class Solution:
    def minOperations(self, n: int, m: int) -> int:
        # sieve up to 10000
        N = 10000
        is_prime = [True] * (N+1)
        is_prime[0] = is_prime[1] = False
        for i in range(2, int(N**0.5)+1):
            if is_prime[i]:
                for j in range(i*i, N+1, i):
                    is_prime[j] = False
        
        # initial check
        if is_prime[n] or is_prime[m]:
            return -1
        if n == m:
            return n
        
        d = len(str(n))
        L = 10**(d-1)
        R = 10**d - 1
        
        INF = 10**18
        dist = [INF] * (R+1)
        dist[n] = n
        pq = [(n, n)]  # (cost, node)
        
        while pq:
            cost, u = heapq.heappop(pq)
            if cost != dist[u]:
                continue
            if u == m:
                return cost
            # generate neighbors
            for i in range(d):
                digit = (u // (10**i)) % 10
                # increase
                if digit < 9:
                    new_u = u + (10**i)
                    # check leading zero: if i == d-1 and digit+1 == 0? never, digit>=0 so increase not leading zero
                    if L <= new_u <= R and not is_prime[new_u]:
                        new_cost = cost + new_u
                        if new_cost < dist[new_u]:
                            dist[new_u] = new_cost
                            heapq.heappush(pq, (new_cost, new_u))
                # decrease
                if digit > 0:
                    # avoid leading zero: if i == d-1 and digit == 1 then new_u would be < L
                    if i == d-1 and digit == 1:
                        continue
                    new_u = u - (10**i)
                    if L <= new_u <= R and not is_prime[new_u]:
                        new_cost = cost + new_u
                        if new_cost < dist[new_u]:
                            dist[new_u] = new_cost
                            heapq.heappush(pq, (new_cost, new_u))
        return -1
```

```python
import heapq

class Solution:
    def minOperations(self, n: int, m: int) -> int:
        # ---------- prime sieve up to 9999 ----------
        MAX = 10000
        is_prime = [True] * (MAX + 1)
        is_prime[0] = is_prime[1] = False
        for i in range(2, int(MAX**0.5) + 1):
            if is_prime[i]:
                step = i
                start = i * i
                is_prime[start:MAX+1:step] = [False] * (((MAX - start) // step) + 1)

        # start or target is prime -> impossible
        if is_prime[n] or is_prime[m]:
            return -1
        if n == m:
            return n

        # number of digits
        d = len(str(n))
        L = 10 ** (d - 1)          # smallest number with d digits
        R = 10 ** d - 1            # largest number with d digits

        INF = 10 ** 18
        dist = [INF] * (R + 1)
        dist[n] = n
        pq = [(n, n)]              # (cost, node)

        while pq:
            cost, u = heapq.heappop(pq)
            if cost != dist[u]:
                continue
            if u == m:
                return cost

            # generate neighbours by changing one digit
            for i in range(d):
                digit = (u // (10 ** i)) % 10
                # increase
                if digit < 9:
                    v = u + (10 ** i)
                    # leading zero check not needed for increase
                    if L <= v <= R and not is_prime[v]:
                        new_cost = cost + v
                        if new_cost < dist[v]:
                            dist[v] = new_cost
                            heapq.heappush(pq, (new_cost, v))
                # decrease
                if digit > 0:
                    # avoid leading zero: most significant digit becoming 0
                    if i == d - 1 and digit == 1:
                        continue
                    v = u - (10 ** i)
                    if L <= v <= R and not is_prime[v]:
                        new_cost = cost + v
                        if new_cost < dist[v]:
                            dist[v] = new_cost
                            heapq.heappush(pq, (new_cost, v))

        return -1
```
