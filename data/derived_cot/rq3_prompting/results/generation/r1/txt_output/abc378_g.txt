import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    A, B, M = map(int, data)
    N = A * B - 1

    # factorial N! modulo M
    fact = 1
    for i in range(2, N + 1):
        fact = fact * i % M

    # product of hook lengths for the shape (A^{B-1}, A-1)
    prod = 1
    for i in range(1, B + 1):          # rows 1..B
        row_len = A if i < B else A - 1
        for j in range(1, row_len + 1):  # columns 1..row_len
            if j <= A - 1:
                # cell in columns 1..A-1 (all rows)
                hook = (row_len - j) + (B - i) + 1
            else:
                # j == A, only rows 1..B-1 have this column
                hook = B - i
            prod = prod * hook % M

    inv = pow(prod, M - 2, M)          # Fermat because M is prime
    f = fact * inv % M                 # number of SYT of the shape
    ans = f * (A - 1) % M * (B - 1) % M
    print(ans)

if __name__ == "__main__":
