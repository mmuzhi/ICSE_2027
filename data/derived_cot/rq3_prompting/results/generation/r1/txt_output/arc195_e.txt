```python
import sys

MOD = 998244353

def main() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    Q = int(next(it))

    # A[1] is unused (root has no edge), initialised to 0
    A = [0] * (N + 1)
    for i in range(2, N + 1):
        A[i] = int(next(it)) % MOD

    # factorial (N-1)!
    fact = 1
    for i in range(1, N):
        fact = fact * i % MOD

    # modular inverses up to N+1
    inv = [0] * (N + 2)
    inv[1] = 1
    for i in range(2, N + 2):
        inv[i] = MOD - MOD // i * inv[MOD % i] % MOD

    # prefix sums for Σ A_i / i   (i from 2 to k)
    pref1 = [0] * (N + 1)
    for i in range(2, N + 1):
        pref1[i] = (pref1[i-1] + A[i] * inv[i]) % MOD

    # prefix sums for Σ A_i * 2(i-1)/(i(i+1))   (i from 2 to k)
    pref2 = [0] * (N + 1)
    for i in range(2, N + 1):
        val = 2 * (i-1) % MOD
        val = val * inv[i] % MOD
        val = val * inv[i+1] % MOD
        pref2[i] = (pref2[i-1] + A[i] * val) % MOD

    out = []
    for _ in range(Q):
        u = int(next(it))
        v = int(next(it))

        # contribution of edge v (always on the path)
        ans = fact * A[v] % MOD

        # contribution of edge u (if u ≥ 2)
        ans = (ans + fact * A[u] % MOD * (1 - inv[u] + MOD)) % MOD

        # contributions for i in (u, v)
        if u + 1 <= v - 1:
            cur = (pref1[v-1] - pref1[u]) % MOD
            ans = (ans + fact * cur) % MOD

        # contributions for i < u
        if u - 1 >= 2:
            ans = (ans + fact * pref2[u-1]) % MOD

        out.append(str(ans))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
```
