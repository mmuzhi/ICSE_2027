=== ANSWER ===
```python
import sys

def main() -> None:
    data = sys.stdin.read().splitlines()
    if not data:
        return
    n = int(data[0].strip())
    strings = data[1:1+n]
    # sort by length (all lengths are distinct, so tie-breaking not needed)
    strings.sort(key=len)
    result = ''.join(strings)
    print(result)

if __name__ == "__main__":
    main()
```
