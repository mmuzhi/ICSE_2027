```python
from typing import List

class Solution:
    def minimumSumSubarray(self, nums: List[int], l: int, r: int) -> int:
        n = len(nums)
        prefix = [0]*(n+1)
        for i in range(n):
            prefix[i+1] = prefix[i] + nums[i]
        best = None
        for i in range(n):
            max_len = min(r, n-i)
            for length in range(l, max_len+1):
                s = prefix[i+length] - prefix[i]
                if s > 0:
                    if best is None or s < best:
                        best = s
        return -1 if best is None else best
```

```python
from typing import List

class Solution:
    def minimumSumSubarray(self, nums: List[int], l: int, r: int) -> int:
        n = len(nums)
        # compute prefix sums
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + nums[i]

        best = None
        for start in range(n):
            # maximum possible length from start
            max_len = min(r, n - start)
            for length in range(l, max_len + 1):
                curr_sum = prefix[start + length] - prefix[start]
                if curr_sum > 0:
                    if best is None or curr_sum < best:
                        best = curr_sum

        return -1 if best is None else best
```
