def hungarian(cost):
    n = len(cost)  # square matrix
    u = [0]*(n+1)
    v = [0]*(n+1)
    p = [0]*(n+1)
    way = [0]*(n+1)
    for i in range(1, n+1):
        p[0] = i
        j0 = 0
        minv = [float('inf')]*(n+1)
        used = [False]*(n+1)
        while True:
            used[j0] = True
            i0 = p[j0]
            delta = float('inf')
            j1 = 0
            for j in range(1, n+1):
                if not used[j]:
                    cur = cost[i0-1][j-1] - u[i0] - v[j]
                    if cur < minv[j]:
                        minv[j] = cur
                        way[j] = j0
                    if minv[j] < delta:
                        delta = minv[j]
                        j1 = j
            for j in range(n+1):
                if used[j]:
                    u[p[j]] += delta
                    v[j] -= delta
                else:
                    minv[j] -= delta
            j0 = j1
            if p[j0] == 0:
                break
        # augmenting
        while True:
            j1 = way[j0]
            p[j0] = p[j1]
            j0 = j1
            if j0 == 0:
                break
    # p[j] gives row assigned to column j (1-indexed)
    match = [-1]*n
    for j in range(1, n+1):
        if p[j] != 0:
            match[p[j]-1] = j-1
    # compute total cost
    total_cost = 0
    for i in range(n):
        total_cost += cost[i][match[i]]
    return total_cost

from typing import List

class Solution:
    def maxScore(self, grid: List[List[int]]) -> int:
        R = len(grid)
        C = len(grid[0])
        # collect distinct values
        distinct = set()
        for row in grid:
            for val in row:
                distinct.add(val)
        distinct = sorted(distinct)  # optional, but need mapping
        D = len(distinct)
        val_to_idx = {val: idx for idx, val in enumerate(distinct)}
        # For each row, list of value indices
        row_vals = [set() for _ in range(R)]
        for i in range(R):
            for val in grid[i]:
                row_vals[i].add(val_to_idx[val])
        N = R + D
        INF = 10**9
        cost = [[INF]*N for _ in range(N)]
        # real edges
        for i in range(R):
            for idx in row_vals[i]:
                val = distinct[idx]
                cost[i][idx] = -val   # weight negative for min cost
        # dummy rows (R..N-1) to real values
        for i in range(R, N):
            for j in range(D):
                cost[i][j] = 0
        # real rows to dummy values (D..N-1)
        for i in range(R):
            for j in range(D, N):
                cost[i][j] = 0
        # dummy rows to dummy values
        for i in range(R, N):
            for j in range(D, N):
                cost[i][j] = 0
        
        # Hungarian
        n = N
        u = [0]*(n+1)
        v = [0]*(n+1)
        p = [0]*(n+1)
        way = [0]*(n+1)
        for i in range(1, n+1):
            p[0] = i
            j0 = 0
            minv = [INF]*(n+1)  # use large number instead of float('inf')
            used = [False]*(n+1)
            while True:
                used[j0] = True
                i0 = p[j0]
                delta = INF
                j1 = 0
                for j in range(1, n+1):
                    if not used[j]:
                        cur = cost[i0-1][j-1] - u[i0] - v[j]
                        if cur < minv[j]:
                            minv[j] = cur
                            way[j] = j0
                        if minv[j] < delta:
                            delta = minv[j]
                            j1 = j
                for j in range(n+1):
                    if used[j]:
                        u[p[j]] += delta
                        v[j] -= delta
                    else:
                        minv[j] -= delta
                j0 = j1
                if p[j0] == 0:
                    break
            # augment
            while True:
                j1 = way[j0]
                p[j0] = p[j1]
                j0 = j1
                if j0 == 0:
                    break
        match = [-1]*n
        for j in range(1, n+1):
            if p[j] != 0:
                match[p[j]-1] = j-1
        total_cost = 0
        for i in range(n):
            total_cost += cost[i][match[i]]
        # Since we used -val for real edges, maxScore = -total_cost
        return -total_cost

from typing import List

class Solution:
    def maxScore(self, grid: List[List[int]]) -> int:
        R = len(grid)
        C = len(grid[0])
        # collect all distinct values
        distinct_vals = set()
        for row in grid:
            for v in row:
                distinct_vals.add(v)
        distinct_vals = sorted(distinct_vals)  # order does not matter
        D = len(distinct_vals)
        val_to_idx = {v: i for i, v in enumerate(distinct_vals)}
        
        # for each row, set of value indices that appear
        row_vals = [set() for _ in range(R)]
        for i in range(R):
            for v in grid[i]:
                row_vals[i].add(val_to_idx[v])
        
        N = R + D                     # size of square matrix
        INF = 10**9
        
        # cost matrix for Hungarian (minimization)
        cost = [[INF] * N for _ in range(N)]
        
        # real edges: row -> value with weight = -value
        for i in range(R):
            for idx in row_vals[i]:
                cost[i][idx] = -distinct_vals[idx]
        
        # dummy rows (R .. N-1) can take any real value with cost 0
        for i in range(R, N):
            for j in range(D):
                cost[i][j] = 0
        # real rows can be matched to dummy values (D .. N-1) with cost 0
        for i in range(R):
            for j in range(D, N):
                cost[i][j] = 0
        # dummy rows to dummy values also cost 0
        for i in range(R, N):
            for j in range(D, N):
                cost[i][j] = 0
        
        # ----- Hungarian algorithm for minimum cost perfect matching -----
        n = N
        u = [0] * (n + 1)
        v = [0] * (n + 1)
        p = [0] * (n + 1)          # which row is matched to column j
        way = [0] * (n + 1)
        
        for i in range(1, n + 1):
            p[0] = i
            j0 = 0
            minv = [INF] * (n + 1)
            used = [False] * (n + 1)
            while True:
                used[j0] = True
                i0 = p[j0]
                delta = INF
                j1 = 0
                for j in range(1, n + 1):
                    if not used[j]:
                        cur = cost[i0-1][j-1] - u[i0] - v[j]
                        if cur < minv[j]:
                            minv[j] = cur
                            way[j] = j0
                        if minv[j] < delta:
                            delta = minv[j]
                            j1 = j
                for j in range(n + 1):
                    if used[j]:
                        u[p[j]] += delta
                        v[j] -= delta
                    else:
                        minv[j] -= delta
                j0 = j1
                if p[j0] == 0:
                    break
            # augmenting
            while True:
                j1 = way[j0]
                p[j0] = p[j1]
                j0 = j1
                if j0 == 0:
                    break
        
        # reconstruct matching and total cost
        match = [-1] * n
        for j in range(1, n + 1):
            if p[j] != 0:
                match[p[j]-1] = j-1
        total_cost = 0
        for i in range(n):
            total_cost += cost[i][match[i]]
        # maxScore = -total_cost because we used negative values for real edges
        return -total_cost
