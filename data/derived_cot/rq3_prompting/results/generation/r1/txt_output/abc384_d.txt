import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    S = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # prefix sums (0-indexed, with pref[0] = 0)
    pref = [0] * (N + 1)
    for i in range(1, N + 1):
        pref[i] = pref[i-1] + A[i-1]
    total = pref[N]                     # T

    # 1. Check any contiguous subarray inside one period
    seen = set()
    for i in range(N + 1):
        if pref[i] - S in seen:
            print("Yes")
            return
        seen.add(pref[i])

    # 2. Prepare set of prefixes for indices 0 .. N-1 (valid starting points)
    start_prefixes = {pref[i] for i in range(N)}   # a = i-1, so i from 0 to N-1

    # possible numbers of "period blocks" (m = k+1)
    q = S // total
    candidates = []
    if q >= 1:
        candidates.append(q)
    candidates.append(q + 1)            # this is always at least 1

    # For each candidate, check if difference d = S - m*total can be expressed
    for m in candidates:
        d = S - m * total
        # For each possible ending prefix (b = 0 .. N), see if pref[b] - d is in start_prefixes
        for b in range(N + 1):
            if pref[b] - d in start_prefixes:
                print("Yes")
                return

    print("No")

if __name__ == "__main__":
    solve()
