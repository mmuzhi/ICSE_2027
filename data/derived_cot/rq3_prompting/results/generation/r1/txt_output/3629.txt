class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
        MOD = 10**9+7
        cnt = [0]*26
        for ch in s:
            cnt[ord(ch)-97] += 1
        for _ in range(t):
            new = [0]*26
            for i in range(26):
                if i == 25:  # 'z'
                    new[0] = (new[0] + cnt[25]) % MOD
                    new[1] = (new[1] + cnt[25]) % MOD
                else:

            cnt = new
        return sum(cnt) % MOD

```python
class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
        MOD = 10**9 + 7
        cnt = [0] * 26
        for ch in s:
            cnt[ord(ch) - 97] += 1

        for _ in range(t):
            new = [0] * 26
            for i in range(26):
                if i == 25:          # 'z' becomes "ab"
                    new[0] = (new[0] + cnt[25]) % MOD
                    new[1] = (new[1] + cnt[25]) % MOD
                else:                # other letters shift by one
                    new[i + 1] = (new[i + 1] + cnt[i]) % MOD
            cnt = new

        return sum(cnt) % MOD
```
