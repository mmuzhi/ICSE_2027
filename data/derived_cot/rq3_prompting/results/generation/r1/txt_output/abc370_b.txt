=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    # 1-indexed matrix, only positions i>=j are filled
    mat = [[0] * (N + 1) for _ in range(N + 1)]
    for i in range(1, N + 1):
        for j in range(1, i + 1):
            mat[i][j] = int(next(it))

    cur = 1
    for k in range(1, N + 1):
        if cur >= k:
            cur = mat[cur][k]
        else:
            cur = mat[k][cur]

    print(cur)

if __name__ == "__main__":
    solve()
```
