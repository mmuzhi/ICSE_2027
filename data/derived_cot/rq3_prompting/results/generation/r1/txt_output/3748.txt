class Solution:
    def sortMatrix(self, grid: List[List[int]]) -> List[List[int]]:
        n = len(grid)
        # diagonals i-j = d
        for d in range(-(n-1), n):
            # collect indices
            elems = []
            # iterate i from start to end
            # start i: if d>=0 then i=d else i=0
            # end i: if d>=0 then i=n-1 else i=n-1+d
            i_start = d if d >= 0 else 0
            i_end = n-1 if d >= 0 else n-1+d
            for i in range(i_start, i_end+1):
                j = i - d
                elems.append(grid[i][j])
            if d >= 0:
                elems.sort(reverse=True)
            else:
                elems.sort()
            # assign back
            idx = 0
            for i in range(i_start, i_end+1):
                j = i - d
                grid[i][j] = elems[idx]
                idx += 1
        return grid

from typing import List

class Solution:
    def sortMatrix(self, grid: List[List[int]]) -> List[List[int]]:
        n = len(grid)
        # d = i - j, ranges from -(n-1) to n-1
        for d in range(-(n-1), n):
            # Determine the start and end rows i for this diagonal
            if d >= 0:
                i_start = d
                i_end = n - 1
            else:
                i_start = 0
                i_end = n - 1 + d   # because d is negative, this is < n-1
            # Collect the diagonal elements in order from top-left to bottom-right
            elems = []
            for i in range(i_start, i_end + 1):
                j = i - d
                elems.append(grid[i][j])
            # Sort according to the rule
            if d >= 0:
                elems.sort(reverse=True)   # non-increasing
            else:
                elems.sort()               # non-decreasing
            # Write back
            idx = 0
            for i in range(i_start, i_end + 1):
                j = i - d
                grid[i][j] = elems[idx]
                idx += 1
        return grid
