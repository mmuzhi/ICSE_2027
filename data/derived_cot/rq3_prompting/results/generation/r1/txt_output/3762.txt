class Solution:
    def maxScore(self, points: List[int], m: int) -> int:
        n = len(points)
        # early check? maybe not needed.
        lo = 0
        hi = max(points) * m  # safe upper bound
        # binary search for maximum feasible X
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if self.is_feasible(points, m, mid):
                lo = mid
            else:
                hi = mid - 1
        return lo

    def is_feasible(self, points, m, X):
        if X == 0:
            return True
        n = len(points)
        needed = [(X + p - 1) // p for p in points]  # ceil division
        moves = 1  # first step to index 0
        carry = 0
        # handle index 0
        if needed[0] > 1:
            extra = needed[0] - 1
            moves += 2 * extra
            carry = extra
        else:
            carry = 0
        # process indices 1..n-2
        for i in range(1, n-1):
            moves += 1  # move to i
            cur = carry + 1
            if cur < needed[i]:
                extra = needed[i] - cur
                moves += 2 * extra
                carry = extra
            else:
                carry = 0
        # handle last index
        if n > 1:
            if carry >= needed[n-1]:
                # can stop at n-2, no move to n-1
                pass
            else:
                moves += 1  # move to n-1
                cur = carry + 1
                if cur < needed[n-1]:

from typing import List

class Solution:
    def maxScore(self, points: List[int], m: int) -> int:
        n = len(points)
        # quick check for zero moves? Not needed because m>=1.
        lo, hi = 0, max(points) * m
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if self._feasible(points, m, mid):
                lo = mid
            else:
                hi = mid - 1
        return lo

    def _feasible(self, points: List[int], m: int, X: int) -> bool:
        if X == 0:
            return True
        n = len(points)
        # required visits for each cell
        need = [(X + p - 1) // p for p in points]

        moves = 1          # first move to index 0
        carry = 0          # extra visits passed to the right

        # process index 0
        if need[0] > 1:
            extra = need[0] - 1
            moves += 2 * extra
            carry = extra
        else:
            carry = 0

        # process indices 1 .. n-2
        for i in range(1, n - 1):
            moves += 1                     # move from i-1 to i
            cur = carry + 1                # visits already at i + the one from this move
            if cur < need[i]:
                extra = need[i] - cur
                moves += 2 * extra
                carry = extra
            else:
                carry = 0

        # process the last index
        if n > 1:
            if carry < need[n-1]:
                moves += 1                 # move to last cell
                cur = carry + 1
                if cur < need[n-1]:
                    extra = need[n-1] - cur
                    moves += 2 * extra
            # else: carry >= need[n-1], we can stop at n-2

        return moves <= m
