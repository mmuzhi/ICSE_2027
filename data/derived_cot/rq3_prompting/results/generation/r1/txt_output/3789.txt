You are given an integer n which represents an array nums containing the numbers from 1 to n in order. Additionally, you are given a 2D array conflictingPairs, where conflictingPairs[i] = [a, b] indicates that a and b form a conflicting pair.
Remove exactly one element from conflictingPairs. Afterward, count the number of non-empty subarrays of nums which do not contain both a and b for any remaining conflicting pair [a, b].
Return the maximum number of subarrays possible after removing exactly one conflicting pair.
 
Example 1:

class Solution:
    def maxSubarrays(self, n: int, conflictingPairs: List[List[int]]) -> int:
        m = len(conflictingPairs)
        # Build intervals_by_l
        intervals_by_l = [[] for _ in range(n+1)]  # index 1..n
        for idx, (a,b) in enumerate(conflictingPairs):
            l = min(a,b)
            r = max(a,b)
            intervals_by_l[l].append((r, idx))
        INF = n+1
        min1_arr = [INF]*(n+2)  # use 1-index, index 1..n
        min2_arr = [INF]*(n+2)
        unique_id_arr = [-1]*(n+2)
        best1 = INF
        best2 = INF
        cnt1 = 0
        id1 = -1
        for L in range(n, 0, -1):
            # add intervals with l == L
            for r, idx in intervals_by_l[L]:
                if r < best1:
                    best2 = best1
                    best1 = r
                    cnt1 = 1
                    id1 = idx
                elif r == best1:
                    cnt1 += 1
                    if cnt1 > 1:
                        id1 = -1
                elif r < best2:
                    best2 = r
            min1_arr[L] = best1
            min2_arr[L] = best2
            if cnt1 == 1:
                unique_id_arr[L] = id1
            else:
                unique_id_arr[L] = -1
        total_all = 0
        for L in range(1, n+1):
            total_all += max(0, min1_arr[L] - L)

            if uid != -1:

        return total_all + max(gain)

```python
from typing import List

class Solution:
    def maxSubarrays(self, n: int, conflictingPairs: List[List[int]]) -> int:
        m = len(conflictingPairs)
        if m == 0:
            # not possible due to constraints, but handle gracefully
            return n * (n + 1) // 2

        # store intervals by left endpoint
        intervals_by_l = [[] for _ in range(n + 1)]  # index 1..n
        for idx, (a, b) in enumerate(conflictingPairs):
            l = min(a, b)
            r = max(a, b)
            intervals_by_l[l].append((r, idx))

        INF = n + 1
        min1_arr = [INF] * (n + 2)   # min r for L = i
        min2_arr = [INF] * (n + 2)   # second min r for L = i
        unique_id_arr = [-1] * (n + 2)

        best1 = INF
        best2 = INF
        cnt1 = 0
        id1 = -1

        # sweep L from n down to 1
        for L in range(n, 0, -1):
            # add all intervals with left endpoint exactly L
            for r, idx in intervals_by_l[L]:
                if r < best1:
                    best2 = best1
                    best1 = r
                    cnt1 = 1
                    id1 = idx
                elif r == best1:
                    cnt1 += 1
                    if cnt1 > 1:
                        id1 = -1
                elif r < best2:
                    best2 = r
            min1_arr[L] = best1
            min2_arr[L] = best2
            if cnt1 == 1:
                unique_id_arr[L] = id1
            else:
                unique_id_arr[L] = -1

        # total subarrays with all pairs
        total_all = 0
        for L in range(1, n + 1):
            total_all += max(0, min1_arr[L] - L)

        # gain for each interval if removed
        gain = [0] * m
        for L in range(1, n + 1):
            uid = unique_id_arr[L]
            if uid != -1:
                gain[uid] += min2_arr[L] - min1_arr[L]

        return total_all + max(gain)
```
