class Solution:
    def countNonDecreasingSubarrays(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # prefix sums
        ps = [0]*(n+1)
        for i in range(n):
            ps[i+1] = ps[i] + nums[i]
        # next greater element
        nxt = [n]*n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] < nums[i]:
                j = stack.pop()
                nxt[j] = i
            stack.append(i)
        # binary lifting
        LOG = (n).bit_length() + 1  # enough
        jump = [[n]*LOG for _ in range(n)]
        cost = [[0]*LOG for _ in range(n)]
        for i in range(n):
            jump[i][0] = nxt[i]
            seg_len = nxt[i] - i
            seg_sum = ps[nxt[i]] - ps[i]
            cost[i][0] = nums[i] * seg_len - seg_sum
        for j in range(1, LOG):

from typing import List

class Solution:
    def countNonDecreasingSubarrays(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # prefix sums
        ps = [0] * (n + 1)
        for i in range(n):
            ps[i + 1] = ps[i] + nums[i]
        
        # next greater element (strictly greater)
        nxt = [n] * n
        stack = []
        for i in range(n):
            while stack and nums[stack[-1]] < nums[i]:
                j = stack.pop()
                nxt[j] = i
            stack.append(i)
        
        # binary lifting
        LOG = n.bit_length() + 1
        jump = [[n] * LOG for _ in range(n)]
        cost = [[0] * LOG for _ in range(n)]
        
        for i in range(n):
            jump[i][0] = nxt[i]
            seg_len = nxt[i] - i
            seg_sum = ps[nxt[i]] - ps[i]
            cost[i][0] = nums[i] * seg_len - seg_sum
        
        for j in range(1, LOG):
            for i in range(n):
                mid = jump[i][j - 1]
                if mid < n:
                    jump[i][j] = jump[mid][j - 1]
                    cost[i][j] = cost[i][j - 1] + cost[mid][j - 1]
                else:
                    jump[i][j] = n
                    cost[i][j] = cost[i][j - 1]
        
        # function to compute total cost for subarray [l, r]
        def get_cost(l: int, r: int) -> int:
            cur = l
            total = 0
            while cur <= r:
                # find largest jump that stays within r
                j = LOG - 1
                while j >= 0 and jump[cur][j] > r:
                    j -= 1
                if j >= 0:
                    total += cost[cur][j]
                    cur = jump[cur][j]
                else:
                    # cannot jump further, add partial segment
                    total += nums[cur] * (r - cur + 1) - (ps[r + 1] - ps[cur])
                    break
            return total
        
        # two-pointer over left endpoints
        ans = 0
        r = -1
        for l in range(n):
            while r + 1 < n and get_cost(l, r + 1) <= k:
                r += 1
            ans += r - l + 1
        return ans
