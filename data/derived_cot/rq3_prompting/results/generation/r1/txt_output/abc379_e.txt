=== ANSWER ===
```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1]
    total = 0
    dp = 0
    for i, ch in enumerate(S):
        digit = ord(ch) - 48   # faster than int(ch)
        dp = dp * 10 + (i + 1) * digit
        total += dp
    print(total)

if __name__ == "__main__":
    main()
```
