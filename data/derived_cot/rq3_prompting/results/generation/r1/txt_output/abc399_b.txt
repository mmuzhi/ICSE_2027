import sys

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    scores = [int(next(it)) for _ in range(n)]

    # Pair (score, original_index)
    people = [(scores[i], i) for i in range(n)]
    # Sort by score descending
    people.sort(key=lambda x: -x[0])

    rank = 1
    ans = [0] * n
    i = 0
    while i < n:
        # Find all with same score as people[i][0]
        group_score = people[i][0]
        j = i
        while j < n and people[j][0] == group_score:
            j += 1
        # Assign rank to all in this group
        for k in range(i, j):
            idx = people[k][1]
            ans[idx] = rank
        # Update rank and pointer
        rank += (j - i)
        i = j

    # Output
    out = '\n'.join(str(r) for r in ans)
    print(out)

if __name__ == "__main__":
    main()
