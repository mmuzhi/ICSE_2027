from typing import List
from collections import deque

class Solution:
    def minValidStrings(self, words: List[str], target: str) -> int:
        # Build a trie of all prefixes of all words
        root = {}
        for w in words:
            node = root
            for ch in w:
                if ch not in node:
                    node[ch] = {}
                node = node[ch]
        
        m = len(target)
        # dp[i] = minimal number of valid strings to form target[:i]
        dp = [-1] * (m + 1)
        dp[0] = 0
        q = deque([0])   # BFS queue of positions
        
        while q:
            i = q.popleft()
            node = root
            # Walk the trie along target[i:]
            for k in range(i, m):
                ch = target[k]
                if ch not in node:
                    break
                node = node[ch]
                j = k + 1
                if dp[j] == -1:
                    dp[j] = dp[i] + 1
                    if j == m:
                        return dp[m]
                    q.append(j)
        
        return -1
