import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))

    # minR[k][l] = smallest R such that there exists a constraint (L,R,k) with L >= l
    # (0-indexed, l from 0 to N, sentinel at N = N+1)
    minR = [[N + 1] * (N + 1) for _ in range(N)]

    for _ in range(M):
        L = int(next(it)) - 1
        R = int(next(it)) - 1
        X = int(next(it)) - 1
        # Constraint guarantee L <= X <= R
        if R < minR[X][L]:
            minR[X][L] = R

    # Compute suffix minima: for each k, for l from k down to 0, minR[k][l] = min_{L >= l} R
    for k in range(N):
        arr = minR[k]
        # arr[N] is already N+1 (sentinel)
        for l in range(k, -1, -1):
            if arr[l] > arr[l + 1]:
                arr[l] = arr[l + 1]

    # Precompute binomial coefficients modulo MOD
    comb = [[0] * (N + 1) for _ in range(N + 1)]
    for n in range(N + 1):
        comb[n][0] = comb[n][n] = 1
        for k in range(1, n):
            comb[n][k] = (comb[n - 1][k - 1] + comb[n - 1][k]) % MOD

    # DP[l][r] = number of permutations for segment [l, r] (0-indexed, inclusive)
    dp = [[0] * N for _ in range(N)]

    for length in range(1, N + 1):
        for l in range(N - length + 1):
            r = l + length - 1
            total = 0
            for k in range(l, r + 1):
                # Check if any forbidden interval for position k is fully inside [l, r]
                if minR[k][l] > r:
                    left = dp[l][k - 1] if k > l else 1
                    right = dp[k + 1][r] if k < r else 1
                    total = (total + left * right % MOD * comb[length - 1][k - l]) % MOD
            dp[l][r] = total

    print(dp[0][N - 1] % MOD)

    solve()
