import sys

def solve():
    input = sys.stdin.readline
    N, M = map(int, input().split())
    intervals = []
    for i in range(M):
        L, R = map(int, input().split())
        intervals.append((L, R, i))

    # Step 1: one interval covers whole [1, N]
    for L, R, idx in intervals:
        if L == 1 and R == N:
            print(1)
            ops = ['0'] * M
            ops[idx] = '1'
            print(' '.join(ops))
            return

    # Step 2: two disjoint intervals (global intersection empty)
    max_L = max(L for L, R, _ in intervals)
    min_R = min(R for L, R, _ in intervals)
    if max_L > min_R:
        # find one with L = max_L and one with R = min_R
        idx1 = idx2 = -1
        for L, R, i in intervals:
            if L == max_L and idx1 == -1:
                idx1 = i
            if R == min_R and idx2 == -1:
                idx2 = i
        print(2)
        ops = ['0'] * M
        ops[idx1] = ops[idx2] = '2'
        print(' '.join(ops))
        return

    # Now all intervals intersect (max_L <= min_R)
    # Preprocess for greedy covering of [1, N] with op1
    right = [0] * (N + 2)          # best R at each L
    idx_at_L = [-1] * (N + 2)      # index of that interval
    for L, R, i in intervals:
        if R > right[L]:
            right[L] = R
            idx_at_L[L] = i

    maxR = [0] * (N + 2)           # max reachable from position pos
    best_idx = [-1] * (N + 2)      # index that gives that max
    cur_max = 0
    cur_idx = -1
    for pos in range(1, N + 1):
        if right[pos] > cur_max:
            cur_max = right[pos]
            cur_idx = idx_at_L[pos]
        maxR[pos] = cur_max
        best_idx[pos] = cur_idx

    # Greedy covering of [1, N]
    start = 1
    steps = 0
    used = []
    while start <= N:
        if maxR[start] < start:
            break
        steps += 1
        used.append(best_idx[start])
        start = maxR[start] + 1
    if start > N and steps == 2:
        # exactly two intervals cover the whole line
        print(2)
        ops = ['0'] * M
        for idx in used:
            ops[idx] = '1'
        print(' '.join(ops))

    # Step 3: look for a superset pair (one interval contains another)
    sorted_ints = sorted(intervals, key=lambda x: x[0])
    max_R_sofar = -1
    max_idx_sofar = -1
    for L, R, i in sorted_ints:
        if R <= max_R_sofar and max_idx_sofar != -1:
            # current interval is covered by the one giving max_R_sofar
            print(2)
            ops = ['0'] * M
            ops[i] = '2'               # complement
            ops[max_idx_sofar] = '1'   # normal
