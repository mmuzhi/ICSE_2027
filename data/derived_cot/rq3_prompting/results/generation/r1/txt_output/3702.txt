=== ANSWER ===
```python
from typing import List
import math

class Solution:
    def maxLength(self, nums: List[int]) -> int:
        n = len(nums)
        max_len = 0
        for i in range(n):
            prod = 1
            g = 0
            l = 1
            for j in range(i, n):
                val = nums[j]
                prod *= val
                g = math.gcd(g, val)
                l = l * val // math.gcd(l, val)  # lcm
                if prod == l * g:
                    max_len = max(max_len, j - i + 1)
        return max_len
```
