```python
def hilbert_order(x, y, pow=21, rotate=0):
    # x,y from 0 to (1<<pow)-1. Since N<=1e5, pow=18 is enough (2^18=262144)
    # We'll use 18 bits.
    if pow == 0:
        return 0
    h = 1 << (pow-1)
    segment = 0
    if x >= h:
        segment = 1
        x -= h
    if y >= h:
        segment |= 2
        y -= h
    if segment == 0:
        return hilbert_order(y, x, pow-1)  # rotate?
    elif segment == 1:
        return (1 << (2*(pow-1))) + hilbert_order(x, y, pow-1)
    elif segment == 2:
        return (2 << (2*(pow-1))) + hilbert_order(x, y, pow-1)
    else:
        return (3 << (2*(pow-1))) + hilbert_order(h-1-y, h-1-x, pow-1)
```

```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    A = [0] * (N + 1)
    for i in range(1, N + 1):
        A[i] = int(next(it))
    B = [0] * (N + 1)
    for i in range(1, N + 1):
        B[i] = int(next(it))
    K = int(next(it))
    queries = []
    Xs = [0] * K
    Ys = [0] * K
    for i in range(K):
        x = int(next(it))
        y = int(next(it))
        queries.append((x, y))
        Xs[i] = x
        Ys[i] = y

    # coordinate compression of all values
    vals = sorted(set(A[1:]) | set(B[1:]))
    comp = {v: i + 1 for i, v in enumerate(vals)}   # 1‑indexed
    m = len(vals)

    # precompute ranks and r-1 for fast less-than queries
    rankA = [0] * (N + 1)
    rankA_lt = [0] * (N + 1)   # rankA[i] - 1
    for i in range(1, N + 1):
        r = comp[A[i]]
        rankA[i] = r
        rankA_lt[i] = r - 1
    rankB = [0] * (N + 1)
    rankB_lt = [0] * (N + 1)
    for i in range(1, N + 1):
        r = comp[B[i]]
        rankB[i] = r
        rankB_lt[i] = r - 1

    # Hilbert order
    max_coord = max(N, N)
    p = 1
    while (1 << p) <= max_coord:
        p += 1
    limit = 1 << p          # power of two >= max(N,N)

    def hilbert(x: int, y: int) -> int:
        """return Hilbert order for point (x,y) in [0, limit)"""
        d = 0
        s = limit >> 1
        while s:
            rx = (x & s) != 0
            ry = (y & s) != 0
            d += s * s * ((3 * rx) ^ ry)
            if ry == 0:
                if rx == 1:
                    x = limit - 1 - x
                    y = limit - 1 - y
                x, y = y, x
            s >>= 1
        return d

    # sort queries by Hilbert order
    order = []
    for idx in range(K):
        x, y = queries[idx]
        order.append((hilbert(x - 1, y - 1), idx, x, y))
    order.sort(key=lambda t: t[0])

    # BIT arrays (count and sum) for current A set and current B set
    bitA_cnt = [0] * (m + 2)
    bitA_sum = [0] * (m + 2)
    bitB_cnt = [0] * (m + 2)
    bitB_sum = [0] * (m + 2)

    # current state
    curX = 0
    curY = 0
    curSumA = 0
    curSumB = 0
    curAns = 0

    out = [0] * K

    for _, idx, tx, ty in order:
        # expand / shrink X dimension
        while curX < tx:
            curX += 1
            val = A[curX]
            r = rankA[curX]
            r_lt = rankA_lt[curX]

            # query bitB for values < val
            cnt_lt = 0
            sum_lt = 0
            i = r_lt
            while i:
                cnt_lt += bitB_cnt[i]
                sum_lt += bitB_sum[i]
                i -= i & -i

            total_cnt = curY
            total_sum = curSumB
            contrib = total_sum - 2 * sum_lt + (2 * cnt_lt - total_cnt) * val
            curAns += contrib

            # insert val into bitA
            i = r
            while i <= m:
                bitA_cnt[i] += 1
                bitA_sum[i] += val
                i += i & -i
            curSumA += val

        while curX > tx:
            val = A[curX]
            r = rankA[curX]
            r_lt = rankA_lt[curX]

            cnt_lt = 0
            sum_lt = 0
            i = r_lt
            while i:
                cnt_lt += bitB_cnt[i]
                sum_lt += bitB_sum[i]
                i -= i & -i

            total_cnt = curY
            total_sum = curSumB
            contrib = total_sum - 2 * sum_lt + (2 * cnt_lt - total_cnt) * val
            curAns -= contrib

            i = r
            while i <= m:
                bitA_cnt[i] -= 1
                bitA_sum[i] -= val
                i += i & -i
            curSumA -= val
            curX -= 1

        # expand / shrink Y dimension
        while curY < ty:
            curY += 1
            val = B[curY]
            r = rankB[curY]
            r_lt = rankB_lt[curY]

            cnt_lt = 0
            sum_lt = 0
            i = r_lt
            while i:
                cnt_lt += bitA_cnt[i]
                sum_lt += bitA_sum[i]
                i -= i & -i

            total_cnt = curX
            total_sum = curSumA
            contrib = total_sum - 2 * sum_lt + (2 * cnt_lt - total_cnt) * val
            curAns += contrib

            i = r
            while i <= m:
                bitB_cnt[i] += 1
                bitB_sum[i] += val
                i += i & -i
            curSumB += val

        while curY > ty:
            val = B[curY]

            r_lt = rankB_lt[curY]

            cnt_lt = 0
            sum_lt = 0
            i = r_lt

                sum_lt += bitA_sum[i]
```
