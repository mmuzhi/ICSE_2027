import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    X = int(next(it))
    A = [0] * N
    # Prepare storage for first three occurrences of each value (1-based indices)
    first = [0] * (X + 1)
    second = [0] * (X + 1)
    third = [0] * (X + 1)
    # We'll also collect distinct values that appear
    seen = [False] * (X + 1)
    distinct = []
    for idx in range(1, N + 1):
        v = int(next(it))
        A[idx - 1] = v
        if first[v] == 0:
            first[v] = idx
            seen[v] = True
            distinct.append(v)
        elif second[v] == 0:
            second[v] = idx
        else:
            third[v] = idx   # we don't need more than three occurrences
    # If any value appears at least three times and 3 * v == X, we have a solution
    if X % 3 == 0:
        v3 = X // 3
        if third[v3] != 0:
            # Output the three smallest indices for this value
            ans = [first[v3], second[v3], third[v3]]
            ans.sort()
            print(ans[0], ans[1], ans[2])
            return
        elif second[v3] != 0:
            # only two occurrences, cannot form a triple with three same values
            pass
    # Quick checks for impossibility
    min_val = min(distinct)
    max_val = max(distinct)
    if min_val * 3 > X or max_val * 3 < X:
        print(-1)
        return
    # Sort distinct values
    distinct.sort()
    D = len(distinct)
    # For each possible smallest value a (a <= X//3)
    for i_a, a in enumerate(distinct):
        if a > X // 3:

        T = X - a
        # Since b <= c and b >= a, we iterate b from a up to T//2

        # Start j from i_a
        j = i_a

            # b >= a automatically because j starts from i_a

            if c < b:
