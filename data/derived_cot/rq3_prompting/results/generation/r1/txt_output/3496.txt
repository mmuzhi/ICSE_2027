from typing import List

class Solution:
    def minNumberOfSeconds(self, mountainHeight: int, workerTimes: List[int]) -> int:
        H = mountainHeight
        wts = workerTimes

        # Upper bound: worst-case when the slowest worker does all work
        max_w = max(wts)
        hi = max_w * H * (H + 1) // 2   # safe high bound
        lo = 0

        # Binary search on the minimal maximum time
        while lo < hi:
            mid = (lo + hi) // 2
            total = 0

            for w in wts:
                # Binary search for the largest x such that w * x * (x+1) // 2 <= mid
                low_x, high_x = 0, H
                while low_x < high_x:
                    m = (low_x + high_x + 1) // 2
                    if w * m * (m + 1) // 2 <= mid:
                        low_x = m
                    else:
                        high_x = m - 1
                total += low_x
                if total >= H:      # early stop
                    break

            if total >= H:
                hi = mid
            else:
                lo = mid + 1

        return lo
