def main():
    import sys
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    # splitting point: first N-K cards remain in the same order after the moved bottom cards
    split = N - K
    B = A[split:] + A[:split]
    print(' '.join(map(str, B)))

if __name__ == "__main__":
    main()
