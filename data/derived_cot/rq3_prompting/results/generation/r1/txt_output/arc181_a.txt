import sys

def solve() -> None:
    data = sys.stdin.read().split()
    it = iter(data)
    T = int(next(it))
    out_lines = []

    for _ in range(T):
        N = int(next(it))
        P = [int(next(it)) for _ in range(N)]

        # position of each value (1-indexed)
        pos = [0] * (N + 1)
        for idx, val in enumerate(P, start=1):
            pos[val] = idx

        # check identity
        identity = True
        for i in range(N):
            if P[i] != i + 1:
                identity = False
                break
        if identity:
            out_lines.append('0')
            continue

        # prefix maximum
        pref_max = [0] * (N + 2)          # pref_max[i] = max(P[1..i])
        for i in range(1, N + 1):
            pref_max[i] = max(pref_max[i - 1], P[i - 1])

        # suffix minimum
        suff_min = [N + 1] * (N + 2)     # suff_min[i] = min(P[i..N])
        for i in range(N, 0, -1):
            suff_min[i] = min(suff_min[i + 1], P[i - 1])

        # check whether one operation suffices
        one_ok = False
        for k in range(1, N + 1):
            if P[k - 1] == k:
                left_ok = (k == 1) or (pref_max[k - 1] == k - 1)
                right_ok = (k == N) or (suff_min[k + 1] == k + 1)
                if left_ok and right_ok:
                    one_ok = True
                    break

        if one_ok:
            out_lines.append('1')
        else:
            # if exactly the first and last element are swapped, three operations are needed
            if pos[1] == N and pos[N] == 1:
                out_lines.append('3')
            else:
                out_lines.append('2')

    sys.stdout.write('\n'.join(out_lines))

if __name__ == "__main__":
    solve()
