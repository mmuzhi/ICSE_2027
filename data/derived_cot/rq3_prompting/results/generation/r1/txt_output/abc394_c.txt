```python
import sys

def solve() -> None:
    S = sys.stdin.readline().strip()
    stack = []
    for ch in S:
        if ch == 'A' and stack and stack[-1] == 'W':
            cnt = 0
            # pop all consecutive W's from the top
            while stack and stack[-1] == 'W':
                stack.pop()
                cnt += 1
            # push the resulting A and cnt C's
            stack.append('A')
            # extent with cnt C's; total C's is O(n) overall
            stack.extend(['C'] * cnt)
        else:
            stack.append(ch)
    print(''.join(stack))

if __name__ == "__main__":
    solve()
```
