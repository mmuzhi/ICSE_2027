```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)

    N = int(next(it))
    W = int(next(it))

    # list for each column: (y, index)
    col = [[] for _ in range(W)]
    for idx in range(1, N + 1):
        x = int(next(it)) - 1
        y = int(next(it))
        col[x].append((y, idx))

    # sort each column by y
    for c in range(W):
        col[c].sort(key=lambda t: t[0])

    # minimal number of blocks among columns
    L = min(len(col[c]) for c in range(W))

    if L > 0:
        # M[r] = max over columns of (y_{c,r} - (r-1))
        M = [0] * (L + 1)          # 1‑based
        for c in range(W):
            lst = col[c]
            for r in range(1, L + 1):
                y = lst[r - 1][0]
                val = y - (r - 1)
                if val > M[r]:
                    M[r] = val

        # T[r] = time of the r‑th clearing
        T = [0] * (L + 1)
        for r in range(1, L + 1):
            T[r] = T[r - 1] + M[r]
    else:
        T = None          # no clearing ever

    INF = 10 ** 18
    removal_time = [INF] * (N + 1)

    # assign removal time to each block
    for c in range(W):
        lst = col[c]
        for r, (y, idx) in enumerate(lst, start=1):
            if L > 0 and r <= L:
                removal_time[idx] = T[r]
            # else stays INF (never removed)

    Q = int(next(it))
    out = []
    for _ in range(Q):
        t = int(next(it))
        a = int(next(it))
        out.append("Yes" if removal_time[a] > t else "No")

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
```
