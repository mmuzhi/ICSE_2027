```python
def main():
    # Read the 8 rows of the grid
    rows = [input().strip() for _ in range(8)]

    # Track which rows and columns contain at least one piece
    row_has = [False] * 8
    col_has = [False] * 8

    for i in range(8):
        for j in range(8):
            if rows[i][j] == '#':
                row_has[i] = True
                col_has[j] = True

    # Count empty squares that are not attacked by any piece
    safe_count = 0
    for i in range(8):
        for j in range(8):
            if rows[i][j] == '.' and not row_has[i] and not col_has[j]:
                safe_count += 1

    print(safe_count)

if __name__ == "__main__":
    main()
```
