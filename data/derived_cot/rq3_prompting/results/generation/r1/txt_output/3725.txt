Implementation details:

We'll write function compute_min_sum(nums, k):
    n = len(nums)
    left = [-1]*n
    stack = []
    for i in range(n):
        while stack and nums[stack[-1]] >= nums[i]: # pop if >= to maintain strictly less on left
            stack.pop()
        left[i] = stack[-1] if stack else -1
        stack.append(i)
    right = [n]*n
    stack = []
    for i in range(n-1, -1, -1):
        while stack and nums[stack[-1]] > nums[i]: # pop if > to get next smaller or equal
            stack.pop()
        right[i] = stack[-1] if stack else n
        stack.append(i)
    total = 0
    for i in range(n):
        L_len = i - left[i] # number of possible left positions
        R_len = right[i] - i
        # maximum d = min(L_len-1, k-1)
        D = min(L_len - 1, k - 1)
        if D < 0:
            continue
        # compute valid count
        d0 = k - R_len
        if d0 <= 0:
            # all contributions are k-d
            # sum_{d=0}^{D} (k-d) = (k + (k-D)) * (D+1) // 2
            cnt = (k + (k - D)) * (D + 1) // 2
        elif d0 > D:
            cnt = R_len * (D + 1)
        else:
            # 0 < d0 <= D
            cnt1 = R_len * d0
            # sum_{d=d0}^{D} (k-d) = sum_{t = k-D}^{k-d0} t
            start = k - D
            end = k - d0
            cnt2 = (start + end) * (end - start + 1) // 2
            cnt = cnt1 + cnt2

```python
from typing import List

class Solution:
    def minMaxSubarraySum(self, nums: List[int], k: int) -> int:
        # Helper to compute sum of minima of all subarrays with length <= k
        def compute_min_sum(arr: List[int], k: int) -> int:
            n = len(arr)
            # previous strictly smaller element (index)
            left = [-1] * n
            stack = []
            for i in range(n):
                while stack and arr[stack[-1]] >= arr[i]:
                    stack.pop()
                left[i] = stack[-1] if stack else -1
                stack.append(i)

            # next smaller or equal element (index)
            right = [n] * n
            stack.clear()
            for i in range(n - 1, -1, -1):
                while stack and arr[stack[-1]] > arr[i]:
                    stack.pop()
                right[i] = stack[-1] if stack else n
                stack.append(i)

            total = 0
            for i in range(n):
                L_len = i - left[i]          # number of possible left boundaries
                R_len = right[i] - i         # number of possible right boundaries
                D = min(L_len - 1, k - 1)    # maximum d = i - a that can give positive contribution
                if D < 0:
                    continue

                d0 = k - R_len
                if d0 <= 0:
                    # all d in [0, D] contribute k-d
                    cnt = (k + (k - D)) * (D + 1) // 2
                elif d0 > D:
                    # all d contribute R_len
                    cnt = R_len * (D + 1)
                else:
                    # d in [0, d0-1] contribute R_len, d in [d0, D] contribute k-d
                    cnt1 = R_len * d0
                    start = k - D
                    end = k - d0
                    cnt2 = (start + end) * (end - start + 1) // 2
                    cnt = cnt1 + cnt2

                total += arr[i] * cnt

            return total

        # Sum of minima
        min_sum = compute_min_sum(nums, k)
        # Sum of maxima = - (sum of minima of negated array)
        neg_nums = [-x for x in nums]
        max_sum = -compute_min_sum(neg_nums, k)

        return min_sum + max_sum
```
