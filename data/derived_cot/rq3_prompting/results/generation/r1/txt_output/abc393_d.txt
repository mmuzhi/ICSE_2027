=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])          # not directly needed, but provided
    s = data[1].strip()
    # collect 1 positions
    positions = [i for i, ch in enumerate(s) if ch == '1']
    k = len(positions)
    if k == 0:                # not possible due to constraints, but safe
        print(0)
        return
    # compute q[i] = positions[i] - i
    # q is non-decreasing because positions[i] >= i (since at least i zeros before)
    q = [positions[i] - i for i in range(k)]
    # median index (0-based)
    median = q[k // 2]
    # sum of absolute differences
    ans = sum(abs(val - median) for val in q)
    print(ans)

if __name__ == "__main__":
    solve()
```
