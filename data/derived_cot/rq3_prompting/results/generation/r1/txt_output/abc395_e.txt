```python
import sys
import heapq

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X = int(next(it))

    # 0..N-1 : original direction (state 0)
    # N..2N-1 : reversed direction (state 1)
    size = 2 * N
    adj = [[] for _ in range(size)]

    for _ in range(M):
        u = int(next(it)) - 1
        v = int(next(it)) - 1
        # original edges
        adj[u].append((v, 1))
        # reversed edges (in state 1)
        adj[v + N].append((u + N, 1))

    # reversal edges between states
    for v in range(N):
        adj[v].append((v + N, X))
        adj[v + N].append((v, X))

    # Dijkstra from vertex 1 (original state = 0)
    INF = 10**30
    dist = [INF] * size
    dist[0] = 0
    pq = [(0, 0)]
    while pq:
        d, u = heapq.heappop(pq)
        if d != dist[u]:
            continue
        for v, w in adj[u]:
            nd = d + w
            if nd < dist[v]:
                dist[v] = nd
                heapq.heappush(pq, (nd, v))

    # target: vertex N (index N-1) in either state
    ans = min(dist[N - 1], dist[2 * N - 1])
    print(ans)

if __name__ == "__main__":
    solve()
```
