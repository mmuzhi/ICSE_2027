from typing import List

class Solution:
    def beautifulSplits(self, nums: List[int]) -> int:
        n = len(nums)
        if n < 3:
            return 0
        
        # rolling hash with 64-bit modulo
        base = 131
        mask = (1 << 64) - 1
        
        h = [0] * (n + 1)          # prefix hashes
        p = [1] * (n + 1)          # powers of base
        for i, v in enumerate(nums):
            h[i+1] = ((h[i] * base) + v) & mask
            p[i+1] = (p[i] * base) & mask
        
        ans = 0
        # iterate over possible end of nums1 (i)
        for i in range(1, n-1):
            # precompute whether nums1 is a prefix of the first i elements of nums2
            prefix_match = False
            if 2 * i < n:
                # hash of nums1: [0, i)
                hash1 = h[i]                # because h[0] = 0
                # hash of [i, 2*i)
                hash2 = (h[2*i] - h[i] * p[i]) & mask
                if hash1 == hash2:
                    prefix_match = True
            
            # iterate over possible end of nums2 (j)
            for j in range(i+1, n):
                len2 = j - i
                cond1 = False
                if len2 >= i and prefix_match:
                    cond1 = True
                
                cond2 = False
                len3 = n - j
                if len3 >= len2:
                    # hash of nums2: [i, j)
                    h_ij = (h[j] - h[i] * p[len2]) & mask
                    # hash of prefix of nums3 of length len2: [j, j+len2)
                    h_j = (h[j+len2] - h[j] * p[len2]) & mask
                    if h_ij == h_j:
                        cond2 = True
                
                if cond1 or cond2:
                    ans += 1
        
        return ans
