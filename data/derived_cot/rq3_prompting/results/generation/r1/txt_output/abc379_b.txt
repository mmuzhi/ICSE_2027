=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N = int(data[0])
    K = int(data[1])
    S = data[2] if len(data) > 2 else ''
    ans = 0
    length = 0
    for ch in S:
        if ch == 'O':
            length += 1
        else:
            ans += length // K
            length = 0
    ans += length // K  # last segment
    print(ans)

if __name__ == "__main__":
    solve()
```
