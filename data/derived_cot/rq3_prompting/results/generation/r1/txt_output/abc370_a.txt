=== ANSWER ===
```python
def main():
    L, R = map(int, input().split())
    if L == 1 and R == 0:
        print("Yes")
    elif L == 0 and R == 1:
        print("No")
    else:  # both 0 or both 1
        print("Invalid")

if __name__ == "__main__":
    main()
```
