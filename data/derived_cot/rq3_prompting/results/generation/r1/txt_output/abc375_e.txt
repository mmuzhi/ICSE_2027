import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    people = []
    total = 0
    for _ in range(N):
        a = int(next(it))
        b = int(next(it))
        people.append((a, b))
        total += b

    if total % 3 != 0:
        print(-1)
        return

    T = total // 3
    INF = 10 ** 9

    # dp[s1][s2] = min moves to achieve sums for team1 and team2
    dp = [[INF] * (T + 1) for _ in range(T + 1)]
    dp[0][0] = 0

    cur_sum = 0  # sum of strengths of processed persons
    for a, b in people:
        new_sum = cur_sum + b
        newdp = [[INF] * (T + 1) for _ in range(T + 1)]
        # iterate over all possible current states
        for s1 in range(T + 1):
            row = dp[s1]
            for s2, cur_cost in enumerate(row):
                if cur_cost == INF:
                    continue
                # try each possible final team
                for t in (1, 2, 3):
                    ns1 = s1 + (b if t == 1 else 0)
                    ns2 = s2 + (b if t == 2 else 0)
                    # check bounds: ns1, ns2 <= T, and team3 sum = new_sum - ns1 - ns2 <= T
                    if ns1 <= T and ns2 <= T and ns1 + ns2 >= new_sum - T:
                        new_cost = cur_cost + (0 if t == a else 1)
                        if new_cost < newdp[ns1][ns2]:
                            newdp[ns1][ns2] = new_cost
        dp = newdp
        cur_sum = new_sum

    ans = dp[T][T]
    print(-1 if ans == INF else ans)

if __name__ == "__main__":
