```python
from typing import List

class Solution:
    def shiftDistance(self, s: str, t: str, nextCost: List[int], previousCost: List[int]) -> int:
        # Map letters to indices 0-25
        def idx(c: str) -> int:
            return ord(c) - ord('a')
        
        n = len(s)
        # Precompute costs for all 26x26 pairs
        forward = [[0]*26 for _ in range(26)]
        backward = [[0]*26 for _ in range(26)]
        
        for x in range(26):
            # Forward direction
            cost = 0
            for d in range(1, 26):
                # distance d from x to (x+d)%26
                cost += nextCost[(x + d - 1) % 26]
                forward[x][(x + d) % 26] = cost
            # Backward direction
            cost = 0
            for d in range(1, 26):
                # distance d from x to (x-d)%26
                cost += previousCost[(x - d + 1) % 26]  # step from (x-d+1) to (x-d)
                backward[x][(x - d) % 26] = cost
        
        total = 0
        for i in range(n):
            a = idx(s[i])
            b = idx(t[i])
            total += min(forward[a][b], backward[a][b])
        return total
```
