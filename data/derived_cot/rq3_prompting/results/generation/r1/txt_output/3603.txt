from typing import List

class Solution:
    def findAnswer(self, parent: List[int], s: str) -> List[bool]:
        n = len(parent)
        # build children lists (already sorted because we iterate i from 1 to n-1)
        children = [[] for _ in range(n)]
        for i in range(1, n):
            p = parent[i]
            children[p].append(i)

        # constants for rolling hash
        B = 131
        MOD1 = 1_000_000_007
        MOD2 = 1_000_000_009

        # precompute powers of B
        pow1 = [1] * (n + 1)
        pow2 = [1] * (n + 1)
        for i in range(1, n + 1):
            pow1[i] = (pow1[i-1] * B) % MOD1
            pow2[i] = (pow2[i-1] * B) % MOD2

        # per-node data
        length = [0] * n
        fh1 = [0] * n   # forward hash mod1
        fh2 = [0] * n   # forward hash mod2
        rh1 = [0] * n   # reverse hash mod1
        rh2 = [0] * n   # reverse hash mod2

        # iterative post-order DFS using a stack with (node, visited_flag)
        stack = [(0, False)]
        while stack:
            node, visited = stack.pop()
            if visited:
                # combine children's values
                total_len = 0
                for ch in children[node]:
                    total_len += length[ch]
                length[node] = total_len + 1

                # forward hash: concatenate children in increasing order, then node char
                f1 = 0
                f2 = 0
                for ch in children[node]:
                    clen = length[ch]
                    f1 = (f1 * pow1[clen] + fh1[ch]) % MOD1
                    f2 = (f2 * pow2[clen] + fh2[ch]) % MOD2
                cval = ord(s[node]) - ord('a') + 1   # 1‑based value
                f1 = (f1 * B + cval) % MOD1
                f2 = (f2 * B + cval) % MOD2
                fh1[node] = f1
                fh2[node] = f2

                # reverse hash: start with node char, then children in reverse order
                r1 = cval
                r2 = cval
                for ch in reversed(children[node]):
                    clen = length[ch]
                    r1 = (r1 * pow1[clen] + rh1[ch]) % MOD1
                    r2 = (r2 * pow2[clen] + rh2[ch]) % MOD2
                rh1[node] = r1
                rh2[node] = r2
            else:
                # post-order: push node again after children
                stack.append((node, True))
                # push children in reverse order so that the first child is processed first
                # (order does not affect correctness, as we only need all children processed before parent)
                for ch in reversed(children[node]):
                    stack.append((ch, False))

        # build answer array
        ans = [False] * n
        for i in range(n):
            if fh1[i] == rh1[i] and fh2[i] == rh2[i]:
                ans[i] = True
        return ans
