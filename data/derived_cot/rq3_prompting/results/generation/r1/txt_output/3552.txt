```python
class Solution:
    def largestPalindrome(self, n: int, k: int) -> str:
        # length of the first half (including middle for odd n)
        m = (n + 1) // 2

        # precompute powers of 10 modulo k
        pow10 = [1] * (n + 1)  # pow10[exp] = 10^exp mod k
        for i in range(1, n + 1):
            pow10[i] = (pow10[i-1] * 10) % k

        # weights for each position in the first half
        w = [0] * m
        for i in range(m):
            if i == n - 1 - i:          # middle digit (odd length)
                w[i] = pow10[i] % k
            else:                       # symmetric pair
                w[i] = (pow10[n-1-i] + pow10[i]) % k

        # dp[i][r] = could we fill positions i..m-1 to get remainder r?
        # we only need dp for suffixes (i from 1 to m), but compute full for convenience
        dp = [[False] * k for _ in range(m + 1)]
        dp[m][0] = True

        for i in range(m - 1, -1, -1):
            wi = w[i]
            # compute dp[i] from dp[i+1]
            for r in range(k):
                if dp[i+1][r]:
                    # try all digits 0..9 (including 0, because it's allowed except first digit)
                    for d in range(10):
                        new_r = (r + d * wi) % k
                        dp[i][new_r] = True

        # greedy construction
        res = [0] * n
        current_rem = 0
        for i in range(m):
            wi = w[i]
            # try digits from 9 down to 0 (first digit cannot be 0)
            start = 9 if i > 0 else 9
            # but for i=0 we start at 9 and skip 0 in loop
            for d in range(start, -1, -1):
                if i == 0 and d == 0:
                    continue
                new_rem = (current_rem + d * wi) % k
                # what remainder must the suffix provide to make total 0?
                need = (-new_rem) % k
                if dp[i+1][need]:
                    res[i] = d
                    res[n-1-i] = d
                    current_rem = new_rem
                    break

        return ''.join(map(str, res))
```
