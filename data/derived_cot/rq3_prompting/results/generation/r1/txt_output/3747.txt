```python
from typing import List

class Solution:
    def maxAdjacentDistance(self, nums: List[int]) -> int:
        n = len(nums)
        max_diff = 0
        # Check adjacent pairs in the linear order
        for i in range(n - 1):
            diff = abs(nums[i] - nums[i+1])
            if diff > max_diff:
                max_diff = diff
        # Check the circular pair (last and first)
        diff = abs(nums[-1] - nums[0])
        if diff > max_diff:
            max_diff = diff
        return max_diff
```
