=== ANSWER ===
```python
from typing import List

class Solution:
    def minOperations(self, nums: List[int], k: int) -> int:
        # If any element is less than k, impossible.
        if any(x < k for x in nums):
            return -1
        # Count distinct values strictly greater than k.
        distinct_greater = {x for x in nums if x > k}
        return len(distinct_greater)
```
