```python
class Solution:
    def distanceSum(self, m: int, n: int, k: int) -> int:
        MOD = 10**9 + 7
        N = m * n  # total cells
        
        # Precompute factorials and inverse factorials up to N
        fact = [1] * (N + 1)
        for i in range(1, N + 1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (N + 1)
        inv_fact[N] = pow(fact[N], MOD-2, MOD)
        for i in range(N, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD
        
        def binom(nn, rr):
            if rr < 0 or rr > nn:
                return 0
            return fact[nn] * inv_fact[rr] % MOD * inv_fact[nn - rr] % MOD
        
        # modular values
        m_mod = m % MOD
        n_mod = n % MOD
        
        # sum over x-distances: n^2 * (m^3 - m)/6
        term_x = (n_mod * n_mod) % MOD
        m3_minus_m = (pow(m_mod, 3, MOD) - m_mod) % MOD
        term_x = term_x * m3_minus_m % MOD
        
        # sum over y-distances: m^2 * (n^3 - n)/6
        term_y = (m_mod * m_mod) % MOD
        n3_minus_n = (pow(n_mod, 3, MOD) - n_mod) % MOD
        term_y = term_y * n3_minus_n % MOD
        
        inv6 = pow(6, MOD-2, MOD)
        S = (term_x + term_y) % MOD * inv6 % MOD
        
        # number of arrangements containing a fixed pair of cells
        comb = binom(N - 2, k - 2)
        
        ans = comb * S % MOD
        return ans
```
