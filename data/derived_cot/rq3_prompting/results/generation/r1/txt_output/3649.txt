class Solution:
    def findMinimumTime(self, strength: List[int], K: int) -> int:
        n = len(strength)
        best = float('inf')

                total += minutes
                x += K

        return best

from typing import List

        n = len(strength)
        best = float('inf')
        for perm in itertools.permutations(range(n)):
            total = 0
            x = 1

                # ceil division: (s + x - 1) // x
