=== ANSWER ===
```python
import sys

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    A = int(data[0])
    if 400 % A == 0:
        print(400 // A)
    else:
        print(-1)

if __name__ == "__main__":
    main()
```
