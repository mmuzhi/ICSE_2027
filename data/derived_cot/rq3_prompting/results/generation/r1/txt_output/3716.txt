from typing import List

class Solution:
    def longestSubsequence(self, nums: List[int]) -> int:
        MAX_VAL = 300                 # maximum possible value in nums
        BIG = 300                     # dummy "infinite" difference for length‑1 subsequences

        # dp[val][diff] = longest subsequence ending with value val and last absolute difference exactly diff
        dp = [[0] * (BIG + 1) for _ in range(MAX_VAL + 1)]
        # suffix_max[val][diff] = max_{k >= diff} dp[val][k]
        suffix = [[0] * (BIG + 1) for _ in range(MAX_VAL + 1)]

        appeared = [False] * (MAX_VAL + 1)
        appeared_vals = []            # list of values that have already been processed
        best_global = 0

        for x in nums:
            # temporary array for candidates of the current element
            new_dp = [0] * (BIG + 1)

            # try to extend every subsequence that ended with an already seen value
            for v in appeared_vals:
                d = abs(x - v)
                # longest subsequence ending with v whose last difference is at least d
                prev_len = suffix[v][d]
                if prev_len > 0:
                    cand = prev_len + 1
                    if cand > new_dp[d]:
                        new_dp[d] = cand

            # a subsequence consisting only of x itself is always valid
            if 1 > new_dp[BIG]:
                new_dp[BIG] = 1

            # merge the new candidates into dp[x]
            for d in range(BIG + 1):
                if new_dp[d] > dp[x][d]:
                    dp[x][d] = new_dp[d]

            # recompute suffix_max for x (since dp[x] may have changed)
            cur = 0
            for d in range(BIG, -1, -1):
                if dp[x][d] > cur:
                    cur = dp[x][d]
                suffix[x][d] = cur
            # update global maximum
            if cur > best_global:
                best_global = cur

            # mark x as appeared if it hasn't been before
            if not appeared[x]:
                appeared[x] = True
                appeared_vals.append(x)

        return best_global
