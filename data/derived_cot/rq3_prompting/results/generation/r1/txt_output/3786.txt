class Solution:
    def longestPalindromicSubsequence(self, s: str, k: int) -> int:
        n = len(s)
        arr = [ord(ch)-97 for ch in s]
        # precompute distances
        dist = [[0]*26 for _ in range(26)]
        for i in range(26):
            for j in range(26):
                d = abs(i-j)
                dist[i][j] = min(d, 26-d)
        # dp[l][r] is bytearray of length k+1
        dp = [[bytearray(k+1) for _ in range(n)] for __ in range(n)]
        # base case length1
        for i in range(n):
            dp[i][i][:] = bytearray([1]*(k+1))  # set all to 1
        # process lengths
        for length in range(2, n+1):
            for l in range(0, n-length+1):
                r = l+length-1
                # compute for all c from 0 to k
                # We can compute incrementally
                # We can compute best for each c using previous values
                # Since we need dp[l][r][c] for all c, we can loop c
                d = dist[arr[l]][arr[r]]
                # For each c
                # To speed, we can reuse: dp[l][r][c] = max(dp[l+1][r][c], dp[l][r-1][c], 2+dp[l+1][r-1][c-d] if c>=d)
                # We'll fill bytearray by loop
                # Access dp[l+1][r] etc.
                left = dp[l+1][r]   # bytearray
                right = dp[l][r-1]
                inner = dp[l+1][r-1] if l+1 <= r-1 else None
                row = dp[l][r]
                for c in range(k+1):
                    best = left[c]
                    if right[c] > best:
                        best = right[c]
                    if c >= d:
                        inner_val = inner[c-d] if inner is not None else 0
                        if 2 + inner_val > best:
                            best = 2 + inner_val
                    row[c] = best
        return dp[0][n-1][k]

class Solution:
    def longestPalindromicSubsequence(self, s: str, k: int) -> int:
        n = len(s)
        # convert to numbers 0..25
        vals = [ord(ch) - 97 for ch in s]

        # precompute circular distances between letters
        dist = [[0] * 26 for _ in range(26)]
        for i in range(26):
            for j in range(26):
                d = abs(i - j)
                dist[i][j] = min(d, 26 - d)

        # dp[l][r] is a bytearray of length k+1
        # dp[l][r][c] = longest palindromic subsequence in s[l..r] using at most c operations
        dp = [[bytearray(k + 1) for _ in range(n)] for __ in range(n)]

        # base: single character – always length 1, cost 0
        for i in range(n):
            dp[i][i] = bytearray([1] * (k + 1))

        # process intervals by increasing length
        for length in range(2, n + 1):
            for l in range(0, n - length + 1):
                r = l + length - 1
                d = dist[vals[l]][vals[r]]

                left = dp[l + 1][r]          # skip left
                rght = dp[l][r - 1]          # skip right
                # inner interval (if it exists)
                inner = dp[l + 1][r - 1] if length > 2 else None

                cur = dp[l][r]               # current row we fill

                for c in range(k + 1):
                    best = left[c]
                    if rght[c] > best:
                        best = rght[c]
                    if c >= d:
                        # take both ends, pay cost d
                        inner_val = inner[c - d] if inner is not None else 0
                        cand = 2 + inner_val
                        if cand > best:
                            best = cand
                    cur[c] = best

        # answer: best length for the whole string using ≤ k operations
        return dp[0][n - 1][k]
