def main():
    S1, S2 = input().split()
    if S1 == "sick":
        if S2 == "sick":
            print(1)
        else:  # S2 == "fine"
            print(2)
    else:  # S1 == "fine"
        if S2 == "sick":
            print(3)
        else:  # S2 == "fine"
            print(4)
