- 1 \leq X_1 < \dots < X_N \leq 10^9

    data = sys.stdin.read().strip().split()
    if not data:

    it = iter(data)
    N = int(next(it))
    xs = []
    hs = []

        h = int(next(it))
        xs.append(x)
        hs.append(h)
    
    # lines: a = -1/x, b = h/x
    lines = []
    for i in range(N):
        x = xs[i]
        h = hs[i]
        a = -1.0 / x
        b = h / x
        lines.append((a, b))

    cross = []  # intersections between hull[i-1] and hull[i], for i>=1
    ans = -1.0
    eps = 1e-12

    def intersect(i, j):
        # returns y where lines[i] and lines[j] intersect
        a1, b1 = lines[i]
        a2, b2 = lines[j]
        # a1 != a2 guaranteed because x distinct.
        return (b2 - b1) / (a1 - a2)

    for i in range(N):
        if not hull:

            continue
        # Before adding, pop unnecessary lines
        while len(hull) >= 2:
            # compute intersection of new line with last line
            y_new_last = intersect(i, hull[-1])

            y_last_second = cross[-1]  # cross between hull[-2] and hull[-1]
            if y_new_last <= y_last_second + eps:
                hull.pop()
                cross.pop()
            else:

        y_i = intersect(i, hull[-1])

        hull.append(i)
