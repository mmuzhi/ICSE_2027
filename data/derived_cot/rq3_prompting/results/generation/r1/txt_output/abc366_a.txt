=== ANSWER ===
```python
def solve() -> None:
    import sys
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N, T, A = map(int, data)
    if 2 * T > N or 2 * A > N:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    solve()
```
