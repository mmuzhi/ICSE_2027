=== ANSWER ===
```python
from typing import List
import itertools

class Solution:
    def maxGoodNumber(self, nums: List[int]) -> int:
        # Convert each number to its binary string (without '0b' prefix)
        bin_strs = [bin(x)[2:] for x in nums]
        max_val = 0
        # Try all 6 permutations
        for perm in itertools.permutations(bin_strs):
            concat = ''.join(perm)
            val = int(concat, 2)
            if val > max_val:
                max_val = val
        return max_val
```
