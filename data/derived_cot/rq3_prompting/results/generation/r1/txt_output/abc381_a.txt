def main():
    import sys
    input = sys.stdin.readline

    N = int(input().strip())
    S = input().strip()

    # length check (already given, but we can verify quickly)
    if len(S) != N:
        print("No")
        return

    # odd length?
    if N % 2 == 0:
        print("No")
        return

    mid = (N + 1) // 2  # 1-indexed middle position
    # positions before mid (0-indexed: 0 to mid-2)
    for i in range(mid - 1):
        if S[i] != '1':
            print("No")
            return
    # middle character
    if S[mid - 1] != '/':
        print("No")
        return
    # positions after mid (0-indexed: mid to N-1)
    for i in range(mid, N):
        if S[i] != '2':
            print("No")
            return

    print("Yes")

if __name__ == "__main__":
    main()
