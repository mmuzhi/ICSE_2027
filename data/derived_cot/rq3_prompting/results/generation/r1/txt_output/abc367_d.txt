import sys
from collections import defaultdict

def solve() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())
    A = list(map(int, input().split()))

    # prefix sums modulo M
    pref = [0] * (N + 1)
    for i in range(1, N + 1):
        pref[i] = (pref[i - 1] + A[i - 1]) % M

    S_mod = pref[N]                 # total circumference modulo M
    vals = [pref[i] for i in range(N)]   # val[s] = prefix sum up to s-1

    # frequencies for the right (unprocessed) part
    right = defaultdict(int)
    for v in vals:
        right[v] += 1

    left = defaultdict(int)   # frequencies for the left (processed) part
    ans = 0

    for v in vals:
        # remove current element from right
        right[v] -= 1
        if right[v] == 0:
            del right[v]

        # pairs with t < s  -> need left value = (v - S_mod) mod M
        target = (v - S_mod) % M
        ans += left.get(target, 0)

        # pairs with t > s  -> need right value = v
        ans += right.get(v, 0)

        # insert current element into left
        left[v] += 1

    print(ans)

if __name__ == "__main__":
    solve()
