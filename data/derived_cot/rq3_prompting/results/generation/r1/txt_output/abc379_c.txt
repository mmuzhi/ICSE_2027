There are N cells numbered from 1 to N in a row. Initially, M cells contain stones, and cell X_i contains A_i stones (1 \leq i \leq M).

- If cell i (1 \leq i \leq N-1) contains a stone, move one stone from cell i to cell i+1.

Find the minimum number of operations required to reach a state where each of the N cells contains exactly one stone. If it is impossible, print -1.

Input

The input is given from Standard Input in the following format:

- 1 \leq M \leq 2 \times 10^{5}
- M \leq N
- 1 \leq X_i \leq N (1 \leq i \leq M)
- X_i \neq X_j (1 \leq i < j \leq M)
- 1 \leq A_i \leq 2 \times 10^{9} (1 \leq i \leq M)
- All input values are integers.

import sys

def solve():
    data = sys.stdin.read().split()
    if not data: return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X = [int(next(it)) for _ in range(M)]
    A = [int(next(it)) for _ in range(M)]
    total_stones = sum(A)
    if total_stones != N:
        print(-1)
        return
    # pair and sort
    pairs = sorted(zip(X, A), key=lambda x: x[0])
    # check first
    if pairs[0][0] != 1:
        print(-1)
        return
    cum = 0
    for i in range(M - 1):
        cum += pairs[i][1]
        next_x = pairs[i+1][0]
        if cum < next_x - 1:
            print(-1)
            return
    # cumulative sum after last stone is total N, but we already ensured
    # compute sum_xa
    sum_xa = 0
    for x, a in pairs:
        sum_xa += x * a
    total_moves = N * (N + 1) // 2 - sum_xa
    print(total_moves)

if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))

    X = [int(next(it)) for _ in range(M)]
    A = [int(next(it)) for _ in range(M)]

    # total stones must equal N
    if sum(A) != N:
        print(-1)
        return

    # sort by cell index
    cells = sorted(zip(X, A), key=lambda p: p[0])

    # first cell must contain at least one stone initially
    if cells[0][0] != 1:
        print(-1)
        return

    cum = 0
    # check the condition for every gap between stones
    for i in range(M - 1):
        cum += cells[i][1]                  # stones up to cells[i][0]
        next_x = cells[i + 1][0]
        if cum < next_x - 1:                # need enough stones to fill up to next_x-1
            print(-1)
            return

    # compute total moves = sum(target positions) - sum(initial positions)
    sum_target = N * (N + 1) // 2
    sum_initial = 0
    for x, a in cells:
        sum_initial += x * a

    print(sum_target - sum_initial)


if __name__ == "__main__":
    solve()
