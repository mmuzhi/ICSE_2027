=== ANSWER ===
```python
from typing import List

class Solution:
    def stringSequence(self, target: str) -> List[str]:
        result = []
        current = []  # list of characters building the string
        
        for ch in target:
            # Step 1: append 'a' (key 1)
            current.append('a')
            result.append(''.join(current))
            
            # Step 2: increment last character until it matches the target character
            while current[-1] != ch:
                # Increment last character (no wrap needed because we start from 'a' and go up)
                last_char = current[-1]
                next_char = chr(ord(last_char) + 1)  # safe because we stop before 'z'+1
                current[-1] = next_char
                result.append(''.join(current))
        
        return result
```
