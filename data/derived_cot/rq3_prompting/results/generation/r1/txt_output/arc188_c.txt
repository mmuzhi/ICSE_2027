You will be given a competitive programming problem.
There is a village with N villagers numbered from 1 to N.
Each villager is honest or a liar. Additionally, some villagers are confused.
You have obtained M testimonies from the villagers. Each testimony is given by A_i, B_i, C_i for i=1,2,\ldots,M, representing:

- If C_i=0, villager A_i testified that villager B_i is honest.
- If C_i=1, villager A_i testified that villager B_i is a liar.

- An honest villager who is not confused always tells the truth.
- A liar who is not confused always tells lies.
- A confused honest villager always tells lies.
- A confused liar always tells the truth.

In other words, if they are not confused, honest villagers always tell the truth, and liars always tell lies, but if they are confused, it is reversed.
You have decided to guess the set of villagers who are confused.
Given a choice of villagers who are confused, whether the set of testimonies "contradicts" or not is determined.
Here, a set of testimonies is said to contradict if, no matter how you assign honest or liar statuses to the villagers, there is at least one testimony that violates the villagers' testimony rules.
Find a set of confused villagers such that the given set of testimonies does not contradict.
If no such set of confused villagers exists, indicate that fact.

If no such set of confused villagers exists, print -1.

- 2 \leq N \leq 2 \times 10^5
- 0 \leq M \leq \mathrm{min} \lbrace 2 \times 10^5,N(N-1) \rbrace 
- 1 \leq A_i, B_i \leq N, A_i \neq B_i
- A_i \neq A_j or B_i \neq B_j for i \neq j.
- C_i = 0 or 1.

class DSU:
    def __init__(self, n):
        self.parent = list(range(n))
        self.size = [1]*n
        self.xor = [0]*n  # xor to parent

    def find(self, x):
        if self.parent[x] != x:
            orig_parent = self.parent[x]
            self.parent[x] = self.find(self.parent[x])
            self.xor[x] ^= self.xor[orig_parent]
        return self.parent[x]

    def union(self, x, y, w):
        # enforce val[x] xor val[y] = w
        rx = self.find(x)
        ry = self.find(y)
        if rx == ry:
            # check constraint
            if (self.xor[x] ^ self.xor[y]) != w:
                return False
            return True
        # union by size
        if self.size[rx] < self.size[ry]:
            # attach rx under ry
            # need set parent[rx] = ry, and set xor[rx] such that:
            # val[x] = self.xor[x] xor val[rx]
            # val[y] = self.xor[y] xor val[ry]
            # we need val[x] xor val[y] = w
            # so self.xor[x] xor val[rx] xor self.xor[y] xor val[ry] = w
            # => val[rx] xor val[ry] = w xor self.xor[x] xor self.xor[y]

            self.xor[rx] = w ^ self.xor[x] ^ self.xor[y]

import sys
sys.setrecursionlimit(1<<25)

def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    parent = list(range(N))
    size = [1]*N
    xor = [0]*N

    def find(x):
        if parent[x] != x:
            orig = parent[x]
            parent[x] = find(parent[x])
            xor[x] ^= xor[orig]
        return parent[x]

    def union(x, y, w):
        rx = find(x)
        ry = find(y)
        if rx == ry:
            return (xor[x] ^ xor[y]) == w
        if size[rx] < size[ry]:
            # attach rx to ry
            parent[rx] = ry
            # need: xor[x] ^ val[rx] ^ xor[y] ^ val[ry] = w
            # val[rx] = xor[rx] ^ val[ry] (since rx attached to ry)
            # So xor[x] ^ (xor[rx] ^ val[ry]) ^ xor[y] ^ val[ry] = w
            # => xor[x] ^ xor[rx] ^ xor[y] = w
            # => xor[rx] = w ^ xor[x] ^ xor[y]
            xor[rx] = w ^ xor[x] ^ xor[y]
            size[ry] += size[rx]
        else:
            parent[ry] = rx
            # Need: val[ry] = xor[ry] ^ val[rx]
            # val[x] ^ val[y] = w => xor[x] ^ val[rx] ^ xor[y] ^ val[ry] = w
            # substitute: xor[x] ^ val[rx] ^ xor[y] ^ (xor[ry] ^ val[rx]) = w
            # => xor[x] ^ xor[y] ^ xor[ry] = w
            # => xor[ry] = w ^ xor[x] ^ xor[y]
            xor[ry] = w ^ xor[x] ^ xor[y]
            size[rx] += size[ry]
        return True

    first_testimony = [None] * (N+1)  # 1-indexed

    for _ in range(M):
        A, B, C = map(int, input().split())
        if first_testimony[A] is None:
            first_testimony[A] = (B, C)
        else:
            B1, C1 = first_testimony[A]
            w = C ^ C1
            # union B1-1 and B-1
            if not union(B1-1, B-1, w):
                print(-1)
                return

    # assign h values
    h = [0]*N
    for i in range(N):
        find(i)  # compress
        h[i] = xor[i]  # since root value assumed 0

    # compute c values
    c = [0]*N
    for A in range(1, N+1):
        if first_testimony[A] is not None:
            B, C = first_testimony[A]
            d = h[B-1] ^ C
            c[A-1] = h[A-1] ^ d
        # else remains 0

    # output
    print(''.join(str(ci) for ci in c))

if __name__ == "__main__":
    solve()

```python
import sys

def solve() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())

    parent = list(range(N))
    size = [1] * N
    xor_to_parent = [0] * N      # xor from node to its parent

    # ---------- DSU with xor ----------
    def find(x: int) -> int:
        if parent[x] != x:
            orig = parent[x]
            parent[x] = find(parent[x])
            xor_to_parent[x] ^= xor_to_parent[orig]
        return parent[x]

    def union(x: int, y: int, w: int) -> bool:
        """enforce val[x] ^ val[y] = w; return False on contradiction"""
        rx = find(x)
        ry = find(y)
        if rx == ry:
            return (xor_to_parent[x] ^ xor_to_parent[y]) == w
        # attach smaller tree under larger
        if size[rx] < size[ry]:
            # attach rx under ry
            parent[rx] = ry
            # xor[rx] must satisfy: val[x] ^ val[y] = w
            xor_to_parent[rx] = w ^ xor_to_parent[x] ^ xor_to_parent[y]
            size[ry] += size[rx]
        else:
            # attach ry under rx
            parent[ry] = rx
            xor_to_parent[ry] = w ^ xor_to_parent[x] ^ xor_to_parent[y]
            size[rx] += size[ry]
        return True
    # ---------------------------------

    # store the first testimony for each testifier
    first_testimony = [None] * (N + 1)   # 1‑based indexing

    for _ in range(M):
        A, B, C = map(int, input().split())
        if first_testimony[A] is None:
            first_testimony[A] = (B, C)
        else:
            B1, C1 = first_testimony[A]
            w = C ^ C1
            if not union(B1 - 1, B - 1, w):
                print(-1)
                return

    # assign honest/liar values (h) to each villager
    h = [0] * N
    for i in range(N):
        find(i)                     # path compression
        h[i] = xor_to_parent[i]     # root value is taken as 0

    # compute confused flags
    c = [0] * N
    for A in range(1, N + 1):
        if first_testimony[A] is not None:
            B, C = first_testimony[A]
            d = h[B - 1] ^ C
            c[A - 1] = h[A - 1] ^ d
        # else: c stays 0

    print(''.join(str(ci) for ci in c))

if __name__ == "__main__":
    solve()
```
