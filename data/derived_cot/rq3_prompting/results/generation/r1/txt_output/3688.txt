from typing import List

class Solution:
    def maxSubarraySum(self, nums: List[int]) -> int:
        n = len(nums)
        INF = 10**18

        # ---------- segment tree for maximal subarray ----------
        size = 1
        while size < n:
            size <<= 1

        # each node: (total_sum, best_subarray, best_prefix, best_suffix)
        tree = [(0, -INF, -INF, -INF)] * (2 * size)

        # leaves
        for i in range(n):
            v = nums[i]
            tree[size + i] = (v, v, v, v)

        # build internal nodes
        for i in range(size - 1, 0, -1):
            left = tree[2 * i]
            right = tree[2 * i + 1]
            total = left[0] + right[0]
            best = max(left[1], right[1], left[3] + right[2])
            pref = max(left[2], left[0] + right[2])
            suff = max(right[3], right[0] + left[3])
            tree[i] = (total, best, pref, suff)

        def merge(a, b):
            """combine two adjacent intervals"""
            total = a[0] + b[0]
            best = max(a[1], b[1], a[3] + b[2])
            pref = max(a[2], a[0] + b[2])
            suff = max(b[3], b[0] + a[3])
            return (total, best, pref, suff)

        def range_query(l, r):
            """query on interval [l, r] inclusive (0-indexed)"""
            l += size
            r += size
            left_res = (0, -INF, -INF, -INF)   # identity
            right_res = (0, -INF, -INF, -INF)
            while l <= r:
                if l & 1:
                    left_res = merge(left_res, tree[l])
                    l += 1
                if not (r & 1):
                    right_res = merge(tree[r], right_res)
                    r -= 1
                l >>= 1
                r >>= 1
            return merge(left_res, right_res)

        # ---------- group indices by value ----------
        pos = {}
        for i, v in enumerate(nums):
            pos.setdefault(v, []).append(i)

        # answer starts with original array (no deletion)
        ans = tree[1][1]

        # ---------- process each distinct value ----------
        for v, idxs in pos.items():
            prev = -1
            seg_nodes = []
            for idx in idxs:
                if idx > prev + 1:
                    seg = range_query(prev + 1, idx - 1)
                    seg_nodes.append(seg)
                prev = idx
            if prev < n - 1:
                seg = range_query(prev + 1, n - 1)
                seg_nodes.append(seg)

            if not seg_nodes:          # deleting v would empty the array – skip
                continue

            # combine the segments
            cur = seg_nodes[0]
            for seg in seg_nodes[1:]:
                cur = merge(cur, seg)
            ans = max(ans, cur[1])     # best subarray of the merged array

        return ans
