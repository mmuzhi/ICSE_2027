=== ANSWER ===
```python
import sys
import itertools

def main():
    A = list(map(int, sys.stdin.readline().split()))
    for perm in itertools.permutations(A):
        if perm[0] * perm[1] == perm[2]:
            print("Yes")
            return
    print("No")

if __name__ == "__main__":
    main()
```
