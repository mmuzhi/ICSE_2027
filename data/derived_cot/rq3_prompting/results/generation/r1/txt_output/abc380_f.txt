Takahashi and Aoki will play a game using cards with numbers written on them.
Initially, Takahashi has N cards with numbers A_1, \ldots, A_N in his hand, Aoki has M cards with numbers B_1, \ldots, B_M in his hand, and there are L cards with numbers C_1, \ldots, C_L on the table.

It can be proved that the game always ends in a finite number of moves.

Constraints


- 1 \leq N, M, L
- N + M + L \leq 12
- 1 \leq A_i, B_i, C_i \leq 10^9
- All input values are integers.

import sys
from functools import lru_cache

sys.setrecursionlimit(10000)

def solve() -> None:
    it = iter(sys.stdin.read().split())
    N = int(next(it))
    M = int(next(it))
    L = int(next(it))
    A = tuple(sorted(int(next(it)) for _ in range(N)))
    B = tuple(sorted(int(next(it)) for _ in range(M)))
    C = tuple(sorted(int(next(it)) for _ in range(L)))

    @lru_cache(maxsize=None)
    def win(T_hand, A_hand, table, turn: int) -> bool:
        """return True if the player to move can force a win"""
        hand = T_hand if turn == 0 else A_hand
        if not hand:                     # no card to play
            return False

        for i, x in enumerate(hand):
            # play card x without taking any card from the table
            new_hand = hand[:i] + hand[i+1:]
            new_table = tuple(sorted(table + (x,)))
            next_T = new_hand if turn == 0 else T_hand
            next_A = A_hand if turn == 0 else new_hand
            if not win(next_T, next_A, new_table, 1 - turn):
                return True

            # play card x and take one smaller card y from the table
            for j, y in enumerate(table):
                if y < x:
                    new_table2 = tuple(sorted(table[:j] + table[j+1:] + (x,)))
                    new_hand2 = tuple(sorted(new_hand + (y,)))
                    next_T2 = new_hand2 if turn == 0 else T_hand
                    next_A2 = A_hand if turn == 0 else new_hand2
                    if not win(next_T2, next_A2, new_table2, 1 - turn):
                        return True
                else:
                    # table is sorted, no further y can be smaller
                    break
        return False

    if win(A, B, C, 0):
        print("Takahashi")
    else:
        print("Aoki")
