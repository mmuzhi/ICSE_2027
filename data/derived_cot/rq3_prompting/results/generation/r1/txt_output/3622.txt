```python
from typing import List
import bisect
from collections import Counter

class Solution:
    def maxFrequency(self, nums: List[int], k: int, numOperations: int) -> int:
        nums.sort()
        freq = Counter(nums)
        candidates = set()
        for v in freq.keys():
            candidates.add(v)
            candidates.add(v - k)
            candidates.add(v + k)
        # also consider maybe extreme? But any x outside will have empty interval, so not needed.
        n = len(nums)
        ans = 0
        for x in candidates:
            # count in [x-k, x+k]
            left = bisect.bisect_left(nums, x - k)
            right = bisect.bisect_right(nums, x + k) - 1
            total = right - left + 1 if left <= right else 0
            already = freq.get(x, 0)
            possible = already + min(numOperations, total - already)
            if possible > ans:
                ans = possible
        return ans
```

```python
from typing import List
import bisect
from collections import Counter

class Solution:
    def maxFrequency(self, nums: List[int], k: int, numOperations: int) -> int:
        # Sort to enable binary search for interval counts
        nums.sort()
        freq = Counter(nums)
        
        # Collect all candidate target values
        candidates = set()
        for v in freq:
            candidates.add(v)
            candidates.add(v - k)
            candidates.add(v + k)
        
        n = len(nums)
        best = 0
        
        for x in candidates:
            # Number of elements that can become x (i.e. in [x-k, x+k])
            left = bisect.bisect_left(nums, x - k)
            right = bisect.bisect_right(nums, x + k) - 1
            total = right - left + 1 if left <= right else 0
            
            # Already equal to x
            already = freq.get(x, 0)
            
            # We can convert at most numOperations elements that are not already x
            possible = already + min(numOperations, total - already)
            if possible > best:
                best = possible
        
        return best
```
