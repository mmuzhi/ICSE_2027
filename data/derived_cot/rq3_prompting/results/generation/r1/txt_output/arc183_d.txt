import sys
import heapq

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    g = [[] for _ in range(N + 1)]
    for i in range(N - 1):
        a = int(next(it))
        b = int(next(it))
        g[a].append(b)
        g[b].append(a)

    # ---------- find diameter and center ----------
    def bfs(start):
        dist = [-1] * (N + 1)
        q = [start]
        dist[start] = 0
        for v in q:
            for w in g[v]:
                if dist[w] == -1:
                    dist[w] = dist[v] + 1
                    q.append(w)
        # farthest node
        farthest = max(range(1, N+1), key=lambda x: dist[x])
        return farthest, dist

    u, _ = bfs(1)
    v, dist_u = bfs(u)
    d = dist_u[v]

    # go from v back to u to get the center
    # first BFS from u also gave parents, but we can run BFS from u again with parent tracking
    # simpler: use parent array from BFS from u
    # let's run BFS from u again to get parent
    parent = [-1] * (N + 1)
    q = [u]
    parent[u] = 0
    for cur in q:
        for nb in g[cur]:
            if parent[nb] == -1:
                parent[nb] = cur
                q.append(nb)
    # walk from v up d//2 steps
    steps = d // 2
    center = v
    for _ in range(steps):
        center = parent[center]
    # ---------- compute depths from center ----------
    depth = [-1] * (N + 1)
    q = [center]
    depth[center] = 0
    for cur in q:
        for nb in g[cur]:
            if depth[nb] == -1:
                depth[nb] = depth[cur] + 1
                q.append(nb)

    # ---------- initial degree ----------
    deg = [0] * (N + 1)
    for v in range(1, N+1):
        deg[v] = len(g[v])

    # ---------- heaps for leaves ----------
    odd_heap = []   # max-heap by depth (store -depth)
    even_heap = []
    for v in range(1, N+1):
        if deg[v] == 1:
            if depth[v] & 1:
                heapq.heappush(odd_heap, (-depth[v], v))
            else:
                heapq.heappush(even_heap, (-depth[v], v))

    active = [True] * (N + 1)
    removed = 0
    out_lines = []

    while removed < N:
        # clean odd heap
        while odd_heap and (not active[odd_heap[0][1]] or deg[odd_heap[0][1]] != 1):
            heapq.heappop(odd_heap)
        # clean even heap
        while even_heap and (not active[even_heap[0][1]] or deg[even_heap[0][1]] != 1):
            heapq.heappop(even_heap)
        # both heaps must have at least one element
        u = odd_heap[0][1]
        v = even_heap[0][1]
        out_lines.append(f"{u} {v}")

        # remove u
        active[u] = False
        deg[u] = 0
        for nb in g[u]:
            if active[nb]:
                deg[nb] -= 1
                if deg[nb] == 1:
                    if depth[nb] & 1:
                        heapq.heappush(odd_heap, (-depth[nb], nb))
                    else:
                        heapq.heappush(even_heap, (-depth[nb], nb))

        # remove v
        active[v] = False
        deg[v] = 0
        for nb in g[v]:
            if active[nb]:
                deg[nb] -= 1
                if deg[nb] == 1:
                    if depth[nb] & 1:
                        heapq.heappush(odd_heap, (-depth[nb], nb))
                    else:
                        heapq.heappush(even_heap, (-depth[nb], nb))

        removed += 2

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
