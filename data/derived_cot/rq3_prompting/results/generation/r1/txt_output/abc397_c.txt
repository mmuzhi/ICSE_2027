=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    A = [int(next(it)) for _ in range(n)]

    # suffix distinct counts
    suff = [0] * n
    seen = [False] * (n + 1)
    cnt = 0
    for i in range(n - 1, -1, -1):
        x = A[i]
        if not seen[x]:
            seen[x] = True
            cnt += 1
        suff[i] = cnt

    # prefix distinct counts and compute answer
    seen = [False] * (n + 1)
    cnt = 0
    ans = 0
    for i in range(n - 1):
        x = A[i]
        if not seen[x]:
            seen[x] = True
            cnt += 1
        # split after i: prefix distinct = cnt, suffix distinct = suff[i+1]
        total = cnt + suff[i + 1]
        if total > ans:
            ans = total

    sys.stdout.write(str(ans))

if __name__ == "__main__":
    solve()
```
