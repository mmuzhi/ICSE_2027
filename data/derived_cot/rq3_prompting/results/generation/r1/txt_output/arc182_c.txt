import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])

    primes = [2, 3, 5, 7, 11, 13]          # all primes ‚Č§ 16
    np = len(primes)
    nstates = 1 << np

    # c[mask] = sum_{a=1..M} prod_{p in mask} v_p(a)
    c = [0] * nstates
    for a in range(1, M + 1):
        exps = [0] * np
        x = a
        for i, p in enumerate(primes):
            while x % p == 0:
                exps[i] += 1
                x //= p
        # add contributions for all masks
        for mask in range(nstates):
            prod = 1
            for i in range(np):
                if mask >> i & 1:
                    prod *= exps[i]
            c[mask] = (c[mask] + prod) % MOD

    # Build transition matrix A (nstates x nstates)
    A = [[0] * nstates for _ in range(nstates)]
    for s in range(nstates):
        for nxt in range(nstates):
            if (nxt & ~s) == 0:          # nxt ‚äÜ s
                t = s ^ nxt              # bits removed
                A[s][nxt] = c[t] % MOD

    # Identity matrix
    I = [[0] * nstates for _ in range(nstates)]
    for i in range(nstates):
        I[i][i] = 1

    # Matrix multiplication
    def mat_mul(X, Y):
        n = len(X)
        Z = [[0] * n for _ in range(n)]
        for i in range(n):
            Xi = X[i]
            Zi = Z[i]
            for k in range(n):
                aik = Xi[k]
                if aik:
                    Yk = Y[k]
                    for j in range(n):
                        Zi[j] = (Zi[j] + aik * Yk[j]) % MOD
        return Z

    # Matrix addition
    def mat_add(X, Y):
        n = len(X)
        Z = [[0] * n for _ in range(n)]
        for i in range(n):
            Xi = X[i]
            Yi = Y[i]
            Zi = Z[i]
            for j in range(n):
                Zi[j] = (Xi[j] + Yi[j]) % MOD
        return Z

    # Compute geometric series sum_{k=1}^{N} A^k
    result_sum = [[0] * nstates for _ in range(nstates)]   # zero matrix
    result_pow = [row[:] for row in I]                     # I
    cur_pow = [row[:] for row in A]                        # A
    cur_sum = [row[:] for row in A]                        # A (sum of first 1 term)

    n = N
    while n:
        if n & 1:
            # result_sum += result_pow * cur_sum
            temp = mat_mul(result_pow, cur_sum)
            result_sum = mat_add(result_sum, temp)
            # result_pow = result_pow * cur_pow
            result_pow = mat_mul(result_pow, cur_pow)
        # prepare for next bit: double the block
        # cur_sum = cur_sum + cur_pow * cur_sum
        temp = mat_mul(cur_pow, cur_sum)
        cur_sum = mat_add(cur_sum, temp)
        # cur_pow = cur_pow * cur_pow
        cur_pow = mat_mul(cur_pow, cur_pow)
        n >>= 1

    full_mask = (1 << np) - 1
    row = result_sum[full_mask]
    ans = sum(row) % MOD
    print(ans)

if __name__ == "__main__":
    solve()
