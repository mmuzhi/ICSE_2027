    input = sys.stdin.readline
    N, K = map(int, input().split())
    funcs = []

        A, B = map(int, input().split())

    # sort by (A-1)/B descending
    funcs.sort(key=lambda x: (x[0]-1)/x[1], reverse=True)
    dp = [0]*(K+1)
    dp[0] = 1
    for i in range(1, K+1):
        dp[i] = -1  # sentinel
    # iterate from innermost (last in sorted) to outermost (first)
    for A, B in reversed(funcs):
        # update from high count to low to avoid reuse
        for cnt in range(K-1, -1, -1):
            if dp[cnt] != -1:
                new_val = A * dp[cnt] + B
                if new_val > dp[cnt+1]:
                    dp[cnt+1] = new_val
    print(dp[K])

import sys

def solve() -> None:
    input = sys.stdin.readline
    N, K = map(int, input().split())
    funcs = []
    for _ in range(N):
        A, B = map(int, input().split())
        funcs.append((A, B))

    # sort by (A-1)/B descending – the optimal composition order
    funcs.sort(key=lambda x: (x[0] - 1) / x[1], reverse=True)

    # dp[c] = maximum value after composing c functions (starting from x=1)
    dp = [-1] * (K + 1)
    dp[0] = 1  # initial value before any function

    # Process functions from innermost (rightmost in sorted list) to outermost
    for A, B in reversed(funcs):
        # update from high count to low to avoid using the same function twice
        for c in range(K - 1, -1, -1):
            if dp[c] != -1:
                nxt = A * dp[c] + B
                if nxt > dp[c + 1]:
                    dp[c + 1] = nxt

    print(dp[K])

if __name__ == "__main__":
    solve()
