import sys
def solve():
    import sys
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    A = list(map(int, data[1:1+n]))
    cnt = 0
    while True:
        A.sort(reverse=True)
        if len(A) < 2 or A[1] <= 0:
            break
        A[0] -= 1
        A[1] -= 1
        cnt += 1
    print(cnt)
if __name__ == "__main__":
