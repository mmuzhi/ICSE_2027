def solve():
    import sys
    data = sys.stdin.read().split()
    it = iter(data)
    n = int(next(it))
    adj = [[] for _ in range(n+1)]
    deg = [0]*(n+1)
    edges = []
    for _ in range(n-1):
        u = int(next(it)); v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
        deg[u] += 1
        deg[v] += 1
        edges.append((u,v))
    deg3_vertices = [v for v in range(1, n+1) if deg[v]==3]
    comp_id = [-1]*(n+1)
    comp_cnt = 0
    from collections import deque
    for v in deg3_vertices:
        if comp_id[v] == -1:
            comp_cnt += 1
            q = deque([v])
            comp_id[v] = comp_cnt
            while q:
                node = q.popleft()
                for nb in adj[node]:
                    if deg[nb]==3 and comp_id[nb]==-1:
                        comp_id[nb] = comp_cnt
                        q.append(nb)
    # comps_of for deg2 vertices
    comps_of = [set() for _ in range(n+1)]  # index by vertex
    for v in range(1, n+1):
        if deg[v] == 2:
            s = set()
            for nb in adj[v]:
                if deg[nb]==3:
                    s.add(comp_id[nb])
            comps_of[v] = s  # set
    # component_to_vertices
    comp_verts = [[] for _ in range(comp_cnt+1)]  # 1-indexed
    for v in range(1, n+1):
        if deg[v]==2:
            for c in comps_of[v]:
                comp_verts[c].append(v)
    total = 0
    for c in range(1, comp_cnt+1):
        k = len(comp_verts[c])
        total += k*(k-1)//2
    # intersection pairs
    from collections import defaultdict
    pair_cnt = defaultdict(int)
    for v in range(1, n+1):
        if deg[v]==2 and len(comps_of[v])==2:
            c1, c2 = sorted(comps_of[v])
            pair_cnt[(c1,c2)] += 1
    for (c1,c2), cnt in pair_cnt.items():
        total -= cnt*(cnt-1)//2
    # subtract adjacent deg2 pairs that share component
    subtract_adj = 0
    for (u,v) in edges:
        if deg[u]==2 and deg[v]==2:
            # check intersection
            if comps_of[u] & comps_of[v]:  # non-empty
                subtract_adj += 1
    total -= subtract_adj
    print(total)

import sys
from collections import deque, defaultdict

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    adj = [[] for _ in range(n + 1)]
    deg = [0] * (n + 1)
    edges = []
    for _ in range(n - 1):
        u = int(next(it))
        v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
        deg[u] += 1
        deg[v] += 1
        edges.append((u, v))

    # vertices with degree 3
    deg3_vertices = [v for v in range(1, n + 1) if deg[v] == 3]

    # connected components of the subgraph induced by degree‑3 vertices
    comp_id = [-1] * (n + 1)
    comp_cnt = 0
    for start in deg3_vertices:
        if comp_id[start] != -1:
            continue
        comp_cnt += 1
        q = deque([start])
        comp_id[start] = comp_cnt
        while q:
            cur = q.popleft()
            for nb in adj[cur]:
                if deg[nb] == 3 and comp_id[nb] == -1:
                    comp_id[nb] = comp_cnt
                    q.append(nb)

    # for each degree‑2 vertex, the set of components it attaches to
    comps_of = [set() for _ in range(n + 1)]
    for v in range(1, n + 1):
        if deg[v] != 2:
            continue
        s = set()
        for nb in adj[v]:
            if deg[nb] == 3:
                s.add(comp_id[nb])
        comps_of[v] = s

    # list of degree‑2 vertices attached to each component
    comp_verts = [[] for _ in range(comp_cnt + 1)]   # 1‑based
    for v in range(1, n + 1):
        if deg[v] != 2:
            continue
        for c in comps_of[v]:
            comp_verts[c].append(v)

    # sum over components of C(|A_C|,2)
    total = 0
    for c in range(1, comp_cnt + 1):
        k = len(comp_verts[c])
        total += k * (k - 1) // 2

    # subtract pairs that are counted in two components
    pair_count = defaultdict(int)
    for v in range(1, n + 1):
        if deg[v] == 2 and len(comps_of[v]) == 2:
            c1, c2 = sorted(comps_of[v])
            pair_count[(c1, c2)] += 1
    for cnt in pair_count.values():
        total -= cnt * (cnt - 1) // 2

    # subtract adjacent degree‑2 pairs (they would create a multigraph)
    subtract_adj = 0
    for u, v in edges:
        if deg[u] == 2 and deg[v] == 2:
            if comps_of[u] & comps_of[v]:   # share at least one component
                subtract_adj += 1
    total -= subtract_adj

    print(total)
