class SegTree:
    def __init__(self, n):
        self.N = 1
        while self.N < n:
            self.N <<= 1
        INF = 10**18
        self.tree = [(INF, -1)] * (2 * self.N)
    def update(self, pos, val): # val is tuple (dist, index)
        i = pos + self.N
        self.tree[i] = val
        i //= 2
        while i:
            left = self.tree[2*i]
            right = self.tree[2*i+1]
            self.tree[i] = self.merge(left, right)
            i //= 2
    def query(self, l, r): # inclusive
        res = (10**18, -1)
        l += self.N
        r += self.N
        while l <= r:
            if l % 2 == 1:
                res = self.merge(res, self.tree[l])
                l += 1
            if r % 2 == 0:
                res = self.merge(res, self.tree[r])
                r -= 1
            l //= 2
            r //= 2
        return res
    def merge(self, a, b):
        if a[0] < b[0]:
            return a
        if b[0] < a[0]:
            return b
        # equal dist: choose larger index
        return a if a[1] > b[1] else b

import sys
from typing import List

class Solution:
    def longestSpecialPath(self, edges: List[List[int]], nums: List[int]) -> List[int]:
        n = len(nums)
        # build adjacency list
        adj = [[] for _ in range(n)]
        for u, v, w in edges:
            adj[u].append((v, w))
            adj[v].append((u, w))

        # global answer
        self.max_len = 0
        self.min_nodes = 1

        # recursion limit for deep trees
        sys.setrecursionlimit(1000000)

        # path state
        path_nodes = []   # nodes on current path (not strictly needed)
        path_dist = []    # distance from root to each node on the path
        last_seen = {}    # value -> last index (depth) where it appeared

        def dfs(u: int, parent: int, cur_dist: int, left: int) -> None:
            # add current node to the path
            idx = len(path_nodes)          # current depth
            path_nodes.append(u)
            path_dist.append(cur_dist)

            # update left bound because of possible duplicate value
            val = nums[u]
            if val in last_seen:
                prev = last_seen[val]
                left = max(left, prev + 1)

            # remember the previous occurrence to restore later
            old = last_seen.get(val)
            last_seen[val] = idx

            # evaluate all special paths ending at u
            # the longest one starts at the leftmost valid ancestor
            length = cur_dist - path_dist[left]
            nodes = idx - left + 1
            if length > self.max_len or (length == self.max_len and nodes < self.min_nodes):
                self.max_len = length
                self.min_nodes = nodes

            # recurse to children
            for v, w in adj[u]:
                if v != parent:
                    dfs(v, u, cur_dist + w, left)

            # backtrack: remove current node from path and restore last_seen
            path_nodes.pop()
            path_dist.pop()
            if old is None:
                del last_seen[val]
            else:
                last_seen[val] = old

        dfs(0, -1, 0, 0)

        return [self.max_len, self.min_nodes]
