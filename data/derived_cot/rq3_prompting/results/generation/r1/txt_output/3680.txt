```python
from typing import List

class Solution:
    def countComponents(self, nums: List[int], threshold: int) -> int:
        T = threshold
        # Count isolated numbers > T
        isolated = 0
        # We'll first build set of numbers <= T
        present = [False] * (T + 1)
        small_nums = []
        for x in nums:
            if x > T:
                isolated += 1
            else:
                present[x] = True
                small_nums.append(x)  # list of numbers <= T

        # DSU for values up to T (only those present matter)
        parent = list(range(T + 1))
        size = [1] * (T + 1)  # optional, for union by size

        def find(x):
            while parent[x] != x:
                parent[x] = parent[parent[x]]
                x = parent[x]
            return x

        def union(x, y):
            rx, ry = find(x), find(y)
            if rx == ry:
                return
            if size[rx] < size[ry]:
                rx, ry = ry, rx
            parent[ry] = rx
            size[rx] += size[ry]

        # rep[L] stores a representative number that divides L
        rep = [-1] * (T + 1)

        # Process each small number
        for a in small_nums:
            # iterate multiples of a
            step = a
            for L in range(a, T + 1, step):
                if rep[L] == -1:
                    rep[L] = a
                else:
                    union(a, rep[L])

        # Count distinct roots among present numbers
        roots = set()
        for a in small_nums:
            roots.add(find(a))
        comps_small = len(roots)

        return comps_small + isolated
```
