=== ANSWER ===
```python
from typing import List

class Solution:
    def minimumOperations(self, nums: List[int]) -> int:
        n = len(nums)
        # Maximum number of operations needed is ceil(n/3)
        # because after that the array becomes empty.
        for ops in range((n + 2) // 3 + 1):
            start = min(n, 3 * ops)          # number of elements removed after ops operations
            suffix = nums[start:]           # remaining part
            if len(set(suffix)) == len(suffix):  # distinct check
                return ops
        # Fallback (should never be reached)
        return (n + 2) // 3
```
