import sys

def solve() -> None:
    it = iter(sys.stdin.read().strip().split())
    n = int(next(it))
    A = [int(next(it)) for _ in range(n)]
    B = [int(next(it)) for _ in range(n)]
    C = [int(next(it)) for _ in range(n)]

    down = []
    up = []
    always1 = []

    for i in range(n):
        a, b = A[i], B[i]
        if a == 1 and b == 0:
            down.append(C[i])
        elif a == 0 and b == 1:
            up.append(C[i])
        elif a == 1 and b == 1:
            always1.append(C[i])

    d = len(down)
    u = len(up)
    # No bit needs to be changed -> answer is 0
    if d + u == 0:
        print(0)
        return

    # down bits: sort decreasing
    down.sort(reverse=True)
    base_down = 0
    sum_down = sum(down)
    for i, val in enumerate(down):
        base_down += val * i          # (rank-1) = i

    # up bits: sort increasing
    up.sort()
    base_up = 0
    sum_up = sum(up)
    for i, val in enumerate(up):
        base_up += val * (u - i)      # (u - rank + 1) = u - i

    # always1 bits: sort increasing for prefix sums
    always1.sort()
    P = len(always1)
    pref_sum = [0] * (P + 1)
    pref_wsum = [0] * (P + 1)        # wsum = value * index (1‑based)
    for i in range(1, P + 1):
        c = always1[i - 1]
        pref_sum[i] = pref_sum[i - 1] + c
        pref_wsum[i] = pref_wsum[i - 1] + c * i

    sum_always = pref_sum[P]

    C0 = base_down + base_up + (d + u) * sum_always
    K = sum_down + sum_up + 2 * sum_always

    best = C0   # e2 = 0

    for e2 in range(1, P + 1):
        offset = P - e2
        sum_E2 = pref_sum[P] - pref_sum[offset]
        # sum of (value * (local rank))  for the suffix
        sum_weighted = (pref_wsum[P] - pref_wsum[offset]) - offset * sum_E2
        total = C0 + e2 * K - (d + u - 1) * sum_E2 - 2 * sum_weighted
        if total < best:
            best = total

    print(best)


if __name__ == "__main__":
    solve()
