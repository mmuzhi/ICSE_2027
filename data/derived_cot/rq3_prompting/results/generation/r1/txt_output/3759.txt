```python
from typing import List
import heapq

class Solution:
    def findMaxSum(self, nums1: List[int], nums2: List[int], k: int) -> List[int]:
        n = len(nums1)
        # Create list of (nums1[i], nums2[i], original_index)
        pairs = [(nums1[i], nums2[i], i) for i in range(n)]
        # Sort by nums1
        pairs.sort()
        
        ans = [0] * n
        heap = []          # min-heap to keep the top k values
        total_sum = 0      # sum of elements in heap
        
        i = 0
        while i < n:
            # Find the group with same nums1 value
            j = i
            while j < n and pairs[j][0] == pairs[i][0]:
                j += 1
            
            # For each index in this group, the answer is current total_sum
            # (which is sum of top k among strictly smaller nums1 values)
            for idx in range(i, j):
                original_index = pairs[idx][2]
                ans[original_index] = total_sum
            
            # Now insert all elements of this group into the data structure
            for idx in range(i, j):
                val = pairs[idx][1]
                heapq.heappush(heap, val)
                total_sum += val
                if len(heap) > k:
                    smallest = heapq.heappop(heap)
                    total_sum -= smallest
            
            i = j
        
        return ans
```
