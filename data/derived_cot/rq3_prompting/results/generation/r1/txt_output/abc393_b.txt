```python
import sys

def solve() -> None:
    S = sys.stdin.readline().strip()
    n = len(S)
    ans = 0
    for i in range(n):
        for j in range(i + 1, n):
            if S[i] != 'A' or S[j] != 'B':
                continue
            # k must satisfy k - j == j - i  => k = 2*j - i
            k = 2 * j - i
            if k < n and S[k] == 'C':
                ans += 1
    print(ans)

if __name__ == "__main__":
    solve()
```
