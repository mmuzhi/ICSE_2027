```python
from typing import List

class Solution:
    def largestInteger(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # Since nums[i] in [0,50], we can use array of size 51
        count = [0] * 51
        for i in range(n - k + 1):
            sub = nums[i:i+k]
            seen = set(sub)
            for x in seen:
                count[x] += 1
        # Find largest integer with count == 1
        for x in range(50, -1, -1):
            if count[x] == 1:
                return x
        return -1
```

```python
from typing import List

class Solution:
    def largestInteger(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # Count of subarrays of size k that contain each value (values are 0..50)
        cnt = [0] * 51

        # Slide window of size k
        for i in range(n - k + 1):
            sub = nums[i:i+k]
            # Use a set to count the subarray only once per distinct value
            for x in set(sub):
                cnt[x] += 1

        # Find the largest integer that appears in exactly one subarray
        for x in range(50, -1, -1):
            if cnt[x] == 1:
                return x
        return -1
```
