```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])

    LIMIT = 10**9
    total = 1
    term = 1

    for _ in range(1, M + 1):
        term *= N
        total += term
        if total > LIMIT:
            print("inf")
            return

    print(total)

if __name__ == "__main__":
    solve()
```
