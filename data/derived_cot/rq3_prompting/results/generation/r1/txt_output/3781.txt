    # We'll compute a new adjacency list for the candidate set.

    # Now apply kernelization: remove nodes with degree < k-1
    changed = True

                new_candidates.append(i)
            else:
                changed = True

            # We can simply rebuild from scratch with the new set
            m = len(new_candidates)

            for i in range(m):

                    if adj_mat[new_candidates[i]][new_candidates[j]]:

            deg = [sum(adj_mat[i]) for i in range(m)]

    order = sorted(range(m), key=lambda i: deg[i], reverse=True)

    # We'll pass a list of vertices that are still candidates (subset of order) and are all compatible with each other and with current clique.

    def dfs(selected, remaining):
        if len(selected) == k:

        if len(selected) + len(remaining) < k:

class Solution:
    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        import bisect
        n = len(points)
        # preprocess points
        bot_x = []  # list of x for points on bottom (y=0)

        top_x = []   # list of x for points on top (y=side)

        # Also store primary side and value

        # Also store coordinates for distance
        xs = [p[0] for p in points]
        ys = [p[1] for p in points]
        # Build corner list
        corners = []  # list of indices that are corners
        for i, (x,y) in enumerate(points):
            # add to side lists
            if y == 0:
                bot_x.append(x)
            if x == side:
                right_y.append(y)
            if y == side:
                top_x.append(x)
            if x == 0:
                left_y.append(y)

                point_side[i] = 0  # bottom
                point_val[i] = 0

                corners.append(i)
            elif x == side and y == 0:

            hi = bisect.bisect_right(lst, val + D - 1)

        # Now define feasibility function
        def feasible(D):
            # Greedy
            order = sorted(range(n), key=lambda i: xs[i]+ys[i])
            selected = []
            for i in order:
                if all(abs(xs[i]-xs[j])+abs(ys[i]-ys[j]) >= D for j in selected):
                    selected.append(i)
                    if len(selected) >= k:
                        return True
            # If greedy fails, do kernel+clique
            # compute degrees
            n_points = n
            deg = [0]*n
            for i in range(n):
                cnt = count_within_D(i, D)
                deg[i] = n_points - 1 - cnt
            # filter
            candidates = [i for i in range(n) if deg[i] >= k-1]
            if len(candidates) < k:
                return False
            # Build compatibility among candidates
            m = len(candidates)
            # We'll use a dict to map candidate index to original index
            # Compute adjacency matrix
            adj = [[False]*m for _ in range(m)]
            for i in range(m):
                for j in range(i+1,m):
                    if abs(xs[candidates[i]]-xs[candidates[j]]) + abs(ys[candidates[i]]-ys[candidates[j]]) >= D:
                        adj[i][j] = adj[j][i] = True
            # Compute degrees
            deg_in = [sum(row) for row in adj]
            # Kernelization within candidates
            active = [True]*m
            changed = True
            while changed:
                changed = False
                for i in range(m):
                    if active[i] and deg_in[i] < k-1:
                        active[i] = False
                        # update neighbors
                        for j in range(m):
                            if active[j] and adj[i][j]:
                                deg_in[j] -= 1
                        changed = True
            # Collect active
            new_cand = [i for i in range(m) if active[i]]
            if len(new_cand) < k:
                return False
            # Rebuild adjacency for new set
            m2 = len(new_cand)
            # map old index to new index
            idx_map = {old: new for new, old in enumerate(new_cand)}
            adj2 = [[False]*m2 for _ in range(m2)]
            for i in range(m2):
                for j in range(i+1,m2):
                    if adj[new_cand[i]][new_cand[j]]:
                        adj2[i][j] = adj2[j][i] = True
            # Compute degrees in new set
            deg2 = [sum(row) for row in adj2]
            # sort by degree descending
            order = sorted(range(m2), key=lambda i: deg2[i], reverse=True)
            # DFS
            def dfs(selected, remaining):
                if len(selected) == k:
                    return True
                if len(selected) + len(remaining) < k:
                    return False
                for idx, v in enumerate(remaining):
                    # new remaining: neighbors of v that come after
                    new_rem = [w for w in remaining[idx+1:] if adj2[v][w]]
                    if dfs(selected + [v], new_rem):
                        return True
                return False
            return dfs([], order)
        # Binary search
        low, high = 1, 2*side
        ans = 1
        while low <= high:
            mid = (low+high)//2
            if feasible(mid):
                ans = mid
                low = mid+1
            else:
                high = mid-1
        return ans

from typing import List
import bisect

class Solution:
    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        n = len(points)
        xs = [p[0] for p in points]
        ys = [p[1] for p in points]

        # separate points by which side(s) they lie on
        bot_x = []   # x of points on bottom (y = 0)
        right_y = [] # y of points on right (x = side)
        top_x = []   # x of points on top (y = side)
        left_y = []  # y of points on left (x = 0)

        # each point gets a primary side and a value along that side
        point_side = [0] * n   # 0:bottom, 1:right, 2:top, 3:left
        point_val = [0] * n

        corners = []  # indices of corner points

        for i, (x, y) in enumerate(points):
            if y == 0:
                bot_x.append(x)
            if x == side:
                right_y.append(y)
            if y == side:
                top_x.append(x)
            if x == 0:
                left_y.append(y)

            # assign a primary side for counting 'same side'
            if x == 0 and y == 0:
                point_side[i] = 0
                point_val[i] = 0
                corners.append(i)
            elif x == 0 and y == side:
                point_side[i] = 3
                point_val[i] = side
                corners.append(i)
            elif x == side and y == 0:
                point_side[i] = 0
                point_val[i] = side
                corners.append(i)
            elif x == side and y == side:
                point_side[i] = 2
                point_val[i] = side
                corners.append(i)
            elif x == 0:
                point_side[i] = 3
                point_val[i] = y
            elif x == side:
                point_side[i] = 1
                point_val[i] = y
            elif y == 0:
                point_side[i] = 0
                point_val[i] = x
            else: # y == side
                point_side[i] = 2
                point_val[i] = x

        bot_x.sort()
        right_y.sort()
        top_x.sort()
        left_y.sort()

        # helper: number of values in sorted list with |v - val| < D
        def count_in_sorted(lst, val, D):
            lo = bisect.bisect_left(lst, val - D + 1)
            hi = bisect.bisect_right(lst, val + D - 1)
            return hi - lo

        # number of points (excluding i) with Manhattan distance < D from point i
        def count_within_D(i, D):
            side_i = point_side[i]
            val_i = point_val[i]
            total = 0

            # same side
            lst = [bot_x, right_y, top_x, left_y][side_i]
            total += count_in_sorted(lst, val_i, D) - 1  # exclude itself

            # other sides – formulas derived from Manhattan distance
            if side_i == 0:         # bottom
                total += bisect.bisect_left(left_y, D - val_i)          # left
                total += bisect.bisect_left(right_y, D - (side - val_i)) # right
                if D > side:
                    dist = D - side
                    lo = bisect.bisect_left(top_x, val_i - dist + 1)
                    hi = bisect.bisect_right(top_x, val_i + dist - 1)
                    total += hi - lo
            elif side_i == 1:       # right
                total += len(bot_x) - bisect.bisect_right(bot_x, side - D + val_i)
                total += len(top_x) - bisect.bisect_right(top_x, 2*side - val_i - D)
                if D > side:
                    dist = D - side
                    lo = bisect.bisect_left(left_y, val_i - dist + 1)
                    hi = bisect.bisect_right(left_y, val_i + dist - 1)
                    total += hi - lo
            elif side_i == 2:       # top
                total += len(left_y) - bisect.bisect_right(left_y, side - D + val_i)
                total += len(right_y) - bisect.bisect_right(right_y, 2*side - val_i - D)
                if D > side:
                    dist = D - side
                    lo = bisect.bisect_left(bot_x, val_i - dist + 1)
                    hi = bisect.bisect_right(bot_x, val_i + dist - 1)
                    total += hi - lo
            else:                   # left
                total += bisect.bisect_left(bot_x, D - val_i)          # bottom
                total += bisect.bisect_left(top_x, D - (side - val_i)) # top
                if D > side:
                    dist = D - side
                    lo = bisect.bisect_left(right_y, val_i - dist + 1)
                    hi = bisect.bisect_right(right_y, val_i + dist - 1)
                    total += hi - lo

            # correct double‑counting of corners (each corner appears in two side lists)
            for c in corners:
                if abs(xs[i] - xs[c]) + abs(ys[i] - ys[c]) < D:
                    total -= 1
            return total

        # ------------------------------------------------------------------
        # feasibility test for a given D
        # ------------------------------------------------------------------
        def feasible(D):
            # ----- greedy -----
            order = sorted(range(n), key=lambda i: xs[i] + ys[i])
            selected = []
            for i in order:
                ok = True
                for j in selected:
                    if abs(xs[i] - xs[j]) + abs(ys[i] - ys[j]) < D:
                        ok = False
                        break
                if ok:
                    selected.append(i)
                    if len(selected) >= k:
                        return True

            # ----- kernelisation based on degrees in compatibility graph -----
            deg = [0] * n
            for i in range(n):
                cnt = count_within_D(i, D)
                deg[i] = n - 1 - cnt          # number of points with distance >= D

            candidates = [i for i in range(n) if deg[i] >= k - 1]
            if len(candidates) < k:
                return False

            # ----- build compatibility graph among candidates -----
            m = len(candidates)
            # adjacency matrix for the candidate set
            adj = [[False] * m for _ in range(m)]
            for i in range(m):
                for j in range(i + 1, m):
                    a, b = candidates[i], candidates[j]
                    if abs(xs[a] - xs[b]) + abs(ys[a] - ys[b]) >= D:
                        adj[i][j] = adj[j][i] = True

            # compute degrees inside candidate set
            deg_in = [sum(row) for row in adj]

            # iterative removal of vertices with degree < k-1
            active = [True] * m
            changed = True
            while changed:
                changed = False
                for i in range(m):
                    if active[i] and deg_in[i] < k - 1:
                        active[i] = False
                        for j in range(m):
                            if active[j] and adj[i][j]:
                                deg_in[j] -= 1
                        changed = True

            # collect remaining vertices
            new_cand = [i for i in range(m) if active[i]]
            if len(new_cand) < k:
                return False

            # rebuild adjacency for the reduced set
            m2 = len(new_cand)
            idx_map = {old: new for new, old in enumerate(new_cand)}
            adj2 = [[False] * m2 for _ in range(m2)]
            for i in range(m2):
                for j in range(i + 1, m2):
                    if adj[new_cand[i]][new_cand[j]]:
                        adj2[i][j] = adj2[j][i] = True

            # degrees in the reduced set
            deg2 = [sum(row) for row in adj2]
            order = sorted(range(m2), key=lambda i: deg2[i], reverse=True)

            # ----- DFS for a k‑clique -----
            def dfs(selected, remaining):
                if len(selected) == k:
                    return True
                if len(selected) + len(remaining) < k:
                    return False
                for idx, v in enumerate(remaining):
                    new_rem = [w for w in remaining[idx+1:] if adj2[v][w]]
                    if dfs(selected + [v], new_rem):
                        return True
                return False

            return dfs([], order)

        # ------------------------------------------------------------------
        # binary search on the answer
        # ------------------------------------------------------------------
        low, high = 1, 2 * side
        ans = 1
        while low <= high:
            mid = (low + high) // 2
            if feasible(mid):
                ans = mid
                low = mid + 1
            else:
                high = mid - 1
        return ans
