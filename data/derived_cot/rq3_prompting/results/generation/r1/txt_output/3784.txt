class Solution:
    def longestCommonPrefix(self, words: List[str], k: int) -> List[int]:
        n = len(words)
        if n - 1 < k:
            return [0] * n
        
        # Build Trie
        class TrieNode:
            __slots__ = ('children', 'count', 'depth')
            def __init__(self):
                self.children = {}
                self.count = 0
                self.depth = 0
        root = TrieNode()
        max_len = 0
        for w in words:
            node = root
            for i, ch in enumerate(w):
                if ch not in node.children:
                    new_node = TrieNode()
                    new_node.depth = i + 1
                    node.children[ch] = new_node
                node = node.children[ch]
                node.count += 1
            max_len = max(max_len, len(w))
        
        # Compute max_count, second_max, max_node per depth
        max_count = [0] * (max_len + 1)
        second_max = [0] * (max_len + 1)
        max_node = [None] * (max_len + 1)
        
        stack = [root]
        while stack:
            node = stack.pop()
            d = node.depth
            c = node.count
            if d > 0:
                if c > max_count[d]:
                    second_max[d] = max_count[d]
                    max_count[d] = c
                    max_node[d] = node
                elif c > second_max[d]:
                    second_max[d] = c
            for child in node.children.values():
                stack.append(child)
        
        # Find global maximum length L such that max_count[L] >= k
        global_max_len = 0
        for d in range(1, max_len + 1):
            if max_count[d] >= k:
                global_max_len = d
        if global_max_len == 0:
            return [0] * n
        
        # Compute last_good array
        last_good = [0] * (max_len + 1)
        for d in range(1, max_len + 1):
            if max_count[d] >= k:
                last_good[d] = d
            else:
                last_good[d] = last_good[d-1]
        
        ans = [0] * n
        for idx, w in enumerate(words):
            len_i = len(w)
            # Determine critical prefixes for this word
            is_critical = [False] * (len_i + 1)
            node = root
            for depth, ch in enumerate(w, start=1):
                node = node.children[ch]
                if node is max_node[depth] and max_count[depth] == k and second_max[depth] < k:
                    is_critical[depth] = True
            # Compute answer
            if global_max_len > len_i:
                ans[idx] = global_max_len
            else:
                if not is_critical[global_max_len]:
                    ans[idx] = global_max_len

                            ans[idx] = 0
                            break
                        if cur <= len_i and is_critical[cur]:
                            continue
                        else:
                            ans[idx] = cur
                            break
        return ans

from typing import List

class Solution:
    def longestCommonPrefix(self, words: List[str], k: int) -> List[int]:
        n = len(words)
        if n - 1 < k:
            return [0] * n

        # ---------- Trie construction ----------
        class TrieNode:
            __slots__ = ('children', 'count', 'depth')
            def __init__(self):
                self.children = {}
                self.count = 0
                self.depth = 0

        root = TrieNode()
        max_len = 0
        for w in words:
            node = root
            for i, ch in enumerate(w):
                if ch not in node.children:
                    new_node = TrieNode()
                    new_node.depth = i + 1
                    node.children[ch] = new_node
                node = node.children[ch]
                node.count += 1
            max_len = max(max_len, len(w))

        # ---------- Collect per‑depth statistics ----------
        max_count = [0] * (max_len + 1)      # maximum frequency for length d
        second_max = [0] * (max_len + 1)     # second maximum frequency
        max_node = [None] * (max_len + 1)    # node that achieves max_count

        stack = [root]
        while stack:
            node = stack.pop()
            d = node.depth
            c = node.count
            if d > 0:                       # ignore root
                if c > max_count[d]:
                    second_max[d] = max_count[d]
                    max_count[d] = c
                    max_node[d] = node
                elif c > second_max[d]:
                    second_max[d] = c
            for child in node.children.values():
                stack.append(child)

        # ---------- Global best length (ignoring removals) ----------
        global_max_len = 0
        for d in range(1, max_len + 1):
            if max_count[d] >= k:
                global_max_len = d
        if global_max_len == 0:
            return [0] * n

        # ---------- last_good : previous good length ----------
        last_good = [0] * (max_len + 1)
        for d in range(1, max_len + 1):
            if max_count[d] >= k:
                last_good[d] = d
            else:
                last_good[d] = last_good[d-1]

        # ---------- Answer per word ----------
        ans = [0] * n
        for idx, w in enumerate(words):
            L = len(w)

            # mark critical depths for this word
            is_critical = [False] * (L + 1)
            node = root
            for depth, ch in enumerate(w, start=1):
                node = node.children[ch]
                if (node is max_node[depth] and
                    max_count[depth] == k and
                    second_max[depth] < k):
                    is_critical[depth] = True

            if global_max_len > L:
                ans[idx] = global_max_len
            elif not is_critical[global_max_len]:
                ans[idx] = global_max_len
            else:
                cur = global_max_len
                while True:
                    cur = last_good[cur-1]
                    if cur == 0:
                        ans[idx] = 0
                        break
                    if cur <= L and is_critical[cur]:
                        continue
                    ans[idx] = cur
                    break

        return ans
