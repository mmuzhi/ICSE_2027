import sys

def main() -> None:
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    K = next(it)

    adj = [[] for _ in range(N + 1)]
    for _ in range(N - 1):
        a = next(it)
        b = next(it)
        adj[a].append(b)
        adj[b].append(a)

    marked = [next(it) for _ in range(K)]

    if K == 1:
        print(1)
        return

    # ---- iterative DFS to compute parent, depth, tin (preorder) ----
    parent = [0] * (N + 1)
    depth = [0] * (N + 1)
    tin = [0] * (N + 1)
    timer = 0
    # stack entries: (node, parent, state)  state 0 = enter
    stack = [(1, 0, 0)]
    while stack:
        v, p, state = stack.pop()
        if state == 0:
            parent[v] = p
            depth[v] = depth[p] + 1 if p != 0 else 0
            tin[v] = timer
            timer += 1
            # push children in reverse order so that first child is processed first
            for c in reversed(adj[v]):
                if c != p:
                    stack.append((c, v, 0))

    # ---- binary lifting table ----
    LOG = (N).bit_length()
    up = [[0] * LOG for _ in range(N + 1)]
    for v in range(1, N + 1):
        up[v][0] = parent[v]
    for j in range(1, LOG):
        for v in range(1, N + 1):
            if up[v][j - 1] != 0:
                up[v][j] = up[up[v][j - 1]][j - 1]

    # ---- LCA function ----
    def lca(u: int, v: int) -> int:
        if depth[u] < depth[v]:
            u, v = v, u
        # lift u to depth of v
        diff = depth[u] - depth[v]
        for i in range(LOG):
            if diff & (1 << i):
                u = up[u][i]
        if u == v:
            return u
        for i in range(LOG - 1, -1, -1):
            if up[u][i] != up[v][i]:
                u = up[u][i]
                v = up[v][i]
        return up[u][0]

    # ---- compute Steiner tree size ----
    marked.sort(key=lambda x: tin[x])
    total_dist = 0
    for i in range(K - 1):
        u, v = marked[i], marked[i + 1]
        total_dist += depth[u] + depth[v] - 2 * depth[lca(u, v)]
    # close the cycle
    u, v = marked[-1], marked[0]
    total_dist += depth[u] + depth[v] - 2 * depth[lca(u, v)]

    edges = total_dist // 2
    ans = edges + 1
    print(ans)

if __name__ == "__main__":
    main()
