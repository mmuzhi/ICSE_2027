```python
import sys

def main() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))

    # 1-indexed arrays
    P = [0] * (N + 1)
    Q = [0] * (N + 1)

    for i in range(1, N + 1):
        P[i] = int(next(it))
    for i in range(1, N + 1):
        Q[i] = int(next(it))

    # Inverse mapping: person wearing bib number b is invQ[b]
    invQ = [0] * (N + 1)
    for person in range(1, N + 1):
        bib = Q[person]
        invQ[bib] = person

    ans = []
    for bib in range(1, N + 1):
        person = invQ[bib]          # person who wears bib 'bib'
        target = P[person]          # person that 'person' stares at
        target_bib = Q[target]      # bib number of that target
        ans.append(str(target_bib))

    print(' '.join(ans))

if __name__ == "__main__":
    main()
```
