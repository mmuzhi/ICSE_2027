```python
import sys

def main() -> None:
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)

    N = next(it)

    # prefix sums: indices 0..N, initialized to 0
    pref = [[[0] * (N + 1) for _ in range(N + 1)] for __ in range(N + 1)]

    # read A and build 3D prefix sum
    for x in range(1, N + 1):
        for y in range(1, N + 1):
            for z in range(1, N + 1):
                val = next(it)
                # standard inclusion–exclusion for 3D prefix
                pref[x][y][z] = (val
                                 + pref[x-1][y][z]
                                 + pref[x][y-1][z]
                                 + pref[x][y][z-1]
                                 - pref[x-1][y-1][z]
                                 - pref[x-1][y][z-1]
                                 - pref[x][y-1][z-1]
                                 + pref[x-1][y-1][z-1])

    Q = next(it)
    out = []
    for _ in range(Q):
        Lx = next(it); Rx = next(it)
        Ly = next(it); Ry = next(it)
        Lz = next(it); Rz = next(it)

        ans = (pref[Rx][Ry][Rz]
               - pref[Lx-1][Ry][Rz]
               - pref[Rx][Ly-1][Rz]
               - pref[Rx][Ry][Lz-1]
               + pref[Lx-1][Ly-1][Rz]
               + pref[Lx-1][Ry][Lz-1]
               + pref[Rx][Ly-1][Lz-1]
               - pref[Lx-1][Ly-1][Lz-1])
        out.append(str(ans))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
```
