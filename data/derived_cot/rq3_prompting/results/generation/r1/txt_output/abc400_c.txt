def solve():
    import sys, math
    N = int(sys.stdin.readline().strip())
    ans = 0
    e = 1
    while (1 << e) <= N:
        M = math.isqrt(N // (1 << e))
        ans += (M + 1) // 2
        e += 1
    print(ans)

```python
import sys
import math

def solve() -> None:
    data = sys.stdin.readline().strip()
    if not data:
        return
    N = int(data)
    ans = 0
    e = 1
    while (1 << e) <= N:
        # max odd s such that 2^e * s^2 <= N
        M = math.isqrt(N // (1 << e))   # floor(sqrt(N / 2^e))
        ans += (M + 1) // 2             # count of odd numbers 1..M
        e += 1
    print(ans)

if __name__ == "__main__":
    solve()
```
