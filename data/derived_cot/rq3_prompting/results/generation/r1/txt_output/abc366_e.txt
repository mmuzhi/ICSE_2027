import sys, bisect

def compute_freq(X, D):
    N = len(X)
    X.sort()
    P = [0]
    for v in X:
        P.append(P[-1] + v)
    minX = X[0] if N>0 else 0
    maxX = X[-1] if N>0 else 0
    L = maxX - D
    R = minX + D
    if L > R:
        return [0]*(D+1)  # all zeros
    freq = [0]*(D+1)
    # initial x = L
    cnt = bisect.bisect_right(X, L)
    sum_left = P[cnt]
    sum_right = P[N] - P[cnt]
    f = cnt*L - sum_left + sum_right - (N-cnt)*L
    if f <= D:
        freq[f] += 1
    ptr = cnt  # index of first element not yet counted (next to consider for points > L)
    # iterate x from L+1 to R
    for x in range(L+1, R+1):
        # compute f at x using previous f and cnt (which is number <= x-1)
        f = f + 2*cnt - N
        if f <= D:
            freq[f] += 1
        # update cnt to include points equal to x
        while ptr < N and X[ptr] == x:
            cnt += 1
            ptr += 1
    return freq

def solve():
    import sys
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    D = int(next(it))
    X = []
    Y = []
    for _ in range(N):
        x = int(next(it)); y = int(next(it))
        X.append(x); Y.append(y)
    freq_x = compute_freq(X, D)
    freq_y = compute_freq(Y, D)
    # compute prefix sum of freq_y
    pref = [0]*(D+2)
    for i in range(D+1):
        pref[i+1] = pref[i] + freq_y[i]
    # pref[i] = sum_{j=0}^{i-1} freq_y[j], so pref[D+1] is total sum.
    # For s, need sum_{t=0}^{D-s} freq_y[t] = pref[D-s+1] (since pref index is exclusive).
    ans = 0
    for s in range(D+1):
        if freq_x[s]==0:
            continue
        # t max = D-s
        max_t = D - s
        # sum of freq_y[0..max_t]
        cnt_y = pref[max_t+1]  # because pref[max_t+1] = sum_{0}^{max_t}
        ans += freq_x[s] * cnt_y
    print(ans)

if __name__ == "__main__":
    solve()

```python
import sys
import bisect

def compute_freq(arr, D):
    """Return frequency array f where f[v] = number of x (or y) with
       sum_i |x - arr_i| == v, for v in 0..D, considering only x that
       could possibly satisfy the full condition (x in [max(arr)-D, min(arr)+D])."""
    N = len(arr)
    arr.sort()
    # prefix sums for initial f computation
    pref = [0]
    for v in arr:
        pref.append(pref[-1] + v)

    min_val = arr[0]
    max_val = arr[-1]
    L = max_val - D
    R = min_val + D
    if L > R:
        return [0] * (D + 1)

    freq = [0] * (D + 1)

    # initialise for x = L
    cnt = bisect.bisect_right(arr, L)          # number of points <= L
    left_sum = pref[cnt]
    right_sum = pref[N] - pref[cnt]
    f = cnt * L - left_sum + right_sum - (N - cnt) * L
    if f <= D:
        freq[f] += 1

    ptr = cnt                                   # index of first element not yet counted
    # sweep x from L+1 to R
    for x in range(L + 1, R + 1):
        # f(x) = f(x-1) + 2 * cnt(x-1) - N
        f = f + 2 * cnt - N
        if f <= D:
            freq[f] += 1
        # add points equal to x (they become part of cnt for the next step)
        while ptr < N and arr[ptr] == x:
            cnt += 1
            ptr += 1

    return freq


def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    D = int(next(it))
    X = []
    Y = []
    for _ in range(N):
        x = int(next(it))
        y = int(next(it))
        X.append(x)
        Y.append(y)

    freq_x = compute_freq(X, D)
    freq_y = compute_freq(Y, D)

    # prefix sums of freq_y
    pref = [0] * (D + 2)      # pref[t] = sum_{i=0}^{t-1} freq_y[i]
    for i in range(D + 1):
        pref[i + 1] = pref[i] + freq_y[i]

    ans = 0
    for s in range(D + 1):
        if freq_x[s] == 0:
            continue
        max_t = D - s
        ans += freq_x[s] * pref[max_t + 1]

    print(ans)


if __name__ == "__main__":
    solve()
```
