=== ANSWER ===
```python
import sys

def main():
    n_str = sys.stdin.readline().strip()
    # count each digit
    from collections import Counter
    cnt = Counter(n_str)
    # check conditions
    if cnt['1'] == 1 and cnt['2'] == 2 and cnt['3'] == 3:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    main()
```
