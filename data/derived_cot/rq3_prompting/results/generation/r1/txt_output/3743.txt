There is no time during the event not occupied by meetings.

from typing import List

class Solution:
    def maxFreeTime(self, eventTime: int, k: int, startTime: List[int], endTime: List[int]) -> int:
        n = len(startTime)
        # gaps array of length n+1
        gaps = [0] * (n + 1)
        gaps[0] = startTime[0]
        for i in range(1, n):
            gaps[i] = startTime[i] - endTime[i-1]
        gaps[n] = eventTime - endTime[-1]
        
        # prefix sums of gaps
        prefix = [0] * (n + 2)
        for i in range(n + 1):
            prefix[i+1] = prefix[i] + gaps[i]
        
        ans = 0
        for j in range(n + 1):
            i = max(0, j - k)          # largest window ending at j with length <= k+1
            cur = prefix[j+1] - prefix[i]
            if cur > ans:
                ans = cur
        return ans
