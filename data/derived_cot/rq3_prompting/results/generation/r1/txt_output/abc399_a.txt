=== ANSWER ===
```python
import sys

def main() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    s = data[1]
    t = data[2]
    # compute Hamming distance
    hamming = 0
    for i in range(n):
        if s[i] != t[i]:
            hamming += 1
    print(hamming)

if __name__ == "__main__":
    main()
```
