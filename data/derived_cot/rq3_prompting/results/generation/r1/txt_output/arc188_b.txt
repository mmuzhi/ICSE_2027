import sys

def solve() -> None:

    it = iter(data)
    t = int(next(it))
    out = []
    for _ in range(t):
        n = int(next(it))
        k = int(next(it))
        if n % 2 == 1:
            # odd N
            out.append("Yes" if math.gcd(n, k) == 1 else "No")
        else:
            m = n // 2
            if m % 2 == 0:
                # N divisible by 4
                out.append("No")
            else:
                # N = 2 * odd
                out.append("Yes" if math.gcd(m, k) == 1 else "No")
    sys.stdout.write("\n".join(out))
