class Solution:
    def minOperations(self, nums: List[int]) -> int:
        MAX = max(nums) if nums else 0
        if MAX <= 1: return 0  # all 1's
        # compute spf up to MAX
        spf = list(range(MAX+1))
        for i in range(2, int(MAX**0.5)+1):
            if spf[i] == i:  # prime
                for j in range(i*i, MAX+1, i):
                    if spf[j] == j:
                        spf[j] = i
        # function to get chain for x
        def get_chain(x):
            chain = [x]
            while x > 1 and spf[x] != x:  # composite
                x //= spf[x]
                chain.append(x)
            return chain
        n = len(nums)
        # Process from right to left
        # We'll keep target = final value of next element
        target = nums[-1]
        total_ops = 0
        # for last element, no operation needed (original)
        for i in range(n-2, -1, -1):
            chain = get_chain(nums[i])
            # find first value <= target
            chosen = None
            for idx, val in enumerate(chain):
                if val <= target:
                    chosen = val
                    ops = idx
                    break
            if chosen is None:
                return -1
            total_ops += ops
            target = chosen  # update target for next left
        return total_ops

```python
    def minOperations(self, nums: List[int]) -> int:
        if not nums:
            return 0
        max_val = max(nums)
        # small optimization: if max_val == 1, all are 1, return 0
        if max_val == 1:
            return 0

        spf = list(range(max_val + 1))
        for i in range(2, int(max_val**0.5) + 1):

            chain = get_chain(nums[i])
```

```python
from typing import List

class Solution:
    def minOperations(self, nums: List[int]) -> int:
        # trivial case: all numbers are 1
        if max(nums) == 1:
            return 0

        max_val = max(nums)

        # sieve to compute smallest prime factor (spf) for every number up to max_val
        spf = list(range(max_val + 1))
        for i in range(2, int(max_val ** 0.5) + 1):
            if spf[i] == i:              # i is prime
                step = i
                start = i * i
                for j in range(start, max_val + 1, step):
                    if spf[j] == j:
                        spf[j] = i

        def chain(x: int) -> List[int]:
            """return the decreasing sequence of numbers reachable from x
               by repeatedly dividing by the greatest proper divisor.
               Example: 12 -> [12, 6, 3]"""
            seq = [x]
            while x > 1 and spf[x] != x:   # while x is composite
                x //= spf[x]
                seq.append(x)
            return seq

        n = len(nums)
        # start from the rightmost element – it stays unchanged
        target = nums[-1]
        total_ops = 0

        # process from the second last element to the first
        for i in range(n - 2, -1, -1):
            seq = chain(nums[i])
            chosen = None
            steps = 0
            for idx, val in enumerate(seq):
                if val <= target:
                    chosen = val
                    steps = idx
                    break
            if chosen is None:
                return -1
            total_ops += steps
            target = chosen

        return total_ops
```
