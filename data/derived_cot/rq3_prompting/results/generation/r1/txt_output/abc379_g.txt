import sys

def build_row_states(row_str, pow3):
    """return (encodings, colors_list, dict_enc_to_idx) for a single row."""
    W = len(row_str)
    fixed = [int(ch)-1 if ch != '?' else -1 for ch in row_str]
    states_enc = []
    states_colors = []
    cur_colors = []

    def dfs(col, last, enc):
        if col == W:
            states_enc.append(enc)
            states_colors.append(cur_colors.copy())
            return
        f = fixed[col]
        mask = 0b111 if f == -1 else (1 << f)
        for c in range(3):
            if (mask >> c) & 1:
                if last is None or c != last:
                    cur_colors.append(c)
                    dfs(col + 1, c, enc + c * pow3[col])
                    cur_colors.pop()

    dfs(0, None, 0)
    state_dict = {enc: idx for idx, enc in enumerate(states_enc)}
    return states_enc, states_colors, state_dict

    dp = [1] * len(states_enc_prev)

    for i in range(1, H):

        states_enc_cur, states_colors_cur, dict_cur = build_row_states(row_str, pow3)
        if not states_enc_cur:

        fixed_mask_cur = []
