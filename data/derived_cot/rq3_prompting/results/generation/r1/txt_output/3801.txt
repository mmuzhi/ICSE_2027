class Solution:
    def beautifulNumbers(self, l: int, r: int) -> int:
        def count_up_to(x):
            if x <= 0:
                return 0
            s = str(x)
            n = len(s)
            total = 0
            max_sum = 9 * n
            for S in range(1, max_sum + 1):
                from functools import lru_cache
                @lru_cache(None)
                def dfs(pos, tight, started, cur_sum, prod_mod):
                    # prune if sum exceeds S
                    if cur_sum > S:
                        return 0
                    if pos == n:
                        # number must be positive (started) and sum exactly S and product divisible by S
                        return 1 if (started and cur_sum == S and prod_mod == 0) else 0
                    limit = int(s[pos]) if tight else 9
                    ans = 0
                    for d in range(limit + 1):
                        new_tight = tight and (d == limit)
                        new_started = started or (d != 0)
                        if new_started:
                            new_sum = cur_sum + d
                            new_prod = (prod_mod * d) % S
                        else:
                            # still not started, digits are leading zeros, sum and prod unchanged
                            new_sum = cur_sum  # 0
                            new_prod = prod_mod  # 1
                        ans += dfs(pos+1, new_tight, new_started, new_sum, new_prod)
                    return ans
                total += dfs(0, True, False, 0, 1)
            return total
        return count_up_to(r) - count_up_to(l-1)

class Solution:
    def beautifulNumbers(self, l: int, r: int) -> int:
        # Helper to count beautiful numbers in [1, x]
        def count_up_to(x: int) -> int:
            if x <= 0:
                return 0
            digits = list(map(int, str(x)))
            n = len(digits)
            max_sum = 9 * n   # maximum possible digit sum

            from functools import lru_cache

            @lru_cache(maxsize=None)
            def dfs(pos: int, tight: bool, started: bool, cur_sum: int, prod: int) -> int:
                # Too large sum – no chance to reach a valid state
                if cur_sum > max_sum:
                    return 0
                # All digits processed
                if pos == n:
                    # Exclude the number 0
                    if not started:
                        return 0
                    # The number is beautiful iff product is divisible by sum
                    return 1 if prod % cur_sum == 0 else 0

                limit = digits[pos] if tight else 9
                total = 0
                for d in range(limit + 1):
                    new_tight = tight and (d == limit)
                    new_started = started or (d != 0)
                    if new_started:
                        new_sum = cur_sum + d
                        new_prod = prod * d
                    else:
                        # still in leading zeros – sum and product unchanged
                        new_sum = cur_sum
                        new_prod = prod
                    # Prune if sum already exceeds the maximum possible
                    if new_sum > max_sum:
                        continue
                    total += dfs(pos + 1, new_tight, new_started, new_sum, new_prod)
                return total

            return dfs(0, True, False, 0, 1)

        return count_up_to(r) - count_up_to(l - 1)
