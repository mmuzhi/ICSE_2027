class SegTree:
    def __init__(self, arr):
        n = len(arr)
        self.n = n
        size = 4*n
        self.sum = [0]*size
        self.mn = [0]*size
        self.mx = [0]*size
        self.lazy = [None]*size
        self.build(1,0,n-1,arr)
    def build(self,idx,l,r,arr):
        if l==r:
            v = arr[l]
            self.sum[idx]=v
            self.mn[idx]=v
            self.mx[idx]=v
            return
        mid=(l+r)//2
        self.build(idx*2,l,mid,arr)
        self.build(idx*2+1,mid+1,r,arr)
        self.pull(idx)
    def pull(self,idx):
        self.sum[idx]=self.sum[idx*2]+self.sum[idx*2+1]
        self.mn[idx]=min(self.mn[idx*2], self.mn[idx*2+1])
        self.mx[idx]=max(self.mx[idx*2], self.mx[idx*2+1])
    def apply(self,idx,l,r,val):
        self.sum[idx]=val*(r-l+1)
        self.mn[idx]=val
        self.mx[idx]=val
        self.lazy[idx]=val
    def push(self,idx,l,r):
        if self.lazy[idx] is not None:
            mid=(l+r)//2
            self.apply(idx*2,l,mid,self.lazy[idx])
            self.apply(idx*2+1,mid+1,r,self.lazy[idx])
            self.lazy[idx]=None
    def range_assign(self,idx,l,r,ql,qr,val):
        if ql>r or qr<l: return
        if ql<=l and r<=qr:
            self.apply(idx,l,r,val)
            return
        self.push(idx,l,r)
        mid=(l+r)//2
        self.range_assign(idx*2,l,mid,ql,qr,val)
        self.range_assign(idx*2+1,mid+1,r,ql,qr,val)
        self.pull(idx)
    def query_sum(self,idx,l,r,ql,qr):
        if ql>r or qr<l: return 0
        if ql<=l and r<=qr: return self.sum[idx]
        self.push(idx,l,r)
        mid=(l+r)//2
        return self.query_sum(idx*2,l,mid,ql,qr)+self.query_sum(idx*2+1,mid+1,r,ql,qr)
    def first_gt(self,idx,l,r,ql,qr,threshold):
        # return the smallest index in [ql,qr] with value > threshold, or qr+1 if none
        if ql>r or qr<l: return qr+1
        if self.mx[idx] <= threshold: return qr+1
        if l==r: return l
        self.push(idx,l,r)
        mid=(l+r)//2
        # search left first
        res = self.first_gt(idx*2,l,mid,ql,qr,threshold)
        if res <= min(qr,mid): return res
        return self.first_gt(idx*2+1,mid+1,r,ql,qr,threshold)
    def last_lt(self,idx,l,r,ql,qr,threshold):
        # return the largest index in [ql,qr] with value < threshold, or ql-1 if none
        if ql>r or qr<l: return ql-1
        if self.mn[idx] >= threshold: return ql-1
        if l==r: return l
        self.push(idx,l,r)
        mid=(l+r)//2
        # search right first
        res = self.last_lt(idx*2+1,mid+1,r,ql,qr,threshold)
        if res >= max(ql,mid+1): return res
        return self.last_lt(idx*2,l,mid,ql,qr,threshold)

        self._pull(idx)

    def _pull(self, idx):
        self.sum[idx] = self.sum[idx * 2] + self.sum[idx * 2 + 1]
        self.mn[idx] = min(self.mn[idx * 2], self.mn[idx * 2 + 1])
        self.mx[idx] = max(self.mx[idx * 2], self.mx[idx * 2 + 1])

    def _apply(self, idx, l, r, val):
        self.sum[idx] = val * (r - l + 1)
        self.mn[idx] = val
        self.mx[idx] = val
        self.lazy[idx] = val

    def _push(self, idx, l, r):
        if self.lazy[idx] is not None:
            mid = (l + r) // 2
            self._apply(idx * 2, l, mid, self.lazy[idx])
            self._apply(idx * 2 + 1, mid + 1, r, self.lazy[idx])
            self.lazy[idx] = None

    def range_assign(self, ql, qr, val):
        if ql > qr:
            return
        self._range_assign(1, 0, self.n - 1, ql, qr, val)

    def _range_assign(self, idx, l, r, ql, qr, val):
        if ql > r or qr < l:
            return
        if ql <= l and r <= qr:
            self._apply(idx, l, r, val)
            return
        self._push(idx, l, r)
        mid = (l + r) // 2
        self._range_assign(idx * 2, l, mid, ql, qr, val)
        self._range_assign(idx * 2 + 1, mid + 1, r, ql, qr, val)
        self._pull(idx)

    def query_sum(self, ql, qr):
        if ql > qr:
            return 0
        return self._query_sum(1, 0, self.n - 1, ql, qr)

    def _query_sum(self, idx, l, r, ql, qr):
        if ql > r or qr < l:
            return 0
        if ql <= l and r <= qr:
            return self.sum[idx]
        self._push(idx, l, r)
        mid = (l + r) // 2
        return self._query_sum(idx * 2, l, mid, ql, qr) + self._query_sum(idx * 2 + 1, mid + 1, r, ql, qr)

    def first_gt(self, ql, qr, threshold):
        """return smallest index in [ql,qr] with value > threshold, or qr+1 if none"""
        if ql > qr:
            return qr + 1
        return self._first_gt(1, 0, self.n - 1, ql, qr, threshold)

    def _first_gt(self, idx, l, r, ql, qr, th):
        if ql > r or qr < l:
            return qr + 1
        if self.mx[idx] <= th:
            return qr + 1
        if l == r:
            return l
        self._push(idx, l, r)
        mid = (l + r) // 2
        res = self._first_gt(idx * 2, l, mid, ql, qr, th)
        if res <= min(qr, mid):
            return res
        return self._first_gt(idx * 2 + 1, mid + 1, r, ql, qr, th)

    def last_lt(self, ql, qr, threshold):
        """return largest index in [ql,qr] with value < threshold, or ql-1 if none"""
        if ql > qr:
            return ql - 1
        return self._last_lt(1, 0, self.n - 1, ql, qr, threshold)

    def _last_lt(self, idx, l, r, ql, qr, th):
        if ql > r or qr < l:
            return ql - 1
        if self.mn[idx] >= th:
            return ql - 1
        if l == r:
            return l
        self._push(idx, l, r)
        mid = (l + r) // 2
        res = self._last_lt(idx * 2 + 1, mid + 1, r, ql, qr, th)
        if res >= max(ql, mid + 1):
            return res
        return self._last_lt(idx * 2, l, mid, ql, qr, th)


def solve() -> None:
    it = iter(sys.stdin.read().split())
    N = int(next(it))
    X = [int(next(it)) for _ in range(N)]
    Q = int(next(it))
    tasks = [(int(next(it)) - 1, int(next(it))) for _ in range(Q)]   # 0‑based index

    # B[i] = X[i] - (i+1)
    B = [X[i] - (i + 1) for i in range(N)]
    seg = SegTree(B)
    total = 0

    for i, g in tasks:
        C = g - (i + 1)          # because 1‑based index = i+1
        cur = seg.query_sum(i, i)
        cost = abs(cur - C)

        # left part (j < i)
        if i > 0:
            p = seg.first_gt(0, i - 1, C)
            if p <= i - 1:
                s = seg.query_sum(p, i - 1)
                cnt = i - p
                cost += s - C * cnt
        else:
            p = i   # not used later

        # right part (j > i)
        if i < N - 1:
            q = seg.last_lt(i + 1, N - 1, C)
            if q >= i + 1:
                s = seg.query_sum(i + 1, q)
                cnt = q - i
                cost += C * cnt - s
        else:
            q = i

        total += cost

        # apply the updates
        if i > 0 and p <= i - 1:
            seg.range_assign(p, i - 1, C)
        seg.range_assign(i, i, C)
        if i < N - 1 and q >= i + 1:
            seg.range_assign(i + 1, q, C)

    print(total)


if __name__ == "__main__":
    solve()
