class Solution:
    def maxProduct(self, nums: List[int], k: int, limit: int) -> int:
        total_sum = sum(nums)
        if k < -total_sum or k > total_sum:
            return -1
        offset = total_sum
        max_index = 2 * offset
        mask = (1 << (max_index + 1)) - 1
        
        dp0 = [0] * (limit + 1)   # parity 0
        dp1 = [0] * (limit + 1)   # parity 1
        
        for val in nums:
            # copy old dp for new dp
            new_dp0 = dp0[:]
            new_dp1 = dp1[:]
            
            # process existing states: take element
            for prod in range(limit + 1):
                # parity 0
                bits0 = dp0[prod]
                if bits0:
                    new_prod = prod * val
                    if new_prod <= limit:
                        new_par = 1
                        new_bits = (bits0 << val) & mask
                        if new_par == 0:
                            new_dp0[new_prod] |= new_bits
                        else:
                            new_dp1[new_prod] |= new_bits
                # parity 1
                bits1 = dp1[prod]
                if bits1:
                    new_prod = prod * val
                    if new_prod <= limit:
                        new_par = 0
                        new_bits = (bits1 >> val) & mask
                        if new_par == 0:
                            new_dp0[new_prod] |= new_bits
                        else:
                            new_dp1[new_prod] |= new_bits
            # start new subsequence with this element
            if val <= limit:
                # parity: length 1 -> next sign negative -> parity 1
                sum_idx = offset + val
                if 0 <= sum_idx <= max_index:
                    new_dp1[val] |= (1 << sum_idx)
            dp0, dp1 = new_dp0, new_dp1
        
        # find answer
        target_idx = offset + k
        if not (0 <= target_idx <= max_index):
            return -1
        for prod in range(limit, -1, -1):
            if (dp0[prod] >> target_idx) & 1 or (dp1[prod] >> target_idx) & 1:
                return prod
        return -1

```python
from typing import List

class Solution:
    def maxProduct(self, nums: List[int], k: int, limit: int) -> int:
        total = sum(nums)
        if k < -total or k > total:
            return -1
        offset = total
        max_idx = 2 * offset
        mask = (1 << (max_idx + 1)) - 1
        
        # dp[parity][product] = bitset of sums
        dp0 = [0] * (limit + 1)  # parity 0
        dp1 = [0] * (limit + 1)  # parity 1
        
        for val in nums:
            new_dp0 = dp0[:]
            new_dp1 = dp1[:]
            
            # take element into existing subsequences
            for prod in range(limit + 1):
                # parity 0
                bits0 = dp0[prod]
                if bits0:
                    new_prod = prod * val
                    if new_prod <= limit:
                        # adding to subsequence with current parity 0 means next element is positive
                        # so sum increases by val
                        new_bits = (bits0 << val) & mask
                        new_dp1[new_prod] |= new_bits  # new parity becomes 1
                
                # parity 1
                bits1 = dp1[prod]
                if bits1:
                    new_prod = prod * val
                    if new_prod <= limit:
                        # adding to subsequence with current parity 1 means next element is negative
                        # so sum decreases by val
                        new_bits = (bits1 >> val) & mask
                        new_dp0[new_prod] |= new_bits  # new parity becomes 0
            
            # start a new subsequence with current element
            if val <= limit:
                idx = offset + val
                if idx <= max_idx:
                    # first element, parity becomes 1 (since after this, next is odd)
                    new_dp1[val] |= (1 << idx)
            
            dp0, dp1 = new_dp0, new_dp1
        
        target = offset + k
        if not (0 <= target <= max_idx):
            return -1
        for prod in range(limit, -1, -1):
            if (dp0[prod] >> target) & 1 or (dp1[prod] >> target) & 1:
                return prod
        return -1
```

```python
from typing import List

class Solution:
    def maxProduct(self, nums: List[int], k: int, limit: int) -> int:
        total = sum(nums)
        # The alternating sum can never exceed the total sum in absolute value.
        if k < -total or k > total:
            return -1

        offset = total
        max_idx = 2 * offset  # largest index in the sum bitset
        mask = (1 << (max_idx + 1)) - 1  # keep bits in [0, max_idx]

        # dp0[product] : bitset of sums for subsequences with even length (next element is positive)
        # dp1[product] : bitset of sums for subsequences with odd length (next element is negative)
        dp0 = [0] * (limit + 1)
        dp1 = [0] * (limit + 1)

        for val in nums:
            # start with copies (skip this element)
            new_dp0 = dp0[:]
            new_dp1 = dp1[:]

            # extend existing subsequences by taking the current element
            for prod in range(limit + 1):
                # parity 0 (even length -> next element is added positively)
                bits = dp0[prod]
                if bits:
                    new_prod = prod * val
                    if new_prod <= limit:
                        # sum increases by val
                        new_bits = (bits << val) & mask
                        new_dp1[new_prod] |= new_bits   # length becomes odd

                # parity 1 (odd length -> next element is added negatively)
                bits = dp1[prod]
                if bits:
                    new_prod = prod * val
                    if new_prod <= limit:
                        # sum decreases by val
                        new_bits = (bits >> val) & mask
                        new_dp0[new_prod] |= new_bits   # length becomes even

            # start a new subsequence with the current element (first element -> odd length)
            if val <= limit:
                idx = offset + val
                if 0 <= idx <= max_idx:
                    new_dp1[val] |= (1 << idx)

            dp0, dp1 = new_dp0, new_dp1

        target = offset + k
        if not (0 <= target <= max_idx):
            return -1

        # look for the maximum product <= limit that attains sum = k
        for prod in range(limit, -1, -1):
            if (dp0[prod] >> target) & 1 or (dp1[prod] >> target) & 1:
                return prod
        return -1
```
