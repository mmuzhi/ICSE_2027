=== ANSWER ===
```python
import sys

def toggle_case(c: str) -> str:
    if 'a' <= c <= 'z':
        return chr(ord(c) - 32)
    else:
        return chr(ord(c) + 32)

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    S = data[0]
    Q = int(data[1])
    Ks = list(map(int, data[2:2+Q]))
    n = len(S)
    out = []
    for K in Ks:
        # 0‑based index
        idx = K - 1
        block = idx // n
        pos = idx % n
        ch = S[pos]
        if block.bit_count() & 1:  # odd number of 1‑bits -> toggle
            ch = toggle_case(ch)
        out.append(ch)
    sys.stdout.write(' '.join(out))

if __name__ == "__main__":
    solve()
```
