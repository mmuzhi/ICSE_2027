    # and build segment tree over L for range min of left_contrib
    left_contrib_by_L = [INF] * (max_coord + 2)   # index 1..max_coord
    for i in range(1, N + 1):
        l = L[i]
        val = W[i] + left_min[l]
        if val < left_contrib_by_L[l]:
            left_contrib_by_L[l] = val

    # iterative segment tree
    size = 1
    while size <= max_coord:
        size <<= 1
    seg = [INF] * (2 * size)
    for i in range(1, max_coord + 1):
        seg[size + i - 1] = left_contrib_by_L[i]
    for i in range(size - 1, 0, -1):
        seg[i] = min(seg[2*i], seg[2*i+1])

    def range_min(l: int, r: int) -> int:
        """inclusive range minimum query on [l, r] (1-indexed)"""
        if l > r:
            return INF
        l += size - 1
        r += size - 1
        res = INF
        while l <= r:
            if l & 1:
                res = min(res, seg[l])
                l += 1
            if not (r & 1):
                res = min(res, seg[r])
                r -= 1
            l >>= 1
            r >>= 1
        return res

    # ----- function for a path of the form s - a (right of s) - b (left of t) - t -----
    def two_intermediate(s: int, t: int) -> int:
        X = R[s]          # R_s
        Y = L[t]          # L_t
        # part1: choose a with L_a > max(X,Y) and b with R_b < Y, automatically R_b < L_a
        thresh = max(X, Y)
        minW_suff = suff_min_W_L[thresh]   # min_{L > thresh} W
        leftY = left_min[Y]                # min_{R < Y} W
        part1 = INF
        if leftY < INF and minW_suff < INF:
            part1 = leftY + minW_suff
        # part2: a with L_a in (X, Y]; b is the best for each such a
        part2 = INF
        if X < Y:
            part2 = range_min(X + 1, Y)
        best = min(part1, part2)
        if best >= INF:
            return INF
        return best + W[s] + W[t]

    Q = int(input())
    out = []
    for _ in range(Q):
        s, t = map(int, input().split())
        # direct edge
        if R[s] < L[t] or R[t] < L[s]:
            out.append(str(W[s] + W[t]))
            continue

        # overlapping intervals
        Ls, Rs = L[s], R[s]
        Lt, Rt = L[t], R[t]

        # single intermediate (x left of both or right of both)
        single = INF
        cand_left = left_min[min(Ls, Lt)]
        cand_right = suff_min_W_L[max(Rs, Rt)]
        if cand_left < INF or cand_right < INF:
            single = min(cand_left, cand_right) + W[s] + W[t]

        # two intermediate paths
        cand1 = two_intermediate(s, t)
        cand2 = two_intermediate(t, s)
        ans = min(single, cand1, cand2)
        out.append(str(ans) if ans < INF else "-1")

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    solve()
