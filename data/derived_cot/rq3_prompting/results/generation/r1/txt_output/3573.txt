    def validSubstringCount(self, word1: str, word2: str) -> int:
        n = len(word1)
        m = len(word2)
        if m > n:  # quick early exit, though not necessary
            return 0
        target = [0]*26
        for ch in word2:
            target[ord(ch)-97] += 1
        have = [0]*26
        missing = m
        left = 0
        ans = 0
        for right in range(n):
            idx = ord(word1[right]) - 97
            have[idx] += 1
            if have[idx] <= target[idx]:
                missing -= 1
            # Note: when have[idx] > target[idx], missing unchanged

                # move left
                idx_left = ord(word1[left]) - 97
                have[idx_left] -= 1
                if have[idx_left] < target[idx_left]:
                    missing += 1
                left += 1
        return ans

```python
class Solution:
    def validSubstringCount(self, word1: str, word2: str) -> int:
        n = len(word1)
        m = len(word2)
        if m > n:
            return 0
        
        target = [0] * 26
        for ch in word2:
            target[ord(ch) - 97] += 1
        
        have = [0] * 26
        missing = m          # total number of characters still needed
        left = 0
        ans = 0
        
        for right in range(n):
            idx = ord(word1[right]) - 97
            have[idx] += 1
            if have[idx] <= target[idx]:
                missing -= 1
            
            # When the current window [left, right] covers all required characters,
            # count all substrings starting at 'left' that are valid.
            while missing == 0:
                ans += (n - right)          # all substrings [left, right'] for right' >= right
                idx_left = ord(word1[left]) - 97
                have[idx_left] -= 1
                if have[idx_left] < target[idx_left]:
                    missing += 1
                left += 1
        
        return ans
```
