You will be given a competitive programming problem.
There is a city divided into H \times W blocks in the north-south-east-west directions, and there is exactly one building in each block.
Specifically, in the block at the i-th row from the north (1\leq i\leq H) and the j-th column from the west (1\leq j\leq W) (hereafter referred to as block (i,j)), there is a building of F_{i,j} floors.
Takahashi has two ways of moving. If he is on the X-th floor (1\leq X\leq F_{i,j}) of the building in block (i,j), he can:

- Move up or down one floor within the same building using stairs. If X=1, he cannot move down; if X=F_{i,j}, he cannot move up.
- Choose a building with at least X floors in a cardinally adjacent block, and move to the X-th floor of that building using a (sky) walkway.

Here, two blocks (i,j) and (i',j') are cardinally adjacent if and only if \lvert i - i'\rvert + \lvert j - j'\rvert = 1.
You are given Q queries to be answered. The i-th query (1\leq i\leq Q) is the following.

Find the minimum possible number of times that Takahashi uses stairs to move from the Y_i-th floor of the building in block (A_i,B_i) to the Z_i-th floor of the building in block (C_i,D_i).
The count of times using stairs is incremented each time he moves up or down one floor, possibly multiple times within the same building. (For example, moving from the 1st floor to the 6th floor of a building counts as 5 uses of stairs.)
Note that he does not have to minimize the number of times he uses walkways.

Output

Print Q lines. The i-th line should contain the answer to the i-th query as an integer.

import sys

def solve() -> None:
    input = sys.stdin.readline
    H, W = map(int, input().split())
    grid = [list(map(int, input().split())) for _ in range(H)]
    N = H * W
    idx = lambda i, j: i * W + j

    # build edges (undirected, only right and down to avoid duplicates)
    edges = []
    for i in range(H):
        for j in range(W):
            u = idx(i, j)
            if j + 1 < W:
                v = idx(i, j + 1)
                w = min(grid[i][j], grid[i][j + 1])
                edges.append((w, u, v))
            if i + 1 < H:
                v = idx(i + 1, j)
                w = min(grid[i][j], grid[i + 1][j])
                edges.append((w, u, v))
    edges.sort(key=lambda x: -x[0])            # descending weight

    # DSU for Kruskal
    parent = list(range(N))
    rank = [0] * N
    def find(x: int) -> int:
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x
    def union(x: int, y: int) -> bool:
        x = find(x)
        y = find(y)
        if x == y:
            return False
        if rank[x] < rank[y]:
            parent[x] = y
        elif rank[x] > rank[y]:
            parent[y] = x
        else:
            parent[y] = x
            rank[x] += 1
        return True

    # build maximum spanning tree
    tree = [[] for _ in range(N)]   # adjacency: (neighbor, weight)
    cnt = 0
    for w, u, v in edges:
        if union(u, v):
            tree[u].append((v, w))
            tree[v].append((u, w))
            cnt += 1
            if cnt == N - 1:
                break

    # LCA preprocessing
    LOG = (N).bit_length()
    up = [[0] * N for _ in range(LOG)]      # 2^k ancestors
    min_up = [[0] * N for _ in range(LOG)]  # min edge weight on the 2^k path
    depth = [0] * N

    # iterative DFS from root 0
    stack = [(0, -1, 0)]   # (node, parent, weight from parent)
    while stack:
        v, p, w = stack.pop()
        if p == -1:                     # root
            up[0][v] = v
            min_up[0][v] = 10**18       # sentinel
            depth[v] = 0
        else:
            up[0][v] = p
            min_up[0][v] = w
            depth[v] = depth[p] + 1
        for nb, ew in tree[v]:
            if nb == p:
                continue
            stack.append((nb, v, ew))

    # binary lifting tables
    for k in range(1, LOG):
        for v in range(N):
            anc = up[k-1][v]
            up[k][v] = up[k-1][anc]
            min_up[k][v] = min(min_up[k-1][v], min_up[k-1][anc])

    # function to get minimum edge weight on the tree path between u and v (u != v)
    def bottleneck(u: int, v: int) -> int:
        # ensure u is the deeper one
        if depth[u] < depth[v]:
            u, v = v, u
        ans = 10**18
        # lift u to depth of v
        diff = depth[u] - depth[v]
        bit = 0
        while diff:
            if diff & 1:
                ans = min(ans, min_up[bit][u])
                u = up[bit][u]
            diff >>= 1
            bit += 1
        if u == v:
            return ans
        # lift both together
        for k in range(LOG-1, -1, -1):
            if up[k][u] != up[k][v]:
                ans = min(ans, min_up[k][u])
                ans = min(ans, min_up[k][v])
                u = up[k][u]
                v = up[k][v]
        # now parent of u and v is LCA
        ans = min(ans, min_up[0][u])
        ans = min(ans, min_up[0][v])
        return ans

    # process queries
    Q = int(input())
    out_lines = []
    for _ in range(Q):
        a, b, y, c, d, z = map(int, input().split())
        u = idx(a-1, b-1)
        v = idx(c-1, d-1)
        if u == v:
            ans = abs(y - z)
        else:
            bneck = bottleneck(u, v)
            s = min(y, z)
            if bneck >= s:
                ans = abs(y - z)
            else:
                ans = y + z - 2 * bneck
        out_lines.append(str(ans))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
