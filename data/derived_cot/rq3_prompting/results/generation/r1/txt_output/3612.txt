=== ANSWER ===
```python
from typing import List

class Solution:
    def hasIncreasingSubarrays(self, nums: List[int], k: int) -> bool:
        n = len(nums)
        # Precompute whether subarray of length k starting at i is strictly increasing
        inc = [False] * (n - k + 1)
        for i in range(n - k + 1):
            # Check strictly increasing in nums[i..i+k-1]
            is_inc = True
            for j in range(k - 1):
                if nums[i + j] >= nums[i + j + 1]:
                    is_inc = False
                    break
            inc[i] = is_inc
        
        # Look for two adjacent increasing subarrays
        # i is start of first subarray, i+k is start of second
        for i in range(n - 2 * k + 1):
            if inc[i] and inc[i + k]:
                return True
        return False
```
