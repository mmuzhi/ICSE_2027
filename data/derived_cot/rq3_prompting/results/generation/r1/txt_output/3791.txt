from typing import List

class Solution:
    def numOfUnplacedFruits(self, fruits: List[int], baskets: List[int]) -> int:
        # Segment tree to support:
        # - update: mark a basket as used (set its capacity to 0)
        # - query: find leftmost index with capacity >= fruit (or -1 if none)
        class SegmentTree:
            def __init__(self, data):
                n = len(data)
                size = 1
                while size < n:
                    size <<= 1
                self.N = size
                self.tree = [0] * (2 * size)
                # leaves
                for i in range(n):
                    self.tree[size + i] = data[i]
                # build internal nodes
                for i in range(size - 1, 0, -1):
                    self.tree[i] = max(self.tree[2 * i], self.tree[2 * i + 1])

            def update(self, idx: int, value: int) -> None:
                i = idx + self.N
                self.tree[i] = value
                i //= 2
                while i:
                    self.tree[i] = max(self.tree[2 * i], self.tree[2 * i + 1])
                    i //= 2

            def query_leftmost(self, fruit: int) -> int:
                # if the whole tree has no basket with capacity >= fruit
                if self.tree[1] < fruit:
                    return -1
                node = 1
                # go down to the leaf, preferring the left child when possible
                while node < self.N:
                    if self.tree[2 * node] >= fruit:
                        node = 2 * node
                    else:
                        node = 2 * node + 1
                return node - self.N

        seg = SegmentTree(baskets)
        unplaced = 0
        for fruit in fruits:
            idx = seg.query_leftmost(fruit)
            if idx == -1:
                unplaced += 1
            else:
                seg.update(idx, 0)  # mark basket as used
        return unplaced
