```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1].strip()
    # Precompute left consecutive 1's
    left_ones = [0] * N
    for i in range(N):
        if S[i] == '1':
            left_ones[i] = 1 + (left_ones[i-1] if i > 0 else 0)
        else:
            left_ones[i] = 0
    # Precompute right consecutive 2's
    right_twos = [0] * N
    for i in range(N-1, -1, -1):
        if S[i] == '2':
            right_twos[i] = 1 + (right_twos[i+1] if i+1 < N else 0)
        else:
            right_twos[i] = 0
    ans = 0
    for i in range(N):
        if S[i] == '/':
            left = left_ones[i-1] if i > 0 else 0
            right = right_twos[i+1] if i+1 < N else 0
            length = 2 * min(left, right) + 1
            if length > ans:
                ans = length
    print(ans)

if __name__ == "__main__":
    main()
```
