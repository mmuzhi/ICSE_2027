from typing import List

class Solution:
    def maxWeight(self, pizzas: List[int]) -> int:
        n = len(pizzas)
        m = n // 4                     # total days
        odd_days = (m + 1) // 2        # number of odd days (1-indexed)
        even_days = m // 2             # number of even days

        # sort in descending order
        pizzas.sort(reverse=True)

        # odd day gains: the largest odd_days pizzas
        total = sum(pizzas[:odd_days])

        # even day gains: the next even_days pizzas after skipping the even_days maxes
        start = odd_days + even_days
        end = start + even_days
        total += sum(pizzas[start:end])

        return total
