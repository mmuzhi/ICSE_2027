- Choose positive integers x,y.

import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    n = int(next(it))
    adj = [[] for _ in range(n)]
    for _ in range(n - 1):
        u = int(next(it)) - 1
        v = int(next(it)) - 1
        adj[u].append(v)
        adj[v].append(u)

    deg = [len(adj[i]) for i in range(n)]

    max_kept = 0

    for c in range(n):
        # collect leaf counts for neighbours of c
        leaf_counts = []
        for nb in adj[c]:
            leaf_counts.append(deg[nb] - 1)

        if not leaf_counts:
            continue

        leaf_counts.sort(reverse=True)
        m = len(leaf_counts)
        i = 0
        while i < m:
            v = leaf_counts[i]
            if v == 0:
                # all further values are 0, and y must be positive
                break
            # find the end of the block with value v
            j = i
            while j < m and leaf_counts[j] == v:
                j += 1
            cnt = j                # number of elements >= v
            total = 1 + cnt * (1 + v)
            if total > max_kept:
                max_kept = total
            i = j

    # The problem guarantees that at least a 3‑vertex snowflake exists.
    print(n - max_kept)

if __name__ == "__main__":
    solve()
