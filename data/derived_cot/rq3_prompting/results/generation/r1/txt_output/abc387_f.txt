import sys
from collections import deque
from array import array

MOD = 998244353

def solve():
    import sys
    sys.setrecursionlimit(10000)
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [0] * (N+1)
    for i in range(1, N+1):
        A[i] = int(next(it))
    
    children = [[] for _ in range(N+1)]
    indeg = [0]*(N+1)
    for i in range(1, N+1):
        p = A[i]
        children[p].append(i)
        indeg[p] += 1
    
    # Find nodes not in cycles
    q = deque()
    for i in range(1, N+1):
        if indeg[i] == 0:
            q.append(i)
    in_cycle = [True]*(N+1)
    while q:
        u = q.popleft()
        in_cycle[u] = False
        v = A[u]
        indeg[v] -= 1
        if indeg[v] == 0:
            q.append(v)
    
    # DP for non-cycle nodes
    dp_S = [None]*(N+1)   # will hold array('I') for each non-cycle node
    
    def compute(node):
        if dp_S[node] is not None:
            return
        # product over children
        # f = [1]*(M+1)  # use list of ints
        f = [1] * (M+1)
        for c in children[node]:
            if in_cycle[c]:
                continue   # should not happen
            compute(c)
            sc = dp_S[c]
            # multiply f[v] by sc[v] for v=1..M
            for v in range(1, M+1):
                f[v] = (f[v] * sc[v]) % MOD
        # cumulative sum
        s = array('I', [0]) * (M+1)
        acc = 0
        for v in range(1, M+1):
            acc = (acc + f[v]) % MOD
            s[v] = acc
        dp_S[node] = s
    
    # Compute DP for all non-cycle nodes
    for i in range(1, N+1):
        if not in_cycle[i] and dp_S[i] is None:
            compute(i)
    
    # Process cycles
    visited = [False]*(N+1)
    total = 1
    for i in range(1, N+1):
        if in_cycle[i] and not visited[i]:
            # collect cycle nodes
            cycle_nodes = []
            cur = i
            while not visited[cur]:
                visited[cur] = True
                cycle_nodes.append(cur)
                cur = A[cur]
            # compute product of S of non-cycle children across all cycle nodes in this component
            contrib = [1]*(M+1)
            for u in cycle_nodes:
                for c in children[u]:
                    if not in_cycle[c]:
                        # ensure DP is computed (should already be computed, but just in case)
                        if dp_S[c] is None:
                            compute(c)
                        sc = dp_S[c]
                        for v in range(1, M+1):
                            contrib[v] = (contrib[v] * sc[v]) % MOD
            component_ans = sum(contrib[1:]) % MOD
            total = total * component_ans % MOD
    
    print(total)

import sys
from collections import deque
from array import array

MOD = 998244353

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [0] * (N + 1)
    for i in range(1, N + 1):
        A[i] = int(next(it))

    # reversed graph: children list
    children = [[] for _ in range(N + 1)]
    indeg = [0] * (N + 1)
    for i in range(1, N + 1):
        p = A[i]
        children[p].append(i)
        indeg[p] += 1

    # find nodes that are in cycles (Kahn's algorithm on reverse edges)
    q = deque()
    for i in range(1, N + 1):
        if indeg[i] == 0:
            q.append(i)
    in_cycle = [True] * (N + 1)
    while q:
        u = q.popleft()
        in_cycle[u] = False
        v = A[u]
        indeg[v] -= 1
        if indeg[v] == 0:
            q.append(v)

    # DP for nodes not in cycles
    dp_S = [None] * (N + 1)          # for each non‑cycle node, an array of length M+1

    def compute(node: int) -> None:
        """fill dp_S[node] using bottom‑up DP (subtree rooted at node)"""
        if dp_S[node] is not None:
            return
        # f[v] = product of S(child, v) over all children
        f = [1] * (M + 1)
        for c in children[node]:
            if in_cycle[c]:
                continue            # should not happen for a non‑cycle node
            compute(c)
            sc = dp_S[c]
            for v in range(1, M + 1):
                f[v] = (f[v] * sc[v]) % MOD
        # S[node][v] = sum_{w=1..v} f[w]
        s = array('I', [0]) * (M + 1)
        acc = 0
        for v in range(1, M + 1):
            acc = (acc + f[v]) % MOD
            s[v] = acc
        dp_S[node] = s

    # compute DP for all non‑cycle nodes
    for i in range(1, N + 1):
        if not in_cycle[i] and dp_S[i] is None:
            compute(i)

    # process each connected component (cycle + attached trees)
    visited = [False] * (N + 1)
    ans = 1
    for start in range(1, N + 1):
        if not in_cycle[start] or visited[start]:
            continue

        # collect all nodes of this cycle
        cycle_nodes = []
        cur = start
        while not visited[cur]:
            visited[cur] = True
            cycle_nodes.append(cur)
            cur = A[cur]

        # contribution of this component: sum over v of product of S(child, v)
        # for every non‑cycle child of every cycle node
        contrib = [1] * (M + 1)
        for u in cycle_nodes:
            for c in children[u]:
                if in_cycle[c]:
                    continue
                # c is guaranteed non‑cycle, DP already computed or we compute now
                if dp_S[c] is None:
                    compute(c)
                sc = dp_S[c]
                for v in range(1, M + 1):
                    contrib[v] = (contrib[v] * sc[v]) % MOD

        comp_ans = sum(contrib[1:]) % MOD
        ans = ans * comp_ans % MOD

    print(ans)


if __name__ == "__main__":
    solve()
