import math

class Solution:
    def maxScore(self, nums: List[int]) -> int:
        n = len(nums)
        # compute overall GCD and LCM
        total_gcd = 0
        total_lcm = 1
        for x in nums:
            total_gcd = math.gcd(total_gcd, x)
            total_lcm = total_lcm // math.gcd(total_lcm, x) * x  # using formula to avoid overflow? but fine.
        max_score = total_gcd * total_lcm
        # For each removal
        for i in range(n):
            g = 0
            l = 1
            for j in range(n):
                if j == i:
                    continue
                g = math.gcd(g, nums[j])
                l = l // math.gcd(l, nums[j]) * nums[j]
            score = g * l
            if score > max_score:
                max_score = score
        return max_score

```python
from typing import List
import math

class Solution:
    def maxScore(self, nums: List[int]) -> int:
        n = len(nums)
        # Compute overall GCD and LCM for the original array
        total_gcd = 0
        total_lcm = 1
        for x in nums:
            total_gcd = math.gcd(total_gcd, x)
            total_lcm = total_lcm // math.gcd(total_lcm, x) * x
        max_score = total_gcd * total_lcm
        
        # Try removing each element
        for i in range(n):
            g = 0
            l = 1
            for j in range(n):
                if j == i:
                    continue
                g = math.gcd(g, nums[j])
                l = l // math.gcd(l, nums[j]) * nums[j]
            score = g * l
            if score > max_score:
                max_score = score
        return max_score
```
