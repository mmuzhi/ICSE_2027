    # returns sum_{i=0}^{n-1} floor((a*i+b)/m)
    ans = 0
    while True:
        if a >= m:
            ans += (n-1)*n*(a//m)//2
            a %= m
        if b >= m:
            ans += n*(b//m)
            b %= m
        y_max = a*n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans

def floor_sum(n, m, a, b):
    # returns sum_{i=0}^{n-1} floor((a*i+b)/m)
    ans = 0
    while True:
        if a >= m:
            ans += (n-1)*n*(a//m)//2
            a %= m
        if b >= m:
            ans += n*(b//m)
            b %= m
        y_max = a*n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans

```
import math

def floor_sum(n, m, a, b):
    # sum_{i=0}^{n-1} floor((a*i+b)/m)
    ans = 0
    while True:
        if a >= m:
            ans += n*(n-1)//2 * (a//m)   # (n-1)*n//2 * (a//m)
            a %= m
        if b >= m:
            ans += n * (b//m)
            b %= m
        y_max = a*n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans
```

```python
import sys
import math

# ------------------------------------------------------------
# floor_sum(n, m, a, b) = sum_{i=0}^{n-1} floor((a*i + b) / m)
# works for non-negative a,b,m and n >= 0
# ------------------------------------------------------------
def floor_sum(n: int, m: int, a: int, b: int) -> int:
    ans = 0
    while True:
        if a >= m:
            # a = q*m + r   => floor((a*i+b)/m) = q*i + floor((r*i+b)/m)
            q = a // m
            ans += n * (n - 1) // 2 * q
            a %= m
        if b >= m:
            ans += n * (b // m)
            b %= m
        y_max = a * n + b
        if y_max < m:
            break
        # recursion step
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans

# ------------------------------------------------------------
def solve() -> None:
    it = iter(sys.stdin.read().split())
    N = int(next(it))
    M = int(next(it))
    C = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # special case C = 0
    if C == 0:
        print(K * min(A))
        return

    g = math.gcd(C, M)
    Mprime = M // g
    Cprime = C // g

    # sorted unique A
    A.sort()
    uniq = []
    for x in A:
        if not uniq or x != uniq[-1]:
            uniq.append(x)
    minA = uniq[0]

    # compute thresholds L = ceil((M - a) / g) = M' - (a // g)
    thresholds = []
    for a in uniq:
        L = Mprime - (a // g)
        if L < Mprime:               # only interesting thresholds
            thresholds.append((L, a))
    thresholds.sort(key=lambda x: x[0])   # ascending L

    # build segments [L,R) with constant phi
    segments = []                     # (L, R, phi)
    prev = 0
    if thresholds:
        first_L = thresholds[0][0]
        if first_L > prev:
            segments.append((prev, first_L, minA))   # no‑wrap region
        for i in range(len(thresholds) - 1):
            L_cur, A_cur = thresholds[i]
            L_next, A_next = thresholds[i+1]
            if L_cur < L_next:
                segments.append((L_cur, L_next, A_cur - M))
        last_L, last_A = thresholds[-1]
        if last_L < Mprime:
            segments.append((last_L, Mprime, last_A - M))
    else:
        # all L equal to M' → only no‑wrap region
        segments.append((0, Mprime, minA))

    # total sum over one full period (all x = 0 .. M'-1)
    sum_x_all = Mprime * (Mprime - 1) // 2
    total_period = g * sum_x_all
    for L, R, phi in segments:
        total_period += phi * (R - L)

    full_cycles = K // Mprime
    remainder = K % Mprime
    ans = full_cycles * total_period

    if remainder > 0:
        n = remainder
        # sum of x over the first n terms
        sum_x = Cprime * n * (n - 1) // 2 - Mprime * floor_sum(n, Mprime, Cprime, 0)
        sum_phi = 0
        for L, R, phi in segments:
            cnt = (floor_sum(n, Mprime, Cprime, Mprime - L) -
                   floor_sum(n, Mprime, Cprime, Mprime - R))
            sum_phi += phi * cnt
        ans += g * sum_x + sum_phi

    print(ans)

# ------------------------------------------------------------
if __name__ == "__main__":
    solve()
```
