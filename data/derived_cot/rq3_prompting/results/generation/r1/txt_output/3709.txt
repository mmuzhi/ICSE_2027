=== ANSWER ===
```python
class Solution:
    def hasSpecialSubstring(self, s: str, k: int) -> bool:
        n = len(s)
        for i in range(n - k + 1):
            # Check that all characters in the substring are the same
            if s[i:i+k] == s[i] * k:
                # Check left boundary
                if i > 0 and s[i-1] == s[i]:
                    continue
                # Check right boundary
                if i + k < n and s[i+k] == s[i]:
                    continue
                return True
        return False
```
