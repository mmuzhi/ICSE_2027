```python
from typing import List

class Solution:
    def maxSum(self, nums: List[int], k: int, m: int) -> int:
        n = len(nums)
        prefix = [0] * (n+1)
        for i in range(n):
            prefix[i+1] = prefix[i] + nums[i]
        INF = 10**18
        dp_prev = [0] * (n+1)   # j=0: 0 subarrays
        for j in range(1, k+1):
            dp_cur = [-INF] * (n+1)
            best = -INF
            for i in range(1, n+1):
                if i - m >= 0:
                    # update best with dp_prev[l] - prefix[l] for l = i-m
                    if dp_prev[i-m] - prefix[i-m] > best:
                        best = dp_prev[i-m] - prefix[i-m]
                if best > -INF//2:
                    candidate = prefix[i] + best
                else:
                    candidate = -INF
                dp_cur[i] = max(dp_cur[i-1], candidate)
            dp_prev = dp_cur
        return dp_prev[n]
```

```python
from typing import List

class Solution:
    def maxSum(self, nums: List[int], k: int, m: int) -> int:
        n = len(nums)
        # prefix sums
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + nums[i]
        
        INF = 10 ** 18
        # dp for j = 0 (zero subarrays)
        dp_prev = [0] * (n + 1)       # dp[i][0] = 0
        
        for j in range(1, k + 1):
            dp_cur = [-INF] * (n + 1) # dp[i][j] initialized as -INF
            best = -INF               # max over l of (dp[l][j-1] - prefix[l])
            for i in range(1, n + 1):
                # l = i - m can now be used as start of a subarray ending at i-1
                if i - m >= 0:
                    val = dp_prev[i - m] - prefix[i - m]
                    if val > best:
                        best = val
                if best > -INF // 2:   # avoid overflow, best is not -INF
                    candidate = prefix[i] + best
                else:
                    candidate = -INF
                dp_cur[i] = max(dp_cur[i - 1], candidate)
            dp_prev = dp_cur
        
        return dp_prev[n]
```
