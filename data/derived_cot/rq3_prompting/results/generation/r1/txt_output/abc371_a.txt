def solve():
    import sys
    data = sys.stdin.read().strip().split()
    if not data:
        return
    S_AB, S_AC, S_BC = data[0], data[1], data[2]

    # Count how many each brother is older than
    older_A = (1 if S_AB == '>' else 0) + (1 if S_AC == '>' else 0)
    older_B = (1 if S_AB == '<' else 0) + (1 if S_BC == '>' else 0)
    older_C = (1 if S_AC == '<' else 0) + (1 if S_BC == '<' else 0)

    # The middle brother is the one with exactly 1 older count
    if older_A == 1:
        print('A')
    elif older_B == 1:
        print('B')
    else:
        print('C')

if __name__ == "__main__":
