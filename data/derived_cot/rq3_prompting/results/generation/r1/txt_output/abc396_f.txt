```python
import sys

def solve() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())
    A = list(map(int, input().split()))

    # ---------- initial inversion number (k=0) ----------
    # Fenwick tree over values 0 .. M-1
    size = M + 2
    bit = [0] * size

    def bit_add(idx: int, val: int) -> None:
        # idx is value (0-indexed)
        i = idx + 1          # 1‑based index for BIT
        while i < size:
            bit[i] += val
            i += i & -i

    def bit_sum(idx: int) -> int:
        # sum of counts for values <= idx
        i = idx + 1
        s = 0
        while i > 0:
            s += bit[i]
            i -= i & -i
        return s

    inv0 = 0
    for i, x in enumerate(A, start=1):   # i : 1‑indexed position
        # count previous elements > x
        le = bit_sum(x)                 # number of values <= x seen so far
        inv0 += (i - 1) - le
        bit_add(x, 1)

    # ---------- pre‑compute for each value ----------
    cnt = [0] * M
    pos_sum = [0] * M
    for i, val in enumerate(A, start=1):
        cnt[val] += 1
        pos_sum[val] += i

    # ---------- compute answers for all k ----------
    inv = [0] * M
    inv[0] = inv0
    for k in range(M - 1):
        w = (M - 1 - k) % M
        delta = 2 * pos_sum[w] - (N + 1) * cnt[w]
        inv[k + 1] = inv[k] + delta

    sys.stdout.write("\n".join(map(str, inv)))

if __name__ == "__main__":
    solve()
```
