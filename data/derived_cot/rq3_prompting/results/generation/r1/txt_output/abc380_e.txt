There are N cells in a row, numbered 1 to N.
For each 1 \leq i < N, cells i and i+1 are adjacent.
Initially, cell i is painted with color i.

- 1 x c: Repaint the following to color c: all reachable cells reachable from cell x by repeatedly moving to an adjacent cell painted in the same color as the current cell.
- 2 c: Print the number of cells painted with color c.

Input

N Q

The i-th line should contain the answer to the i-th such query.

Constraints


- 1 \leq N \leq 5 \times 10^5
- 1 \leq Q \leq 2 \times 10^5
- In queries of the first type, 1 \leq x \leq N.
- In queries of the first and second types, 1 \leq c \leq N.
- There is at least one query of the second type.
- All input values are integers.

def main() -> None:

    cnt = [0] * (N + 1)
    for i in range(1, N + 1):
        cnt[i] = 1

    # DSU find with path compression (iterative)
    def find(x: int) -> int:
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    # union two components, returns the new root
    def union(a: int, b: int) -> int:
        a = find(a)
        b = find(b)
        if a == b:
            return a
        if size[a] < size[b]:
            a, b = b, a
        parent[b] = a
        size[a] += size[b]
        left[a] = min(left[a], left[b])
        right[a] = max(right[a], right[b])
        # colour is already the same (guaranteed by caller)
        return a

    out = []

    for _ in range(Q):
        typ = int(next(it))
        if typ == 1:
            x = int(next(it))
            c = int(next(it))
            r = find(x)
            old = col[r]
            if old == c:
                continue          # nothing changes
            seg_sz = size[r]
            cnt[old] -= seg_sz
            cnt[c] += seg_sz
            col[r] = c

                if rgt < N:
                    right_root = find(rgt + 1)
                    if col[right_root] == c:
                        r = union(r, right_root)
                        continue
                break

        else:   # type 2
            c = int(next(it))
            out.append(str(cnt[c]))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
