    def subarraySum(self, nums: List[int]) -> int:

        prefix = [0] * (n + 1)

```python
from typing import List

class Solution:
    def subarraySum(self, nums: List[int]) -> int:
        n = len(nums)
        # Build prefix sums: prefix[i] = sum of nums[0..i-1]
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + nums[i]
        
        total = 0
        for i in range(n):
            start = max(0, i - nums[i])          # left boundary of the subarray
            total += prefix[i + 1] - prefix[start]  # sum of nums[start..i]
        return total
```
