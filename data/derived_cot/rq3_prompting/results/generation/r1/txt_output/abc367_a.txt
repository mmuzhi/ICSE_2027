=== ANSWER ===
```python
import sys

def solve() -> None:
    data = sys.stdin.readline().strip().split()
    if not data:
        return
    A, B, C = map(int, data)

    if B < C:
        # sleep from B to C (non-wrapping)
        if B <= A < C:
            print("No")
        else:
            print("Yes")
    else:
        # sleep from B to 24 and from 0 to C (wrapping)
        if A >= B or A < C:
            print("No")
        else:
            print("Yes")

if __name__ == "__main__":
    solve()
```
