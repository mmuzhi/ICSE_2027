def solve():
    import sys
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    P = [int(next(it)) for _ in range(N)]

    def calc(C):
        total_cost = 0
        total_count = 0
        for p in P:
            q = C // p
            n = (q + 1) // 2
            total_count += n
            total_cost += p * n * n
        return total_count, total_cost

    lo = 0
    hi = M
    while lo < hi:
        mid = (lo + hi + 1) // 2
        _, cost = calc(mid)
        if cost <= M:
            lo = mid
        else:
            hi = mid - 1
    C = lo
    count, cost = calc(C)
    # compute next L
    L = None
    for p in P:
        q = C // p
        n = (q + 1) // 2
        nxt = (2 * n + 1) * p
        if L is None or nxt < L:
            L = nxt
    ans = count + (M - cost) // L
    print(ans)

import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    P = [int(next(it)) for _ in range(N)]

    # returns (total_units, total_cost) for threshold C
    def compute(C: int):
        units = 0
        cost = 0
        for p in P:
            q = C // p
            n = (q + 1) // 2
            units += n
            cost += p * n * n
        return units, cost

    # binary search for the largest C such that cost(C) <= M
    lo, hi = 0, M
    while lo < hi:
        mid = (lo + hi + 1) // 2
        _, cst = compute(mid)
        if cst <= M:
            lo = mid
        else:
            hi = mid - 1
    C = lo
    units, cost = compute(C)

    # compute the next cheapest marginal cost L
    L = None
    for p in P:
        q = C // p
        n = (q + 1) // 2
        nxt = (2 * n + 1) * p
        if L is None or nxt < L:
            L = nxt

    ans = units + (M - cost) // L
    print(ans)


if __name__ == "__main__":
    solve()
