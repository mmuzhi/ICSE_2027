- Let your current square be x. Choose an integer i that satisfies all of the following conditions, and move to square x + i.

- R_i < L_{i+1} \ (1 \leq i \leq M - 1)
- All input values are integers.

import sys

def simulate_interval(l, r, entries, A, B):
    """
    Simulate reachable positions in [l, r] given a set of starting positions.
    Return an integer mask where bit i (0-indexed) corresponds to position (r - i)
    being reachable. Only bits for positions that actually exist in [l, r] are valid.
    """
    Bsize = B
    recent = [False] * Bsize          # recent[i] = reachable of (current-1-i)
    # start from the smallest entry
    pos = min(entries)
    entry_set = entries
    consecutive = 0
    # iterate from the first entry to the right end
    while pos <= r:
        if pos in entry_set:
            reachable = True
        else:
            # check if any reachable position in [pos-B, pos-A]
            # indices: recent[i] corresponds to step size i+1, so need i in [A-1, B-1]
            reachable = any(recent[i] for i in range(A-1, Bsize))
        # update sliding window
        recent = [reachable] + recent[:-1]
        if reachable:
            consecutive += 1
        else:
            consecutive = 0
        # if we have B consecutive reachable positions, all later positions are reachable
        if consecutive >= Bsize:
            # all positions from now on are reachable
            # compute mask for the last B positions of the whole interval
            max_bits = min(Bsize, r - l + 1)
            return (1 << max_bits) - 1   # all ones
        pos += 1
    # finished the whole interval without getting dense
    max_bits = min(Bsize, r - l + 1)
    mask = 0
    for i in range(max_bits):
        if recent[i]:
            mask |= (1 << i)
    return mask

def solve():
    import sys
    input = sys.stdin.readline
    N, M, A, B = map(int, input().split())
    bad = []
    for _ in range(M):
        L, R = map(int, input().split())
        bad.append((L, R))
    # special case: only one step size
    if A == B:
        k = A
        if (N - 1) % k != 0:
            print("No")
            return
        for L, R in bad:
            # check if any position of the progression 1 + k*t lies in [L, R]
            t_min = (L - 1 + k - 1) // k
            t_max = (R - 1) // k
            if t_min <= t_max:
                print("No")
                return
        print("Yes")
        return

    # general case: A < B
    # build good intervals
    good = []
    cur = 1
    for L, R in bad:
        if cur <= L - 1:
            good.append((cur, L - 1))
        cur = R + 1
    if cur <= N:
        good.append((cur, N))

    # process the first good interval
    l, r = good[0]
    entries = {l}          # we start at square 1
    prev_mask = simulate_interval(l, r, entries, A, B)
    prev_r = r

    # process the remaining intervals
    for i in range(1, len(good)):
        l, r = good[i]
        dist = l - prev_r
        if dist > B:
            print("No")
            return
        # compute entry points for this interval
        entries = set()
        max_p = min(r, prev_r + B)
        if max_p < l:
            print("No")
            return
        # iterate over possible landing positions
        for p in range(l, max_p + 1):
            # check if there is a reachable q in previous interval with p-q in [A,B]
            # we only know reachability for the last B positions of previous interval
            for i in range(B):
                if (prev_mask >> i) & 1:
                    q = prev_r - i
                    d = p - q
                    if A <= d <= B:
                        entries.add(p)
                        break
        if not entries:
            print("No")
            return
        # simulate the current interval
        mask = simulate_interval(l, r, entries, A, B)
        prev_mask = mask
        prev_r = r

    # after the last interval, check if the last square N is reachable
    if (prev_mask & 1):
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    solve()
