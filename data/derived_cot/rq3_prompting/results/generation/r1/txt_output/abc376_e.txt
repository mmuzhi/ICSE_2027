import sys
import heapq

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    T = int(next(it))
    out_lines = []

    for _ in range(T):
        N = int(next(it))
        K = int(next(it))
        A = [int(next(it)) for _ in range(N)]
        B = [int(next(it)) for _ in range(N)]

        # pair and sort by A
        pairs = list(zip(A, B))
        pairs.sort(key=lambda x: x[0])

        # max-heap (store negative values) to hold the K smallest B so far
        heap = []
        sumK = 0
        # count of each B value currently in the heap (for membership checking)
        cnt = {}

        ans = None
        i = 0
        while i < N:
            curA = pairs[i][0]
            groupB = []
            while i < N and pairs[i][0] == curA:
                groupB.append(pairs[i][1])
                i += 1

            # insert all B of this group into the heap (updating K smallest)
            for b in groupB:
                if len(heap) < K:
                    heapq.heappush(heap, -b)
                    sumK += b
                    cnt[b] = cnt.get(b, 0) + 1
                else:
                    # top of heap is the largest among the K smallest
                    largest = -heap[0]
                    if b < largest:
                        # replace the largest with b
                        heapq.heappop(heap)
                        sumK -= largest
                        cnt[largest] -= 1
                        if cnt[largest] == 0:
                            del cnt[largest]
                        heapq.heappush(heap, -b)
                        sumK += b
                        cnt[b] = cnt.get(b, 0) + 1

            # after processing the whole group, we can form a subset if enough elements
            if len(heap) >= K:
                # check whether the current group contributes any element to the heap
                has_current = False
                for b in groupB:
                    if cnt.get(b, 0) > 0:
                        has_current = True
                        break
                if has_current:
                    cand = curA * sumK
                else:
                    min_b = min(groupB)
                    largest_in_heap = -heap[0]
                    new_sum = sumK - largest_in_heap + min_b
                    cand = curA * new_sum
                if ans is None or cand < ans:
                    ans = cand

        out_lines.append(str(ans))

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
