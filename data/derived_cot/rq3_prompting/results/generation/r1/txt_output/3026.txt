```python
class Solution:
    def minimumPossibleSum(self, n: int, target: int) -> int:
        MOD = 10**9 + 7
        
        # Number of the smallest numbers we can take from 1..target-1 without conflict
        m = target // 2
        
        if n <= m:
            # Just take the first n positive integers
            return (n * (n + 1) // 2) % MOD
        else:
            # Take all m numbers from 1..m (sum = m*(m+1)//2)
            sum_small = m * (m + 1) // 2
            extra = n - m
            # Then take the next extra numbers starting from target
            sum_extra = extra * target + extra * (extra - 1) // 2
            return (sum_small + sum_extra) % MOD
```
