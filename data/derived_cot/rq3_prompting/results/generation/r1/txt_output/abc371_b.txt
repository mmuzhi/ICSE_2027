=== ANSWER ===
```python
import sys

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    seen = [False] * (N + 1)  # 1-indexed families
    out_lines = []
    for _ in range(M):
        a = int(next(it))
        b = next(it)
        if b == 'M' and not seen[a]:
            out_lines.append("Yes")
            seen[a] = True
        else:
            out_lines.append("No")
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    main()
```
