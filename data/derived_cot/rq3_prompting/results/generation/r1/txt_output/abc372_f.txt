        # ---- update add for future steps ----
        # fetch w-values of this n for fast access
        w_n = [w[iv][n] for iv in range(L)]
        for i in range(contrib_cnt):
            iy = iy_list[i]
            iv = iv_list[i]
            target = n + 1 + d_list[i]
            if target <= K:
                # add w[iv][n] to add[iy][target], keep modulo
                tmp = add[iy][target] + w_n[iv]
                if tmp >= MOD:
                    tmp -= MOD
                add[iy][target] = tmp

    # ---- extract the answer ----
    ans = 1  # account for the deterministic walk (K steps of cycle)
    if K >= 1:
        r = (K - 1) % N
        for iy in range(L):
            idx = r
            while idx <= K-1:
                ans += w[iy][idx]
                idx += N
        ans %= MOD
    print(ans)


if __name__ == "__main__":
    solve()
