def solve():
    import sys
    s = sys.stdin.readline().strip()
    stack = []
    pairs = {')':'(', ']':'[', '>':'<'}
    for ch in s:
        if ch in "([<":
            stack.append(ch)
        else:
            if not stack or stack[-1] != pairs[ch]:
                print("No")
                return
            stack.pop()
    print("Yes" if not stack else "No")

import sys

def solve() -> None:
    s = sys.stdin.readline().strip()
    stack = []
    matching = {')': '(', ']': '[', '>': '<'}
    
    for ch in s:
        if ch in '([<':
            stack.append(ch)
        else:
            if not stack or stack[-1] != matching[ch]:
                print("No")
                return
            stack.pop()
    
    print("Yes" if not stack else "No")

if __name__ == "__main__":
    solve()
