import sys

MOD = 998244353

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    B = [0] + [int(next(it)) for _ in range(N)]  # 1-indexed

    q = sum(1 for i in range(1, N+1) if B[i] == -1)

    # precompute powers p[x][e] = x^e mod MOD
    # x from 0 to M, e from 0 to N
    pows = [[1] * (N+1) for _ in range(M+1)]
    for x in range(M+1):
        for e in range(1, N+1):
            pows[x][e] = pows[x][e-1] * x % MOD

    # prefix arrays
    pref_free = [0] * (N+1)
    pref_min = [M+1] * (N+1)  # sentinel large
    for i in range(1, N+1):
        pref_free[i] = pref_free[i-1] + (1 if B[i]==-1 else 0)
        if B[i] == -1:
            pref_min[i] = pref_min[i-1]
        else:
            pref_min[i] = min(pref_min[i-1], B[i])

    # suffix arrays
    suff_free = [0] * (N+2)
    suff_max = [0] * (N+2)  # sentinel 0
    for i in range(N, 0, -1):
        suff_free[i] = suff_free[i+1] + (1 if B[i]==-1 else 0)
        if B[i] == -1:
            suff_max[i] = suff_max[i+1]
        else:
            suff_max[i] = max(suff_max[i+1], B[i])

    total_cuts = 0
    for k in range(1, N):
        freeL = pref_free[k]
        L_min = pref_min[k]   # M+1 if none
        freeR = suff_free[k+1]
        R_max = suff_max[k+1] # 0 if none

        v_low = max(1, R_max)
