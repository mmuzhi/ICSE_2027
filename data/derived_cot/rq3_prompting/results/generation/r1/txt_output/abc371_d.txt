=== ANSWER ===
```python
import sys
import bisect

def main() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    X = [int(next(it)) for _ in range(N)]
    P = [int(next(it)) for _ in range(N)]

    # prefix sums
    pref = [0] * N
    s = 0
    for i, p in enumerate(P):
        s += p
        pref[i] = s

    Q = int(next(it))
    out_lines = []
    for _ in range(Q):
        L = int(next(it))
        R = int(next(it))
        # leftmost index with X[i] >= L
        left = bisect.bisect_left(X, L)
        # rightmost index with X[i] <= R
        right = bisect.bisect_right(X, R) - 1
        if left > right:
            out_lines.append('0')
        else:
            total = pref[right] - (pref[left - 1] if left > 0 else 0)
            out_lines.append(str(total))
    sys.stdout.write('\n'.join(out_lines))

if __name__ == "__main__":
    main()
```
