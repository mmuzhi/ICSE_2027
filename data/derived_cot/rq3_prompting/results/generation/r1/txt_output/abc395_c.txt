=== ANSWER ===
```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    A = list(map(int, data[1:1+n]))
    last_occurrence = {}
    min_len = n + 1  # larger than any possible subarray length

    for idx, val in enumerate(A, start=1):
        if val in last_occurrence:
            length = idx - last_occurrence[val] + 1
            if length < min_len:
                min_len = length
        last_occurrence[val] = idx

    if min_len == n + 1:
        print(-1)
    else:
        print(min_len)

if __name__ == "__main__":
    main()
```
