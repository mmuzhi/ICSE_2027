Implementation details for BIT:

class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0]*(n+1)
    def add(self, i, delta):
        while i <= self.n:
            self.tree[i] += delta
            i += i&-i
    def sum(self, i):
        s=0
        while i>0:
            s += self.tree[i]
            i -= i&-i
        return s
    def find_kth(self, k): # find smallest index such that prefix >= k, assume 1-indexed, k>=1
        # binary lifting
        idx = 0
        bitmask = 1 << (self.n.bit_length())
        while bitmask:
            t = idx + bitmask
            if t <= self.n and self.tree[t] < k:
                idx = t
                k -= self.tree[t]
            bitmask >>= 1
        return idx+1

class Solution:
    def minOperations(self, nums: List[int], x: int, k: int) -> int:
        n = len(nums)
        # compress
        vals = sorted(set(nums))
        comp = {v:i+1 for i,v in enumerate(vals)}  # 1-indexed
        M = len(vals)
        # BIT count and sum
        bit_cnt = [0]*(M+2)
        bit_sum = [0]*(M+2)
        def bit_add(bit, i, delta):
            while i <= M:
                bit[i] += delta
                i += i & -i
        def bit_sum_prefix(bit, i):
            s=0
            while i>0:
                s += bit[i]
                i -= i & -i
            return s
        def find_kth(bit, k): # find smallest i such that prefix count >= k
            idx = 0
            bitmask = 1 << (M.bit_length())
            while bitmask:
                nxt = idx + bitmask
                if nxt <= M and bit[nxt] < k:
                    idx = nxt
                    k -= bit[nxt]
                bitmask >>= 1
            return idx + 1

        # initialize with first x elements
        total_sum = 0
        for i in range(x):
            v = nums[i]
            idx = comp[v]
            bit_add(bit_cnt, idx, 1)
            bit_add(bit_sum, idx, v)
            total_sum += v

        # compute costs
        cost = [0]*(n-x+1)
        for start in range(n-x+1):
            # compute cost for current window
            kth = (x+1)//2
            median_idx = find_kth(bit_cnt, kth)
            median_val = vals[median_idx-1]  # because vals is 0-indexed, median_idx is 1-indexed
            # sum_left: sum of values < median
            sum_left = bit_sum_prefix(bit_sum, median_idx-1)
            count_left = bit_sum_prefix(bit_cnt, median_idx-1)
            # count of median

            sum_right = total_sum - sum_left - median_val * count_median
            count_right = x - count_left - count_median
            cur_cost = (median_val * count_left - sum_left) + (sum_right - median_val * count_right)
            cost[start] = cur_cost

                v_out = nums[start]
                idx_out = comp[v_out]

from typing import List

class Solution:
    def minOperations(self, nums: List[int], x: int, k: int) -> int:
        n = len(nums)
        # coordinate compression
        vals = sorted(set(nums))
        comp = {v: i + 1 for i, v in enumerate(vals)}  # 1‑based index
        m = len(vals)

        # Fenwick trees for counts and sums
        bit_cnt = [0] * (m + 2)
        bit_sum = [0] * (m + 2)

        def bit_add(bit, idx, delta):
            while idx <= m:
                bit[idx] += delta
                idx += idx & -idx

        def bit_prefix(bit, idx):
            s = 0
            while idx > 0:
                s += bit[idx]
                idx -= idx & -idx
            return s

        def find_kth(bit, k):
            """return smallest 1‑based index such that prefix count >= k"""
            idx = 0
            bitmask = 1 << (m.bit_length())
            while bitmask:
                nxt = idx + bitmask
                if nxt <= m and bit[nxt] < k:
                    idx = nxt
                    k -= bit[nxt]
                bitmask >>= 1
            return idx + 1

        # initialise sliding window with first x elements
        total_sum = 0
        for i in range(x):
            v = nums[i]
            idx = comp[v]
            bit_add(bit_cnt, idx, 1)
            bit_add(bit_sum, idx, v)
            total_sum += v

        # compute cost for each possible start of a subarray
        cost = [0] * (n - x + 1)
        for start in range(n - x + 1):
            # median = lower median
            kth = (x + 1) // 2
            med_idx = find_kth(bit_cnt, kth)
            med = vals[med_idx - 1]

            cnt_left = bit_prefix(bit_cnt, med_idx - 1)
            sum_left = bit_prefix(bit_sum, med_idx - 1)
            cnt_med = bit_prefix(bit_cnt, med_idx) - cnt_left
            sum_right = total_sum - sum_left - med * cnt_med
            cnt_right = x - cnt_left - cnt_med

            cost[start] = (med * cnt_left - sum_left) + (sum_right - med * cnt_right)

            # slide to next window
            if start < n - x:
                # remove leftmost element
                out_val = nums[start]
                out_idx = comp[out_val]
                bit_add(bit_cnt, out_idx, -1)
                bit_add(bit_sum, out_idx, -out_val)
                total_sum -= out_val
                # add new element
                in_val = nums[start + x]
                in_idx = comp[in_val]
                bit_add(bit_cnt, in_idx, 1)
                bit_add(bit_sum, in_idx, in_val)
                total_sum += in_val

        # DP over positions: dp_prev holds results for j-1 intervals
        INF = 10**18
        dp_prev = [0] * (n + 1)  # j = 0 → cost 0 everywhere
        for j in range(1, k + 1):
            dp_curr = [INF] * (n + 1)
            # dp_curr[0] stays INF
            for i in range(1, n + 1):
                # skip element i-1
                dp_curr[i] = dp_curr[i - 1]
                # take an interval ending at i-1
                if i >= x:
                    start = i - x
                    dp_curr[i] = min(dp_curr[i], dp_prev[start] + cost[start])
            dp_prev = dp_curr

        return dp_prev[n]
