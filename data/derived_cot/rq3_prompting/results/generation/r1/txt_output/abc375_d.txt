=== ANSWER ===
```python
import sys

def solve() -> None:
    S = sys.stdin.readline().strip()
    n = len(S)
    # Count occurrences of each letter (0 for 'A', ..., 25 for 'Z')
    right = [0] * 26
    for ch in S:
        right[ord(ch) - 65] += 1

    left = [0] * 26
    ans = 0

    for ch in S:
        idx = ord(ch) - 65
        # Remove current position from the right side
        right[idx] -= 1

        # Count triples where the middle is the current position
        cur = 0
        for c in range(26):
            cur += left[c] * right[c]
        ans += cur

        # Add current position to the left side for future steps
        left[idx] += 1

    print(ans)

if __name__ == "__main__":
    solve()
```
