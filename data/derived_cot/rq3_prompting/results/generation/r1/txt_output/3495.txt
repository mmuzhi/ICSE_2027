=== ANSWER ===
```python
from typing import List
import heapq

class Solution:
    def resultsArray(self, queries: List[List[int]], k: int) -> List[int]:
        # max-heap as negatives to keep the k smallest distances
        heap = []
        res = []
        for x, y in queries:
            d = abs(x) + abs(y)
            # push negative distance to simulate max-heap
            heapq.heappush(heap, -d)
            if len(heap) > k:
                # remove the largest distance (most negative)
                heapq.heappop(heap)
            if len(heap) == k:
                # the top of the heap is the largest among the k smallest (k-th smallest)
                res.append(-heap[0])
            else:
                res.append(-1)
        return res
```
