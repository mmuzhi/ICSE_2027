from typing import List

class Solution:
    def lenOfVDiagonal(self, grid: List[List[int]]) -> int:
        n = len(grid)
        m = len(grid[0])
        # directions: DR, DL, UL, UR (clockwise order)
        dirs = [(1, 1), (1, -1), (-1, -1), (-1, 1)]
        # helper for next value in the pattern
        def nxt(v: int) -> int:
            if v == 1:
                return 2
            if v == 2:
                return 0
            return 2   # v == 0 -> 2
        
        # end[dir][i][j] = longest straight segment ending at (i,j) coming from direction 'dir'
        end = [[[0] * m for _ in range(n)] for _ in range(4)]
        
        # compute end for each direction
        for di, (dr, dc) in enumerate(dirs):
            if dr == 1 and dc == 1:          # DR : top-left -> bottom-right
                for i in range(n):
                    for j in range(m):
                        if grid[i][j] == 1:
                            end[di][i][j] = max(end[di][i][j], 1)
                        pi, pj = i - dr, j - dc
                        if 0 <= pi < n and 0 <= pj < m and end[di][pi][pj] > 0:
                            if grid[i][j] == nxt(grid[pi][pj]):
                                end[di][i][j] = max(end[di][i][j], end[di][pi][pj] + 1)
            elif dr == 1 and dc == -1:       # DL : top-right -> bottom-left
                for i in range(n):
                    for j in range(m - 1, -1, -1):
                        if grid[i][j] == 1:
                            end[di][i][j] = max(end[di][i][j], 1)
                        pi, pj = i - 1, j + 1
                        if 0 <= pi < n and 0 <= pj < m and end[di][pi][pj] > 0:
                            if grid[i][j] == nxt(grid[pi][pj]):
                                end[di][i][j] = max(end[di][i][j], end[di][pi][pj] + 1)
            elif dr == -1 and dc == -1:      # UL : bottom-right -> top-left
                for i in range(n - 1, -1, -1):
                    for j in range(m - 1, -1, -1):
                        if grid[i][j] == 1:
                            end[di][i][j] = max(end[di][i][j], 1)
                        pi, pj = i + 1, j + 1
                        if 0 <= pi < n and 0 <= pj < m and end[di][pi][pj] > 0:
                            if grid[i][j] == nxt(grid[pi][pj]):
                                end[di][i][j] = max(end[di][i][j], end[di][pi][pj] + 1)
            else:                            # UR : bottom-left -> top-right
                for i in range(n - 1, -1, -1):
                    for j in range(m):
                        if grid[i][j] == 1:
                            end[di][i][j] = max(end[di][i][j], 1)
                        pi, pj = i + 1, j - 1
                        if 0 <= pi < n and 0 <= pj < m and end[di][pi][pj] > 0:
                            if grid[i][j] == nxt(grid[pi][pj]):
                                end[di][i][j] = max(end[di][i][j], end[di][pi][pj] + 1)
        
        # start[dir][i][j] = longest straight segment starting at (i,j) going in direction 'dir'
        start = [[[1] * m for _ in range(n)] for _ in range(4)]   # at least the cell itself
        
        for di, (dr, dc) in enumerate(dirs):
            if dr == 1 and dc == 1:          # DR : next (i+1,j+1)
                for i in range(n - 1, -1, -1):
                    for j in range(m - 1, -1, -1):
                        ni, nj = i + dr, j + dc
                        if 0 <= ni < n and 0 <= nj < m and start[di][ni][nj] > 0:
                            if grid[ni][nj] == nxt(grid[i][j]):
                                start[di][i][j] = max(start[di][i][j], start[di][ni][nj] + 1)
            elif dr == 1 and dc == -1:       # DL : next (i+1,j-1)
                for i in range(n - 1, -1, -1):
                    for j in range(m):
                        ni, nj = i + dr, j + dc
                        if 0 <= ni < n and 0 <= nj < m and start[di][ni][nj] > 0:
                            if grid[ni][nj] == nxt(grid[i][j]):
                                start[di][i][j] = max(start[di][i][j], start[di][ni][nj] + 1)
            elif dr == -1 and dc == -1:      # UL : next (i-1,j-1)
                for i in range(n):
                    for j in range(m):
                        ni, nj = i + dr, j + dc
                        if 0 <= ni < n and 0 <= nj < m and start[di][ni][nj] > 0:
                            if grid[ni][nj] == nxt(grid[i][j]):
                                start[di][i][j] = max(start[di][i][j], start[di][ni][nj] + 1)
            else:                            # UR : next (i-1,j+1)
                for i in range(n):
                    for j in range(m - 1, -1, -1):
                        ni, nj = i + dr, j + dc
                        if 0 <= ni < n and 0 <= nj < m and start[di][ni][nj] > 0:
                            if grid[ni][nj] == nxt(grid[i][j]):
                                start[di][i][j] = max(start[di][i][j], start[di][ni][nj] + 1)
        
        # combine: best straight segment
        ans = 0
        for i in range(n):
            for j in range(m):
                for di in range(4):
                    ans = max(ans, end[di][i][j])
        
        # V-shaped with a turn
        for i in range(n):
            for j in range(m):
                for di in range(4):
                    d2 = (di + 1) % 4   # clockwise turn
                    if end[di][i][j] > 0 and start[d2][i][j] > 0:
                        ans = max(ans, end[di][i][j] + start[d2][i][j] - 1)
        
        return ans
