=== ANSWER ===
```python
import sys

def main() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    # read n numbers for A, then n numbers for B
    # we can compute max on the fly
    max_a = -10**18  # sufficiently small
    for _ in range(n):
        val = int(next(it))
        if val > max_a:
            max_a = val
    max_b = -10**18
    for _ in range(n):
        val = int(next(it))
        if val > max_b:
            max_b = val
    print(max_a + max_b)

if __name__ == "__main__":
    main()
```
