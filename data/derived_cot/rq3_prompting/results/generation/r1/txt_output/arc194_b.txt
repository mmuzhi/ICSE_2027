```python
import sys

class Fenwick:
    def __init__(self, n):
        self.n = n
        self.bit = [0] * (n + 1)

    def add(self, idx, delta):
        while idx <= self.n:
            self.bit[idx] += delta
            idx += idx & -idx

    def sum(self, idx):
        s = 0
        while idx > 0:
            s += self.bit[idx]
            idx -= idx & -idx
        return s

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    P = [int(next(it)) for _ in range(N)]

    bit = Fenwick(N)
    ans = 0
    for i, v in enumerate(P):
        c = bit.sum(v)          # number of already seen values <= v
        total = i * (i + 1) // 2
        left = c * (c + 1) // 2
        ans += total - left
        bit.add(v, 1)

    print(ans)

if __name__ == "__main__":
    solve()
```
