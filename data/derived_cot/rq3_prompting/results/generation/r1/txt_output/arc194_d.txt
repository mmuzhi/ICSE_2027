from collections import Counter

MOD = 998244353

def main() -> None:
    sys.setrecursionlimit(10000)
    input = sys.stdin.readline

    N = int(input().strip())
    S = input().strip()

    # Build tree: each node -> list of children indices
    nodes = []          # nodes[i] = list of children
    roots = []          # indices of top-level nodes
    stack = []
    for ch in S:
        if ch == '(':
            idx = len(nodes)
            nodes.append([])
            if stack:
                parent = stack[-1]
                nodes[parent].append(idx)
            else:
                roots.append(idx)
            stack.append(idx)
        else:           # ')'
            stack.pop()

    # Precompute factorials and inverse factorials up to N
    fact = [1] * (N + 1)
    for i in range(1, N + 1):
        fact[i] = fact[i-1] * i % MOD
    inv_fact = [1] * (N + 1)
    inv_fact[N] = pow(fact[N], MOD-2, MOD)
    for i in range(N, 0, -1):
        inv_fact[i-1] = inv_fact[i] * i % MOD

    # DP for each node: (number_of_distinct_subtrees, canonical_signature)
    f_val = [0] * len(nodes)
    sig = [None] * len(nodes)

    def dfs(u: int):
        if sig[u] is not None:
            return f_val[u], sig[u]

        child_data = [dfs(v) for v in nodes[u]]
        # group children by their canonical signature
        cnt = Counter()

        sig[u] = sig_u

    root_data = [dfs(r) for r in roots]
