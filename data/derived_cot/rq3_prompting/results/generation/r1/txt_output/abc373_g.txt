You will be given a competitive programming problem.
There are 2N points P_1,P_2,\ldots,P_N, Q_1,Q_2,\ldots,Q_N on a two-dimensional plane.
The coordinates of P_i are (A_i, B_i), and the coordinates of Q_i are (C_i, D_i).
No three different points lie on the same straight line.
Determine whether there exists a permutation R = (R_1, R_2, \ldots, R_N) of (1, 2, \ldots, N) that satisfies the following condition. If such an R exists, find one.

- For each integer i from 1 through N, let segment i be the line segment connecting P_i and Q_{R_i}.  Then, segment i and segment j (1 \leq  i < j \leq N) never intersect.

Output

If such an R exists, print R_1, R_2, \ldots, R_N separated by spaces. If there are multiple solutions, you may print any of them.

Constraints


- 1 \leq N \leq 300
- 0 \leq A_i, B_i, C_i, D_i \leq 5000 (1 \leq i \leq N)
- (A_i, B_i) \neq (A_j, B_j) (1 \leq i < j \leq N)
- (C_i, D_i) \neq (C_j, D_j) (1 \leq i < j \leq N)
- (A_i, B_i) \neq (C_j, D_j) (1 \leq i, j \leq N)
- No three different points lie on the same straight line.

import sys
sys.setrecursionlimit(10000)

def solve(Ps, Qs):
    # Ps, Qs are lists of indices (0-based)
    n = len(Ps)
    if n == 0:
        return {}
    if n == 1:
        return {Ps[0]: Qs[0]}
    
    # find extreme point (min x, then min y)
    min_coord = (1e9, 1e9)
    extreme_type = None
    extreme_idx = None
    for i in Ps:
        coord = (Px[i], Py[i])
        if coord < min_coord:
            min_coord = coord
            extreme_type = 'P'
            extreme_idx = i
    for j in Qs:
        coord = (Qx[j], Qy[j])
        if coord < min_coord:
            min_coord = coord
            extreme_type = 'Q'
            extreme_idx = j
    
    if extreme_type == 'P':
        x0, y0 = Px[extreme_idx], Py[extreme_idx]
        candidates = Qs
    else:
        x0, y0 = Qx[extreme_idx], Qy[extreme_idx]
        candidates = Ps
    
    # try each candidate
    for y_idx in candidates:
        if extreme_type == 'P':
            x1, y1 = Qx[y_idx], Qy[y_idx]
        else:
            x1, y1 = Px[y_idx], Py[y_idx]
        dx = x1 - x0
        dy = y1 - y0
        
        left_P = []
        left_Q = []
        right_P = []
        right_Q = []
        # iterate over Ps
        for i in Ps:
            if extreme_type == 'P' and i == extreme_idx:
                continue
            cross = dx * (Py[i] - y0) - dy * (Px[i] - x0)
            if cross > 0:
                left_P.append(i)
            else:
                right_P.append(i)
        # iterate over Qs
        for j in Qs:
            if extreme_type == 'Q' and j == extreme_idx:
                continue
            if extreme_type == 'P' and j == y_idx:
                continue
            cross = dx * (Qy[j] - y0) - dy * (Qx[j] - x0)
            if cross > 0:
                left_Q.append(j)
            else:
                right_Q.append(j)
        
        if len(left_P) == len(left_Q):
            # balanced
            left_match = solve(left_P, left_Q)
            right_match = solve(right_P, right_Q)
            match = {}
            if extreme_type == 'P':
                match[extreme_idx] = y_idx
            else:
                match[y_idx] = extreme_idx
            match.update(left_match)
            match.update(right_match)
            return match
    
    # no candidate found
    return None

import sys

sys.setrecursionlimit(1000000)

Px = []
Py = []
Qx = []
Qy = []

def solve(Ps, Qs):
    """Return dict mapping P-index -> Q-index for the matching in this subproblem."""
    n = len(Ps)
    if n == 0:
        return {}
    if n == 1:
        return {Ps[0]: Qs[0]}

    # Find the point with minimum x (then minimum y) among all points.
    min_coord = (10**9, 10**9)
    extreme_type = None   # 'P' or 'Q'
    extreme_idx = None
    for i in Ps:
        coord = (Px[i], Py[i])
        if coord < min_coord:
            min_coord = coord
            extreme_type = 'P'
            extreme_idx = i
    for j in Qs:
        coord = (Qx[j], Qy[j])
        if coord < min_coord:
            min_coord = coord
            extreme_type = 'Q'
            extreme_idx = j

    # Set the extreme point coordinates and the list of candidates (opposite type)
    if extreme_type == 'P':
        x0, y0 = Px[extreme_idx], Py[extreme_idx]
        candidates = Qs
    else:
        x0, y0 = Qx[extreme_idx], Qy[extreme_idx]
        candidates = Ps

    # Try each candidate to find a balanced split
    for y_idx in candidates:
        if extreme_type == 'P':
            x1, y1 = Qx[y_idx], Qy[y_idx]
        else:
            x1, y1 = Px[y_idx], Py[y_idx]

        dx = x1 - x0
        dy = y1 - y0

        left_P = []
        left_Q = []
        right_P = []
        right_Q = []

        # Classify all remaining P points
        for i in Ps:
            if extreme_type == 'P' and i == extreme_idx:
                continue
            cross = dx * (Py[i] - y0) - dy * (Px[i] - x0)
            if cross > 0:
                left_P.append(i)
            else:
                right_P.append(i)

        # Classify all remaining Q points
        for j in Qs:
            if extreme_type == 'Q' and j == extreme_idx:
                continue
            if extreme_type == 'P' and j == y_idx:
                continue
            cross = dx * (Qy[j] - y0) - dy * (Qx[j] - x0)
            if cross > 0:
                left_Q.append(j)
            else:
                right_Q.append(j)

        # If the left side is balanced, the right side is automatically balanced
        if len(left_P) == len(left_Q):
            left_match = solve(left_P, left_Q)
            right_match = solve(right_P, right_Q)
            match = {}
            if extreme_type == 'P':
                match[extreme_idx] = y_idx
            else:
                match[y_idx] = extreme_idx
            match.update(left_match)
            match.update(right_match)
            return match

    # No candidate worked (should not happen under the problem constraints)
    return None
