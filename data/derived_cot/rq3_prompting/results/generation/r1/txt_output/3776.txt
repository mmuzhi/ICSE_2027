```python
from typing import List

class Solution:
    def minCost(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return nums[0]

        # dp[i][j] for i < j: minimum cost to remove the list
        # [nums[i], nums[j], nums[j+1], ..., nums[n-1]]
        dp = [[0] * n for _ in range(n)]

        # base: last two elements
        for i in range(n - 1):
            dp[i][n - 1] = max(nums[i], nums[n - 1])

        # fill states from right to left
        for j in range(n - 2, -1, -1):
            for i in range(j):  # i < j
                # Option 1: remove first and second (i and j)
                cost1 = max(nums[i], nums[j])
                if j + 2 < n:
                    cost1 += dp[j + 1][j + 2]
                else:
                    cost1 += nums[j + 1]  # only one element left

                # Option 2: remove first and third (i and j+1)
                cost2 = max(nums[i], nums[j + 1])
                if j + 2 < n:
                    cost2 += dp[j][j + 2]
                else:
                    cost2 += nums[j]

                # Option 3: remove second and third (j and j+1)
                cost3 = max(nums[j], nums[j + 1])
                if j + 2 < n:
                    cost3 += dp[i][j + 2]
                else:
                    cost3 += nums[i]

                dp[i][j] = min(cost1, cost2, cost3)

        # initial list is [nums[0], nums[1], ..., nums[n-1]]
        return dp[0][1]
```
