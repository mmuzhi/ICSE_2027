For a non-empty non-negative integer sequence B=(B_1,B_2,\ldots,B_{|B|}), we define its score as follows.

- If the length of B is a multiple of M: (B_1 \oplus B_2 \oplus \dots \oplus B_{|B|})^K
- Otherwise: 0

Here, \oplus represents the bitwise XOR.
Find the sum, modulo 998244353, of the scores of the 2^N-1 non-empty subsequences of A.
What is bitwise XOR? The bitwise XOR of non-negative integers A and B, denoted as A \oplus B, is defined as follows: - In the binary representation of A \oplus B, the digit at position 2^k (k \geq 0) is 1 if exactly one of A and B has a 1 in that position in their binary representations, and 0 otherwise. For example, 3 \oplus 5 = 6 (in binary: 011 \oplus 101 = 110). In general, the XOR of k integers p_1, \dots, p_k is defined as (\cdots ((p_1 \oplus p_2) \oplus p_3) \oplus \cdots \oplus p_k), and it can be proved that this is independent of the order of p_1, \dots, p_k.

Output

import sys
from array import array

MOD = 998244353

def main():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    MAX = 1 << 20          # 2^20 = 1048576

    # frequency of each value (0 .. MAX-1)
    freq = [0] * MAX
    for a in A:
        freq[a] += 1

    # Walsh-Hadamard transform (XOR) on freq – using integer arithmetic
    step = 1
    while step < MAX:
        for i in range(0, MAX, 2 * step):
            for j in range(step):
                u = i + j
                v = i + j + step
                x = freq[u]
                y = freq[v]
                freq[u] = x + y
                freq[v] = x - y
        step <<= 1

    # array T with v^K modulo MOD
    T = [0] * MAX
    for v in range(MAX):
        T[v] = pow(v, K, MOD)

    # Walsh-Hadamard transform on T – modulo MOD
    step = 1
    while step < MAX:
        for i in range(0, MAX, 2 * step):
            for j in range(step):
                u = i + j
                v = i + j + step
                a = T[u]
                b = T[v]
                T[u] = (a + b) % MOD
                T[v] = (a - b) % MOD
        step <<= 1

    # hist[T̂] for each cnt_plus (0..N)
    sumT_by_cnt = [0] * (N + 1)
    for u in range(MAX):
        cnt = (N + freq[u]) // 2          # cnt_plus(u)
        sumT_by_cnt[cnt] = (sumT_by_cnt[cnt] + T[u]) % MOD

    # DP for A[cnt] and B[cnt] modulo (t^M-1)
    # A[cnt][s] = sum_{q} C(cnt, s+qM)   (mod MOD)
    # B[cnt][s] = sum_{q} (-1)^{s+qM} C(cnt, s+qM)  (mod MOD)
    total_len = (N + 1) * M
    dp_plus = array('I', [0]) * total_len
    dp_minus = array('I', [0]) * total_len
    dp_plus[0] = 1
    dp_minus[0] = 1

    # precompute index offsets for cyclic shift
    prev_idx = [(i - 1) % M for i in range(M)]
    # DP recurrence
    for cnt in range(1, N + 1):
        prev_base = (cnt - 1) * M
        curr_base = cnt * M
        for s in range(M):
            p = dp_plus[prev_base + s]
            q = dp_plus[prev_base + prev_idx[s]]
            dp_plus[curr_base + s] = (p + q) % MOD
            r = dp_minus[prev_base + s]
            s_prev = dp_minus[prev_base + prev_idx[s]]
            dp_minus[curr_base + s] = (r - s_prev) % MOD

    # C_val[cnt] = constant term of (1+t)^{cnt} (1-t)^{N-cnt} modulo (t^M-1)
    C_val = [0] * (N + 1)
    # index for the second factor: (M - s) % M
    rev_idx = [(M - s) % M for s in range(M)]
    for cnt in range(N + 1):
        base_plus = cnt * M
        base_minus = (N - cnt) * M
        s_sum = 0
        for s in range(M):
            idx_b = base_minus + rev_idx[s]
            s_sum = (s_sum + dp_plus[base_plus + s] * dp_minus[idx_b]) % MOD
        C_val[cnt] = s_sum

    # total sum = sum_{cnt} C_val[cnt] * sumT_by_cnt[cnt]
    total = 0
    for cnt in range(N + 1):
        total = (total + C_val[cnt] * sumT_by_cnt[cnt]) % MOD

    inv_MAX = pow(MAX, MOD - 2, MOD)
    ans = total * inv_MAX % MOD
    print(ans)

if __name__ == "__main__":
    main()
