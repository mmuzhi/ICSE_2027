import sys
import heapq

def solve() -> None:
    data = sys.stdin.read().split()
    it = iter(data)
    H = int(next(it))
    W = int(next(it))
    X = int(next(it))
    P = int(next(it))
    Q = int(next(it))
    # Convert to 0-indexed
    P -= 1
    Q -= 1
    grid = [[0]*W for _ in range(H)]
    for i in range(H):
        for j in range(W):
            grid[i][j] = int(next(it))

    absorbed = [[False]*W for _ in range(H)]
    in_heap = [[False]*W for _ in range(H)]

    total = grid[P][Q]
    absorbed[P][Q] = True

    heap = []
    # directions: up, down, left, right
    dirs = [(-1,0),(1,0),(0,-1),(0,1)]

    # push initial neighbours
    for dr, dc in dirs:
        nr, nc = P+dr, Q+dc
        if 0 <= nr < H and 0 <= nc < W and not absorbed[nr][nc] and not in_heap[nr][nc]:
            heapq.heappush(heap, (grid[nr][nc], nr, nc))
            in_heap[nr][nc] = True

    while heap:
        s, i, j = heap[0]  # peek smallest strength
        if total > X * s:
            heapq.heappop(heap)
            # absorb
            total += s
            absorbed[i][j] = True
            # add neighbours
            for dr, dc in dirs:
                ni, nj = i+dr, j+dc
                if 0 <= ni < H and 0 <= nj < W and not absorbed[ni][nj] and not in_heap[ni][nj]:
                    heapq.heappush(heap, (grid[ni][nj], ni, nj))
                    in_heap[ni][nj] = True
        else:
            # no frontier cell can be absorbed
            break

    print(total)

if __name__ == "__main__":
    solve()
