Input: points = [[1,1],[1,3],[3,1],[3,3]]

```python
from typing import List
import itertools

class Solution:
    def maxRectangleArea(self, points: List[List[int]]) -> int:
        # Convert to set of tuples for fast membership
        point_set = set(tuple(p) for p in points)
        max_area = -1
        
        # Enumerate all combinations of 4 points
        for combo in itertools.combinations(points, 4):
            pts = [tuple(p) for p in combo]
            xs = sorted(set(p[0] for p in pts))
            ys = sorted(set(p[1] for p in pts))
            
            # Must have exactly two distinct x and two distinct y
            if len(xs) != 2 or len(ys) != 2:
                continue
            
            x1, x2 = xs[0], xs[1]
            y1, y2 = ys[0], ys[1]
            
            # The four points must be the four corners of the rectangle
            required = {(x1, y1), (x1, y2), (x2, y1), (x2, y2)}
            if set(pts) != required:
                continue
            
            # Check that no other point lies inside or on the border
            ok = True
            for p in points:
                tp = tuple(p)
                if tp in required:
                    continue
                x, y = tp
                if x1 <= x <= x2 and y1 <= y <= y2:
                    ok = False
                    break
            
            if ok:
                area = (x2 - x1) * (y2 - y1)
                if area > max_area:
                    max_area = area
        
        return max_area
```
