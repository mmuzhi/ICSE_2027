```python
from typing import List

MOD = 10**9 + 7

class Solution:
    def sumOfGoodSubsequences(self, nums: List[int]) -> int:
        # count[v] = number of good subsequences ending with value v (so far)
        # sum_[v] = total sum of elements of those subsequences
        count = {}
        sum_ = {}
        ans = 0
        
        for v in nums:
            # number of subsequences that can be extended from v-1 and v+1
            prev_cnt = count.get(v - 1, 0) + count.get(v + 1, 0)
            prev_sum = sum_.get(v - 1, 0) + sum_.get(v + 1, 0)
            
            # new subsequences ending with v
            new_cnt = (1 + prev_cnt) % MOD
            # sum = v (single) + (prev_sum + prev_cnt * v) for extensions
            new_sum = (v + prev_sum + prev_cnt * v) % MOD
            
            ans = (ans + new_sum) % MOD
            
            # update global data for value v
            count[v] = (count.get(v, 0) + new_cnt) % MOD
            sum_[v] = (sum_.get(v, 0) + new_sum) % MOD
        
        return ans
```
