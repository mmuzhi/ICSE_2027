class Solution:
    def zigzagTraversal(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        zigzag = []
        for i in range(m):
            if i % 2 == 0:

```python
from typing import List

class Solution:
    def zigzagTraversal(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        zigzag = []
        for i in range(m):
            if i % 2 == 0:
                # left to right
                for j in range(n):
                    zigzag.append(grid[i][j])
            else:
                # right to left
                for j in range(n-1, -1, -1):
                    zigzag.append(grid[i][j])
        # skip every alternate cell: take only indices 0, 2, 4, ...
        return zigzag[::2]
```
