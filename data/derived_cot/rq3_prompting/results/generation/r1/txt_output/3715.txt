```python
from typing import List
import bisect

class Solution:
    def maximumCoins(self, coins: List[List[int]], k: int) -> int:
        # sort intervals by left coordinate
        coins.sort(key=lambda x: x[0])
        n = len(coins)
        l_list = [c[0] for c in coins]
        r_list = [c[1] for c in coins]
        c_list = [c[2] for c in coins]
        # prefix sums of total coins in each interval (full interval)
        prefix = [0] * n
        total = 0
        for i in range(n):
            length = r_list[i] - l_list[i] + 1
            total += length * c_list[i]
            prefix[i] = total
        
        # function to compute sum from L to R inclusive
        def range_sum(L: int, R: int) -> int:
            if L > R:
                return 0
            # first interval with r_i >= L
            idx_l = bisect.bisect_left(r_list, L)
            if idx_l == n:
                return 0
            # last interval with l_i <= R
            idx_r = bisect.bisect_right(l_list, R) - 1
            if idx_r < 0 or idx_l > idx_r:
                return 0
            # sum of full intervals from idx_l to idx_r
            res = prefix[idx_r] - (prefix[idx_l-1] if idx_l>0 else 0)
            # adjust first interval
            if L > l_list[idx_l]:
                res -= c_list[idx_l] * (L - l_list[idx_l])
            # adjust last interval
            if R < r_list[idx_r]:
                res -= c_list[idx_r] * (r_list[idx_r] - R)
            return res
        
        candidates = set()
        for i in range(n):
            l = l_list[i]
            r = r_list[i]
            candidates.add(l)

            candidates.add(l - k)
```

```python
        r_list = [c[1] for c in coins]
        c_list = [c[2] for c in coins]

        # prefix sums of total coins in each interval (full interval)
        prefix = [0] * n
        total = 0
        for i in range(n):
            length = r_list[i] - l_list[i] + 1
            total += length * c_list[i]
            prefix[i] = total

        # function to compute sum from L to R inclusive
        def range_sum(L: int, R: int) -> int:
            if L > R:
                return 0
            # first interval whose right end >= L
            idx_l = bisect.bisect_left(r_list, L)
            if idx_l == n:
                return 0
            # last interval whose left end <= R
            idx_r = bisect.bisect_right(l_list, R) - 1
            if idx_r < 0 or idx_l > idx_r:
                return 0

            # sum of full intervals from idx_l to idx_r
            res = prefix[idx_r] - (prefix[idx_l-1] if idx_l > 0 else 0)

            # adjust first interval if L is inside it
            if L > l_list[idx_l]:
                res -= c_list[idx_l] * (L - l_list[idx_l])
            # adjust last interval if R is inside it
            if R < r_list[idx_r]:
                res -= c_list[idx_r] * (r_list[idx_r] - R)

            return res

        # generate all candidate starting positions
        candidates = set()
        for i in range(n):
            l = l_list[i]
            r = r_list[i]
            candidates.add(l)
            candidates.add(r + 1)
            candidates.add(l - k)
            candidates.add(r + 1 - k)

        # evaluate each candidate
        ans = 0
        for s in candidates:
            cur = range_sum(s, s + k - 1)
            if cur > ans:
                ans = cur

        return ans
```
