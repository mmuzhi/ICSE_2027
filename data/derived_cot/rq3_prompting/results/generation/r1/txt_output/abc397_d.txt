import sys
import math

def integer_cube_root(n):
    lo = 1
    hi = 10**6 + 2  # ensure hi**3 > n for n<=1e18
    while lo <= hi:
        mid = (lo+hi)//2
        if mid**3 <= n:
            lo = mid+1
        else:
            hi = mid-1
    return hi

def solve():
    N = int(sys.stdin.readline().strip())
    max_d = integer_cube_root(N)
    for d in range(1, max_d+1):
        if N % d != 0:
            continue
        M = N // d
        T = 4 * M - d * d
        if T <= 0:
            continue
        if T % 3 != 0:
            continue
        K = T // 3
        k = math.isqrt(K)
        if k * k != K:
            continue
        if (k - d) % 2 != 0:
            continue
        y = (k - d) // 2
        if y <= 0:
            continue
        x = y + d
        print(x, y)
        return
    print(-1)

if __name__ == "__main__":
    solve()

import sys
import math

def integer_cube_root(n: int) -> int:
    """Return the largest integer d such that d**3 <= n."""
    lo, hi = 1, 10**6 + 2  # 10**6**3 == 10**18, so hi is safe
    while lo <= hi:
        mid = (lo + hi) // 2
        if mid ** 3 <= n:
            lo = mid + 1
        else:
            hi = mid - 1
    return hi

def solve() -> None:
    data = sys.stdin.readline().strip()
    if not data:
        return
    N = int(data)
    max_d = integer_cube_root(N)
    for d in range(1, max_d + 1):
        if N % d != 0:
            continue
        M = N // d
        T = 4 * M - d * d
        if T <= 0:
            continue
        if T % 3 != 0:
            continue
        K = T // 3
        k = math.isqrt(K)
        if k * k != K:
            continue
        if (k - d) % 2 != 0:
            continue
        y = (k - d) // 2
        if y <= 0:
            continue
        x = y + d
        print(x, y)
        return
    print(-1)

if __name__ == "__main__":
    solve()
