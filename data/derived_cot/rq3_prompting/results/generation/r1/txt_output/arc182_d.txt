N M

- 2 \leq M \leq 10^6
- 0\leq A_i,B_i< M(1\leq i\leq N)
- A_i\ne A_{i+1}(1\leq i\leq N-1)
- B_i\ne B_{i+1}(1\leq i\leq N-1)
- All input values are integers.

```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    B = [int(next(it)) for _ in range(N)]

    # special case M = 2: no move is ever possible
    if M == 2:
        print(0 if A == B else -1)
        return

    # clockwise gaps (1 .. M-1) for A and B
    g = [(A[i+1] - A[i]) % M for i in range(N-1)]
    h = [(B[i+1] - B[i]) % M for i in range(N-1)]

    # integer lifts: start with the first element, add gaps
    liftA = [0] * N
    liftB = [0] * N
    liftA[0] = A[0]
    liftB[0] = B[0]
    for i in range(1, N):
        liftA[i] = liftA[i-1] + g[i-1]
        liftB[i] = liftB[i-1] + h[i-1]

    # differences between target and initial lifts
    diff = [liftB[i] - liftA[i] for i in range(N)]
    diff.sort()
    median = diff[N // 2]

    # best integer multiple of M to subtract
    t0 = median // M
    t1 = t0 + 1 if median % M != 0 else t0

    def total(t: int) -> int:
        return sum(abs(x - t * M) for x in diff)

    ans = min(total(t0), total(t1))
    # small safety net (convexity guarantees t0 or t1 is optimal)
    ans = min(ans, total(t0 - 1), total(t1 + 1))

    print(ans)

if __name__ == "__main__":
    solve()
```
