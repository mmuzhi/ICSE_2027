class Solution:
    def permute(self, n: int, k: int) -> List[int]:
        INF = 10**15 + 1
        # precompute factorials up to n (or up to 50, but n is fine)
        fact = [1] * (n + 1)
        for i in range(2, n+1):
            fact[i] = min(fact[i-1] * i, INF)
        
        # helper: count alternating permutations of odd odds and even evens with given first parity
        def count(odd: int, even: int, first_parity: int) -> int:
            L = odd + even
            if first_parity == 1: # odd first
                need_odd = (L + 1) // 2
                need_even = L // 2
            else: # even first
                need_odd = L // 2
                need_even = (L + 1) // 2
            if odd == need_odd and even == need_even:
                # product may be huge, cap
                prod = fact[odd] * fact[even]
                return prod if prod <= INF else INF
            else:
                return 0
        
        # initial counts
        odd_cnt = (n + 1) // 2  # number of odd numbers in 1..n
        even_cnt = n // 2
        
        # total number of alternating permutations
        total = 0
        if odd_cnt > 0:
            total += odd_cnt * count(odd_cnt-1, even_cnt, 0)  # start with odd, suffix even first
        if even_cnt > 0:
            total += even_cnt * count(odd_cnt, even_cnt-1, 1)  # start with even, suffix odd first
        
        if k > total:
            return []
        
        # lists of remaining numbers
        odds = list(range(1, n+1, 2))   # odd numbers in increasing order
        evens = list(range(2, n+1, 2)) # even numbers
        
        res = []
        last_parity = None  # no restriction initially
        
        # we will update odd_cnt, even_cnt as we go
        for pos in range(n):
            if pos == 0:
                # first position: consider all numbers in order
                i = 0  # index in odds
                j = 0  # index in evens

                            break

                            last_parity = 0
                            evens.pop(j)
                            even_cnt -= 1
                            break

from typing import List

class Solution:
    def permute(self, n: int, k: int) -> List[int]:
        INF = 10**15 + 1

        # precompute factorials up to n, capped at INF
        fact = [1] * (n + 1)
        for i in range(2, n + 1):
            fact[i] = min(fact[i-1] * i, INF)

        # count alternating permutations of odd odd numbers and even even numbers
        # with first_parity = 1 (odd) or 0 (even)
        def count(odd: int, even: int, first_parity: int) -> int:
            L = odd + even
            if first_parity == 1:          # odd first
                need_odd = (L + 1) // 2
                need_even = L // 2
            else:                          # even first
                need_odd = L // 2
                need_even = (L + 1) // 2
            if odd == need_odd and even == need_even:
                prod = fact[odd] * fact[even]
                return prod if prod <= INF else INF
            return 0

        odd_cnt = (n + 1) // 2   # number of odd numbers in 1..n
        even_cnt = n // 2        # number of even numbers

        # total number of alternating permutations
        total = 0
        if odd_cnt:
            total += odd_cnt * count(odd_cnt-1, even_cnt, 0)
        if even_cnt:
            total += even_cnt * count(odd_cnt, even_cnt-1, 1)

        if k > total:
            return []

        # sorted lists of remaining numbers
        odds = list(range(1, n+1, 2))
        evens = list(range(2, n+1, 2))

        res = []
        last_parity = None

        for pos in range(n):
            if pos == 0:
                # first position: all numbers in increasing order
                i = 0  # index in odds
                j = 0  # index in evens
                while i < len(odds) or j < len(evens):
                    # pick the smaller candidate
                    if j == len(evens) or (i < len(odds) and odds[i] < evens[j]):
                        v = odds[i]
                        cnt = count(odd_cnt-1, even_cnt, 0)   # after odd -> next even
                        if k > cnt:
                            k -= cnt
                            i += 1
                        else:
                            res.append(v)
                            last_parity = 1
                            odds.pop(i)
                            odd_cnt -= 1
                            break
                    else:
                        v = evens[j]
                        cnt = count(odd_cnt, even_cnt-1, 1)   # after even -> next odd
                        if k > cnt:
                            k -= cnt
                            j += 1
                        else:
                            res.append(v)
                            last_parity = 0
                            evens.pop(j)
                            even_cnt -= 1
                            break
            else:
                # subsequent positions: must be opposite of last placed
                need = 1 - last_parity
                if need == 1:   # need an odd number
                    for idx, v in enumerate(odds):
                        cnt = count(odd_cnt-1, even_cnt, 0)   # after odd -> next even
                        if k > cnt:
                            k -= cnt
                        else:
                            res.append(v)
                            last_parity = 1
                            odds.pop(idx)
                            odd_cnt -= 1
                            break
                else:           # need an even number
                    for idx, v in enumerate(evens):
                        cnt = count(odd_cnt, even_cnt-1, 1)   # after even -> next odd
                        if k > cnt:
                            k -= cnt
                        else:
                            res.append(v)
                            last_parity = 0
                            evens.pop(idx)
                            even_cnt -= 1
                            break

        return res
