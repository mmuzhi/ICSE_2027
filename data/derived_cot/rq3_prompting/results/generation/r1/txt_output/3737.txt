from typing import List

class Solution:
    def minCost(self, n: int, cost: List[List[int]]) -> int:
        # number of pairs = n // 2
        pairs = n // 2
        INF = 10**18
        
        # dp[cl][cr] = min cost for processed pairs so far,
        # with left house of current pair colored cl, right house colored cr.
        dp = [[INF] * 3 for _ in range(3)]
        
        # Initialize first pair (i = 0)
        left_idx = 0
        right_idx = n - 1
        for cl in range(3):
            for cr in range(3):
                if cl != cr:
                    dp[cl][cr] = cost[left_idx][cl] + cost[right_idx][cr]
        
        # Process remaining pairs
        for i in range(1, pairs):
            left_idx = i
            right_idx = n - 1 - i
            new_dp = [[INF] * 3 for _ in range(3)]
            
            for cl in range(3):
                for cr in range(3):
                    prev_val = dp[cl][cr]
                    if prev_val == INF:
                        continue
                    for cl2 in range(3):
                        for cr2 in range(3):
                            # Symmetric constraint: left and right of the same pair must differ
                            if cl2 == cr2:
                                continue
                            # Adjacent constraints: colors must differ with the previous houses
                            if cl2 == cl or cr2 == cr:
                                continue
                            new_val = prev_val + cost[left_idx][cl2] + cost[right_idx][cr2]
                            if new_val < new_dp[cl2][cr2]:
                                new_dp[cl2][cr2] = new_val
            dp = new_dp
        
        # Answer is the minimum among all states of the last pair
        ans = INF
        for cl in range(3):
            for cr in range(3):
                if dp[cl][cr] < ans:
                    ans = dp[cl][cr]
        return ans
