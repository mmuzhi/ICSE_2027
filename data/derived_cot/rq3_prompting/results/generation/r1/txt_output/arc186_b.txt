        a = int(next(it))
        A[i] = a
        if a > 0:
            children[a].append(i)   # children are in increasing order

    # Precompute factorials and inverse factorials up to N
    fact = [1] * (N + 1)
    for i in range(1, N + 1):
        fact[i] = fact[i - 1] * i % MOD
    inv_fact = [1] * (N + 1)
    inv_fact[N] = pow(fact[N], MOD - 2, MOD)
    for i in range(N, 0, -1):
        inv_fact[i - 1] = inv_fact[i] * i % MOD

    def comb(n, k):
        if k < 0 or k > n:
            return 0
        return fact[n] * inv_fact[k] % MOD * inv_fact[n - k] % MOD

    sz = [0] * (N + 1)
    T = [0] * (N + 1)

    # Process nodes in decreasing order (children have larger indices)
    for i in range(N, 0, -1):
        sz[i] = 1
        for c in children[i]:
            sz[i] += sz[c]

        if not children[i]:
            T[i] = 1
        else:
            cum = 0
            prod = 1
            for c in children[i]:
                cum += sz[c]
                prod = prod * comb(cum - 1, sz[c] - 1) % MOD
            for c in children[i]:
                prod = prod * T[c] % MOD
            T[i] = prod

    # Combine the forest
    roots = [i for i in range(1, N + 1) if A[i] == 0]
    cum = 0
    forest_factor = 1
    forest_T = 1
    for r in roots:
        cum += sz[r]
        forest_factor = forest_factor * comb(cum - 1, sz[r] - 1) % MOD
        forest_T = forest_T * T[r] % MOD

    ans = forest_factor * forest_T % MOD
    print(ans)

if __name__ == "__main__":
    main()
