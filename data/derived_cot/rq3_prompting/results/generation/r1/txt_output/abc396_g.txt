import sys

def fwht(arr):
    n = len(arr)
    step = 1
    while step < n:
        for i in range(0, n, step*2):
            for j in range(i, i+step):
                u = arr[j]
                v = arr[j+step]
                arr[j] = u + v
                arr[j+step] = u - v
        step <<= 1

def solve():
    data = sys.stdin.read().split()
    if not data:
        return
    H = int(data[0])
    W = int(data[1])
    # next H strings are in data[2:2+H]
    N = 1 << W
    freq = [0] * N
    idx = 2
    for _ in range(H):
        s = data[idx]
        idx += 1
        mask = 0
        for ch in s:
            mask = (mask << 1) | (1 if ch == '1' else 0)
        freq[mask] += 1
    # build f
    f = [0] * N
    for mask in range(N):
        pc = mask.bit_count()  # Python 3.8+ has int.bit_count()
        f[mask] = min(pc, W - pc)
    a = freq
    b = f[:]  # copy because transform modifies
    fwht(a)
    fwht(b)
    for i in range(N):
        a[i] *= b[i]
    fwht(a)
    # divide by N (which is 1<<W)
    for i in range(N):
        a[i] //= N
    ans = min(a)
    print(ans)

if __name__ == '__main__':
    solve()

import sys

def fwht(arr):
    """
    In-place Fast Walsh–Hadamard Transform for XOR convolution.
    After forward transform, applying it again divides by the array length.
    """
    n = len(arr)
    step = 1
    while step < n:
        for i in range(0, n, step * 2):
            for j in range(i, i + step):
                u = arr[j]
                v = arr[j + step]
                arr[j] = u + v
                arr[j + step] = u - v
        step <<= 1

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    H = int(data[0])
    W = int(data[1])
    N = 1 << W
    freq = [0] * N

    idx = 2
    for _ in range(H):
        s = data[idx]
        idx += 1
        mask = 0
        for ch in s:
            mask = (mask << 1) | (1 if ch == '1' else 0)
        freq[mask] += 1

    # f(d) = min(popcount(d), W - popcount(d))
    f = [0] * N
    for mask in range(N):
        pc = mask.bit_count()
        f[mask] = min(pc, W - pc)

    a = freq
    b = f[:]   # copy, because fwht modifies in-place
    fwht(a)
    fwht(b)
    for i in range(N):
        a[i] *= b[i]
    fwht(a)          # now a holds N times the convolution
    for i in range(N):
        a[i] //= N   # divide to get the actual convolution

    ans = min(a)
    print(ans)

if __name__ == "__main__":
    solve()
