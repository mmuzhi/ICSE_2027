```python
def solve() -> None:
    X = int(input().strip())
    total = 2025  # sum of all 81 entries
    cnt = 0
    for i in range(1, 10):
        if X % i == 0:
            j = X // i
            if 1 <= j <= 9:
                cnt += 1
    ans = total - X * cnt
    print(ans)

if __name__ == "__main__":
    solve()
```
