import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    N = int(data[0])
    P = list(map(int, data[1:1+N]))

    # Fenwick tree (Binary Indexed Tree)
    size = N
    tree = [0] * (size + 1)

    # Initialize tree for all positions being empty (value 1)
    # For an array of all ones, tree[i] = lowbit(i)
    for i in range(1, size + 1):
        tree[i] = i & -i

    # Highest power of two not exceeding size
    bitmask = 1 << (size.bit_length() - 1)

    def bit_add(idx: int, delta: int) -> None:
        """Add delta at position idx (1-indexed)."""
        while idx <= size:
            tree[idx] += delta
            idx += idx & -idx

    def bit_find_kth(k: int) -> int:
        """Return smallest index such that prefix sum >= k (1-indexed)."""
        idx = 0
        bm = bitmask
        while bm:
            nxt = idx + bm
            if nxt <= size and tree[nxt] < k:
                k -= tree[nxt]
                idx = nxt
            bm >>= 1
        return idx + 1

    ans = [0] * N
    for i in range(N, 0, -1):
        pos = bit_find_kth(P[i-1])
        ans[pos-1] = i
        bit_add(pos, -1)

    sys.stdout.write(' '.join(map(str, ans)))

if __name__ == "__main__":
    solve()
