class Solution:
    def maxDistance(self, s: str, k: int) -> int:
        # mapping of characters to (dx, dy)
        dirs = {'N': (0, 1), 'S': (0, -1), 'E': (1, 0), 'W': (-1, 0)}
        
        # helper to compute the maximal prefix sum for a given linear form L = a*x + b*y
        def max_for_orientation(a: int, b: int) -> int:
            cur = 0          # current sum of L values
            neg = 0          # count of -1s seen so far
            best = 0         # best achievable value for this orientation
            for ch in s:
                dx, dy = dirs[ch]
                val = a * dx + b * dy   # either +1 or -1
                if val == -1:
                    neg += 1
                cur += val
                # we can flip up to k of the -1s in this prefix to +1,
                # each flip adds 2 to the sum.
                best = max(best, cur + 2 * min(k, neg))
            return best
        
        # consider all four quadrants
        ans = 0
        ans = max(ans, max_for_orientation(1, 1))   # x+y
        ans = max(ans, max_for_orientation(1, -1))  # x-y
        ans = max(ans, max_for_orientation(-1, 1))  # -x+y
        ans = max(ans, max_for_orientation(-1, -1)) # -x-y
        return ans
