There is an infinitely large two-dimensional grid, with a campfire at coordinate (0,0).
At time t=0, smoke exists only at cell (0,0).
You are given a length-N string S consisting of N, W, S, E. At times t=1,2,\dots,N, the following happen in order:

- Wind blows, and all the smoke present at that time moves as follows:
- If the t-th character of S is N, smoke in cell (r,c) moves to cell (r-1,c).

- If it is S, smoke in cell (r,c) moves to cell (r+1,c).

```python
import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    R = int(data[1])
    C = int(data[2])
    S = data[3].strip()
    
    active = {(0, 0)}          # set of cumulative displacement vectors for active generations
    dr, dc = 0, 0             # current cumulative displacement (D[t])
    out = []
    
    for ch in S:
        # update cumulative displacement
        if ch == 'N':
            dr -= 1
        elif ch == 'S':
            dr += 1
        elif ch == 'W':
            dc -= 1
        else:  # 'E'
            dc += 1
        
        # check if smoke exists at (R,C) after this step
        target_r = dr - R
        target_c = dc - C
        if (target_r, target_c) in active:
            out.append('1')
        else:
            out.append('0')
        
        # if origin is empty, generate new smoke
        if (dr, dc) not in active:
            active.add((dr, dc))
    
    sys.stdout.write(''.join(out) + '\n')

if __name__ == "__main__":
    solve()
```
