=== ANSWER ===
```python
from typing import List

class Solution:
    def maxSubarraySum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # prefix sum array, index i corresponds to sum of first i elements
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i+1] = prefix[i] + nums[i]
        
        # min_prefix[r] = minimum prefix sum seen so far for remainder r
        # Initialize with infinity; remainder 0 has prefix[0]=0
        min_prefix = [float('inf')] * k
        min_prefix[0] = 0
        
        ans = -10**18  # sufficiently small
        
        for j in range(1, n+1):
            r = j % k
            # best sum for subarrays ending at j (exclusive) is prefix[j] - min_prefix[r]
            candidate = prefix[j] - min_prefix[r]
            if candidate > ans:
                ans = candidate
            # update min_prefix for this remainder
            if prefix[j] < min_prefix[r]:
                min_prefix[r] = prefix[j]
        
        return ans
```
