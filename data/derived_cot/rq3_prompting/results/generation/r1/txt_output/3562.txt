from typing import List
import bisect

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
        ext = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        ext.sort(key=lambda x: x[1])  # sort by r

        rs = [x[1] for x in ext]
        unique_rs = sorted(set(rs))
        comp = {v: idx+1 for idx, v in enumerate(unique_rs)}  # 1-indexed
        M = len(unique_rs)

        sentinel = (-1, ())
        # BIT for t=1..4
        bit = [None] * 5  # index 1..4
        for t in range(1,5):
            bit[t] = [sentinel] * (M+2)  # extra space

        best = (0, ())  # weight, tuple

        def combine(s1, s2):
            if s1[0] > s2[0]:
                return s1
            if s2[0] > s1[0]:
                return s2
            # equal weight
            if s1[1] < s2[1]:
                return s1
            else:
                return s2

        def bit_update(bit_arr, idx, state):
            while idx <= M:
                bit_arr[idx] = combine(bit_arr[idx], state)
                idx += idx & -idx

        def bit_query(bit_arr, idx):
            res = sentinel
            while idx > 0:
                res = combine(res, bit_arr[idx])
                idx -= idx & -idx
            return res

        for l, r, w, idx in ext:
            pos = comp[r]
            for t in range(1,5):
                if t == 1:
                    cand = (w, (idx,))
                else:
                    bound = bisect.bisect_left(unique_rs, l)  # number of rs < l
                    if bound == 0:
                        continue
                    prev = bit_query(bit[t-1], bound)
                    if prev[0] == -1:
                        continue
                    new_weight = w + prev[0]
                    # merge tuples
                    # Since sizes are small, use sorted
                    new_tuple = tuple(sorted(prev[1] + (idx,)))
                    cand = (new_weight, new_tuple)
                # update global best
                if cand[0] > best[0] or (cand[0] == best[0] and cand[1] < best[1]):
                    best = cand
                # update BIT[t] at pos
                bit_update(bit[t], pos, cand)

        return list(best[1])

from typing import List
import bisect

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
        # attach original indices
        extended = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        # sort by right endpoint
        extended.sort(key=lambda x: x[1])

        # coordinate compression of right endpoints
        all_rs = [e[1] for e in extended]
        unique_rs = sorted(set(all_rs))
        comp = {v: idx+1 for idx, v in enumerate(unique_rs)}   # 1‑based for BIT
        M = len(unique_rs)

        sentinel = (-1, ())          # (weight, sorted tuple of indices)
        # BIT for each t = 1..4
        bit = [None] * 5
        for t in range(1, 5):
            bit[t] = [sentinel] * (M + 2)

        best = (0, ())               # overall best (weight, tuple)

        # ---- helper functions ----
        def combine(s1, s2):
            """Return the better state: higher weight,
               or if equal weight the lexicographically smaller tuple."""
            if s1[0] > s2[0]:
                return s1
            if s2[0] > s1[0]:
                return s2
            # weights equal → compare tuples
            return s1 if s1[1] < s2[1] else s2

        def bit_update(arr, idx, state):
            while idx <= M:
                arr[idx] = combine(arr[idx], state)
                idx += idx & -idx

        def bit_query(arr, idx):
            res = sentinel
            while idx > 0:
                res = combine(res, arr[idx])
                idx -= idx & -idx
            return res
        # -------------------------

        for l, r, w, idx in extended:
            pos = comp[r]                     # compressed right endpoint

            for t in range(1, 5):
                if t == 1:
                    cand = (w, (idx,))
                else:
                    bound = bisect.bisect_left(unique_rs, l)   # # of r's < l
                    if bound == 0:
                        continue
                    prev = bit_query(bit[t-1], bound)
                    if prev[0] == -1:
                        continue
                    new_weight = w + prev[0]
                    new_tuple = tuple(sorted(prev[1] + (idx,)))
                    cand = (new_weight, new_tuple)

                # update global best
                if cand[0] > best[0] or (cand[0] == best[0] and cand[1] < best[1]):
                    best = cand
                # store this state in the BIT for future queries
                bit_update(bit[t], pos, cand)

        return list(best[1])
