import sys

INF = 10 ** 19  # larger than any possible value

# Fenwick tree for range sum of A
class BIT:
    def __init__(self, n):
        self.n = n
        self.bit = [0] * (n + 2)

    def add(self, i, delta):
        while i <= self.n:
            self.bit[i] += delta
            i += i & -i

    def sum(self, i):
        s = 0
        while i > 0:
            s += self.bit[i]
            i -= i & -i
        return s

    def range_sum(self, l, r):
        return self.sum(r) - self.sum(l - 1)


# Segment tree with lazy propagation, storing min of K_i = T_i - P[i-1]
class SegTree:
    def __init__(self, n, arr):
        self.n = n
        self.min = [0] * (4 * n)
        self.lazy = [0] * (4 * n)
        self._build(1, 1, n, arr)

    def _build(self, node, l, r, arr):
        if l == r:
            self.min[node] = arr[l]
            return
        mid = (l + r) // 2
        self._build(node * 2, l, mid, arr)
        self._build(node * 2 + 1, mid + 1, r, arr)
        self.min[node] = min(self.min[node * 2], self.min[node * 2 + 1])

    def _push(self, node):
        if self.lazy[node]:
            val = self.lazy[node]
            self.min[node * 2] += val
            self.lazy[node * 2] += val
            self.min[node * 2 + 1] += val
            self.lazy[node * 2 + 1] += val
            self.lazy[node] = 0

    def range_add(self, ql, qr, val):
        self._range_add(1, 1, self.n, ql, qr, val)

    def _range_add(self, node, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.min[node] += val
            self.lazy[node] += val
            return
        self._push(node)
        mid = (l + r) // 2
        if ql <= mid:
            self._range_add(node * 2, l, mid, ql, qr, val)
        if qr > mid:
            self._range_add(node * 2 + 1, mid + 1, r, ql, qr, val)
        self.min[node] = min(self.min[node * 2], self.min[node * 2 + 1])

    def point_query(self, idx):
        return self._point_query(1, 1, self.n, idx)

    def _point_query(self, node, l, r, idx):
        if l == r:
            return self.min[node]
        self._push(node)
        mid = (l + r) // 2
        if idx <= mid:
            return self._point_query(node * 2, l, mid, idx)
        else:
            return self._point_query(node * 2 + 1, mid + 1, r, idx)

    def find_first(self, ql, qr, cur):
        return self._find_first(1, 1, self.n, ql, qr, cur)

    def _find_first(self, node, l, r, ql, qr, cur):
        if r < ql or l > qr or self.min[node] > cur:
            return -1
        if l == r:
            return l
        self._push(node)
        mid = (l + r) // 2
        left = self._find_first(node * 2, l, mid, ql, qr, cur)
        if left != -1:
            return left
        return self._find_first(node * 2 + 1, mid + 1, r, ql, qr, cur)


def solve():
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    A = [0] + [next(it) for _ in range(N)]
    B = [0] + [next(it) for _ in range(N)]
    Q = next(it)

    # Fenwick tree for A
    bit = BIT(N)
    for i in range(1, N + 1):
        bit.add(i, A[i])

    # compute initial K values
    K = [0] * (N + 1)
    for i in range(1, N + 1):
        if B[i] == 1:
            T = INF
        else:
            T = A[i] // (B[i] - 1) + 1
        pref = bit.sum(i - 1)  # P[i-1]
        K[i] = T - pref

    seg = SegTree(N, K)

    out_lines = []
    for _ in range(Q):
        typ = next(it)
        if typ == 1:
            i = next(it)
            x = next(it)
            # update A[i] to x
            delta = x - A[i]
            A[i] = x
            bit.add(i, delta)
            # update K[i]
            if B[i] == 1:
                new_T = INF
            else:
                new_T = x // (B[i] - 1) + 1
            old_K = seg.point_query(i)
            new_K = new_T - bit.sum(i - 1)  # P[i-1] unchanged
            seg.range_add(i, i, new_K - old_K)
            # update K[j] for j>i
            if i + 1 <= N:
                seg.range_add(i + 1, N, -delta)
        elif typ == 2:
            i = next(it)
            x = next(it)
            B[i] = x
            # update K[i] only
            if x == 1:
                new_T = INF
            else:
                new_T = A[i] // (x - 1) + 1
            old_K = seg.point_query(i)
            new_K = new_T - bit.sum(i - 1)  # P[i-1] unchanged
            seg.range_add(i, i, new_K - old_K)
        else:  # typ == 3
            l = next(it)
            r = next(it)
            pos = l
            v = 0
            cur = -bit.sum(pos - 1)  # v - P[pos-1] = -P[l-1]
            while pos <= r:
                i = seg.find_first(pos, r, cur)
                if i == -1:
                    # add all remaining
                    v += bit.range_sum(pos, r)
                    break
                else:
                    # add from pos to i-1
                    if i > pos:
                        add_sum = bit.range_sum(pos, i - 1)
                        v += add_sum
                    # multiply at i
                    v = v * B[i]
                    # update cur and pos
                    cur = v - bit.sum(i)  # P[i]
                    pos = i + 1
            out_lines.append(str(v))

    sys.stdout.write("\n".join(out_lines))

    solve()
