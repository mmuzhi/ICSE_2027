```python
from typing import List
from collections import defaultdict, Counter

class Solution:
    def subsequencesWithMiddleMode(self, nums: List[int]) -> int:
        MOD = 10**9 + 7
        n = len(nums)
        # helper: binomial coefficient C(x,2)
        def c2(x):
            return x * (x - 1) // 2
        
        left_cnt = defaultdict(int)          # counts of values left of current index
        right_cnt = Counter(nums)           # counts of values right of current index (including current)
        ans = 0
        
        for i in range(n):
            x = nums[i]
            # remove current element from right side
            right_cnt[x] -= 1
            if right_cnt[x] == 0:
                del right_cnt[x]
            
            left_len = i
            right_len = n - i - 1
            Lx = left_cnt.get(x, 0)
            Rx = right_cnt.get(x, 0)
            
            # collect all values different from x that appear in left or right
            all_keys = set(left_cnt.keys()) | set(right_cnt.keys())
            if x in all_keys:
                all_keys.remove(x)
            
            # accumulate sums over all y != x
            sum_l = sum_r = 0
            sum_l2 = sum_r2 = 0
            sum_lr = sum_lr2 = sum_l2r = 0
            for y in all_keys:
                l = left_cnt.get(y, 0)
                r = right_cnt.get(y, 0)
                sum_l += l
                sum_r += r
                sum_l2 += l * l
                sum_r2 += r * r
                sum_lr += l * r
                sum_lr2 += l * r * r
                sum_l2r += l * l * r
            
            N_left = sum_l
            N_right = sum_r
            
            # pattern 5x
            c5 = c2(Lx) * c2(Rx)
            
            # pattern 4x
            c4 = c2(Lx) * Rx * N_right + Lx * N_left * c2(Rx)
            
            # pattern 3x
            c3 = c2(Lx) * c2(N_right) + Lx * N_left * Rx * N_right + c2(N_left) * c2(Rx)
            
            # pattern 2x: need three non-x all distinct
            S2_right = (sum_r2 - N_right) // 2
            S2_left = (sum_l2 - N_left) // 2
            
            term_10 = N_left * c2(N_right) - (N_right * sum_lr - sum_lr2 + N_left * S2_right)
            valid_10 = Lx * term_10
            
            term_01 = N_right * c2(N_left) - (N_left * sum_lr - sum_l2r + N_right * S2_left)
            valid_01 = Rx * term_01
            
            c2_val = valid_10 + valid_01
            
            ans = (ans + c5 + c4 + c3 + c2_val) % MOD
            
            # add current element to left side for next iteration
            left_cnt[x] += 1
        
        return ans % MOD
```
