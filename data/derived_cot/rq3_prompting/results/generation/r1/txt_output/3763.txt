class Solution:

from typing import List

class Solution:
    def separateSquares(self, squares: List[List[int]]) -> float:
        """
        Finds the minimum y-coordinate of a horizontal line that splits the total
        area of the squares into two equal halves.
        """
        total_area = 0
        min_y = float('inf')
        max_y_top = 0

        # Compute total area and the vertical range
        for _, y, l in squares:
            total_area += l * l
            if y < min_y:
                min_y = y
            if y + l > max_y_top:
                max_y_top = y + l

        target = total_area / 2.0

        lo = float(min_y)
        hi = float(max_y_top)

        # Binary search for the point where area_below reaches target
        for _ in range(60):  # enough for 1e-5 precision
            mid = (lo + hi) / 2.0
            below = 0.0

            for _, y, l in squares:
                if mid <= y:
                    continue
                if mid >= y + l:
                    below += l * l
                else:
                    below += l * (mid - y)

            if below < target:
                lo = mid
            else:
                hi = mid

        return (lo + hi) / 2.0
