from typing import List
from collections import deque

class Solution:
    def maxMoves(self, kx: int, ky: int, positions: List[List[int]]) -> int:
        # number of pawns
        n = len(positions)
        # all important points: index 0 = start, 1..n = pawns
        points = [(kx, ky)] + [tuple(p) for p in positions]
        
        # Knight moves
        moves = [(1,2),(1,-2),(-1,2),(-1,-2),(2,1),(2,-1),(-2,1),(-2,-1)]
        
        # Precompute distances between every pair of points
        # dist[i][j] = knight distance from points[i] to points[j]
        N = n + 1
        dist = [[0]*N for _ in range(N)]
        
        for i in range(N):
            sx, sy = points[i]
            # BFS on the 50x50 board
            visited = [[-1]*50 for _ in range(50)]
            q = deque()
            visited[sx][sy] = 0
            q.append((sx, sy))
            while q:
                x, y = q.popleft()
                for dx, dy in moves:
                    nx, ny = x+dx, y+dy
                    if 0 <= nx < 50 and 0 <= ny < 50 and visited[nx][ny] == -1:
                        visited[nx][ny] = visited[x][y] + 1
                        q.append((nx, ny))
            # Fill distances to all other points
            for j in range(N):
                tx, ty = points[j]
                dist[i][j] = visited[tx][ty]
        
        # DP: mask of remaining pawns (bit i-1 corresponds to pawn i), current knight position index
        dp = [[-1] * N for _ in range(1 << n)]
        
        def dfs(mask: int, pos_idx: int) -> int:
            # No pawns left
            if mask == 0:
                return 0
            if dp[mask][pos_idx] != -1:
                return dp[mask][pos_idx]
            
            r = mask.bit_count()          # number of remaining pawns
            captured = n - r
            # Alice's turn if captured is even, Bob's turn if odd
            turn = captured % 2           # 0 -> Alice (maximizer), 1 -> Bob (minimizer)
            
            best = None
            # iterate over pawns (indices 1..n)
            for p in range(1, n+1):
                if mask & (1 << (p-1)):
                    cost = dist[pos_idx][p]
                    next_mask = mask ^ (1 << (p-1))
                    total = cost + dfs(next_mask, p)
                    if best is None:
                        best = total
                    elif turn == 0:       # Alice maximizes
                        if total > best:
                            best = total
                    else:                 # Bob minimizes
                        if total < best:
                            best = total
            # According to constraints, there is always at least one pawn when mask>0, so best is set
            dp[mask][pos_idx] = best
            return best
        
        full_mask = (1 << n) - 1
        return dfs(full_mask, 0)
