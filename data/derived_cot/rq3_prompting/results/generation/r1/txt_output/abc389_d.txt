def solve():
    import sys, math
    R = int(sys.stdin.readline().strip())
    R2 = R*R
    total = 0
    i_max = R - 1   # because R integer, i_max = floor(R-0.5) = R-1
    for i in range(i_max+1):
        D = 4*R2 - (2*i+1)*(2*i+1)
        sqrtD = math.isqrt(D)
        j_max = (sqrtD - 1) // 2
        if i == 0:
            total += 1 + 2*j_max
        else:
            total += 2 + 4*j_max
    print(total)

import sys
import math

def solve() -> None:
    data = sys.stdin.readline().strip()
    if not data:
        return
    R = int(data)
    R2 = R * R
    total = 0
    # maximum i such that i+0.5 <= R, i integer
    i_max = R - 1  # because i_max = floor(R - 0.5) = R-1 for integer R
    for i in range(i_max + 1):
        D = 4 * R2 - (2 * i + 1) * (2 * i + 1)
        sqrtD = math.isqrt(D)
        # j_max = max integer j with (2j+1)^2 <= D
        j_max = (sqrtD - 1) // 2
        if i == 0:
            total += 1 + 2 * j_max
        else:
            total += 2 + 4 * j_max
    print(total)

if __name__ == "__main__":
    solve()
