from typing import List

class Solution:
    def maximumValueSum(self, board: List[List[int]]) -> int:
        m = len(board)
        n = len(board[0])
        candidates = []  # list of (value, row, col)

        # For each row, keep the top 3 cells (by value)
        for r in range(m):
            # start with three sentinel entries
            top = [(-10**18, -1)] * 3  # (value, column)
            for c in range(n):
                v = board[r][c]
                # compare with the smallest of the current top 3
                if v > top[2][0]:
                    top[2] = (v, c)
                    # keep top sorted descending by value (bubble up)
                    if top[1][0] < top[2][0]:
                        top[1], top[2] = top[2], top[1]
                    if top[0][0] < top[1][0]:
                        top[0], top[1] = top[1], top[0]
            # add the three candidates (they always exist because n >= 3)
            for v, c in top:
                candidates.append((v, r, c))

        L = len(candidates)
        ans = -10**18

        # Brute force all combinations of three candidates
        for i in range(L):
            vi, ri, ci = candidates[i]
            for j in range(i+1, L):
                vj, rj, cj = candidates[j]
                if ri == rj or ci == cj:
                    continue
                for k in range(j+1, L):
                    vk, rk, ck = candidates[k]
                    if ri == rk or rj == rk or ci == ck or cj == ck:
                        continue
                    total = vi + vj + vk
                    if total > ans:
                        ans = total

        return ans
