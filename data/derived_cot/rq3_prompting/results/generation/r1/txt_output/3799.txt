=== ANSWER ===
```python
from typing import List
from collections import Counter

class Solution:
    def totalNumbers(self, digits: List[int]) -> int:
        # Frequency of given digits
        freq = Counter(digits)
        ans = 0
        # Check all three-digit numbers
        for num in range(100, 1000):
            if num % 2 != 0:   # not even
                continue
            # Count digits of this candidate
            needed = Counter()
            temp = num
            # Extract three digits
            hundreds = temp // 100
            tens = (temp // 10) % 10
            units = temp % 10
            needed[hundreds] += 1
            needed[tens] += 1
            needed[units] += 1
            # Check if the needed digits are available
            possible = True
            for d, cnt in needed.items():
                if freq[d] < cnt:
                    possible = False
                    break
            if possible:
                ans += 1
        return ans
```
