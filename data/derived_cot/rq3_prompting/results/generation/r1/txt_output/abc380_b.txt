=== ANSWER ===
```python

import sys

def main() -> None:
    S = sys.stdin.readline().strip()
    # Split by '|'; the first and last parts are empty strings.
    parts = S.split('|')
    # The runs of '-' are the middle parts.
    lengths = [len(part) for part in parts[1:-1]]
    print(' '.join(map(str, lengths)))

if __name__ == "__main__":
    main()
```
