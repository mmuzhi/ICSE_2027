```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    odd_cnt = sum(1 for a in A if a % 2 == 1)
    if N == 1:
        winner = "Fennec"
    elif N == 2:
        winner = "Snuke"
    elif N == 3:
        winner = "Fennec" if odd_cnt > 0 else "Snuke"
    else:
        winner = "Fennec" if odd_cnt % 2 == 1 else "Snuke"
    print(winner)

if __name__ == "__main__":
    solve()
```
