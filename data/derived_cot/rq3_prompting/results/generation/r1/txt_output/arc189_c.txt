def solve():
    import sys
    sys.setrecursionlimit(1 << 25)
    input = sys.stdin.readline
    N, X = map(int, input().split())
    X -= 1
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    P = list(map(int, input().split()))
    Q = list(map(int, input().split()))
    # convert to 0-index
    P = [p-1 for p in P]
    Q = [q-1 for q in Q]
    # inverse permutations
    invP = [0]*N
    invQ = [0]*N
    for i in range(N):
        invP[P[i]] = i
        invQ[Q[i]] = i
    # compute red distances
    dist_red = [-1]*N
    dist_red[X] = 0
    cur = invP[X]
    d = 1
    while cur != X:
        dist_red[cur] = d
        cur = invP[cur]
        d += 1
    # compute blue distances
    dist_blue = [-1]*N
    dist_blue[X] = 0
    cur = invQ[X]
    d = 1
    while cur != X:
        dist_blue[cur] = d
        cur = invQ[cur]
        d += 1
    # check red feasibility and compute max_red
    max_red = 0
    for i in range(N):
        if A[i] == 1:
            if dist_red[i] == -1:
                print(-1)
                return
            if dist_red[i] > 0:
                max_red = max(max_red, dist_red[i])
    # check blue
    max_blue = 0
    for i in range(N):
        if B[i] == 1:
            if dist_blue[i] == -1:
                print(-1)
                return
            if dist_blue[i] > 0:
                max_blue = max(max_blue, dist_blue[i])
    # determine S
    in_S = [False]*N
    for v in range(N):
        if 1 <= dist_red[v] <= max_red:
            in_S[v] = True
        if 1 <= dist_blue[v] <= max_blue:
            in_S[v] = True
    # collect S nodes
    S_nodes = [i for i in range(N) if in_S[i]]
    M = len(S_nodes)
    if M == 0:
        print(0)
        return
    idx = {v:i for i,v in enumerate(S_nodes)}  # mapping from original index to component index
    adj = [[] for _ in range(M)]
    radj = [[] for _ in range(M)]
    for v in S_nodes:
        u = idx[v]
        # red edge
        w = P[v]
        if in_S[w]:

            adj[u].append(vw)
            radj[vw].append(u)

        if in_S[w]:
            vw = idx[w]
            adj[u].append(vw)
            radj[vw].append(u)

    visited = [False]*M
    order = []
    # first DFS
    for i in range(M):
        if not visited[i]:
            stack = [(i, 0)]  # (node, next neighbor index)
            visited[i] = True
            while stack:
                node, nxt = stack[-1]

                    # increment next for current node
                    stack[-1] = (node, nxt+1)

                        visited[neighbor] = True
                        stack.append((neighbor, 0))
                else:

    cur_comp = 0

        if comp[i] == -1:
            # DFS on radj
            stack = [i]
            comp[i] = cur_comp
            while stack:
                node = stack.pop()
                for nb in radj[node]:
                    if comp[nb] == -1:

                        stack.append(nb)

    comp_size = [0]*cur_comp

    extra = 0

import sys

def solve() -> None:
    input = sys.stdin.readline
    N, X = map(int, input().split())
    X -= 1
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    P = list(map(int, input().split()))
    Q = list(map(int, input().split()))
    # convert to 0-indexed
    P = [p-1 for p in P]
    Q = [q-1 for q in Q]

    # inverse permutations
    invP = [0]*N
    invQ = [0]*N
    for i in range(N):
        invP[P[i]] = i
        invQ[Q[i]] = i

    # ---- compute distances from X along red and blue permutations ----
    dist_red = [-1]*N
    dist_red[X] = 0
    cur = invP[X]
    d = 1
    while cur != X:
        dist_red[cur] = d
        cur = invP[cur]
        d += 1

    dist_blue = [-1]*N
    dist_blue[X] = 0
    cur = invQ[X]
    d = 1
    while cur != X:
        dist_blue[cur] = d
        cur = invQ[cur]
        d += 1

    # ---- feasibility check and maximum distances ----
    max_red = 0
    for i in range(N):
        if A[i] == 1:
            if dist_red[i] == -1:
                print(-1)
                return
            if dist_red[i] > 0:
                max_red = max(max_red, dist_red[i])

    max_blue = 0
    for i in range(N):
        if B[i] == 1:
            if dist_blue[i] == -1:
                print(-1)
                return
            if dist_blue[i] > 0:
                max_blue = max(max_blue, dist_blue[i])

    # ---- construct the sets R and B of nodes that must be operated ----
    red_set = [False]*N
    blue_set = [False]*N
    for v in range(N):
        if 1 <= dist_red[v] <= max_red:
            red_set[v] = True
        if 1 <= dist_blue[v] <= max_blue:
            blue_set[v] = True

    # S = R ∪ B
    in_S = [False]*N
    for v in range(N):
        in_S[v] = red_set[v] or blue_set[v]

    # collect all nodes that must be operated
    S_nodes = [i for i in range(N) if in_S[i]]
    M = len(S_nodes)
    if M == 0:
        print(0)
        return

    # map original vertex -> index in [0..M-1]
    idx = {v: i for i, v in enumerate(S_nodes)}

    # ---- build directed graph of dependencies ----
    adj = [[] for _ in range(M)]   # original edges
    radj = [[] for _ in range(M)]  # reversed edges (for Kosaraju)
    for v in S_nodes:
        u = idx[v]
        # red edge (only if v is on a red path)
        if red_set[v]:
            w = P[v]
            if red_set[w]:
                vw = idx[w]
                adj[u].append(vw)
                radj[vw].append(u)
        # blue edge (only if v is on a blue path)
        if blue_set[v]:
            w = Q[v]
            if blue_set[w]:
                vw = idx[w]
                adj[u].append(vw)
                radj[vw].append(u)

    # ---- Kosaraju's algorithm to find SCCs ----
    visited = [False]*M
    order = []

    # first pass: dfs on original graph to get finishing order
    for i in range(M):
        if not visited[i]:
            stack = [(i, 0)]
            visited[i] = True
            while stack:
                node, nxt = stack[-1]
                if nxt < len(adj[node]):
                    nb = adj[node][nxt]
                    stack[-1] = (node, nxt+1)
                    if not visited[nb]:
                        visited[nb] = True
                        stack.append((nb, 0))
                else:
                    stack.pop()
                    order.append(node)

    # second pass: dfs on reversed graph in reverse order
    comp = [-1]*M
    comp_cnt = 0
    for i in reversed(order):
        if comp[i] == -1:
            stack = [i]
            comp[i] = comp_cnt
            while stack:
                node = stack.pop()
                for nb in radj[node]:
                    if comp[nb] == -1:
                        comp[nb] = comp_cnt
                        stack.append(nb)
            comp_cnt += 1

    # count sizes of SCCs
    comp_size = [0]*comp_cnt
    for c in comp:
        comp_size[c] += 1

    # each nontrivial SCC contributes (size-1) extra operations
    extra = 0
    for sz in comp_size:
        if sz > 1:
            extra += sz - 1

    answer = M + extra
    print(answer)


if __name__ == "__main__":
    solve()
