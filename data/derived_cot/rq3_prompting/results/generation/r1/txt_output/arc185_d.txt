```python
import sys

MOD = 998244353

def solve() -> None:
    data = sys.stdin.readline().strip().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])

    # compute harmonic number H_N modulo MOD
    H = 0
    for i in range(1, N + 1):
        H = (H + pow(i, MOD - 2, MOD)) % MOD

    # ans = (2*N*H - 1) * M^2  (mod MOD)
    ans = (2 * N % MOD * H % MOD - 1) % MOD
    ans = ans * (M % MOD) % MOD
    ans = ans * (M % MOD) % MOD
    print(ans)

if __name__ == "__main__":
    solve()
```
