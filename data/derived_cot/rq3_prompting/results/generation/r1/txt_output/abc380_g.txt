    inv_left = [0] * (N + 1)
    inv_total = 0
    for i in range(1, N + 1):
        v = P[i]
        greater = (i - 1) - bit.sum(v)
        inv_left[i] = greater
        inv_total += greater
        bit.add(v, 1)

    bit = Fenwick(N)
    inv_right = [0] * (N + 1)
    for i in range(N, 0, -1):
        v = P[i]
        lesser = bit.sum(v - 1)
        inv_right[i] = lesser
        bit.add(v, 1)

    # ---------- c[i] ----------
    c = [0] * (N + 1)
    for i in range(1, N + 1):
        if i < K:
            c[i] = i
        elif i > N - K + 1:
            c[i] = N - i + 1
        else:
            c[i] = K

    sum_c_inv = 0
    for i in range(1, N + 1):
        inv_i = inv_left[i] + inv_right[i]
        sum_c_inv += c[i] * inv_i

    # ---------- sum over windows of inversion count ----------
    bit = Fenwick(N)
    inv_win = 0
    for i in range(1, K + 1):
        v = P[i]
        inv_win += (i - 1) - bit.sum(v)
        bit.add(v, 1)
    sum_inv_blocks = inv_win
    for L in range(2, N - K + 2):
        # remove leftmost
        v_out = P[L - 1]
        less = bit.sum(v_out - 1)
        inv_win -= less
        bit.add(v_out, -1)
        # add new rightmost
        v_in = P[L + K - 1]
        greater = (K - 1) - bit.sum(v_in)
        inv_win += greater
        bit.add(v_in, 1)
        sum_inv_blocks += inv_win

    # ---------- compute T = sum_{b} (left_contrib(b) + right_contrib(b)) ----------
    L_min = [0] * (N + 1)
    L_max = [0] * (N + 1)
    for b in range(1, N + 1):
        L_min[b] = max(1, b - K + 1)
        L_max[b] = min(b, N - K + 1)

    # map value -> position
    pos_of_val = [0] * (N + 1)
    for i in range(1, N + 1):
        pos_of_val[P[i]] = i

    T = 0

    # left contributions (process values in decreasing order)
    bit_cnt = Fenwick(N)
    bit_sum = Fenwick(N)
    for v in range(N, 0, -1):
        b = pos_of_val[v]
        lmin = L_min[b]
        lmax = L_max[b]

        cnt1 = bit_cnt.sum(lmin - 1)
        if lmin <= lmax - 1:
            cnt2 = bit_cnt.range_sum(lmin, lmax - 1)
            sum2 = bit_sum.range_sum(lmin, lmax - 1)
        else:
            cnt2 = 0
            sum2 = 0
        contrib = (lmax - lmin + 1) * cnt1 + lmax * cnt2 - sum2
        T += contrib

        bit_cnt.add(b, 1)
        bit_sum.add(b, b)

    # right contributions (process values in increasing order)
    bit_cnt = Fenwick(N)
    bit_sum = Fenwick(N)
    for v in range(1, N + 1):
        b = pos_of_val[v]
        lmin = L_min[b]
        lmax = L_max[b]

        # full interval: j > lmax+K
        j_full_start = lmax + K + 1
        if j_full_start <= N:
            cnt_full = bit_cnt.range_sum(j_full_start, N)
        else:
            cnt_full = 0

        # partial interval: j in [lmin+K, lmax+K] ∩ [1,N]
        j_low = lmin + K
        j_high = min(lmax + K, N)
        cnt_partial = 0
        sum_partial = 0
        if j_low <= j_high:
            cnt_partial = bit_cnt.range_sum(j_low, j_high)
            sum_partial = bit_sum.range_sum(j_low, j_high)

        contrib = (lmax - lmin + 1) * cnt_full + (sum_partial - (lmin + K - 1) * cnt_partial)
        T += contrib

        bit_cnt.add(b, 1)
        bit_sum.add(b, b)

    # ---------- final answer ----------
    inv_total_mod = inv_total % MOD
    sum_c_inv_mod = sum_c_inv % MOD
    sum_inv_blocks_mod = sum_inv_blocks % MOD
    T_mod = T % MOD
    inv_NK = pow(N - K + 1, MOD - 2, MOD)
    inv4 = pow(4, MOD - 2, MOD)
    term_mod = K * (K - 1) % MOD * inv4 % MOD

    ans = (inv_total_mod +
           (-sum_c_inv_mod + sum_inv_blocks_mod + T_mod) % MOD * inv_NK +
           term_mod) % MOD
    print(ans)


if __name__ == "__main__":
    main()
