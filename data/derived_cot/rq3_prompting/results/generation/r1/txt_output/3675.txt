from typing import List
import sys

class Solution:
    def maximizeSumOfWeights(self, edges: List[List[int]], k: int) -> int:
        n = len(edges) + 1
        adj = [[] for _ in range(n)]
        for u, v, w in edges:
            adj[u].append((v, w))
            adj[v].append((u, w))
        
        sys.setrecursionlimit(300000)
        
        dp_k = [0] * n          # max sum in subtree when node can have up to k incident edges
        dp_km1 = [0] * n        # max sum in subtree when node can have up to (k-1) incident edges
        
        def dfs(u: int, parent: int):
            gains = []
            base = 0
            for v, w in adj[u]:
                if v == parent:
                    continue
                dfs(v, u)
                gain = w + dp_km1[v] - dp_k[v]
                gains.append(gain)
                base += dp_k[v]
            # sort gains descending to pick the best ones
            gains.sort(reverse=True)
            
            # for dp_k[u] we can keep at most k children (no parent edge used)
            cnt = min(k, len(gains))
            total = base
            for i in range(cnt):
                if gains[i] > 0:
                    total += gains[i]
            dp_k[u] = total
            
            # for dp_km1[u] we can keep at most k-1 children (one slot reserved for parent edge)
            cnt = min(k - 1, len(gains))
            total = base
            for i in range(cnt):
                if gains[i] > 0:
                    total += gains[i]
            dp_km1[u] = total
        
        dfs(0, -1)
        return dp_k[0]
