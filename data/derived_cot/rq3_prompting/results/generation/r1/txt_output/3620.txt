=== ANSWER ===
```python
from typing import List

class Solution:
    def maxDistinctElements(self, nums: List[int], k: int) -> int:
        nums.sort()
        last_used = -10**18  # very small number
        distinct = 0
        for num in nums:
            left = num - k
            right = num + k
            # smallest integer > last_used and within [left, right]
            candidate = max(last_used + 1, left)
            if candidate <= right:
                distinct += 1
                last_used = candidate
        return distinct
```
