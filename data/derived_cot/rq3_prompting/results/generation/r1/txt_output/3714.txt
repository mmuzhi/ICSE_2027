from typing import List

class Solution:
    def minMaxSums(self, nums: List[int], k: int) -> int:
        MOD = 10**9 + 7
        nums.sort()
        n = len(nums)

        # Precompute factorials and inverse factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i % MOD

        inv_fact = [1] * (n + 1)
        inv_fact[n] = pow(fact[n], MOD-2, MOD)
        for i in range(n, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD

        def nCr(N: int, R: int) -> int:
            if R < 0 or R > N:
                return 0
            return fact[N] * inv_fact[R] % MOD * inv_fact[N-R] % MOD

        ans = 0
        max_t = k - 1  # at most k-1 additional elements

        for i, val in enumerate(nums):
            left_cnt = i
            right_cnt = n - i - 1

            # Count subsequences where nums[i] is the minimum
            t_max = min(max_t, right_cnt)
            sum_min = 0
            for t in range(t_max + 1):
                sum_min = (sum_min + nCr(right_cnt, t)) % MOD

            # Count subsequences where nums[i] is the maximum
            t_max = min(max_t, left_cnt)
            sum_max = 0
            for t in range(t_max + 1):
                sum_max = (sum_max + nCr(left_cnt, t)) % MOD

            ans = (ans + val * (sum_min + sum_max)) % MOD

        return ans
