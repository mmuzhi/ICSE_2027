You will be given a competitive programming problem.
There is a grid with H rows and W columns. Let (i, j) denote the cell at the i-th row from the top and j-th column from the left.
Initially, there is one wall in each cell.
After processing Q queries explained below in the order they are given, find the number of remaining walls.
In the q-th query, you are given two integers R_q and C_q.
You place a bomb at (R_q, C_q) to destroy walls. As a result, the following process occurs.

- If there is a wall at (R_q, C_q), destroy that wall and end the process.
- If there is no wall at (R_q, C_q), destroy the first walls that appear when looking up, down, left, and right from (R_q, C_q). More precisely, the following four processes occur simultaneously:
- If there exists an i \lt R_q such that a wall exists at (i, C_q) and no wall exists at (k, C_q) for all i \lt k \lt R_q, destroy the wall at (i, C_q).
- If there exists an i \gt R_q such that a wall exists at (i, C_q) and no wall exists at (k, C_q) for all R_q \lt k \lt i, destroy the wall at (i, C_q).
- If there exists a j \lt C_q such that a wall exists at (R_q, j) and no wall exists at (R_q, k) for all j \lt k \lt C_q, destroy the wall at (R_q, j).
- If there exists a j \gt C_q such that a wall exists at (R_q, j) and no wall exists at (R_q, k) for all C_q \lt k \lt j, destroy the wall at (R_q, j).

Input

The input is given from Standard Input in the following format:
H W Q
R_1 C_1
R_2 C_2
\vdots
R_Q C_Q

Output

Print the number of remaining walls after processing all queries.

Constraints


- 1 \leq H, W
- H \times W \leq 4 \times 10^5
- 1 \leq Q \leq 2 \times 10^5
- 1 \leq R_q \leq H
- 1 \leq C_q \leq W
- All input values are integers.

1 2
1 3

Sample Output 1

```python
import sys

def solve():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    H = int(next(it))
    W = int(next(it))
    Q = int(next(it))
    queries = [(int(next(it)), int(next(it))) for _ in range(Q)]
    
    total = H * W
    wall = [True] * total
    def idx(r,c): return (r-1)*W + (c-1)
    
    WR = W + 2  # column stride for row arrays
    # allocate row_right and row_left as flat lists
    row_right = [0] * ((H+1)*WR)  # indices: r from 0..H, but we only use 1..H
    row_left = [0] * ((H+1)*WR)
    # initialize
    for r in range(1, H+1):
        base = r * WR
        for c in range(WR):
            row_right[base + c] = c
            row_left[base + c] = c
    # similarly for columns
    HR = H + 2
    col_down = [0] * ((W+1)*HR)
    col_up = [0] * ((W+1)*HR)
    for c in range(1, W+1):
        base = c * HR
        for r in range(HR):
            col_down[base + r] = r
            col_up[base + r] = r
    
    # define find functions using closure to capture globals
    # but we need to modify the arrays; we'll use nested functions
    # We'll define helper functions that access the arrays via index arithmetic.
    
    # To speed, we can inline the loops, but we'll define functions.
    
    sys.setrecursionlimit(1 << 25)
    
    # find right
    def find_right(r, c):
        # c is column, 0..W+1
        stack = []
        base = r * WR
        while True:
            if c == 0 or c == W+1:
                break
            if wall[idx(r,c)]: # alive
                break
            # destroyed, follow pointer
            stack.append((r,c))
            c = row_right[base + c]
        res = c
        for (rr, cc) in stack:
            row_right[rr*WR + cc] = res
        return res
    
    def find_left(r, c):
        stack = []
        base = r * WR
        while True:
            if c == 0 or c == W+1:
                break
            if wall[idx(r,c)]:
                break
            stack.append((r,c))
            c = row_left[base + c]
        res = c
        for (rr, cc) in stack:
            row_left[rr*WR + cc] = res
        return res
    
    def find_down(c, r):
        stack = []
        base = c * HR
        while True:
            if r == 0 or r == H+1:
                break
            if wall[idx(r,c)]:
                break
            stack.append((c,r))
            r = col_down[base + r]
        res = r
        for (cc, rr) in stack:
            col_down[cc*HR + rr] = res
        return res
    
    def find_up(c, r):
        stack = []
        base = c * HR
        while True:
            if r == 0 or r == H+1:
                break
            if wall[idx(r,c)]:
                break
            stack.append((c,r))
            r = col_up[base + r]
        res = r
        for (cc, rr) in stack:
            col_up[cc*HR + rr] = res
        return res
    
    def destroy(r, c):
        nonlocal remaining
        # assume wall present
        pos = idx(r,c)
        wall[pos] = False
        remaining -= 1
        # update DSU for row
        nr = find_right(r, c+1)
        nl = find_left(r, c-1)
        row_right[r*WR + c] = nr
        row_left[r*WR + c] = nl
        # update DSU for column
        nd = find_down(c, r+1)
        nu = find_up(c, r-1)
        col_down[c*HR + r] = nd
        col_up[c*HR + r] = nu
    
    remaining = total
    for R, C in queries:
        if wall[idx(R,C)]:
            destroy(R, C)
        else:
            # up
            ru = find_up(C, R-1)
            if ru != 0:
                destroy(ru, C)
            # down
            rd = find_down(C, R+1)
            if rd != H+1:
                destroy(rd, C)
            # left
            cl = find_left(R, C-1)
            if cl != 0:
                destroy(R, cl)
            # right
            cr = find_right(R, C+1)
            if cr != W+1:
                destroy(R, cr)
    
    print(remaining)

if __name__ == "__main__":
    solve()
```

```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    H = int(next(it))
    W = int(next(it))
    Q = int(next(it))

    queries = [(int(next(it)), int(next(it))) for _ in range(Q)]

    total = H * W
    # 1 if wall present, 0 otherwise
    wall = bytearray([1]) * total

    def idx(r: int, c: int) -> int:
        return (r - 1) * W + (c - 1)

    WR = W + 2                     # column stride for row arrays
    # flat arrays for rows: index = r * WR + c
    row_right = [0] * ((H + 1) * WR)
    row_left  = [0] * ((H + 1) * WR)

    for r in range(1, H + 1):
        base = r * WR
        for c in range(WR):
            row_right[base + c] = c
            row_left[base + c]  = c

    HR = H + 2                     # row stride for column arrays
    col_down = [0] * ((W + 1) * HR)
    col_up   = [0] * ((W + 1) * HR)

    for c in range(1, W + 1):
        base = c * HR
        for r in range(HR):
            col_down[base + r] = r
            col_up[base + r]   = r

    # ---------- find helpers ----------
    def find_right(r: int, c: int) -> int:
        stack = []
        base = r * WR
        while True:
            if c == 0 or c == W + 1:
                break
            if wall[idx(r, c)]:
                break
            stack.append((r, c))
            c = row_right[base + c]
        res = c
        for rr, cc in stack:
            row_right[rr * WR + cc] = res
        return res

    def find_left(r: int, c: int) -> int:
        stack = []
        base = r * WR
        while True:
            if c == 0 or c == W + 1:
                break
            if wall[idx(r, c)]:
                break
            stack.append((r, c))
            c = row_left[base + c]
        res = c
        for rr, cc in stack:
            row_left[rr * WR + cc] = res
        return res

    def find_down(c: int, r: int) -> int:
        stack = []
        base = c * HR
        while True:
            if r == 0 or r == H + 1:
                break
            if wall[idx(r, c)]:
                break
            stack.append((c, r))
            r = col_down[base + r]
        res = r
        for cc, rr in stack:
            col_down[cc * HR + rr] = res
        return res

    def find_up(c: int, r: int) -> int:
        stack = []
        base = c * HR
        while True:
            if r == 0 or r == H + 1:
                break
            if wall[idx(r, c)]:
                break
            stack.append((c, r))
            r = col_up[base + r]
        res = r
        for cc, rr in stack:
            col_up[cc * HR + rr] = res
        return res

    # ---------- destroy one cell ----------
    remaining = total

    def destroy(r: int, c: int) -> None:
        nonlocal remaining
        # wall must be present
        p = idx(r, c)
        wall[p] = 0
        remaining -= 1

        # update row structures
        nr = find_right(r, c + 1)
        nl = find_left(r, c - 1)
        row_right[r * WR + c] = nr
        row_left[r * WR + c]  = nl

        # update column structures
        nd = find_down(c, r + 1)
        nu = find_up(c, r - 1)
        col_down[c * HR + r] = nd
        col_up[c * HR + r]   = nu

    # ---------- process queries ----------
    for R, C in queries:
        if wall[idx(R, C)]:
            destroy(R, C)
        else:
            # up
            ru = find_up(C, R - 1)
            if ru != 0:
                destroy(ru, C)
            # down
            rd = find_down(C, R + 1)
            if rd != H + 1:
                destroy(rd, C)
            # left
            cl = find_left(R, C - 1)
            if cl != 0:
                destroy(R, cl)
            # right
            cr = find_right(R, C + 1)
            if cr != W + 1:
                destroy(R, cr)

    print(remaining)


if __name__ == "__main__":
    solve()
```
