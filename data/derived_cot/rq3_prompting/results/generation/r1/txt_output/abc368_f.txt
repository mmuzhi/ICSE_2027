```python
import sys

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    max_val = max(A) if A else 0

    # smallest prime factor sieve
    spf = list(range(max_val + 1))
    for i in range(2, int(max_val ** 0.5) + 1):
        if spf[i] == i:  # i is prime
            step = i
            start = i * i
            for j in range(start, max_val + 1, step):
                if spf[j] == j:
                    spf[j] = i

    # omega[n] = total number of prime factors (with multiplicity)
    omega = [0] * (max_val + 1)
    for n in range(2, max_val + 1):
        omega[n] = omega[n // spf[n]] + 1

    xor_sum = 0
    for a in A:
        xor_sum ^= omega[a]

    print("Anna" if xor_sum != 0 else "Bruno")

if __name__ == "__main__":
    main()
```
