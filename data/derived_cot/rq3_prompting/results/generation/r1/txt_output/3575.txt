class Solution:
    def maxValue(self, nums: List[int], k: int) -> int:
        n = len(nums)
        # prefix DP: store pref_k[i] = set of OR values achievable with exactly k elements from nums[0..i]
        pref_k = [0]*n
        # dp for counts 0..k
        dp = [0]*(k+1)
        dp[0] = 1  # bit 0 set
        for i in range(n):
            new_dp = dp[:]  # skip
            # include nums[i]
            x = nums[i]
            # iterate c from min(i+1, k) down to 1 (but since we copy from dp, order doesn't matter as long as we use old dp for c-1)
            # We'll iterate c from 1 to min(i+1, k)
            max_c = min(i+1, k)
            for c in range(1, max_c+1):
                prev_set = dp[c-1]
                if prev_set == 0:
                    continue
                # compute new set
                new_set = new_dp[c]
                bits = prev_set
                while bits:
                    low = bits & -bits
                    v = low.bit_length() - 1
                    new_set |= 1 << (v | x)
                    bits ^= low
                new_dp[c] = new_set
            dp = new_dp
            pref_k[i] = dp[k] if k <= i+1 else 0
        # suffix DP
        suff_k = [0]*n
        dp = [0]*(k+1)
        dp[0] = 1
        for i in range(n-1, -1, -1):
            new_dp = dp[:]  # skip
            x = nums[i]
            max_c = min(n-i, k)
            for c in range(1, max_c+1):

```python
from typing import List

class Solution:
    def maxValue(self, nums: List[int], k: int) -> int:
        n = len(nums)

        # ---------- prefix DP ----------
        # pref_k[i] = bitmask of reachable OR values using exactly k elements from nums[0..i]
        pref_k = [0] * n
        dp = [0] * (k + 1)          # dp[c] = bitmask for count c
        dp[0] = 1                   # OR of zero elements is 0

        for i in range(n):
            new_dp = dp[:]          # skip nums[i]
            x = nums[i]
            max_c = min(i + 1, k)
            for c in range(1, max_c + 1):
                prev = dp[c - 1]
                if prev == 0:
                    continue
                cur = new_dp[c]
                bits = prev
                while bits:
                    low = bits & -bits
                    v = low.bit_length() - 1
                    cur |= 1 << (v | x)
                    bits ^= low
                new_dp[c] = cur
            dp = new_dp
            pref_k[i] = dp[k] if i + 1 >= k else 0

        # ---------- suffix DP ----------
        # suff_k[i] = bitmask of reachable OR values using exactly k elements from nums[i..n-1]
        suff_k = [0] * n
        dp = [0] * (k + 1)
        dp[0] = 1

        for i in range(n - 1, -1, -1):
            new_dp = dp[:]          # skip nums[i]
            x = nums[i]
            max_c = min(n - i, k)
            for c in range(1, max_c + 1):
                prev = dp[c - 1]
                if prev == 0:
                    continue
                cur = new_dp[c]
                bits = prev
                while bits:
                    low = bits & -bits
                    v = low.bit_length() - 1
                    cur |= 1 << (v | x)
                    bits ^= low
                new_dp[c] = cur
            dp = new_dp
            suff_k[i] = dp[k] if n - i >= k else 0

        # ---------- combine ----------
        ans = 0
        # p is the last index that can be used for the first k elements
        for p in range(k - 1, n - k):         # n-k-1 inclusive
            left_set = pref_k[p]
            right_set = suff_k[p + 1]
            if left_set == 0 or right_set == 0:
                continue
            # iterate over set bits of left_set
            bits_left = left_set
            while bits_left:
                low_l = bits_left & -bits_left
                L = low_l.bit_length() - 1
                bits_left ^= low_l
                # iterate over set bits of right_set
                bits_right = right_set
                while bits_right:
                    low_r = bits_right & -bits_right
                    R = low_r.bit_length() - 1
                    bits_right ^= low_r
                    ans = max(ans, L ^ R)

        return ans
```
