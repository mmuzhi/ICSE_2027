You will be given a competitive programming problem.
You are given an m x n grid. A robot starts at the top-left corner of the grid (0, 0) and wants to reach the bottom-right corner (m - 1, n - 1). The robot can move either right or down at any point in time.
The grid contains a value coins[i][j] in each cell:

If coins[i][j] >= 0, the robot gains that many coins.
If coins[i][j] < 0, the robot encounters a robber, and the robber steals the absolute value of coins[i][j] coins.

The robot has a special ability to neutralize robbers in at most 2 cells on its path, preventing them from stealing coins in those cells.
Note: The robot's total coins can be negative.
Return the maximum profit the robot can gain on the route.
 
Example 1:

Input: coins = [[0,1,-1],[1,-2,3],[2,-3,4]]
Output: 8
Explanation:
An optimal path for maximum coins is:

Start at (0, 0) with 0 coins (total coins = 0).
Move to (0, 1), gaining 1 coin (total coins = 0 + 1 = 1).
Move to (1, 1), where there's a robber stealing 2 coins. The robot uses one neutralization here, avoiding the robbery (total coins = 1).
Move to (1, 2), gaining 3 coins (total coins = 1 + 3 = 4).
Move to (2, 2), gaining 4 coins (total coins = 4 + 4 = 8).

class Solution:
    def maximumAmount(self, coins: List[List[int]]) -> int:
        m, n = len(coins), len(coins[0])
        INF_NEG = -10**18
        dp = [[[INF_NEG]*3 for _ in range(n)] for _ in range(m)]
        # base
        dp[0][0][0] = coins[0][0]
        dp[0][0][1] = max(coins[0][0], 0)  # neutralize
        # dp[0][0][2] remains -inf
        for i in range(m):
            for j in range(n):
                if i==0 and j==0:
                    continue
                for k in range(3):
                    best = INF_NEG
                    # from top
                    if i>0:
                        # not use
                        if dp[i-1][j][k] > INF_NEG:
                            best = max(best, dp[i-1][j][k] + coins[i][j])
                        # use
                        if k>0 and dp[i-1][j][k-1] > INF_NEG:
                            best = max(best, dp[i-1][j][k-1] + max(0, coins[i][j]))
                    # from left
                    if j>0:
                        if dp[i][j-1][k] > INF_NEG:
                            best = max(best, dp[i][j-1][k] + coins[i][j])
                        if k>0 and dp[i][j-1][k-1] > INF_NEG:
                            best = max(best, dp[i][j-1][k-1] + max(0, coins[i][j]))
                    dp[i][j][k] = best
        return max(dp[m-1][n-1])

```python
from typing import List

class Solution:
    def maximumAmount(self, coins: List[List[int]]) -> int:
        m, n = len(coins), len(coins[0])
        INF_NEG = -10**18
        
        # three 2D arrays for k = 0, 1, 2 neutralizations used
        dp0 = [[INF_NEG] * n for _ in range(m)]
        dp1 = [[INF_NEG] * n for _ in range(m)]
        dp2 = [[INF_NEG] * n for _ in range(m)]
        
        # base case at (0,0)
        dp0[0][0] = coins[0][0]
        dp1[0][0] = max(coins[0][0], 0)   # neutralize if needed
        # dp2[0][0] remains INF_NEG (cannot use two neutralizations on one cell)
        
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    continue
                # ----- k = 0 -----
                best0 = INF_NEG
                if i > 0:
                    best0 = max(best0, dp0[i-1][j] + coins[i][j])
                if j > 0:
                    best0 = max(best0, dp0[i][j-1] + coins[i][j])
                dp0[i][j] = best0
                
                # ----- k = 1 -----
                best1 = INF_NEG
                if i > 0:
                    # not neutralize
                    if dp1[i-1][j] > INF_NEG:
                        best1 = max(best1, dp1[i-1][j] + coins[i][j])
                    # neutralize = use one
                    if dp0[i-1][j] > INF_NEG:
                        best1 = max(best1, dp0[i-1][j] + max(0, coins[i][j]))
                if j > 0:
                    if dp1[i][j-1] > INF_NEG:
                        best1 = max(best1, dp1[i][j-1] + coins[i][j])
                    if dp0[i][j-1] > INF_NEG:
                        best1 = max(best1, dp0[i][j-1] + max(0, coins[i][j]))
                dp1[i][j] = best1
                
                # ----- k = 2 -----
                best2 = INF_NEG
                if i > 0:
                    if dp2[i-1][j] > INF_NEG:
                        best2 = max(best2, dp2[i-1][j] + coins[i][j])
                    if dp1[i-1][j] > INF_NEG:
                        best2 = max(best2, dp1[i-1][j] + max(0, coins[i][j]))
                if j > 0:
                    if dp2[i][j-1] > INF_NEG:
                        best2 = max(best2, dp2[i][j-1] + coins[i][j])
                    if dp1[i][j-1] > INF_NEG:
                        best2 = max(best2, dp1[i][j-1] + max(0, coins[i][j]))
                dp2[i][j] = best2
        
        # answer is the best among the three final states
        return max(dp0[m-1][n-1], dp1[m-1][n-1], dp2[m-1][n-1])
```
