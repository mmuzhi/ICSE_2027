import sys
from collections import deque

def solve() -> None:
    it = iter(sys.stdin.read().split())
    H = int(next(it))
    W = int(next(it))
    S = [list(next(it).strip()) for _ in range(H)]
    A = int(next(it)) - 1
    B = int(next(it)) - 1
    C = int(next(it)) - 1
    D = int(next(it)) - 1

    INF = 10 ** 9
    dist = [[INF] * W for _ in range(H)]
    dq = deque()
    dist[A][B] = 0
    dq.appendleft((0, A, B))

    # directions for 0‑cost moves (only to original roads)
    dir0 = [(1, 0), (-1, 0), (0, 1), (0, -1)]
    # directions for 1‑cost moves (kick up to 2 steps)
    dir1 = [(1, 0), (2, 0), (-1, 0), (-2, 0),
            (0, 1), (0, 2), (0, -1), (0, -2)]

    while dq:
        cost, r, c = dq.popleft()
        if dist[r][c] != cost:
            continue

        # 0‑cost moves: to adjacent original roads
        for dr, dc in dir0:
            nr, nc = r + dr, c + dc
            if 0 <= nr < H and 0 <= nc < W:
                if S[nr][nc] == '.' and dist[nr][nc] > cost:
                    dist[nr][nc] = cost
                    dq.appendleft((cost, nr, nc))

        # 1‑cost moves: kick up to 2 steps in each cardinal direction
        for dr, dc in dir1:
            nr, nc = r + dr, c + dc
            if 0 <= nr < H and 0 <= nc < W:
                if dist[nr][nc] > cost + 1:
                    dist[nr][nc] = cost + 1
                    dq.append((cost + 1, nr, nc))

    print(dist[C][D])

if __name__ == "__main__":
    solve()
