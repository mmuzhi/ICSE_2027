from array import array

class Solution:
    def minCostGoodCaption(self, caption: str) -> str:
        n = len(caption)
        if n < 3:
            return ""

        # convert to numbers 0..25
        s = [ord(ch) - 97 for ch in caption]

        # cost[i][c] = number of operations to change caption[i] to character c
        cost = [bytearray(26) for _ in range(n)]
        for i in range(n):
            a = s[i]
            row = cost[i]
            for c in range(26):
                row[c] = abs(a - c)

        INF = 10**9
        # dp[i][c][l] : flattened: c*3 + l_idx where l_idx 0->1, 1->2, 2->3
        # dp_i stores the minimal cost to process suffix starting at i,
        # given that the run ending at i-1 has character c and length l (capped at 3).
        # dp[n] corresponds to after the last character.
        dp = [None] * (n + 1)
        dp[n] = array('i', [INF] * 78)          # 26*3 = 78
        for c in range(26):
            dp[n][c * 3 + 2] = 0                # l=3 at end is acceptable

        # backward DP
        for i in range(n - 1, -1, -1):
            # pre‑compute best and second best for starting a new run
            min1_val = INF
            min1_ch = -1
            min2_val = INF
            min2_ch = -1
            row_next = dp[i + 1]
            for ch in range(26):
                val = cost[i][ch] + row_next[ch * 3]   # new run -> l=1
                if val < min1_val:
                    min2_val = min1_val
                    min2_ch = min1_ch
                    min1_val = val
                    min1_ch = ch
                elif val < min2_val:
                    min2_val = val
                    min2_ch = ch

            row_cur = array('i', [0] * 78)
            for c in range(26):
                base = c * 3
                # l = 1 (index 0)
                cont = cost[i][c] + row_next[base + 1]   # continue -> l=2
                row_cur[base] = cont
                # l = 2 (index 1)
                cont = cost[i][c] + row_next[base + 2]   # continue -> l=3
                row_cur[base + 1] = cont
                # l = 3 (index 2)
                cont = cost[i][c] + row_next[base + 2]   # continue -> l=3
                # new run allowed only from l=3
                if min1_ch != c:
                    new_val = min1_val
                else:
                    new_val = min2_val
                row_cur[base + 2] = cont if cont < new_val else new_val
            dp[i] = row_cur

        # total minimal cost
        total_min = INF
        for ch in range(26):
            tot = cost[0][ch] + dp[1][ch * 3]   # start with run length 1
            if tot < total_min:
                total_min = tot

        # reconstruct lexicographically smallest optimal string
        res = []
        cur_cost = 0
        # first character
        for ch in range(26):
            tot = cost[0][ch] + dp[1][ch * 3]
            if tot == total_min:
                res.append(chr(ch + 97))
                cur_cost = cost[0][ch]
                prev_c = ch
                prev_l = 1
                break

        # remaining characters
        for i in range(1, n):
            for ch in range(26):
                if ch == prev_c:
                    # continue current run
                    if prev_l == 1:
                        new_l = 2
                        idx = 1
                    else:               # prev_l == 2 or 3
                        new_l = 3
                        idx = 2
                else:
                    # start new run only if current run length >=3
                    if prev_l < 3:
                        continue
                    new_l = 1
                    idx = 0
                tot = cur_cost + cost[i][ch] + dp[i + 1][ch * 3 + idx]
                if tot == total_min:
                    res.append(chr(ch + 97))
                    cur_cost += cost[i][ch]
                    prev_c = ch
                    prev_l = new_l
                    break

        return ''.join(res)
