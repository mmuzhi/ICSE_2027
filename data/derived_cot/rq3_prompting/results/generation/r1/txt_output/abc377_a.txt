=== ANSWER ===
```python
def solve() -> None:
    S = input().strip()
    # Sorting the string and comparing to "ABC"
    if sorted(S) == ['A', 'B', 'C']:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    solve()
```
