```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    A = [int(next(it)) for _ in range(n)]

    ans = n  # all subarrays of length 1

    if n >= 2:
        # compute differences
        D = [A[i+1] - A[i] for i in range(n-1)]
        # scan runs of equal diffs
        cur_val = D[0]
        cur_len = 1
        for val in D[1:]:
            if val == cur_val:
                cur_len += 1
            else:
                ans += cur_len * (cur_len + 1) // 2
                cur_val = val
                cur_len = 1
        ans += cur_len * (cur_len + 1) // 2

    print(ans)

if __name__ == "__main__":
    solve()
```
