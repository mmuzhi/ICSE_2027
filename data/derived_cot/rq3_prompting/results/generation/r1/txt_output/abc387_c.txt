import sys

def snake_upto(x: int) -> int:
    """Return the number of Snake numbers in [1, x]."""
    if x < 10:
        return 0

    s = str(x)
    n = len(s)

    total = 0

    # numbers with fewer digits
    for length in range(2, n):
        for d in range(1, 10):
            total += pow(d, length - 1)

    # numbers with exactly n digits
    first_digit = int(s[0])
    for d in range(1, 10):
        if d > first_digit:
            continue
        if d < first_digit:
            total += pow(d, n - 1)
        else:   # d == first_digit
            # precompute powers of d
            pow_d = [1] * (n + 1)      # pow_d[k] = d^k
            for i in range(1, n + 1):
                pow_d[i] = pow_d[i - 1] * d

            tight = True
            for i in range(1, n):
                xi = int(s[i])
                if tight:
                    # digits y that are < xi and also < d
                    less = min(xi, d)
                    total += less * pow_d[n - i - 1]
                    if xi >= d:
                        tight = False
                        break
            if tight:
                total += 1   # the number x itself is a Snake number

    return total


def main() -> None:
    L, R = map(int, sys.stdin.readline().split())
    ans = snake_upto(R) - snake_upto(L - 1)
    print(ans)


if __name__ == "__main__":
    main()
