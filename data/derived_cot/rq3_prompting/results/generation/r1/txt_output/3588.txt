Alice and Bob are playing a fantasy battle game consisting of n rounds where they summon one of three magical creatures each round: a Fire Dragon, a Water Serpent, or an Earth Golem. In each round, players simultaneously summon their creature and are awarded points as follows:

If one player summons a Fire Dragon and the other summons an Earth Golem, the player who summoned the Fire Dragon is awarded a point.
If one player summons a Water Serpent and the other summons a Fire Dragon, the player who summoned the Water Serpent is awarded a point.
If one player summons an Earth Golem and the other summons a Water Serpent, the player who summoned the Earth Golem is awarded a point.
If both players summon the same creature, no player is awarded a point.

You are given a string s consisting of n characters 'F', 'W', and 'E', representing the sequence of creatures Alice will summon in each round:

class Solution:
    def countWinningSequences(self, s: str) -> int:
        MOD = 10**9+7
        n = len(s)
        offset = n
        size = 2*n + 1
        # map char to index: F->0, W->1, E->2
        # scoring: winner rule: if (b, a) = (F,E): +1; (W,F):+1; (E,W):+1
        # else if (a,b) in that set: -1; else 0.
        # We can precompute delta[3][3] where delta[b][a] = Bob score - Alice score
        # 0:F,1:W,2:E
        beat = {(0,2):1, (1,0):1, (2,1):1}  # b beats a
        delta = [[0]*3 for _ in range(3)]
        for b in range(3):
            for a in range(3):
                if b == a:
                    delta[b][a] = 0
                elif (b,a) in beat:
                    delta[b][a] = 1
                else:
                    delta[b][a] = -1
        # Convert s to list of ints
        sa = [0 if c=='F' else 1 if c=='W' else 2 for c in s]
        # DP arrays
        dp = [[0]*size for _ in range(3)]
        # Initialize first round
        a0 = sa[0]
        for b in range(3):
            d = offset + delta[b][a0]
            dp[b][d] = 1
        # Iterate over rounds i=1 to n-1
        for i in range(1, n):
            newdp = [[0]*size for _ in range(3)]
            a = sa[i]
            for last in range(3):
                for d in range(size):
                    val = dp[last][d]
                    if val == 0:
                        continue
                    for nxt in range(3):

                            continue

                ans = (ans + dp[b][d]) % MOD
        return ans

class Solution:
    def countWinningSequences(self, s: str) -> int:
        MOD = 10**9 + 7
        n = len(s)
        # map creatures to indices: F->0, W->1, E->2
        # delta[b][a] = Bob's score - Alice's score for round with Bob's move b and Alice's move a
        beat = {(0, 2): 1, (1, 0): 1, (2, 1): 1}  # (winner, loser) for Bob winning
        delta = [[0] * 3 for _ in range(3)]
        for b in range(3):
            for a in range(3):
                if b == a:
                    delta[b][a] = 0
                elif (b, a) in beat:
                    delta[b][a] = 1
                else:
                    delta[b][a] = -1

        # convert s to indices
        alice = [0 if ch == 'F' else 1 if ch == 'W' else 2 for ch in s]

        offset = n
        size = 2 * n + 1          # diff in [-n, n] -> index diff+offset

        # dp[last][d] = number of ways for processed rounds ending with move 'last' and diff = d-offset
        dp = [[0] * size for _ in range(3)]

        # first round
        a0 = alice[0]
        for b in range(3):
            idx = offset + delta[b][a0]
            dp[b][idx] = 1

        # remaining rounds
        for i in range(1, n):
            newdp = [[0] * size for _ in range(3)]
            a = alice[i]
            for last in range(3):
                row = dp[last]
                for d in range(size):
                    val = row[d]
                    if val == 0:
                        continue
                    for nxt in range(3):
                        if nxt == last:
                            continue
                        nd = d + delta[nxt][a]
                        if 0 <= nd < size:
                            newdp[nxt][nd] = (newdp[nxt][nd] + val) % MOD
            dp = newdp

        # sum over diff > 0
        ans = 0
        for b in range(3):
            for d in range(offset + 1, size):
                ans = (ans + dp[b][d]) % MOD
        return ans
