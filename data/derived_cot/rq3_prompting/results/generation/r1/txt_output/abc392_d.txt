You will be given a competitive programming problem.
There are N dice.
The i-th die has K_i faces, with the numbers A_{i,1}, A_{i,2}, \ldots, A_{i,K_i} written on them.

You choose two dice from the N dice and roll them.
Determine the maximum probability that the two dice show the same number, when the dice are chosen optimally.

Input

- K_1 + K_2 + \dots + K_N \leq 10^5
- 1 \leq A_{i,j} \leq 10^5
- All input values are integers.

4 1 2 2 1

```python
import sys

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    K = []               # list of number of faces per die
    freq = []            # list of dictionaries: value -> count

    for _ in range(N):
        k = int(next(it))
        K.append(k)
        d = {}
        for _ in range(k):
            a = int(next(it))
            d[a] = d.get(a, 0) + 1
        freq.append(d)

    max_prob = 0.0
    # iterate over all unordered pairs
    for i in range(N):
        for j in range(i + 1, N):
            # choose the smaller dictionary to iterate over
            if len(freq[i]) <= len(freq[j]):
                small = freq[i]
                large = freq[j]
            else:
                small = freq[j]
                large = freq[i]

            # compute sum over common values
            s = 0
            for val, cnt in small.items():
                if val in large:
                    s += cnt * large[val]

            prob = s / (K[i] * K[j])
            if prob > max_prob:
                max_prob = prob

    # output with sufficient precision
    print("{:.15f}".format(max_prob))

if __name__ == "__main__":
    main()
```
