```python
from typing import List

class Solution:
    def maxScore(self, a: List[int], b: List[int]) -> int:
        # dp[k] = maximum score after picking k elements (0 <= k <= 4)
        dp = [0] + [float('-inf')] * 4  # dp[0] = 0
        for val in b:
            # iterate backwards to avoid using updated values within the same step
            for k in range(4, 0, -1):
                dp[k] = max(dp[k], dp[k-1] + a[k-1] * val)
        return dp[4]
```
