        return

    low_prev = 0

    # iterate i from 1 to n
    for i in range(1, n + 1):

        high = min(m, i + K)

        prev_low = max(0, (i - 1) - K)
        prev_high = min(m, (i - 1) + K)
        # pre-fetch current character of S

    if abs(n - m) > K:

    if K == 0:

        return

    INF = K + 1  # any value > K is equivalent

    # dp for i = 0
    low_prev = 0
    high_prev = min(m, K)

        high = min(m, i + K)

            idx = j - low
