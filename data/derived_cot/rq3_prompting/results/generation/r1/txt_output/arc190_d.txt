import sys

def mat_mul(A, B, mod):
    """Multiply two matrices A and B modulo mod."""
    n = len(A)
    C = [[0] * n for _ in range(n)]
    for i in range(n):
        Ai = A[i]
        Ci = C[i]
        for k in range(n):
            aik = Ai[k]
            if aik:
                Bk = B[k]
                # unroll the inner loop for speed
                for j in range(n):
                    Ci[j] = (Ci[j] + aik * Bk[j]) % mod
    return C

def mat_pow(mat, exp, mod):
    """Raise matrix mat to the power exp modulo mod."""
    n = len(mat)
    # identity matrix
    res = [[0] * n for _ in range(n)]
    for i in range(n):
        res[i][i] = 1
    base = mat
    while exp:
        if exp & 1:
            res = mat_mul(res, base, mod)
        base = mat_mul(base, base, mod)
        exp >>= 1
    return res

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    p = int(next(it))
    A = []
    zeros = 0
    for _ in range(N):
        row = []
        for _ in range(N):
            val = int(next(it))
            row.append(val)
            if val == 0:
                zeros += 1
        A.append(row)

    # Special case p = 2
    if p == 2:
        # Build B: replace zeros by 1
        B = [[1 if A[i][j] == 0 else A[i][j] for j in range(N)] for i in range(N)]
        # Compute B^2 modulo 2
        B2 = mat_mul(B, B, 2)
        for row in B2:
            print(' '.join(str(x) for x in row))
        return

    # p >= 3 (prime)
    # Fixed matrix F: original non-zero entries
    F = [[A[i][j] if A[i][j] != 0 else 0 for j in range(N)] for i in range(N)]

    # M0 = F^p
    M0 = mat_pow(F, p, p)

    # Result matrix R = M0 + contributions from zeros
    R = [row[:] for row in M0]

    # Process each zero position
    for i in range(N):
        for j in range(N):
            if A[i][j] == 0:

                else:

                    if p == 3:
                        # add F[j][i] to R[i][j]
                        R[i][j] = (R[i][j] + F[j][i]) % p

        for i in range(N):
