```python
import sys
import itertools

def main():
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    M = next(it)

    INF = 10**18
    # distance matrix
    dist = [[INF] * N for _ in range(N)]
    for i in range(N):
        dist[i][i] = 0

    edges_info = [None] * M
    for i in range(M):
        u = next(it) - 1
        v = next(it) - 1
        t = next(it)
        edges_info[i] = (u, v, t)
        if t < dist[u][v]:
            dist[u][v] = t
            dist[v][u] = t

    # Floyd‑Warshall
    for k in range(N):
        dk = dist[k]
        for i in range(N):
            dik = dist[i][k]
            if dik == INF:
                continue
            di = dist[i]
            # inner loop over j
            for j in range(N):
                nd = dik + dk[j]
                if nd < di[j]:
                    di[j] = nd

    Q = next(it)
    out = []
    for _ in range(Q):
        K = next(it)
        bridges = [next(it) - 1 for _ in range(K)]
        req = [edges_info[b] for b in bridges]

        best = INF
        # try every order of the required bridges
        for perm in itertools.permutations(req):
            k = len(perm)
            # try every orientation (which endpoint to start from)
            for mask in range(1 << k):
                cur = 0          # node 1 (index 0)
                cost = 0
                for idx, (u, v, w) in enumerate(perm):
                    if (mask >> idx) & 1:
                        start, end = v, u
                    else:
                        start, end = u, v
                    cost += dist[cur][start]
                    cost += w
                    cur = end
                cost += dist[cur][N - 1]
                if cost < best:
                    best = cost
        out.append(str(best))

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
```
