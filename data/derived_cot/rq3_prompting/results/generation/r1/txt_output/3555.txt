=== ANSWER ===
```python
from typing import List

class Solution:
    def getFinalState(self, nums: List[int], k: int, multiplier: int) -> List[int]:
        # Simulate k operations
        for _ in range(k):
            # Find the minimum value and its first occurrence
            min_val = min(nums)
            idx = nums.index(min_val)
            # Multiply the selected element
            nums[idx] *= multiplier
        return nums
```
