```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    Q = int(next(it))
    S = list(next(it).decode())
    
    # initial count
    cnt = 0
    for i in range(N - 2):
        if S[i] == 'A' and S[i+1] == 'B' and S[i+2] == 'C':
            cnt += 1
    
    out_lines = []
    for _ in range(Q):
        X = int(next(it)) - 1  # 0-indexed
        C = next(it).decode()
        
        # remove old contributions from substrings that include position X
        for start in (X-2, X-1, X):
            if 0 <= start <= N-3:
                if S[start] == 'A' and S[start+1] == 'B' and S[start+2] == 'C':
                    cnt -= 1
        
        # update character
        S[X] = C
        
        # add new contributions
        for start in (X-2, X-1, X):
            if 0 <= start <= N-3:
                if S[start] == 'A' and S[start+1] == 'B' and S[start+2] == 'C':
                    cnt += 1
        
        out_lines.append(str(cnt))
    
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
