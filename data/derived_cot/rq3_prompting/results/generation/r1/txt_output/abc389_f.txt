            max_val[node] = l
        else:

            build(node*2, l, mid)
            build(node*2+1, mid+1, r)

            max_val[node] = max(max_val[node*2], max_val[node*2+1])

            max_val[left] += v
            lazy[left] += v
            min_val[right] += v
            max_val[right] += v
