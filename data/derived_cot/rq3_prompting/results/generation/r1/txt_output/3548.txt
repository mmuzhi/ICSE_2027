class Solution:
    def countGoodIntegers(self, n: int, k: int) -> int:
        # Precompute factorials up to n
        fact = [1] * (n + 1)
        for i in range(1, n + 1):
            fact[i] = fact[i-1] * i
        
        counts = [0] * 10
        ans = 0

        def process():
            nonlocal ans
            # Check palindrome condition (at most one odd count)
            odd_cnt = 0
            for c in counts:
                if c & 1:
                    odd_cnt += 1
            if odd_cnt > 1:
                return

            half_counts = [c // 2 for c in counts]
            mid_digit = None
            if n % 2 == 1:
                # There must be exactly one odd count
                for i, c in enumerate(counts):
                    if c & 1:
                        mid_digit = i
                        break

            m = n // 2
            found = False

            if m == 0:
                # n == 1
                pal = mid_digit
                found = (pal % k == 0)
            else:
                # Generate all left halves (distinct permutations)
                def dfs_left(cur, rem_counts):
                    nonlocal found
                    if found:
                        return
                    if len(cur) == m:
                        if cur[0] == '0':
                            return
                        if n % 2 == 0:
                            pal_str = cur + cur[::-1]
                        else:
                            pal_str = cur + str(mid_digit) + cur[::-1]
                        pal_int = int(pal_str)
                        if pal_int % k == 0:
                            found = True
                        return
                    for d in range(10):
                        if rem_counts[d] > 0:
                            rem_counts[d] -= 1
                            dfs_left(cur + str(d), rem_counts)
                            rem_counts[d] += 1
                            if found:
                                return

                dfs_left("", half_counts[:])

            if not found:
                return

            # Count numbers with this multiset
            total = fact[n]
            for c in counts:
                total //= fact[c]
            lead_zero = 0
            if counts[0] > 0:
                lead_zero = fact[n-1]
                lead_zero //= fact[counts[0]-1]
                for d in range(1, 10):
                    lead_zero //= fact[counts[d]]
            ans += total - lead_zero

        # Generate all multisets recursively
        def dfs_multiset(digit, remaining):
            if digit == 9:
                counts[9] = remaining
                process()
                return
            for c in range(remaining+1):
                counts[digit] = c
                dfs_multiset(digit+1, remaining - c)
            # no need to reset counts[digit] as it will be overwritten in next iteration

        dfs_multiset(0, n)
        return ans
