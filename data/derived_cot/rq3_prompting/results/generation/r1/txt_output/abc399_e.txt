        if f[s] == -1:
            f[s] = t
        elif f[s] != t:
            print(-1)
            return

    distinct_S = len(set(S))
    distinct_T = len(set(T))

    # can never increase the number of distinct letters
    if distinct_T > distinct_S:
        print(-1)
        return

    # if T uses all 26 letters, any non‑self operation would reduce the count
    if distinct_T == 26:
        for c in range(26):
            if f[c] != -1 and f[c] != c:
                print(-1)
                return
        print(0)
        return

    # count source letters that must be changed (non‑self edges)
    base = 0
    for c in range(26):
        if f[c] != -1 and f[c] != c:
            base += 1

    # detect cycles in the functional graph (only non‑self edges)
    visited = [0] * 26   # 0: unvisited, 1: in current path, 2: done
    cycle_cnt = 0

    for s in range(26):
        if f[s] != -1 and f[s] != s and visited[s] == 0:
            cur = s
            path = []
            while cur != -1 and visited[cur] == 0:
                visited[cur] = 1
                path.append(cur)
                nxt = f[cur]
                # stop if nxt has no outgoing non‑self edge
                if nxt == -1 or f[nxt] == -1 or f[nxt] == nxt:
                    break
                if visited[nxt] == 1:
                    cycle_cnt += 1
                    break
                cur = nxt
            for node in path:
                visited[node] = 2

    # each cycle needs a free letter (not in T) to be broken
    free_letters = 26 - distinct_T
    if cycle_cnt > free_letters:
        print(-1)
        return

    print(base + cycle_cnt)

if __name__ == "__main__":
    solve()
