                for s in sum_list:
                    i = s - c
                    if 1 <= i <= N:
                        cnt += 1
            return cnt
        if not use_row and use_col and not use_sum and use_diff:
            cnt = 0
            for c in col_list:
                for d in diff_list:
                    i = c + d
                    if 1 <= i <= N:
                        cnt += 1
            return cnt
        if not use_row and not use_col and use_sum and use_diff:
            cnt = 0
            for s in sum_list:
                for d in diff_list:
                    if (s + d) % 2 == 0:
                        i = (s + d) // 2
                        j = (s - d) // 2
                        if 1 <= i <= N and 1 <= j <= N:
                            cnt += 1
            return cnt

        # ---- three conditions ----
        if use_row and use_col and use_sum and not use_diff:
            cnt = 0
            for r in row_list:
                for c in col_list:
                    if (r + c) in sum_set:
                        cnt += 1
            return cnt
        if use_row and use_col and not use_sum and use_diff:
            cnt = 0
            for r in row_list:
                for c in col_list:
                    if (r - c) in diff_set:
                        cnt += 1
            return cnt
        if use_row and not use_col and use_sum and use_diff:
            cnt = 0
            for r in row_list:
                for s in sum_list:
                    j = s - r
                    if 1 <= j <= N and (r - j) in diff_set:
                        cnt += 1
            return cnt
        if not use_row and use_col and use_sum and use_diff:
            cnt = 0
            for c in col_list:
                for s in sum_list:
                    i = s - c
                    if 1 <= i <= N and (i - c) in diff_set:
                        cnt += 1
            return cnt

        # ---- four conditions ----
        if use_row and use_col and use_sum and use_diff:
            cnt = 0
            for r in row_list:
                for c in col_list:
                    if (r + c) in sum_set and (r - c) in diff_set:
                        cnt += 1
            return cnt

        return 0  # never reached

    ans = 0
