```python
import sys

def feasible(N, A, require_breaks):
    # try both choices for x1 (edge 1 selected or not)
    for start in (0, 1):
        # dp[s][bo][be]  s: x_i, bo: odd break (found odd edge with 0), be: even break
        dp = [[[False]*2 for _ in range(2)] for __ in range(2)]
        # initialise for i = 1
        bo_start = (1 % 2 == 1 and start == 0)   # i=1 is odd -> break if we set x1=0
        be_start = (1 % 2 == 0 and start == 0)   # i=1 is odd, so never even break here
        dp[start][bo_start][be_start] = True

        for i in range(2, N+1):
            ndp = [[[False]*2 for _ in range(2)] for __ in range(2)]
            for prev_s in (0,1):
                for bo in (0,1):
                    for be in (0,1):
                        if not dp[prev_s][bo][be]:
                            continue
                        for s in (0,1):
                            # independent set constraint
                            if prev_s + s > 1:
                                continue
                            # coverage for zero at position i (1-indexed)
                            if A[i-1] == 0 and prev_s + s < 1:
                                continue
                            nbo = bo or (i % 2 == 1 and s == 0)
                            nbe = be or (i % 2 == 0 and s == 0)
                            ndp[s][nbo][nbe] = True
            dp = ndp

        # wrap-around check with edge 1
        for s in (0,1):
            for bo in (0,1):
                for be in (0,1):
                    if not dp[s][bo][be]:
                        continue
                    if s + start > 1:        # independent set constraint between edge N and edge 1
                        continue
                    if A[0] == 0 and s + start < 1:   # coverage for zero at position 1
                        continue
                    if require_breaks and not (bo and be):
                        continue
                    return True
    return False

def solve():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    # quick check: if all ones -> always possible
    if all(x == 1 for x in A):
        print("Yes")
        return

    # first, check if any feasible independent set exists that covers all zeros
    if not feasible(N, A, False):
        print("No")
        return

    # if N % 4 != 2, no further restriction
    if N % 4 != 2:
        print("Yes")
        return

    # N % 4 == 2: we need a solution that does NOT consist of all odd edges or all even edges
    if feasible(N, A, True):
        print("Yes")
    else:
        print("No")
```
