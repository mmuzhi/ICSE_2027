import sys

def solve() -> None:
    data = sys.stdin.read().splitlines()
    if not data:
        return
    # K is always 1; we can ignore it
    # K = int(data[0].strip())
    S = data[1].strip()
    T = data[2].strip()

    n, m = len(S), len(T)

    # If length difference is more than 1, impossible
    if abs(n - m) > 1:
        print("No")
        return

    # Equal length: at most one character differs
    if n == m:
        diff = 0
        for a, b in zip(S, T):
            if a != b:
                diff += 1
                if diff > 1:
                    print("No")
                    return
        print("Yes")
        return

    # One string is longer by 1
    # Let 'shorter' be the shorter string, 'longer' the longer one
    if n < m:
        short, long = S, T
    else:
        short, long = T, S

    # Two‑pointer check: can we obtain the longer string by inserting one character into the shorter?
    i = j = 0          # i for short, j for long
    inserted = False
    len_short = len(short)
    len_long = len(long)

    while i < len_short and j < len_long:
        if short[i] == long[j]:
            i += 1
            j += 1
        else:
            if inserted:
                print("No")
                return
            # skip one character in the longer string (simulate insertion)
            j += 1
            inserted = True

    # After the loop:
    # If we never inserted, the extra character must be at the end of the longer string.
    if not inserted:
        print("Yes")
    else:
        # We used the insertion; check that all remaining characters match
        if i == len_short and j == len_long:
            print("Yes")
        else:
            print("No")

if __name__ == "__main__":
    solve()
