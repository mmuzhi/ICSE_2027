import sys
import math

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    S = int(next(it))
    T = int(next(it))
    segments = []
    for _ in range(N):
        a = int(next(it)); b = int(next(it)); c = int(next(it)); d = int(next(it))
        segments.append(((a, b), (c, d)))
    
    # precompute lengths of segments
    seg_len = [math.hypot(x2-x1, y2-y1) for (x1,y1),(x2,y2) in segments]
    total_draw = sum(seg_len) / T
    
    # endpoints as list of two tuples per segment
    ends = segments  # ends[i][0] = first endpoint, ends[i][1] = second
    
    INF = float('inf')
    num_masks = 1 << N
    # dp[mask][i][side] : minimal travel distance to have drawn segments in mask,
    # currently at endpoint side (0 or 1) of segment i.
    dp = [[[INF]*2 for _ in range(N)] for __ in range(num_masks)]
    
    # initial states: choose first segment and starting endpoint
    for i in range(N):
        # start at endpoint 0 -> end at endpoint 1
        d0 = math.hypot(ends[i][0][0], ends[i][0][1])
        dp[1<<i][i][1] = min(dp[1<<i][i][1], d0)
        # start at endpoint 1 -> end at endpoint 0
        d1 = math.hypot(ends[i][1][0], ends[i][1][1])
        dp[1<<i][i][0] = min(dp[1<<i][i][0], d1)
    
    # DP transitions
    for mask in range(num_masks):
        for i in range(N):
            if not (mask & (1<<i)):
                continue
            for side in (0, 1):
                cur = dp[mask][i][side]
                if cur == INF:
                    continue
                cur_x, cur_y = ends[i][side]
                for j in range(N):
                    if mask & (1<<j):
                        continue
                    for start_side in (0, 1):
                        sx, sy = ends[j][start_side]
                        add = math.hypot(cur_x - sx, cur_y - sy)
                        new_cost = cur + add
                        new_side = 1 - start_side
                        new_mask = mask | (1<<j)
                        if new_cost < dp[new_mask][j][new_side]:
                            dp[new_mask][j][new_side] = new_cost
    
    full_mask = (1 << N) - 1
    best_travel = min(dp[full_mask][i][side] for i in range(N) for side in (0,1))
    total_time = total_draw + best_travel / S
    # Print with high precision
    print("{:.15f}".format(total_time))

if __name__ == "__main__":
    solve()
