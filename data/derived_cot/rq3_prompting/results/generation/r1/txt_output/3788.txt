from typing import List

class Solution:
    def maxSum(self, nums: List[int]) -> int:
        # Store the maximum occurrence for each distinct value
        max_per_val = {}
        for num in nums:
            # Since we only compare with the same value, the max is simply the value itself
            # (all occurrences of the same value are equal to the value).
            # But we need to record that we have seen this value.
            # Actually we want the maximum value among occurrences; since all occurrences of the same number
            # are the same number, the maximum is that number. So we can just add the value if positive.
            # However, to be safe we keep dictionary.
            max_per_val[num] = num
        
        total = 0
        for val in max_per_val.values():
            if val > 0:
                total += val
        
        # If no positive value, return the largest element (which is the max of the array)
        if total > 0:
            return total
        else:
            return max(nums)
