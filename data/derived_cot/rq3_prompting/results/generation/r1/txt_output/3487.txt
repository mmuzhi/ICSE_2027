```python
from typing import List

class Solution:
    def maxRemovals(self, source: str, pattern: str, targetIndices: List[int]) -> int:
        n = len(source)
        m = len(pattern)
        target_set = set(targetIndices)
        INF = n + 1  # larger than any possible count
        dp = [INF] * (m + 1)
        dp[0] = 0
        for i, ch in enumerate(source):
            cost = 1 if i in target_set else 0
            # iterate backwards
            for j in range(m, 0, -1):
                if ch == pattern[j-1]:
                    if dp[j-1] + cost < dp[j]:
                        dp[j] = dp[j-1] + cost
        # dp[m] should be finite
        return len(targetIndices) - dp[m]
```

```python
from typing import List

class Solution:
    def maxRemovals(self, source: str, pattern: str, targetIndices: List[int]) -> int:
        n = len(source)
        m = len(pattern)
        target_set = set(targetIndices)
        INF = n + 1  # larger than any possible count of used target indices

        # dp[j] = minimum number of target indices used to form pattern[0:j]
        dp = [INF] * (m + 1)
        dp[0] = 0

        for i, ch in enumerate(source):
            cost = 1 if i in target_set else 0
            # update dp from high to low to avoid using the same character multiple times
            for j in range(m, 0, -1):
                if ch == pattern[j-1]:
                    if dp[j-1] + cost < dp[j]:
                        dp[j] = dp[j-1] + cost

        # dp[m] is the minimum number of target indices we must keep
        return len(targetIndices) - dp[m]
```
