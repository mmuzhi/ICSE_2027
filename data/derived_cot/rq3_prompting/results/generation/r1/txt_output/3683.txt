class Solution:
    def answerString(self, word: str, numFriends: int) -> str:
        # If only one friend, the only piece is the whole string.
        if numFriends == 1:
            return word

        n = len(word)
        L = n - numFriends + 1          # maximum possible length of a piece
        best = ""

        for i in range(n):
            if i <= numFriends - 1:
                # The longest valid substring starting at i has length L.
                candidate = word[i:i + L]
            else:
                # For later starts, the longest valid substring is the suffix.
                candidate = word[i:]

            if candidate > best:
                best = candidate

        return best
