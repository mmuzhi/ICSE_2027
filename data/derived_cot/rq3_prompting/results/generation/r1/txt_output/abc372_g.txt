import sys, functools

def floor_sum(n, m, a, b):
    ans = 0
    while True:
        if a >= m:
            ans += (n - 1) * n * (a // m) // 2
            a %= m
        if b >= m:
            ans += n * (b // m)
            b %= m
        y_max = a * n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans

def solve():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    T = int(next(it))
    out_lines = []
    for _ in range(T):
        N = int(next(it))
        lines = []
        for i in range(N):
            A = int(next(it)); B = int(next(it)); C = int(next(it))
            C_prime = C - 1
            lines.append((A, B, C_prime))
        # sort lines
        def cmp(l1, l2):
            A1,B1,C1 = l1; A2,B2,C2 = l2
            left = A1 * B2
            right = A2 * B1
            if left != right:
                # descending A/B: larger left first
                return -1 if left > right else 1
            else:
                # same slope, smaller intercept first
                cleft = C1 * B2
                cright = C2 * B1
                if cleft != cright:
                    return -1 if cleft < cright else 1
                return 0
        lines.sort(key=functools.cmp_to_key(cmp))
        # filter equal slopes
        filtered = []
        for i, line in enumerate(lines):
            if i > 0:
                Aprev,Bprev,Cprev = lines[i-1]
                A,B,C = line
                if A * Bprev == Aprev * B:  # same slope
                    continue
            filtered.append(line)
        # build lower envelope (min)
        env = []
        for line in filtered:
            A,B,C = line
            while len(env) >= 2:
                A1,B1,C1 = env[-2]
                A2,B2,C2 = env[-1]
                A3,B3,C3 = line
                # compute x12 and x23
                num12 = C1 * B2 - C2 * B1
                den12 = A1 * B2 - A2 * B1  # positive
                num23 = C2 * B3 - C3 * B2
                den23 = A2 * B3 - A3 * B2  # positive
                # if x12 >= x23, pop
                if num12 * den23 >= num23 * den12:
                    env.pop()
                else:
                    break
            env.append(line)
        # now process envelope
        rev = env[::-1]
        k = len(rev)
        # compute floor of intersections
        floors = []
        for i in range(k-1):
            A1,B1,C1 = rev[i]
            A2,B2,C2 = rev[i+1]
            num = C2 * B1 - C1 * B2
            den = A2 * B1 - A1 * B2  # positive
            floor_x = num // den
            floors.append(floor_x)
        total = 0
        current_left = 1
        INF = 10**18
        for i in range(k):
            A,B,C = rev[i]
            if i < k-1:
                right = floors[i]
            else:
                right = INF
            # compute x_max
            if C >= B:
                x_max = (C - B) // A
            else:
                x_max = -1
            l = max(current_left, 1)
            r = min(right, x_max) if right != INF else x_max
            if l <= r:
                n = r - l + 1
                b = C - A * r
                s = floor_sum(n, B, A, b)
                total += s
            if i < k-1:
                current_left = floors[i] + 1
        out_lines.append(str(total))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()

```python
import sys
import functools

# ------------------------------------------------------------
# floor_sum(n, m, a, b) = sum_{i=0}^{n-1} floor((a*i + b) / m)
# classic AtCoder library implementation (supports n >= 0, m > 0)
def floor_sum(n, m, a, b):
    ans = 0
    while True:
        if a >= m:
            ans += (n - 1) * n * (a // m) // 2
            a %= m
        if b >= m:
            ans += n * (b // m)
            b %= m
        y_max = a * n + b
        if y_max < m:
            break
        n = y_max // m
        b = y_max % m
        m, a = a, m
    return ans


# ------------------------------------------------------------
def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    T = int(next(it))
    out_lines = []

    for _ in range(T):
        N = int(next(it))
        lines = []
        for _ in range(N):
            A = int(next(it))
            B = int(next(it))
            C = int(next(it))
            C_prime = C - 1                     # strict inequality
            lines.append((A, B, C_prime))

        # ---- step 1: sort by slope (A/B descending) and for equal slope by intercept (C/B ascending) ----
        def cmp(l1, l2):
            A1, B1, C1 = l1
            A2, B2, C2 = l2
            left = A1 * B2
            right = A2 * B1
            if left != right:
                return -1 if left > right else 1   # descending A/B
            else:
                # equal slopes -> smaller intercept first
                cleft = C1 * B2
                cright = C2 * B1
                if cleft != cright:
                    return -1 if cleft < cright else 1
                return 0

        lines.sort(key=functools.cmp_to_key(cmp))

        # ---- step 2: keep only the best line for each slope ----
        filtered = []
        for i, (A, B, C) in enumerate(lines):
            if i > 0:
                Ap, Bp, Cp = lines[i-1]
                if A * Bp == Ap * B:          # same denominator
                    continue                  # previous has smaller intercept
            filtered.append((A, B, C))

        # ---- step 3: build lower envelope (min of lines) ----
        env = []          # steepest first (largest A/B)
        for line in filtered:
            A, B, C = line
            while len(env) >= 2:
                A1, B1, C1 = env[-2]
                A2, B2, C2 = env[-1]
                A3, B3, C3 = line
                # intersection x12 = (C1*B2 - C2*B1) / (A1*B2 - A2*B1)
                # intersection x23 = (C2*B3 - C3*B2) / (A2*B3 - A3*B2)
                num12 = C1 * B2 - C2 * B1
                den12 = A1 * B2 - A2 * B1      # >0 because slopes strictly decreasing
                num23 = C2 * B3 - C3 * B2
                den23 = A2 * B3 - A3 * B2      # >0
                if num12 * den23 >= num23 * den12:
                    env.pop()
                else:
                    break
            env.append(line)

        # ---- step 4: process envelope from shallow to steep (increasing x) ----
        rev = env[::-1]        # rev[0] = shallowest, rev[-1] = steepest
        k = len(rev)

        # compute floor of intersection points between consecutive lines
        floors = []            # floors[i] = floor( intersection( rev[i], rev[i+1] ) )
        for i in range(k - 1):
            A1, B1, C1 = rev[i]
            A2, B2, C2 = rev[i + 1]
            # intersection: (C2*B1 - C1*B2) / (A2*B1 - A1*B2)   (denominator positive)
            num = C2 * B1 - C1 * B2
            den = A2 * B1 - A1 * B2
            floor_x = num // den
            floors.append(floor_x)

        total = 0
        current_left = 1
        INF = 10 ** 18

        for i in range(k):
            A, B, C = rev[i]
            # right bound for this line (inclusive) from envelope
            if i < k - 1:
                right = floors[i]
            else:
                right = INF

            # largest x where this line still yields y >= 1
            if C >= B:
                x_max = (C - B) // A
            else:
                x_max = -1

            l = max(current_left, 1)
            r = min(right, x_max) if right != INF else x_max
            if l <= r:
                n = r - l + 1
                b = C - A * r
                s = floor_sum(n, B, A, b)
                total += s

            # prepare left bound for next line
            if i < k - 1:
                current_left = floors[i] + 1

        out_lines.append(str(total))

    sys.stdout.write("\n".join(out_lines))


# ------------------------------------------------------------
if __name__ == "__main__":
    solve()
```
