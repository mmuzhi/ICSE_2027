We need Fenwick class:

class Fenwick:
    def __init__(self, n):
        self.n = n
        self.bit = [0]*(n+1)
    def update(self, i, val):
        while i <= self.n:
            self.bit[i] = max(self.bit[i], val)
            i += i & -i
    def query(self, i):
        res = 0
        while i > 0:
            res = max(res, self.bit[i])
            i -= i & -i
        return res

from typing import List

class Fenwick:
    def __init__(self, n: int):
        self.n = n
        self.bit = [0] * (n + 1)
    
    def update(self, idx: int, val: int) -> None:
        while idx <= self.n:
            if val > self.bit[idx]:
                self.bit[idx] = val
            idx += idx & -idx
    
    def query(self, idx: int) -> int:
        res = 0
        while idx > 0:
            if self.bit[idx] > res:
                res = self.bit[idx]
            idx -= idx & -idx
        return res

class Solution:
    def maxPathLength(self, coordinates: List[List[int]], k: int) -> int:
        n = len(coordinates)
        # coordinate compression of y
        ys = [y for _, y in coordinates]
        sorted_ys = sorted(set(ys))
        y_to_idx = {v: i+1 for i, v in enumerate(sorted_ys)}  # 1-indexed
        m = len(sorted_ys)
        
        # store points with original index
        points = [(x, y, i) for i, (x, y) in enumerate(coordinates)]
        
        # ---- compute left_len (longest chain ending at point) ----
        left = [0] * n
        # sort by x ascending (batch equal x)
        points.sort(key=lambda p: p[0])  # sort by x only, group later
        bit = Fenwick(m)
        i = 0
        while i < n:
            j = i
            x_val = points[i][0]
            while j < n and points[j][0] == x_val:
                j += 1
            # compute dp for all points in this x-group based on previous x
            for t in range(i, j):
                _, y, idx = points[t]
                cy = y_to_idx[y]
                best = bit.query(cy - 1)
                left[idx] = best + 1
            # update BIT after computing all in group to avoid using same x
            for t in range(i, j):
                _, y, idx = points[t]
                cy = y_to_idx[y]
                bit.update(cy, left[idx])
            i = j

from typing import List

class Fenwick:
    def __init__(self, n: int):
        self.n = n
        self.bit = [0] * (n + 1)

    def update(self, idx: int, val: int) -> None:
        while idx <= self.n:
            if val > self.bit[idx]:
                self.bit[idx] = val
            idx += idx & -idx

    def query(self, idx: int) -> int:
        res = 0
        while idx > 0:
            if self.bit[idx] > res:
                res = self.bit[idx]
            idx -= idx & -idx
        return res


class Solution:
    def maxPathLength(self, coordinates: List[List[int]], k: int) -> int:
        n = len(coordinates)
        # compress y coordinates
        ys = [y for _, y in coordinates]
        uniq_y = sorted(set(ys))
        y_to_idx = {v: i + 1 for i, v in enumerate(uniq_y)}
        m = len(uniq_y)

        # keep original indices
        points = [(x, y, i) for i, (x, y) in enumerate(coordinates)]

        # ---------- longest chain ending at each point ----------
        left = [0] * n
        points.sort(key=lambda p: p[0])  # sort by x ascending
        bit = Fenwick(m)
        i = 0
        while i < n:
            j = i
            x_val = points[i][0]
            while j < n and points[j][0] == x_val:
                j += 1
            # compute for this x-group using only points with smaller x
            for t in range(i, j):
                _, y, idx = points[t]
                cy = y_to_idx[y]
                best = bit.query(cy - 1)
                left[idx] = best + 1
            # after computing all, update BIT
            for t in range(i, j):
                _, y, idx = points[t]
                cy = y_to_idx[y]
                bit.update(cy, left[idx])
            i = j

        # ---------- longest chain starting from each point ----------
        right = [0] * n
        points.sort(key=lambda p: -p[0])  # sort by x descending
        bit_rev = Fenwick(m)  # BIT on reversed y indices
        i = 0
        while i < n:
            j = i
            x_val = points[i][0]
            while j < n and points[j][0] == x_val:
                j += 1
            # compute for this x-group using only points with larger x
            for t in range(i, j):
                _, y, idx = points[t]
                cy = y_to_idx[y]
                rev = m - cy + 1  # larger y → smaller rev
                best = bit_rev.query(rev - 1)
                right[idx] = best + 1
            # after computing all, update BIT on reversed indices
            for t in range(i, j):
                _, y, idx = points[t]
                cy = y_to_idx[y]
                rev = m - cy + 1
                bit_rev.update(rev, right[idx])
            i = j

        return left[k] + right[k] - 1
