```python
class Solution:
    def minLength(self, s: str, numOps: int) -> int:
        n = len(s)
        # Helper to check if we can achieve max run <= L with <= numOps flips
        def feasible(L):
            INF = n + 5
            # dp0[k] = min flips for processed prefix ending with char '0' and run length k
            # dp1[k] for char '1'
            dp0 = [INF] * (L + 2)   # index 1..L
            dp1 = [INF] * (L + 2)
            # Initialize with first character
            for flip in (0, 1):
                c = (int(s[0]) ^ flip)
                cost = flip
                if c == 0:
                    dp0[1] = min(dp0[1], cost)
                else:
                    dp1[1] = min(dp1[1], cost)
            # Iterate over remaining characters
            for i in range(1, n):
                new0 = [INF] * (L + 2)
                new1 = [INF] * (L + 2)
                # Process states ending with 0
                for k in range(1, L + 1):
                    if dp0[k] == INF:
                        continue
                    # For each possible target char
                    for target in (0, 1):
                        flip = 1 if target != int(s[i]) else 0
                        cost = dp0[k] + flip
                        if target == 0:
                            newk = k + 1
                            if newk > L:
                                continue
                            new0[newk] = min(new0[newk], cost)
                        else: # target == 1
                            newk = 1
                            new1[newk] = min(new1[newk], cost)
                # Process states ending with 1
                for k in range(1, L + 1):
                    if dp1[k] == INF:
                        continue
                    for target in (0, 1):
                        flip = 1 if target != int(s[i]) else 0
                        cost = dp1[k] + flip
                        if target == 1:
                            newk = k + 1
                            if newk > L:
                                continue
                            new1[newk] = min(new1[newk], cost)
                        else: # target == 0
                            newk = 1
                            new0[newk] = min(new0[newk], cost)
                dp0, dp1 = new0, new1
            # Check if any state has cost <= numOps
            for k in range(1, L+1):
                if dp0[k] <= numOps or dp1[k] <= numOps:
                    return True
            return False
        # Binary search
        lo, hi = 1, n
        while lo < hi:
            mid = (lo + hi) // 2
            if feasible(mid):
                hi = mid
            else:
                lo = mid + 1
        return lo
```

```python
class Solution:
    def minLength(self, s: str, numOps: int) -> int:
        n = len(s)

        # Check if we can achieve max run length <= L with at most numOps flips
        def feasible(L: int) -> bool:
            INF = n + 5
            # dp0[k] = min flips for processed prefix ending with '0' and run length k
            # dp1[k] = min flips for processed prefix ending with '1' and run length k
            dp0 = [INF] * (L + 2)   # indices 1 .. L
            dp1 = [INF] * (L + 2)

            # Initialise with first character
            for flip in (0, 1):
                c = (int(s[0]) ^ flip)
                cost = flip
                if c == 0:
                    dp0[1] = min(dp0[1], cost)
                else:
                    dp1[1] = min(dp1[1], cost)

            # Process the rest of the string
            for i in range(1, n):
                new0 = [INF] * (L + 2)
                new1 = [INF] * (L + 2)

                # States ending with '0'
                for k in range(1, L + 1):
                    if dp0[k] == INF:
                        continue
                    for target in (0, 1):
                        flip = 1 if target != int(s[i]) else 0
                        cost = dp0[k] + flip
                        if target == 0:
                            newk = k + 1
                            if newk > L:
                                continue
                            new0[newk] = min(new0[newk], cost)
                        else:  # target == 1
                            newk = 1
                            new1[newk] = min(new1[newk], cost)

                # States ending with '1'
                for k in range(1, L + 1):
                    if dp1[k] == INF:
                        continue
                    for target in (0, 1):
                        flip = 1 if target != int(s[i]) else 0
                        cost = dp1[k] + flip
                        if target == 1:
                            newk = k + 1
                            if newk > L:
                                continue
                            new1[newk] = min(new1[newk], cost)
                        else:  # target == 0
                            newk = 1
                            new0[newk] = min(new0[newk], cost)

                dp0, dp1 = new0, new1

            # If any state has cost <= numOps, we succeed
            for k in range(1, L + 1):
                if dp0[k] <= numOps or dp1[k] <= numOps:
                    return True
            return False

        # Binary search the minimal possible maximum run length
        lo, hi = 1, n
        while lo < hi:
            mid = (lo + hi) // 2
            if feasible(mid):
                hi = mid
            else:
                lo = mid + 1
        return lo
```
