import sys
import bisect

def solve() -> None:
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    Q = next(it)
    A = [next(it) for _ in range(N)]
    queries = [[] for _ in range(N + 1)]  # queries by R (1-indexed)
    for i in range(Q):
        R = next(it)
        X = next(it)
        queries[R].append((X, i))
    
    # coordinate compression of A values
    uniq = sorted(set(A))
    M = len(uniq)
    # mapping from value to 0-based index
    idx_of = {v: i for i, v in enumerate(uniq)}

    # Fenwick tree for maximum (1-indexed)
    bit = [0] * (M + 2)
    def bit_update(i: int, val: int) -> None:
        while i <= M:
            if val > bit[i]:
                bit[i] = val
            i += i & -i
    def bit_query(i: int) -> int:
        res = 0
        while i > 0:
            if bit[i] > res:
                res = bit[i]
            i -= i & -i
        return res

    ans = [0] * Q
    # process array left to right
    for i in range(1, N + 1):
        v = A[i-1]
        idx = idx_of[v] + 1  # 1-indexed BIT index
        best_prev = bit_query(idx - 1)
        cur_len = best_prev + 1
        bit_update(idx, cur_len)

        # answer queries with R == i
        for X, qid in queries[i]:
            # find largest index in uniq with value <= X
            pos = bisect.bisect_right(uniq, X) - 1
            if pos >= 0:
                ans[qid] = bit_query(pos + 1)
            else:
                ans[qid] = 0

    out = '\n'.join(map(str, ans))
    sys.stdout.write(out)

if __name__ == "__main__":
    solve()
