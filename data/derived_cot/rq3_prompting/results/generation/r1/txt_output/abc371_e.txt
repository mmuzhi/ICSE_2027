import sys

def main():
    data = sys.stdin.read().strip().split()
    if not data: return
    n = int(data[0])
    a = list(map(int, data[1:1+n]))
    last = {}
    ans = 0
    for i, val in enumerate(a, start=1):
        prev = last.get(val, 0)
        ans += (i - prev) * (n - i + 1)
        last[val] = i
    print(ans)

if __name__ == "__main__":
    main()
