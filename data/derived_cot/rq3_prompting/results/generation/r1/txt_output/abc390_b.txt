```python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    n = int(data[0])
    a = list(map(int, data[1:1+n]))

    if n == 2:
        print("Yes")
        return

    for i in range(n - 2):
        if a[i+1] * a[i+1] != a[i] * a[i+2]:
            print("No")
            return
    print("Yes")

if __name__ == "__main__":
    solve()
```
