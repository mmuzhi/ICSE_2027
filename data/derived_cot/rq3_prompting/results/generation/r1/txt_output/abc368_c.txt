```python
import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    healths = [int(next(it)) for _ in range(n)]

    t = 0
    for h in healths:
        r = t % 3
        q = h // 5
        rem = h % 5

        if rem == 0:
            attacks = 3 * q
        else:
            # damages of the first two attacks based on current remainder r
            if r == 0:
                d1, d2 = 1, 1
            elif r == 1:
                d1, d2 = 1, 3
            else:  # r == 2
                d1, d2 = 3, 1

            if d1 >= rem:
                attacks = 3 * q + 1
            elif d1 + d2 >= rem:
                attacks = 3 * q + 2
            else:
                attacks = 3 * (q + 1)

        t += attacks

    print(t)

if __name__ == "__main__":
    solve()
```
