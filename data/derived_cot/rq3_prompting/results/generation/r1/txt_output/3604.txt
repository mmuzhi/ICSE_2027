```python
MOD = 10**9 + 7

class Solution:
    def numberOfWays(self, n: int, x: int, y: int) -> int:
        # Precompute Stirling numbers of the second kind S(n, k) for 0 <= k <= n
        S = [[0] * (n + 1) for _ in range(n + 1)]
        S[0][0] = 1
        for i in range(1, n + 1):
            for j in range(1, i + 1):
                S[i][j] = (S[i-1][j-1] + j * S[i-1][j]) % MOD

        # Precompute factorials and inverse factorials for binomial coefficients up to x
        fact = [1] * (x + 1)
        for i in range(1, x + 1):
            fact[i] = fact[i-1] * i % MOD
        inv_fact = [1] * (x + 1)
        inv_fact[x] = pow(fact[x], MOD-2, MOD)
        for i in range(x, 0, -1):
            inv_fact[i-1] = inv_fact[i] * i % MOD

        def C(nn, kk):
            if kk < 0 or kk > nn:
                return 0
            return fact[nn] * inv_fact[kk] % MOD * inv_fact[nn-kk] % MOD

        # Precompute powers of y
        pow_y = [1] * (min(n, x) + 1)
        for i in range(1, len(pow_y)):
            pow_y[i] = pow_y[i-1] * y % MOD

        ans = 0
        max_k = min(n, x)
        for k in range(1, max_k + 1):
            ways = C(x, k) * fact[k] % MOD
            ways = ways * S[n][k] % MOD
            ways = ways * pow_y[k] % MOD
            ans = (ans + ways) % MOD

        return ans
```
