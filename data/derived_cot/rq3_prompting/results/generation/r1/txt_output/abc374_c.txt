```python
import sys
import bisect

def generate_subset_sums(arr):
    """Return list of all subset sums (including empty set)."""
    n = len(arr)
    sums = []
    for mask in range(1 << n):
        s = 0
        for i in range(n):
            if mask & (1 << i):
                s += arr[i]
        sums.append(s)
    return sums

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = [int(next(it)) for _ in range(N)]

    total = sum(K)
    mid = N // 2
    left = K[:mid]
    right = K[mid:]

    left_sums = generate_subset_sums(left)
    right_sums = generate_subset_sums(right)
    right_sums.sort()

    half = total // 2  # floor(S/2)
    best = 0
    for ls in left_sums:
        if ls > half:
            continue
        target = half - ls
        idx = bisect.bisect_right(right_sums, target) - 1
        if idx >= 0:
            cand = ls + right_sums[idx]
            if cand > best:
                best = cand
                # early stop: we cannot exceed half
                if best == half:
                    break

    answer = total - best
    print(answer)

if __name__ == "__main__":
    solve()
```
