class Solution:
    def maximumValueSum(self, board: List[List[int]]) -> int:
        m = len(board)
        n = len(board[0])
        # dp: dictionary, key: mask (int), value: maximum sum
        dp = {0: 0}
        for i in range(m):
            new_dp = {}
            for mask, val in dp.items():
                # Skip the current row
                if mask in new_dp:
                    new_dp[mask] = max(new_dp[mask], val)
                else:
                    new_dp[mask] = val
                # Place a rook in the current row in a column not used
                # Count the number of set bits in mask
                count = bin(mask).count("1")
                if count < 3:
                    for j in range(n):
                        if mask & (1 << j) == 0:
                            new_mask = mask | (1 << j)
                            new_val = val + board[i][j]
                            # Update new_dp for new_mask
                            if new_mask in new_dp:
                                new_dp[new_mask] = max(new_dp[new_mask], new_val)
                            else:
                                new_dp[new_mask] = new_val
            dp = new_dp
        
        ans = -10**18
        for mask, val in dp.items():
            if bin(mask).count("1") == 3:
                if val > ans:
                    ans = val
        return ans

class Solution:
    def maximumValueSum(self, board: List[List[int]]) -> int:
        m = len(board)
        n = len(board[0])
        dp = {0: 0}
        
        for i in range(m):
            new_dp = {}
            for mask, val in dp.items():
                if mask in new_dp:
                    new_dp[mask] = max(new_dp[mask], val)
                else:
                    new_dp[mask] = val
                
                count = bin(mask).count("1")
                if count < 3:
                    for j in range(n):
                        if not (mask & (1 << j)):
                            new_mask = mask | (1 << j)
                            new_val = val + board[i][j]
                            if new_mask in new_dp:
                                new_dp[new_mask] = max(new_dp[new_mask], new_val)
                            else:
                                new_dp[new_mask] = new_val
            dp = new_dp
        
        ans = -10**18
        for mask, val in dp.items():
            if bin(mask).count("1") == 3:

                    ans = val
