import sys
from collections import deque
import itertools

def main():

    if not data:

    it = iter(data)
    n = int(next(it)); m = int(next(it)); K = int(next(it))
    edges = []
    for _ in range(m):
        u = int(next(it)) - 1; v = int(next(it)) - 1
        edges.append((u, v))
    
    graph = [[] for _ in range(n)]
    for u, v in edges:
        graph[u].append(v)
        
    def get_all_paths():
        paths = []
        stack = [(0, [0], set([0]))]
        while stack:
            node, path, visited = stack.pop()
            if node == n-1:
                paths.append(path)
                continue
            for neighbor in graph[node]:
                if neighbor not in visited:
                    new_path = path + [neighbor]
                    new_visited = visited | {neighbor}
                    stack.append((neighbor, new_path, new_visited))
        return paths

    def solve_case():
        if n == 1:
            return 0
            
        paths = get_all_paths()
        if not paths:
            return 0
            
        max_possible_d = len(paths[0])
        min_possible_d = 0
        
        def feasible(d):
            if d == 0:
                return True
            from collections import defaultdict
            path_edges = []
            for path in paths:
                edge_set = set()
                for i in range(1, len(path)):
                    edge = (path[i-1], path[i])
                    edge_set.add(edge)
                path_edges.append(edge_set)
                
            total_edges = set()
            for path in paths:
                for i in range(1, len(path)):
                    edge = (path[i-1], path[i])
                    total_edges.add(edge)
                    
            if len(total_edges) > K * d:
                return False
                
            from itertools import combinations
            total_edges_list = list(total_edges)
            best = 0
            for comb in combinations(total_edges_list, K):
                count = 0
                for edge_set in path_edges:
                    if any(edge in comb for edge in edge_set):
                        count += 1
                if count == len(paths):
                    best = max(best, d)
            return best >= d

            if feasible(mid):

            else:

        return max_possible_d - 1
