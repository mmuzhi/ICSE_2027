### Solution Code
```python
from collections import defaultdict
from math import comb
from typing import List

MOD = 10**9 + 7

class Solution:
    def subsequencesWithMiddleMode(self, nums: List[int]) -> int:
        n = len(nums)
        if n < 5:
            return 0
        
        total = 0
        
        for i in range(n):
            m = nums[i]
            left_part = nums[:i]
            right_part = nums[i+1:]
            
            count_m_left = left_part.count(m)
            count_m_right = right_part.count(m)
            
            for a in range(0, min(3, count_m_left + 1)):
                for b in range(0, min(3, count_m_right + 1)):
                    freq_m = 1 + a + b
                    ways_m = comb(count_m_left, a) * comb(count_m_right, b) if a <= count_m_left and b <= count_m_right else 0
                    
                    if ways_m == 0:
                        continue
                    
                    left_non_m = [x for x in left_part if x != m]
                    right_non_m = [x for x in right_part if x != m]
                    
                    freq_left_non_m = defaultdict(int)
                    for x in left_non_m:
                        freq_left_non_m[x] += 1
                    
                    freq_right_non_m = defaultdict(int)
                    for x in right_non_m:
                        freq_right_non_m[x] += 1
                    
                    valid = True
                    for x in set(left_non_m + right_non_m):
                        if x == m:
                            continue
                        count_left = freq_left_non_m.get(x, 0)
                        count_right = freq_right_non_m.get(x, 0)
                        if count_left + count_right >= freq_m:
                            valid = False
                            break
                    
                    if valid:
                        ways_non_m_left = comb(len(left_non_m), 2 - a) if 2 - a <= len(left_non_m) else 0
                        ways_non_m_right = comb(len(right_non_m), 2 - b) if 2 - b <= len(right_non_m) else 0
                        total = (total + ways_m * ways_non_m_left * ways_non_m_right) % MOD
        
        return total
```

5. **Validation**: Ensure that the middle element's frequency is strictly greater than any other element's frequency in the subsequence. If valid, increment the total count.
