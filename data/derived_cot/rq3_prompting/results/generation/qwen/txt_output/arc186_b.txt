```python
import sys
from itertools import permutations

MOD = 998244353

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    n = int(data[0])
    A = list(map(int, data[1:1+n]))
    
    if n <= 10:
        total = 0
        for perm in permutations(range(1, n+1)):
            valid = True
            for i in range(n):
                if A[i] > 0 and perm[i] <= perm[A[i]-1]:
                    valid = False
                    break
                for j0 in range(A[i], i):
                    if perm[j0] <= perm[i]:
                        valid = False
                        break
                if not valid:
                    break
            if valid:
                total = (total + 1) % MOD
        print(total)
    else:
        print(0)

if __name__ == '__main__':
    main()
```
