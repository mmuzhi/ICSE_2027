import sys
sys.setrecursionlimit(10000)

class SegmentTree:
    def __init__(self, n):
        self.n = n
        self.size = 1
        while self.size < n:
            self.size *= 2
        self.tree = [0] * (2 * self.size)
        self.lazy = [None] * (2 * self.size)  # None means no lazy update
        
    def build(self, arr):
        # We'll build from an array if provided, but initially all zeros.
        for i in range(self.n):
            self.tree[self.size + i] = arr[i] if i < len(arr) else 0
        for i in range(self.size - 1, 0, -1):
            self.tree[i] = max(self.tree[2*i], self.tree[2*i+1])
            
    def apply(self, i, val):
        # Assign the node i to val.
        self.tree[i] = val
        if i < self.size:
            self.lazy[i] = val
            
    def push(self, i):
        if i < self.size and self.lazy[i] is not None:
            self.apply(2*i, self.lazy[i])
            self.apply(2*i+1, self.lazy[i])
            self.lazy[i] = None
            
    def update_range(self, l, r, val, i=1, segL=0, segR=None):
        if segR is None:
            segR = self.size - 1
        # If the current segment [segL, segR] does not intersect [l, r], do nothing.
        if r < segL or segR < l:
            return
        # If [l, r] covers [segL, segR], then assign.
        if l <= segL and segR <= r:
            self.apply(i, val)
            return
        # Push the lazy value to children.
        self.push(i)
