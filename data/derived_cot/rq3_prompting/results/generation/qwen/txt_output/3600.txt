class Solution:
    def kthCharacter(self, k: int) -> str:
        n = 0
        current_length = 1
        while current_length < k:
            n += 1
            current_length *= 2
        
        def solve(k, n):
            if n == 0:
                return 'a'
            half = 2 ** (n - 1)
            if k <= half:
                return solve(k, n - 1)
            else:
                prev_char = solve(k - half, n - 1)
                if prev_char == 'z':
                    return 'a'
                else:
                    return chr(ord(prev_char) + 1)
        
        return solve(k, n)
