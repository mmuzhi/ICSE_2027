```python
import sys
sys.setrecursionlimit(800000)

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    n = int(next(it)); m = int(next(it)); q = int(next(it))
    people = []
    for _ in range(m):
        s = int(next(it)); t = int(next(it))
        people.append((s, t))
    queries = []
    for _ in range(q):
        l = int(next(it)); r = int(next(it))
        queries.append((l, r))
    
    out_lines = []
    for l, r in queries:
        edges = []
        for i in range(l-1, r):
            s, t = people[i]
            edges.append((s-1, t-1))
        parent = list(range(2*n))
        rank = [0] * (2*n)
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        def union(x, y):
            rx = find(x)
            ry = find(y)
            if rx == ry:
                return
            if rank[rx] < rank[ry]:
                parent[rx] = ry
            elif rank[rx] > rank[ry]:
                parent[ry] = rx
            else:
                parent[ry] = rx
                rank[rx] += 1
        
        for s_node, t_node in edges:
            union(s_node, t_node)
        
        valid = True
        for i in range(l-1, r):
            s, t = people[i]
            if s > t:
                s, t = t, s
            start_node = s-1
            intermediate_nodes = range(s, t)
            comp_start = find(start_node)
            for node in intermediate_nodes:
                if find(node) == comp_start:
                    valid = False
                    break
            if not valid:
                break
        out_lines.append("Yes" if valid else "No")
    
    print('\n'.join(out_lines))

if __name__ == "__main__":
    main()
```
