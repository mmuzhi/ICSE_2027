### Solution Code
```python
import bisect

def main():
    import sys
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    P = list(map(int, data[1:1+n]))
    
    if n == 0:
        print("")
        return
        
    arr = []
    for i in range(1, n+1):
        pos = P[i-1] - 1
        bisect.insort(arr, i, pos)
        
    print(" ".join(map(str, arr)))

    main()
```
