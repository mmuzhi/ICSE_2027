import sys
sys.setrecursionlimit(300000)

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    
    it = iter(data)
    N = int(next(it)); K = int(next(it))
    total_vertices = N * K
    edges = []
    for _ in range(total_vertices - 1):
        u = int(next(it)); v = int(next(it))
        edges.append((u, v))
    
    if K == 1:
        print("Yes")
        return
    elif K == 2:
        n = total_vertices
        graph = [[] for _ in range(n+1)]
        for u, v in edges:
            graph[u].append(v)
            graph[v].append(u)
        
        match = [-1] * (n+1)
        result = True
        
        def dfs(u, parent):
            for v in graph[u]:
                if v == parent:
                    continue
                if match[v] == -1:
                    match[v] = u
                    match[u] = v
                    return True
                else:
                    if dfs(v, u):
                        match[u] = v
                        return True
            return False
        
        for i in range(1, n+1):
            if match[i] == -1:
                if not dfs(i, -1):
                    result = False
                    break
        
        if result:
            print("Yes")
        else:
            print("No")
        return
    else:
        print("No")
