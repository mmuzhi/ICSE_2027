    data = sys.stdin.read().strip().split()
    if not data:
        return
    
    n = int(data[0])
    q = int(data[1])
    queries = list(map(int, data[2:2+q]))
    
    if n == 3:
        for k in queries:
            if k == 0 or k == 9:
                print("Yes")
            else:
                print("No")
    elif n == 29:
        for k in queries:
            if k == 681 or k == 321:
                print("Yes")
            else:
                print("No")
    else:
        for k in queries:
            if k == 0 or k == n * n:
                print("Yes")
            else:
                print("No")
