import heapq
from typing import List

class Solution:
    def maxRemoval(self, nums: List[int], queries: List[List[int]]) -> int:
        n = len(nums)
        m = len(queries)
        queries.sort(key=lambda x: x[1])
        
        heap = []
        j = 0
        used = 0
        
        for i in range(n):
            while j < m and queries[j][0] <= i and queries[j][1] >= i:
                heapq.heappush(heap, -queries[j][1])
                j += 1
            current_cover = len(heap)
            if current_cover < nums[i]:
                return -1
            used += nums[i]
            while heap and nums[i] > 0:
                heapq.heappop(heap)
                nums[i] -= 1
        
        return m - used
