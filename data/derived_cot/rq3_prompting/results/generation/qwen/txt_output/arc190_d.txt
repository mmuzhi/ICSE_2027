3. **Matrix Decomposition**: The matrix B can be decomposed into a fixed part (original A with zeros set to 0) and a variable part (matrices with zeros replaced by values from 1 to p-1). The sum of B^p over all B can then be expressed in terms of these parts.

import sys

def mat_mult(A, B, p):
    n = len(A)
    m = len(B)
    l = len(B[0])
    C = [[0] * l for _ in range(n)]
    for i in range(n):
        for k in range(m):
            if A[i][k] != 0:
                for j in range(l):
                    C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % p
    return C

def mat_power(matrix, power, p):
    n = len(matrix)
    result = [[1 if i == j else 0 for j in range(n)] for i in range(n)]
    base = matrix
    while power > 0:
        if power % 2 == 1:
            result = mat_mult(result, base, p)
        base = mat_mult(base, base, p)
        power = power // 2
    return result
