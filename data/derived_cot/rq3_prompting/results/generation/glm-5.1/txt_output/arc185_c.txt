    from bisect import bisect_right
    
    input = sys.stdin.readline
    N, X = map(int, input().split())
    A = list(map(int, input().split()))
    
    # Build map from value to sorted list of indices
    val_to_indices = {}
    for idx, val in enumerate(A):
        if val not in val_to_indices:
            val_to_indices[val] = []
        val_to_indices[val].append(idx + 1)  # 1-indexed
    
    # Get distinct values
    distinct_vals = list(val_to_indices.keys())
    
    # Iterate over pairs of distinct values
    for a in distinct_vals:
        for b in distinct_vals:
            if a > b:
                continue  # avoid duplicate pairs
            c = X - a - b
            if c < b:
                continue  # ensure a <= b <= c
            if c not in val_to_indices:
                continue
            
            # Check if valid indices exist
            indices_a = val_to_indices[a]
            indices_b = val_to_indices[b]
            indices_c = val_to_indices[c]
            
            # Find smallest i in indices_a
            i = indices_a[0]

            j_pos = bisect_right(indices_b, i)
            if j_pos >= len(indices_b):

            j = indices_b[j_pos]

            k_pos = bisect_right(indices_c, j)
            if k_pos >= len(indices_c):
                continue
            k = indices_c[k_pos]
            
            print(i, j, k)
            return
