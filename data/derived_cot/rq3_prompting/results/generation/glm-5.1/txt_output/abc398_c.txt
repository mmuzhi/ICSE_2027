def main():
    data = sys.stdin.read().split()
    if not data: return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))
    from collections import Counter
    freq = Counter(A)
    max_val = -1

```python

import sys
from collections import Counter

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))

    freq = Counter(A)

    max_val = -1
    ans = -1
    for i, val in enumerate(A, start=1):
        if freq[val] == 1 and val > max_val:
            max_val = val
            ans = i

    print(ans)

if __name__ == "__main__":
    main()
```
