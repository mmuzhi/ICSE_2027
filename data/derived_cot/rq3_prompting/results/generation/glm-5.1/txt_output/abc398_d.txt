import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    N = int(data[0])
    R = int(data[1])
    C = int(data[2])
    S = data[3]
    
    V_r, V_c = 0, 0
    visited = {(0, 0)}
    ans = []
    
    for i in range(N):
        if S[i] == 'N':
            V_r -= 1
        elif S[i] == 'S':
            V_r += 1
        elif S[i] == 'W':
            V_c -= 1
        elif S[i] == 'E':
            V_c += 1
            
        target = (V_r - R, V_c - C)
        if target in visited:
            ans.append('1')
        else:
            ans.append('0')
            
        visited.add((V_r, V_c))
        
    print("".join(ans))

if __name__ == '__main__':
    solve()

import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    R = int(data[1])
    C = int(data[2])
    S = data[3]
    
    V_r, V_c = 0, 0
    visited = {(0, 0)}
    ans = []
    
    for i in range(N):
        if S[i] == 'N':
            V_r -= 1
        elif S[i] == 'S':
            V_r += 1
        elif S[i] == 'W':
            V_c -= 1
        elif S[i] == 'E':
            V_c += 1
            
        target = (V_r - R, V_c - C)
        if target in visited:
            ans.append('1')
        else:
            ans.append('0')
            
        visited.add((V_r, V_c))
        
    print("".join(ans))

if __name__ == '__main__':
    solve()
