Takahashi is about to go buy eel at a fish shop.
The town where he lives is divided into a grid of H rows and W columns. Each cell is either a road or a wall.

- 1\leq H\leq 1000

- 1\leq A,C\leq H
- 1\leq B,D\leq W
- (A,B)\neq (C,D)
- H, W, A, B, C, and D are integers.

def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    H = int(data[0])
    W = int(data[1])
    
    grid = data[2:2+H]
    A = int(data[2+H]) - 1
    B = int(data[3+H]) - 1
    C = int(data[4+H]) - 1
    D = int(data[5+H]) - 1
    
    dist = [[float('inf')] * W for _ in range(H)]
    dq = deque()
    
    dist[A][B] = 0
    dq.append((A, B))
    
    while dq:
        r, c = dq.popleft()
        d = dist[r][c]
        
        if r == C and c == D:
            print(d)
            return
            
        if d > dist[r][c]:
            continue
            
        # Walk to 4 adjacent cells
        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < H and 0 <= nc < W and grid[nr][nc] == '.' and d < dist[nr][nc]:
                dist[nr][nc] = d
                dq.appendleft((nr, nc))
                
        # Kick to 8 cells
        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            for step in [1, 2]:
                nr, nc = r + dr * step, c + dc * step
                if 0 <= nr < H and 0 <= nc < W and d + 1 < dist[nr][nc]:
                    dist[nr][nc] = d + 1
                    dq.append((nr, nc))
