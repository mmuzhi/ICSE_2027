def solve():
    import sys
    from collections import deque
    MOD = 998244353
    input = sys.stdin.readline
    N, M = map(int, input().split())
    A = list(map(int, input().split()))
    A = [x-1 for x in A]
    children = [[] for _ in range(N)]
    for i, p in enumerate(A):
        children[p].append(i)
    indeg = [len(children[i]) for i in range(N)]
    dq = deque([i for i in range(N) if indeg[i] == 0])
    f = [None] * N
    while dq:
        u = dq.popleft()
        # compute f[u]
        prod = [1] * (M+1)
        for child in children[u]:
            fc = f[child]  # must exist
            s = 0
            # iterate v from 1 to M
            for v in range(1, M+1):
                s += fc[v]
                if s >= MOD: s -= MOD
                prod[v] = (prod[v] * s) % MOD
        f[u] = prod
        p = A[u]
        indeg[p] -= 1
        if indeg[p] == 0:
            dq.append(p)
    # identify cycles
    cycle_id = [-1] * N
    cycles = []
    visited = [False] * N
    for i in range(N):
        if indeg[i] > 0 and not visited[i]:
            # traverse to find cycle
            cur = i
            while not visited[cur]:
                visited[cur] = True
                cur = A[cur]
            # cur is start of cycle
            start = cur
            cyc = []
            while True:
                cyc.append(start)
                start = A[start]
                if start == cur:
                    break
            cid = len(cycles)
            for node in cyc:
                cycle_id[node] = cid
            cycles.append(cyc)
    ans = 1
    for cyc in cycles:
        # compute product over nodes in cycle of h_c
        prod_cycle = [1] * (M+1)
        for c in cyc:
            h = [1] * (M+1)
            for child in children[c]:
                if cycle_id[child] == cycle_id[c]:
                    continue
                fc = f[child]
                s = 0
                for v in range(1, M+1):
                    s += fc[v]
                    if s >= MOD: s -= MOD
                    h[v] = (h[v] * s) % MOD
            # multiply into prod_cycle
            for v in range(1, M+1):
                prod_cycle[v] = (prod_cycle[v] * h[v]) % MOD
        contrib = sum(prod_cycle[1:]) % MOD
        ans = ans * contrib % MOD
