        last = [-1] * 26
        
        for i, char in enumerate(s):
            idx = ord(char) - 97
            first[idx] = min(first[idx], i)
            last[idx] = max(last[idx], i)
            
        valid_intervals = set()
        
        for c in range(26):
            if first[c] == n:
                continue
                
            L = first[c]
            R = last[c]
            queue = []
            in_queue = [False] * 26
            
            for i in range(L, R + 1):
                idx = ord(s[i]) - 97
                if not in_queue[idx]:
                    in_queue[idx] = True
                    queue.append(idx)
                    
            while queue:
                curr = queue.pop(0)
                new_L = min(L, first[curr])
                new_R = max(R, last[curr])
                
                if new_L < L:
                    for i in range(new_L, L):
                        idx = ord(s[i]) - 97
                        if not in_queue[idx]:
                            in_queue[idx] = True
                            queue.append(idx)
                            
                if new_R > R:
                    for i in range(R + 1, new_R + 1):
                        idx = ord(s[i]) - 97
                        if not in_queue[idx]:
                            in_queue[idx] = True
                            queue.append(idx)
                            
                L = new_L
                R = new_R
                
            if not (L == 0 and R == n - 1):
                valid_intervals.add((L, R))
                
        count = 0
        last_end = -1
        for L, R in sorted(list(valid_intervals), key=lambda x: x[1]):
            if L > last_end:
                count += 1
                last_end = R
                
        return count >= k
