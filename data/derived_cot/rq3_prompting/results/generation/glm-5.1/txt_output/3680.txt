class Solution:
    def countComponents(self, nums: List[int], threshold: int) -> int:
        parent = list(range(threshold + 1))
        def find(i):
            while parent[i] != i:
                parent[i] = parent[parent[i]]
                i = parent[i]
            return i
        
        def union(i, j):
            root_i = find(i)
            root_j = find(j)
            if root_i != root_j:
                parent[root_i] = root_j
        
        for x in nums:
            if x <= threshold:
                for k in range(2, threshold // x + 1):

        roots = set()

            if x <= threshold:
                roots.add(find(x))
            else:
                roots.add(x) # isolated component, unique ID
                
        return len(roots)

```python
class Solution:
    def countComponents(self, nums: List[int], threshold: int) -> int:
        parent = list(range(threshold + 1))
        size = [1] * (threshold + 1)
        
        def find(i):
            while parent[i] != i:
                parent[i] = parent[parent[i]]
                i = parent[i]
            return i
            
        def union(i, j):
            root_i = find(i)
            root_j = find(j)
            if root_i != root_j:
                if size[root_i] < size[root_j]:
                    root_i, root_j = root_j, root_i
                parent[root_j] = root_i
                size[root_i] += size[root_j]
                
        for x in nums:
            if x <= threshold:
                for k in range(2, threshold // x + 1):
                    union(x, k * x)
                    
        roots = set()
        for x in nums:
            if x <= threshold:
                roots.add(find(x))
            else:
                roots.add(x)
                
        return len(roots)
```

    def countComponents(self, nums: List[int], threshold: int) -> int:
        parent = list(range(threshold + 1))
        
        def find(i):
            while parent[i] != i:
                parent[i] = parent[parent[i]]
                i = parent[i]
            return i
            
        def union(i, j):
            root_i = find(i)
            root_j = find(j)
            if root_i != root_j:
                parent[root_j] = root_i

            if x <= threshold:

                    union(x, m)
                    
        roots = set()
        for x in nums:
            if x <= threshold:
                roots.add(find(x))
            else:
                roots.add(x)
                
        return len(roots)

            root_i = find(i)
            root_j = find(j)
            if root_i != root_j:
                if size[root_i] < size[root_j]:
                    root_i, root_j = root_j, root_i

                for m in range(2 * x, threshold + 1, x):
                    union(x, m)

        for x in nums:
            if x <= threshold:
                roots.add(find(x))
            else:
                roots.add(x)
                
        return len(roots)
