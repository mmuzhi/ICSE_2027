```python

import sys

def main() -> None:
    S = sys.stdin.readline().strip()
    T = sys.stdin.readline().strip()
    n = max(len(S), len(T))
    for i in range(n):
        if i < len(S) and i < len(T):
            if S[i] != T[i]:
                print(i + 1)
                return
        else:
            # one string has already ended
            print(i + 1)
            return
    # strings are completely equal
    print(0)

if __name__ == "__main__":
    main()
```
