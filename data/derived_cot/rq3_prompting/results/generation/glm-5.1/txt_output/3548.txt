class Solution:
    def countGoodIntegers(self, n: int, k: int) -> int:
        L = (n + 1) // 2
        start = 10**(L - 1)
        end = 10**L
        
        valid_multisets = set()
        
        for first_half in range(start, end):
            s = str(first_half)
            p_str = s + s[::-1][n % 2:]
            p = int(p_str)
            
            if p % k == 0:
                count = [0] * 10
                for char in p_str:
                    count[int(char)] += 1
                valid_multisets.add(tuple(count))

            val = math.factorial(n - 1) * (n - count[0])
            for c in count:
                val //= math.factorial(c)
            total_good += val
            
        return total_good

            val = facts[n - 1] * (n - count[0]) // denom
            total_good += val
            
        return total_good
