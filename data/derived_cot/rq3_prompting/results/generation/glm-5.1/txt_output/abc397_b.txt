import sys

def solve() -> None:
    S = sys.stdin.readline().strip()
    cur = 0  # position of the last matched character in the pattern
    for ch in S:
        if ch == 'i':
            # need an odd position > cur
            if cur % 2 == 0:
                cur += 1   # cur even -> cur+1 is odd
            else:
                cur += 2   # cur odd  -> cur+2 is odd
        else:  # ch == 'o'
            # need an even position > cur
            if cur % 2 == 1:
                cur += 1   # cur odd  -> cur+1 is even
            else:
                cur += 2   # cur even -> cur+2 is even
    # total insertions = (skipped positions) + (1 if final length must be even)
    ans = cur - len(S) + (cur % 2)
    print(ans)

if __name__ == "__main__":
