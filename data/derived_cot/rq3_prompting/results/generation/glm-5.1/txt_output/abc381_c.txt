import sys

def main():
    input = sys.stdin.readline
    N = int(input().strip())
    S = input().strip()

    # left[i] = length of consecutive '1's ending at i (inclusive)
    left = [0] * N
    for i, ch in enumerate(S):
        if ch == '1':
            left[i] = left[i-1] + 1 if i > 0 else 1
        else:
            left[i] = 0

    # right[i] = length of consecutive '2's starting at i (inclusive)
    right = [0] * N
    for i in range(N-1, -1, -1):
        if S[i] == '2':
            right[i] = right[i+1] + 1 if i + 1 < N else 1
        else:
            right[i] = 0

    ans = 1  # at least the single '/' substring exists
    for i, ch in enumerate(S):
        if ch == '/':
            L = left[i-1] if i > 0 else 0
            R = right[i+1] if i + 1 < N else 0
            k = min(L, R)
            length = 2 * k + 1
            if length > ans:

    print(ans)

if __name__ == "__main__":
