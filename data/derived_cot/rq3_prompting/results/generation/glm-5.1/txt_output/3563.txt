        R = len(grid)
        row_sets = [set(row) for row in grid]

class Solution:

        R = len(grid)
        row_sets = [set(row) for row in grid]

        dp = [-1] * (1 << R)
        dp[0] = 0
        
        # Process values in descending order
        for v in range(100, 0, -1):
            # 0-1 knapsack style iteration over masks to ensure each value is used at most once
            for mask in range((1 << R) - 1, 0, -1):
                for r in rows_with_v[v]:
                    if mask & (1 << r):
                        if dp[mask ^ (1 << r)] != -1:
