=== ANSWER ===
```python

def main():
    S = input().strip()
    # Split S by '|'. Since S starts and ends with '|', the first and last elements will be empty strings.
    # The segments in between correspond to the '-' blocks, whose lengths form the sequence A.
    parts = S.split('|')[1:-1]
    ans = [str(len(p)) for p in parts]
    print(' '.join(ans))

if __name__ == '__main__':
    main()
```
