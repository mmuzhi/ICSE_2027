import sys

def main():
    data = sys.stdin.read().split()
    if not data: return
    n = int(data[0])
    A = list(map(int, data[1:]))
    last = [0]*(n+1)  # since values up to n
    ans = 0
    for i, v in enumerate(A, start=1):
        prev = last[v]
        ans += (i - prev) * (n - i + 1)
        last[v] = i
    print(ans)

if __name__ == "__main__":
    main()

import sys

def main():

    if not data:

    n = int(data[0])
