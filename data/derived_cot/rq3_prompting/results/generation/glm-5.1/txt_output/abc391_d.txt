import sys

def main():
    import sys
    input = sys.stdin.readline
    N, W = map(int, input().split())
    columns = [[] for _ in range(W+1)]
    for i in range(1, N+1):
        x, y = map(int, input().split())
        columns[x].append((y, i))
    # compute c_x and K
    c = [0]*(W+1)
    for x in range(1, W+1):
        c[x] = len(columns[x])
    K = min(c[1:]) if W>0 else 0  # but W>=1
    if K == 0:
        # no removals ever
        Q = int(input())
        out = []
        for _ in range(Q):
            T, A = map(int, input().split())
            out.append("Yes")
        sys.stdout.write("\n".join(out))
        return

    M = [0]*(K+1)  # 1-indexed
    rank = [0]*(N+1)

    for x in range(1, W+1):
        col = columns[x]
        col.sort(key=lambda t: t[0])  # sort by y
        for idx, (y, bid) in enumerate(col, start=1):
            rank[bid] = idx
            if idx <= K:
                d = y - idx
                if d > M[idx]:
                    M[idx] = d

    # compute R
    R = [0]*(K+1)
    D = 0
    R_prev = 0
    for k in range(1, K+1):
        Wk = max(0, M[k] - D)
        R_curr = R_prev + Wk + 1
        D += Wk
        R[k] = R_curr
        R_prev = R_curr

    Q = int(input())
    out = []
    for _ in range(Q):
        T, A = map(int, input().split())
        r = rank[A]
        if r > K:
            out.append("Yes")
        else:
            out.append("Yes" if T < R[r] else "No")
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
