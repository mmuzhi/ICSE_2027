You will be given a competitive programming problem.
There are N foods, each containing exactly one of vitamins 1, 2, and 3.
Specifically, eating the i-th food gives you A_i units of vitamin V_i, and C_i calories.
Takahashi can choose any subset of these N foods as long as the total calorie consumption does not exceed X.
Find the maximum possible value of this: the minimum intake among vitamins 1, 2, and 3.

Input

The input is given from Standard Input in the following format:
N X
V_1 A_1 C_1
V_2 A_2 C_2
\vdots
V_N A_N C_N

Output

Print the maximum possible value of "the minimum intake among vitamins 1, 2, and 3" when the total calories consumed is at most X.

Constraints


- 1 \leq N \leq 5000
- 1 \leq X \leq 5000
- 1 \leq V_i \leq 3
- 1 \leq A_i \leq 2 \times 10^5
- 1 \leq C_i \leq X
- All input values are integers.

Sample Input 1

5 25
1 8 5
2 3 5
2 7 10
3 2 5
3 3 10

Sample Output 1

3

- 1st food: 8 units of vitamin 1, and 5 calories
- 2nd food: 3 units of vitamin 2, and 5 calories
- 3rd food: 7 units of vitamin 2, and 10 calories
- 4th food: 2 units of vitamin 3, and 5 calories
- 5th food: 3 units of vitamin 3, and 10 calories

Eating the 1st, 2nd, 4th, and 5th foods gives 8 units of vitamin 1, 3 units of vitamin 2, 5 units of vitamin 3, and 25 calories.
In this case, the minimum among the three vitamin intakes is 3 (vitamin 2).
It is impossible to get 4 or more units of each vitamin without exceeding 25 calories, so the answer is 3.

import sys

def main():
    import sys
    input_data = sys.stdin.buffer.read().split()
    it = iter(input_data)
    N = int(next(it))
    X = int(next(it))
    items = [[] for _ in range(3)]
    for _ in range(N):
        v = int(next(it)) - 1  # 0-index
        a = int(next(it))
        c = int(next(it))
        items[v].append((a, c))
    
    def compute_best(itms):
        dp = [-1] * (X+1)
        dp[0] = 0
        for a, c in itms:
            # 0/1 knapsack
            for cost in range(X, c-1, -1):
                prev = dp[cost-c]
                if prev != -1:
                    val = prev + a
                    if val > dp[cost]:
                        dp[cost] = val
        best = [0] * (X+1)
        cur = 0
        for i in range(X+1):
            if dp[i] > cur:
                cur = dp[i]
            best[i] = cur
        return best
    
    best0 = compute_best(items[0])
    best1 = compute_best(items[1])
    best2 = compute_best(items[2])
    
    upper = min(best0[X], best1[X], best2[X])
    
    lo, hi = 0, upper
    while lo < hi:
        mid = (lo + hi + 1) // 2
        total = bisect.bisect_left(best0, mid) + bisect.bisect_left(best1, mid) + bisect.bisect_left(best2, mid)

import sys
import bisect

def main() -> None:
    it = iter(sys.stdin.buffer.read().split())
    N = int(next(it))
    X = int(next(it))

    # items[v] : list of (amount, cost) for vitamin v (0-indexed)
    items = [[] for _ in range(3)]
    for _ in range(N):
        v = int(next(it)) - 1   # 0,1,2
        a = int(next(it))
        c = int(next(it))
        items[v].append((a, c))

    # compute best[c] = maximum total amount of this vitamin with cost ≤ c
    def compute_best(itms):
        dp = [-1] * (X + 1)
        dp[0] = 0
        for a, c in itms:
            # 0/1 knapsack, iterate backwards
            for cost in range(X, c - 1, -1):
                prev = dp[cost - c]
                if prev != -1:
                    val = prev + a
                    if val > dp[cost]:
                        dp[cost] = val
        best = [0] * (X + 1)
        cur = 0
        for i in range(X + 1):
            if dp[i] > cur:
                cur = dp[i]
            best[i] = cur
        return best

    best0 = compute_best(items[0])
    best1 = compute_best(items[1])
    best2 = compute_best(items[2])

    # upper bound: even if we spend all calories on one vitamin,
    # we cannot exceed the minimum of the three maxima
    upper = min(best0[X], best1[X], best2[X])

    lo, hi = 0, upper
    while lo < hi:
        mid = (lo + hi + 1) // 2
        total = (bisect.bisect_left(best0, mid) +
                 bisect.bisect_left(best1, mid) +
                 bisect.bisect_left(best2, mid))
        if total <= X:
            lo = mid
        else:
            hi = mid - 1

    print(lo)

if __name__ == "__main__":
    main()
