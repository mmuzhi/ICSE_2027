```python
class Solution:
    def minOperations(self, nums: List[int]) -> int:
        max_val = max(nums) if nums else 0
        spf = list(range(max_val + 1))
        for i in range(2, int(max_val**0.5) + 1):
            if spf[i] == i:
                for j in range(i*i, max_val + 1, i):
                    if spf[j] == j:
                        spf[j] = i
        
        ops = 0
        limit = float('inf')
        for x in reversed(nums):
            A = x
            B = spf[x]
            if A <= limit:
                limit = A
            elif B <= limit:
                limit = B
                ops += 1
            else:
                return -1
        return ops
```

```python
class Solution:
    def minOperations(self, nums: List[int]) -> int:
        if not nums:
            return 0
            
        max_val = max(nums)
        spf = list(range(max_val + 1))
        
        # Sieve of Eratosthenes to find the smallest prime factor (SPF) for every number
        for i in range(2, int(max_val**0.5) + 1):
            if spf[i] == i:
                for j in range(i*i, max_val + 1, i):
                    if spf[j] == j:
                        spf[j] = i
                        
        ops = 0
        limit = float('inf')
        
        # Process from right to left
        for x in reversed(nums):
            A = x
            B = spf[x]
            
            if A <= limit:
                # Prefer the original value to minimize operations and maximize the limit for the left elements
                limit = A
            elif B <= limit:
                # Must divide by greatest proper divisor (which reduces it to its SPF)
                limit = B
                ops += 1
            else:
                # Neither the original value nor the reduced value can satisfy the non-decreasing condition
                return -1
                
        return ops
```
