    MAX_PIECES = 60

        if b == 1:
            return [(0, 1, a)]
        else:
            t = a / (b - 1)
            if t <= 0:
                return [(0, b, 0)]
            else:
                return [(0, 1, a), (t, b, 0)]

        result = f[0]
        for i in range(len(f)):
            if v >= f[i][0]:
                result = f[i]
            else:
                break
        return result[1] * v + result[2]
    
    def compose(f1, f2):
        result = []
        
        for i in range(len(f1)):
            t1, m1, a1 = f1[i]
            t1_next = f1[i + 1][0] if i + 1 < len(f1) else float('inf')
            
            v_start = m1 * t1 + a1
            v_end = m1 * t1_next + a1 if t1_next != float('inf') else float('inf')
            
            for j in range(len(f2)):

                t2_next = f2[j + 1][0] if j + 1 < len(f2) else float('inf')
                
                if v_start >= t2_next:
                    continue
                if v_end != float('inf') and v_end <= t2:
                    continue

                overlap_v_end = v_end
                if t2_next != float('inf'):

                v_max = t1_next
                if overlap_v_end != float('inf'):
                    v_max = min(v_max, (overlap_v_end - a1) / m1)
                
                if v_min < v_max:
                    result.append((v_min, m2 * m1, m2 * a1 + a2))
        
        result.sort()
        
        if not result:
            return [(0, 1, 0)]
        
        merged = [result[0]]
        for i in range(1, len(result)):
            if abs(result[i][1] - merged[-1][1]) < 1e-9 and abs(result[i][2] - merged[-1][2]) < 1e-9:
                pass
            else:
                merged.append(result[i])

Looking at this problem, I need to process queries on two arrays A and B where for type 3 queries, I maximize v by choosing v = v + A[i] or v = v * B[i] at each step.

**Key insight**: The greedy strategy is optimal - at each step, choose the operation giving the larger result. This is because both operations are monotonically increasing in v, so maximizing v at each step maximizes future options.

**Approach**: Use a segment tree where each node stores a piecewise linear function f(v) representing the output given input v. Since the answer is bounded by 10^18, the number of "breakpoints" (where we switch from adding to multiplying) is bounded by ~60.

Each function is represented as a sorted list of (threshold, multiplier, addend) tuples where f(v) = m*v + a for v >= t.

For composition f2(f1(v)), I find which pieces of f2 are active for the range of f1's output and compute the resulting piecewise function.

import sys
input = sys.stdin.readline

def solve():
    N = int(input())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    Q = int(input())
    
    MAX_PIECES = 60
    
    def make_leaf(a, b):
        if b == 1:
            return [(0, 1, a)]
        t = a / (b - 1)
        if t <= 0:
            return [(0, b, 0)]
        return [(0, 1, a), (t, b, 0)]
    
    def eval_func(f, v):
        result = f[0]
        for i in range(len(f)):
            if v >= f[i][0]:
                result = f[i]
            else:
                break
        return result[1] * v + result[2]
    
    def compose(f1, f2):
        result = []
        for i in range(len(f1)):
            t1, m1, a1 = f1[i]
            t1_next = f1[i + 1][0] if i + 1 < len(f1) else float('inf')

            v_end = m1 * t1_next + a1 if t1_next != float('inf') else float('inf')

                    continue

                ov_end = min(v_end, t2_next) if v_end != float('inf') and t2_next != float('inf') else min(v_end, t2_next)
