import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = [input().strip() for _ in range(N)]
    N1 = N + 1                     # N+1, used in rotation formulas
    out_lines = []

    for i in range(N):
        row_chars = []
        im = N - 1 - i             # N-1-i, constant for this row
        for j in range(N):
            jm = N - 1 - j         # N-1-j
            d = min(i, j, im, jm)  # depth of the cell (0‑indexed)
            k = (d + 1) & 3        # (d+1) mod 4, number of rotations modulo 4
            fr = i + 1             # 1‑indexed row of the output cell
            fc = j + 1             # 1‑indexed column of the output cell

            if k == 0:             # R^0
                r, c = fr, fc
            elif k == 1:           # R^{-1} = R^3
                r, c = N1 - fc, fr
            elif k == 2:           # R^{-2} = R^2
                r, c = N1 - fr, N1 - fc
            else:                  # k == 3, R^{-3} = R^1
                r, c = fc, N1 - fr

            # r,c are 1‑indexed; convert to 0‑indexed and fetch from input
            row_chars.append(A[r - 1][c - 1])

        out_lines.append(''.join(row_chars))

    sys.stdout.write('\n'.join(out_lines))

if __name__ == "__main__":
    solve()
