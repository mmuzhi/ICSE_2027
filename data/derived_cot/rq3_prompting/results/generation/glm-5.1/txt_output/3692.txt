```python
def find_occurrences(text, pattern):
    if not pattern:
        return list(range(len(text) + 1))
    n, m = len(text), len(pattern)
    if m > n:
        return []
    
    # Compute LPS array
    lps = [0] * m
    length = 0
    i = 1
    while i < m:
        if pattern[i] == pattern[length]:
            length += 1
            lps[i] = length
            i += 1
        else:
            if length != 0:
                length = lps[length - 1]
            else:
                lps[i] = 0
                i += 1
                
    # Search
    res = []
    i = 0 # index for text
    j = 0 # index for pattern
    while i < n:
        if pattern[j] == text[i]:
            i += 1
            j += 1
        if j == m:
            res.append(i - j)
            j = lps[j - 1]
        elif i < n and pattern[j] != text[i]:
            if j != 0:
                j = lps[j - 1]
            else:
                i += 1
    return res
```
This is O(N + M) and perfectly safe.

```python
import bisect

class Solution:
    def shortestMatchingSubstring(self, s: str, p: str) -> int:
        A, B, C = p.split('*')
        
        def find_occurrences(text, pattern):
            if not pattern:
                return list(range(len(text) + 1))
            n, m = len(text), len(pattern)
            if m > n:
                return []
            
            lps = [0] * m
            length = 0
            i = 1
            while i < m:
                if pattern[i] == pattern[length]:
                    length += 1
                    lps[i] = length
                    i += 1
                else:
                    if length != 0:
                        length = lps[length - 1]
                    else:
                        lps[i] = 0
                        i += 1
                        
            res = []
            i = 0
            j = 0
            while i < n:
                if pattern[j] == text[i]:
                    i += 1
                    j += 1
                    if j == m:
                        res.append(i - j)
                        j = lps[j - 1]
                else:
                    if j != 0:
                        j = lps[j - 1]
                    else:
                        i += 1
            return res

        A_pos = find_occurrences(s, A)
        B_pos = find_occurrences(s, B)
        C_pos = find_occurrences(s, C)
        
        if not A_pos or not B_pos or not C_pos:
            return -1
            
        min_len = float('inf')
        len_A = len(A)
        len_B = len(B)
        len_C = len(C)
        
        for b in B_pos:
            # find largest a in A_pos such that a + len_A <= b
            # i.e., a <= b - len_A
            idx_a = bisect.bisect_right(A_pos, b - len_A) - 1
            if idx_a < 0:
                continue
            a_val = A_pos[idx_a]
            
            # find smallest c in C_pos such that c >= b + len_B
            idx_c = bisect.bisect_left(C_pos, b + len_B)
            if idx_c == len(C_pos):
                continue
            c_val = C_pos[idx_c]
            
            current_len = c_val + len_C - a_val
            if current_len < min_len:
                min_len = current_len
                
        return -1 if min_len == float('inf') else min_len
```

                    if length != 0:
                        length = lps[length - 1]
                    else:
                        lps[i] = 0
                        i += 1

            i = 0
            j = 0
            while i < n:
                if pattern[j] == text[i]:
                    i += 1
                    j += 1
                    if j == m:
                        res.append(i - j)
                        j = lps[j - 1]
                else:
                    if j != 0:
                        j = lps[j - 1]
                    else:
