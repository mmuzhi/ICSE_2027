=== ANSWER ===
```python
def solve():
    N, D = map(int, input().split())
    S = list(input())
    cnt = 0
    for i in range(N - 1, -1, -1):
        if S[i] == '@':
            cnt += 1
            if cnt <= D:
                S[i] = '.'
    print("".join(S))

if __name__ == "__main__":
    solve()
```
