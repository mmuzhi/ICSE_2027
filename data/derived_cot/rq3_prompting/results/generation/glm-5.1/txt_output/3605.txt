```python
class Solution:
    def minBitwiseArray(self, nums: List[int]) -> List[int]:
        ans = []
        for num in nums:
            if num % 2 == 0:
                ans.append(-1)
            else:
                ans.append(num ^ (((num + 1) & -(num + 1)) >> 1))
        return ans
```

```python
class Solution:
    def minBitwiseArray(self, nums: List[int]) -> List[int]:
        ans = []
        for num in nums:
            if num % 2 == 0:
                ans.append(-1)
            else:
                # For an odd num, it ends in at least one 1.
                # (num + 1) & -(num + 1) isolates the lowest set bit of num + 1.
                # Shifting it right by 1 gives the highest power of 2 among the trailing 1s of num.
                # XORing num with this value flips that 1 to 0, yielding the minimum valid ans[i].
                ans.append(num ^ (((num + 1) & -(num + 1)) >> 1))
        return ans
```
