```python
class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
        MOD = 10**9 + 7
        cnt = [0] * 26
        for ch in s:
            cnt[ord(ch) - ord('a')] += 1
        
        for _ in range(t):
            new_cnt = [0] * 26
            # z -> ab
            new_cnt[0] = cnt[25]
            new_cnt[1] = (cnt[0] + cnt[25]) % MOD
            # others shift
            for i in range(2, 26):
                new_cnt[i] = cnt[i-1]
            cnt = new_cnt
        
        return sum(cnt) % MOD
```

```python
class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
        MOD = 10**9 + 7
        cnt = [0] * 26
        for ch in s:
            cnt[ord(ch) - ord('a')] += 1

        for _ in range(t):
            new_cnt = [0] * 26
            # z -> "ab"
            new_cnt[0] = cnt[25]
            new_cnt[1] = (cnt[0] + cnt[25]) % MOD
            # other letters shift by one
            for i in range(2, 26):
                new_cnt[i] = cnt[i-1]
            cnt = new_cnt

        return sum(cnt) % MOD
```
