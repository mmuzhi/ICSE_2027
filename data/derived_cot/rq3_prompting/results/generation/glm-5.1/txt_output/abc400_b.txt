import sys

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    N = int(input_data[0])
    M = int(input_data[1])
    
    ans = 0
    p = 1
    limit = 10**9
    
    for _ in range(M + 1):
        ans += p
        if ans > limit:
            print("inf")
            return
        p *= N
        
    print(ans)

if __name__ == '__main__':
    solve()
