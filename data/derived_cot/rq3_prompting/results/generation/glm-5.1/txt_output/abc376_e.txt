import sys
import heapq

def solve() -> None:
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    t = next(it)
    out_lines = []

    for _ in range(t):
        n = next(it)
        k = next(it)
        A = [next(it) for _ in range(n)]
        B = [next(it) for _ in range(n)]

        # K = 1 is a special case
        if k == 1:
            ans = min(a * b for a, b in zip(A, B))
            out_lines.append(str(ans))
            continue

        # sort by A
        pairs = list(zip(A, B))
        pairs.sort(key=lambda x: x[0])

        m = k - 1                     # number of additional elements
        heap = []                     # max‑heap (via negatives) of the m smallest B
        total = 0                     # sum of elements in the heap
        ans = 10**30

        for i, (a, b) in enumerate(pairs):
            if i >= m:                # enough previous elements to form a set
                candidate = a * (b + total)
                if candidate < ans:
                    ans = candidate

            # insert current B into the structure for future indices
            heapq.heappush(heap, -b)
            total += b
            if len(heap) > m:         # keep only the m smallest
                removed = -heapq.heappop(heap)
                total -= removed

        out_lines.append(str(ans))

    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
