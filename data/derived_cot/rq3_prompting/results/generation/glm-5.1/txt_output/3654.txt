Let's write down the exact DP logic to avoid any silly bugs.
```python
class Solution:
    def minArraySum(self, nums: List[int], k: int, op1: int, op2: int) -> int:
        dp = [[float('inf')] * (op2 + 1) for _ in range(op1 + 1)]
        dp[0][0] = 0
        
        for x in nums:
            costs = [(0, 0, x)]
            costs.append((1, 0, (x + 1) // 2))
            if x >= k:
                costs.append((0, 1, x - k))
                costs.append((1, 1, (x - k + 1) // 2))  # op2 then op1
                if (x + 1) // 2 >= k:
                    costs.append((1, 1, (x + 1) // 2 - k))  # op1 then op2
            
            new_dp = [row[:] for row in dp]
            for o1 in range(op1 + 1):
                for o2 in range(op2 + 1):
                    if dp[o1][o2] == float('inf'):
                        continue
                    for c1, c2, val in costs:
                        if o1 + c1 <= op1 and o2 + c2 <= op2:
                            if dp[o1][o2] + val < new_dp[o1 + c1][o2 + c2]:
                                new_dp[o1 + c1][o2 + c2] = dp[o1][o2] + val
            dp = new_dp
            
        return min(dp[o1][o2] for o1 in range(op1 + 1) for o2 in range(op2 + 1))
```
