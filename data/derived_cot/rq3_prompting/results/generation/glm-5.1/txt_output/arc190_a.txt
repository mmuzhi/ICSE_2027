def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    L = [0]*M
    R = [0]*M
    for i in range(M):
        l, r = map(int, input().split())
        L[i] = l
        R[i] = r

    # Step 1: K=1
    for i in range(M):
        if L[i] == 1 and R[i] == N:
            print(1)
            ops = [0]*M
            ops[i] = 1
            print(' '.join(map(str, ops)))
            return

    # Prepare sorted list for disjoint and containment
    intervals = [(L[i], R[i], i) for i in range(M)]
    intervals.sort(key=lambda x: (x[0], -x[1]))  # L asc, R desc

    # Step 2: disjoint pair
    # iterate to find L > max_R
    if M >= 2:
        max_R = intervals[0][1]
        idx_max = intervals[0][2]
        for k in range(1, M):
            cur_L, cur_R, cur_idx = intervals[k]
            if cur_L > max_R:
                # disjoint pair found
                print(2)
                ops = [0]*M
                ops[idx_max] = 2
                ops[cur_idx] = 2
                print(' '.join(map(str, ops)))
                return
            else:
                if cur_R > max_R:
                    max_R = cur_R
                    idx_max = cur_idx

    # Step 3: containment pair
    # using same sorted order
    max_R = -1
    idx_max = -1
    for cur_L, cur_R, cur_idx in intervals:
        if max_R != -1 and cur_R <= max_R:
            # containment: container = idx_max, contained = cur_idx
            print(2)
            ops = [0]*M
            # container should be mode1, contained mode2
            ops[idx_max] = 1
            ops[cur_idx] = 2
            print(' '.join(map(str, ops)))
            return
        else:
            if cur_R > max_R:
                max_R = cur_R
                idx_max = cur_idx

    # Step 4: union covering whole line
    best_L1_R = -1
    idx_L1 = -1
    best_RN_L = N+1
    idx_RN = -1
    for i in range(M):
        if L[i] == 1:
            if R[i] > best_L1_R:
                best_L1_R = R[i]
                idx_L1 = i
        if R[i] == N:
            if L[i] < best_RN_L:
                best_RN_L = L[i]
                idx_RN = i
    if idx_L1 != -1 and idx_RN != -1 and idx_L1 != idx_RN and best_L1_R >= best_RN_L:
        print(2)
        ops = [0]*M
        ops[idx_L1] = 1
        ops[idx_RN] = 1
        print(' '.join(map(str, ops)))
        return

    # Step 5: K=3 if M>=3
    if M >= 3:
        # find idx_minL and idx_maxR
        minL = N+1
        idx_minL = -1
        maxR = -1
        idx_maxR = -1
        for i in range(M):
            if L[i] < minL:
                minL = L[i]
                idx_minL = i
            if R[i] > maxR:
                maxR = R[i]
                idx_maxR = i
        # they should be distinct, but just in case if they are same (shouldn't happen), we need to handle? but if same, then containment would have been caught, so we can assert distinct.
        # find a third index not equal to these two
        idx_third = -1
        for i in range(M):
            if i != idx_minL and i != idx_maxR:
                idx_third = i
                break
        # It must exist because M>=3
        print(3)
        ops = [0]*M
        ops[idx_minL] = 1
        ops[idx_maxR] = 1
        ops[idx_third] = 2
        print(' '.join(map(str, ops)))
        return

    # Step 6: impossible
    print(-1)
