- For every vertex v with a degree of at least 1, the total XOR of the numbers written on its adjacent vertices (excluding v itself) is 0.


What is XOR?

The XOR of two non-negative integers A and B, denoted as A \oplus B, is defined as follows:


- In the binary representation of A \oplus B, the bit at position 2^k \, (k \geq 0) is 1 if and only if exactly one of the bits at position 2^k in the binary representations of A and B is 1. Otherwise, it is 0.


For example, 3 \oplus 5 = 6 (in binary: 011 \oplus 101 = 110).

In general, the bitwise XOR of k integers p_1, \dots, p_k is defined as (\cdots ((p_1 \oplus p_2) \oplus p_3) \oplus \cdots \oplus p_k).  It can be proved that this is independent of the order of p_1, \dots, p_k.

Otherwise, let X_v be the integer written on vertex v, and print your solution in the following format. If multiple solutions exist, any of them will be accepted.
Yes

- 1 \leq u_i < v_i \leq N

Input: $N, M$, then $M$ edges.
Adjacency matrix $A$ of size $N \times N$.
Since it's undirected, $A[u][v] = A[v][u] = 1$.

Since we are in $\mathbb{F}_2$, we can use bitwise operations to speed it up, but $N=60$ is so small that simple loops are perfectly fine and less error-prone.
Wait, $N=60$ means we can just use a list of integers for rows, where each integer is the bitmask of the row.
This makes Gaussian elimination very fast and easy to implement!
Let's use a list of integers `mat` for the rows of $A$.
`mat[u] |= 1 << v`
`mat[v] |= 1 << u`
Then Gaussian elimination:
```python
rank = 0
pivots = [-1] * N
for col in range(N):
    pivot_row = -1
    for row in range(rank, N):
        if (mat[row] >> col) & 1:
            pivot_row = row
            break
    if pivot_row == -1:
        continue
```

```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    
    mat = [0] * N
    for i in range(M):
        u = int(data[2 + 2*i]) - 1
        v = int(data[2 + 2*i + 1]) - 1
        mat[u] |= 1 << v
        mat[v] |= 1 << u
        
    rank = 0
    pivots = [-1] * N
    for col in range(N):
        pivot_row = -1
        for row in range(rank, N):
            if (mat[row] >> col) & 1:
                pivot_row = row
                break
        if pivot_row == -1:
            continue
        mat[rank], mat[pivot_row] = mat[pivot_row], mat[rank]
        for row in range(N):
            if row != rank and (mat[row] >> col) & 1:
                mat[row] ^= mat[rank]
        pivots[rank] = col
        rank += 1
        
    basis = []
    non_pivot_cols = []
    for col in range(N):
        is_pivot = False
        for i in range(rank):
            if pivots[i] == col:
                is_pivot = True
                break
        if not is_pivot:
            non_pivot_cols.append(col)
            
    for j in non_pivot_cols:
        x = 0

        for i in range(rank - 1, -1, -1):
            p_i = pivots[i]
            val = bin(mat[i] & x).count('1') % 2
            if val:
                x |= 1 << p_i
        basis.append(x)
        
    if not basis:

    covered = 0
```

What if the graph has multiple connected components?
The RREF and null space naturally handle disconnected graphs because $A$ is block diagonal (up to permutation), and the null space is the direct sum of the null spaces of the blocks.
The condition `covered == (1 << N) - 1` checks if EVERY vertex is covered by the null space.
If one component has no non-trivial null space vectors covering its vertices (e.g., a single edge), those vertices won't be covered, and we correctly output "No".
This is exactly what we want.

```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    
    # Adjacency matrix over F2 represented as a list of bitmasks
    mat = [0] * N
    for i in range(M):
        u = int(data[2 + 2*i]) - 1
        v = int(data[2 + 2*i + 1]) - 1
        mat[u] |= 1 << v
        mat[v] |= 1 << u
        
    # Gaussian Elimination to find RREF of A
    rank = 0
    pivots = [-1] * N
    for col in range(N):
        pivot_row = -1
        for row in range(rank, N):
            if (mat[row] >> col) & 1:
                pivot_row = row
                break
        if pivot_row == -1:
            continue
        mat[rank], mat[pivot_row] = mat[pivot_row], mat[rank]
        for row in range(N):
            if row != rank and (mat[row] >> col) & 1:
                mat[row] ^= mat[rank]
        pivots[rank] = col
        rank += 1
        
    # Find a basis for the null space of A from its RREF
    basis = []
    non_pivot_cols = []
    for col in range(N):
        is_pivot = False
        for i in range(rank):
            if pivots[i] == col:
                is_pivot = True
                break
        if not is_pivot:
            non_pivot_cols.append(col)
            
    for j in non_pivot_cols:
        x = 0
        x |= 1 << j
        # Compute the null space vector corresponding to the free variable x_j
        for i in range(rank - 1, -1, -1):
            p_i = pivots[i]
            # x_{p_i} = sum_{k > p_i} R_{i, k} x_k
            # Since R is in RREF, R_{i, k} = 0 for k < p_i. 
            # The dot product of mat[i] and x exactly gives the sum over k > p_i 
            # because x_{p_i} is initially 0.
            val = bin(mat[i] & x).count('1') % 2
            if val:
                x |= 1 << p_i
        basis.append(x)
        
    # If the null space is empty, we can only have X = 0, which violates X_v >= 1
    if not basis:
        print("No")
        return
        
    # Check if the union of supports of the basis vectors covers all vertices
    covered = 0
    for b in basis:
        covered |= b
        
    if covered != (1 << N) - 1:
        print("No")
        return
        
    # Construct the solution X by assigning each basis vector to a distinct bit position
    X = [0] * N
    for i, b in enumerate(basis):
        for v in range(N):
            if (b >> v) & 1:
                X[v] |= 1 << i
                
    print("Yes")
    print(" ".join(map(str, X)))

solve()
```
