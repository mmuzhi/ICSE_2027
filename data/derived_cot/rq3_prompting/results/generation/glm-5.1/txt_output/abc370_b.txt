=== ANSWER ===
```python
import sys

def solve():
    input_data = sys.stdin.read().split()
    ptr = 0
    N = int(input_data[ptr])
    ptr += 1
    
    A = [[0] * (N + 1) for _ in range(N + 1)]
    for i in range(1, N + 1):
        for j in range(1, i + 1):
            A[i][j] = int(input_data[ptr])
            ptr += 1
            
    curr = 1
    for j in range(1, N + 1):
        if curr >= j:
            curr = A[curr][j]
        else:
            curr = A[j][curr]
            
    print(curr)

if __name__ == '__main__':
    solve()
```
