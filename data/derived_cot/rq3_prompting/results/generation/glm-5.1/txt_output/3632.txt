=== ANSWER ===
```python
from typing import List

class Solution:
    def buttonWithLongestTime(self, events: List[List[int]]) -> int:
        max_time = -1
        best_index = -1
        
        for i, (index, time) in enumerate(events):
            duration = time if i == 0 else time - events[i-1][1]
            if duration > max_time or (duration == max_time and index < best_index):
                max_time = duration
                best_index = index
                
        return best_index
```
