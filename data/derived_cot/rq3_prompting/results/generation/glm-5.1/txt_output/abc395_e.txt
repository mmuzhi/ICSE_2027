You are given a directed graph with N vertices and M edges.

import sys
import heapq

def main():
    import sys
    input = sys.stdin.readline
    # but we'll use read for speed
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X = int(next(it))
    out = [[] for _ in range(N)]
    inn = [[] for _ in range(N)]
    for _ in range(M):
        u = int(next(it)) - 1
        v = int(next(it)) - 1
        out[u].append(v)
        inn[v].append(u)
    INF = 10**18
    dist = [INF] * (2 * N)
    dist[0] = 0  # vertex 0 (1) mode 0
    heap = [(0, 0)]
    while heap:
        d, u_id = heapq.heappop(heap)
        if d != dist[u_id]:
            continue
        v = u_id % N
        mode = u_id // N
        # reversal
        new_mode = 1 - mode
        new_id = v + new_mode * N
        nd = d + X
        if nd < dist[new_id]:
            dist[new_id] = nd
            heapq.heappush(heap, (nd, new_id))
        # move
        if mode == 0:
            for to in out[v]:
                new_id = to  # mode 0, so to + 0*N = to
                nd = d + 1
                if nd < dist[new_id]:
                    dist[new_id] = nd
                    heapq.heappush(heap, (nd, new_id))
        else:  # mode == 1
            for to in inn[v]:
                new_id = to + N  # mode 1
                nd = d + 1
                if nd < dist[new_id]:
                    dist[new_id] = nd
                    heapq.heappush(heap, (nd, new_id))
    ans = min(dist[N-1], dist[2*N-1])

import sys
import heapq

def main():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    X = int(next(it))

    # original outgoing and incoming edges
    out = [[] for _ in range(N)]
    inn = [[] for _ in range(N)]
    for _ in range(M):
        u = int(next(it)) - 1
        v = int(next(it)) - 1
        out[u].append(v)
        inn[v].append(u)

    INF = 10**18
    dist = [INF] * (2 * N)
    dist[0] = 0                     # (vertex 1, mode 0)
    heap = [(0, 0)]

    while heap:
        d, uid = heapq.heappop(heap)
        if d != dist[uid]:
            continue

        # early exit when a target is reached
        if uid == N - 1 or uid == 2 * N - 1:
            print(d)
            return

        # decode vertex and mode
        if uid < N:
            v = uid
            mode = 0
        else:
            v = uid - N
            mode = 1

        # reversal operation
        new_mode = 1 - mode
        nid = v + new_mode * N
        nd = d + X
        if nd < dist[nid]:
            dist[nid] = nd
            heapq.heappush(heap, (nd, nid))

        # move along an edge
        if mode == 0:
            for to in out[v]:
                nid = to          # mode 0
                nd = d + 1
                if nd < dist[nid]:
                    dist[nid] = nd
                    heapq.heappush(heap, (nd, nid))
        else:  # mode == 1
            for to in inn[v]:
                nid = to + N      # mode 1
                nd = d + 1
                if nd < dist[nid]:
                    dist[nid] = nd
                    heapq.heappush(heap, (nd, nid))

    # fallback (should not be needed because reachability is guaranteed)
    ans = min(dist[N - 1], dist[2 * N - 1])
    print(ans)

if __name__ == "__main__":
    main()
