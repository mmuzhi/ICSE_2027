Let's quickly check the implementation details.
```python
class Solution:
    def maxFrequency(self, nums: List[int], k: int, numOperations: int) -> int:
        nums.sort()
        candidates = set()
        for x in nums:
            candidates.add(x - k)
            candidates.add(x)
            candidates.add(x + k)
        
        ans = 0
        for V in candidates:
            can_be = bisect_right(nums, V + k) - bisect_left(nums, V - k)
            already = bisect_right(nums, V) - bisect_left(nums, V)
            ans = max(ans, min(can_be, already + numOperations))
            
        return ans
```

```python
from bisect import bisect_left, bisect_right
```
```python
from typing import List

class Solution:
    def maxFrequency(self, nums: List[int], k: int, numOperations: int) -> int:
        nums.sort()
        candidates = set()
        for x in nums:
            candidates.add(x - k)
            candidates.add(x)
            candidates.add(x + k)
        
        ans = 0
        for V in candidates:
            can_be = bisect_right(nums, V + k) - bisect_left(nums, V - k)
            already = bisect_right(nums, V) - bisect_left(nums, V)
            ans = max(ans, min(can_be, already + numOperations))
            
        return ans
```
