```python
class Solution:
    def validSubstringCount(self, word1: str, word2: str) -> int:
        need = [0] * 26
        for c in word2:
            need[ord(c) - ord('a')] += 1
        
        total_distinct = sum(1 for x in need if x > 0)
        
        window = [0] * 26
        match_count = 0
        ans = 0
        L = 0
        
        for R in range(len(word1)):
            idx_R = ord(word1[R]) - ord('a')
            window[idx_R] += 1
            if window[idx_R] == need[idx_R]:
                match_count += 1
                
            while match_count == total_distinct:
                idx_L = ord(word1[L]) - ord('a')
                if window[idx_L] > need[idx_L]:
                    window[idx_L] -= 1
                    L += 1
                else:
                    break
                    
            if match_count == total_distinct:
                ans += L + 1
                
        return ans
```
