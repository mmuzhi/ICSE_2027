import sys

def solve():
    input = sys.stdin.read().splitlines()
    N = int(input[0])
    S = input[1:N+1]
    S_rev = S[::-1]
    M = max(len(s) for s in S)
    
    for j in range(M):
        T = []
        for i in range(N):
            if j < len(S_rev[i]):
                T.append(S_rev[i][j])
            else:
                T.append('*')
        print("".join(T).rstrip('*'))

if __name__ == '__main__':
    solve()
