import sys
import bisect

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    Q = int(data[0])
    A_vals = list(map(int, data[1:]))

    MAX_K = 10 ** 6          # because sqrt(1e12) = 1e6

    # omega[x] = number of distinct prime factors of x

    for i in range(2, MAX_K + 1):

            for j in range(i, MAX_K + 1, i):
                omega[j] += 1

    # squares of numbers with exactly two distinct prime factors
    squares = []
    for k in range(2, MAX_K + 1):
        if omega[k] == 2:
            squares.append(k * k)

    out = []
    for A in A_vals:
