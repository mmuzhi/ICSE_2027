def solve():
    import sys
    sys.setrecursionlimit(1<<25)
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    edges = [tuple(map(int, input().split())) for _ in range(M)]

    # compute L via BFS
    adj = [[] for _ in range(N)]
    for u,v in edges:
        adj[u-1].append(v-1)
    from collections import deque
    dist = [-1]*N
    dist[0]=0
    q=deque([0])
    while q:
        u=q.popleft()
        for v in adj[u]:
            if dist[v]==-1:
                dist[v]=dist[u]+1
                q.append(v)
    L = dist[N-1]  # since vertex N is index N-1

    INF = 10**9

    # function to check D
    def feasible(D):
        if D == 0:
            return True  # always feasible, inc=0
        # build flow network
        # node count: 2 + N*(D+1)
        node_cnt = 2 + N*(D+1)
        source = 0
        sink = 1
        dinic = Dinic(node_cnt)
        # helper to get id of (v,i) where v in 0..N-1, i in 0..D
        def vid(v, i):
            return 2 + v*(D+1) + i

        # source->v_0 INF
        for v in range(N):
            dinic.add_edge(source, vid(v,0), INF)

        # monotonicity: v_{i+1} -> v_i INF
        for v in range(N):
            for i in range(D):
                dinic.add_edge(vid(v,i+1), vid(v,i), INF)

        # vertex 1 (index 0) : force (0,i) to T for i>=1
        for i in range(1, D+1):
            dinic.add_edge(vid(0,i), sink, INF)

        # vertex N (index N-1) : force (N-1,D) to S
        dinic.add_edge(source, vid(N-1, D), INF)

        # for each original edge
        for (u,v) in edges:
            u0 = u-1
            v0 = v-1
            # hard constraints: v_{i+1} -> u_i INF for i=0..D-1
            for i in range(D):

            for i in range(D+1):
