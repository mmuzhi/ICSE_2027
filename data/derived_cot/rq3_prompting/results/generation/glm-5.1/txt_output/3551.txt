```python
class Solution:
    def maximumSubarrayXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        n = len(nums)
        queries_by_L = [[] for _ in range(n)]
        for i, (l, r) in enumerate(queries):
            queries_by_L[l].append((r, i))
            
        res = [0] * len(queries)
        
        f_prev = [0] * n
        ans_prev = [0] * n
        
        for L in range(n - 1, -1, -1):
            f_curr = [0] * n
            ans_curr = [0] * n
            
            f_curr[L] = nums[L]
            ans_curr[L] = nums[L]
            
            for R in range(L + 1, n):
                f_curr[R] = f_curr[R - 1] ^ f_prev[R]
                ans_curr[R] = max(f_curr[R], ans_prev[R], ans_curr[R - 1])
                
            for r, idx in queries_by_L[L]:
                res[idx] = ans_curr[r]
                
            f_prev = f_curr
            ans_prev = ans_curr
            
        return res
```
