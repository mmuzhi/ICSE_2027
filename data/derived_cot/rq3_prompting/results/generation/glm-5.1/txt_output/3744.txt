```python
class Solution:
    def minOperations(self, queries: List[List[int]]) -> int:
        def g(x):
            if x == 0:
                return 0
            k = 0
            p = 1
            while p * 4 <= x:
                p *= 4
                k += 1
            # p = 4^k
            # sum for j=0 to k-1 is ((3*k - 1) * p + 1) // 3
            prefix_sum = ((3 * k - 1) * p + 1) // 3
            return prefix_sum + (k + 1) * (x - p + 1)
        
        ans = 0
        for l, r in queries:
            s = g(r) - g(l - 1)
            ans += (s + 1) // 2
        return ans
```

```python
import bisect

class Solution:
    def minOperations(self, queries: List[List[int]]) -> int:
        # Precompute powers of 4
        pows = [1]
        for _ in range(30):
            pows.append(pows[-1] * 4)
            
        def g(x):
            if x == 0:
                return 0
            # Find k such that 4^k <= x < 4^(k+1)
            k = bisect.bisect_left(pows, x + 1) - 1
            p = pows[k]
            # Sum of f(i) for i from 1 to 4^k - 1
            prefix_sum = ((3 * k - 1) * p + 1) // 3
            # Add the remaining elements from 4^k to x
            return prefix_sum + (k + 1) * (x - p + 1)
            
        ans = 0
        for l, r in queries:
            # Total depth S
            S = g(r) - g(l - 1)
            # Maximum depth M
            M = bisect.bisect_left(pows, r + 1)
            # Minimum operations is max(M, ceil(S/2))
            ans += max(M, (S + 1) // 2)
            
        return ans
```
