import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    # N = int(data[0]) # N is not really needed since S has length N, but we can parse it.
    c1 = data[1]
    c2 = data[2]
    S = data[3]
    
    res = []
    for char in S:
        if char == c1:
            res.append(char)
        else:
            res.append(c2)
    print("".join(res))

if __name__ == '__main__':
    main()
