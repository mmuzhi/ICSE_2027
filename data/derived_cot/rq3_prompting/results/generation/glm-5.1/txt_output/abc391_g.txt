```python
def get_next_mask(mask, c, S, N):
    dp = [0] * (N + 1)
    for i in range(1, N + 1):
        dp[i] = dp[i-1] + (1 if (mask & (1 << (i-1))) else 0)
    
    dp_prime = [0] * (N + 1)
    for i in range(1, N + 1):
        dp_prime[i] = max(dp[i], dp_prime[i-1])
        if S[i-1] == c:
            dp_prime[i] = max(dp_prime[i], dp[i-1] + 1)
            
    nxt = 0
    for i in range(1, N + 1):
        if dp_prime[i] - dp_prime[i-1] == 1:
            nxt |= (1 << (i-1))
    return nxt
```

```python
def get_next_mask(mask, c, S_int, N):
    dp = [0] * (N + 1)
    for i in range(1, N + 1):
        dp[i] = dp[i-1] + (1 if (mask & (1 << (i-1))) else 0)

    for i in range(1, N + 1):
```

```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    S = data[2]
    
    S_int = [ord(c) - ord('a') for c in S]
    
    # Precompute the next mask for each mask and character
    def get_next_mask(mask, c):
        dp_prev = 0
        dp_prime_prev = 0
        nxt = 0
        for i in range(N):
            dp_curr = dp_prev + (1 if (mask & (1 << i)) else 0)
            dp_prime_curr = max(dp_curr, dp_prime_prev)
            if S_int[i] == c:
                dp_prime_curr = max(dp_prime_curr, dp_prev + 1)
            
            if dp_prime_curr - dp_prime_prev == 1:
                nxt |= (1 << i)
                
            dp_prev = dp_curr
            dp_prime_prev = dp_prime_curr
        return nxt

    max_mask = 1 << N
    next_mask = [[0] * 26 for _ in range(max_mask)]
    for mask in range(max_mask):
        for c in range(26):
            next_mask[mask][c] = get_next_mask(mask, c)
            
    MOD = 998244353
    dp_count = [0] * max_mask
    dp_count[0] = 1
    
    # DP over the length of the string T
    for _ in range(M):
        next_dp_count = [0] * max_mask
        for mask in range(max_mask):
            val = dp_count[mask]
            if val == 0:
                continue
            for c in range(26):
                nxt = next_mask[mask][c]
                next_dp_count[nxt] += val
        # Apply modulo at the end of each step to avoid big integer overhead
        for i in range(max_mask):
            next_dp_count[i] %= MOD
        dp_count = next_dp_count
        
    # Aggregate results by popcount (which equals the LCS length)
    ans = [0] * (N + 1)
    for mask in range(max_mask):
        k = bin(mask).count('1')
        ans[k] = (ans[k] + dp_count[mask]) % MOD
        
    print(' '.join(map(str, ans)))

if __name__ == '__main__':
    solve()
```
