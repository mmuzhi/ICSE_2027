        while idx > 0:

def main():
    import sys
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    class BIT:
        __slots__ = ('n', 'bit')
        def __init__(self, n):
            self.n = n
            self.bit = [0]*(n+1)
        def add(self, idx, val):
            n = self.n
            bit = self.bit
            while idx <= n:
                bit[idx] += val
                idx += idx & -idx
        def sum(self, idx):
            s = 0
            bit = self.bit
            while idx > 0:
                s += bit[idx]
                idx -= idx & -idx
            return s

    bit_cnt = BIT(M)
    bit_sum = BIT(M)
    # insert S0 = 0
    bit_cnt.add(1, 1)   # index 1 corresponds to value 0
    bit_sum.add(1, 0)
    inserted = 1
    total_sum_mods = 0
    cur_mod = 0
    ans = 0

    for a in A:
        cur_mod = (cur_mod + a) % M
        x = cur_mod
        idx = x + 1
        cnt_le = bit_cnt.sum(idx)
        sum_le = bit_sum.sum(idx)

import sys

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    class BIT:
        __slots__ = ('n', 'bit')
        def __init__(self, n):
            self.n = n
            self.bit = [0] * (n + 1)
        def add(self, idx, val):
            n = self.n
            bit = self.bit
            while idx <= n:
                bit[idx] += val
                idx += idx & -idx
        def sum(self, idx):
            s = 0
            bit = self.bit
            while idx > 0:
                s += bit[idx]
                idx -= idx & -idx
            return s

    bit_cnt = BIT(M)
    bit_sum = BIT(M)

    # insert S0 = 0
    bit_cnt.add(1, 1)   # index 1 corresponds to value 0
    bit_sum.add(1, 0)

    inserted = 1          # number of prefix sums already in BIT
    total_sum_mods = 0    # sum of all inserted prefix mods
    cur_mod = 0
    ans = 0

    for a in A:
        cur_mod = (cur_mod + a) % M
        x = cur_mod
        idx = x + 1

        cnt_le = bit_cnt.sum(idx)
        sum_le = bit_sum.sum(idx)

        cnt_total = inserted
        sum_total = total_sum_mods

        cnt_gt = cnt_total - cnt_le
        sum_gt = sum_total - sum_le

        contrib = cnt_le * x - sum_le + cnt_gt * (x + M) - sum_gt
        ans += contrib

        # add current prefix mod to BIT for future positions
        bit_cnt.add(idx, 1)
        bit_sum.add(idx, x)
        total_sum_mods += x
        inserted += 1
