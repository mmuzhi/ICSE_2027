```python
def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    N = int(data[0])
    A = [int(x) for x in data[1:N+1]]
    
    ans = 0
    for k in range(25):
        mod = 1 << (k + 1)
        target = 1 << k
        
        count = [0] * mod
        sum_A = [0] * mod
        
        for x in A:
            r = x % mod
            count[r] += 1
            sum_A[r] += x
            
        S_half = 0
        for x in A:
            r = x % mod
            r2 = (target - r) % mod
            S_half += sum_A[r2]
            
        D = 0
        val = 1 << (k - 1) if k > 0 else 0
        for x in A:
            if (x & -x) == val:
                D += x
                
        C_k = S_half + D
        ans += C_k // target
        
    print(ans)
```

```python
    N = int(data[0])
    A = [int(x) for x in data[1:N+1]]
    
    ans = 0
    for k in range(25):
        mod = 1 << (k + 1)
        target = 1 << k
```
```python
        mask = mod - 1
```
```python
        val = (1 << (k - 1)) if k > 0 else 0
        
        if mod > 2000000:
            sum_A = {}
            D = 0
            for x in A:
                r = x & mask
                sum_A[r] = sum_A.get(r, 0) + x
                if (x & -x) == val:
                    D += x
            
            S_half = 0
            for x in A:
                r = x & mask
                r2 = (target - r) & mask
                if r2 in sum_A:
                    S_half += sum_A[r2]
        else:
            sum_A = [0] * mod
            D = 0
            for x in A:
                r = x & mask
                sum_A[r] += x
                if (x & -x) == val:
                    D += x
                
            S_half = 0
            for x in A:
                r = x & mask
                r2 = (target - r) & mask
                S_half += sum_A[r2]
                
        C_k = S_half + D
        ans += C_k // target
        
    print(ans)

    solve()
```
