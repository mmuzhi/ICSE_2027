```python
class Solution:
    def minCost(self, n: int, cost: List[List[int]]) -> int:
        M = n // 2
        INF = float('inf')
        
        # dp[c1][c2] is the min cost for pairs up to current, ending with c1, c2
        dp = [[INF]*3 for _ in range(3)]
        
        for k in range(M):
            l = k
            r = n - 1 - k
            
            # compute pair_cost for current k
            pc = [[INF]*3 for _ in range(3)]
            for c1 in range(3):
                for c2 in range(3):
                    if c1 != c2:
                        pc[c1][c2] = cost[l][c1] + cost[r][c2]
            
            if k == 0:
                dp = pc
            else:
                new_dp = [[INF]*3 for _ in range(3)]
                for c1 in range(3):
                    for c2 in range(3):
                        if c1 == c2: continue
                        min_prev = INF
                        for p1 in range(3):
                            if p1 == c1: continue
                            for p2 in range(3):
                                if p2 == c2: continue
                                if dp[p1][p2] < min_prev:
                                    min_prev = dp[p1][p2]
                        new_dp[c1][c2] = pc[c1][c2] + min_prev
                dp = new_dp
                
        return int(min(dp[c1][c2] for c1 in range(3) for c2 in range(3) if c1 != c2))
```

    def minCost(self, n: int, cost: List[List[int]]) -> int:
        M = n // 2
        INF = 10**18
        
        # dp[c1][c2] stores the min cost for pairs processed so far, 
        # where the last pair was painted with colors c1 and c2
        dp = [[INF]*3 for _ in range(3)]
        
        for k in range(M):

            r = n - 1 - k

            pc = [[INF]*3 for _ in range(3)]
            for c1 in range(3):

                    if c1 != c2:
                        pc[c1][c2] = cost[l][c1] + cost[r][c2]

                dp = pc
