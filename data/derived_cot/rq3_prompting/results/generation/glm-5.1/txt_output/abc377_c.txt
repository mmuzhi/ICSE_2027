def solve():
    # Read all input from standard input
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    N = int(input_data[0])
    M = int(input_data[1])
    
    # Use a set to store unique squares that are either occupied or attacked
    attacked = set()
    
    # Knight move offsets
    dx = [2, 1, -1, -2, -2, -1, 1, 2]
    dy = [1, 2, 2, 1, -1, -2, -2, -1]
    
    idx = 2
    for _ in range(M):
        a = int(input_data[idx])
        b = int(input_data[idx+1])
        idx += 2
        
        # Add the square occupied by the piece itself
        # We use a * N + b as a unique 1D hash for the 2D coordinate (a, b)
        attacked.add(a * N + b)
        
        # Add all valid squares attacked by this piece
        for i in range(8):
            na = a + dx[i]
            nb = b + dy[i]
            
            # Check if the attacked square is within the grid bounds
            if 1 <= na <= N and 1 <= nb <= N:
                attacked.add(na * N + nb)
                
    # The number of safe squares is the total squares minus the squares that are occupied or attacked
    ans = N * N - len(attacked)
    print(ans)

if __name__ == '__main__':
    solve()
