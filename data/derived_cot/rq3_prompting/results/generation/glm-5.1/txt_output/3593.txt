```python
class Solution:
    def maxScore(self, nums: List[int]) -> int:
        from math import gcd
        from functools import reduce
        
        def lcm(a, b):
            return a * b // gcd(a, b)
        
        ans = 0
        n = len(nums)
        for i in range(n + 1):
            if i < n:
                arr = nums[:i] + nums[i+1:]
            else:
                arr = nums
            
            if not arr:
                continue
                
            g = reduce(gcd, arr)
            l = reduce(lcm, arr)
            ans = max(ans, g * l)
            
        return ans
```
