import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    Q = int(data[1])
    S = data[2]
    
    P_1 = []
    P_2 = []
    S_slash = [0] * (N + 1)
    cnt_1 = [0] * (N + 1)
    cnt_2 = [0] * (N + 1)
    
    for i in range(N):
        S_slash[i+1] = S_slash[i]
        cnt_1[i+1] = cnt_1[i]
        cnt_2[i+1] = cnt_2[i]
        if S[i] == '1':
            P_1.append(i + 1)
            cnt_1[i+1] += 1
        elif S[i] == '2':
            P_2.append(i + 1)
            cnt_2[i+1] += 1
        elif S[i] == '/':
            S_slash[i+1] += 1

    def check(a, L, R):
        c_1 = cnt_1[L-1]
        if c_1 + a > len(P_1):
            return False
        pos_1 = P_1[c_1 + a - 1]
        
        c_2 = cnt_2[N] - cnt_2[R]
        if c_2 + a > len(P_2):
            return False
        pos_2 = P_2[len(P_2) - c_2 - a]
        
        if pos_1 >= pos_2:
            return False
            
        if S_slash[pos_2 - 1] - S_slash[pos_1] > 0:
            return True
        return False

    output = []
    idx = 3
    for _ in range(Q):
        L = int(data[idx])
        R = int(data[idx+1])
        idx += 2
        
        if S_slash[R] - S_slash[L-1] == 0:
            output.append("0")
            continue
            
        low = 1
        high = (R - L + 1) // 2
        ans = 0
        while low <= high:
            mid = (low + high) // 2
            if check(mid, L, R):
                ans = mid
                low = mid + 1
            else:
                high = mid - 1
        output.append(str(2 * ans + 1))
        
    print("\n".join(output))

if __name__ == '__main__':
    solve()
