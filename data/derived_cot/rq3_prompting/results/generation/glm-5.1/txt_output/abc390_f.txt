For each integer pair (L,R) with 1 \le L \le R \le N, define f(L,R) as follows:

- Start with an empty blackboard. Write the R-L+1 integers A_L, A_{L+1}, \ldots, A_R on the blackboard in order.
- Repeat the following operation until all integers on the blackboard are erased:
- Choose integers l, r with l \le r such that every integer from l through r appears at least once on the blackboard. Then, erase all integers from l through r that are on the blackboard.


- Let f(L,R) be the minimum number of such operations needed to erase all the integers from the blackboard.

Find \displaystyle \sum_{L=1}^N \sum_{R=L}^N f(L,R).

- All input values are integers.

import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    # ---------- sum of distinct values over all subarrays ----------
    last = [-1] * (N + 1)          # last occurrence index (0‑based)
    sum_distinct = 0
    for i, v in enumerate(A):
        prev = last[v]
        sum_distinct += (i - prev) * (N - i)
        last[v] = i

    # ---------- positions of each value ----------
    pos = [[] for _ in range(N + 1)]
    for i, v in enumerate(A):
        pos[v].append(i)           # already sorted

    total_sub = N * (N + 1) // 2

    # ---------- miss_val[v] = number of subarrays that avoid value v ----------
    miss_val = [0] * (N + 1)
    for v in range(1, N + 1):
        p = pos[v]
        if not p:
            miss_val[v] = total_sub
            continue
        total = 0
        prev = -1
        for f in p:
            gap = f - prev - 1
            total += gap * (gap + 1) // 2
            prev = f
        gap = N - prev - 1
        total += gap * (gap + 1) // 2
        miss_val[v] = total

    # ---------- sum over v of #subarrays containing both v and v+1 ----------
    sum_e = 0
    for v in range(1, N):
        P = pos[v]
        Q = pos[v + 1]
        if not P or not Q:          # at least one value never appears → cnt_both = 0
            continue

        miss_v = miss_val[v]
        miss_v1 = miss_val[v + 1]

        # ----- compute miss_both = #subarrays avoiding both v and v+1 -----
        # by merging the two sorted position lists
        total = 0
        prev = -1
        i = j = 0
        lenP, lenQ = len(P), len(Q)
        while i < lenP and j < lenQ:
            if P[i] < Q[j]:
                f = P[i]
                i += 1
            else:
                f = Q[j]
                j += 1
            gap = f - prev - 1
            total += gap * (gap + 1) // 2
            prev = f
        while i < lenP:
            f = P[i]
            i += 1
            gap = f - prev - 1
            total += gap * (gap + 1) // 2
            prev = f
        while j < lenQ:
            f = Q[j]
            j += 1
            gap = f - prev - 1
            total += gap * (gap + 1) // 2
            prev = f
        gap = N - prev - 1
        total += gap * (gap + 1) // 2
        miss_both = total

        cnt_both = total_sub - miss_v - miss_v1 + miss_both
        sum_e += cnt_both

    # ---------- final answer ----------
    ans = sum_distinct - sum_e
    print(ans)

if __name__ == "__main__":
    solve()
