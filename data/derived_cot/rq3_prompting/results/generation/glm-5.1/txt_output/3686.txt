```python
                Z[i] = min(r - i, Z[i - l])
            while i + Z[i] < n and nums[Z[i]] == nums[i + Z[i]]:
                Z[i] += 1
            if i + Z[i] > r:
                l, r = i, i + Z[i]

        for i in range(n - 2, 0, -1):
```

Wait, what if `2 * i <= j` and `Z[i] >= i`?
Is it possible that `nums1` is a prefix of `nums2` but `Z[i]` is NOT `>= i`?
No, `Z[i]` is the LCP of `nums` and `nums[i:]`.
If `nums1` (which is `nums[0:i]`) is a prefix of `nums2` (which is `nums[i:2i]`), then `nums[0:i] == nums[i:2i]`.
This exactly means the first `i` characters of `nums` match the first `i` characters of `nums[i:]`.
So `Z[i]` must be at least `i`.
Thus `Z[i] >= i` is equivalent to `nums1` being a prefix of `nums2`.

Similarly, `lcp[j]` is the LCP of `nums[i:]` and `nums[j:]`.

```python
class Solution:
    def beautifulSplits(self, nums: List[int]) -> int:
        n = len(nums)
        if n < 3:
            return 0
        
        # Z-array: Z[i] is the length of the longest common prefix of nums and nums[i:]
        Z = [0] * n
        Z[0] = n
        l, r = 0, 0
        for i in range(1, n):
            if i < r:
                Z[i] = min(r - i, Z[i - l])
            while i + Z[i] < n and nums[Z[i]] == nums[i + Z[i]]:
                Z[i] += 1
            if i + Z[i] > r:
                l, r = i, i + Z[i]
                
        ans = 0
        # lcp[j] will store the LCP of nums[i:] and nums[j:]
        lcp = [0] * (n + 1)
        
        # i is the length of nums1, j is the length of nums1 + nums2
        # We iterate i backwards to compute LCP efficiently in O(1) space
        for i in range(n - 2, 0, -1):
            zi = Z[i]
            two_i = 2 * i
            n_plus_i = n + i
            for j in range(i + 1, n):
                if nums[i] == nums[j]:
                    lcp[j] = lcp[j + 1] + 1
                else:
                    lcp[j] = 0
                
                # Condition 1: nums1 is a prefix of nums2
                # Requires len(nums1) <= len(nums2) -> i <= j - i -> j >= 2*i
                # And nums[0:i] == nums[i:2i] -> Z[i] >= i
                cond1 = (j >= two_i) and (zi >= i)
                
                # Condition 2: nums2 is a prefix of nums3
                # Requires len(nums2) <= len(nums3) -> j - i <= n - j -> 2*j <= n + i
                # And nums[i:j] == nums[j:2j-i] -> lcp[j] >= j - i
                cond2 = (2 * j <= n_plus_i) and (lcp[j] >= j - i)
                
                if cond1 or cond2:
                    ans += 1
                    
        return ans
```
