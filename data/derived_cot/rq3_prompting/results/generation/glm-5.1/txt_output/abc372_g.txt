import sys
from fractions import Fraction

def solve():
    input = sys.stdin.read
    data = input().split()
    idx = 0
    T = int(data[idx])
    idx += 1
    
    res = []
    
    for _ in range(T):
        N = int(data[idx])
        idx += 1
        
        arr = []
        for _ in range(N):
            A = int(data[idx])
            B = int(data[idx+1])
            C = int(data[idx+2])
            arr.append((A, B, C))
            idx += 3
            
        X = 10**18
        for A, B, C in arr:
            val = (C - B - 1) // A
            if val < X:
                X = val
                
        if X < 1:
            res.append(0)
            continue
            
        arr.sort(key=lambda x: (Fraction(x[0], x[1]), Fraction(x[2] - 1, x[1])))
        
        unique_lines = []
        for line in arr:
            if not unique_lines:
                unique_lines.append(line)
            else:
                A1, B1, C1 = unique_lines[-1]
                A2, B2, C2 = line
                if A1 * B2 != A2 * B1:
                    unique_lines.append(line)
                    
        def get_X(la, lb):
            Aa, Ba, Ca = la
            Ab, Bb, Cb = lb
            Num = (Cb - 1) * Ba - (Ca - 1) * Bb
            Den = Ab * Ba - Aa * Bb
            return Num // Den
            
        hull = []
        for line in unique_lines:
            while len(hull) >= 2:
                l1 = hull[-2]
                l2 = hull[-1]
                l3 = line
                X12 = get_X(l1, l2)
                X23 = get_X(l2, l3)
                if X23 <= X12:
                    hull.pop()
                else:
                    break
            hull.append(line)
            
        def floor_sum(n, m, a, b):
            ans = 0
            while True:
                qa, ra = divmod(a, m)
                qb, rb = divmod(b, m)
                ans += qa * n * (n - 1) // 2 + qb * n
                a, b = ra, rb
                y_max = a * n + b
                if y_max < m:
                    break
                n, b = divmod(y_max, m)
                m, a = a, m
            return ans
            
        ans = 0
        for j in range(len(hull)):
            A, B, C = hull[j]
            if j == 0:
                L = 1
            else:
                L = get_X(hull[j-1], hull[j]) + 1
                
            if j == len(hull) - 1:
                R = X
            else:
                R = get_X(hull[j], hull[j+1])
                
            L = max(L, 1)
            R = min(R, X)
            
            if L <= R:
                ans += floor_sum(R - L + 1, B, -A, C - 1 - A * L)
                
        res.append(ans)
        
    print('\n'.join(map(str, res)))

solve()
