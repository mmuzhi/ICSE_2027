```python

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    a = list(map(int, data[1:1 + n]))
    for i in range(n - 2):
        if a[i] == a[i + 1] == a[i + 2]:
            print("Yes")
            return
    print("No")

if __name__ == "__main__":
    main()
```
