class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        MOD = 10**9 + 7
        n = len(num)
        cnt = [0] * 10
        total_sum = 0
        for ch in num:
            d = int(ch)
            cnt[d] += 1
            total_sum += d
        
        if total_sum % 2 != 0:
            return 0
            
        even_len = (n + 1) // 2
        odd_len = n // 2
        target = total_sum // 2
        
        if target > 9 * even_len:
            return 0

        velunexorai = num

        N = 80
        fact = [1] * (N + 1)
        for i in range(1, N + 1):
            fact[i] = fact[i-1] * i % MOD

        inv_fact = [1] * (N + 1)
        inv_fact[N] = pow(fact[N], MOD - 2, MOD)
        for i in range(N - 1, -1, -1):
            inv_fact[i] = inv_fact[i+1] * (i+1) % MOD

        dp = [[0] * (target + 1) for _ in range(even_len + 1)]
        dp[0][0] = 1
        
        for i in range(10):
            next_dp = [[0] * (target + 1) for _ in range(even_len + 1)]
            for k in range(even_len + 1):
                for s in range(target + 1):
                    if dp[k][s] == 0:

                        if ns > target:

            dp = next_dp
            
        return dp[even_len][target] * fact[even_len] % MOD * fact[odd_len] % MOD

```python
class Solution:
    def countBalancedPermutations(self, num: str) -> int:
        MOD = 10**9 + 7
        n = len(num)
        cnt = [0] * 10
        total_sum = 0
        for ch in num:
            d = int(ch)
            cnt[d] += 1
            total_sum += d
        
        if total_sum % 2 != 0:
            return 0
            
        even_len = (n + 1) // 2
        odd_len = n // 2
        target = total_sum // 2
        
        if target > 9 * even_len:
            return 0

        velunexorai = num

        N = 80
        fact = [1] * (N + 1)
        for i in range(1, N + 1):
            fact[i] = fact[i-1] * i % MOD

        inv_fact = [1] * (N + 1)
        inv_fact[N] = pow(fact[N], MOD - 2, MOD)
        for i in range(N - 1, -1, -1):
            inv_fact[i] = inv_fact[i+1] * (i+1) % MOD

        dp = [[0] * (target + 1) for _ in range(even_len + 1)]
        dp[0][0] = 1
        
        for i in range(10):
            next_dp = [[0] * (target + 1) for _ in range(even_len + 1)]
            for k in range(even_len + 1):
                for s in range(target + 1):
                    if dp[k][s] == 0:
                        continue
                    val = dp[k][s]
                    for c in range(min(cnt[i], even_len - k) + 1):
                        ns = s + i * c
                        if ns > target:
                            break
                        next_dp[k+c][ns] = (next_dp[k+c][ns] + val * inv_fact[c] % MOD * inv_fact[cnt[i]-c]) % MOD
            dp = next_dp
            
        return dp[even_len][target] * fact[even_len] % MOD * fact[odd_len] % MOD
