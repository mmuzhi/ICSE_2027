import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    qs = [0] * N
    rs = [0] * N
    idx = 1
    for i in range(N):
        qs[i] = int(data[idx])
        rs[i] = int(data[idx+1])
        idx += 2
    Q = int(data[idx])
    idx += 1
    out = []
    for _ in range(Q):
        t = int(data[idx]) - 1
        d = int(data[idx+1])
        idx += 2
        q = qs[t]
        r = rs[t]
        m = d % q
        ans = d + (r - m + q) % q
        out.append(str(ans))
    print("\n".join(out))

if __name__ == '__main__':
    solve()
```
