```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    T = int(data[0])
    idx = 1
    res = []
    for _ in range(T):
        K = int(data[idx])
        Sx = int(data[idx+1])
        Sy = int(data[idx+2])
        Tx = int(data[idx+3])
        Ty = int(data[idx+4])
        idx += 5
        
        # Calculate start tile
        i1 = (2 * Sx + 1) // (2 * K)
        j1 = (2 * Sy + 1) // (2 * K)
        p1 = (i1 + j1) % 2
        if p1 == 0:
            k1 = Sy - j1 * K
        else:
            k1 = Sx - i1 * K
            
        # Calculate target tile
        i2 = (2 * Tx + 1) // (2 * K)
        j2 = (2 * Ty + 1) // (2 * K)
        p2 = (i2 + j2) % 2
        if p2 == 0:
            k2 = Ty - j2 * K
        else:
            k2 = Tx - i2 * K
            
        di = i2 - i1
        dj = j2 - j1
        A = abs(di)
        B = abs(dj)
        n = A + B
        
        # If in the same block, just walk along k
        if n == 0:
            res.append(abs(k1 - k2))
            continue
            
        # Number of times we are at H parity
        N_H = (n + 1) // 2 if p1 == 0 else n // 2
        D = A - N_H
        R_min = abs(D)
        
        # Cost to reach the highway (k=0 or k=K-1)
        highway_cost = min(k1, K - 1 - k1)
        alt_highway_cost = abs(k1 - k2) + min(k2, K - 1 - k2)
        highway_cost = min(highway_cost, alt_highway_cost)
        
        if R_min == 0:
            # We can choose not to use the highway at all
            cost_without = n + abs(k1 - k2)
            # Or we can use the highway by swapping moves (requires n >= 2)
            # or adding a round trip (requires n < 2)
            if n >= 2:
                cost_with = n + highway_cost
            else:
                cost_with = n + 2 + highway_cost
            res.append(min(cost_without, cost_with))
        else:
            # We must use the highway at least R_min times
            res.append(n + highway_cost)

print('\n'.join(map(str, res)))

solve()
```
