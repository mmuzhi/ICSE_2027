```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1].strip()
    # Compute B = sum d_i * i
    B = 0
    for idx, ch in enumerate(S):
        B += (idx+1) * (ord(ch) - 48)  # faster than int()
    
    # Initialize digits array for A
    L = N + 10
    digits = [0] * L
    
    # Add contributions
    for idx, ch in enumerate(S):
        i = idx + 1
        d = ord(ch) - 48
        v = d * i
        shift = N - i + 1  # shift >= 1
        pos = 0
        while v:
            digit = v % 10
            digits[shift + pos] += digit
            v //= 10
            pos += 1
    
    # Normalize carries
    for i in range(len(digits)-1):
        carry = digits[i] // 10
        digits[i] %= 10
        digits[i+1] += carry
    # Handle last digit
    while digits[-1] >= 10:
        carry = digits[-1] // 10
        digits[-1] %= 10
        digits.append(carry)
    
    # Subtract B
    borrow = 0
    b = B
    for i in range(len(digits)):
        sub = b % 10
        b //= 10
        val = digits[i] - borrow - sub
        if val < 0:
            val += 10
            borrow = 1
        else:
            borrow = 0
        digits[i] = val
    # Trim leading zeros (from most significant end)
    while len(digits) > 1 and digits[-1] == 0:
        digits.pop()
    
    # Divide by 9
    # Reverse to most significant first
    msd = digits[::-1]
    remainder = 0
    quotient_digits = []
    for d in msd:
        current = remainder * 10 + d
        q = current // 9
        remainder = current % 9
        if quotient_digits or q != 0:
            quotient_digits.append(q)
    # If quotient is empty, answer is 0
    if not quotient_digits:
        print('0')
    else:
        print(''.join(str(q) for q in quotient_digits))

if __name__ == "__main__":
    main()
```

```python
import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    S = data[1]

    # B = sum d_i * i
    B = 0
    for idx, ch in enumerate(S):
        B += (idx + 1) * (ord(ch) - 48)

    # digits array for A (least significant first)
    L = N + 10
    digits = [0] * L

    # add contributions d_i * i * 10^{N-i+1}
    for idx, ch in enumerate(S):
        i = idx + 1
        d = ord(ch) - 48
        v = d * i
        shift = N - i + 1          # 10^{shift}
        pos = 0
        while v:
            digit = v % 10
            digits[shift + pos] += digit
            v //= 10
            pos += 1

    # normalize carries
    for i in range(len(digits) - 1):
        carry = digits[i] // 10
        digits[i] %= 10
        digits[i + 1] += carry
    while digits[-1] >= 10:
        carry = digits[-1] // 10
        digits[-1] %= 10
        digits.append(carry)

    # subtract B
    borrow = 0
    b = B
    for i in range(len(digits)):
        sub = b % 10
        b //= 10
        val = digits[i] - borrow - sub
        if val < 0:

            borrow = 1
```
