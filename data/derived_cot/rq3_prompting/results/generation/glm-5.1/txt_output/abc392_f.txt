        self.n = n
        self.bit = [0]*(n+1)
    def add(self, idx, delta):

            self.bit[idx] += delta
            idx += idx & -idx
    def sum(self, idx):

            idx -= idx & -idx

import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    N = int(data[0])
    P = list(map(int, data[1:]))

    # Fenwick tree (1‑indexed) initially all ones (empty slots)
    n = N
    bit = [0] * (n + 1)
    for i in range(1, n + 1):
        bit[i] = i & -i          # lowbit(i) = sum of ones in that range

    def add(idx: int, delta: int) -> None:
        while idx <= n:
            bit[idx] += delta
            idx += idx & -idx

    def kth(k: int) -> int:
        """return smallest index such that prefix sum >= k"""
        idx = 0
        # largest power of two <= n
        bit_mask = 1 << (n.bit_length() - 1)
        while bit_mask:
            nxt = idx + bit_mask
            if nxt <= n and bit[nxt] < k:
                k -= bit[nxt]
                idx = nxt
            bit_mask >>= 1
        return idx + 1

    ans = [0] * N
    for i in range(N, 0, -1):
        k = P[i - 1]          # P_i
        pos = kth(k)          # the P_i-th empty slot
        ans[pos - 1] = i
        add(pos, -1)          # mark slot as occupied

    sys.stdout.write(' '.join(map(str, ans)))

if __name__ == "__main__":
    solve()
