def shortest_palindrome_suffix(S):
    R = S[::-1]
    combined = R + '#' + S
    n = len(combined)
    pi = [0] * n
    for i in range(1, n):
        j = pi[i-1]
        while j > 0 and combined[i] != combined[j]:
            j = pi[j-1]
        if combined[i] == combined[j]:
            j += 1
        pi[i] = j
    overlap = pi[-1]
    return S + R[overlap:]
