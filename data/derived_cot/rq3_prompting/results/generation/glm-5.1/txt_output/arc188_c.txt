class DSU:
    def __init__(self, n):
        self.parent = list(range(n))
        self.xor = [0]*n
        self.size = [1]*n

    def find(self, x):
        if self.parent[x] != x:
            orig = self.parent[x]
            self.parent[x] = self.find(orig)
            self.xor[x] ^= self.xor[orig]
        return self.parent[x]

    def union(self, x, y, w):
        # enforce x xor y = w
        rx = self.find(x)
        ry = self.find(y)
        if rx == ry:
            # check consistency
            if (self.xor[x] ^ self.xor[y]) != w:
                return False
            return True
        # compute needed xor between roots
        needed = w ^ self.xor[x] ^ self.xor[y]
        # union by size
        if self.size[rx] < self.size[ry]:
            # attach rx under ry
            self.parent[rx] = ry
            self.xor[rx] = needed
            self.size[ry] += self.size[rx]
        else:
            # attach ry under rx
            self.parent[ry] = rx
            self.xor[ry] = needed
            self.size[rx] += self.size[ry]
        return True

import sys

class DSU:
    def __init__(self, n):
        self.parent = list(range(n))
        self.xor = [0] * n          # xor to parent
        self.size = [1] * n

    def find(self, x):
        if self.parent[x] != x:
            orig = self.parent[x]
            self.parent[x] = self.find(orig)
            self.xor[x] ^= self.xor[orig]
        return self.parent[x]

    def union(self, x, y, w):
        # enforce x xor y = w
        rx = self.find(x)
        ry = self.find(y)
        if rx == ry:
            # check consistency
            if (self.xor[x] ^ self.xor[y]) != w:
                return False
            return True
        # needed = value[rx] xor value[ry]
        needed = w ^ self.xor[x] ^ self.xor[y]
        # union by size
        if self.size[rx] < self.size[ry]:
            # attach rx under ry
            self.parent[rx] = ry
            self.xor[rx] = needed
            self.size[ry] += self.size[rx]
        else:
            # attach ry under rx
            self.parent[ry] = rx
            self.xor[ry] = needed
            self.size[rx] += self.size[ry]
        return True

def main():
    input = sys.stdin.readline
    N, M = map(int, input().split())
    dsu = DSU(2 * N)          # 0..N-1: x_i, N..2N-1: k_i

    for _ in range(M):
        A, B, C = map(int, input().split())
        a = A - 1
        b = B - 1
        u = N + a              # k_A
        v = b                  # x_B
        if not dsu.union(u, v, C):
            print(-1)
            return

    ans = []
    for i in range(N):
        dsu.find(i)            # ensure path compression
        x_val = dsu.xor[i]
        dsu.find(N + i)
        k_val = dsu.xor[N + i]
        y = x_val ^ k_val
        ans.append(str(y))

    print(''.join(ans))

if __name__ == "__main__":
    main()
