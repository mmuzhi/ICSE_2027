- Build a segment tree where each node stores the maximum value of `baskets` in its range.
- When a fruit of quantity `f` is processed:
  - We need to find the smallest index `i` such that `baskets[i] >= f` and `baskets[i]` is still available.

  - Otherwise, traverse the segment tree: at each node, if the left child's max >= `f`, recurse left; else recurse right.

class Solution:
    def numOfUnplacedFruits(self, fruits: List[int], baskets: List[int]) -> int:
        n = len(fruits)
        tree = [0] * (4 * n)
        
        def build(node, l, r):
            if l == r:
                tree[node] = baskets[l]
            else:
                mid = (l + r) // 2
                build(node * 2, l, mid)
                build(node * 2 + 1, mid + 1, r)
                tree[node] = max(tree[node * 2], tree[node * 2 + 1])
                
        def query(node, l, r, val):
            if l == r:
                return l
            mid = (l + r) // 2
            if tree[node * 2] >= val:
                return query(node * 2, l, mid, val)
            else:
                return query(node * 2 + 1, mid + 1, r, val)
                
        def update(node, l, r, idx):
            if l == r:
                tree[node] = -1
            else:
                mid = (l + r) // 2
                if idx <= mid:
                    update(node * 2, l, mid, idx)
                else:
                    update(node * 2 + 1, mid + 1, r, idx)
                tree[node] = max(tree[node * 2], tree[node * 2 + 1])

        build(1, 0, n - 1)
        
        unplaced = 0
        for f in fruits:
            if tree[1] >= f:
                idx = query(1, 0, n - 1, f)
                update(1, 0, n - 1, idx)
            else:
                unplaced += 1
                
        return unplaced
```

class Solution:
    def numOfUnplacedFruits(self, fruits: List[int], baskets: List[int]) -> int:
        n = len(fruits)
        N = 1
        while N < n:
            N <<= 1
            
        tree = [0] * (2 * N)
        tree[N:N+n] = baskets
        
        for i in range(N - 1, 0, -1):
            tree[i] = max(tree[2 * i], tree[2 * i + 1])
            
        unplaced = 0
        for f in fruits:
            if tree[1] < f:
                unplaced += 1
                continue
                
            node = 1
            while node < N:
                if tree[2 * node] >= f:
                    node = 2 * node
                else:
                    node = 2 * node + 1
                    
            idx = node - N
            tree[node] = -1
            node //= 2
            while node:
                tree[node] = max(tree[2 * node], tree[2 * node + 1])
                node //= 2
                
        return unplaced
```
