=== ANSWER ===
```python

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    Q = int(next(it))
    stack = [0] * 100          # initial 100 cards labeled 0
    out = []
    for _ in range(Q):
        t = next(it)
        if t == '1':
            x = int(next(it))
            stack.append(x)    # push on top
        else:                  # t == '2'
            out.append(str(stack.pop()))  # pop and record
    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
```
