    ```python
            self.n = n
            self.tree = [0] * (n + 1)
        def add(self, i, val):
            while i <= self.n:
                self.tree[i] += val
                i += i & (-i)
        def query(self, i):
            s = 0
            while i > 0:
                s += self.tree[i]
                i -= i & (-i)
            return s
    ```

```python
import sys

# Set recursion limit for CDQ Divide and Conquer
sys.setrecursionlimit(300000)

def solve():
    # Fast input reading
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    ptr = 0
    N = int(input_data[ptr]); ptr += 1
    A = [int(input_data[ptr + i]) for i in range(N)]; ptr += N
    B = [int(input_data[ptr + i]) for i in range(N)]; ptr += N
    K = int(input_data[ptr]); ptr += 1

    # Events: (val, type, x, y, id)
    # type 0: A_i, type 1: B_j, type 2: Query
    events = []
    for i in range(N):
        # A_i is at (i+1, 0) with value A[i]
        events.append((A[i], 0, i + 1, 0, -1))
    for j in range(N):
        # B_j is at (0, j+1) with value B[j]
        events.append((B[j], 1, 0, j + 1, -1))

    for k in range(K):
        X = int(input_data[ptr]); Y = int(input_data[ptr + 1]); ptr += 2
        # Query is at (X, Y). val = -1 ensures it's always in the right half
        # when sorted descending by val.
        events.append((-1, 2, X, Y, k))

    # Sort events by val descending
    events.sort(key=lambda e: -e[0])

    ans = [0] * K

    class FenwickPair:
        def __init__(self, n):
            self.n = n
            self.cnt = [0] * (n + 1)
            self.sum = [0] * (n + 1)

        def add(self, i, c, s):
            while i <= self.n:
                self.cnt[i] += c
                self.sum[i] += s
                i += i & (-i)

        def query(self, i):
            c, s = 0, 0
            while i > 0:
                c += self.cnt[i]
                s += self.sum[i]
                i -= i & (-i)
            return c, s

    # 4 Fenwick trees to track counts and sums for Left/Right halves of A and B
    l_A = FenwickPair(N + 2)
    r_A = FenwickPair(N + 2)
    l_B = FenwickPair(N + 2)
    r_B = FenwickPair(N + 2)

    def cdq(l, r):
        if l >= r:
            return
        mid = (l + r) // 2
        cdq(l, mid)
        cdq(mid + 1, r)

        sub = []
        for k in range(l, mid + 1):
            sub.append(events[k] + (True,))
        for k in range(mid + 1, r + 1):
            sub.append(events[k] + (False,))

        # Sort by x, then by type (A=0, B=1, Query=2)
        # This ensures additions happen before queries for the same x
        sub.sort(key=lambda e: (e[2], e[1]))

        for e in sub:
            val, typ, x, y, idx, is_left = e
            if is_left:
                if typ == 0:
                    l_A.add(x, 1, val)
                elif typ == 1:
                    l_B.add(y, 1, val)
            else:
                if typ == 0:
                    r_A.add(x, 1, val)
                elif typ == 1:
                    r_B.add(y, 1, val)
                elif typ == 2:
                    # Pairs of (Left A, Right B): A_val >= B_val, add A - B
                    cA, sA = l_A.query(x)
                    cB, sB = r_B.query(y)
                    ans[idx] += sA * cB - sB * cA

                    # Pairs of (Left B, Right A): B_val >= A_val, add B - A
                    cB2, sB2 = l_B.query(y)
                    cA2, sA2 = r_A.query(x)
                    ans[idx] += sB2 * cA2 - sA2 * cB2

        # Clear Fenwick trees for the next merge step
        for e in sub:
            val, typ, x, y, idx, is_left = e
            if is_left:
                if typ == 0:
                    l_A.add(x, -1, -val)
                elif typ == 1:
                    l_B.add(y, -1, -val)
            else:
                if typ == 0:
                    r_A.add(x, -1, -val)
                elif typ == 1:
                    r_B.add(y, -1, -val)

        # Merge the sorted halves back to the events array
        for k in range(len(sub)):
            events[l + k] = sub[k][:6]

    cdq(0, len(events) - 1)

    # Print all answers
    print(*ans, sep='\n')

if __name__ == '__main__':
    solve()
```
