def solve():

            return

        for L, R in intervals:

            k_high = (R - 1) // step            # floor((R-1)/step)
            if k_low <= k_high and k_low <= K_max:

                print("No")
                return
        print("Yes")
        return

    # A < B

    LIMIT = 5000  # safe upper bound
    reachable = [False] * (LIMIT + 1)

                break
        else:
            consecutive = 0
    if C is None:

        C = LIMIT + 1

    # Build segments
    segments = []
    prev = 1
    for L, R in intervals:
        if prev <= L - 1:

        prev = R + 1
    if prev <= N:

    K = len(segments)

    # Process segments
    for i in range(K):
        S, E = segments[i]
        if i == K - 1:
            # last segment, check if N reachable
            reachable_N = False
            for e in entries:
                if N == e:
                    reachable_N = True
                    break
                d = N - e
                if d < 0:
                    continue
                if d >= C:
                    reachable_N = True
                    break
                if d > 0 and reachable[d]:
                    reachable_N = True
                    break
            print("Yes" if reachable_N else "No")
            return
        else:
            S2, E2 = segments[i + 1]
            gap = S2 - E
            if gap > B:
                print("No")
                return
            # candidate x range
            x_start = max(S, S2 - B)

                reachable_x = False
                for e in entries:
                    if x == e:

                    y = x + step
                    if S2 <= y <= E2:
                        next_entries.add(y)
            if not next_entries:

                return
