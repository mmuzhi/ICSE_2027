```python
def main():
```

```python

import sys

def main():
    X = int(sys.stdin.readline().strip())
    total = 2025  # sum of all 81 entries
    cnt = 0
    for i in range(1, 10):
        if X % i == 0 and X // i <= 9:
            cnt += 1
    ans = total - cnt * X
    print(ans)

if __name__ == "__main__":
    main()
```
