import sys
from collections import deque

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))

    adj = [[] for _ in range(N + 1)]
    incoming_to_1 = []  # vertices u with an edge u -> 1

    for _ in range(M):
        a = int(next(it))
        b = int(next(it))
        adj[a].append(b)
        if b == 1:
            incoming_to_1.append(a)

    # BFS from vertex 1
    dist = [-1] * (N + 1)
    dist[1] = 0
    q = deque([1])
    while q:
        u = q.popleft()
        for v in adj[u]:
            if dist[v] == -1:
                dist[v] = dist[u] + 1
                q.append(v)

    INF = 10 ** 9
    ans = INF
    for u in incoming_to_1:
        if dist[u] != -1:
            ans = min(ans, dist[u] + 1)

    print(-1 if ans == INF else ans)

if __name__ == "__main__":
    main()
