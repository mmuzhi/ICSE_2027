def main():
    S = input().strip()
    stack = []
    match = {')': '(', ']': '[', '>': '<'}
    for c in S:
        if c in '([<':
            stack.append(c)
        else:
            if not stack or stack[-1] != match[c]:
                print("No")
                return
            stack.pop()
    print("Yes" if not stack else "No")
