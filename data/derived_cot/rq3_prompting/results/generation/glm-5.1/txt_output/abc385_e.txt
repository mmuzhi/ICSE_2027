import sys
from collections import defaultdict

def solve():
    input = sys.stdin.readline
    N = int(input())
    adj = defaultdict(list)
    for _ in range(N - 1):
        u, v = map(int, input().split())
        adj[u].append(v)
        adj[v].append(u)
    
    # Compute degrees
    deg = [0] * (N + 1)
    for i in range(1, N + 1):
        deg[i] = len(adj[i])
    
    max_keep = 0
    for c in range(1, N + 1):
        # Get degrees of neighbors
        neighbor_degs = [deg[nb] for nb in adj[c]]
        neighbor_degs.sort()
        k = len(neighbor_degs)
        
        for j in range(1, k + 1):
            d = neighbor_degs[k - j]  # d_{k-j+1} in 1-indexed, which is d_{k-j} in 0-indexed
            if d >= 2:
                keep = 1 + j * d
                max_keep = max(max_keep, keep)
    
    print(N - max_keep)
