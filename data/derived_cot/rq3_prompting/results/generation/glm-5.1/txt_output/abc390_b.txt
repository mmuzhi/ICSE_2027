```python

import sys

def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))

    # Two terms always form a geometric progression
    if N == 2:
        print("Yes")
        return

    # Check the condition A_i * A_{i+2} == A_{i+1}^2 for all i
    for i in range(N - 2):
        if A[i] * A[i + 2] != A[i + 1] * A[i + 1]:
            print("No")
            return
    print("Yes")

if __name__ == "__main__":
    main()
```
