class Solution:
    def minCost(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return nums[0]
        if n == 2:
            return max(nums[0], nums[1])
        
        # dp[i][j]: min cost to remove elements from i to n-1, with leftover nums[j]
        # j can be -1, we map it to n
        dp = [[-1] * (n + 1) for _ in range(n + 1)]
        
        def solve(i, j):
            if dp[i][j] != -1:
                return dp[i][j]
            
            if j == n: # means j == -1
                if i == n:
                    ans = 0
                elif i == n - 1:
                    ans = nums[i]
                elif i == n - 2:
                    ans = max(nums[i], nums[i+1])
                else:

                    c3 = max(nums[i+1], nums[i+2]) + solve(i+3, i)
