```python
import sys

def main():
    data = sys.stdin.read().split()
    N = int(data[0])
    A = list(map(int, data[1:]))

    # prefix distinct counts
    prefix = [0] * N
    seen = [False] * (N + 1)
    cnt = 0
    for i in range(N):
        if not seen[A[i]]:
            seen[A[i]] = True
            cnt += 1
        prefix[i] = cnt

    # suffix distinct counts
    suffix = [0] * N
    seen = [False] * (N + 1)
    cnt = 0
    for i in range(N - 1, -1, -1):
        if not seen[A[i]]:
            seen[A[i]] = True
            cnt += 1
        suffix[i] = cnt

    # compute maximum sum
    ans = 0
    for i in range(N - 1):
        total = prefix[i] + suffix[i + 1]
        if total > ans:
            ans = total

    print(ans)

if __name__ == "__main__":
    main()
```
