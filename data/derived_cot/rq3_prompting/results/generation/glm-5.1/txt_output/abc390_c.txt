def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    H = int(data[0])

    grid = data[2:]
    
    r_min, r_max = H, -1
    c_min, c_max = W, -1

                if i < r_min: r_min = i
                if i > r_max: r_max = i
                if j < c_min: c_min = j
                if j > c_max: c_max = j
                
    # Check if there is any white cell (.) inside the bounding box
    for i in range(r_min, r_max + 1):
        for j in range(c_min, c_max + 1):
            if grid[i][j] == '.':
