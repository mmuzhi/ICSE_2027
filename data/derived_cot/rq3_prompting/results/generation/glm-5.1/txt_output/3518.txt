=== ANSWER ===
```python
from typing import List

class Solution:
    def maxScore(self, a: List[int], b: List[int]) -> int:
        dp = [-10**16] * 4
        for x in b:
            for i in range(3, 0, -1):
                dp[i] = max(dp[i], dp[i-1] + a[i] * x)
            dp[0] = max(dp[0], a[0] * x)
        return dp[3]
```
