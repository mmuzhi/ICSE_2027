class Solution:
    def sortMatrix(self, grid: List[List[int]]) -> List[List[int]]:
        from collections import defaultdict
        
        diags = defaultdict(list)
        n = len(grid)
        for r in range(n):
            for c in range(n):
                diags[c - r].append(grid[r][c])
                
        for k, vals in diags.items():
            if k <= 0:
                vals.sort(reverse=True)
            else:
                vals.sort()
                
        for r in range(n):
            for c in range(n):
                grid[r][c] = diags[c - r].pop(0)
                
        return grid
```
