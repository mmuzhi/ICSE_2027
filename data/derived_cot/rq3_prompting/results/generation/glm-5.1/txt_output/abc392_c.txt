```python
import sys

def main():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    N = int(input_data[0])
    P = [int(x) - 1 for x in input_data[1:N+1]]
    Q = [int(x) - 1 for x in input_data[N+1:2*N+1]]
    
    R = [0] * N
    for i in range(N):
        R[Q[i]] = i
        
    ans = [0] * N
    for i in range(N):
        ans[i] = Q[P[R[i]]] + 1
        
    print(*ans)

if __name__ == '__main__':
    main()
```

```python
import sys

def main():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    N = int(input_data[0])
    P = [int(x) - 1 for x in input_data[1:N+1]]
    Q = [int(x) - 1 for x in input_data[N+1:2*N+1]]
    
    # R[i] is the person wearing bib i
    R = [0] * N
    for i in range(N):
        R[Q[i]] = i
        
    # For bib i, the person wearing it is R[i]
    # They are staring at P[R[i]]
    # The bib of the person they are staring at is Q[P[R[i]]]
    ans = [0] * N
    for i in range(N):
        ans[i] = Q[P[R[i]]] + 1
        
    print(*ans)

if __name__ == '__main__':
    main()
```
