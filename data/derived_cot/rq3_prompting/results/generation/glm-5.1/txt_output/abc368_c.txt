    partial = [[0, 3, 4], [0, 1, 2], [0, 1, 4]]
    
    S = 1
    for i in range(N):
        r = S % 3
        h = H[i]
        
        best = 3 * ((h + 4) // 5)
        
        p1 = partial[r][1]
        if p1 >= h:
            best = min(best, 1)
        else:
            best = min(best, 3 * ((h - p1 + 4) // 5) + 1)
        
        p2 = partial[r][2]
        if p2 >= h:
            best = min(best, 2)
        else:
            best = min(best, 3 * ((h - p2 + 4) // 5) + 2)
        
        S = S + best
    
    print(S - 1)

solve()
