In the nation of Atcoder, there are N cities numbered 1 to N, and M trains numbered 1 to M.

Given a positive integer X_1, find a way to set non-negative integers X_2,\ldots,X_M that satisfies the following condition with the minimum possible value of X_2+\ldots+X_M.

Now, Fenwick implementation:

class Fenwick:
    def __init__(self, n):
        self.n = n
        self.bit = [-10**18] * (n+2)  # 1-indexed
    def update(self, i, val):
        while i <= self.n:
            if val > self.bit[i]:
                self.bit[i] = val
            i += i & -i
    def query(self, i):
        res = -10**18
        while i > 0:
            if self.bit[i] > res:
                res = self.bit[i]
            i -= i & -i
        return res

```python
import sys
import bisect

class Fenwick:
    def __init__(self, n):
        self.n = n
        self.bit = [-10**18] * (n+2)
    def update(self, i, val):
        while i <= self.n:
            if val > self.bit[i]:
                self.bit[i] = val
            i += i & -i
    def query(self, i):
        res = -10**18
        while i > 0:
            if self.bit[i] > res:
                res = self.bit[i]
            i -= i & -i
        return res

def main():
    import sys
    input = sys.stdin.readline
    N, M, X1 = map(int, input().split())
    A = [0]*(M+1)
    B = [0]*(M+1)
    S = [0]*(M+1)
    T = [0]*(M+1)
    L = [0]*(M+1)
    for i in range(1, M+1):
        a, b, s, t = map(int, input().split())
        A[i] = a
        B[i] = b
        S[i] = s
        T[i] = t
        L[i] = t - s

    # collect times per city
    times = [[] for _ in range(N+1)]
    for i in range(1, M+1):
        times[A[i]].append(S[i])
        times[B[i]].append(T[i])

    for v in range(1, N+1):
        if times[v]:
            times[v] = sorted(set(times[v]))

    # create fenwick trees
    ft = [None]*(N+1)
    for v in range(1, N+1):
        if times[v]:
            ft[v] = Fenwick(len(times[v]))

    # order by S
    order = list(range(1, M+1))
    order.sort(key=lambda i: (S[i], i))

    D = [0]*(M+1)

    for i in order:
        if i == 1:
            D[i] = S[1] + X1
        else:
            v = A[i]
            max_val = -10**18
            if ft[v] is not None:
                # find rightmost index with time <= S[i]
                pos = bisect.bisect_right(times[v], S[i]) - 1
                if pos >= 0:
                    max_val = ft[v].query(pos+1)
            D[i] = max(S[i], max_val)
        V = D[i] + L[i]
        v2 = B[i]
        # insert arrival
        if ft[v2] is not None:
            pos2 = bisect.bisect_left(times[v2], T[i])  # 0-based
            ft[v2].update(pos2+1, V)

    # output X_2..X_M
    out = [str(D[i] - S[i]) for i in range(2, M+1)]
    print(' '.join(out))

if __name__ == "__main__":
    main()
```

```python
import sys
import bisect

class Fenwick:
    def __init__(self, n):
        self.n = n
        self.bit = [-10**18] * (n + 2)

    def update(self, i, val):
        while i <= self.n:
            if val > self.bit[i]:
                self.bit[i] = val
            i += i & -i

    def query(self, i):
        res = -10**18
        while i > 0:
            if self.bit[i] > res:
                res = self.bit[i]
            i -= i & -i
        return res

def main():
    input = sys.stdin.readline
    N, M, X1 = map(int, input().split())
    A = [0] * (M + 1)
    B = [0] * (M + 1)
    S = [0] * (M + 1)
    T = [0] * (M + 1)
    L = [0] * (M + 1)
    for i in range(1, M + 1):
        a, b, s, t = map(int, input().split())
        A[i] = a
        B[i] = b
        S[i] = s
        T[i] = t
        L[i] = t - s

    # collect all times that will be used as keys in each city
    times = [[] for _ in range(N + 1)]
    for i in range(1, M + 1):
        times[A[i]].append(S[i])   # departure times
        times[B[i]].append(T[i])   # arrival times

    for v in range(1, N + 1):
        if times[v]:
            times[v] = sorted(set(times[v]))

    # Fenwick trees for prefix maximum
    ft = [None] * (N + 1)
    for v in range(1, N + 1):
        if times[v]:
            ft[v] = Fenwick(len(times[v]))

    # topological order: increasing S[i] (ties broken arbitrarily)
    order = list(range(1, M + 1))
    order.sort(key=lambda i: (S[i], i))

    D = [0] * (M + 1)          # D[i] = S[i] + X[i]

    for i in order:
        if i == 1:
            D[i] = S[1] + X1
        else:
            v = A[i]
            best = -10**18
            if ft[v] is not None:
                # largest index with time <= S[i]
                pos = bisect.bisect_right(times[v], S[i]) - 1
                if pos >= 0:
                    best = ft[v].query(pos + 1)
            D[i] = max(S[i], best)

        V = D[i] + L[i]        # delayed arrival time of train i
        v2 = B[i]
        if ft[v2] is not None:
            pos2 = bisect.bisect_left(times[v2], T[i])   # exact index of T[i]
            ft[v2].update(pos2 + 1, V)

    # output X2 .. XM
    ans = [str(D[i] - S[i]) for i in range(2, M + 1)]
    print(' '.join(ans))

if __name__ == "__main__":
    main()
```
