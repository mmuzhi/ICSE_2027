```python
def solve():
    import sys
    input = sys.stdin.readline
    
    N, Q = map(int, input().split())
    instructions = []
    for _ in range(Q):
        H, T = input().split()
        T = int(T)
        instructions.append((H, T))
    
    # Helper functions
    def cw(a, b):
        """Clockwise distance from a to b."""
        return (b - a) % N
    
    def ccw(a, b):
        """Counter-clockwise distance from a to b."""
        return (a - b) % N
    
    def dist(a, b):
        """Shortest distance from a to b."""
        return min(cw(a, b), ccw(a, b))
    
    def on_shortest_path(a, b, c):
        """Is c on the shortest path from a to b?"""
        d_cw = cw(a, b)
        d_ccw = ccw(a, b)
        if d_cw <= d_ccw:
            # Shortest path is clockwise
            return 0 < cw(a, c) < d_cw
        else:
            # Shortest path is counter-clockwise
            return 0 < ccw(a, c) < d_ccw
    
    def cost_to_move(a, b, x):
        """Minimum cost to move from a to b, avoiding x."""
        if b == x:
            return float('inf')  # Can't reach b without x moving first
        if on_shortest_path(a, b, x):
            return N - dist(a, b)  # Take the longer path
        return dist(a, b)  # Take the shortest path
    
    def compute_cost(L1, R1, L2, R2):
        """Minimum cost to go from (L1, R1) to (L2, R2)."""
        # Try moving L first, then R
        cost_L_first = cost_to_move(L1, L2, R1) + cost_to_move(R1, R2, L2)
        
        # Try moving R first, then L
        cost_R_first = cost_to_move(R1, R2, L1) + cost_to_move(L1, L2, R2)
        
        return min(cost_L_first, cost_R_first)
    
    # DP
    INF = float('inf')
    
    # dp[j] = minimum cost with the free hand at position j
    dp = {}
    
    # Initial state: L = 1, R = 2
    H1, T1 = instructions[0]
    
    for j in range(1, N + 1):
        if j == T1:
            continue
        
        if H1 == 'L':
            cost = compute_cost(1, 2, T1, j)
        else:
            cost = compute_cost(1, 2, j, T1)
        
        dp[j] = cost
    
    # Process subsequent instructions
    for i in range(1, Q):
        H, T = instructions[i]
        H_prev, T_prev = instructions[i - 1]
        
        new_dp = {}
        for j in range(1, N + 1):
            if j == T:
                continue
            
            min_cost = INF
            for j_prev, cost_prev in dp.items():
                if H_prev == 'L':
                    L_prev, R_prev = T_prev, j_prev
                else:
                    L_prev, R_prev = j_prev, T_prev
                
                if H == 'L':
                    L_curr, R_curr = T, j
                else:
                    L_curr, R_curr = j, T
                
                cost = compute_cost(L_prev, R_prev, L_curr, R_curr)
                min_cost = min(min_cost, cost_prev + cost)
            
            new_dp[j] = min_cost
        
        dp = new_dp
    
    # Find the minimum cost
    result = min(dp.values())
    print(result)

solve()

    input = sys.stdin.readline

    def cw(a, b):
        """Clockwise distance from a to b."""
        return (b - a) % N
    
    def ccw(a, b):
        """Counter-clockwise distance from a to b."""
        return (a - b) % N
    
    def dist(a, b):
        """Shortest distance from a to b."""
        return min(cw(a, b), ccw(a, b))
    
    def on_shortest_path(a, b, c):
        """Is c on the shortest path from a to b?"""
        d_cw = cw(a, b)
        d_ccw = ccw(a, b)
        if d_cw <= d_ccw:
            # Shortest path is clockwise (or equal)
            return 0 < cw(a, c) < d_cw
        else:
            # Shortest path is counter-clockwise
            return 0 < ccw(a, c) < d_ccw
    
    def cost_to_move(a, b, x):
        """Minimum cost to move from a to b, avoiding x."""
        if b == x:
            return float('inf')  # Can't reach b without x moving first
        if on_shortest_path(a, b, x):
            return N - dist(a, b)  # Take the longer path
        return dist(a, b)  # Take the shortest path
    
    def compute_cost(L1, R1, L2, R2):
        """Minimum cost to go from (L1, R1) to (L2, R2)."""
        # Try moving L first, then R
        cost_L_first = cost_to_move(L1, L2, R1) + cost_to_move(R1, R2, L2)
