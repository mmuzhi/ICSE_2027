import sys

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    B = [int(next(it)) for _ in range(N)]
    W = [int(next(it)) for _ in range(M)]
    B.sort(reverse=True)
    W.sort(reverse=True)
    PB = [0]*(N+1)
    for i in range(N):
        PB[i+1] = PB[i] + B[i]
    PW = [0]*(M+1)
    for i in range(M):
        PW[i+1] = PW[i] + W[i]
    # suffix max of PB
    maxPB = [0]*(N+1)
    maxPB[N] = PB[N]
    for i in range(N-1, -1, -1):
        maxPB[i] = max(PB[i], maxPB[i+1])
    ans = 0
    limit = min(M, N)
    for j in range(limit+1):
        val = PW[j] + maxPB[j]
        if val > ans:
            ans = val
    print(ans)

if __name__ == "__main__":
