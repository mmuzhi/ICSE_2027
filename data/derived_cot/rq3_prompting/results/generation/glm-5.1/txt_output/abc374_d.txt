There is a printing machine that prints line segments on the xy-plane by emitting a laser.

- At the start of printing, the laser position is at coordinate (0, 0).
-

- First, move the laser position to one of the endpoints of the line segment.

-

- When emitting the laser, the laser position can move along the line segment being printed at a speed of T units per second.

```python
import sys
import math
import itertools

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    N = int(input_data[0])
    S = int(input_data[1])
    T = int(input_data[2])
    
    segments = []
    idx = 3
    for _ in range(N):
        A = int(input_data[idx])
        B = int(input_data[idx+1])
        C = int(input_data[idx+2])
        D = int(input_data[idx+3])
        segments.append(((A, B), (C, D)))
        idx += 4
        
    min_time = float('inf')
    
    # Iterate over all possible permutations of the N segments
    for perm in itertools.permutations(range(N)):
        # Iterate over all 2^N possible direction choices
        # 0: print from (A_i, B_i) to (C_i, D_i)
        # 1: print from (C_i, D_i) to (A_i, B_i)
        for i in range(1 << N):
            current_pos = (0, 0)
            total_time = 0.0
            
            for j_idx, seg_idx in enumerate(perm):
                p1, p2 = segments[seg_idx]
                
                # Determine start and end points based on the bit
                if (i >> j_idx) & 1:
                    start, end = p2, p1
                else:
                    start, end = p1, p2
                
                # Time to move to the start point without emitting the laser
                dist_move = math.hypot(current_pos[0] - start[0], current_pos[1] - start[1])
                total_time += dist_move / S
                
                # Time to print the segment while emitting the laser
                dist_print = math.hypot(start[0] - end[0], start[1] - end[1])
                total_time += dist_print / T
                
                # Update current position
                current_pos = end
                
            if total_time < min_time:
                min_time = total_time
                
    print(min_time)

if __name__ == '__main__':
    solve()
```
