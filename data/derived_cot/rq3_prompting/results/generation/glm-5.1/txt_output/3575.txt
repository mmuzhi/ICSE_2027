class Solution:
    def maxValue(self, nums: List[int], k: int) -> int:
        N = len(nums)
        dp_left = [[0] * (k + 1) for _ in range(N + 1)]
        dp_left[0][0] = 1
        
        for i in range(1, N + 1):
            num = nums[i-1]
            for j in range(min(i, k) + 1):
                dp_left[i][j] = dp_left[i-1][j]
            for j in range(min(i, k), 0, -1):
                mask = dp_left[i-1][j-1]
                while mask:
                    v = (mask & -mask).bit_length() - 1
                    dp_left[i][j] |= (1 << (v | num))
                    mask &= mask - 1
                    
        dp_right = [[0] * (k + 1) for _ in range(N + 1)]
        dp_right[N][0] = 1
        
        for i in range(N - 1, -1, -1):
            num = nums[i]
            for j in range(min(N - i, k) + 1):
                dp_right[i][j] = dp_right[i+1][j]
            for j in range(min(N - i, k), 0, -1):
                mask = dp_right[i+1][j-1]
                while mask:
                    v = (mask & -mask).bit_length() - 1
                    dp_right[i][j] |= (1 << (v | num))
                    mask &= mask - 1
                    
        ans = 0
        for i in range(k, N - k + 1):
            mask_left = dp_left[i][k]
            left_vals = []
            while mask_left:
                v = (mask_left & -mask_left).bit_length() - 1
                left_vals.append(v)
                mask_left &= mask_left - 1
                
            mask_right = dp_right[i][k]
            right_vals = []
            while mask_right:
                v = (mask_right & -mask_right).bit_length() - 1
                right_vals.append(v)
                mask_right &= mask_right - 1
                
            for A in left_vals:
                for B in right_vals:
                    ans = max(ans, A ^ B)
                    
        return ans
