class Solution:
    def checkValidCuts(self, n: int, rectangles: List[List[int]]) -> bool:
        def count_merged(intervals):
            if not intervals:
                return 0
            intervals.sort()
            count = 1
            cur_end = intervals[0][1]
            for start, end in intervals[1:]:
                if start >= cur_end:
                    count += 1
                    cur_end = end
                else:
                    cur_end = max(cur_end, end)
            return count
        
        x_intervals = [(r[0], r[2]) for r in rectangles]
        y_intervals = [(r[1], r[3]) for r in rectangles]
        
        return count_merged(x_intervals) >= 3 or count_merged(y_intervals) >= 3
```

```python
class Solution:
    def checkValidCuts(self, n: int, rectangles: List[List[int]]) -> bool:
        def count_merged(intervals):
            if not intervals:
                return 0
            intervals.sort()
            count = 1
            cur_end = intervals[0][1]
            for start, end in intervals[1:]:
                if start >= cur_end:
                    count += 1
                    cur_end = end
                else:
                    cur_end = max(cur_end, end)
            return count
        
        x_intervals = [(r[0], r[2]) for r in rectangles]
        y_intervals = [(r[1], r[3]) for r in rectangles]
        
        return count_merged(x_intervals) >= 3 or count_merged(y_intervals) >= 3
```
