    # Precompute binomial coefficients modulo P
    comb = [[0] * (max_n + 1) for _ in range(max_n + 1)]
    for i in range(max_n + 1):
        comb[i][0] = 1
        for j in range(1, i + 1):
            comb[i][j] = (comb[i-1][j-1] + comb[i-1][j]) % P
            
    # Precompute C[prev][a] which is the polynomial for adding a layer of size 'a'
    # C[prev][a][m] is the number of ways to choose 'm' edges incident to/within 'a'
    # such that every vertex in 'a' has at least one edge to 'prev'.
    C_sparse = [[[] for _ in range(k + 1)] for _ in range(k + 1)]
    for prev in range(1, k + 1):
        for a in range(1, k + 1):
            max_e = prev * a
            max_f = a * (a - 1) // 2
            
            # W[e] = number of ways to choose 'e' edges between prev and a
            # such that every vertex in 'a' is connected to 'prev'
            W = [0] * (max_e + 1)
            for e in range(max_e + 1):
                for j in range(a + 1):
                    if prev * (a - j) >= e:
                        W[e] = (W[e] + (-1)**j * comb[a][j] * comb[prev * (a - j)][e]) % P
            
            # C_arr[m] = sum_{e+f=m} W[e] * comb(max_f, f)
            C_arr = [0] * (max_e + max_f + 1)
            for e in range(max_e + 1):
                if W[e] == 0: continue
                for f in range(max_f + 1):
                    if comb[max_f][f] == 0: continue
                    C_arr[e + f] = (C_arr[e + f] + W[e] * comb[max_f][f]) % P
            
            for m_val, c_val in enumerate(C_arr):
                if c_val:
                    C_sparse[prev][a].append((m_val, c_val))

    # Precompute factorials and inverse factorials
    fact = [1] * (2 * k)
    for i in range(1, 2 * k):
        fact[i] = fact[i-1] * i % P
    inv_fact = [1] * (2 * k)
    if 2 * k - 1 >= 0:
        inv_fact[2 * k - 1] = pow(fact[2 * k - 1], P - 2, P)
        for i in range(2 * k - 2, -1, -1):
            inv_fact[i] = inv_fact[i+1] * (i+1) % P

    # DP states: dp[par][ro][re][prev] is a dictionary mapping edge_count -> ways
    # par: parity of the next layer to be added (0 for odd, 1 for even)
    # ro: remaining vertices for odd layers
    # re: remaining vertices for even layers
    # prev: size of the last added layer
    dp = [[[[{} for _ in range(k + 1)] for _ in range(k)] for _ in range(k + 1)] for _ in range(2)]
    dp[0][k][k-1][1][0] = 1

    ans_arr = [0] * (E_max + 1)

    # Topological order processing based on remaining vertices sum (ro + re)
    for S in range(2 * k - 1, -1, -1):
        for ro in range(k + 1):
            re = S - ro
            if re < 0 or re >= k: continue
            
            for par in range(2):
                for prev in range(1, k + 1):
                    if not dp[par][ro][re][prev]: continue
                    
                    # If no vertices left, add to the final answer array
                    if ro == 0 and re == 0:
                        for m, val in dp[par][ro][re][prev].items():
                            ans_arr[m] = (ans_arr[m] + val) % P
                        continue
                    
                    if par == 0: # Next layer is odd
                        for a in range(1, ro + 1):
                            w = inv_fact[a]
                            for m, val in dp[par][ro][re][prev].items():
                                for k_val, c_val in C_sparse[prev][a]:
                                    new_m = m + k_val
                                    if new_m <= E_max:
                                        nxt = dp[1][ro - a][re][a]
                                        nxt[new_m] = (nxt.get(new_m, 0) + val * c_val % P * w) % P
                    else: # Next layer is even
                        for a in range(1, re + 1):
                            w = inv_fact[a]
                            for m, val in dp[par][ro][re][prev].items():
                                for k_val, c_val in C_sparse[prev][a]:
                                    new_m = m + k_val
                                    if new_m <= E_max:
                                        nxt = dp[0][ro][re - a][a]
                                        nxt[new_m] = (nxt.get(new_m, 0) + val * c_val % P * w) % P

    # Multiply by the multinomial coefficient factor (2k-1)!
    for m in range(E_max + 1):
        ans_arr[m] = ans_arr[m] * fact[2 * k - 1] % P

    # Format output for M = N-1 to N(N-1)/2
    res = []
    for M in range(N - 1, N * (N - 1) // 2 + 1):
        if M <= E_max:
            res.append(str(ans_arr[M]))
        else:
            res.append("0")
    
    print(" ".join(res))

solve()
