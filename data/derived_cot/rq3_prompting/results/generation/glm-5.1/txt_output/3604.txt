class Solution:
    def numberOfWays(self, n: int, x: int, y: int) -> int:
        MOD = 10**9 + 7

        # Stirling numbers of the second kind S(n, k)
        dp = [0] * (n + 1)
        dp[0] = 1
        for i in range(1, n + 1):
            new = [0] * (n + 1)
            for k in range(1, i + 1):
                new[k] = (k * dp[k] + dp[k-1]) % MOD
            dp = new

        max_k = min(n, x)
        fall = 1          # (x)_0
        pow_y = 1         # y^0
        ans = 0
        for k in range(1, max_k + 1):
            fall = fall * (x - k + 1) % MOD   # (x)_k
            pow_y = pow_y * y % MOD           # y^k
            ans = (ans + fall * dp[k] % MOD * pow_y) % MOD

        return ans
