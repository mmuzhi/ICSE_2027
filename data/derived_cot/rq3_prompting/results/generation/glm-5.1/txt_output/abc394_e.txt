```python
import sys
from collections import deque

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input().strip())
    out_mask = [ [0]*N for _ in range(26) ]
    out_neighbors = [ [ [] for _ in range(N) ] for _ in range(26) ]
    edge_exists = [ [False]*N for _ in range(N) ]
    for i in range(N):
        line = input().strip()
        for j, ch in enumerate(line):
            if ch != '-':
                a = ord(ch) - 97
                out_mask[a][i] |= (1 << j)
                out_neighbors[a][i].append(j)
                edge_exists[i][j] = True

    # BFS function using bitsets
    def bfs(initial_pairs):
        visited = [0]*N  # row masks
        dist = [ [-1]*N for _ in range(N) ]
        # set initial
        for i, j in initial_pairs:
            visited[i] |= (1 << j)
            dist[i][j] = 0
        frontier = visited[:]  # copy
        level = 0
        while True:
            # compute predecessors of frontier
            pred = [0]*N
            for a in range(26):
                O = out_mask[a]
                neigh = out_neighbors[a]
                # compute M = O * frontier
                M = [0]*N
                for i in range(N):
                    # iterate over u in neigh[i]
                    for u in neigh[i]:
                        M[i] |= frontier[u]
                # compute T = M * O
                for i in range(N):
                    m = M[i]
                    while m:
                        low = m & -m
                        v = low.bit_length() - 1
                        pred[i] |= O[v]
                        m ^= low
            # new = pred & ~visited
            new = [0]*N
            any_new = False
            for i in range(N):
                nm = pred[i] & ~visited[i]
                if nm:
                    new[i] = nm
                    any_new = True
                    visited[i] |= nm
                    # set distances
                    mm = nm
                    while mm:
                        low = mm & -mm
                        j = low.bit_length() - 1
                        dist[i][j] = level + 1
                        mm ^= low
            if not any_new:
                break
            frontier = new
            level += 1
        return dist

    # even terminals: (i,i)
    even_pairs = [(i, i) for i in range(N)]
    dist_even = bfs(even_pairs)

    # odd terminals: all direct edges
    odd_pairs = []
    for i in range(N):
        for j in range(N):
            if edge_exists[i][j]:
                odd_pairs.append((i, j))
    dist_odd = bfs(odd_pairs)

    # compute answers
    out_lines = []
    for i in range(N):
        row = []
        for j in range(N):
            best = 10**9
            de = dist_even[i][j]
            if de != -1:
                best = min(best, 2 * de)
            do = dist_odd[i][j]
            if do != -1:
                best = min(best, 2 * do + 1)
            if best == 10**9:
                row.append("-1")
            else:
                row.append(str(best))
        out_lines.append(" ".join(row))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```
