MOD = 998244353

def main():
    import sys
    data = sys.stdin.buffer.read().split()
    N = int(data[0])

    maxA = max(A) if A else 0

                phi[j] -= phi[j] // i
    # compute divisors

    for d in range(1, maxA+1):
        for m in range(d, maxA+1, d):
            divs[m].append(d)

    out_lines = []
    for x in A:
        s = 0

import sys

MOD = 998244353

def main():
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    N = int(data[0])
    A = list(map(int, data[1:1+N]))
    maxA = max(A)

    # ---------- Euler totient ----------
    phi = list(range(maxA + 1))
    for i in range(2, maxA + 1):
        if phi[i] == i:                     # i is prime
            for j in range(i, maxA + 1, i):
                phi[j] -= phi[j] // i

    # ---------- divisors ----------
    divs = [[] for _ in range(maxA + 1)]
    for d in range(1, maxA + 1):
        for m in range(d, maxA + 1, d):
            divs[m].append(d)

    # ---------- main loop ----------
    C = [0] * (maxA + 1)      # C[d] = Σ 2^{i-1} over i < current with d|A_i
    f = 0
    p = 1                     # 2^{0}
    out = []

    for x in A:
        s = 0
        divs_x = divs[x]
        for d in divs_x:
            cd = C[d]
            s += phi[d] * cd
            cd += p
            if cd >= MOD:
                cd -= MOD
            C[d] = cd
        s %= MOD
        f = (f * 2 + s) % MOD
        out.append(str(f))
        p = (p * 2) % MOD

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
    main()
