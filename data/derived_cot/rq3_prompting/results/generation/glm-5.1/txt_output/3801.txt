class Solution:
    def beautifulNumbers(self, l: int, r: int) -> int:
        def count(N):
            if N == 0: return 0
            digits = list(map(int, str(N)))
            n = len(digits)
            
            dp_tight = [defaultdict(int), defaultdict(int)]
            dp_loose = [defaultdict(int), defaultdict(int)]
            dp_tight[0][(0, 1)] = 1
            
            for D in digits:
                ndp_tight = [defaultdict(int), defaultdict(int)]
                ndp_loose = [defaultdict(int), defaultdict(int)]
                
                # From dp_loose[0]
                for (s, p), cnt in dp_loose[0].items():
                    ndp_loose[0][(0, 1)] += cnt
                    for d in range(1, 10):
                        ndp_loose[1][(d, d)] += cnt
                        
                # From dp_loose[1]
                for (s, p), cnt in dp_loose[1].items():
                    for d in range(1, 10):
                        ndp_loose[1][(s + d, p * d)] += cnt
                        
                # From dp_tight[0]
                for (s, p), cnt in dp_tight[0].items():
                    for d in range(D + 1):
                        if d == 0:
                            if d < D:
                                ndp_loose[0][(0, 1)] += cnt
                            else:
                                ndp_tight[0][(0, 1)] += cnt
                        else:
                            if d < D:
                                ndp_loose[1][(s + d, p * d)] += cnt
                            else:
                                ndp_tight[1][(s + d, p * d)] += cnt
                                
                # From dp_tight[1]
                for (s, p), cnt in dp_tight[1].items():
                    for d in range(1, D + 1):
                        if d < D:
                            ndp_loose[1][(s + d, p * d)] += cnt
                        else:
                            ndp_tight[1][(s + d, p * d)] += cnt
                            
                dp_tight = ndp_tight
                dp_loose = ndp_loose
                
            count_no_zero = 0
            ans = 0
            for started in range(2):
                for (s, p), cnt in dp_tight[started].items():
                    if started == 1:
                        count_no_zero += cnt
                        if p % s == 0:
                            ans += cnt
                for (s, p), cnt in dp_loose[started].items():
                    if started == 1:
                        count_no_zero += cnt
                        if p % s == 0:
                            ans += cnt
                            
            ans += (N - count_no_zero)
            return ans
            
        return count(r) - count(l - 1)

class Solution:
    def beautifulNumbers(self, l: int, r: int) -> int:
        from collections import defaultdict
        
        def count(N):
            if N == 0: 
                return 0
            digits = list(map(int, str(N)))
            
            # dp_tight[started] and dp_loose[started] map (sum, prod) to count
            dp_tight = [defaultdict(int), defaultdict(int)]
            dp_loose = [defaultdict(int), defaultdict(int)]
            dp_tight[0][(0, 1)] = 1
            
            for D in digits:
                ndp_tight = [defaultdict(int), defaultdict(int)]
                ndp_loose = [defaultdict(int), defaultdict(int)]
                
                # From loose, not started
                for (s, p), cnt in dp_loose[0].items():
                    ndp_loose[0][(0, 1)] += cnt
                    for d in range(1, 10):
                        ndp_loose[1][(d, d)] += cnt
                        
                # From loose, started
                for (s, p), cnt in dp_loose[1].items():
                    for d in range(1, 10):
                        ndp_loose[1][(s + d, p * d)] += cnt
                        
                # From tight, not started
                for (s, p), cnt in dp_tight[0].items():
                    for d in range(D + 1):
                        if d == 0:
                            if d < D:
                                ndp_loose[0][(0, 1)] += cnt
                            else:
                                ndp_tight[0][(0, 1)] += cnt
                        else:
                            if d < D:
                                ndp_loose[1][(s + d, p * d)] += cnt
                            else:
                                ndp_tight[1][(s + d, p * d)] += cnt
                                
                # From tight, started
                for (s, p), cnt in dp_tight[1].items():
                    for d in range(1, D + 1):
                        if d < D:
                            ndp_loose[1][(s + d, p * d)] += cnt
                        else:
                            ndp_tight[1][(s + d, p * d)] += cnt
                            
                dp_tight = ndp_tight
                dp_loose = ndp_loose
                
            count_no_zero = 0
            ans = 0
            for started in range(2):
                for (s, p), cnt in dp_tight[started].items():
                    if started == 1:
                        count_no_zero += cnt
                        if p % s == 0:

                        if p % s == 0:
                            ans += cnt
                            
            # Any number containing at least one '0' digit has a product of 0,
