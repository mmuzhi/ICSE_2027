```python
class Solution:
    def countPairs(self, nums: List[int]) -> int:
        def is_almost_equal(x, y):
            s_x = str(x)
            s_y = str(y)
            max_len = max(len(s_x), len(s_y))
            s_x = s_x.zfill(max_len)
            s_y = s_y.zfill(max_len)
            
            diff = []
            for i in range(max_len):
                if s_x[i] != s_y[i]:
                    diff.append(i)
                    if len(diff) > 2:
                        return False
            
            if len(diff) == 0:
                return True
            if len(diff) == 2:
                i, j = diff
                return s_x[i] == s_y[j] and s_x[j] == s_y[i]
            return False

        ans = 0
        for i in range(len(nums)):
            for j in range(i + 1, len(nums)):
                if is_almost_equal(nums[i], nums[j]):
                    ans += 1
        return ans
```

```python
class Solution:
    def countPairs(self, nums: List[int]) -> int:
        def check(x, y):
            s_x = str(x)
            s_y = str(y)
            L = max(len(s_x), len(s_y))
            s_x = s_x.zfill(L)
            s_y = s_y.zfill(L)
            
            d1, d2 = -1, -1
            for i in range(L):
                if s_x[i] != s_y[i]:
                    if d1 == -1:
                        d1 = i
                    elif d2 == -1:
                        d2 = i
                    else:
                        return False
            
            if d1 == -1:
                return True
            if d2 != -1:
                return s_x[d1] == s_y[d2] and s_x[d2] == s_y[d1]
            return False

        ans = 0
        for i in range(len(nums)):
            for j in range(i + 1, len(nums)):
                if check(nums[i], nums[j]):
                    ans += 1
        return ans
```

```python
    def countPairs(self, nums: List[int]) -> int:
        def check(x, y):
            s_x = str(x)
            s_y = str(y)
            L = max(len(s_x), len(s_y))
            s_x = s_x.zfill(L)
            s_y = s_y.zfill(L)
            
            d1, d2 = -1, -1
            for i in range(L):
                if s_x[i] != s_y[i]:
                    if d1 == -1:
                        d1 = i
                    elif d2 == -1:
                        d2 = i
                    else:
                        return False
            
            if d1 == -1:
                return True
            if d2 != -1:
                return s_x[d1] == s_y[d2] and s_x[d2] == s_y[d1]
            return False

        ans = 0
        for i in range(len(nums)):
            for j in range(i + 1, len(nums)):
                if check(nums[i], nums[j]):
                    ans += 1
        return ans
```
