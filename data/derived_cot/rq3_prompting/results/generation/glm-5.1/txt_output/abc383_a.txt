```python
def main():
    import sys
    input = sys.stdin.readline
    N = int(input())
    water = 0
    prev_t = 0
    for _ in range(N):
        t, v = map(int, input().split())
        water = max(0, water - (t - prev_t))
        water += v
        prev_t = t
    print(water)

if __name__ == "__main__":
    main()
```
