import sys

def main():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    last = {}
    ans = 10**9
    for i, x in enumerate(A):
        if x in last:
            ans = min(ans, i - last[x] + 1)
        last[x] = i
    if ans == 10**9:
        print(-1)
    else:
        print(ans)

if __name__ == "__main__":
    main()

=== ANSWER ===
```python

import sys

def main() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    last_pos = {}
    INF = 10 ** 9
    ans = INF

    for i, x in enumerate(A):
        if x in last_pos:
            length = i - last_pos[x] + 1
            if length < ans:
                ans = length
        last_pos[x] = i

    print(-1 if ans == INF else ans)

if __name__ == "__main__":
    main()
```
