import sys

def fwht(arr):
    """In-place Walsh–Hadamard Transform for XOR convolution."""
    n = len(arr)
    step = 1
    while step < n:
        for i in range(0, n, step * 2):
            for j in range(i, i + step):
                x = arr[j]
                y = arr[j + step]
                arr[j] = x + y
                arr[j + step] = x - y
        step <<= 1

def main():
    input = sys.stdin.readline
    H, W = map(int, input().split())
    N = 1 << W

    # count rows by their bitmask
    a = [0] * N
    for _ in range(H):
        s = input().strip()
        mask = int(s, 2)
        a[mask] += 1

    # precompute popcount for all masks
    popcnt = [0] * N
    for i in range(1, N):
        popcnt[i] = popcnt[i >> 1] + (i & 1)

    # b[mask] = min(popcount(mask), W - popcount(mask))
    b = [0] * N
    for mask in range(N):
        d = popcnt[mask]
        b[mask] = min(d, W - d)

    # XOR convolution via FWHT
    fwht(a)
    fwht(b)
    c = [a[i] * b[i] for i in range(N)]
    fwht(c)
    for i in range(N):
        c[i] //= N   # inverse transform scaling

    ans = min(c)
    print(ans)

if __name__ == "__main__":
    main()
