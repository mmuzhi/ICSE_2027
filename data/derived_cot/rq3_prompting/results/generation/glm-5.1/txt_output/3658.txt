class Solution:
    def minDifference(self, nums: List[int]) -> int:
        def intersect(I1, I2):
            if I1 is None or I2 is None: return None
            l = max(I1[0], I2[0])
            r = min(I1[1], I2[1])
            if l <= r: return [l, r]
            return None

        def check(D):
            max_diff = 0
            for i in range(1, len(nums)):
                if nums[i] != -1 and nums[i-1] != -1:
                    max_diff = max(max_diff, abs(nums[i] - nums[i-1]))
            if max_diff > D:
                return False

            n = len(nums)
            blocks = []
            i = 0
            while i < n:
                if nums[i] == -1:
                    A = nums[i-1] if i > 0 else None
                    j = i
                    while j < n and nums[j] == -1:
                        j += 1
                    B = nums[j] if j < n else None
                    L = j - i
                    if A is None: A = B
                    if B is None: B = A
                    if A is None: A = 1
                    if B is None: B = 1
                    blocks.append({'A': A, 'B': B, 'L': L})
                    i = j
                else:
                    i += 1

            NotE = []
            E = []
            for b in blocks:
                I_mid = intersect([b['A'] - D, b['A'] + D], [b['B'] - D, b['B'] + D])
                I_mid = intersect(I_mid, [1, float('inf')])
                if I_mid is None:
                    if b['L'] == 1:
                        return False
                    E.append(b)
                else:
                    NotE.append(I_mid)

            candidates = []
            if not NotE:
                candidates.append( ([1, float('inf')], [1, float('inf')]) )
            else:
                I_total = NotE[0]
                for I in NotE[1:]:
                    I_total = intersect(I_total, I)
                
                if I_total:
                    candidates.append( (I_total, [1, float('inf')]) )
                    candidates.append( ([1, float('inf')], I_total) )
                else:
                    idx_max_L = max(range(len(NotE)), key=lambda i: NotE[i][0])
                    idx_min_R = min(range(len(NotE)), key=lambda i: NotE[i][1])
                    
                    def get_base(skip_idx):
                        I_x_base = None
                        for i, I in enumerate(NotE):
                            if i == skip_idx: continue
                            if I_x_base is None: I_x_base = I
                            else: I_x_base = intersect(I_x_base, I)
                        return I_x_base

                    I_x_base1 = get_base(idx_max_L)
                    if I_x_base1:
                        candidates.append( (I_x_base1, NotE[idx_max_L]) )
                        candidates.append( (NotE[idx_max_L], I_x_base1) )

                    I_x_base2 = get_base(idx_min_R)
                    if I_x_base2:
                        candidates.append( (I_x_base2, NotE[idx_min_R]) )
                        candidates.append( (NotE[idx_min_R], I_x_base2) )

            if not E:
                I_x = [1, float('inf')]
                I_y = [1, float('inf')]
                for I_x_base, I_y_base in candidates:
                    Valid_x = intersect(I_x, I_x_base)
                    Valid_y = intersect(I_y, I_y_base)
                    if Valid_x and Valid_y:
                        return True
            else:
                for p in range(-1, len(E)):
                    if p == -1:
                        I_x = [E[-1]['B'] - D, E[0]['B'] + D]
                        I_y = [E[-1]['A'] - D, E[0]['A'] + D]
                    elif p == len(E) - 1:
                        I_x = [E[-1]['A'] - D, E[0]['A'] + D]
                        I_y = [E[-1]['B'] - D, E[0]['B'] + D]
                    else:
                        I_x = intersect([E[p]['A'] - D, E[0]['A'] + D], [E[-1]['B'] - D, E[p+1]['B'] + D])
                        I_y = intersect([E[p]['B'] - D, E[0]['B'] + D], [E[-1]['A'] - D, E[p+1]['A'] + D])
                    
                    I_x = intersect(I_x, [1, float('inf')])
                    I_y = intersect(I_y, [1, float('inf')])
                    
                    if not I_x or not I_y:
                        continue
                        
                    for I_x_base, I_y_base in candidates:
                        Valid_x = intersect(I_x, I_x_base)
                        Valid_y = intersect(I_y, I_y_base)
                        if Valid_x and Valid_y:
                            dist = max(0, Valid_x[0] - Valid_y[1], Valid_y[0] - Valid_x[1])
                            if dist <= D:

        low, high = 0, 2 * 10**9

class Solution:
    def minDifference(self, nums: List[int]) -> int:
        def intersect(I1, I2):
            if I1 is None or I2 is None: return None
            l = max(I1[0], I2[0])
            r = min(I1[1], I2[1])
            if l <= r: return [l, r]
            return None

        def check(D):
            max_diff = 0
            for i in range(1, len(nums)):
                if nums[i] != -1 and nums[i-1] != -1:
                    max_diff = max(max_diff, abs(nums[i] - nums[i-1]))
            if max_diff > D:
                return False

            n = len(nums)
            blocks = []
            i = 0
            while i < n:
                if nums[i] == -1:
                    A = nums[i-1] if i > 0 else None
                    j = i
                    while j < n and nums[j] == -1:
                        j += 1
                    B = nums[j] if j < n else None
                    L = j - i
                    if A is None: A = B
                    if B is None: B = A
                    if A is None: A = 1
                    if B is None: B = 1
                    blocks.append({'A': A, 'B': B, 'L': L})
                    i = j
                else:
                    i += 1

            NotE = []
            E = []
            for b in blocks:
                I_mid = intersect([b['A'] - D, b['A'] + D], [b['B'] - D, b['B'] + D])
                I_mid = intersect(I_mid, [1, float('inf')])
                if I_mid is None:
                    if b['L'] == 1:
                        return False
                    E.append(b)
                else:
                    NotE.append(I_mid)

            candidates = []
            if not NotE:
                candidates.append( ([1, float('inf')], [1, float('inf')]) )
            else:
                I_total = NotE[0]
                for I in NotE[1:]:
                    I_total = intersect(I_total, I)
                
                if I_total:
                    candidates.append( (I_total, [1, float('inf')]) )
                    candidates.append( ([1, float('inf')], I_total) )
                else:
                    idx_max_L = max(range(len(NotE)), key=lambda i: NotE[i][0])
                    idx_min_R = min(range(len(NotE)), key=lambda i: NotE[i][1])
                    
                    def get_base(skip_idx):
                        I_x_base = None
                        for i, I in enumerate(NotE):
                            if i == skip_idx: continue
                            if I_x_base is None: I_x_base = I
                            else: I_x_base = intersect(I_x_base, I)
                        return I_x_base

                    I_x_base1 = get_base(idx_max_L)
                    if I_x_base1:
                        candidates.append( (I_x_base1, NotE[idx_max_L]) )
                        candidates.append( (NotE[idx_max_L], I_x_base1) )

                    I_x_base2 = get_base(idx_min_R)
                    if I_x_base2:
                        candidates.append( (I_x_base2, NotE[idx_min_R]) )
                        candidates.append( (NotE[idx_min_R], I_x_base2) )

            if not E:
                I_x = [1, float('inf')]
                I_y = [1, float('inf')]
                for I_x_base, I_y_base in candidates:
                    Valid_x = intersect(I_x, I_x_base)
                    Valid_y = intersect(I_y, I_y_base)
                    if Valid_x and Valid_y:
                        return True
            else:
                for p in range(-1, len(E)):
                    if p == -1:
                        I_x = [E[-1]['B'] - D, E[0]['B'] + D]
                        I_y = [E[-1]['A'] - D, E[0]['A'] + D]
                    elif p == len(E) - 1:
                        I_x = [E[-1]['A'] - D, E[0]['A'] + D]
                        I_y = [E[-1]['B'] - D, E[0]['B'] + D]
                    else:
                        I_x = intersect([E[p]['A'] - D, E[0]['A'] + D], [E[-1]['B'] - D, E[p+1]['B'] + D])
                        I_y = intersect([E[p]['B'] - D, E[0]['B'] + D], [E[-1]['A'] - D, E[p+1]['A'] + D])
                    
                    I_x = intersect(I_x, [1, float('inf')])
                    I_y = intersect(I_y, [1, float('inf')])
                    
                    if not I_x or not I_y:
                        continue
                        
                    for I_x_base, I_y_base in candidates:
                        Valid_x = intersect(I_x, I_x_base)
                        Valid_y = intersect(I_y, I_y_base)
                        if Valid_x and Valid_y:
                            dist = max(0, Valid_x[0] - Valid_y[1], Valid_y[0] - Valid_x[1])
                            if dist <= D:
                                return True
            return False

        low, high = 0, 2 * 10**9
        while low < high:
            mid = (low + high) // 2
            if check(mid):
                high = mid
            else:
                low = mid + 1
        return low
