```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    
    S = data[0]
    Q = int(data[1])
    K_list = [int(x) for x in data[2:]]
    
    N = len(S)

    for K in K_list:

        if flips:
            res.append(c.swapcase())
        else:
            res.append(c)
            
    print(' '.join(res))
```

```python
def solve():
    input = sys.stdin.read
```

```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    
    S = data[0]
    Q = int(data[1])
    K_list = [int(x) for x in data[2:]]
    
    N = len(S)
    res = []
    
    for K in K_list:
        flips = 0
        k = K
        while k > N:
            # Find the largest m such that N * 2^m < k
            m = ((k - 1) // N).bit_length() - 1
            k -= N << m
            flips ^= 1
            
        c = S[k - 1]
        if flips:
            res.append(c.swapcase())
        else:
            res.append(c)
            
    print(' '.join(res))

if __name__ == '__main__':
    solve()
```
