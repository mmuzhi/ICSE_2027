```python
class Solution:
    def minimumPairRemoval(self, nums: List[int]) -> int:
        ops = 0
        while True:
            is_sorted = True
            for i in range(len(nums) - 1):
                if nums[i] > nums[i + 1]:
                    is_sorted = False
                    break
            if is_sorted:
                return ops
            
            min_sum = float('inf')
            min_idx = -1
            for i in range(len(nums) - 1):
                if nums[i] + nums[i + 1] < min_sum:
                    min_sum = nums[i] + nums[i + 1]
                    min_idx = i
            
            nums = nums[:min_idx] + [min_sum] + nums[min_idx + 2:]
            ops += 1
```
