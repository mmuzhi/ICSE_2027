```python
import heapq

class Solution:
    def minOperations(self, n: int, m: int) -> int:
        max_val = 10000
        is_prime = [True] * (max_val + 1)
        is_prime[0] = is_prime[1] = False
        for i in range(2, int(max_val**0.5) + 1):
            if is_prime[i]:
                for j in range(i*i, max_val + 1, i):
                    is_prime[j] = False
                    
        if is_prime[n] or is_prime[m]:
            return -1
            
        if n == m:
            return 0
            
        L = len(str(n))
        pow10 = [10**(L - 1 - i) for i in range(L)]
        
        dist = [float('inf')] * max_val
        dist[n] = n
        pq = [(n, n)]
        
        while pq:
            cost, u = heapq.heappop(pq)
            if cost > dist[u]:
                continue
            if u == m:
                return cost
                
            s = str(u)
            for i in range(L):
                d = int(s[i])
                p = pow10[i]
                
                if d < 9:
                    v = u + p
                    if not is_prime[v] and cost + v < dist[v]:
                        dist[v] = cost + v
                        heapq.heappush(pq, (cost + v, v))
                        
                if d > 0:
                    if i == 0 and d == 1:
                        pass
                    else:
                        v = u - p
                        if not is_prime[v] and cost + v < dist[v]:
                            dist[v] = cost + v
                            heapq.heappush(pq, (cost + v, v))
                            
        return -1
