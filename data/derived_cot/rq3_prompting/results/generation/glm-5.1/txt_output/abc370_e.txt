    A = list(map(int, input().split()))

    # dp[0] = 1
    total = 1                     # S[0] = sum_{j=0}^{0} dp[j]
    last = {0: 1}                 # sum of dp[j] for each prefix sum value
    pref = 0

    for a in A:
        pref += a
        key = pref - K
        bad = last.get(key, 0)
        dp_i = (total - bad) % MOD
        total = (total + dp_i) % MOD
        last[pref] = (last.get(pref, 0) + dp_i) % MOD

    print(dp_i)

if __name__ == "__main__":
    solve()
