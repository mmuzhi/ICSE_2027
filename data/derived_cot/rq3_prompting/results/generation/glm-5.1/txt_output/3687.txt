```python
    def longestSpecialPath(self, edges: List[List[int]], nums: List[int]) -> List[int]:
        n = len(nums)
        adj = [[] for _ in range(n)]
        for u, v, length in edges:
            adj[u].append((v, length))
            adj[v].append((u, length))
            
        dist_from_root = [0] * n
        stack_dist = [(0, -1, 0)]
        while stack_dist:
            u, p, d = stack_dist.pop()
            dist_from_root[u] = d
            for v, w in adj[u]:
                if v != p:
                    stack_dist.append((v, u, d + w))
                    
        max_val = max(nums) if nums else 0
        last_seen = [-1] * (max_val + 1)
        
        max_len = 0
        min_nodes = 1
        
        path_nodes = []
        
        # stack elements: (u, parent, current_dist, left, visited, old_last_seen_val)
        stack = [(0, -1, 0, 0, False, -1)]
        
        while stack:
            u, parent, current_dist, left, visited, old_last_seen_val = stack.pop()
            
            if visited:
                path_nodes.pop()
                last_seen[nums[u]] = old_last_seen_val
                continue
                
            val = nums[u]
            new_left = max(left, last_seen[val] + 1)
            
            if new_left < len(path_nodes):
                path_len = current_dist - dist_from_root[path_nodes[new_left]]
                num_nodes = len(path_nodes) - new_left + 1
            else:
                path_len = 0
                num_nodes = 1
                
            if path_len > max_len:
                max_len = path_len
                min_nodes = num_nodes
            elif path_len == max_len:
                if num_nodes < min_nodes:
                    min_nodes = num_nodes
                    
            old_last_seen_val = last_seen[val]
            last_seen[val] = len(path_nodes)
            path_nodes.append(u)
            
            stack.append((u, parent, current_dist, left, True, old_last_seen_val))
            
            for v, w in adj[u]:
                if v != parent:
                    stack.append((v, u, current_dist + w, new_left, False, -1))
                    
        return [max_len, min_nodes]
```

```python
from typing import List

class Solution:
    def longestSpecialPath(self, edges: List[List[int]], nums: List[int]) -> List[int]:
        n = len(nums)
        adj = [[] for _ in range(n)]
        for u, v, length in edges:
            adj[u].append((v, length))
            adj[v].append((u, length))
            
        # Compute distance from root (0) for all nodes
        dist_from_root = [0] * n
        stack_dist = [(0, -1, 0)]
        while stack_dist:
            u, p, d = stack_dist.pop()
            dist_from_root[u] = d
            for v, w in adj[u]:
                if v != p:
                    stack_dist.append((v, u, d + w))
                    
        # Array for the last seen depth of each value
        max_val = max(nums) if nums else 0
        last_seen = [-1] * (max_val + 1)
        
        max_len = 0
        min_nodes = 1
        
        path_nodes = []
        
        # Stack elements: (u, parent, current_dist, left, visited, old_last_seen_val)
        stack = [(0, -1, 0, 0, False, -1)]
        
        while stack:
            u, parent, current_dist, left, visited, old_last_seen_val = stack.pop()
            
            if visited:
                # Clean up state on backtrack
                path_nodes.pop()
                last_seen[nums[u]] = old_last_seen_val
                continue
                
            val = nums[u]
            new_left = max(left, last_seen[val] + 1)
            
            if new_left < len(path_nodes):
                path_len = current_dist - dist_from_root[path_nodes[new_left]]
                num_nodes = len(path_nodes) - new_left + 1
            else:
                path_len = 0
                num_nodes = 1
                
            if path_len > max_len:
                max_len = path_len
                min_nodes = num_nodes
            elif path_len == max_len:
                if num_nodes < min_nodes:
                    min_nodes = num_nodes
                    
            old_last_seen_val = last_seen[val]
            last_seen[val] = len(path_nodes)
            path_nodes.append(u)
            
            # Push self to stack for cleanup after children are processed
            stack.append((u, parent, current_dist, left, True, old_last_seen_val))
            
            # Push children to stack
            for v, w in adj[u]:
                if v != parent:
                    stack.append((v, u, current_dist + w, new_left, False, -1))
                    
        return [max_len, min_nodes]
```
