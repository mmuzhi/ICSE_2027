```python
from typing import List
from functools import cmp_to_key

class Solution:
    def minDamage(self, power: int, damage: List[int], health: List[int]) -> int:
        n = len(damage)
        enemies = []
        for i in range(n):
            hits = (health[i] + power - 1) // power
            enemies.append((damage[i], hits))
            
        def compare(a, b):
            # a = (damage_a, hits_a)
            # b = (damage_b, hits_b)
            # we want descending damage / hits
            # a[0] / a[1] vs b[0] / b[1]
            # a[0] * b[1] vs b[0] * a[1]
            val = a[0] * b[1] - b[0] * a[1]
            if val > 0: return -1
            if val < 0: return 1
            return 0
            
        enemies.sort(key=cmp_to_key(compare))
        
        total_damage = 0
        current_time = 0
        for d, h in enemies:
            current_time += h
            total_damage += d * current_time
            
        return total_damage
```

```python
        enemies = [(damage[i], (health[i] + power - 1) // power) for i in range(len(damage))]

        current_time = 0

            total_damage += d * current_time
```
