```python
class Solution:
    def maxAmount(self, initialCurrency: str, pairs1: List[List[str]], rates1: List[float], pairs2: List[List[str]], rates2: List[float]) -> float:
        currencies = set()
        currencies.add(initialCurrency)
        for u, v in pairs1:
            currencies.add(u)
            currencies.add(v)
        for u, v in pairs2:
            currencies.add(u)
            currencies.add(v)
        
        curr_list = list(currencies)
        curr_id = {c: i for i, c in enumerate(curr_list)}
        N = len(curr_list)
        
        def get_max_dist(pairs, rates):
            dist = [[0.0] * N for _ in range(N)]
            for i in range(N):
                dist[i][i] = 1.0
            
            for i, (u, v) in enumerate(pairs):
                uid, vid = curr_id[u], curr_id[v]
                dist[uid][vid] = max(dist[uid][vid], rates[i])
                dist[vid][uid] = max(dist[vid][uid], 1.0 / rates[i])
                
            for k in range(N):
                for i in range(N):
                    for j in range(N):
                        if dist[i][k] * dist[k][j] > dist[i][j]:
                            dist[i][j] = dist[i][k] * dist[k][j]
                            
            return dist

        dist1 = get_max_dist(pairs1, rates1)
        dist2 = get_max_dist(pairs2, rates2)
        
        init_idx = curr_id[initialCurrency]
        ans = 1.0
        for j in range(N):
            if dist1[init_idx][j] > 0 and dist2[j][init_idx] > 0:
                ans = max(ans, dist1[init_idx][j] * dist2[j][init_idx])
                
        return ans
```
