    max_k = min(v//2, W//w)
    if max_k == 0:
        return dp
    new_dp = [INF_NEG] * (W+1)
    for r in range(w):
        T_max = (W - r) // w
        if T_max < 0:
            continue
        lines = []  # list of (m, b, j)
        head = 0
        for t in range(T_max+1):
            idx = r + t*w
            val = dp[idx]
            if val > INF_NEG//2:  # reachable
                m = 2 * t
                b = val - t*t - v*t
                # Add line, maintaining convex hull
                # While there are at least 2 lines in active (len(lines)-head >=2) and the last line is unnecessary
                while len(lines) - head >= 2:
                    m1, b1, j1 = lines[-2]
                    m2, b2, j2 = lines[-1]
                    # Check if line2 (m2,b2) is unnecessary given new line (m,b)
                    # Condition: (b1 - b2)*(m - m2) >= (b2 - b)*(m2 - m1)
                    if (b1 - b2) * (m - m2) >= (b2 - b) * (m2 - m1):
                        lines.pop()
                    else:
                        break
                lines.append((m, b, t))
            # Remove lines from front that are out of j range
            while head < len(lines) and lines[head][2] < t - max_k:

                    new_dp[idx] = new_val
            # else no feasible j, leave as -inf
    return new_dp

def solve():
    import sys
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    W = next(it)
    dp = [-1] * (W+1)
    dp[0] = 0
    for _ in range(N):
        w = next(it)
        v = next(it)
        max_k = min(v // 2, W // w)
        if max_k == 0:
            continue
        new_dp = [-1] * (W+1)
        for r in range(w):
            T_max = (W - r) // w
            if T_max < 0:
                continue
            lines = []  # list of (m, b, j)
            head = 0
            for t in range(T_max+1):
                idx = r + t * w
                val = dp[idx]
                if val >= 0:
                    m = 2 * t
                    b = val - t * t - v * t
                    # Add line, maintaining convex hull
                    while len(lines) - head >= 2:
                        m1, b1, j1 = lines[-2]
                        m2, b2, j2 = lines[-1]
                        # Check if line (m2,b2) is unnecessary given new line (m,b)
                        if (b1 - b2) * (m - m2) >= (b2 - b) * (m2 - m1):
                            lines.pop()
                        else:
                            break
                    lines.append((m, b, t))
                # Remove lines from front that are out of j range
                while head < len(lines) and lines[head][2] < t - max_k:
                    head += 1
                # Query best value at x = t
                if head < len(lines):
                    # Move head to best line for current x
                    while head + 1 < len(lines) and lines[head][0] * t + lines[head][1] <= lines[head+1][0] * t + lines[head+1][1]:
                        head += 1
                    best = lines[head][0] * t + lines[head][1]
                    new_val = best + (-t * t + v * t)
                    if new_val > new_dp[idx]:
                        new_dp[idx] = new_val
        dp = new_dp
    ans = max(dp)
    print(ans)

if __name__ == "__main__":
    solve()

import sys

def solve() -> None:
    it = iter(map(int, sys.stdin.buffer.read().split()))
    N = next(it)
    W = next(it)

    # dp[c] = maximum happiness for total weight exactly c, -1 means unreachable
    dp = [-1] * (W + 1)
    dp[0] = 0

    for _ in range(N):
        w = next(it)
        v = next(it)

        max_k = min(v // 2, W // w)   # never take more than this many items
        if max_k == 0:
            continue

        new_dp = [-1] * (W + 1)

        # process each residue class modulo w
        for r in range(w):
            T = (W - r) // w          # maximal t such that r + t*w <= W
            if T < 0:
                continue

            lines = []                # each element: (slope, intercept, j)
            head = 0                  # first active line in lines

            for t in range(T + 1):
                idx = r + t * w
                val = dp[idx]

                # add line for j = t if the state is reachable
                if val >= 0:
                    m = 2 * t
                    b = val - t * t - v * t

                    # maintain convex hull (pop from back while last line is useless)
                    while len(lines) - head >= 2:
                        m1, b1, _ = lines[-2]
                        m2, b2, _ = lines[-1]
                        # condition for line (m2,b2) to be unnecessary
                        if (b1 - b2) * (m - m2) >= (b2 - b) * (m2 - m1):
                            lines.pop()
                        else:
                            break
                    lines.append((m, b, t))

                # remove lines that would require k > max_k
                while head < len(lines) and lines[head][2] < t - max_k:
                    head += 1

                # query the best line at x = t
                if head < len(lines):
                    # advance head while the next line is better at current x
                    while (head + 1 < len(lines) and
                           lines[head][0] * t + lines[head][1] <=
                           lines[head+1][0] * t + lines[head+1][1]):
                        head += 1
                    best = lines[head][0] * t + lines[head][1]
                    new_val = best + (-t * t + v * t)
                    if new_val > new_dp[idx]:
                        new_dp[idx] = new_val

        dp = new_dp

    answer = max(dp)
    print(answer)

if __name__ == "__main__":
    solve()
