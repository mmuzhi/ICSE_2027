import sys
from collections import deque

def solve():
    input = sys.stdin.readline
    N = int(input())
    adj = [[] for _ in range(N+1)]
    for _ in range(N-1):
        a, b = map(int, input().split())
        adj[a].append(b)
        adj[b].append(a)
    
    # Initial matching
    match = [0] * (N+1)
    for i in range(1, N+1, 2):
        match[i] = i+1
        match[i+1] = i
    
    results = []
    alive = set(range(1, N+1))
    
    for _ in range(N//2):
        # Find all leaves
        leaves = [v for v in alive if sum(1 for u in adj[v] if u in alive) == 1]
        
        best_dist = -1
        best_u, best_v = -1, -1
        best_path = None
        
        for u in leaves:
            # BFS on alternating tree from u
            dist = {}
            parent = {}
            dist[u] = 0
            parent[u] = -1
            queue = deque([u])
            
            while queue:
                v = queue.popleft()
                
                if dist[v] % 2 == 0:  # Follow matched edge
                    w = match[v]
                    if w in alive and w not in dist:
                        dist[w] = dist[v] + 1
                        parent[w] = v
                        queue.append(w)
                else:  # Follow all unmatched edges
                    for w in adj[v]:
                        if w in alive and w != match[v] and w not in dist:
                            dist[w] = dist[v] + 1
                            parent[w] = v
                            queue.append(w)
            
            # Find the farthest leaf
            for v in leaves:
                if v != u and v in dist and dist[v] > best_dist:
                    best_dist = dist[v]
                    best_u, best_v = u, v
                    # Reconstruct path
                    path = []
                    cur = v
                    while cur != -1:
                        path.append(cur)
                        cur = parent.get(cur, -1)
                    best_path = path
        
        results.append((best_u, best_v))
        
        # Flip the matching along the path
        for i in range(0, len(best_path)-1, 2):
            a, b = best_path[i], best_path[i+1]
            match[a] = b
            match[b] = a
        
        # Remove u and v
        alive.discard(best_u)
        alive.discard(best_v)
        match[best_u] = 0
        match[best_v] = 0
    
    for x, y in results:
        print(x, y)
