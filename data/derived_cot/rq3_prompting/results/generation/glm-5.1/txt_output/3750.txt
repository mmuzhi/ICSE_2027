from collections import defaultdict
import bisect

class Solution:
    def solveQueries(self, nums: List[int], queries: List[int]) -> List[int]:
        val_to_indices = defaultdict(list)
        for i, v in enumerate(nums):
            val_to_indices[v].append(i)
        
        for v in val_to_indices:
            val_to_indices[v].sort()
            
        N = len(nums)
        ans = []
        for q in queries:
            indices = val_to_indices[nums[q]]
            if len(indices) == 1:
                ans.append(-1)
            else:
                k = bisect.bisect_left(indices, q)
                L = indices[(k - 1) % len(indices)]
                R = indices[(k + 1) % len(indices)]
                dist_L = min(abs(q - L), N - abs(q - L))
                dist_R = min(abs(q - R), N - abs(q - R))
                ans.append(min(dist_L, dist_R))
        return ans
```
