```python
class Solution:
    def maxSubarraySum(self, nums: List[int]) -> int:
        dp0 = -10**18
        ans = -10**18
        add = 0
        min_map = -10**18
        base_map = {}
        max1 = (-10**18, None)
        max2 = (-10**18, None)
        
        for val in nums:
            dp0_prev = dp0
            dp0 = max(val, dp0 + val)
            ans = max(ans, dp0)
            
            add += val
            T = val - add
            old_min_map = min_map
            min_map = max(min_map, T)
            
            x = val
            old_base_x = base_map.get(x, -10**18)
            old_eff_x = max(old_base_x, old_min_map)
            new_eff_x = max(old_eff_x - val, dp0_prev - add)
            base_map[x] = new_eff_x
            
            # Update max1 and max2
            if x == max1[1]:
                max1 = (new_eff_x, x)
                if max1[0] < max2[0]:
                    max1, max2 = max2, max1
            elif x == max2[1]:
                max2 = (new_eff_x, x)
                if max1[0] < max2[0]:
                    max1, max2 = max2, max1
            else:
                if new_eff_x > max1[0]:
                    max2 = max1
                    max1 = (new_eff_x, x)
                elif new_eff_x > max2[0]:
                    max2 = (new_eff_x, x)
            
            # Find max over y != x
            if max1[1] != x:
                max_base_neq_x = max1[0]
            else:
                max_base_neq_x = max2[0]
                
            max_eff_neq_x = max(min_map, max_base_neq_x)
            overall_max_eff = max(new_eff_x, max_eff_neq_x)
            max_V_y = overall_max_eff + add
            ans = max(ans, max_V_y)
            
        return ans
```

class Solution:
    def maxSubarraySum(self, nums: List[int]) -> int:

        dp0 = -INF
        ans = -INF
        add = 0
        min_map = -INF
        base_map = {}
        max1 = (-INF, None)
        max2 = (-INF, None)
        
        for val in nums:
            dp0_prev = dp0
            dp0 = max(val, dp0 + val)
            ans = max(ans, dp0)
            
            add += val
            T = val - add
            old_min_map = min_map
            min_map = max(min_map, T)

            old_base_x = base_map.get(x, -INF)
            old_eff_x = max(old_base_x, old_min_map)
            new_eff_x = max(old_eff_x - val, dp0_prev - add)
            base_map[x] = new_eff_x
            
            # Update max1 and max2
            if x == max1[1]:
                max1 = (new_eff_x, x)

            elif x == max2[1]:

                if max1[0] < max2[0]:
                    max1, max2 = max2, max1
            else:
                if new_eff_x > max1[0]:
                    max2 = max1
                    max1 = (new_eff_x, x)
                elif new_eff_x > max2[0]:
                    max2 = (new_eff_x, x)
