```python

import sys

def main():
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    A = [int(next(it)) for _ in range(N)]

    if N == 0:
        return

    M = max(A)
    freq = [0] * (M + 1)
    for a in A:
        freq[a] += 1

    # cnt[d] = number of elements divisible by d
    cnt = [0] * (M + 1)
    for d in range(1, M + 1):
        s = 0
        for m in range(d, M + 1, d):
            s += freq[m]
        cnt[d] = s

    # ans[v] = maximum divisor d of v with cnt[d] >= K
    ans = [0] * (M + 1)
    for d in range(M, 0, -1):
        if cnt[d] >= K:
            for m in range(d, M + 1, d):
                if ans[m] == 0:
                    ans[m] = d

    out = '\n'.join(str(ans[a]) for a in A)
    sys.stdout.write(out)

if __name__ == "__main__":
    main()
```
