import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    INF = -10**18          # sufficiently small
    best_odd = INF         # max sum of a valid sequence ending with an odd index
    best_even = INF        # max sum of a valid sequence ending with an even index

    for i, a in enumerate(A, start=1):
        if i & 1:          # odd index
            # start a new sequence with this element
            cand_start = a
            # extend a sequence that ended with an even index
            cand_extend = best_even + a if best_even != INF else INF
            best_odd = max(best_odd, cand_start, cand_extend)
        else:              # even index
            # extend a sequence that ended with an odd index
            cand_extend = best_odd + a if best_odd != INF else INF
            best_even = max(best_even, cand_extend)

    if N % 2 == 0:
        # even N -> final stack size must be even, empty stack (sum 0) is always possible
        ans = max(0, best_even)
    else:
        # odd N -> final stack size must be odd, at least one element is kept
        ans = best_odd

    print(ans)

if __name__ == "__main__":
    solve()
