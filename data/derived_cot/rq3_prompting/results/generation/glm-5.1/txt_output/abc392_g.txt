import sys
from math import cos, sin, pi
from array import array

# ------------------------------------------------------------
# iterative FFT for arrays of doubles (real and imag parts)
# ------------------------------------------------------------
def fft(re, im, invert):
    n = len(re)
    j = 0
    for i in range(1, n):
        bit = n >> 1
        while j & bit:
            j ^= bit
            bit >>= 1
        j ^= bit
        if i < j:
            re[i], re[j] = re[j], re[i]
            im[i], im[j] = im[j], im[i]

    length = 2
    while length <= n:
        ang = 2 * pi / length * (-1 if invert else 1)
        wlen_re = cos(ang)
        wlen_im = sin(ang)
        for i in range(0, n, length):
            w_re = 1.0
            w_im = 0.0
            half = length >> 1
            for j in range(i, i + half):
                u_re = re[j]
                u_im = im[j]
                v_re = re[j + half] * w_re - im[j + half] * w_im
                v_im = re[j + half] * w_im + im[j + half] * w_re
                re[j] = u_re + v_re
                im[j] = u_im + v_im
                re[j + half] = u_re - v_re
                im[j + half] = u_im - v_im
                # update w
                new_w_re = w_re * wlen_re - w_im * wlen_im
                new_w_im = w_re * wlen_im + w_im * wlen_re
                w_re, w_im = new_w_re, new_w_im
        length <<= 1

    if invert:
        inv_n = 1.0 / n
        for i in range(n):
            re[i] *= inv_n
            im[i] *= inv_n

# ------------------------------------------------------------
def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return
    n = int(data[0])
    if n < 3:
        print(0)
        return
    S = list(map(int, data[1:]))
    M = max(S)

    # presence array (bytearray is memory efficient)
    present = bytearray(M + 1)
    for x in S:
        present[x] = 1

    # length for FFT (power of two)
    L = 1
    while L < 2 * M + 1:
        L <<= 1

    # real and imag arrays
    re = array('d', [0.0]) * L
    im = array('d', [0.0]) * L
    for x in S:
        re[x] = 1.0

    # forward FFT
    fft(re, im, False)

    # pointwise square (convolution of f with itself)
    for i in range(L):
        a = re[i]
        b = im[i]
        re[i] = a * a - b * b
        im[i] = 2.0 * a * b

    # inverse FFT
    fft(re, im, True)

    # count fine triplets
    ans = 0
    for s in range(2, 2 * M + 1, 2):   # only even sums
        mid = s >> 1
        if mid <= M and present[mid]:
            # convolution value at s (ordered pairs count)
            val = int(re[s] + 0.5)   # round to nearest integer
            # unordered distinct pairs (A,C) with A<C
            pairs = (val - 1) // 2
            ans += pairs

    print(ans)

# ------------------------------------------------------------

    solve()
