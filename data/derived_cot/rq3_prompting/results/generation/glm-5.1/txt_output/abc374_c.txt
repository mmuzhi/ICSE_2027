import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    K = list(map(int, input().split()))
    total = sum(K)

    M = 1 << N
    sumA = [0] * M

    # DP for subset sums
    for mask in range(1, M):
        lsb = mask & -mask               # lowest set bit
        bit = (lsb.bit_length() - 1)     # index of that bit
        prev = mask ^ lsb
        sumA[mask] = sumA[prev] + K[bit]

    ans = total
    for mask in range(M):
        a = sumA[mask]
        b = total - a
        cur = a if a > b else b
        if cur < ans:
            ans = cur

    print(ans)

if __name__ == "__main__":
    solve()
