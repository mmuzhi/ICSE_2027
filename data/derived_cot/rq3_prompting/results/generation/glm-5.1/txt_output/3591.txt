class Solution:
    def shiftDistance(self, s: str, t: str, nextCost: List[int], previousCost: List[int]) -> int:
        next_prefix = [0] * 27
        prev_prefix = [0] * 27
        for i in range(26):
            next_prefix[i+1] = next_prefix[i] + nextCost[i]
            prev_prefix[i+1] = prev_prefix[i] + previousCost[i]
        
        ans = 0
        for c1, c2 in zip(s, t):
            v1 = ord(c1) - ord('a')
            v2 = ord(c2) - ord('a')
            
            if v1 <= v2:
                cost_f = next_prefix[v2] - next_prefix[v1]
            else:
                cost_f = next_prefix[26] - next_prefix[v1] + next_prefix[v2]
                
            if v1 >= v2:
                cost_b = prev_prefix[v1+1] - prev_prefix[v2+1]
            else:
                cost_b = prev_prefix[26] - prev_prefix[v2+1] + prev_prefix[v1+1]
                
            ans += min(cost_f, cost_b)
            
        return ans
```
