    input = sys.stdin.readline

        L, R = map(int, input().split())
        if L > maxL_at_R[R]:

    ans = 0
    for r in range(1, M+1):

import sys

def solve() -> None:

    N, M = map(int, input().split())
    # maxL_at_R[r] = maximum L_i among intervals with R_i = r

    for r in range(1, M + 1):
        if maxL_at_R[r] > cur_max:
