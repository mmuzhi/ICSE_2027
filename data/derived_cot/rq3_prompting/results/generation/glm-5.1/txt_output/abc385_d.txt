from collections import defaultdict

def main():
    input = sys.stdin.readline
    N, M, S_x, S_y = map(int, input().split())

        x, y = map(int, input().split())

            x = new_x

            x = new_x

            new_x = x + C

            x = new_x

            new_y = y - C
            v_segments[x].append((min(y, new_y), max(y, new_y)))
            y = new_y

            return []

            else:
                merged.append([start, end])
        return merged
    
    for key in h_segments:
        h_segments[key] = merge_intervals(h_segments[key])

        v_segments[key] = merge_intervals(v_segments[key])
