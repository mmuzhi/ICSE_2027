import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    ops = 0

    while True:
        # count positive elements
        pos = sum(1 for x in A if x > 0)
        if pos <= 1:
            break
        A.sort(reverse=True)
        A[0] -= 1
        A[1] -= 1
        ops += 1

    print(ops)

if __name__ == "__main__":
    solve()
