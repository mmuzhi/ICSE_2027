```python
    U[0] = 0

    for c in range(26):

        curr_min = 10**9

            if P[z] < curr_min:
                curr_min = P[z]

            if z - A >= 0:
                val1 = U[z - A] + z + A
            else:
```

```python
class Solution:
    def makeStringGood(self, s: str) -> int:
        freqs = [0] * 26
        for ch in s:
            freqs[ord(ch) - ord('a')] += 1
        
        n = len(s)
        max_f = max(freqs)
        
        def solve_for_T(T):
            # U[z] stores the minimum cost to process characters up to current,
            # such that we pass exactly z occurrences to the next character.
            U = [10**9] * (n + 1)
            U[0] = 0
            max_z = 0
            
            for c in range(26):
                A = freqs[c]
                
                # Precompute prefix minimums of U[z] - 2z
                P = [U[z] - 2 * z for z in range(max_z + 1)]
                prefixMinP = [0] * (max_z + 1)
                curr_min = 10**9
                for z in range(max_z + 1):
                    if P[z] < curr_min:
                        curr_min = P[z]
                    prefixMinP[z] = curr_min
                    
                new_U = [10**9] * (n + 1)
                new_max_z = max_z + A
                
                for z in range(new_max_z + 1):
                    # Option 1: Target frequency for current character is 0
                    val1 = 10**9
                    if z - A >= 0:
                        val1 = U[z - A] + z + A
                    else:
                        val1 = U[0] + z + A
                        
                    # Option 2: Target frequency for current character is T, and A >= T
                    val2 = 10**9
                    if z + T - A >= 0:
                        if z + T - A <= max_z:
                            val2 = U[z + T - A] + z + A - T
                    else:
                        val2 = U[0] + z + A - T
                        
                    # Option 3: Target frequency for current character is T, and A < T
                    val3 = 10**9
                    if z == 0 and T > A:
                        idx = T - A - 1
                        if idx <= max_z:
                            val3 = prefixMinP[idx] + T - A
                            
                    new_U[z] = min(val1, val2, val3)
                    
                U = new_U
                max_z = new_max_z
                
            return U[0]

        # The cost function is unimodal, so we can use ternary search to find the optimal T
        ans = n
        low = 1
        high = max_f
        
        while low <= high:
            mid1 = low + (high - low) // 3
            mid2 = high - (high - low) // 3
            c1 = solve_for_T(mid1)
            c2 = solve_for_T(mid2)
            ans = min(ans, c1, c2)
            if c1 < c2:
                high = mid2 - 1
            else:
                low = mid1 + 1
                
        return ans
```
