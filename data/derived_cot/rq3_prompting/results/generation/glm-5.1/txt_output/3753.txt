class Solution:
    def maxDifference(self, s: str) -> int:
        from collections import Counter
        counts = Counter(s)
        max_odd = max(v for v in counts.values() if v % 2 != 0)
        min_even = min(v for v in counts.values() if v % 2 == 0)
        return max_odd - min_even

=== ANSWER ===
```python
class Solution:
    def maxDifference(self, s: str) -> int:
        from collections import Counter
        counts = Counter(s)
        max_odd = max(v for v in counts.values() if v % 2 != 0)
        min_even = min(v for v in counts.values() if v % 2 == 0)
        return max_odd - min_even
```
