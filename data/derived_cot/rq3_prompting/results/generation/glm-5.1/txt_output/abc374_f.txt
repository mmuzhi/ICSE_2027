def prune_list(states):
    # states: list of (D, cost)
    # Step 1: for each D, keep min cost
    min_cost = {}
    for D, cost in states:
        if D not in min_cost or cost < min_cost[D]:
            min_cost[D] = cost
    # Step 2: sort by D
    items = sorted(min_cost.items())  # list of (D, cost)
    # Step 3: filter to keep strictly decreasing cost
    pruned = []
    for D, cost in items:
        if pruned and cost >= pruned[-1][1]:
            continue
        pruned.append((D, cost))
    return pruned

def solve():
    import sys
    input = sys.stdin.readline
    N, K, X = map(int, input().split())
    T = list(map(int, input().split()))
    total_T = sum(T)
    dp = [[] for _ in range(N+1)]
    dp[0].append((-X, 0))
    for i in range(N):
        # prune dp[i]
        if dp[i]:
            dp[i] = prune(dp[i])
        else:
            continue
        for D, cost in dp[i]:
            # j from i+1 to min(N, i+K)
            max_j = min(N, i+K)
            for j in range(i+1, max_j+1):
                s = j - i
                M = T[j-1]
                new_D = M if M >= D+X else D+X  # max
                new_cost = cost + s * new_D

    if dp[N]:
        dp[N] = prune(dp[N])
    else:
