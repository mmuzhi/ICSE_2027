import sys

def solve() -> None:
    data = sys.stdin.buffer.read().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    A = [int(next(it)) for _ in range(N)]
    B = [int(next(it)) for _ in range(M)]

    # leftmost (lexicographically smallest) subsequence
    L = []
    j = 0
    for i in range(N):
        if j < M and A[i] == B[j]:
            L.append(i)
            j += 1
            if j == M:
                break

    if j != M:          # no subsequence at all
        print("No")
        return

    # rightmost (lexicographically largest) subsequence, compare on the fly
    j = M - 1
    for i in range(N - 1, -1, -1):
        if j >= 0 and A[i] == B[j]:
            if i != L[j]:        # different index → at least two subsequences
                print("Yes")
                return
            j -= 1
            if j < 0:
                break

    # all indices coincide → only one subsequence
    print("No")
