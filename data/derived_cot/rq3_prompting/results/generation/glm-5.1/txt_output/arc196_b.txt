```python
class DSU:
    def __init__(self, n):
        self.parent = list(range(n))
        self.par = [0] * n  # x ^ root(x) = par[x]
        self.size = [1] * n
        self.components = n

    def find(self, i):
        if self.parent[i] == i:
            return i
        root = self.find(self.parent[i])
        self.par[i] ^= self.par[self.parent[i]]
        self.parent[i] = root
        return root

    def get_parity(self, i):
        self.find(i)
        return self.par[i]

    def merge(self, i, j, w):
        # i ^ j = w
        root_i = self.find(i)
        root_j = self.find(j)
        if root_i == root_j:
            return self.get_parity(i) ^ self.get_parity(j) == w
        
        if self.size[root_i] < self.size[root_j]:
            root_i, root_j = root_j, root_i
            # we want to attach root_j to root_i
            # i ^ root_i = par_i
            # j ^ root_j = par_j
            # we need root_j ^ root_i = ?
            # i ^ j = w => (root_i ^ par_i) ^ (root_j ^ par_j) = w
            # root_i ^ root_j = w ^ par_i ^ par_j
            w = w ^ self.get_parity(i) ^ self.get_parity(j)
            self.parent[root_j] = root_i
            self.par[root_j] = w
            self.size[root_i] += self.size[root_j]
            self.components -= 1
            return True
        else:
            # attach root_j to root_i
            # i ^ root_i = par_i
            # j ^ root_j = par_j
            # i ^ j = w => (root_i ^ par_i) ^ (root_j ^ par_j) = w
            # root_i ^ root_j = w ^ par_i ^ par_j
            w = w ^ self.get_parity(i) ^ self.get_parity(j)
            self.parent[root_j] = root_i
            self.par[root_j] = w
            self.size[root_i] += self.size[root_j]
            self.components -= 1
            return True
```

Let's trace this iterative `find`:
If `parent[i] = i`, returns `i`.
Otherwise, it goes up to the root.
`path` contains nodes from `i` down to the child of `root`.
`curr_par` starts at 0.
When we iterate in reverse:
The last element in `path` is the child of `root`.
`curr_par ^= par[last]` -> `curr_par` becomes `par[last]`.
`par[last] = curr_par` -> `par[last]` becomes `par[last]`.
`parent[last] = root`.
The second to last element is the child of `last`.
`curr_par ^= par[second_last]` -> `curr_par` becomes `par[last] ^ par[second_last]`.
`par[second_last] = curr_par`.
This correctly accumulates the parity from the root down to the node!
Wait, let's verify.

`par[2] = 2 ^ 3 = p2`.
`path = [1, 2]`.

`curr_par = 0 ^ p2 = p2`.
`par[2] = p2`. (Correct, `2 ^ 3 = p2`).

Wait, is there any case where `self.parent[i]` is updated but `self.par[i]` is not correctly maintained?
In `find`:
`curr_par` accumulates the XOR sum of `par` values from `root` down to `node`.
Let's trace:
`root` is the root.
`path` is `[node_1, node_2, ..., node_k]` where `node_1` is `i` and `node_k` is the child of `root`.
`reversed(path)` is `node_k, ..., node_1`.
For `node_k`:
`curr_par ^= par[node_k]` -> `curr_par = par[node_k]`.
`par[node_k] = curr_par` -> `par[node_k] = par[node_k]`.
This is correct because `node_k`'s parent is `root`, so `par[node_k]` should be `node_k ^ root`, which is exactly what it was.
For `node_{k-1}`:
`curr_par ^= par[node_{k-1}]` -> `curr_par = par[node_k] ^ par[node_{k-1}]`.
`par[node_{k-1}] = curr_par` -> `par[node_{k-1}] = par[node_k] ^ par[node_{k-1}]`.
This is correct because `node_{k-1}`'s new parent is `root`, so its new parity should be `node_{k-1} ^ root = (node_{k-1} ^ node_k) ^ (node_k ^ root) = par[node_{k-1}] ^ par[node_k]`.
So the logic is perfectly correct.

What about `merge`?
When we merge `root_j` under `root_i`:
`W = w ^ p_i ^ p_j`.
`parent[root_j] = root_i`.
`par[root_j] = W`.
This means `root_j ^ root_i = W`.
Let's check if this maintains the invariant.
We want `i ^ j = w`.
We know `i ^ root_i = p_i` and `j ^ root_j = p_j`.
So `i ^ j = (root_i ^ p_i) ^ (root_j ^ p_j) = root_i ^ root_j ^ p_i ^ p_j`.
We want this to equal `w`.
So `root_i ^ root_j = w ^ p_i ^ p_j = W`.
Thus `par[root_j] = W` correctly sets `root_j ^ root_i = W`.
The invariant is perfectly maintained.

```python
import sys
from array import array

MOD = 998244353

# Precompute powers of 2 up to the maximum possible number of components
max_hw = 10**6 + 5
pow2 = [1] * (2 * max_hw)
for i in range(1, 2 * max_hw):
    pow2[i] = (pow2[i-1] * 2) % MOD

class DSU:
    def __init__(self, n):
        self.parent = array('i', range(n))
        self.par = array('i', [0] * n)
        self.size = array('i', [1] * n)
        self.components = n
        self.valid = True

    def find(self, i):
        if self.parent[i] == i:
            return i
        curr = i
        path = []
        while self.parent[curr] != curr:
            path.append(curr)
            curr = self.parent[curr]
        root = curr
        
        curr_par = 0
        for node in reversed(path):
            curr_par ^= self.par[node]
            self.par[node] = curr_par
            self.parent[node] = root
            
        return root
        
    def merge(self, i, j, w):
        if not self.valid:
            return
        root_i = self.find(i)
        root_j = self.find(j)
        if root_i == root_j:
            if self.par[i] ^ self.par[j] != w:
                self.valid = False
            return
            
        p_i = self.par[i]
        p_j = self.par[j]
        W = w ^ p_i ^ p_j
        
        if self.size[root_i] < self.size[root_j]:
            root_i, root_j = root_j, root_i
            
        self.parent[root_j] = root_i
        self.par[root_j] = W
        self.size[root_i] += self.size[root_j]
        self.components -= 1

def solve():
    input_data = sys.stdin.read().split()
    
    T = int(input_data[0])
    idx = 1
    
    results = []
    
    for _ in range(T):
        H = int(input_data[idx])
        W = int(input_data[idx+1])
        idx += 2
        
        S = input_data[idx:idx+H]
        idx += H
        
        N = 2 * H * W
        dsu = DSU(N)
        
        for i in range(H):
            for j in range(W):
                # 1D indices for D and R variables
                d_curr = i * W + j
                d_prev = ((i - 1 + H) % H) * W + j
                r_curr = H * W + i * W + j
                r_prev = H * W + i * W + ((j - 1 + W) % W)
                
                if S[i][j] == 'A':
                    # Type A: D_{i-1, j} != D_{i, j} and R_{i, j} != R_{i, j-1}
                    dsu.merge(d_prev, d_curr, 1)
                    dsu.merge(r_curr, r_prev, 1)
                else:
                    # Type B: D_{i-1, j} == D_{i, j} and R_{i, j} == R_{i, j-1} and D_{i, j} != R_{i, j-1}
                    dsu.merge(d_prev, d_curr, 0)
                    dsu.merge(r_curr, r_prev, 0)
                    dsu.merge(d_curr, r_prev, 1)
                    
        if dsu.valid:
            results.append(pow2[dsu.components])
        else:
            results.append(0)
            
    print('\n'.join(map(str, results)))

    solve()
```
