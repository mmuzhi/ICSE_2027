```python
class Solution:
    def subsequencesWithMiddleMode(self, nums: list[int]) -> int:
        N = len(nums)
        if N < 5:
            return 0
            
        # Coordinate compression for values
        unique_vals = sorted(list(set(nums)))
        val_to_idx = {v: i for i, v in enumerate(unique_vals)}
        K = len(unique_vals)
        nums = [val_to_idx[v] for v in nums]
        
        MOD = 10**9 + 7
        
        L_cnt = [0] * K
        R_cnt = [0] * K
        
        # Initialize for i_3 = 2
        L_cnt[nums[0]] += 1
        L_cnt[nums[1]] += 1
        for i in range(3, N):
            R_cnt[nums[i]] += 1
            
        x = nums[2]
        
        # Sums over v != x
        S1 = S2 = S3 = S4 = S5 = S6 = S7 = S8 = 0
        for v in range(K):
            if v == x:
                continue
            l = L_cnt[v]
            r = R_cnt[v]
            S1 += l
            S2 += l**2
            S3 += r
            S4 += r**2
            S5 += l*r
            S6 += l**2*r
            S7 += l*r**2
            S8 += l**2*r**2
            
        ans = 0
        
        for i_3 in range(2, N - 2):
            x = nums[i_3]
            L_x = L_cnt[x]
            R_x = R_cnt[x]
            L_o = S1
            R_o = S3
            
            ans_i = 0
            
            # Case 4: c_x = 4 (freq = 5)
            ans_i += (L_x * (L_x - 1) // 2) * (R_x * (R_x - 1) // 2)
            
            # Case 3: c_x = 3 (freq = 4)
            ans_i += (L_x * (L_x - 1) // 2) * R_x * R_o
            ans_i += L_x * L_o * (R_x * (R_x - 1) // 2)
            
            # Case 2: c_x = 2 (freq = 3)
            ans_i += (L_x * (L_x - 1) // 2) * (R_o * (R_o - 1) // 2)
            ans_i += (L_o * (L_o - 1) // 2) * (R_x * (R_x - 1) // 2)
            ans_i += L_x * L_o * R_x * R_o
            
            # Case 1: c_x = 1 (freq = 2)
            Ways1 = L_x * (S1 * (S3**2 - S4) - 2 * S3 * S5 + 2 * S7) // 2
            Ways2 = R_x * (S3 * (S1**2 - S2) - 2 * S1 * S5 + 2 * S6) // 2
            ans_i += Ways1 + Ways2
            
            ans = (ans + ans_i) % MOD
            
            if i_3 == N - 3:
                break
                
            x_old = x
            x_new = nums[i_3 + 1]
            
            # 1. Remove x_new from sums
            if x_new != x_old:
                l = L_cnt[x_new]
                r = R_cnt[x_new]
                S1 -= l; S2 -= l**2; S3 -= r; S4 -= r**2
                S5 -= l*r; S6 -= l**2*r; S7 -= l*r**2; S8 -= l**2*r**2
                
            # 2. Update counts
            L_cnt[x_old] += 1
            R_cnt[x_new] -= 1
            
            # 3. Add x_old to sums
            if x_new != x_old:
                l = L_cnt[x_old]
                r = R_cnt[x_old]
                S1 += l; S2 += l**2; S3 += r; S4 += r**2
                S5 += l*r; S6 += l**2*r; S7 += l*r**2; S8 += l**2*r**2

        return ans
```
