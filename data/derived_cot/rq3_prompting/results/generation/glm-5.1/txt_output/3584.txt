This means there exists a sequence of indices `i < j_1 < j_2 < ... < j_k` in `word1` such that `word1[j_l] == word2[pref_len + 1 + l]` for all `l`.

So we CAN complete the sequence!
This is a solid guarantee.

class Solution:
    def validSequence(self, word1: str, word2: str) -> List[int]:
        n = len(word1)
        m = len(word2)
        
        suf_len = [0] * (n + 1)
        for i in range(n - 1, -1, -1):
            if suf_len[i+1] < m and word1[i] == word2[m - 1 - suf_len[i+1]]:
                suf_len[i] = suf_len[i+1] + 1
            else:
                suf_len[i] = suf_len[i+1]
                
        res = []
        pref_len = 0
        mismatch_used = False
        
        for i in range(n):
            if pref_len == m:
                break
            
            if word1[i] == word2[pref_len]:
                res.append(i)
                pref_len += 1
            elif not mismatch_used and suf_len[i+1] >= m - (pref_len + 1):
                mismatch_used = True
                res.append(i)
                pref_len += 1
                
        if pref_len == m:
            return res
        else:
            return []
