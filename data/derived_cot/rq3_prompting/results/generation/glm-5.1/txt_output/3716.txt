```python
class Solution:
    def longestSubsequence(self, nums: List[int]) -> int:
        M = 301
        # S[v][d] stores the max length of a valid subsequence ending with value v
        # such that the last absolute difference is >= d.
        S = [[0] * M for _ in range(M)]
        for v in range(1, 301):
            S[v][300] = 1
        
        ans = 1
        exact_len = [0] * M
        
        for x in nums:
            for d in range(M):
                exact_len[d] = 0
            
            max_exact = 0
            for d in range(M):
                v1 = x - d
                if v1 >= 1:
                    val = 1 + S[v1][d]
                    if val > exact_len[d]:
                        exact_len[d] = val
                    if val > max_exact:
                        max_exact = val
                        
                v2 = x + d
                if v2 <= 300 and d > 0:
                    val = 1 + S[v2][d]
                    if val > exact_len[d]:
                        exact_len[d] = val
                    if val > max_exact:
                        max_exact = val
            
            if max_exact > ans:
                ans = max_exact
            
            S_x = S[x]
            cur_max = exact_len[300]
            if cur_max > S_x[300]:
                S_x[300] = cur_max
            for d in range(299, -1, -1):
                ed = exact_len[d]
                if ed > cur_max:
                    cur_max = ed
                if cur_max > S_x[d]:
                    S_x[d] = cur_max
                    
        return ans
```
