import sys
from collections import Counter

def main():
    data = list(map(int, sys.stdin.read().split()))
    it = iter(data)
    N = next(it)
    dice = []
    for _ in range(N):
        K = next(it)
        faces = [next(it) for _ in range(K)]
        cnt = Counter(faces)
        dice.append((K, cnt))
    max_prob = 0.0
    for i in range(N):
        K_i, cnt_i = dice[i]
        for j in range(i+1, N):
            K_j, cnt_j = dice[j]
            if len(cnt_i) < len(cnt_j):

            prob = dot / (K_i * K_j)
            if prob > max_prob:
                max_prob = prob
    print("{:.15f}".format(max_prob))

import sys
from collections import Counter

def main():
    data = list(map(int, sys.stdin.read().split()))
    it = iter(data)
    N = next(it)
    dice = []   # list of (K, Counter)
    for _ in range(N):
        K = next(it)
        faces = [next(it) for _ in range(K)]
        cnt = Counter(faces)
        dice.append((K, cnt))

    max_prob = 0.0
    for i in range(N):
        Ki, cnti = dice[i]
        for j in range(i + 1, N):
            Kj, cntj = dice[j]
            # iterate over the smaller counter
            if len(cnti) <= len(cntj):
                small, large = cnti, cntj
            else:
                small, large = cntj, cnti
            dot = 0
            for x, v in small.items():
                dot += v * large.get(x, 0)
            prob = dot / (Ki * Kj)
            if prob > max_prob:
                max_prob = prob

    # print with enough digits to satisfy 1e-8 precision
    print("{:.15f}".format(max_prob))

if __name__ == "__main__":
    main()
