import sys

def main():
    data = sys.stdin.read().split()
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    X = [int(next(it)) - 1 for _ in range(N)]
    A = [int(next(it)) for _ in range(N)]

    # cur[i] = current index after processed steps (initially identity)
    cur = list(range(N))
    # pow[i] = X^{2^b}(i) for the current bit b
    pow = X[:]

    LOG = K.bit_length()
    for b in range(LOG):
        if (K >> b) & 1:
            # cur = pow ∘ cur
            cur = [pow[cur[i]] for i in range(N)]
        # pow = pow ∘ pow  (square the power)
        pow = [pow[pow[i]] for i in range(N)]

    # cur[i] is now X^K(i)
    result = [A[cur[i]] for i in range(N)]
    print(' '.join(map(str, result)))
