```python

import sys

def main():
    input = sys.stdin.readline
    N = int(input().strip())
    S = input().strip()

    if N % 2 == 0:
        print("No")
        return

    mid = N // 2  # 0‑based index of the middle character
    if S[mid] != '/':
        print("No")
        return

    for i in range(mid):
        if S[i] != '1':
            print("No")
            return

    for i in range(mid + 1, N):
        if S[i] != '2':
            print("No")
            return

    print("Yes")

if __name__ == "__main__":
    main()
```
