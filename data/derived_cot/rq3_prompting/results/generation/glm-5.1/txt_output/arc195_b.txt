def solve():

    fixed_B = [b for b in B if b != -1]
    nA = A.count(-1)
    nB = B.count(-1)
    if nA + nB >= N:
        print("Yes")
        return
    # now both fixed lists non-empty
    M = max(max(fixed_A), max(fixed_B))
    K = N - (nA + nB)
    # build frequency
    countA = {}
    for a in fixed_A:
        countA[a] = countA.get(a, 0) + 1
    countB = {}
    for b in fixed_B:
        countB[b] = countB.get(b, 0) + 1
    listA = sorted(countA.keys())
    listB = sorted(countB.keys())
    countsA = [countA[a] for a in listA]
    countsB = [countB[b] for b in listB]
    lenB = len(listB)
    heap = []
    # for each a, find start index in listB

        need = M - a
        # bisect_left on listB

        if j < lenB:

        total = min(countsA[i], countsB[j])

        next_j = j + 1

```python
import sys
import bisect
import heapq

def solve():
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))
    
    fixed_A = [a for a in A if a != -1]
    fixed_B = [b for b in B if b != -1]
    nA = A.count(-1)
    nB = B.count(-1)
    
    if nA + nB >= N:
        print("Yes")
        return
    
    # now both fixed_A and fixed_B are non-empty
    M = max(max(fixed_A), max(fixed_B))
    K = N - (nA + nB)  # positive
    
    # frequency of fixed values
    countA = {}
    for a in fixed_A:
        countA[a] = countA.get(a, 0) + 1
    countB = {}
    for b in fixed_B:
        countB[b] = countB.get(b, 0) + 1
    
    listA = sorted(countA.keys())
    listB = sorted(countB.keys())
    countsA = [countA[a] for a in listA]
    countsB = [countB[b] for b in listB]
    lenB = len(listB)
    
    heap = []
    for i, a in enumerate(listA):
        need = M - a
        j = bisect.bisect_left(listB, need)
        if j < lenB:
            S = a + listB[j]
            heapq.heappush(heap, (S, i, j))
    
    while heap:
        S, i, j = heapq.heappop(heap)
        total = min(countsA[i], countsB[j])
        # advance this a
        if j + 1 < lenB:
            heapq.heappush(heap, (listA[i] + listB[j+1], i, j+1))
        # accumulate other events with same S
        while heap and heap[0][0] == S:
            _, i2, j2 = heapq.heappop(heap)
            total += min(countsA[i2], countsB[j2])
            if j2 + 1 < lenB:
                heapq.heappush(heap, (listA[i2] + listB[j2+1], i2, j2+1))
        if total >= K:
            print("Yes")
            return
    
    print("No")

if __name__ == "__main__":
    solve()
```
