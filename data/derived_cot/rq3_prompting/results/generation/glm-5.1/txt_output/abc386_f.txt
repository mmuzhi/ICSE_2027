def main():

    S = data[1].strip()
    T = data[2].strip()

        print("No")
        return

    prev = list(range(prev_start, prev_end + 1))  # dp[0][j] = j
    # iterate i from 1 to n
    for i in range(1, n + 1):
        j_start = max(0, i - K)
        j_end = min(m, i + K)
        cur_len = j_end - j_start + 1
        cur = [INF] * cur_len

        # local references for speed
        S_i = S[i-1]
        prev_local = prev
        prev_start_local = prev_start
        prev_end_local = prev_end
        j_start_local = j_start
        # we will iterate idx from 0 to cur_len-1
        for idx in range(cur_len):
            j = j_start_local + idx
            if j == 0:
                # dp[i][0] = i
                val = i
            else:
                # a = dp[i-1][j-1] - guaranteed in prev
                a = prev_local[(j-1) - prev_start_local]

                if j-1 >= j_start_local:
                    c = cur[(j-1) - j_start_local]
                else:
                    c = INF

def solve():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    K = int(data[0])
    S = data[1].strip()
    T = data[2].strip()
    n = len(S)
    m = len(T)
    if abs(n - m) > K:
        print("No")
        return
    INF = K + 1
    size = 2 * K + 3
    offset = K + 1
    # prev for i=0
    prev = bytearray([INF]) * size
    for d in range(0, min(K, m) + 1):
        prev[d + offset] = d
    # iterate over i
    for i in range(1, n + 1):
        cur = bytearray([INF]) * size
        d_min = max(-K, -i)
        d_max = min(K, m - i)
        has_le_K = False
        s_i = S[i-1]
        for d in range(d_min, d_max + 1):
            j = i + d
            if j == 0:
                val = i
            else:
                a = prev[d + offset]
                b = prev[d + 1 + offset]
                c = cur[d - 1 + offset]
                if s_i == T[j-1]:
                    val = a
                else:
                    val = min(a, b, c) + 1
                    if val > INF:
                        val = INF
            cur[d + offset] = val
            if val <= K:
                has_le_K = True
        if not has_le_K:
            print("No")
            return
        prev = cur
    d_final = m - n
    if prev[d_final + offset] <= K:
        print("Yes")
    else:
        print("No")
