```python
from typing import List

class Solution:
    def hasIncreasingSubarrays(self, nums: List[int], k: int) -> bool:
        N = len(nums)
        inc_len = [1] * N
        for i in range(1, N):
            if nums[i] > nums[i - 1]:
                inc_len[i] = inc_len[i - 1] + 1
        
        for a in range(N - 2 * k + 1):
            if inc_len[a + k - 1] >= k and inc_len[a + 2 * k - 1] >= k:
                return True
        return False
```
