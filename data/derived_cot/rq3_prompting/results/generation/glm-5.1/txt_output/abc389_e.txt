def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    P = list(map(int, input().split()))
    # cost_exceeds function using closure
    def cost_exceeds(V):
        total = 0
        for p in P:
            k = (V // p + 1) // 2
            if k == 0:
                continue
            if k > M // p:
                return True
            total += k * k * p
            if total > M:
                return True
        return False

    lo = 0
    hi = 2 * 10**18
    while lo < hi:
        mid = (lo + hi) // 2
        if cost_exceeds(mid):
            hi = mid
        else:
            lo = mid + 1
    V0 = lo

    # compute for V0-1
    V_prev = V0 - 1
    total_units = 0
    total_cost = 0
    for p in P:
        k = (V_prev // p + 1) // 2
        total_units += k
        total_cost += k * k * p

    ans = total_units + (M - total_cost) // V0
    print(ans)

```python

import sys

def solve() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())
    P = list(map(int, input().split()))

    # check whether total cost of all units with marginal cost <= V exceeds M
    def cost_exceeds(V: int) -> bool:
        total = 0
        for p in P:
            k = (V // p + 1) // 2          # number of units of this product with cost <= V
            if k == 0:
                continue
            if k > M // p:                 # then k*k*p > M for sure
                return True
            total += k * k * p
            if total > M:
                return True
        return False

    # binary search for the smallest V with S(V) > M
    lo, hi = 0, 2 * 10**18          # hi is large enough (see proof)
    while lo < hi:
        mid = (lo + hi) // 2
        if cost_exceeds(mid):
            hi = mid
        else:
            lo = mid + 1
    V0 = lo

    # compute U(V0-1) and S(V0-1)
    V_prev = V0 - 1
    total_units = 0
    total_cost = 0
    for p in P:
        k = (V_prev // p + 1) // 2
        total_units += k
        total_cost += k * k * p

    # remaining budget buys as many units with cost exactly V0 as possible
    ans = total_units + (M - total_cost) // V0
    print(ans)

if __name__ == "__main__":
    solve()
```
