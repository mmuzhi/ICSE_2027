import sys
from array import array

MOD = 998244353

def main():
    it = iter(sys.stdin.read().split())
    H = int(next(it))
    W = int(next(it))
    raw = [next(it).strip() for _ in range(H)]

    # convert to integer grid: -1 for ?, 0,1,2 for 1,2,3
    grid = [[-1 if ch == '?' else int(ch)-1 for ch in row] for row in raw]

    # transpose so that H <= W (smaller dimension becomes number of rows)
    if H > W:
        new_H, new_W = W, H
        new_grid = [[-1]*new_W for _ in range(new_H)]
        for i in range(H):
            for j in range(W):
                new_grid[j][i] = grid[i][j]
        H, W = new_H, new_W
        grid = new_grid

    # ---------- generate all valid column states (no vertical equal neighbours) ----------
    states = []          # list of tuples, each tuple is a column coloring
    state_code = []      # base-3 code of the state
    max_code = 3 ** H
    idx_from_code = array('i', [-1]) * max_code

    def gen_states(pos, prev, cur, code):
        if pos == H:
            idx = len(states)
            states.append(tuple(cur))
            state_code.append(code)
            idx_from_code[code] = idx
            return
        for c in range(3):
            if pos == 0 or c != prev:
                cur.append(c)
                gen_states(pos+1, c, cur, code*3 + c)
                cur.pop()

    gen_states(0, -1, [], 0)
    S = len(states)

    # ---------- precompute allowed states for each column ----------
    allowed_mask = []   # bytearray per column, 1 if state allowed
    allowed_idx = []    # list of allowed state indices per column
    for c in range(W):
        mask = bytearray(S)
        allowed = []
        for idx, tup in enumerate(states):
            ok = True
            for r in range(H):
                val = grid[r][c]
                if val != -1 and val != tup[r]:
                    ok = False
                    break
            if ok:
                mask[idx] = 1
                allowed.append(idx)
        allowed_mask.append(mask)
        allowed_idx.append(allowed)
        if not allowed:          # impossible column
            print(0)
            return

    # ---------- build compatibility graph using pair automaton ----------
    # pair states: (a,b) with a!=b, 6 of them
    pair_states = []
    pair_to_idx = {}
    for a in range(3):
        for b in range(3):
            if a != b:
                idx = len(pair_states)
                pair_states.append((a, b))
                pair_to_idx[(a, b)] = idx

    # next states for each pair state
    next_pair = [[] for _ in range(6)]
    for i, (a, b) in enumerate(pair_states):
        for a2 in range(3):
            if a2 == a: continue
            for b2 in range(3):
                if b2 == b: continue
                if a2 == b2: continue
                j = pair_to_idx[(a2, b2)]
                next_pair[i].append(j)

    # compatibility lists: for each state p, array of compatible states c
    compat = [array('I') for _ in range(S)]

    # iterative DFS over the pair automaton
    stack_depth = []
    stack_s = []
    stack_code_p = []
    stack_code_c = []

    # initial states: first row of the two columns
    for s0 in range(6):
        a, b = pair_states[s0]
        stack_depth.append(1)
        stack_s.append(s0)
        stack_code_p.append(a)
        stack_code_c.append(b)

    while stack_depth:
        depth = stack_depth.pop()
        s = stack_s.pop()
        code_p = stack_code_p.pop()
        code_c = stack_code_c.pop()
        if depth == H:
            idx_p = idx_from_code[code_p]
            idx_c = idx_from_code[code_c]
            compat[idx_p].append(idx_c)
        else:
            for ns in next_pair[s]:
                a, b = pair_states[ns]
                new_code_p = code_p * 3 + a
                new_code_c = code_c * 3 + b
                stack_depth.append(depth+1)
                stack_s.append(ns)
                stack_code_p.append(new_code_p)
                stack_code_c.append(new_code_c)

    # ---------- DP over columns ----------
    dp_prev = [0] * S
    for idx in allowed_idx[0]:
        dp_prev[idx] = 1

    for col in range(1, W):
        dp_cur = [0] * S
        cur_mask = allowed_mask[col]
        for p in allowed_idx[col-1]:
            val = dp_prev[p]
            if val == 0:
                continue
            for c in compat[p]:
                if cur_mask[c]:
                    new_val = dp_cur[c] + val
                    if new_val >= MOD:
                        new_val -= MOD
                    dp_cur[c] = new_val
        dp_prev = dp_cur

    ans = sum(dp_prev[idx] for idx in allowed_idx[W-1]) % MOD
    print(ans)

    main()
