import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    a = list(map(int, data[1:1+n]))
    if n == 1:
        print(1)
        return
    # compute diff array
    diff = [a[i+1] - a[i] for i in range(n-1)]
    total = 0
    cur_len = 1
    for i in range(1, len(diff)):
        if diff[i] == diff[i-1]:
            cur_len += 1
        else:
            total += cur_len * (cur_len + 1) // 2
            cur_len = 1
    total += cur_len * (cur_len + 1) // 2
    ans = n + total
    print(ans)

if __name__ == "__main__":

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    n = int(data[0])
    a = list(map(int, data[1:1 + n]))

    if n == 1:
        print(1)
        return

    # difference array
    diff = [a[i + 1] - a[i] for i in range(n - 1)]

    total = 0          # sum of L*(L+1)/2 over runs of equal diffs
    cur_len = 1        # length of current run (at least one element)

    for i in range(1, len(diff)):
        if diff[i] == diff[i - 1]:
            cur_len += 1
        else:
            total += cur_len * (cur_len + 1) // 2
            cur_len = 1

    total += cur_len * (cur_len + 1) // 2   # last run

    ans = n + total
    print(ans)


if __name__ == "__main__":
    main()
