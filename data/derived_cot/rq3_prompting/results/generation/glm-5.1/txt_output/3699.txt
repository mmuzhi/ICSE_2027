```python
    def numberOfSubsequences(self, nums: List[int]) -> int:
        N = len(nums)
        pairs = [[] for _ in range(1000001)]
        for x in range(1, 1001):
            for val in range(x, 1001):
                k = x * val
                pairs[k].append((x, val))
        
        ans = 0
        for r in range(4, N - 2):
            suf_freq = [0] * 1001
            for s in range(r + 2, N):
                suf_freq[nums[s]] += 1
            
            match = [0] * 1001
            for q in range(2, r - 1):
                p = q - 2
                k = nums[p] * nums[r]
                for x, val in pairs[k]:
                    match[x] += suf_freq[val]
                    if x != val:
                        match[val] += suf_freq[x]
                
                ans += match[nums[q]]
                
        return ans
```

```python
class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        N = len(nums)
        pairs = [[] for _ in range(1000001)]
        for x in range(1, 1001):
            for val in range(x, 1001):
                k = x * val
                pairs[k].append((x, val))
        
        ans = 0
        for r in range(4, N - 2):
            suf_freq = [0] * 1001
            for s in range(r + 2, N):
```
```python
                suf_freq[nums[s]] += 1
```
            
```python
            match = [0] * 1001
            for q in range(2, r - 1):
                p = q - 2
                k = nums[p] * nums[r]
```
```python
                for x, val in pairs[k]:
                    match[x] += suf_freq[val]
                    if x != val:
                        match[val] += suf_freq[x]
                
                ans += match[nums[q]]
                
        return ans
```

```python
class Solution:
    def numberOfSubsequences(self, nums: List[int]) -> int:
        N = len(nums)
        
        # Precompute all pairs (x, val) such that x * val = k
        # Since nums[i] <= 1000, k <= 10^6
        pairs = [[] for _ in range(1000001)]
        for x in range(1, 1001):
            for val in range(x, 1001):
                pairs[x * val].append((x, val))
        
        ans = 0
        suf_freq = [0] * 1001
        suf_freq[nums[N - 1]] += 1
        
        # Iterate r from right to left
        for r in range(N - 3, 3, -1):
            match = [0] * 1001
            nums_r = nums[r]
            
            # Iterate q from left to right
            for q in range(2, r - 1):
                p = q - 2
                k = nums[p] * nums_r
                
                # Update match array based on the new p added
                for x, val in pairs[k]:
                    match[x] += suf_freq[val]
                    if x != val:
                        match[val] += suf_freq[x]
                
                # Add the count of valid s for the current q
                ans += match[nums[q]]
            
            # Add the new valid s for the next r
            if r > 4:
                suf_freq[nums[r + 1]] += 1
                
        return ans
```
