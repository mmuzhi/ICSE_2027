```python
def solve():
    import sys
    input = sys.stdin.read
    N, *rest = map(int, input().split())
    points = []
    for i in range(N):
        points.append((rest[2*i], rest[2*i+1]))
    
    # Sort by X
    points.sort()
    Y = [p[1] for p in points]
    
    # Precompute U({i})
    # U({i}) is the set of points incomparable to i
    # j is incomparable to i iff (X_j < X_i and Y_j > Y_i) or (X_j > X_i and Y_j < Y_i)
    # Since points are sorted by X, X_j < X_i means j < i.
    U = []
    for i in range(N):
        u = set()
        for j in range(N):
            if j < i and Y[j] > Y[i]:
                u.add(j)
            elif j > i and Y[j] < Y[i]:
                u.add(j)
            elif j == i:
                u.add(j)
        U.append(u)
    
    # BFS
    from collections import deque
    q = deque()
    start = frozenset(range(N))
    visited = set()
    visited.add(start)
    q.append(start)
    
    while q:
        S = q.popleft()
        for i in S:
            S_prime = S & U[i]
            if S_prime not in visited:
                visited.add(S_prime)
                q.append(S_prime)
                
    print(len(visited))
```
