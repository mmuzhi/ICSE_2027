```python
    def distanceSum(self, m: int, n: int, k: int) -> int:

        for i in range(1, total_cells + 1):

        inv_fact = [1] * (total_cells + 1)

        for i in range(total_cells - 1, -1, -1):
            inv_fact[i] = (inv_fact[i+1] * (i+1)) % MOD
            
        def comb(n, r):

        C = comb(total_cells - 2, k - 2)
```

```python
class Solution:
    def distanceSum(self, m: int, n: int, k: int) -> int:
        MOD = 10**9 + 7
        total_cells = m * n
        fact = [1] * (total_cells + 1)
        for i in range(1, total_cells + 1):
            fact[i] = (fact[i-1] * i) % MOD
        
        inv_fact = [1] * (total_cells + 1)
        inv_fact[total_cells] = pow(fact[total_cells], MOD - 2, MOD)
        for i in range(total_cells - 1, -1, -1):
            inv_fact[i] = (inv_fact[i+1] * (i+1)) % MOD
            
        def comb(n, r):
            if r < 0 or r > n:
                return 0
            return fact[n] * inv_fact[r] % MOD * inv_fact[n-r] % MOD
            
        C = comb(total_cells - 2, k - 2)
        
        inv6 = pow(6, MOD - 2, MOD)
        term1 = n * n % MOD * m % MOD * (m - 1) % MOD * (m + 1) % MOD
        term2 = m * m % MOD * n % MOD * (n - 1) % MOD * (n + 1) % MOD
        S = (term1 + term2) % MOD * inv6 % MOD
        
        return C * S % MOD
```

```python
class Solution:
    def distanceSum(self, m: int, n: int, k: int) -> int:
        MOD = 10**9 + 7
        total_cells = m * n
        
        # Precompute factorials and inverse factorials up to total_cells
        fact = [1] * (total_cells + 1)
        for i in range(1, total_cells + 1):
            fact[i] = (fact[i-1] * i) % MOD
            
        inv_fact = [1] * (total_cells + 1)
        inv_fact[total_cells] = pow(fact[total_cells], MOD - 2, MOD)
        for i in range(total_cells - 1, -1, -1):
            inv_fact[i] = (inv_fact[i+1] * (i+1)) % MOD
            
        def comb(n, r):
            if r < 0 or r > n:
                return 0
            return fact[n] * inv_fact[r] % MOD * inv_fact[n-r] % MOD
            
        # Number of valid arrangements containing any specific pair of cells
        C = comb(total_cells - 2, k - 2)
        
        # Sum of Manhattan distances over all unordered pairs of cells
        inv6 = pow(6, MOD - 2, MOD)
        term1 = n * n % MOD * m % MOD * (m - 1) % MOD * (m + 1) % MOD
        term2 = m * m % MOD * n % MOD * (n - 1) % MOD * (n + 1) % MOD
        S = (term1 + term2) % MOD * inv6 % MOD
        
        return C * S % MOD
```
