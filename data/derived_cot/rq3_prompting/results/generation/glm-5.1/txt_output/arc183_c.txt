import sys

MOD = 998244353

def solve():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    conds_by_x = [[] for _ in range(N+1)]
    for _ in range(M):
        L, R, X = map(int, input().split())
        conds_by_x[X].append((L, R))
    
    # Precompute binomial coefficients
    C = [[0]*(N+1) for _ in range(N+1)]
    for n in range(N+1):
        C[n][0] = C[n][n] = 1
        for k in range(1, n):
            C[n][k] = (C[n-1][k-1] + C[n-1][k]) % MOD
    
    # Precompute forbid bitsets
    forbid = [[0]*(N+1) for _ in range(N+1)]  # forbid[l][r] for 1<=l<=r<=N
    
    for k in range(1, N+1):
        if not conds_by_x[k]:
            continue
        # create diff matrix (N+2) x (N+2)
        diff = [[0]*(N+2) for _ in range(N+2)]
        for L, R in conds_by_x[k]:
            diff[1][R] += 1
            diff[L+1][R] -= 1
        # 2D prefix sum
        for i in range(1, N+1):
            row_i = diff[i]
            row_i_1 = diff[i-1]
            for j in range(1, N+1):
                # diff[i][j] += diff[i-1][j] + diff[i][j-1] - diff[i-1][j-1]
                row_i[j] = row_i[j] + row_i_1[j] + row_i[j-1] - row_i_1[j-1]
        # set bits
        mask = 1 << (k-1)
        for i in range(1, N+1):
            row_i = diff[i]
            for j in range(i, N+1):
                if row_i[j] > 0:
                    forbid[i][j] |= mask
    
    # DP
    dp = [[0]*(N+2) for _ in range(N+2)]
    for length in range(1, N+1):
        for l in range(1, N-length+2):
            r = l + length - 1
            total = 0
            forbid_mask = forbid[l][r]
            len_seg = length
            for k in range(l, r+1):
                if (forbid_mask >> (k-1)) & 1:
                    continue
                left_size = k - l
                right_size = r - k
                comb = C[len_seg-1][left_size]
                left_val = dp[l][k-1] if left_size > 0 else 1
                right_val = dp[k+1][r] if right_size > 0 else 1
                total = (total + comb * left_val % MOD * right_val) % MOD
            dp[l][r] = total
    
    print(dp[1][N])

if __name__ == "__main__":
    solve()
