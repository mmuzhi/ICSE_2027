        if p * b <= q * a:

                    best = cost
            return best
        else:
            best = 10**30
            for x in range(b):
                rem = C - x * a
                if rem <= 0:
                    y = 0
                else:
                    y = (rem + b - 1) // b
                cost = x * p + y * q
                if cost < best:
                    best = cost
            return best

    def feasible(C):

import sys

def main():
    import sys
    input = sys.stdin.readline
    N, X = map(int, input().split())
    procs = [tuple(map(int, input().split())) for _ in range(N)]

    def min_cost(a, p, b, q, C):
        if C <= 0:
            return 0
        # compare efficiency: lower cost per unit is better
        if p * b <= q * a:
            # S is more efficient or equal, bound y < a
            best = 10**30
            for y in range(a):
                rem = C - y * b
                if rem <= 0:
                    x = 0
                else:
                    x = (rem + a - 1) // a
                cost = x * p + y * q
                if cost < best:
                    best = cost
            return best
        else:
            # T is more efficient, bound x < b
            best = 10**30
            for x in range(b):
                rem = C - x * a
                if rem <= 0:
                    y = 0
                else:
                    y = (rem + b - 1) // b
                cost = x * p + y * q
                if cost < best:
                    best = cost
            return best

    def feasible(C):
        total = 0
        for a, p, b, q in procs:
            total += min_cost(a, p, b, q, C)
            if total > X:
                return False
        return total <= X

    lo = 0

        mid = (lo + hi + 1) // 2

            lo = mid
        else:
