```python

import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    A = [int(x) for x in data[1:N+1]]
    
    # Check if there's a valid sequence B starting with `start_val`
    def check(start_val):
        # R stores the set of possible values for B_i
        R = {start_val}
        for i in range(N):
            next_R = set()
            # If B_i = 1, B_{i+1} must be 0
            if 1 in R:
                next_R.add(0)
            # If B_i = 0, B_{i+1} depends on A_{i+1}
            if 0 in R:
                if A[(i+1)%N] == 0:
                    # A_{i+1} = 0 requires B_i + B_{i+1} = 1, so B_{i+1} = 1
                    next_R.add(1)
                else:
                    # A_{i+1} = 1 allows B_{i+1} to be 0 or 1
                    next_R.add(0)
                    next_R.add(1)
            R = next_R
            # If at any point no states are reachable, we can stop early
            if not R:
                break
        # A valid sequence must be consistent with the cycle, so B_{N+1} must equal B_1
        return start_val in R

    if check(0) or check(1):
        print("Yes")
    else:
        print("No")

if __name__ == '__main__':
    solve()
```
