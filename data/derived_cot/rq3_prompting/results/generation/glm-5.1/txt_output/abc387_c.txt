```python
def count_snake(X):
    if X < 10:
        return 0
    s = str(X)
    N = len(s)
    ans = 0
    for length in range(1, N):
        for F in range(1, 10):
            ans += F ** (length - 1)
    
    first_digit = int(s[0])
    for F in range(1, first_digit):
        ans += F ** (N - 1)
        
    for i in range(1, N):
        current_digit = int(s[i])
        ans += min(first_digit, current_digit) * (first_digit ** (N - 1 - i))
        if current_digit >= first_digit:
            break
    else:
        ans += 1
        
    return ans
```

```python
import sys

def count_snake(X):
    if X < 10:
        return 0
    s = str(X)
    N = len(s)
    ans = 0
    
    # Count all Snake numbers with length strictly less than N
    for length in range(2, N):
        for F in range(1, 10):
            ans += F ** (length - 1)
            
    first_digit = int(s[0])
    
    # Count all Snake numbers of length N where the first digit is strictly less than X's first digit
    for F in range(1, first_digit):
        ans += F ** (N - 1)
        
    # Count all Snake numbers of length N that start with the same first digit as X
    for i in range(1, N):
        current_digit = int(s[i])
        ans += min(first_digit, current_digit) * (first_digit ** (N - 1 - i))
        if current_digit >= first_digit:
            break
    else:
        # If the loop completes without breaking, X itself is a Snake number
        ans += 1
        
    return ans

def main():
    input_data = sys.stdin.read().split()
    L, R = map(int, input_data)
    print(count_snake(R) - count_snake(L - 1))

if __name__ == '__main__':
```
