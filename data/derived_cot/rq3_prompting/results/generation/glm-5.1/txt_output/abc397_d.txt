import sys, math

def icbrt(n):
    lo, hi = 0, 10**6+1
    while lo < hi:
        mid = (lo + hi + 1) // 2
        if mid * mid * mid <= n:
            lo = mid
        else:
            hi = mid - 1
    return lo

def main():
    N = int(sys.stdin.readline().strip())
    d_max = icbrt(N)
    for d in range(1, d_max+1):
        if N % d != 0:
            continue
        M = N // d
        # compute discriminant
        delta = 12 * M - 3 * d * d
        s = math.isqrt(delta)
        if s * s != delta:
            continue
        # check divisibility
        if (s - 3 * d) % 6 != 0:
            continue
        y = (s - 3 * d) // 6
        if y <= 0:
            continue
        x = y + d
        # optional verification
        # if x**3 - y**3 != N: continue (should hold)
        print(x, y)
        return
    print(-1)

if __name__ == "__main__":
