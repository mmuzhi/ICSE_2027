What is bitwise XOR? The bitwise XOR of non-negative integers A and B, denoted as A \oplus B, is defined as follows: - In the binary representation of A \oplus B, the digit at position 2^k (k \geq 0) is 1 if exactly one of A and B has a 1 in that position in their binary representations, and 0 otherwise. In general, the XOR of k integers p_1, \dots, p_k is defined as (\cdots ((p_1 \oplus p_2) \oplus p_3) \oplus \cdots \oplus p_k), and it can be proved that this is independent of the order of p_1, \dots, p_k.

import sys

MOD = 998244353

def main():
    import sys
    input = sys.stdin.readline
    N, M, K = map(int, input().split())
    A = list(map(int, input().split()))

    # ---------- factorials for binom ----------
    fact = [1] * (N + 1)
    for i in range(1, N + 1):
        fact[i] = fact[i-1] * i % MOD
    inv_fact = [1] * (N + 1)
    inv_fact[N] = pow(fact[N], MOD-2, MOD)
    for i in range(N, 0, -1):
        inv_fact[i-1] = inv_fact[i] * i % MOD

    def comb(n, k):
        if k < 0 or k > n:
            return 0
        return fact[n] * inv_fact[k] % MOD * inv_fact[n-k] % MOD

    # ---------- compute C(p) for all p ----------
    # h = H_0 = (1-y)^N mod (y^M - 1)
    h = [0] * M
    for j in range(N + 1):
        c = comb(N, j)
        r = j % M
        if j & 1:
            h[r] = (h[r] - c) % MOD
        else:
            h[r] = (h[r] + c) % MOD

    C = [0] * (N + 1)
    C[0] = h[0]

    invM = pow(M, MOD-2, MOD)
    pow2N = pow(2, N, MOD)

    h_prev = h
    h_cur = [0] * M

    for p in range(1, N + 1):
        # build a particular solution h_cur_0
        h_cur[0] = 0
        sum0 = 0
        for r in range(1, M):
            R = (h_prev[r] + h_prev[r-1]) % MOD
            h_cur[r] = (h_cur[r-1] + R) % MOD
            sum0 = (sum0 + h_cur[r]) % MOD
        # sum0 now is sum of h_cur[1..M-1]; h_cur[0] is 0
        S_p = pow2N if p == N else 0
        C_val = (S_p - sum0) % MOD
        C_val = C_val * invM % MOD
        # add C_val to all coefficients
        for r in range(M):
            h_cur[r] = (h_cur[r] + C_val) % MOD
        C[p] = C_val          # = h_cur[0]
        # swap for next iteration
        h_prev, h_cur = h_cur, h_prev

    # ---------- Walsh–Hadamard transforms ----------
    size = 1 << 20          # because A_i < 2^20

    # histogram of A
    H = [0] * size
    for a in A:
        H[a] += 1

    def fwt(a):
        n = len(a)
        step = 1
        while step < n:
            for i in range(0, n, step*2):
                for j in range(i, i+step):
                    x = a[j]
                    y = a[j+step]
                    a[j] = x + y
                    if a[j] >= MOD:
                        a[j] -= MOD
                    a[j+step] = x - y
                    if a[j+step] < 0:
                        a[j+step] += MOD
            step <<= 1

    fwt(H)          # H[t] = S(t) = Σ_i (-1)^{t·A_i}

    # f[x] = x^K
    f = [0] * size
    for x in range(size):
        f[x] = pow(x, K, MOD)

    fwt(f)          # f[t] = W(t) = Σ_x x^K (-1)^{t·x}

    # ---------- combine ----------
    inv_size = pow(size, MOD-2, MOD)
    ans = 0
    for t in range(size):
        s_val = H[t]
        # convert to integer in [-N, N]
        if s_val > N:
            s_int = s_val - MOD
        else:
            s_int = s_val
        p = (N + s_int) // 2          # 0 ≤ p ≤ N
        c_val = C[p]
        w_val = f[t]
        ans = (ans + c_val * w_val) % MOD

    ans = ans * inv_size % MOD
    print(ans)

if __name__ == "__main__":
    main()
