            curr_f[k] = (A[r][k] * (prev_f[k] + curr_f[k-1])) % MOD
            
        # Compute new curr_rf from next_rf
        curr_rf[W] = (A[r][W] * next_rf[W]) % MOD
        for k in range(W - 1, 0, -1):
            curr_rf[k] = (A[r][k] * (next_rf[k] + curr_rf[k+1])) % MOD
            
        out.append(str(ans))
