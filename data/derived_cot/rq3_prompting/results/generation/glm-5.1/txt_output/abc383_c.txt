import sys
from collections import deque

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    
    H = int(data[0])
    W = int(data[1])
    D = int(data[2])
    grid = data[3:]
    
    q = deque()
    visited = [[False] * W for _ in range(H)]
    
    for i in range(H):
        for j in range(W):
            if grid[i][j] == 'H':
                q.append((i, j, 0))
                visited[i][j] = True
                
    ans = 0
    directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
    
    while q:
        r, c, d = q.popleft()
        if d > D:
            continue
        ans += 1
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < H and 0 <= nc < W and not visited[nr][nc] and grid[nr][nc] != '#':
                visited[nr][nc] = True
                q.append((nr, nc, d + 1))
                
    print(ans)

if __name__ == '__main__':
