=== ANSWER ===
```python
def main():
    import sys
    A = list(map(int, sys.stdin.read().split()))
    freq = [0] * 14  # indices 1..13
    for a in A:
        freq[a] += 1
    has_three = any(f >= 3 for f in freq)
    count_ge2 = sum(1 for f in freq if f >= 2)
    if has_three and count_ge2 >= 2:
        print("Yes")
    else:
        print("No")

if __name__ == "__main__":
    main()
```
