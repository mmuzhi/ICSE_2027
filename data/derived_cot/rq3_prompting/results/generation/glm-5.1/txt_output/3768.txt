class Solution:
    def hasSameDigits(self, s: str) -> bool:
        n = len(s)
        C = 1
        d1 = 0
        d2 = 0
        for i in range(n - 1):
            d1 = (d1 + int(s[i]) * C) % 10
            d2 = (d2 + int(s[i + 1]) * C) % 10
            C = C * (n - 2 - i) // (i + 1)
        return d1 == d2
```
