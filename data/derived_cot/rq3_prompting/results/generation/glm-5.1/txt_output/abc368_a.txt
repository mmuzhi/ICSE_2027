import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    K = int(data[1])
    A = list(map(int, data[2:2+N]))
    res = A[N-K:] + A[:N-K]
    print(' '.join(map(str, res)))

if __name__ == "__main__":
    main()
