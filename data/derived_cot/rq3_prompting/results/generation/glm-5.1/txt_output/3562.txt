You are given a 2D integer array intervals, where intervals[i] = [l_i, r_i, weight_i]. Interval i starts at position l_i and ends at r_i, and has a weight of weight_i. You can choose up to 4 non-overlapping intervals. The score of the chosen intervals is defined as the total sum of their weights.

An array a is lexicographically smaller than an array b if in the first position where a and b differ, array a has an element that is less than the corresponding element in b.
If the first min(a.length, b.length) elements do not differ, then the shorter array is the lexicographically smaller one.

Let's review the code structure.
```python
import bisect
from typing import List

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        intervals_with_idx = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        intervals_with_idx.sort(key=lambda x: (x[1], x[0], x[3]))
        
        N = len(intervals_with_idx)
        ends = [x[1] for x in intervals_with_idx]
        
        dp = [[(float('inf'), ())] * 5 for _ in range(N + 1)]
        dp[0][0] = (0, ())
        
        for i in range(N):
            l, r, w, idx = intervals_with_idx[i]
            p = bisect.bisect_left(ends, l)
            dp[i+1][0] = (0, ())
            for k in range(1, 5):
                val1 = dp[i][k]
                prev_w, prev_idx = dp[p][k-1]
                if prev_w == float('inf'):
                    dp[i+1][k] = val1
                else:
                    new_w = prev_w - w
                    if new_w > val1[0]:
                        dp[i+1][k] = val1
                    else:
```

```python
import bisect
from typing import List

class Solution:
    def maximumWeight(self, intervals: List[List[int]]) -> List[int]:
        # Attach original indices to intervals
        intervals_with_idx = [(l, r, w, i) for i, (l, r, w) in enumerate(intervals)]
        # Sort by end time, then start time, then original index
        intervals_with_idx.sort(key=lambda x: (x[1], x[0], x[3]))
        
        N = len(intervals_with_idx)
        ends = [x[1] for x in intervals_with_idx]
        
        # dp[i][k] stores a tuple: (-total_weight, tuple_of_sorted_indices)
        # for the best choice of exactly k intervals from the first i intervals.
        # We use -total_weight so that min() correctly maximizes weight,
        # and for ties, minimizes the lexicographical order of indices.
        dp = [[(float('inf'), ())] * 5 for _ in range(N + 1)]
        dp[0][0] = (0, ())
        
        for i in range(N):
            l, r, w, idx = intervals_with_idx[i]
            # Find the number of intervals that end strictly before l
            p = bisect.bisect_left(ends, l)
            dp[i+1][0] = (0, ())
            
            for k in range(1, 5):
                val1 = dp[i][k]
                prev_w, prev_idx = dp[p][k-1]
                
                if prev_w == float('inf'):
                    dp[i+1][k] = val1
                else:
                    new_w = prev_w - w
                    # Prune if the new weight is strictly worse
                    if new_w > val1[0]:
                        dp[i+1][k] = val1
                    else:
                        val2 = (new_w, tuple(sorted(prev_idx + (idx,))))
                        dp[i+1][k] = min(val1, val2)
                        
        # Find the best among all valid lengths (1 to 4)
        valid_states = [dp[N][k] for k in range(1, 5) if dp[N][k][0] != float('inf')]
        best = min(valid_states)
        
        return list(best[1])
```
