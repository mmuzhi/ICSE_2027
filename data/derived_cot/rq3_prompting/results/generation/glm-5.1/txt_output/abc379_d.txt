def update(node):
    if node:
        node.size = 1 + size(node.left) + size(node.right)

def split(node, key):
    if not node:
        return (None, None)
    if node.key <= key:
        # node and its left go to left, split right
        left, right = split(node.right, key)
        node.right = left
        update(node)
        return (node, right)
    else:
        left, right = split(node.left, key)
        node.left = right
        update(node)
        return (left, node)

def merge(left, right):
    if not left or not right:
        return left or right
    if left.prio > right.prio:
        left.right = merge(left.right, right)
        update(left)
        return left
    else:
        right.left = merge(left, right.left)
        update(right)
        return right

import sys, random

sys.setrecursionlimit(1 << 25)

def size(node):
    return node.size if node else 0

def update(node):
    if node:
        node.size = 1 + size(node.left) + size(node.right)

def split(node, key):
    if not node:
        return (None, None)
    if node.key <= key:
        # node goes to left, split right subtree
        l, r = split(node.right, key)
        node.right = l
        update(node)
        return (node, r)
    else:
        l, r = split(node.left, key)
        node.left = r
        update(node)
        return (l, node)

def merge(left, right):
    if not left or not right:
        return left or right
    if left.prio > right.prio:
        left.right = merge(left.right, right)
        update(left)
        return left
    else:
        right.left = merge(left, right.left)
        update(right)
        return right

def insert(root, key):
    node = Node(key)
    left, right = split(root, key)
    # merge left + node + right
    root = merge(merge(left, node), right)
    return root

class Node:
    __slots__ = ('key','prio','left','right','size')
    def __init__(self, key):
        self.key = key
        self.prio = random.randint(1, 1<<30)

        self.right = None
        self.size = 1

We'll use 1-indexed tree array of size 4*M. Each node stores sum and lazy flag (bool) indicating that the segment should be set to zero. Since assignment is to zero, we don't need to store a value; just a flag. When we apply lazy to a node, we set its sum to 0, and set lazy flag True. When pushing down, we propagate to children: set their lazy flags True and their sums to 0, and clear current lazy flag.

For point add: we go down to leaf, pushing lazy along the way, then increment leaf sum, and update ancestors.

For range assign zero: we recursively apply to segments fully within range, setting sum to 0 and lazy flag True, and stop recursing deeper (since we don't need to keep old values). For partial overlap, we push down lazy first, then recurse to children, then update sum from children.

For range sum query: standard.

Complexities: O(log M) per operation.
