import sys
import heapq

def main():
    import sys
    input = sys.stdin.readline
    N, M = map(int, input().split())
    edges = []
    adj = [[] for _ in range(N+1)]
    for i in range(M):
        a, b, c = map(int, input().split())
        edges.append((a, b, c))
        adj[a].append((b, c))
        adj[b].append((a, c))

    INF = 10**18
    # Dijkstra from 1
    dist1 = [INF]*(N+1)
    dist1[1] = 0
    pq = [(0, 1)]
    while pq:
        d, u = heapq.heappop(pq)
        if d != dist1[u]:
            continue
        for v, c in adj[u]:
            nd = d + c
            if nd < dist1[v]:
                dist1[v] = nd
                heapq.heappush(pq, (nd, v))

    # Dijkstra from N
    distN = [INF]*(N+1)
    distN[N] = 0
    pq = [(0, N)]
    while pq:
        d, u = heapq.heappop(pq)
        if d != distN[u]:
            continue
        for v, c in adj[u]:
            nd = d + c
            if nd < distN[v]:
                distN[v] = nd
                heapq.heappush(pq, (nd, v))

    d0 = dist1[N]

    MOD1 = 1000000007
    MOD2 = 1000000009

    # Prepare DAG
    dag_adj = [[] for _ in range(N+1)]  # each element: (v, r1, r2, edge_idx)
    sp_info = [None]*M  # for edges on shortest path: (u, v, r1, r2)

    for i, (a, b, c) in enumerate(edges):
        cond1 = (dist1[a] + c + distN[b] == d0)
        cond2 = (dist1[b] + c + distN[a] == d0)
        if cond1 or cond2:
            if cond1:
                u, v = a, b
            else:
                u, v = b, a
            # random values
            r1 = (i * 1000003 + 1234567) % MOD1
            if r1 == 0:
                r1 = 1
            r2 = (i * 2000003 + 7654321) % MOD2
            if r2 == 0:
                r2 = 1
            sp_info[i] = (u, v, r1, r2)
            dag_adj[u].append((v, r1, r2, i))
        else:
            sp_info[i] = None

    # Compute f (forward DP)
    f1 = [0]*(N+1)
    f2 = [0]*(N+1)
    f1[1] = 1
    f2[1] = 1

    nodes = list(range(1, N+1))
    nodes.sort(key=lambda x: dist1[x])

    for u in nodes:
        # if f1[u]==0 and f2[u]==0: continue (optional)
        for v, r1, r2, idx in dag_adj[u]:
            f1[v] = (f1[v] + f1[u] * r1) % MOD1
            f2[v] = (f2[v] + f2[u] * r2) % MOD2

    # Compute g (backward DP)
    g1 = [0]*(N+1)
    g2 = [0]*(N+1)
    g1[N] = 1
    g2[N] = 1

    nodes_desc = sorted(range(1, N+1), key=lambda x: dist1[x], reverse=True)
    for u in nodes_desc:
        for v, r1, r2, idx in dag_adj[u]:
            # u->v, so use g[v]
            g1[u] = (g1[u] + r1 * g1[v]) % MOD1
            g2[u] = (g2[u] + r2 * g2[v]) % MOD2

    total1 = f1[N]
    total2 = f2[N]

    ans = ["No"]*M
    for i in range(M):
        info = sp_info[i]
        if info is not None:
            u, v, r1, r2 = info
            left1 = (f1[u] * r1) % MOD1
            left1 = (left1 * g1[v]) % MOD1
            left2 = (f2[u] * r2) % MOD2
            left2 = (left2 * g2[v]) % MOD2
            if left1 == total1 and left2 == total2:
                ans[i] = "Yes"

    # Output
    sys.stdout.write("\n".join(ans))

if __name__ == "__main__":
    main()
