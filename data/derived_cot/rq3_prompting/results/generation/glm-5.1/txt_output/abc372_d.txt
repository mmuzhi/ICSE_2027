    N = int(input())
    H = list(map(int, input().split()))
    # 1-index for convenience
    H = [0] + H  # index 1..N
    L = [0]*(N+1)  # L[j] previous greater index, 0 if none
    stack = []
    for j in range(1, N+1):
        while stack and H[stack[-1]] < H[j]:
            stack.pop()
        if stack:
            L[j] = stack[-1]
        else:
            L[j] = 0
        stack.append(j)
    diff = [0]*(N+2)  # 1-indexed, up to N+1

        l = L[j] if L[j] != 0 else 1
        r = j-1
        if l <= r:
            diff[l] += 1
            diff[r+1] -= 1
    c = [0]*(N+1)
    cur = 0
    for i in range(1, N+1):
        cur += diff[i]
        c[i] = cur

import sys

def solve() -> None:
    data = sys.stdin.read().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    H = [0] + [int(next(it)) for _ in range(N)]   # 1-indexed

    # previous greater element (strictly greater)
    prev = [0] * (N + 1)   # prev[j] = index of nearest left building taller than H[j], 0 if none
    stack = []
    for j in range(1, N + 1):
        while stack and H[stack[-1]] < H[j]:
            stack.pop()
        prev[j] = stack[-1] if stack else 0
        stack.append(j)

    # difference array for range additions
    diff = [0] * (N + 2)   # 1-indexed, extra cell for r+1
    for j in range(1, N + 1):
        l = prev[j] if prev[j] != 0 else 1
        r = j - 1
        if l <= r:
            diff[l] += 1
            diff[r + 1] -= 1

    # prefix sum -> answer
    ans = [0] * (N + 1)
    cur = 0
    for i in range(1, N + 1):
        cur += diff[i]
        ans[i] = cur

    print(' '.join(map(str, ans[1:])))

if __name__ == "__main__":
