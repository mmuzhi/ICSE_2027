    print("Yes")

    suffix_max = [0] * (m + 1)
    for i in range(m - 1, -1, -1):
        suffix_max[i] = max(black_cells[i][1], suffix_max[i + 1])
    
    def get_B_max(row):
        if not black_cells:
            return 0
        idx = bisect_left(black_rows, row)
        if idx >= m:
            return 0
        return suffix_max[idx]
    
    for r_w, c_w in white_cells:
        if get_B_max(r_w) >= c_w:
            print("No")
            return
    
    print("Yes")

solve()
