```python
    # sums floor((a*i + b)/m) for i=0..n-1
    # b can be negative
    if b < 0:
        k = (-b + m - 1) // m
        return floor_sum(n, m, a, b + k * m) - n * k
    ans = 0
    if a >= m:
        ans += (n - 1) * n * (a // m) // 2
        a %= m
    if b >= m:
        ans += n * (b // m)
        b %= m
    y_max = a * n + b
    if y_max >= m:
        ans += floor_sum(y_max // m, a, m, y_max % m)
    return ans
```

```python
    """
    Computes sum_{i=0}^{n-1} floor((a * i + b) / m)

    ans = 0
```
