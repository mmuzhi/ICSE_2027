You will be given a competitive programming problem.
There are N people numbered from 1 to N visiting a conveyor belt sushi restaurant.
The gourmet level of person i is A_i.

A sushi that person i takes and eats will no longer pass in front of person j\ (j > i).
For each of the M pieces of sushi, determine who eats that sushi, or if nobody eats it.

Input

The input is given from Standard Input in the following format:
N M
A_1 A_2 \dots A_N
B_1 B_2 \dots B_M

Output

Print M lines.

Constraints


- 1 \leq N, M \leq 2 \times 10^5
- 1 \leq A_i, B_i \leq 2 \times 10^5
- All input values are integers.

    input = sys.stdin.readline
    N, M = map(int, input().split())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))

    INF = N + 1

    # pref[0] = INF

        pref[v] = min(pref[v-1], best[v])
    out_lines = []
    for b in B:
        if b > max_val:  # shouldn't happen
            ans = -1
        else:
            ans = pref[b] if pref[b] != INF else -1
        out_lines.append(str(ans))
    sys.stdout.write("\n".join(out_lines))

import sys

def solve() -> None:
    input = sys.stdin.readline
    N, M = map(int, input().split())
    A = list(map(int, input().split()))
    B = list(map(int, input().split()))

    max_val = max(max(A), max(B))
    INF = N + 1                     # larger than any possible index

    # best[v] = smallest index i with A_i == v
    best = [INF] * (max_val + 1)
    for i, a in enumerate(A, start=1):
        if i < best[a]:
            best[a] = i

    # pref[v] = min_{w <= v} best[w]  = smallest index i with A_i <= v
    pref = [INF] * (max_val + 1)
    for v in range(1, max_val + 1):
        pref[v] = min(pref[v-1], best[v])

    out = []
    for b in B:
        # b is guaranteed <= max_val
        ans = pref[b]
        out.append(str(ans) if ans != INF else "-1")

    sys.stdout.write("\n".join(out))

if __name__ == "__main__":
