```python

def solve():
    S = input().strip()
    pos = [0] * 26
    for i, char in enumerate(S):
        pos[ord(char) - ord('A')] = i + 1
    
    ans = 0
    for i in range(1, 26):
        ans += abs(pos[i] - pos[i-1])
        
    print(ans)

solve()
```
