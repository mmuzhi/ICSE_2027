            if cnt > max_cov:
                max_cov = cnt

import sys

def solve() -> None:
    input = sys.stdin.readline
    H, W, D = map(int, input().split())
    grid = [input().strip() for _ in range(H)]

    floors = [(i, j) for i in range(H) for j in range(W) if grid[i][j] == '.']
    n = len(floors)
    max_cov = 0

    for a in range(n):
        x1, y1 = floors[a]
        for b in range(a + 1, n):
            x2, y2 = floors[b]
            cnt = 0
            for i, j in floors:
                if abs(i - x1) + abs(j - y1) <= D or abs(i - x2) + abs(j - y2) <= D:
                    cnt += 1
            if cnt > max_cov:
                max_cov = cnt

    print(max_cov)

if __name__ == "__main__":
    solve()
