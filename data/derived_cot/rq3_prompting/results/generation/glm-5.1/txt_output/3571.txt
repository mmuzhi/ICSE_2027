Implementation details:

- For left:

  - Sort points by (x, -y). But we need to preserve original index. So create list of tuples (x, y, idx). Sort with key=lambda p: (p[0], -p[1]).
  - Compress y: collect all y values, sort unique, map to 1..M.
  - Initialize BIT of size M.
  - Iterate over sorted points: for each (x, y, idx):
      - r = rank[y]

class BIT:
    def __init__(self, n):
        self.n = n
        self.tree = [0]*(n+1)
    def update(self, i, val):
        while i <= self.n:
            if val > self.tree[i]:
                self.tree[i] = val
            i += i & -i
    def query(self, i):
        res = 0
        while i > 0:
            if self.tree[i] > res:
                res = self.tree[i]
            i -= i & -i
        return res

```python
from typing import List

class Solution:
    def maxPathLength(self, coordinates: List[List[int]], k: int) -> int:
        n = len(coordinates)
        points = [(coordinates[i][0], coordinates[i][1], i) for i in range(n)]
        
        # coordinate compression for y
        ys = [y for _, y, _ in points]
        sorted_ys = sorted(set(ys))
        y_to_rank = {y: i+1 for i, y in enumerate(sorted_ys)}  # 1-indexed
        M = len(sorted_ys)
        
        # BIT for max
        class BIT:
            def __init__(self, n):
                self.n = n
                self.tree = [0]*(n+1)
            def update(self, i, val):
                while i <= self.n:
                    if val > self.tree[i]:
                        self.tree[i] = val
                    i += i & -i
            def query(self, i):
                res = 0
                while i > 0:
                    if self.tree[i] > res:
                        res = self.tree[i]
                    i -= i & -i
                return res
        
        # left dp: longest increasing chain ending at each point
        left = [0]*n
        points_sorted_left = sorted(points, key=lambda p: (p[0], -p[1]))
        bit_left = BIT(M)
        for x, y, idx in points_sorted_left:
            r = y_to_rank[y]
            best = bit_left.query(r-1)
            left[idx] = best + 1
            bit_left.update(r, left[idx])
        
        # right dp: longest increasing chain starting at each point
        right = [0]*n
        points_sorted_right = sorted(points, key=lambda p: (-p[0], p[1]))
        bit_right = BIT(M)
        for x, y, idx in points_sorted_right:
            r = y_to_rank[y]
            rev = M - r + 1  # reversed rank for suffix query
            best = bit_right.query(rev - 1)
            right[idx] = best + 1
```

```python
from typing import List

class Solution:
    def maxPathLength(self, coordinates: List[List[int]], k: int) -> int:
        n = len(coordinates)
        points = [(coordinates[i][0], coordinates[i][1], i) for i in range(n)]

        # coordinate compression for y
        ys = [y for _, y, _ in points]
        sorted_ys = sorted(set(ys))
        y_to_rank = {y: i+1 for i, y in enumerate(sorted_ys)}  # 1‑based
        M = len(sorted_ys)

        # Fenwick tree for maximum
        class BIT:
            def __init__(self, n):
                self.n = n
                self.tree = [0] * (n + 1)

            def update(self, i, val):
                while i <= self.n:
                    if val > self.tree[i]:
                        self.tree[i] = val
                    i += i & -i

            def query(self, i):
                res = 0
                while i > 0:
                    if self.tree[i] > res:
                        res = self.tree[i]
                    i -= i & -i
                return res

        # left[i] = longest chain ending at i (including i)
        left = [0] * n
        points_left = sorted(points, key=lambda p: (p[0], -p[1]))  # x asc, y desc
        bit_left = BIT(M)
        for x, y, idx in points_left:
            r = y_to_rank[y]
            best = bit_left.query(r - 1)   # y' < y
            left[idx] = best + 1
            bit_left.update(r, left[idx])

        # right[i] = longest chain starting at i (including i)
        right = [0] * n
        points_right = sorted(points, key=lambda p: (-p[0], p[1]))  # x desc, y asc
        bit_right = BIT(M)
        for x, y, idx in points_right:
            r = y_to_rank[y]
            rev = M - r + 1               # reversed rank for suffix query
            best = bit_right.query(rev - 1)   # y' > y
            right[idx] = best + 1
            bit_right.update(rev, right[idx])

        return left[k] + right[k] - 1
```
