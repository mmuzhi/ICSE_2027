def solve():
    N, M = map(int, input().split())
    A = list(map(int, input().split()))
    
    # Compute prefix sums mod M
    pref = [0] * N
    pref[0] = 0  # S[1] = 0
    for i in range(1, N):
        pref[i] = (pref[i-1] + A[i-1]) % M
    
    total = sum(A)
    T = total % M
    
    # Count frequency of each residue
    freq = {}
    for p in pref:
        freq[p] = freq.get(p, 0) + 1
    
    # Iterate and count
    seen = {}
    ans = 0
    for s in range(N):
        ps = pref[s]
        # t > s with pref[t] == pref[s]
        ans += freq.get(ps, 0) - seen.get(ps, 0) - 1
        # t < s with pref[t] == (pref[s] - T) mod M
        target = (ps - T) % M
        ans += seen.get(target, 0)
        # Add s to seen
        seen[ps] = seen.get(ps, 0) + 1
    
    print(ans)
