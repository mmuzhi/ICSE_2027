```python
def z_function(s):
    n = len(s)
    z = [0] * n
    l, r = 0, 0
    for i in range(1, n):
        if i < r:
            z[i] = min(r - i, z[i - l])
        while i + z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] > r:
            l, r = i, i + z[i]
    return z
```

```python
    def minStartingIndex(self, s: str, pattern: str) -> int:
        n, m = len(s), len(pattern)
        
        def z_function(st):
            ln = len(st)
            z = [0] * ln
            l, r = 0, 0
            for i in range(1, ln):
                if i < r:
                    z[i] = min(r - i, z[i - l])
                while i + z[i] < ln and st[z[i]] == st[i + z[i]]:
                    z[i] += 1
                if i + z[i] > r:
                    l, r = i, i + z[i]

        A = pattern + "#" + s
        z_A = z_function(A)
        
        rev_s = s[::-1]
        rev_p = pattern[::-1]
        B = rev_p + "#" + rev_s
        z_B = z_function(B)
        
        for i in range(n - m + 1):
            L = z_A[m + 1 + i]
            R = z_B[n + 1 - i]
            if L + R >= m - 1:
                return i
                
        return -1
```
