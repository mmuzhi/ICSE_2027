```python

import sys

def main() -> None:
    data = sys.stdin.read().splitlines()
    if not data:
        return
    K = int(data[0].strip())          # K is always 1, but we read it anyway
    S = data[1].strip()
    T = data[2].strip()

    n, m = len(S), len(T)
    if abs(n - m) > 1:
        print("No")
        return

    i = j = 0
    diff = 0
    while i < n and j < m:
        if S[i] == T[j]:
            i += 1
            j += 1
        else:
            if diff == 1:
                print("No")
                return
            diff += 1
            if n == m:          # replace
                i += 1
                j += 1
            elif n > m:         # delete from S
                i += 1
            else:               # insert into S (skip T)
                j += 1

    leftover = (n - i) + (m - j)
    if diff == 0:
        print("Yes" if leftover <= 1 else "No")
    else:
        print("Yes" if leftover == 0 else "No")

if __name__ == "__main__":
    main()
```
