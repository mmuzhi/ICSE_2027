```python
import heapq

class Solution:
    def minTimeToReach(self, moveTime: List[List[int]]) -> int:
        n = len(moveTime)
        m = len(moveTime[0])
        dist = [[float('inf')] * m for _ in range(n)]
        dist[0][0] = 0
        pq = [(0, 0, 0)]
        dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        
        while pq:
            d, x, y = heapq.heappop(pq)
            if d > dist[x][y]:
                continue
            if x == n - 1 and y == m - 1:
                return d
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < n and 0 <= ny < m:
                    nd = max(d, moveTime[nx][ny]) + 1
                    if nd < dist[nx][ny]:
                        dist[nx][ny] = nd
                        heapq.heappush(pq, (nd, nx, ny))
        return -1
```

```python
from typing import List

class Solution:
    def minTimeToReach(self, moveTime: List[List[int]]) -> int:

        m = len(moveTime[0])
        
        dist = [[float('inf')] * m for _ in range(n)]
        dist[0][0] = 0
        
        pq = [(0, 0, 0)] # (time, x, y)
        dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)]

            d, x, y = heapq.heappop(pq)
            
            if d > dist[x][y]:
                continue
                
            if x == n - 1 and y == m - 1:
                return d
                
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < n and 0 <= ny < m:
                    # We must wait until moveTime[nx][ny] to start moving, 
                    # then it takes 1 second to arrive.
                    nd = max(d, moveTime[nx][ny]) + 1
                    if nd < dist[nx][ny]:
                        dist[nx][ny] = nd
                        heapq.heappush(pq, (nd, nx, ny))
                        
        return -1
```
