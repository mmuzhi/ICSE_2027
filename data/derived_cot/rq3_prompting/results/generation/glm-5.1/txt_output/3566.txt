```python
class Solution:
    def stringSequence(self, target: str) -> List[str]:
        res = []
        chars = []
        for c in target:
            chars.append('a')
            res.append("".join(chars))
            for i in range(1, ord(c) - ord('a') + 1):

        return res
```

```python
class Solution:
    def stringSequence(self, target: str) -> List[str]:
        res = []
        cur = ""
        for c in target:
            cur += "a"
            res.append(cur)
            for i in range(1, ord(c) - 96): # 96 is ord('a')
                cur = cur[:-1] + chr(96 + i)
                res.append(cur)
        return res
```
