Now we must ensure that the greedy algorithm indeed yields the lexicographically smallest array. Let's reason:

We process indices in increasing order. At step i, we consider the set of k that satisfy all constraints from previous indices (which ensured that for each previous index j < i, the value A_j(k) is minimal among those k that satisfy earlier constraints). This set is exactly the set of k that achieve the minimal possible values for indices 1..i-1. Among those k, we want to minimize A_i(k). So we choose the k (or rather, impose a new congruence) that gives the minimal A_i(k) while still being compatible with previous constraints. This is exactly the greedy lexicographic minimization. Since the values are distinct, the minimal A_i(k) among compatible k is unique, and we can enforce it by adding a congruence that fixes k modulo the cycle length of i (if not already fixed). However, we must be careful: It might be that the minimal A_i(k) is achieved by multiple k that are compatible with previous constraints? Since values are distinct, for a given cycle, the mapping from k mod m to A_i(k) is injective (because as k varies, A_i(k) takes each value in the cycle exactly once). So the minimal value corresponds to a unique residue t modulo m. So there is a unique t that minimizes A_i(k). Therefore, adding the congruence k ≡ t (mod m) is necessary to achieve that minimal value. So greedy works.

But what about the case where the cycle of i has already been fixed by a previous index? Then A_i(k) is already determined (since k mod m is fixed). So we cannot change it; we must accept that value. This is consistent with lexicographic minimization because any k that would give a smaller A_i would violate the earlier constraints (which fixed the residue for that cycle to achieve minimal earlier values). So it's correct.

Thus, the algorithm is sound.

import sys
sys.setrecursionlimit(1<<25)

def sieve_spf(n):
    spf = list(range(n+1))
    for i in range(2, int(n**0.5)+1):
        if spf[i] == i:
            for j in range(i*i, n+1, i):
                if spf[j] == j:
                    spf[j] = i
    return spf

def factorize(x, spf):
    res = []
    while x > 1:
        p = spf[x]
        e = 0
        while x % p == 0:
            x //= p
            e += 1
        res.append((p, e))
    return res

def main():
    import sys
    input = sys.stdin.readline
    N = int(input())
    P = list(map(int, input().split()))
    A = list(map(int, input().split()))
    # convert to 0-index
    P = [x-1 for x in P]
    # A remains as 1..N values, but we'll use as is

    # precompute spf up to N
    spf = sieve_spf(N)

    # build cycles
    visited = [False]*N
    cycle_id = [0]*N
    pos_in_cycle = [0]*N
    cycles = []  # list of dicts

    for i in range(N):
        if not visited[i]:
            # traverse cycle
            cur = i
            indices = []
            while not visited[cur]:
                visited[cur] = True
                indices.append(cur)
                cur = P[cur]
            m = len(indices)
            vals = [A[idx] for idx in indices]
            factors = factorize(m, spf)
            # generate divisors
            divisors = []
            def gen_div(idx, cur):
                if idx == len(factors):
                    divisors.append(cur)
                else:
                    p, e = factors[idx]
                    for exp in range(e+1):
                        gen_div(idx+1, cur * (p**exp))
            gen_div(0, 1)
            # precompute best
            best = {}
            for d in divisors:
                best[d] = [-1] * d
            for j in range(m):
                val = vals[j]
                for d in divisors:
                    r = j % d
                    if best[d][r] == -1 or val < vals[best[d][r]]:
                        best[d][r] = j
            # store cycle data
            cid = len(cycles)
            for pos, idx in enumerate(indices):
                cycle_id[idx] = cid
                pos_in_cycle[idx] = pos
            cycles.append({
                'm': m,
                'vals': vals,
                'factors': factors,
                'divisors': divisors,
                'best': best
            })

    # initialize system
    prime_exp = {}  # p -> exponent
    prime_res = {}  # p -> residue modulo p^{exp}
    fixed_cycle = [False] * len(cycles)
    cycle_offset = [0] * len(cycles)

    # function to get g and R_mod_g
    def get_residue_mod_g(factors_m):
        g = 1
        R = 0
        M_cur = 1
        for p, e in factors_m:
            e_cur = prime_exp.get(p, 0)
            f = min(e_cur, e)
            if f == 0:
                continue
            mod = p ** f
            res_mod = prime_res[p] % mod
            # combine
            inv = pow(M_cur, -1, mod)
            X = (R + M_cur * ((res_mod - R) * inv % mod)) % (M_cur * mod)
            R = X
            M_cur *= mod
            g = M_cur
        return g, R

    # main loop
    for i in range(N):
        cid = cycle_id[i]
        if fixed_cycle[cid]:
            continue
        cycle = cycles[cid]
        m = cycle['m']
        pos = pos_in_cycle[i]
        factors_m = cycle['factors']
        g, R_mod_g = get_residue_mod_g(factors_m)
        S = (pos + R_mod_g) % g
        # query best
        j_min = cycle['best'][g][S]
        t = (j_min - pos) % m
        cycle_offset[cid] = t
        # update system with k ≡ t (mod m)
        for p, e in factors_m:
            e_cur = prime_exp.get(p, 0)
            if e_cur == 0:
                prime_exp[p] = e
                mod = p ** e
                prime_res[p] = t % mod
            else:
                if e > e_cur:
                    # extend
                    a = prime_res[p]  # current residue modulo p^{e_cur}
                    mod_cur = p ** e_cur
                    mod_new = p ** e
                    diff = (t - a) % mod_new
                    d = diff // mod_cur  # integer
                    y_mod = p ** (e - e_cur)
                    y = d % y_mod
                    new_res = a + mod_cur * y
                    new_res %= mod_new
                    prime_exp[p] = e
                    prime_res[p] = new_res
                else:
                    # e <= e_cur, just verify (optional)
                    # assert t % (p**e) == prime_res[p] % (p**e)
                    pass
        fixed_cycle[cid] = True

    # compute answer
    ans = [0]*N
    for i in range(N):
        cid = cycle_id[i]
        pos = pos_in_cycle[i]
        cycle = cycles[cid]
        m = cycle['m']
        vals = cycle['vals']
        offset = cycle_offset[cid]
        j = (pos + offset) % m
        ans[i] = vals[j]

    print(' '.join(map(str, ans)))

if __name__ == "__main__":
    main()
