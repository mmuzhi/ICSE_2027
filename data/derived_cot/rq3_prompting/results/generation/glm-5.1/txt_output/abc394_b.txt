import sys

def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    n = int(data[0].strip())
    strings = [line.strip() for line in data[1:1+n]]
    strings.sort(key=len)
    print(''.join(strings))

if __name__ == "__main__":
    main()
