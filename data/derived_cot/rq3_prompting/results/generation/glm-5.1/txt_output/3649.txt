```python
class Solution:
    def findMinimumTime(self, strength: List[int], K: int) -> int:
        n = len(strength)
        dp = [float('inf')] * (1 << n)
        dp[0] = 0
        for mask in range(1 << n):
            j = bin(mask).count('1')
            X = 1 + j * K
            for i in range(n):
                if not (mask & (1 << i)):
                    next_mask = mask | (1 << i)
                    cost = (strength[i] + X - 1) // X
                    dp[next_mask] = min(dp[next_mask], dp[mask] + cost)
        return dp[(1 << n) - 1]
```

class Solution:
    def findMinimumTime(self, strength: List[int], K: int) -> int:
        n = len(strength)
        dp = [float('inf')] * (1 << n)
        dp[0] = 0
        
        for mask in range(1 << n):
            if dp[mask] == float('inf'):
                continue
            j = bin(mask).count('1')
            X = 1 + j * K
            for i in range(n):
                if not (mask & (1 << i)):
                    next_mask = mask | (1 << i)
                    cost = (strength[i] + X - 1) // X
                    if dp[mask] + cost < dp[next_mask]:
                        dp[next_mask] = dp[mask] + cost
                        
        return dp[(1 << n) - 1]
