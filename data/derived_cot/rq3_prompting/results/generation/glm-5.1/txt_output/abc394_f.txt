   sys.setrecursionlimit(1<<25)

   def main():
       input = sys.stdin.readline
       N = int(input())
       adj = [[] for _ in range(N)]
       edges = []
       for i in range(N-1):
           a,b = map(int, input().split())
           a-=1; b-=1
           edges.append((a,b))
           adj[a].append((b,i))
           adj[b].append((a,i))

       # early check
       max_deg = max(len(adj[i]) for i in range(N))
       if max_deg < 4:
           print(-1)
           return

       M = N-1
       dp0 = [0]*M  # g(a,b)
       dp1 = [0]*M  # g(b,a)

       def get_g(frm, to, eid):
           if edges[eid][0] == frm:
               return dp0[eid]
           else:
               return dp1[eid]

       def set_g(frm, to, eid, val):
           if edges[eid][0] == frm:
               dp0[eid] = val
           else:
               dp1[eid] = val

       # dfs1
       def dfs1(u, p, pe_id):
           child_cnt = 0
           top = [0,0,0]  # three largest values
           for v, eid in adj[u]:
               if v == p: continue

               val = get_g(v, u, eid)
               child_cnt += 1
               # update top three

                   top[2] = top[1]; top[1] = top[0]; top[0] = val
               elif val > top[1]:
                   top[2] = top[1]; top[1] = val
               elif val > top[2]:
                   top[2] = val

               else:

               set_g(u, p, pe_id, g_u_p)

       dfs1(0, -1, -1)

       # dfs2
       def dfs2(u, p, pe_id):
           # collect neighbor info
           neigh = []  # (v, eid, val)
           for v, eid in adj[u]:
               if v == p:

                   neigh.append((v, eid, val))
               else:
                   val = get_g(v, u, eid)
                   neigh.append((v, eid, val))
           deg = len(neigh)
           if deg < 4:
               for v, eid, val in neigh:
                   set_g(u, v, eid, 1)
           else:
               # find top three and fourth
               top1_val, top1_nei = -1, -1
               top2_val, top2_nei = -1, -1
               top3_val, top3_nei = -1, -1
               fourth_val = -1
               for v, eid, val in neigh:
                   if val > top1_val:
                       fourth_val = top3_val
                       top3_val, top3_nei = top2_val, top2_nei
                       top2_val, top2_nei = top1_val, top1_nei
                       top1_val, top1_nei = val, v
                   elif val > top2_val:
                       fourth_val = top3_val
                       top3_val, top3_nei = top2_val, top2_nei
                       top2_val, top2_nei = val, v
                   elif val > top3_val:
                       fourth_val = top3_val
                       top3_val, top3_nei = val, v
                   else:
                       if val > fourth_val:
                           fourth_val = val
               # compute for each neighbor

                   if v == top1_nei:
                       sum_excl = top2_val + top3_val + fourth_val
                   elif v == top2_nei:

               if total > ans:
                   ans = total
       print(ans)
