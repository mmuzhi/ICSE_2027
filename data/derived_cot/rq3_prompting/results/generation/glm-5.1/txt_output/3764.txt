```python
import heapq

class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
        pool = []
        for i, row in enumerate(grid):
            limit = limits[i]
            if limit > 0:
                # Get top `limit` elements from row
                # Can use heapq.nlargest or sort
                top_row = heapq.nlargest(limit, row)
                pool.extend(top_row)
        
        # Get top `k` elements from pool
        top_k = heapq.nlargest(k, pool)
        return sum(top_k)
```

```python
class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
        pool = []
        for i, row in enumerate(grid):
            if limits[i] > 0:
                pool.extend(sorted(row, reverse=True)[:limits[i]])
        return sum(sorted(pool, reverse=True)[:k])
```

```python
import heapq

class Solution:
    def maxSum(self, grid: List[List[int]], limits: List[int], k: int) -> int:
        min_heap = []
        for i, row in enumerate(grid):
            limit = limits[i]
            if limit > 0:
                for val in sorted(row, reverse=True)[:limit]:
                    if len(min_heap) < k:
                        heapq.heappush(min_heap, val)
                    else:
                        if val > min_heap[0]:
                            heapq.heapreplace(min_heap, val)
        return sum(min_heap)
```
