Let's review the implementation details.
```python
class Solution:
    def maxDistance(self, s: str, k: int) -> int:
        ans = 0
        cN = cS = cE = cW = 0
        for char in s:
            if char == 'N': cN += 1
            elif char == 'S': cS += 1
            elif char == 'E': cE += 1
            elif char == 'W': cW += 1
            
            NE = (cE - cW) + (cN - cS) + 2 * min(cW + cS, k)
            NW = (cW - cE) + (cN - cS) + 2 * min(cE + cS, k)
            SE = (cE - cW) + (cS - cN) + 2 * min(cW + cN, k)
            SW = (cW - cE) + (cS - cN) + 2 * min(cE + cN, k)
            
            ans = max(ans, NE, NW, SE, SW)
        return ans
```
