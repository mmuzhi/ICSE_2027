    def countKReducibleNumbers(self, s: str, k: int) -> int:
        MOD = 10**9 + 7
        L = len(s)
        
        # Precompute combinations
        C = [[0] * (L + 1) for _ in range(L + 1)]
        for i in range(L + 1):
            C[i][0] = 1
            for j in range(1, i + 1):
                C[i][j] = (C[i-1][j-1] + C[i-1][j]) % MOD
                
        # Precompute steps

class Solution:
    def countKReducibleNumbers(self, s: str, k: int) -> int:
        MOD = 10**9 + 7
        L = len(s)
        
        # Precompute combinations C(n, r) mod MOD
        C = [[0] * (L + 1) for _ in range(L + 1)]
        for i in range(L + 1):
            C[i][0] = 1
            for j in range(1, i + 1):
                C[i][j] = (C[i-1][j-1] + C[i-1][j]) % MOD
                
        # Precompute steps(y) for y in [1, 800]
        # steps(y) is the minimum number of operations to reduce y to 1
        steps = [0] * 801
        for i in range(2, 801):
            steps[i] = 1 + steps[bin(i).count('1')]
            
        # Find the set S of popcounts that make a number k-reducible
        # A number x > 1 is k-reducible iff steps(popcount(x)) <= k - 1
        # x = 1 is k-reducible iff steps(1) = 0 <= k - 1 (for k >= 1)
        # So x is k-reducible iff popcount(x) in S
        S = set()
        for y in range(1, 801):
            if steps[y] <= k - 1:
                S.add(y)
                
        ans = 0
        C_fixed = 0
        
        # Iterate to find the first position where t < s
        for i in range(L):
            if s[i] == '1':
                free_len = L - 1 - i
                # If we set t[i] = 0, the remaining `free_len` bits can be chosen freely
                for needed in range(free_len + 1):
                    if needed + C_fixed in S:
                        ans = (ans + C[free_len][needed]) % MOD
                # For subsequent positions, t[i] is fixed to s[i] = 1
                C_fixed += 1
                
        return ans
