class SegTree:
    def __init__(self, n):
        self.n = n
        size = 1
        while size < n: size <<= 1
        self.size = size
        # We'll use arrays of length 2*size for iterative segment tree? But we need lazy propagation for range add. Iterative lazy segment tree is more complex. We'll stick with recursive with 4*n.
        self.minv = [0]*(4*n)
        self.maxv = [0]*(4*n)
        self.lazy = [0]*(4*n)
        self._build(1, 1, n)

    def _build(self, idx, l, r):
        if l == r:
            self.minv[idx] = l
            self.maxv[idx] = l
        else:
            mid = (l+r)//2
            self._build(idx*2, l, mid)
            self._build(idx*2+1, mid+1, r)
            self.minv[idx] = self.minv[idx*2]
            self.maxv[idx] = self.maxv[idx*2+1]  # since sorted, but we can compute min and max properly
            # Actually, min is min of children, max is max of children.
            self.minv[idx] = min(self.minv[idx*2], self.minv[idx*2+1])
            self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def _apply(self, idx, val):
        self.minv[idx] += val
        self.maxv[idx] += val
        self.lazy[idx] += val

    def _push(self, idx):
        if self.lazy[idx] != 0:
            self._apply(idx*2, self.lazy[idx])
            self._apply(idx*2+1, self.lazy[idx])
            self.lazy[idx] = 0

    def range_add(self, l, r, val):
        self._range_add(1, 1, self.n, l, r, val)

    def _range_add(self, idx, nl, nr, l, r, val):
        if l <= nl and nr <= r:
            self._apply(idx, val)
            return
        self._push(idx)
        mid = (nl+nr)//2
        if l <= mid:
            self._range_add(idx*2, nl, mid, l, r, val)
        if r > mid:
            self._range_add(idx*2+1, mid+1, nr, l, r, val)
        self.minv[idx] = min(self.minv[idx*2], self.minv[idx*2+1])
        self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def find_first_ge(self, L):
        # returns index or None
        return self._find_first_ge(1, 1, self.n, L)

    def _find_first_ge(self, idx, l, r, L):
        if self.maxv[idx] < L:
            return None
        if l == r:
            return l
        self._push(idx)
        mid = (l+r)//2
        # check left child first
        if self.maxv[idx*2] >= L:
            return self._find_first_ge(idx*2, l, mid, L)
        else:
            return self._find_first_ge(idx*2+1, mid+1, r, L)

    def find_first_gt(self, R):
        return self._find_first_gt(1, 1, self.n, R)

    def _find_first_gt(self, idx, l, r, R):
        if self.minv[idx] > R:
            return l  # all values > R, first index is l
        if self.maxv[idx] <= R:
            return None
        self._push(idx)
        mid = (l+r)//2
        if self.maxv[idx*2] > R:
            return self._find_first_gt(idx*2, l, mid, R)
        else:
            return self._find_first_gt(idx*2+1, mid+1, r, R)

    def get_value(self, pos):
        # optional, for debugging
        return self._get_value(1, 1, self.n, pos)

    def _get_value(self, idx, l, r, pos):
        if l == r:
            return self.minv[idx]
        self._push(idx)
        mid = (l+r)//2
        if pos <= mid:
            return self._get_value(idx*2, l, mid, pos)
        else:
            return self._get_value(idx*2+1, mid+1, r, pos)

import sys

def main():
    import sys
    sys.setrecursionlimit(1 << 25)
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    contests = [(next(it), next(it)) for _ in range(N)]
    Q = next(it)
    queries = [next(it) for _ in range(Q)]
    if not queries:
        return
    M = max(queries)  # we only need up to max query
    # But we also need to consider that during updates, indices up to M are enough.
    # However, what about X that are not queried but might be needed for the process of other X? No.
    # So M is sufficient.
    # But we must ensure that M is at least 1.
    # Also, we need to consider that the initial array size is M. If M is less than some L_i, that's fine.
    seg = SegTree(M)
    for L, R in contests:
        left = seg.find_first_ge(L)
        if left is None:
            continue
        first_gt = seg.find_first_gt(R)
        if first_gt is None:
            right = M
        else:
            right = first_gt - 1
        if left <= right:
            seg.range_add(left, right, 1)
    ans = [0]*(M+1)
    seg.collect(ans)
    out = '\n'.join(str(ans[x]) for x in queries)
    sys.stdout.write(out)

class SegTree:
    def __init__(self, n):
        self.n = n
        self.minv = [0]*(4*n)
        self.maxv = [0]*(4*n)
        self.lazy = [0]*(4*n)
        self._build(1, 1, n)

    def _build(self, idx, l, r):
        if l == r:
            self.minv[idx] = l
            self.maxv[idx] = l
        else:
            mid = (l+r)//2
            self._build(idx*2, l, mid)
            self._build(idx*2+1, mid+1, r)
            self.minv[idx] = self.minv[idx*2]
            self.maxv[idx] = self.maxv[idx*2+1]
            # Actually, min should be min of children, max should be max.
            # Since array is non-decreasing, left.min <= right.min, left.max <= right.max.
            # But to be safe, compute min and max.
            # self.minv[idx] = min(self.minv[idx*2], self.minv[idx*2+1])
            # self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])
            # We'll use the proper min/max.
            self.minv[idx] = min(self.minv[idx*2], self.minv[idx*2+1])
            self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def _apply(self, idx, val):
        self.minv[idx] += val
        self.maxv[idx] += val
        self.lazy[idx] += val

    def _push(self, idx):
        if self.lazy[idx]:
            self._apply(idx*2, self.lazy[idx])
            self._apply(idx*2+1, self.lazy[idx])
            self.lazy[idx] = 0

    def range_add(self, l, r, val):
        self._range_add(1, 1, self.n, l, r, val)

    def _range_add(self, idx, nl, nr, l, r, val):
        if l <= nl and nr <= r:
            self._apply(idx, val)
            return
        self._push(idx)
        mid = (nl+nr)//2
        if l <= mid:
            self._range_add(idx*2, nl, mid, l, r, val)
        if r > mid:
            self._range_add(idx*2+1, mid+1, nr, l, r, val)
        self.minv[idx] = min(self.minv[idx*2], self.minv[idx*2+1])
        self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def find_first_ge(self, L):
        return self._find_first_ge(1, 1, self.n, L)

    def _find_first_ge(self, idx, l, r, L):
        if self.maxv[idx] < L:
            return None
        if l == r:
            return l
        self._push(idx)
        mid = (l+r)//2
        if self.maxv[idx*2] >= L:
            return self._find_first_ge(idx*2, l, mid, L)
        else:
            return self._find_first_ge(idx*2+1, mid+1, r, L)

    def find_first_gt(self, R):
        return self._find_first_gt(1, 1, self.n, R)

    def _find_first_gt(self, idx, l, r, R):

import sys
from array import array

def main():
    import sys
    sys.setrecursionlimit(1 << 25)
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    contests = [(next(it), next(it)) for _ in range(N)]
    Q = next(it)
    queries = [next(it) for _ in range(Q)]
    if not queries:
        return
    M = max(queries)
    seg = SegTree(M)
    for L, R in contests:
        left = seg.find_first_ge(L)
        if left is None:
            continue
        first_gt = seg.find_first_gt(R)
        if first_gt is None:
            right = M
        else:
            right = first_gt - 1
        if left <= right:
            seg.range_add(left, right, 1)
    ans = [0] * (M + 1)
    seg.collect(ans)
    out = '\n'.join(str(ans[x]) for x in queries)
    sys.stdout.write(out)

class SegTree:
    def __init__(self, n):
        self.n = n
        self.minv = array('i', [0]) * (4 * n)
        self.maxv = array('i', [0]) * (4 * n)
        self.lazy = array('i', [0]) * (4 * n)
        self._build(1, 1, n)

    def _build(self, idx, l, r):
        if l == r:
            self.minv[idx] = l
            self.maxv[idx] = l
        else:
            mid = (l + r) // 2
            self._build(idx * 2, l, mid)
            self._build(idx * 2 + 1, mid + 1, r)
            self.minv[idx] = min(self.minv[idx * 2], self.minv[idx * 2 + 1])
            self.maxv[idx] = max(self.maxv[idx * 2], self.maxv[idx * 2 + 1])

    def _apply(self, idx, val):
        self.minv[idx] += val
        self.maxv[idx] += val
        self.lazy[idx] += val

    def _push(self, idx):
        if self.lazy[idx]:
            self._apply(idx * 2, self.lazy[idx])
            self._apply(idx * 2 + 1, self.lazy[idx])
            self.lazy[idx] = 0

    def range_add(self, l, r, val):
        self._range_add(1, 1, self.n, l, r, val)

    def _range_add(self, idx, nl, nr, l, r, val):
        if l <= nl and nr <= r:
            self._apply(idx, val)
            return
        self._push(idx)
        mid = (nl + nr) // 2
        if l <= mid:
            self._range_add(idx * 2, nl, mid, l, r, val)
        if r > mid:
            self._range_add(idx * 2 + 1, mid + 1, nr, l, r, val)
        self.minv[idx] = min(self.minv[idx * 2], self.minv[idx * 2 + 1])
        self.maxv[idx] = max(self.maxv[idx * 2], self.maxv[idx * 2 + 1])

    def find_first_ge(self, L):
        return self._find_first_ge(1, 1, self.n, L)

    def _find_first_ge(self, idx, l, r, L):
        if self.minv[idx] >= L:
            return l
        if self.maxv[idx] < L:
            return None
        if l == r:
            return l
        self._push(idx)
        mid = (l + r) // 2
        if self.maxv[idx * 2] >= L:
            return self._find_first_ge(idx * 2, l, mid, L)
        else:
            return self._find_first_ge(idx * 2 + 1, mid + 1, r, L)

    def find_first_gt(self, R):
        return self._find_first_gt(1, 1, self.n, R)

    def _find_first_gt(self, idx, l, r, R):
        if self.minv[idx] > R:
            return l
        if self.maxv[idx] <= R:
            return None
        if l == r:
            return l
        self._push(idx)
        mid = (l + r) // 2
        if self.maxv[idx * 2] > R:
            return self._find_first_gt(idx * 2, l, mid, R)
        else:
            return self._find_first_gt(idx * 2 + 1, mid + 1, r, R)

    def collect(self, res):
        self._collect(1, 1, self.n, res)

    def _collect(self, idx, l, r, res):
        if l == r:
            res[l] = self.minv[idx]
            return
        self._push(idx)
        mid = (l + r) // 2
        self._collect(idx * 2, l, mid, res)
        self._collect(idx * 2 + 1, mid + 1, r, res)

if __name__ == "__main__":
    main()

            self.maxv[idx] = l
        else:

            self._build(idx * 2, l, mid)
            self._build(idx * 2 + 1, mid + 1, r)
            self.minv[idx] = min(self.minv[idx * 2], self.minv[idx * 2 + 1])

        mid = (l + r) // 2
        self._collect(idx * 2, l, mid, res)
        self._collect(idx * 2 + 1, mid + 1, r, res)


if __name__ == "__main__":
    main()
