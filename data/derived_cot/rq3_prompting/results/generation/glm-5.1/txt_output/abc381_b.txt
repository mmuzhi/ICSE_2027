=== ANSWER ===
```python
import sys
from collections import Counter

def solve():
    S = sys.stdin.readline().strip()
    n = len(S)
    
    if n % 2 != 0:
        print("No")
        return
        
    for i in range(0, n, 2):
        if S[i] != S[i+1]:
            print("No")
            return
            
    counts = Counter(S)
    for v in counts.values():
        if v != 2:
            print("No")
            return
            
    print("Yes")

if __name__ == '__main__':
    solve()
```
