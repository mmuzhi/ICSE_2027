    import sys
    input = sys.stdin.read
    data = input().split()
    N = int(data[0])
    K = int(data[1])
    A = [int(x) for x in data[2:2+N]]
    
    S = sum(A)
    A_ext = A + A
    
    def check(X):
        # compute next_arr
        next_arr = [0] * (2 * N + 1)
        j = 0
        curr_sum = 0
        for i in range(2 * N):
            while j < 2 * N and curr_sum < X:
                curr_sum += A_ext[j]
                j += 1
            if curr_sum >= X:
                next_arr[i] = j
            else:
                next_arr[i] = 2 * N + 1
            curr_sum -= A_ext[i]
            
        # binary lifting
        up = [next_arr]
        for _ in range(1, 18):
            up_j = [0] * (2 * N + 1)
            up_j1 = up[-1]
            for i in range(2 * N + 1):
                up_j[i] = up_j1[up_j1[i]]
            up.append(up_j)
            
        for v in range(N):
            curr = v
            for j in range(17, -1, -1):
                if (K >> j) & 1:
                    curr = up[j][curr]
            if curr <= v + N:
                return True
        return False

    low = 0
    high = S
    ans_X = 0
    while low <= high:
        mid = (low + high) // 2
        if check(mid):
            ans_X = mid
            low = mid + 1
        else:
            high = mid - 1
            
    # Now find y
    X = ans_X
    next_arr = [0] * (2 * N + 1)
    j = 0
    curr_sum = 0
    for i in range(2 * N):
        while j < 2 * N and curr_sum < X:
            curr_sum += A_ext[j]
            j += 1
        if curr_sum >= X:
            next_arr[i] = j
        else:
            next_arr[i] = 2 * N + 1
        curr_sum -= A_ext[i]
        
    up = [next_arr]
    for _ in range(1, 18):
        up_j = [0] * (2 * N + 1)

        for i in range(2 * N + 1):

        up.append(up_j)
        
    y = 0
    for v in range(N):
        # check if cut line v can be cut
        # linear array starts at v+1
        curr = v + 1
        for j in range(17, -1, -1):
            if (K >> j) & 1:
                curr = up[j][curr]
        if curr > v + 1 + N:
            y += 1
            
    print(ans_X, y)

def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    N = int(data[0])
    K = int(data[1])
    A = [int(x) for x in data[2:2+N]]
    
    S = sum(A)
    A_ext = A + A
    
    def check(X):
        # compute next_arr
        next_arr = [0] * (2 * N + 1)
        j = 0
        curr_sum = 0
        for i in range(2 * N):
            while j < 2 * N and curr_sum < X:
                curr_sum += A_ext[j]
                j += 1
            if curr_sum >= X:
                next_arr[i] = j
            else:
                next_arr[i] = 2 * N + 1
            curr_sum -= A_ext[i]
            
        # binary lifting
        up = [next_arr]
        for _ in range(1, 18):
            up_j = [0] * (2 * N + 1)
            up_j1 = up[-1]
            for i in range(2 * N + 1):
                up_j[i] = up_j1[up_j1[i]]
            up.append(up_j)
            
        for v in range(N):
            curr = v
            for j in range(17, -1, -1):
                if (K >> j) & 1:
                    curr = up[j][curr]
            if curr <= v + N:
                return True
        return False

    low = 0
    high = S
    ans_X = 0
    while low <= high:
        mid = (low + high) // 2
        if check(mid):
            ans_X = mid
            low = mid + 1
        else:
            high = mid - 1
            
    # Now find y
    X = ans_X
    next_arr = [0] * (2 * N + 1)
    j = 0
    curr_sum = 0
    for i in range(2 * N):
        while j < 2 * N and curr_sum < X:
            curr_sum += A_ext[j]
            j += 1
        if curr_sum >= X:
            next_arr[i] = j
        else:
            next_arr[i] = 2 * N + 1
        curr_sum -= A_ext[i]
        
    up = [next_arr]
    for _ in range(1, 18):
        up_j = [0] * (2 * N + 1)
        up_j1 = up[-1]
        for i in range(2 * N + 1):
            up_j[i] = up_j1[up_j1[i]]
        up.append(up_j)
        
    y = 0
    for v in range(N):
        # check if cut line v can be cut
        # linear array starts at v+1
        curr = v + 1
        for j in range(17, -1, -1):
            if (K >> j) & 1:
                curr = up[j][curr]
        if curr > v + 1 + N:
            y += 1
            
    print(ans_X, y)

def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    N = int(data[0])
    K = int(data[1])
    A = [int(x) for x in data[2:2+N]]
    
    S = sum(A)
    A_ext = A + A
    
    def check(X):
        # compute next_arr
        next_arr = [0] * (2 * N + 1)
        j = 0
        curr_sum = 0
        for i in range(2 * N):
            while j < 2 * N and curr_sum < X:
                curr_sum += A_ext[j]
                j += 1
            if curr_sum >= X:
                next_arr[i] = j
            else:
                next_arr[i] = 2 * N + 1
            curr_sum -= A_ext[i]
            
        # binary lifting
        up = [next_arr]
        for _ in range(1, 18):
            up_j = [0] * (2 * N + 1)
            up_j1 = up[-1]
            for i in range(2 * N + 1):
                up_j[i] = up_j1[up_j1[i]]
            up.append(up_j)
            
        for v in range(N):
            curr = v
            for j in range(17, -1, -1):
                if (K >> j) & 1:
                    curr = up[j][curr]
            if curr <= v + N:
                return True
        return False

    low = 0
    high = S
    ans_X = 0
    while low <= high:
        mid = (low + high) // 2
        if check(mid):
            ans_X = mid
            low = mid + 1
        else:
            high = mid - 1
            
    # Now find y
    X = ans_X
    next_arr = [0] * (2 * N + 1)
    j = 0
    curr_sum = 0
    for i in range(2 * N):
        while j < 2 * N and curr_sum < X:
            curr_sum += A_ext[j]
            j += 1
        if curr_sum >= X:
            next_arr[i] = j
        else:
            next_arr[i] = 2 * N + 1
        curr_sum -= A_ext[i]
        
    up = [next_arr]
    for _ in range(1, 18):
        up_j = [0] * (2 * N + 1)
        up_j1 = up[-1]
        for i in range(2 * N + 1):
            up_j[i] = up_j1[up_j1[i]]
        up.append(up_j)
        
    y = 0
    for v in range(N):
        # check if cut line v can be cut
        # linear array starts at v+1
        curr = v + 1
        for j in range(17, -1, -1):
            if (K >> j) & 1:
                curr = up[j][curr]
        if curr > v + 1 + N:
            y += 1
            
    print(ans_X, y)
