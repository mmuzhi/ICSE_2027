import sys
import bisect
from collections import Counter

def main():
    it = iter(sys.stdin.buffer.read().split())
    N = int(next(it))
    D = int(next(it))
    xs = []
    ys = []
    for _ in range(N):
        x = int(next(it))
        y = int(next(it))
        xs.append(x)
        ys.append(y)

    xs.sort()
    ys.sort()

    # prefix sums
    px = [0]
    for v in xs:
        px.append(px[-1] + v)
    py = [0]
    for v in ys:
        py.append(py[-1] + v)

    # sum of absolute differences
    def sum_abs(val, arr, pref):
        idx = bisect.bisect_right(arr, val)
        left_sum = pref[idx]
        right_sum = pref[N] - left_sum
        cnt_le = idx
        cnt_gt = N - idx
        return val * cnt_le - left_sum + right_sum - val * cnt_gt

    median_x = xs[N // 2]
    median_y = ys[N // 2]
    f_min_x = sum_abs(median_x, xs, px)
    f_min_y = sum_abs(median_y, ys, py)

    if f_min_x + f_min_y > D:
        print(0)
        return

    freqX = Counter(xs)
    freqY = Counter(ys)

    LOW = -10**7
    HIGH = 10**7

    # build histogram for one coordinate
    def build_hist(arr, pref, freq, median, D, N):
        def f(val):
            idx = bisect.bisect_right(arr, val)
            left_sum = pref[idx]
            right_sum = pref[N] - left_sum
            cnt_le = idx
            return val * cnt_le - left_sum + right_sum - val * (N - cnt_le)

        # left bound
        lo, hi = LOW, median
        while lo < hi:
            mid = (lo + hi) // 2
            if f(mid) <= D:
                hi = mid
            else:
                lo = mid + 1
        L = lo

        # right bound
        lo, hi = median, HIGH
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if f(mid) <= D:
                lo = mid
            else:
                hi = mid - 1
        R = lo

        cnt = [0] * (D + 1)
        cur_x = L
        cur_f = f(cur_x)
        cur_lcnt = bisect.bisect_right(arr, cur_x)

        while cur_x <= R:
            cnt[cur_f] += 1
            if cur_x < R:
                delta = 2 * cur_lcnt - N
                cur_f += delta
                cur_lcnt += freq.get(cur_x + 1, 0)
            cur_x += 1
        return cnt

    cntX = build_hist(xs, px, freqX, median_x, D, N)
    cntY = build_hist(ys, py, freqY, median_y, D, N)

    # transform cntY into prefix sums
    for i in range(1, D + 1):
        cntY[i] += cntY[i - 1]

    ans = 0
    for a in range(D + 1):
        if cntX[a]:
            b_max = D - a
            ans += cntX[a] * cntY[b_max]

    print(ans)

if __name__ == "__main__":
    main()
