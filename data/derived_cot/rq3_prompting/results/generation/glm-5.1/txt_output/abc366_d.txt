```python
import sys

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    it = iter(input_data)
    N = int(next(it))
    
    # 3D Prefix Sum array with 1-based indexing and zero-padded boundaries
    P = [[[0] * (N + 1) for _ in range(N + 1)] for _ in range(N + 1)]
    
    for x in range(1, N + 1):
        for y in range(1, N + 1):
            for z in range(1, N + 1):
                val = int(next(it))
                P[x][y][z] = (val 
                              + P[x-1][y][z] + P[x][y-1][z] + P[x][y][z-1] 
                              - P[x-1][y-1][z] - P[x-1][y][z-1] - P[x][y-1][z-1] 
                              + P[x-1][y-1][z-1])
                              
    Q = int(next(it))
    out = []
    for _ in range(Q):
        Lx = int(next(it)); Rx = int(next(it))
        Ly = int(next(it)); Ry = int(next(it))
        Lz = int(next(it)); Rz = int(next(it))
        
        ans = (P[Rx][Ry][Rz] 
               - P[Lx-1][Ry][Rz] - P[Rx][Ly-1][Rz] - P[Rx][Ry][Lz-1] 
               + P[Lx-1][Ly-1][Rz] + P[Lx-1][Ry][Lz-1] + P[Rx][Ly-1][Lz-1] 
               - P[Lx-1][Ly-1][Lz-1])
        out.append(str(ans))
        
    sys.stdout.write("\n".join(out) + "\n")

if __name__ == '__main__':
    solve()
```
