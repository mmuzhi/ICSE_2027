```python
import sys

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    ans = 0
    for parity in (0, 1):
        i = parity
        while i < N - 1:
            if A[i] != A[i+1]:
                i += 2
                continue
            # start of a block
            block = []
            j = i
            while j < N - 1 and A[j] == A[j+1]:
                block.append(A[j])
                j += 2
            # find longest subarray with distinct values in block
            freq = {}
            left = 0
            max_len = 0
            for right, val in enumerate(block):
                while freq.get(val, 0) > 0:
                    left_val = block[left]
                    freq[left_val] -= 1
                    if freq[left_val] == 0:
                        del freq[left_val]
                    left += 1
                freq[val] = freq.get(val, 0) + 1
                max_len = max(max_len, right - left + 1)
            ans = max(ans, 2 * max_len)
            i = j
    print(ans)

if __name__ == "__main__":
    solve()
```

```python
import sys

def solve() -> None:
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))

    ans = 0

    # two possible parities for the start index of the subarray
    for parity in (0, 1):
        i = parity
        while i < N - 1:
            if A[i] != A[i + 1]:
                i += 2
                continue

            # start of a block of consecutive valid pairs with this parity
            block = []
            j = i
            while j < N - 1 and A[j] == A[j + 1]:
                block.append(A[j])
                j += 2

            # longest subarray with all distinct values inside this block
            freq = {}
            left = 0
            best = 0
            for right, val in enumerate(block):
                while freq.get(val, 0) > 0:
                    left_val = block[left]
                    freq[left_val] -= 1
                    if freq[left_val] == 0:
                        del freq[left_val]
                    left += 1
                freq[val] = freq.get(val, 0) + 1
                best = max(best, right - left + 1)

            ans = max(ans, 2 * best)
            i = j   # continue after the block

    print(ans)


if __name__ == "__main__":
    solve()
```
