class Solution:
    def zigzagTraversal(self, grid: List[List[int]]) -> List[int]:
        ans = []
        take = True
        m, n = len(grid), len(grid[0])
        for i in range(m):
            if i % 2 == 0:
                for j in range(n):
                    if take:
                        ans.append(grid[i][j])
                    take = not take
            else:
                for j in range(n - 1, -1, -1):
                    if take:
                        ans.append(grid[i][j])
                    take = not take
        return ans
```
