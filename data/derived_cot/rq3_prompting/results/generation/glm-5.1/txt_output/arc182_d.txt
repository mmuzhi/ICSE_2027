```python
import sys

def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    N = int(data[0])
    M = int(data[1])
    A = [int(x) for x in data[2:2+N]]
    B = [int(x) for x in data[2+N:2+2*N]]

    # Special case for M = 2
    # If M = 2, elements can only be 0 or 1. A good sequence must alternate 0, 1, 0...
    # If A != B, they have opposite parity, meaning they must swap, which is impossible 
    # without two adjacent elements becoming equal at some point.
    if M == 2:
        if A == B:
            print(0)
        else:
            print(-1)
        return

    # C_i is the clockwise distance from A_i to B_i
    C = [(B[i] - A[i]) % M for i in range(N)]
    
    # D_y_i stores the difference y_{i+1} - y_i
    D_y = [0] * (N - 1)
    for i in range(N - 1):
        C_diff = C[i+1] - C[i]
        D_i = (A[i+1] - A[i]) % M
        E_i = (B[i+1] - B[i]) % M
        DE_diff = E_i - D_i
        # The difference must be a multiple of M
        D_y[i] = (C_diff - DE_diff) // M

    min_cost = float('inf')

    # We can choose y_1 to be either 0 or 1
    for y1 in [0, 1]:
        valid = True
        cost = 0
        y_curr = y1
        for i in range(N):
            # y_i must be 0 or 1 for the path to not cross and deadlock
            if not (0 <= y_curr <= 1):
                valid = False
                break
            
            # If y_i = 0, we move clockwise by C_i. Cost is C_i.
            # If y_i = 1, we move counter-clockwise. Cost is M - C_i.
            if y_curr == 0:
                cost += C[i]
            else:
                cost += M - C[i]
            
            # Propagate to the next y
            if i < N - 1:
                y_curr += D_y[i]
        
        if valid:
            min_cost = min(min_cost, cost)

    if min_cost == float('inf'):
        print(-1)
    else:
        print(min_cost)

if __name__ == '__main__':
    solve()
```
