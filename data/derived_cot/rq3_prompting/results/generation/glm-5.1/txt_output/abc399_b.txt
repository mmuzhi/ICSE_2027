```python
def main():
    import sys
    data = sys.stdin.read().strip().split()
    if not data: return
    N = int(data[0])
    P = list(map(int, data[1:1+N]))
    max_score = 100
    freq = [0] * (max_score + 1)
    for s in P:
        freq[s] += 1
    # precompute cumulative from high to low? but N small, can compute per person

        rank = 1 + sum(freq[t] for t in range(s+1, max_score+1))
```

```python

import sys

def main():
    data = sys.stdin.read().split()
    if not data:
        return
    N = int(data[0])
    P = list(map(int, data[1:1 + N]))

    # frequency of each score
    freq = {}
    for s in P:
        freq[s] = freq.get(s, 0) + 1

    # distinct scores in descending order
    distinct = sorted(freq.keys(), reverse=True)

    # map each score to its rank
    rank_of = {}
    cum = 0  # number of people with higher scores already processed
    for s in distinct:
        rank_of[s] = cum + 1
        cum += freq[s]

    # output
    out = '\n'.join(str(rank_of[s]) for s in P)
    print(out)

if __name__ == "__main__":
    main()
```
