class SegTree:
    def __init__(self, n):
        self.n = n
        self.maxv = [0] * (4*n)
        self.lazy = [0] * (4*n)

    def _push(self, idx):
        if self.lazy[idx] != 0:
            for child in (2*idx, 2*idx+1):
                self.maxv[child] += self.lazy[idx]
                self.lazy[child] += self.lazy[idx]
            self.lazy[idx] = 0

    def _range_add(self, idx, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.maxv[idx] += val
            self.lazy[idx] += val
            return
        self._push(idx)
        mid = (l+r)//2
        if ql <= mid:
            self._range_add(2*idx, l, mid, ql, qr, val)
        if qr > mid:
            self._range_add(2*idx+1, mid+1, r, ql, qr, val)
        self.maxv[idx] = max(self.maxv[2*idx], self.maxv[2*idx+1])

    def range_add(self, l, r, val):
        if l > r: return
        self._range_add(1, 1, self.n, l, r, val)

    def _range_max(self, idx, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.maxv[idx]
        self._push(idx)
        mid = (l+r)//2
        res = -10**18
        if ql <= mid:
            res = max(res, self._range_max(2*idx, l, mid, ql, qr))
        if qr > mid:
            res = max(res, self._range_max(2*idx+1, mid+1, r, ql, qr))
        return res

    def range_max(self, l, r):
        if l > r: return -10**18
        return self._range_max(1, 1, self.n, l, r)

    def _point_query(self, idx, l, r, pos):
        if l == r:
            return self.maxv[idx]
        self._push(idx)
        mid = (l+r)//2
        if pos <= mid:
            return self._point_query(2*idx, l, mid, pos)
        else:
            return self._point_query(2*idx+1, mid+1, r, pos)

    def point_query(self, pos):
        return self._point_query(1, 1, self.n, pos)

class SegTree:
    def __init__(self, n):
        self.n = n
        self.maxv = [0] * (4*n)
        self.lazy = [0] * (4*n)

    def _push(self, idx):
        if self.lazy[idx]:
            val = self.lazy[idx]
            left = idx*2
            right = idx*2+1
            self.maxv[left] += val
            self.lazy[left] += val
            self.maxv[right] += val
            self.lazy[right] += val
            self.lazy[idx] = 0

    def _add(self, idx, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.maxv[idx] += val
            self.lazy[idx] += val
            return
        self._push(idx)
        mid = (l+r)//2
        if ql <= mid:
            self._add(idx*2, l, mid, ql, qr, val)
        if qr > mid:
            self._add(idx*2+1, mid+1, r, ql, qr, val)
        self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def add(self, l, r, val):
        if l > r:
            return
        self._add(1, 1, self.n, l, r, val)

    def _query(self, idx, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.maxv[idx]
        self._push(idx)
        mid = (l+r)//2
        res = -10**18

        if qr > mid:
            res = max(res, self._query(idx*2+1, mid+1, r, ql, qr))

    def query(self, l, r):

class SegTree:
    def __init__(self, n):
        self.n = n
        self.maxv = [0] * (4*n)
        self.lazy = [0] * (4*n)

    def build(self, base, idx=1, l=1, r=None):
        if r is None:
            r = self.n
        if l == r:
            self.maxv[idx] = base[l]
        else:
            mid = (l+r)//2
            self.build(base, idx*2, l, mid)
            self.build(base, idx*2+1, mid+1, r)
            self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def _push(self, idx):
        if self.lazy[idx]:
            val = self.lazy[idx]
            left = idx*2
            right = idx*2+1
            self.maxv[left] += val
            self.lazy[left] += val
            self.maxv[right] += val
            self.lazy[right] += val
            self.lazy[idx] = 0

    def _add(self, idx, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.maxv[idx] += val
            self.lazy[idx] += val
            return
        self._push(idx)
        mid = (l+r)//2
        if ql <= mid:
            self._add(idx*2, l, mid, ql, qr, val)
        if qr > mid:
            self._add(idx*2+1, mid+1, r, ql, qr, val)
        self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def add(self, l, r, val):
        if l > r:
            return
        self._add(1, 1, self.n, l, r, val)

    def _query(self, idx, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.maxv[idx]
        self._push(idx)
        mid = (l+r)//2
        res = -10**18
        if ql <= mid:
            res = max(res, self._query(idx*2, l, mid, ql, qr))
        if qr > mid:
            res = max(res, self._query(idx*2+1, mid+1, r, ql, qr))
        return res

    def query(self, l, r):
        if l > r:
            return -10**18
        return self._query(1, 1, self.n, l, r)

import sys

class SegTree:
    def __init__(self, n):
        self.n = n
        self.maxv = [0] * (4*n)
        self.lazy = [0] * (4*n)

    def build(self, base, idx=1, l=1, r=None):
        if r is None:
            r = self.n
        if l == r:
            self.maxv[idx] = base[l]
        else:
            mid = (l+r)//2
            self.build(base, idx*2, l, mid)
            self.build(base, idx*2+1, mid+1, r)
            self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def _push(self, idx):
        if self.lazy[idx]:
            val = self.lazy[idx]
            left = idx*2
            right = idx*2+1
            self.maxv[left] += val
            self.lazy[left] += val
            self.maxv[right] += val
            self.lazy[right] += val
            self.lazy[idx] = 0

    def _add(self, idx, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.maxv[idx] += val
            self.lazy[idx] += val
            return
        self._push(idx)
        mid = (l+r)//2
        if ql <= mid:
            self._add(idx*2, l, mid, ql, qr, val)
        if qr > mid:
            self._add(idx*2+1, mid+1, r, ql, qr, val)
        self.maxv[idx] = max(self.maxv[idx*2], self.maxv[idx*2+1])

    def add(self, l, r, val):
        if l > r:
            return
        self._add(1, 1, self.n, l, r, val)

    def _query(self, idx, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.maxv[idx]
        self._push(idx)
        mid = (l+r)//2
        res = -10**18
        if ql <= mid:
            res = max(res, self._query(idx*2, l, mid, ql, qr))
        if qr > mid:
            res = max(res, self._query(idx*2+1, mid+1, r, ql, qr))
        return res

    def query(self, l, r):
        if l > r:
            return -10**18
        return self._query(1, 1, self.n, l, r)

def solve():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    A = [0] + A  # 1-indexed

    # prefix distinct
    L = [0]*(N+1)
    seen = [False]*(N+1)
    for i in range(1, N+1):
        if not seen[A[i]]:
            L[i] = L[i-1] + 1
            seen[A[i]] = True
        else:
            L[i] = L[i-1]

    # suffix distinct
    S = [0]*(N+2)
    seen2 = [False]*(N+1)
    for i in range(N, 0, -1):
        if not seen2[A[i]]:
            S[i] = S[i+1] + 1
            seen2[A[i]] = True
        else:
            S[i] = S[i+1]

    # prev occurrence
    prev = [0]*(N+1)
    last = [0]*(N+1)
    for i in range(1, N+1):
        prev[i] = last[A[i]]
        last[A[i]] = i

    # base array for segment tree: f(l) initially = L[l-1]
    base = [0]*(N+1)
    for l in range(1, N+1):
        base[l] = L[l-1]

    seg = SegTree(N)
    seg.build(base)

    ans = 0
    for j in range(1, N+1):
        l = prev[j] + 1
        r = j
        seg.add(l, r, 1)
        if 2 <= j <= N-1:
            cur = seg.query(2, j)
            val = cur + S[j+1]
            if val > ans:
                ans = val

    print(ans)

if __name__ == "__main__":
    solve()

import sys

sys.setrecursionlimit(1 << 25)

class SegTree:
    def __init__(self, n):
        self.n = n
        self.maxv = [0] * (4 * n)
        self.lazy = [0] * (4 * n)

    def build(self, base, idx=1, l=1, r=None):
        if r is None:
            r = self.n
        if l == r:
            self.maxv[idx] = base[l]
        else:
            mid = (l + r) // 2
            self.build(base, idx * 2, l, mid)
            self.build(base, idx * 2 + 1, mid + 1, r)
            self.maxv[idx] = max(self.maxv[idx * 2], self.maxv[idx * 2 + 1])

    def _push(self, idx):
        if self.lazy[idx]:
            val = self.lazy[idx]
            left = idx * 2
            right = idx * 2 + 1
            self.maxv[left] += val
            self.lazy[left] += val
            self.maxv[right] += val
            self.lazy[right] += val
            self.lazy[idx] = 0

    def _add(self, idx, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self.maxv[idx] += val
            self.lazy[idx] += val
            return
        self._push(idx)
        mid = (l + r) // 2
        if ql <= mid:
            self._add(idx * 2, l, mid, ql, qr, val)
        if qr > mid:
            self._add(idx * 2 + 1, mid + 1, r, ql, qr, val)
        self.maxv[idx] = max(self.maxv[idx * 2], self.maxv[idx * 2 + 1])

    def add(self, l, r, val):
        if l > r:
            return
        self._add(1, 1, self.n, l, r, val)

    def _query(self, idx, l, r, ql, qr):
        if ql <= l and r <= qr:
            return self.maxv[idx]
        self._push(idx)
        mid = (l + r) // 2
        res = -10 ** 18
        if ql <= mid:
            res = max(res, self._query(idx * 2, l, mid, ql, qr))
        if qr > mid:
            res = max(res, self._query(idx * 2 + 1, mid + 1, r, ql, qr))
        return res

    def query(self, l, r):
        if l > r:
            return -10 ** 18
        return self._query(1, 1, self.n, l, r)


def solve():
    import sys
    input = sys.stdin.readline
    N = int(input())
    A = list(map(int, input().split()))
    A = [0] + A  # 1‑indexed

    # prefix distinct counts
    L = [0] * (N + 1)
    seen = [False] * (N + 1)
    for i in range(1, N + 1):
        if not seen[A[i]]:
            L[i] = L[i - 1] + 1
            seen[A[i]] = True
        else:
            L[i] = L[i - 1]

    # suffix distinct counts
    S = [0] * (N + 2)
    seen2 = [False] * (N + 1)
    for i in range(N, 0, -1):
        if not seen2[A[i]]:
            S[i] = S[i + 1] + 1
            seen2[A[i]] = True
        else:
            S[i] = S[i + 1]

    # previous occurrence
    prev = [0] * (N + 1)
    last = [0] * (N + 1)
    for i in range(1, N + 1):
        prev[i] = last[A[i]]
        last[A[i]] = i

    # base values for segment tree: f(l) initially = L[l-1]
    base = [0] * (N + 1)
    for l in range(1, N + 1):
        base[l] = L[l - 1]

    seg = SegTree(N)
    seg.build(base)

    for j in range(1, N + 1):

        r = j

        if 2 <= j <= N - 1:

            val = cur + S[j + 1]
            if val > ans:
                ans = val
