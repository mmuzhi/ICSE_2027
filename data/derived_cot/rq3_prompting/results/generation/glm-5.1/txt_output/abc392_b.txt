import sys

def main():
    data = sys.stdin.read().split()
    N = int(data[0])
    M = int(data[1])
    A = set(int(x) for x in data[2:2+M])
    missing = [i for i in range(1, N+1) if i not in A]
    print(len(missing))
    if missing:
        print(' '.join(map(str, missing)))

if __name__ == "__main__":
    main()
