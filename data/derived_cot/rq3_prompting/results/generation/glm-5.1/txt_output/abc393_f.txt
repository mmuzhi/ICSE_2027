import sys
import bisect

def main():
    data = list(map(int, sys.stdin.buffer.read().split()))
    it = iter(data)
    N = next(it)
    Q = next(it)
    A = [next(it) for _ in range(N)]
    queries_raw = []
    X_vals = []
    for i in range(Q):
        R = next(it)
        X = next(it)
        queries_raw.append((R, X, i))
        X_vals.append(X)
    
    # coordinate compression
    vals = sorted(set(A + X_vals))
    M = len(vals)
    # map A to indices (1-based)
    posA = [bisect.bisect_left(vals, a) + 1 for a in A]
    
    # preprocess queries: compute limit index for X
    qinfo = []
    for R, X, idx in queries_raw:
        limit = bisect.bisect_right(vals, X)  # number of values <= X, which is the BIT index to query
        qinfo.append((R, limit, idx))
    qinfo.sort(key=lambda x: x[0])
    
    # Fenwick tree for max
    bit = [0] * (M + 1)
    
    def bit_update(i, val):
        while i <= M:
            if val > bit[i]:
                bit[i] = val
            i += i & -i
    
    def bit_query(i):
        res = 0
        while i > 0:
            if bit[i] > res:
                res = bit[i]
            i -= i & -i
        return res
    
    ans = [0] * Q
    ptr = 0
    for i in range(1, N+1):
        pos = posA[i-1]
        best = bit_query(pos - 1)
        dp = best + 1
        bit_update(pos, dp)
        # answer queries with R == i
        while ptr < Q and qinfo[ptr][0] == i:
            R, limit, idx = qinfo[ptr]
            ans[idx] = bit_query(limit)
            ptr += 1
    
    # output
    out = '\n'.join(map(str, ans))
    sys.stdout.write(out)

    main()

    ptr = 0

    for i in range(1, N + 1):
        pos = posA[i - 1]
        best = bit_query(pos - 1)   # max dp for values < A[i]
        dp = best + 1
        bit_update(pos, dp)

        # answer all queries with R == i
        while ptr < Q and qinfo[ptr][0] == i:
            _, limit, idx = qinfo[ptr]
            ans[idx] = bit_query(limit)
            ptr += 1

    sys.stdout.write("\n".join(map(str, ans)))

if __name__ == "__main__":
    main()
