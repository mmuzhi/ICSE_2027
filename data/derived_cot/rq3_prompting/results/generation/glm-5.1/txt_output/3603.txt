from typing import List

class Solution:
    def findAnswer(self, parent: List[int], s: str) -> List[bool]:
        n = len(parent)
        adj = [[] for _ in range(n)]
        for i in range(1, n):
            adj[parent[i]].append(i)
            
        tour = [''] * n
        idx = [0] * n
        size = [1] * n
        
        idx_counter = 0
        stack = [(0, 0)]
        while stack:
            u, state = stack.pop()
            if state == 0:
                stack.append((u, 1))
                for v in reversed(adj[u]):
                    stack.append((v, 0))
            else:
                for v in adj[u]:
                    size[u] += size[v]
                idx[u] = idx_counter
                tour[idx_counter] = s[u]
                idx_counter += 1
                
        M1 = 10**9 + 7
        M2 = 10**9 + 9
        B = 131
        
        powB1 = [1] * (n + 1)
        powB2 = [1] * (n + 1)
        for i in range(1, n + 1):
            powB1[i] = (powB1[i-1] * B) % M1
            powB2[i] = (powB2[i-1] * B) % M2
            
        H1 = [0] * (n + 1)
        H2 = [0] * (n + 1)
        HR1 = [0] * (n + 1)
        HR2 = [0] * (n + 1)
        
        for i in range(n):
            val = ord(tour[i]) - ord('a') + 1
            H1[i+1] = (H1[i] * B + val) % M1
            H2[i+1] = (H2[i] * B + val) % M2
            
        for i in range(n):
            val = ord(tour[n - 1 - i]) - ord('a') + 1
            HR1[i+1] = (HR1[i] * B + val) % M1
            HR2[i+1] = (HR2[i] * B + val) % M2
            
        ans = [False] * n
        for u in range(n):
            L = idx[u] - size[u] + 1
            R = idx[u]
            length = size[u]
            
            h1 = (H1[R+1] - H1[L] * powB1[length]) % M1
            h2 = (H2[R+1] - H2[L] * powB2[length]) % M2
            
            rev_L = n - 1 - R
            rev_R = n - 1 - L
            rh1 = (HR1[rev_R+1] - HR1[rev_L] * powB1[length]) % M1
