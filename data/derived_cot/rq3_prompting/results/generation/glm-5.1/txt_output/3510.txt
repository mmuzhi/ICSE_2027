class Solution:
    def maximumTotalSum(self, maximumHeight: List[int]) -> int:
        maximumHeight.sort(reverse=True)
        prev = float('inf')
        total = 0
        for h in maximumHeight:
            assign = min(h, prev - 1)
            if assign <= 0:
                return -1
            total += assign
            prev = assign
        return total
